// Redis 캐싱 전략
import { createClient } from 'redis';

class ContentAuthCache {
    private redis: any;

    async connect(): Promise {
        this.redis = createClient({
            url: process.env.REDIS_URL
        });
        await this.redis.connect();
        console.log('✅ Redis 연결 완료');
    }

    // 지문 캐싱
    async cacheFingerprint(
        assetId: string,
        fingerprint: any,
        ttl: number = 3600  // 1시간
    ): Promise {
        const key = `fingerprint:${assetId}`;
        await this.redis.setEx(
            key,
            ttl,
            JSON.stringify(fingerprint)
        );
    }

    // 캐시 조회
    async getFingerprint(assetId: string): Promise {
        const key = `fingerprint:${assetId}`;
        const cached = await this.redis.get(key);

        if (cached) {
            console.log('✅ 캐시 히트');
            return JSON.parse(cached);
        }

        console.log('⚠️ 캐시 미스');
        return null;
    }

    // 캐시 무효화
    async invalidate(pattern: string): Promise {
        const keys = await this.redis.keys(pattern);
        if (keys.length > 0) {
            await this.redis.del(keys);
        }
    }
}

// OAuth 2.0 + JWT 인증
import { sign, verify } from 'jsonwebtoken';
import { hash, compare } from 'bcrypt';

class AuthenticationService {
    private jwtSecret: string;

    constructor(jwtSecret: string) {
        this.jwtSecret = jwtSecret;
    }

    // 사용자 인증
    async authenticate(
        username: string,
        password: string
    ): Promise {
        // 데이터베이스에서 사용자 조회
        const user = await this.getUserByUsername(username);
        if (!user) {
            return null;
        }

        // 비밀번호 검증
        const valid = await compare(password, user.passwordHash);
        if (!valid) {
            return null;
        }

        // JWT 토큰 생성
        const token = sign(
            {
                userId: user.id,
                username: user.username,
                roles: user.roles
            },
            this.jwtSecret,
            {
                expiresIn: '1h',
                issuer: 'WIA-AI-017'
            }
        );

        return token;
    }

    // 토큰 검증
    async verifyToken(token: string): Promise {
        try {
            const decoded = verify(token, this.jwtSecret);
            return decoded;
        } catch (error) {
            return null;
        }
    }

    // 역할 기반 접근 제어 (RBAC)
    hasPermission(
        userRoles: string[],
        requiredPermission: string
    ): boolean {
        const permissions: Record = {
            'admin': ['read', 'write', 'delete', 'admin'],
            'editor': ['read', 'write'],
            'viewer': ['read']
        };

        for (const role of userRoles) {
            if (permissions[role]?.includes(requiredPermission)) {
                return true;
            }
        }

        return false;
    }

    private async getUserByUsername(username: string): Promise {
        // 데이터베이스 조회 (구현 생략)
        return {
            id: '123',
            username,
            passwordHash: 'hashed_password',
            roles: ['editor']
        };
    }
}

// Express 미들웨어
import { Request, Response, NextFunction } from 'express';

function authMiddleware(requiredPermission: string) {
    return async (req: Request, res: Response, next: NextFunction) => {
        // Authorization 헤더에서 토큰 추출
        const authHeader = req.headers.authorization;
        if (!authHeader?.startsWith('Bearer ')) {
            return res.status(401).json({ error: '인증 필요' });
        }

        const token = authHeader.substring(7);
        const auth = new AuthenticationService(process.env.JWT_SECRET!);

        // 토큰 검증
        const decoded = await auth.verifyToken(token);
        if (!decoded) {
            return res.status(401).json({ error: '유효하지 않은 토큰' });
        }

        // 권한 확인
        if (!auth.hasPermission(decoded.roles, requiredPermission)) {
            return res.status(403).json({ error: '권한 없음' });
        }

        // 요청 객체에 사용자 정보 추가
        (req as any).user = decoded;
        next();
    };
}

// Prometheus 메트릭 수집
import { Counter, Histogram, Registry } from 'prom-client';

class MetricsCollector {
    private registry: Registry;
    private httpRequestCounter: Counter;
    private httpRequestDuration: Histogram;
    private fingerprintCounter: Counter;

    constructor() {
        this.registry = new Registry();

        // HTTP 요청 카운터
        this.httpRequestCounter = new Counter({
            name: 'http_requests_total',
            help: '총 HTTP 요청 수',
            labelNames: ['method', 'route', 'status'],
            registers: [this.registry]
        });

        // HTTP 요청 지속 시간
        this.httpRequestDuration = new Histogram({
            name: 'http_request_duration_seconds',
            help: 'HTTP 요청 지속 시간',
            labelNames: ['method', 'route'],
            buckets: [0.01, 0.05, 0.1, 0.5, 1, 2, 5],
            registers: [this.registry]
        });

        // 지문 생성 카운터
        this.fingerprintCounter = new Counter({
            name: 'fingerprints_generated_total',
            help: '생성된 지문 수',
            labelNames: ['algorithm', 'status'],
            registers: [this.registry]
        });
    }

    // HTTP 요청 기록
    recordRequest(
        method: string,
        route: string,
        status: number,
        duration: number
    ): void {
        this.httpRequestCounter.inc({
            method,
            route,
            status: status.toString()
        });

        this.httpRequestDuration.observe(
            { method, route },
            duration
        );
    }

    // 지문 생성 기록
    recordFingerprint(
        algorithm: string,
        success: boolean
    ): void {
        this.fingerprintCounter.inc({
            algorithm,
            status: success ? 'success' : 'failure'
        });
    }

    // 메트릭 노출
    async getMetrics(): Promise {
        return await this.registry.metrics();
    }
}

// 자동 백업 시스템
import { S3Client, PutObjectCommand } from '@aws-sdk/client-s3';
import { createGzip } from 'zlib';
import { createReadStream, createWriteStream } from 'fs';
import { pipeline } from 'stream/promises';

class BackupService {
    private s3: S3Client;

    constructor() {
        this.s3 = new S3Client({
            region: process.env.AWS_REGION
        });
    }

    // 데이터베이스 백업
    async backupDatabase(): Promise {
        const timestamp = new Date().toISOString();
        const filename = `backup-${timestamp}.sql.gz`;

        // PostgreSQL 덤프
        const { spawn } = await import('child_process');
        const pg_dump = spawn('pg_dump', [
            '-h', process.env.DB_HOST!,
            '-U', process.env.DB_USER!,
            '-d', process.env.DB_NAME!,
            '-F', 'c'  // 커스텀 형식
        ]);

        // Gzip 압축
        const gzip = createGzip();
        const output = createWriteStream(`/tmp/${filename}`);

        await pipeline(
            pg_dump.stdout,
            gzip,
            output
        );

        // S3에 업로드
        const fileStream = createReadStream(`/tmp/${filename}`);

        await this.s3.send(new PutObjectCommand({
            Bucket: 'content-auth-backups',
            Key: `database/${filename}`,
            Body: fileStream
        }));

        console.log(`✅ 백업 완료: ${filename}`);
    }

    // 자동 백업 스케줄
    scheduleBackups(): void {
        // 매일 오전 2시에 백업
        const cron = require('node-cron');

        cron.schedule('0 2 * * *', async () => {
            console.log('⏰ 자동 백업 시작');
            try {
                await this.backupDatabase();
            } catch (error) {
                console.error('❌ 백업 실패:', error);
                // 알림 전송 (이메일, Slack 등)
            }
        });
    }
}

// 감사 로깅 시스템
class AuditLogger {
    async logAction(
        action: {
            userId: string;
            action: string;
            resourceType: string;
            resourceId: string;
            changes?: any;
            ipAddress: string;
            userAgent: string;
        }
    ): Promise {
        // 데이터베이스에 감사 로그 저장
        const auditEntry = {
            ...action,
            timestamp: new Date().toISOString(),
            success: true
        };

        await this.saveToDatabase(auditEntry);

        // 중요 작업은 별도 알림
        if (this.isCriticalAction(action.action)) {
            await this.notifyAdmins(auditEntry);
        }
    }

    private isCriticalAction(action: string): boolean {
        const criticalActions = [
            'delete_asset',
            'modify_permissions',
            'export_data'
        ];
        return criticalActions.includes(action);
    }
