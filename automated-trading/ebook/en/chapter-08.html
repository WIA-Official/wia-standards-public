Reinforcement Learning for Trading

TypeScript: Reinforcement Learning Trading Agent Framework
interface State {
  prices: number[];
  volumes: number[];
  technicalIndicators: Record;
  position: number;
  cash: number;
  timestamp: number;
}

interface Action {
  type: 'BUY' | 'SELL' | 'HOLD';
  quantity?: number;
}

interface Reward {
  value: number;
  components: {
    pnl: number;
    riskAdjustment: number;
    transactionCost: number;
  };
}

class RLTradingAgent {
  private qTable: Map> = new Map();
  private learningRate: number = 0.1;
  private discountFactor: number = 0.95;
  private epsilon: number = 0.1;  // Exploration rate
  private episodeRewards: number[] = [];

  constructor(config?: {
    learningRate?: number;
    discountFactor?: number;
    epsilon?: number;
  }) {
    if (config) {
      this.learningRate = config.learningRate || 0.1;
      this.discountFactor = config.discountFactor || 0.95;
      this.epsilon = config.epsilon || 0.1;
    }
  }

  selectAction(state: State): Action {
    // Epsilon-greedy policy
    if (Math.random() < this.epsilon) {
      // Explore: random action
      return this.randomAction(state);
    } else {
      // Exploit: best known action
      return this.bestAction(state);
    }
  }

  update(
    state: State,
    action: Action,
    reward: Reward,
    nextState: State
  ): void {
    const stateKey = this.stateToKey(state);
    const actionKey = this.actionToKey(action);
    const nextStateKey = this.stateToKey(nextState);

    // Get current Q-value
    const currentQ = this.getQValue(stateKey, actionKey);

    // Find maximum Q-value for next state
    const maxNextQ = this.getMaxQValue(nextStateKey);

    // Q-learning update
    const newQ = currentQ + this.learningRate * (
      reward.value +
      this.discountFactor * maxNextQ -
      currentQ
    );

    // Store updated Q-value
    this.setQValue(stateKey, actionKey, newQ);
  }

  trainEpisode(
    environment: TradingEnvironment,
    maxSteps: number = 1000
  ): number {
    let state = environment.reset();
    let episodeReward = 0;
    let step = 0;

    while (step < maxSteps && !environment.isDone()) {
      // Select and execute action
      const action = this.selectAction(state);
      const result = environment.step(action);

      // Update Q-values
      this.update(state, action, result.reward, result.nextState);

      // Accumulate reward
      episodeReward += result.reward.value;

      // Move to next state
      state = result.nextState;
      step++;
    }

    this.episodeRewards.push(episodeReward);
    return episodeReward;
  }

  train(
    environment: TradingEnvironment,
    numEpisodes: number
  ): TrainingResult {
    console.log(`Training for ${numEpisodes} episodes...`);

    for (let episode = 0; episode < numEpisodes; episode++) {
      const reward = this.trainEpisode(environment);

      // Decay exploration rate
      this.epsilon = Math.max(0.01, this.epsilon * 0.995);

      if ((episode + 1) % 100 === 0) {
        const avgReward = this.episodeRewards
          .slice(-100)
          .reduce((a, b) => a + b) / 100;
        console.log(
          `Episode ${episode + 1}: Avg Reward = ${avgReward.toFixed(2)}, ` +
          `Epsilon = ${this.epsilon.toFixed(4)}`
        );
      }
    }

    return {
      episodeRewards: this.episodeRewards,
      finalEpsilon: this.epsilon,
      qTableSize: this.qTable.size
    };
  }

  private randomAction(state: State): Action {
    const actions: Action['type'][] = ['BUY', 'SELL', 'HOLD'];
    const type = actions[Math.floor(Math.random() * actions.length)];

    if (type === 'HOLD') {
      return { type };
    }

    // Random quantity (simplified)
    const quantity = Math.floor(Math.random() * 100) + 1;
    return { type, quantity };
  }

  private bestAction(state: State): Action {
    const stateKey = this.stateToKey(state);
    const possibleActions = this.getPossibleActions(state);

    let bestAction = possibleActions[0];
    let bestQ = -Infinity;

    for (const action of possibleActions) {
      const actionKey = this.actionToKey(action);
      const q = this.getQValue(stateKey, actionKey);

      if (q > bestQ) {
        bestQ = q;
        bestAction = action;
      }
    }

    return bestAction;
  }

  private getPossibleActions(state: State): Action[] {
    const actions: Action[] = [{ type: 'HOLD' }];

    // Can buy if have cash
    if (state.cash > 0) {
      actions.push({ type: 'BUY', quantity: 10 });
      actions.push({ type: 'BUY', quantity: 50 });
    }

    // Can sell if have position
    if (state.position > 0) {
      actions.push({ type: 'SELL', quantity: 10 });
      actions.push({ type: 'SELL', quantity: 50 });
    }

    return actions;
  }

  private stateToKey(state: State): string {
    // Discretize continuous state into bins
    const priceBin = Math.floor(state.prices[state.prices.length - 1] / 10);
    const positionBin = Math.floor(state.position / 10);
    const cashBin = Math.floor(state.cash / 1000);

    return `${priceBin}_${positionBin}_${cashBin}`;
  }

  private actionToKey(action: Action): string {
    return `${action.type}_${action.quantity || 0}`;
  }

  private getQValue(stateKey: string, actionKey: string): number {
    if (!this.qTable.has(stateKey)) {
      this.qTable.set(stateKey, new Map());
    }

    const stateActions = this.qTable.get(stateKey)!;
    return stateActions.get(actionKey) || 0;
  }

  private setQValue(
    stateKey: string,
    actionKey: string,
    value: number
  ): void {
    if (!this.qTable.has(stateKey)) {
      this.qTable.set(stateKey, new Map());
    }

    this.qTable.get(stateKey)!.set(actionKey, value);
  }

  private getMaxQValue(stateKey: string): number {
    if (!this.qTable.has(stateKey)) {
      return 0;
    }

    const stateActions = this.qTable.get(stateKey)!;
    const values = Array.from(stateActions.values());

    return values.length > 0 ? Math.max(...values) : 0;
  }

  exportModel(): string {
    const modelData = {
      qTable: Array.from(this.qTable.entries()).map(([state, actions]) => ({
        state,
        actions: Array.from(actions.entries())
      })),
      learningRate: this.learningRate,
      discountFactor: this.discountFactor,
      epsilon: this.epsilon,
      episodeRewards: this.episodeRewards
    };

    return JSON.stringify(modelData, null, 2);
  }
}

interface TradingEnvironment {
  reset(): State;
  step(action: Action): {
    nextState: State;
    reward: Reward;
    done: boolean;
  };
  isDone(): boolean;
}

interface TrainingResult {
  episodeRewards: number[];
  finalEpsilon: number;
  qTableSize: number;
}

2. Decentralized Finance (DeFi) Trading

DeFi protocols on blockchain platforms like Ethereum enable algorithmic trading without traditional intermediaries. Automated Market Makers (AMMs), yield farming, and liquidity provision create new opportunities and challenges.

DeFi Trading Characteristics

Aspect Traditional Finance DeFi
Intermediaries Brokers, exchanges, clearinghouses Smart contracts, no intermediaries
Trading Hours Limited (market hours) 24/7/365
Settlement T+1 or T+2 Instant (block confirmation)
Transparency Order books may be hidden All transactions on-chain
Access KYC/AML requirements Permissionless (wallet only)
Risks Counterparty, operational Smart contract, MEV, impermanent loss

MEV and Sandwich Attacks

Maximal Extractable Value (MEV) refers to profit extracted by reordering, including, or excluding transactions in a block. Sandwich attacks place trades before and after victim transactions to profit from price impact. DeFi traders must design strategies resistant to MEV exploitation.

3. Quantum Computing

Quantum computers could revolutionize algorithmic trading through superior optimization of portfolio allocation, faster Monte Carlo simulations, and breaking current cryptographic schemes. However, practical quantum advantage for trading remains years away.

Potential Quantum Trading Applications

  • Portfolio Optimization: Quantum algorithms could solve large-scale optimization problems exponentially faster
  • Option Pricing: Quantum Monte Carlo for faster derivative pricing
  • Pattern Recognition: Quantum machine learning for market pattern detection
  • Cryptography: Quantum-resistant algorithms needed to protect trading systems
  • Arbitrage Detection: Search massive solution spaces for arbitrage opportunities

5. Market Structure Evolution

Market microstructure continues to evolve with maker-taker pricing, speed bumps, frequent batch auctions, and dark pools changing how algorithms interact with markets. Understanding these structural elements is crucial for strategy design.

6. Regulatory Evolution

Regulatory frameworks will continue adapting to technological change. Expected developments include stricter testing requirements, real-time reporting, algorithm registration, and liability frameworks for AI-driven trading decisions.

7. Democratization of Algorithmic Trading

Cloud computing, APIs, and user-friendly platforms are making algorithmic trading accessible to retail traders. This democratization brings both opportunities and risks as more participants deploy automated strategies.

Democratization Drivers

  • Cloud Infrastructure: AWS, Azure, GCP provide scalable computing without capital investment
  • API Accessibility: Brokers and exchanges offer comprehensive APIs to retail traders
  • Educational Resources: Online courses, documentation, and communities lower barriers to entry
  • Open Source Tools: Libraries like ccxt, TA-Lib, and backtrader provide free infrastructure
  • Low-Code Platforms: Visual strategy builders enable non-programmers to create algorithms

8. Ethical Considerations

As algorithms become more sophisticated and autonomous, ethical questions arise: Should AI trading be transparent? Who is liable for algorithm errors? How do we prevent algorithms from exacerbating market instability? These questions will shape future regulations and industry practices.

Reference Standards and Implementation Notes

This standard is designed to interoperate with established international frameworks. Implementations SHOULD treat the following as the single source of truth and use this PHASE document as the interoperability layer that maps existing data into a machine-readable, signed, and auditable shape.

  • OpenAPI Specification 3.1 — REST API definition
  • JSON Schema 2020-12 — schema validation
  • ISO/IEC 27001:2022 — information security management
  • ISO/IEC 27018:2019 — PII protection in public clouds
  • ISO/IEC 17065:2012 — product certification bodies
  • CycloneDX 1.5 / SPDX 2.3 — software bill of materials
  • Sigstore (DSSE envelope, Rekor transparency log) — signature anchoring
  • in-toto Attestation Framework 1.0 — build provenance
  • IETF RFC 9700 — OAuth 2.1
  • IETF RFC 9457 — Problem Details for HTTP APIs
  • IETF RFC 8615 — well-known URIs

Practical Implementation Checklist

  • Manifest integrity — every published manifest is signed via Sigstore DSSE. Tampering is detectable without trusting an intermediary. Updates are issued as new manifests; the prior manifest is preserved with a supersede marker.
  • Retention — keep signed evidence for at least seven years to satisfy ISO/IEC 17065:2012 audit expectations. Some regulators require longer retention; in that case the deployment policy MUST extend the window rather than relying on the defaults.
  • Versioning — follow Semantic Versioning 2.0.0. Major version bumps require at least a 90-day overlap with the prior major version on every WIA-published reference implementation.
  • Negative-result reporting — a skipped indicator without recorded justification is treated as a failed indicator at audit time.
  • Key rotation — manifest signing keys are rotated per the deployment policy. The previous key is preserved in a published key history so that auditors can verify historical manifests offline.

Reference Topologies

Implementations declare their operational topology in the manifest: single-tenant edge (one runtime per organization), multi-tenant gateway (shared runtime with header-based isolation), federated mesh (multiple runtimes peer to one another and publish manifests to a directory service), or air-gapped batch (signed evidence package transported out-of-band). Each topology ships with reference test vectors so that implementations can validate their deployment matches the canonical pattern.

Open Governance

Issues, errata, and proposals are tracked at github.com/WIA-Official/wia-standards/issues with the appropriate label. The WIA Standards working group reviews open issues at the start of every minor release cycle and publishes the resulting decision log alongside the release notes. Errata are issued as patch releases; new normative requirements trigger minor bumps; backwards-incompatible changes trigger major bumps.
