챕터 2: 시스템 아키텍처 및 설계

학습 목표: 탄소 포집 플랜트 설계 원리 습득, 공정 흐름도 이해, 장비 사양 학습, 기존 산업 인프라와의 통합 패턴 이해

탄소 포집 플랜트 개요

탄소 포집 플랜트는 배기가스 스트림에서 CO2를 효율적으로 분리하기 위해 여러 단위 공정을 통합하는 복잡한 산업 시설입니다. 설계는 포집 효율, 에너지 소비, 자본 비용 및 운영 신뢰성 간의 균형을 맞춰야 합니다. 현대 포집 플랜트는 일반적으로 하루에 수백만 입방미터의 배기가스를 처리하면서 90% 이상의 포집률을 달성합니다.

// 탄소 포집 플랜트 구성
interface CaptureCapture {
    plantId: string;
    capacity: {
        flueGasFlow: number;      // Nm³/시간
        CO2CaptureRate: number;   // 톤/일
        captureEfficiency: number; // 퍼센트
    };

    mainUnits: {
        pretreatment: PretreatmentUnit;
        absorber: AbsorberColumn;
        stripper: StripperColumn;
        compression: CompressionTrain;
        utilities: UtilitySystem;
    };

    performance: {
        energyConsumption: number;  // GJ/톤 CO2
        waterConsumption: number;   // m³/톤 CO2
        solventLoss: number;        // kg/톤 CO2
        availability: number;       // 퍼센트
    };
}

공정 흐름 아키텍처

표준 아민 기반 포집 공정은 잘 확립된 흐름 패턴을 따릅니다. 이 아키텍처를 이해하는 것은 포집 시설의 적절한 설계와 운영에 필수적입니다.

흡수탑 설계

흡수탑은 배기가스에서 CO2가 포집되는 주요 물질 전달 장비입니다. 적절한 설계는 압력 강하와 용매 손실을 최소화하면서 높은 포집 효율을 보장합니다.

// 흡수탑 설계 사양
interface AbsorberDesign {
    geometry: {
        internalDiameter: number;  // 미터
        totalHeight: number;       // 미터
        packingHeight: number;     // 미터
        numberOfBeds: number;
        bedHeight: number;         // 베드당 미터
    };

    packing: {
        type: "structured" | "random";
        material: string;
        specificArea: number;      // m²/m³
        voidFraction: number;      // 무차원
        pressureDrop: number;      // mbar/m
    };

    internals: {
        liquidDistributors: number;
        redistributors: number;
        supportPlates: number;
        holdDownPlates: number;
    };

    operatingLimits: {
        maxGasVelocity: number;    // m/s
        minLiquidRate: number;     // m³/m²/h
        floodingMargin: number;    // 플러딩 이하 퍼센트
    };
}

// 예시: 500 MW 발전소 흡수탑
const absorberSpec: AbsorberDesign = {
    geometry: {
        internalDiameter: 15.5,
        totalHeight: 45,
        packingHeight: 24,
        numberOfBeds: 4,
        bedHeight: 6
    },
    packing: {
        type: "structured",
        material: "스테인레스 스틸 316L",
        specificArea: 250,
        voidFraction: 0.97,
        pressureDrop: 25
    },
    internals: {
        liquidDistributors: 5,
        redistributors: 3,
        supportPlates: 4,
        holdDownPlates: 4
    },
    operatingLimits: {
        maxGasVelocity: 2.8,
        minLiquidRate: 15,
        floodingMargin: 30
    }
};

스트리퍼 컬럼 및 재비기 시스템

스트리퍼 컬럼은 재비기에서 공급되는 열을 사용하여 흡수된 CO2를 방출함으로써 리치 아민 용액을 재생합니다. 이 유닛은 포집 플랜트에서 가장 큰 에너지 소비자를 나타냅니다.

에너지 최적화: 재비기 듀티는 일반적으로 포집 플랜트 총 에너지 소비의 60-70%를 차지합니다. 고급 용매, 공정 수정 및 열 통합을 통해 재비기 듀티를 줄이는 것이 포집 비용을 낮추는 주요 경로입니다. 
// 스트리퍼 및 재비기 사양
interface StripperSystem {
    column: {
        diameter: number;           // 미터
        height: number;             // 미터
        packingHeight: number;      // 미터
        operatingPressure: number;  // bar
        topTemperature: number;     // °C
        bottomTemperature: number;  // °C
    };

    reboiler: {
        type: "kettle" | "thermosiphon" | "forcedCirculation";
        duty: number;               // MW
        steamPressure: number;      // bar
        steamTemperature: number;   // °C
        heatTransferArea: number;   // m²
        U_coefficient: number;      // W/m²/K
    };

    condenser: {
        duty: number;               // MW
        coolingMedium: string;
        coolantTemperature: number; // °C
        refluxRatio: number;        // 무차원
    };
}

// 재비기 듀티 계산
function calculateReboilerDuty(
    CO2CaptureRate: number,      // 톤/시간
    specificDuty: number,        // GJ/톤 CO2
    heatExchangerEfficiency: number  // 분율
): number {
    const grossDuty = CO2CaptureRate * specificDuty * 1000 / 3600;  // MW
    const netDuty = grossDuty / heatExchangerEfficiency;
    return netDuty;  // MW
}

압축 트레인 설계

CO2 압축 트레인은 포집된 CO2의 압력을 대기압 근처에서 파이프라인 운송 압력(일반적으로 100-150 bar)으로 높입니다. 에너지 소비를 최소화하기 위해 인터쿨링이 있는 다단 압축이 사용됩니다.

// CO2 압축 트레인 구성
interface CompressionTrain {
    stages: CompressionStage[];
    totalPower: number;          // MW
    overallEfficiency: number;   // 퍼센트
    coolingDuty: number;         // MW
}

interface CompressionStage {
    stageNumber: number;
    compressorType: "centrifugal" | "reciprocating" | "integrallygeared";
    inletPressure: number;       // bar
    outletPressure: number;      // bar
    compressionRatio: number;    // 무차원
    polytropicEfficiency: number; // 퍼센트
    power: number;               // MW
    intercooler: {
        outletTemperature: number;  // °C
        duty: number;               // MW
    };
}

// 4단 압축 트레인 예시
const compressionTrain: CompressionTrain = {
    stages: [
        { stageNumber: 1, compressorType: "centrifugal", inletPressure: 1.5,
          outletPressure: 5, compressionRatio: 3.33, polytropicEfficiency: 82,
          power: 4.5, intercooler: { outletTemperature: 35, duty: 3.8 }},
        { stageNumber: 2, compressorType: "centrifugal", inletPressure: 5,
          outletPressure: 15, compressionRatio: 3.0, polytropicEfficiency: 84,
          power: 3.8, intercooler: { outletTemperature: 35, duty: 3.2 }},
        { stageNumber: 3, compressorType: "centrifugal", inletPressure: 15,
          outletPressure: 45, compressionRatio: 3.0, polytropicEfficiency: 85,
          power: 3.2, intercooler: { outletTemperature: 35, duty: 2.8 }},
        { stageNumber: 4, compressorType: "centrifugal", inletPressure: 45,
          outletPressure: 150, compressionRatio: 3.33, polytropicEfficiency: 85,
          power: 3.5, intercooler: { outletTemperature: 35, duty: 0 }}
    ],
    totalPower: 15,
    overallEfficiency: 84,
    coolingDuty: 9.8
};

열 통합 전략

효과적인 열 통합은 탄소 포집의 에너지 페널티를 크게 줄일 수 있습니다. 다양한 공정 스트림에서 열을 회수하고 호스트 플랜트의 열 사이클과 통합함으로써 전체 효율을 상당히 개선할 수 있습니다.

열원 온도 (°C) 잠재적 용도 에너지 절감
린 아민 쿨러 50-80 리치 아민 예열 5-10%
CO2 압축기 인터쿨러 80-150 급수 가열 3-5%
스트리퍼 오버헤드 응축기 80-100 리치 아민 예열 8-12%
배기가스 (포집 전) 120-180 용매 재생 10-15% 
// 열교환기 네트워크 최적화
interface HeatExchangerNetwork {
    exchangers: HeatExchanger[];
    totalHeatRecovery: number;    // MW
    minimumApproach: number;      // °C
    annualEnergySaving: number;   // GJ/년
}

interface HeatExchanger {
    id: string;
    hotStream: string;
    coldStream: string;
    duty: number;                 // MW
    LMTD: number;                 // °C
    area: number;                 // m²
    U_coefficient: number;        // W/m²/K
}

// 열교환기 면적 계산
function calculateHeatExchangerArea(
    duty: number,         // MW
    U: number,            // W/m²/K
    LMTD: number          // °C
): number {
    return (duty * 1e6) / (U * LMTD);  // m²
}

계측 및 제어 아키텍처

현대 탄소 포집 플랜트는 최적 운영을 유지하고 안전을 보장하기 위해 정교한 분산 제어 시스템(DCS)을 사용합니다. 제어 아키텍처는 수백 개의 제어 루프를 관리하고 플랜트 전체 시스템과 인터페이스합니다.

제어 철학: 탄소 포집 플랜트는 일반적으로 기본 계층 조절 제어, 감독 고급 공정 제어(APC), 실시간 최적화(RTO) 계층으로 구성된 계층적 제어 구조를 사용하여 안전한 운영을 유지하면서 효율을 극대화합니다. 
// 제어 시스템 아키텍처
interface ControlArchitecture {
    dcs: {
        vendor: string;
        redundancy: "dual" | "triple";
        scanRate: number;           // 밀리초
        ioCount: number;
    };

    controlLoops: {
        flow: number;
        pressure: number;
        temperature: number;
        level: number;
        analytical: number;
        total: number;
    };

    apc: {
        enabled: boolean;
        vendor: string;
        controlledVariables: number;
        manipulatedVariables: number;
        updateRate: number;         // 초
    };

    rto: {
        enabled: boolean;
        optimizationInterval: number;  // 분
        objectiveFunction: string;
    };

    safetySystem: {
        silLevel: 1 | 2 | 3;
        tripPoints: number;
        emergencyShutdowns: number;
    };
}

요약

이 챕터에서는 공정 흐름 설계부터 장비 사양, 제어 시스템 및 부지 배치까지 탄소 포집 시스템의 포괄적인 아키텍처를 다루었습니다. 이러한 구성 요소를 응집력 있고 효율적인 시설로 통합하려면 신중한 엔지니어링과 시스템 간 인터페이스에 대한 주의가 필요합니다.
