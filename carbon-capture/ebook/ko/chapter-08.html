// 에너지 효율 비교
function compareEnergyEfficiency(
    thermalProcess: number,    // kJ/mol CO2
    electrochemicalProcess: number,  // kJ 전기/mol CO2
    gridCarbonIntensity: number      // kg CO2/kWh
): ComparisonResult {
    // 전기를 등가 열로 변환
    const thermalEquivalent = electrochemicalProcess / 0.4;  // 40% 발전 효율

    // 발전 배출을 고려한 순 CO2 포집
    const netCaptureThermal = 1 - (thermalProcess / 1000 / 44) * gridCarbonIntensity * 0.277;
    const netCaptureElectro = 1 - (electrochemicalProcess / 1000 / 44) * gridCarbonIntensity * 0.277;

    return {
        thermalEfficiency: netCaptureThermal * 100,
        electrochemicalEfficiency: netCaptureElectro * 100,
        recommendation: gridCarbonIntensity < 0.2 ? "전기화학적 선호" : "열적이 더 나을 수 있음"
    };
}

// 프로젝트 비용 계산기
function calculateLCOC(
    capex: number,              // $/톤 CO2/년 용량
    fixedOpex: number,          // $/톤 CO2/년
    variableOpex: number,       // $/톤 CO2
    capacityFactor: number,     // 퍼센트
    discountRate: number,       // 퍼센트
    lifetime: number            // 년
): number {
    const CRF = (discountRate * Math.pow(1 + discountRate, lifetime)) /
                (Math.pow(1 + discountRate, lifetime) - 1);
    const annualizedCapex = capex * CRF;
    const LCOC = (annualizedCapex + fixedOpex) / (capacityFactor / 100) + variableOpex;
    return LCOC;  // $/톤 CO2
}

요약

탄소 포집의 미래는 고급 소재, 혁신적인 프로세스, 비용 절감, 전례 없는 규모의 글로벌 배포에 있습니다. 이 챕터에서는 이 비전을 실현하는 데 필요한 기술 경로, 경제적 동인 및 전략적 프레임워크를 설명했습니다. 성공을 위해서는 탄소 포집 기술에 투자하고, 배포하고, 지속적으로 개선하기 위한 정부, 산업 및 사회의 지속적인 노력이 필요합니다.
