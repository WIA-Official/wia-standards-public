// Energy Efficiency Comparison
function compareEnergyEfficiency(
    thermalProcess: number,    // kJ/mol CO2
    electrochemicalProcess: number,  // kJ electrical/mol CO2
    gridCarbonIntensity: number      // kg CO2/kWh
): ComparisonResult {
    // Convert electrical to equivalent thermal
    const thermalEquivalent = electrochemicalProcess / 0.4;  // 40% power plant efficiency

    // Net CO2 captured accounting for power generation emissions
    const netCaptureThermal = 1 - (thermalProcess / 1000 / 44) * gridCarbonIntensity * 0.277;
    const netCaptureElectro = 1 - (electrochemicalProcess / 1000 / 44) * gridCarbonIntensity * 0.277;

    return {
        thermalEfficiency: netCaptureThermal * 100,
        electrochemicalEfficiency: netCaptureElectro * 100,
        recommendation: gridCarbonIntensity < 0.2 ? "Electrochemical preferred" : "Thermal may be better"
    };
}

// Project Cost Calculator
function calculateLCOC(
    capex: number,              // $/tonne CO2/year capacity
    fixedOpex: number,          // $/tonne CO2/year
    variableOpex: number,       // $/tonne CO2
    capacityFactor: number,     // percent
    discountRate: number,       // percent
    lifetime: number            // years
): number {
    const CRF = (discountRate * Math.pow(1 + discountRate, lifetime)) /
                (Math.pow(1 + discountRate, lifetime) - 1);
    const annualizedCapex = capex * CRF;
    const LCOC = (annualizedCapex + fixedOpex) / (capacityFactor / 100) + variableOpex;
    return LCOC;  // $/tonne CO2
}
