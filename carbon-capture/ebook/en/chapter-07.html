// Calculate Capture Efficiency
function calculateCaptureEfficiency(
    inletCO2Flow: number,    // Nm³/hr
    outletCO2Flow: number    // Nm³/hr
): number {
    return ((inletCO2Flow - outletCO2Flow) / inletCO2Flow) * 100;
}

// Calculate Solvent Health Index
function calculateSolventHealth(
    HSSContent: number,      // wt%
    ironContent: number,     // ppm
    foamingIndex: number     // relative scale
): number {
    const HSSScore = Math.max(0, 100 - (HSSContent / 0.05));
    const ironScore = Math.max(0, 100 - (ironContent / 1));
    const foamScore = Math.max(0, 100 - (foamingIndex * 10));

    return (HSSScore + ironScore + foamScore) / 3;  // 0-100 scale
}
Solvent Health Indicator: A Solvent Health Index above 80 indicates good condition. Between 60-80 requires increased monitoring. Below 60 indicates immediate corrective action is needed, potentially including partial or full solvent replacement.

// Energy Optimization Calculation
function optimizeEnergy(
    captureTarget: number,      // tonnes/hr
    steamCost: number,          // $/GJ
    electricityCost: number,    // $/kWh
    currentOperation: OperatingPoint
): OptimizedSetpoints {
    // Simplified optimization logic
    const optimalLG = findOptimalLGRatio(captureTarget);
    const optimalLeanLoading = findOptimalLeanLoading(steamCost);

    return {
        solventFlow: optimalLG * currentOperation.flueGasFlow,
        reboilerDuty: calculateReboilerDuty(optimalLeanLoading),
        expectedCost: calculateOperatingCost(steamCost, electricityCost),
        expectedEfficiency: captureTarget / currentOperation.CO2Inlet * 100
    };
}
