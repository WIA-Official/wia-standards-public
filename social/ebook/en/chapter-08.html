8.1 Installation

NPM Installation
================

npm install @anthropic/wia-social

Or yarn:
yarn add @anthropic/wia-social

Or pnpm:
pnpm add @anthropic/wia-social


Basic Setup
===========

import { WIASocial } from '@anthropic/wia-social';

// Create client
const social = new WIASocial({
  identity: {
    displayName: 'Hong Gildong',
    email: 'hong@example.com'
  }
});

// Connect platforms
await social.connectPlatform('instagram', instagramToken);
await social.connectPlatform('twitter', twitterToken);

// Ready to use!
console.log('Connected:', social.connectedPlatforms);


Configuration Options
=====================

const social = new WIASocial({
  // Identity
  identity: {
    displayName: 'Hong Gildong',
    email: 'hong@example.com'
  },

  // Platforms
  platforms: ['instagram', 'twitter', 'bluesky'],

  // Privacy (Aunt integration)
  privacy: {
    airShield: {
      enabled: true,
      mode: 'balanced'
    },
    defaultVisibility: 'protected'
  },

  // Sync settings
  sync: {
    realtime: true,
    interval: 60  // seconds
  }
});

8.3 Platform Adapters

Adapter Interface
=================

interface PlatformAdapter {
  // Platform info
  platform: Platform;
  version: string;
  capabilities: Capabilities;

  // Authentication
  authenticate(credentials): Promise<Token>;
  refreshToken(token): Promise<Token>;

  // Posting
  createPost(post): Promise<PostId>;
  updatePost(id, post): Promise<void>;
  deletePost(id): Promise<void>;

  // Feed
  getFeed(options): Promise<Post[]>;
  convertToUniversal(post): UniversalPost;

  // Social
  getFollowers(): Promise<User[]>;
  getFollowing(): Promise<User[]>;
  follow(userId): Promise<void>;
  unfollow(userId): Promise<void>;

  // Messaging
  sendMessage(to, message): Promise<void>;
  getMessages(): Promise<Message[]>;

  // Data
  exportData(): Promise<ExportPackage>;
  importData(data): Promise<void>;
}

Custom Adapter
==============

// Create adapter for new platform
class MyPlatformAdapter implements PlatformAdapter {
  platform = 'myplatform';
  version = '1.0.0';

  async createPost(post: UniversalPost) {
    // Convert to platform format
    const platformPost = this.convert(post);
    // Call platform API
    const result = await myPlatformApi.post(platformPost);
    return result.id;
  }

  // ... implement other methods
}

// Register adapter
social.registerAdapter(new MyPlatformAdapter());

8.4 Protocol Bridges

ActivityPub Bridge (Mastodon, etc.)
===================================

import { ActivityPubBridge } from '@anthropic/wia-social/protocols';

const bridge = new ActivityPubBridge();

// Convert WIA post to ActivityPub Activity
const activity = bridge.toActivity(universalPost);

// Convert ActivityPub to WIA format
const wiaPost = bridge.fromActivity(activity);


AT Protocol Bridge (Bluesky)
============================

import { ATProtocolBridge } from '@anthropic/wia-social/protocols';

const bridge = new ATProtocolBridge();

// Convert to AT Protocol record
const record = bridge.toRecord(universalPost);

// Convert from AT Protocol
const wiaPost = bridge.fromRecord(record);

8.5 Family Integration

Father (INTENT) Integration
===========================

import { Intent } from '@anthropic/wia-intent';
import { WIASocial } from '@anthropic/wia-social';

const father = new Intent();
const social = new WIASocial();

// Intent-based posting
const intent = await father.express(`
  Share my lunch photo with close friends.
  Casual tone, no location.
`);

// SOCIAL processes the intent
await social.executeIntent(intent);


Mother (OMNI-API) Integration
=============================

import { OmniApi } from '@anthropic/wia-omni-api';

// OMNI-API handles all platform APIs
const mother = new OmniApi();

// SOCIAL uses OMNI-API for unified access
social.setApiLayer(mother);


Aunt (AIR-SHIELD) Integration
=============================

import { AirShield } from '@anthropic/wia-air-shield';

const aunt = new AirShield({ mode: 'balanced' });

// Enable privacy protection
social.setPrivacyGuard(aunt);

// Now all posts are scanned
await social.post({
  content: myContent
  // Aunt automatically scans before posting
});


Uncle (AIR-POWER) Integration
=============================

import { AirPower } from '@anthropic/wia-air-power';

// Ensure posting even on low battery
const uncle = new AirPower();

social.setPowerManager(uncle);

// Uncle ensures reliable delivery
await social.post({
  content: { text: 'Important update!' },
  priority: 'high'  // Uncle prioritizes
});


Full Family Example
===================

import { Intent } from '@anthropic/wia-intent';
import { OmniApi } from '@anthropic/wia-omni-api';
import { AirPower } from '@anthropic/wia-air-power';
import { AirShield } from '@anthropic/wia-air-shield';
import { WIASocial } from '@anthropic/wia-social';

// The WIA Family
const family = {
  father: new Intent(),
  mother: new OmniApi(),
  uncle: new AirPower(),
  aunt: new AirShield(),
  nephew: new WIASocial()
};

// Configure nephew with family
family.nephew.setFamily(family);

// Intent-based secure posting
await family.father.express(`
  Announce my new job to professional contacts.
  Hide my salary details.
`);

// Flow:
// 1. Father understands intent
// 2. Aunt scans for privacy (salary)
// 3. Mother routes to APIs
// 4. Uncle ensures delivery
// 5. Nephew posts to LinkedIn, Twitter

8.6 TypeScript Support

Full Type Definitions
=====================

import {
  WIASocial,
  UniversalIdentity,
  UniversalPost,
  UnifiedFeed,
  Platform,
  PrivacySettings,
  PlatformAdapter
} from '@anthropic/wia-social';

// All methods fully typed
const social = new WIASocial(config);

// Type-safe posting
const post: UniversalPost = {
  content: {
    text: 'Hello!',
    media: []
  },
  targets: ['instagram', 'twitter']
};

await social.post(post);

// Type-safe feed
const feed: UnifiedFeed = await social.getFeed({
  sorting: 'chronological'
});

// TypeScript catches errors
await social.post({
  content: { text: 'Hello!' },
  targets: ['invalid_platform']  // Error!
});
