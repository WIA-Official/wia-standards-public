1.5 Quick Start

Installation
============

npm install @anthropic/wia-social


Basic Usage
===========

import { WIASocial } from '@anthropic/wia-social';

// Create your identity
const social = new WIASocial({
  identity: {
    displayName: 'Hong Gildong',
    email: 'hong@example.com'
  }
});

// Connect platforms
await social.connectPlatform('instagram', token1);
await social.connectPlatform('twitter', token2);
await social.connectPlatform('tiktok', token3);

// Post to all at once!
await social.post({
  content: {
    text: 'Hello from WIA-SOCIAL!',
    media: [myPhoto]
  },
  targets: ['instagram', 'twitter', 'tiktok']
});

// Get unified feed
const feed = await social.getFeed();

// Find friend across platforms
const friend = await social.findFriendAcrossPlatforms('Kim Minsu');
