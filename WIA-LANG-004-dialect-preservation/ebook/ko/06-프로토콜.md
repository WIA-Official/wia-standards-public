### 6.1.2 요청/응답 패턴

```python
# HTTP 요청/응답 프로토콜 구현
from typing import Dict, Optional, Any
from dataclasses import dataclass
from datetime import datetime
import hashlib
import json

@dataclass
class HTTPRequest:
    """HTTP 요청 구조"""
    method: str  # GET, POST, PUT, PATCH, DELETE
    uri: str
    headers: Dict[str, str]
    body: Optional[Any] = None
    query_params: Optional[Dict[str, str]] = None

    def to_dict(self) -> Dict:
        """딕셔너리로 변환"""
        return {
            "method": self.method,
            "uri": self.uri,
            "headers": self.headers,
            "body": self.body,
            "query_params": self.query_params
        }

@dataclass
class HTTPResponse:
    """HTTP 응답 구조"""
    status_code: int
    headers: Dict[str, str]
    body: Any
    timestamp: datetime = None

    def __post_init__(self):
        if self.timestamp is None:
            self.timestamp = datetime.now()

    def to_dict(self) -> Dict:
        """딕셔너리로 변환"""
        return {
            "status_code": self.status_code,
            "headers": self.headers,
            "body": self.body,
            "timestamp": self.timestamp.isoformat()
        }

class RequestResponseProtocol:
    """요청/응답 프로토콜 핸들러"""

    # HTTP 상태 코드 매핑
    STATUS_CODES = {
        # 2xx Success
        200: "OK",
        201: "Created",
        202: "Accepted",
        204: "No Content",

        # 3xx Redirection
        301: "Moved Permanently",
        304: "Not Modified",

        # 4xx Client Error
        400: "Bad Request",
        401: "Unauthorized",
        403: "Forbidden",
        404: "Not Found",
        409: "Conflict",
        422: "Unprocessable Entity",
        429: "Too Many Requests",

        # 5xx Server Error
        500: "Internal Server Error",
        502: "Bad Gateway",
        503: "Service Unavailable",
        504: "Gateway Timeout"
    }

    @staticmethod
    def create_request(method: str, uri: str, api_key: str,
                      body: Optional[Dict] = None,
                      query_params: Optional[Dict] = None) -> HTTPRequest:
        """표준 HTTP 요청 생성"""

        headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json",
            "Accept": "application/ld+json",
            "User-Agent": "WIA-LANG-004-Client/1.0",
            "X-Request-ID": RequestResponseProtocol._generate_request_id()
        }

        if body:
            # Content-Length 자동 계산
            body_json = json.dumps(body, ensure_ascii=False)
            headers["Content-Length"] = str(len(body_json.encode('utf-8')))

            # Digest (무결성 검증)
            digest = hashlib.sha256(body_json.encode('utf-8')).hexdigest()
            headers["Digest"] = f"sha-256={digest}"

        return HTTPRequest(
            method=method,
            uri=uri,
            headers=headers,
            body=body,
            query_params=query_params
        )

    @staticmethod
    def create_response(status_code: int, body: Any,
                       request_id: Optional[str] = None) -> HTTPResponse:
        """표준 HTTP 응답 생성"""

        headers = {
            "Content-Type": "application/ld+json; charset=utf-8",
            "X-Content-Type-Options": "nosniff",
            "X-Frame-Options": "DENY",
            "X-XSS-Protection": "1; mode=block"
        }

        if request_id:
            headers["X-Request-ID"] = request_id

        # ETag 생성 (캐싱용)
        if body:
            body_json = json.dumps(body, ensure_ascii=False, sort_keys=True)
            etag = hashlib.md5(body_json.encode('utf-8')).hexdigest()
            headers["ETag"] = f'W/"{etag}"'

        # CORS 헤더
        headers["Access-Control-Allow-Origin"] = "*"
        headers["Access-Control-Allow-Methods"] = "GET, POST, PUT, PATCH, DELETE, OPTIONS"
        headers["Access-Control-Allow-Headers"] = "Authorization, Content-Type"

        return HTTPResponse(
            status_code=status_code,
            headers=headers,
            body=body
        )

    @staticmethod
    def _generate_request_id() -> str:
        """고유 요청 ID 생성"""
        import uuid
        return str(uuid.uuid4())

    @staticmethod
    def create_error_response(status_code: int, error_code: str,
                            message: str, details: Optional[Dict] = None) -> HTTPResponse:
        """에러 응답 생성"""

        error_body = {
            "@context": "https://wia-lang.org/ns/lang-004/errors",
            "@type": "Error",
            "status": status_code,
            "error_code": error_code,
            "message": message,
            "timestamp": datetime.now().isoformat()
        }

        if details:
            error_body["details"] = details

        return RequestResponseProtocol.create_response(status_code, error_body)

# 사용 예시
protocol = RequestResponseProtocol()

# 방언 조회 요청 생성
request = protocol.create_request(
    method="GET",
    uri="/v1/dialects/ko-jeju",
    api_key="your_api_key"
)

print("=== HTTP 요청 ===")
print(json.dumps(request.to_dict(), indent=2, ensure_ascii=False))

# 성공 응답 생성
success_response = protocol.create_response(
    status_code=200,
    body={
        "@context": "https://wia-lang.org/ns/lang-004/context.json",
        "@type": "Dialect",
        "@id": "wia-lang://dialects/ko-jeju",
        "name": "제주 방언",
        "endangermentStatus": "severely_endangered"
    },
    request_id=request.headers.get("X-Request-ID")
)

print("\n=== HTTP 응답 (성공) ===")
print(json.dumps(success_response.to_dict(), indent=2, ensure_ascii=False))

# 에러 응답 생성
error_response = protocol.create_error_response(
    status_code=404,
    error_code="DIALECT_NOT_FOUND",
    message="지정한 방언을 찾을 수 없습니다",
    details={
        "dialect_id": "ko-invalid",
        "suggestion": "방언 ID를 확인하세요"
```

## 6.2 데이터 동기화 프로토콜

### 6.2.1 분산 저장소 동기화

```python
# 분산 저장소 동기화 프로토콜
from typing import List, Dict, Optional
from dataclasses import dataclass
from datetime import datetime
from enum import Enum

class SyncOperation(Enum):
    """동기화 작업 유형"""
    CREATE = "create"
    UPDATE = "update"
    DELETE = "delete"
    MERGE = "merge"

@dataclass
class SyncRecord:
    """동기화 레코드"""
    resource_id: str
    resource_type: str  # 'dialect', 'recording', 'transcription'
    operation: SyncOperation
    version: int
    timestamp: datetime
    checksum: str
    data: Optional[Dict] = None

class SyncProtocol:
    """동기화 프로토콜"""

    def __init__(self, node_id: str):
        self.node_id = node_id
        self.sync_log: List[SyncRecord] = []
        self.last_sync_timestamp: Optional[datetime] = None

    def record_change(self, resource_id: str, resource_type: str,
                     operation: SyncOperation, data: Dict) -> SyncRecord:
        """변경사항 기록"""

        # 체크섬 계산
        import json
        import hashlib
        data_json = json.dumps(data, sort_keys=True, ensure_ascii=False)
        checksum = hashlib.sha256(data_json.encode('utf-8')).hexdigest()

        # 버전 계산 (이전 버전 + 1)
        existing_records = [
            r for r in self.sync_log
            if r.resource_id == resource_id
        ]
        version = max([r.version for r in existing_records], default=0) + 1

        # 동기화 레코드 생성
        record = SyncRecord(
            resource_id=resource_id,
            resource_type=resource_type,
            operation=operation,
            version=version,
            timestamp=datetime.now(),
            checksum=checksum,
            data=data
        )

        self.sync_log.append(record)
        return record

    def get_changes_since(self, timestamp: datetime) -> List[SyncRecord]:
        """특정 시점 이후 변경사항 조회"""
        return [
            record for record in self.sync_log
            if record.timestamp > timestamp
        ]

    def generate_sync_request(self, target_node: str) -> Dict:
        """동기화 요청 생성"""
        return {
            "@context": "https://wia-lang.org/ns/lang-004/sync",
            "@type": "SyncRequest",
            "source_node": self.node_id,
            "target_node": target_node,
            "last_sync": self.last_sync_timestamp.isoformat() if self.last_sync_timestamp else None,
            "request_time": datetime.now().isoformat()
        }

    def process_sync_response(self, response: Dict) -> List[SyncRecord]:
        """동기화 응답 처리"""
        changes = response.get("changes", [])
        processed = []

        for change_data in changes:
            record = SyncRecord(
                resource_id=change_data["resource_id"],
                resource_type=change_data["resource_type"],
                operation=SyncOperation(change_data["operation"]),
                version=change_data["version"],
                timestamp=datetime.fromisoformat(change_data["timestamp"]),
                checksum=change_data["checksum"],
                data=change_data.get("data")
            )

            # 충돌 검사
            conflict = self._check_conflict(record)
            if conflict:
                resolved = self._resolve_conflict(record, conflict)
                processed.append(resolved)
            else:
                processed.append(record)
                self.sync_log.append(record)

        # 마지막 동기화 시간 업데이트
        self.last_sync_timestamp = datetime.now()

        return processed

    def _check_conflict(self, incoming: SyncRecord) -> Optional[SyncRecord]:
        """충돌 검사"""
        for local_record in reversed(self.sync_log):
            if (local_record.resource_id == incoming.resource_id and
                local_record.timestamp > incoming.timestamp):
                # 더 최신 로컬 변경사항이 있음
                return local_record
        return None

    def _resolve_conflict(self, incoming: SyncRecord,
                         local: SyncRecord) -> SyncRecord:
        """충돌 해결"""

        # 충돌 해결 전략: 타임스탬프 기반 "최신 승리"
        if incoming.timestamp > local.timestamp:
            # 원격이 더 최신
            print(f"충돌 해결: 원격 버전 채택 ({incoming.resource_id})")
            return incoming
        else:
            # 로컬이 더 최신
            print(f"충돌 해결: 로컬 버전 유지 ({local.resource_id})")
            return local

# 사용 예시
node_a = SyncProtocol(node_id="node-seoul")
node_b = SyncProtocol(node_id="node-jeju")

# Node A에서 제주 방언 데이터 생성
record_a = node_a.record_change(
    resource_id="wia-lang://dialects/ko-jeju",
    resource_type="dialect",
    operation=SyncOperation.CREATE,
    data={
        "name": "제주 방언",
        "endangermentStatus": "severely_endangered",
        "speakers": {"estimatedCount": 5000}
    }
)

print(f"Node A 변경: {record_a.operation.value} @ {record_a.timestamp}")

# Node B에서 동기화 요청
sync_request = node_b.generate_sync_request(target_node="node-seoul")
print(f"\n동기화 요청:")
print(json.dumps(sync_request, indent=2, ensure_ascii=False))

# Node A가 변경사항 전송
import time
time.sleep(0.1)  # 시간 차이 시뮬레이션

if node_b.last_sync_timestamp:
    changes_since = node_a.get_changes_since(node_b.last_sync_timestamp)
else:
    changes_since = node_a.sync_log

sync_response = {
    "@context": "https://wia-lang.org/ns/lang-004/sync",
    "@type": "SyncResponse",
    "source_node": "node-seoul",
    "target_node": "node-jeju",
    "changes": [
        {
            "resource_id": r.resource_id,
            "resource_type": r.resource_type,
            "operation": r.operation.value,
            "version": r.version,
            "timestamp": r.timestamp.isoformat(),
            "checksum": r.checksum,
            "data": r.data
        }
        for r in changes_since
    ]
}

print(f"\n동기화 응답: {len(sync_response['changes'])}개 변경사항")

# Node B가 변경사항 적용
processed = node_b.process_sync_response(sync_response)
print(f"\n동기화 완료: {len(processed)}개 레코드 처리")
```

```python
# Server-Sent Events (SSE) 프로토콜
from typing import Generator, Dict
import json
import time

class SSEProtocol:
    """Server-Sent Events 프로토콜"""

    @staticmethod
    def format_sse_message(event_type: str, data: Dict, event_id: Optional[str] = None) -> str:
        """SSE 메시지 형식화"""
        lines = []

        if event_id:
            lines.append(f"id: {event_id}")

        lines.append(f"event: {event_type}")

        # 데이터를 JSON으로 직렬화
        data_json = json.dumps(data, ensure_ascii=False)

        # 여러 줄인 경우 각 줄에 'data:' 접두사
        for line in data_json.split('\n'):
            lines.append(f"data: {line}")

        # 메시지는 빈 줄로 종료
        lines.append("")
        lines.append("")

        return '\n'.join(lines)

    @staticmethod
    def stream_events(channel: str, filters: Dict = None) -> Generator[str, None, None]:
        """이벤트 스트림 생성 (예시)"""

        # 연결 확인 메시지
        yield SSEProtocol.format_sse_message(
            event_type="connected",
            data={
                "channel": channel,
                "message": "연결됨",
                "timestamp": datetime.now().isoformat()
            },
            event_id="conn-1"
        )

        # 실제로는 데이터베이스나 메시지 큐에서 이벤트를 가져옴
        # 여기서는 시뮬레이션
        event_id = 1
        while True:
            time.sleep(5)  # 5초마다

            # 예시 이벤트
            yield SSEProtocol.format_sse_message(
                event_type="heartbeat",
                data={
                    "message": "연결 유지",
                    "timestamp": datetime.now().isoformat()
                },
                event_id=f"evt-{event_id}"
            )

            event_id += 1

            # 실제 이벤트 예시

                        "annotation_type": "phonetic",
                        "annotator": "linguist_park",
                        "timestamp": datetime.now().isoformat()
                    },
                    event_id=f"evt-{event_id}"
                )
```

### 6.4.2 데이터 무결성 검증

```python
# 데이터 무결성 검증 프로토콜
import hashlib
import hmac
from typing import Dict, Tuple

class IntegrityProtocol:
    """데이터 무결성 검증"""

    @staticmethod
    def calculate_checksum(data: str, algorithm: str = "sha256") -> str:
        """체크섬 계산"""
        if algorithm == "sha256":
            return hashlib.sha256(data.encode('utf-8')).hexdigest()
        elif algorithm == "sha512":
            return hashlib.sha512(data.encode('utf-8')).hexdigest()
        elif algorithm == "md5":
            # MD5는 암호학적으로 안전하지 않으므로 비권장
            return hashlib.md5(data.encode('utf-8')).hexdigest()
        else:
            raise ValueError(f"지원하지 않는 알고리즘: {algorithm}")

    @staticmethod
    def generate_signature(data: str, secret_key: str) -> str:
        """HMAC 서명 생성"""
        return hmac.new(
            secret_key.encode('utf-8'),
            data.encode('utf-8'),
            hashlib.sha256
        ).hexdigest()

    @staticmethod
    def verify_signature(data: str, signature: str, secret_key: str) -> bool:
        """HMAC 서명 검증"""
        expected_signature = IntegrityProtocol.generate_signature(data, secret_key)
        return hmac.compare_digest(signature, expected_signature)

    @staticmethod
    def package_with_integrity(data: Dict, secret_key: str) -> Tuple[Dict, str]:
        """무결성 정보와 함께 데이터 패키징"""
        import json

        # 데이터를 JSON으로 직렬화 (정렬된 키로)
        data_json = json.dumps(data, sort_keys=True, ensure_ascii=False)

        # 체크섬 계산
        checksum = IntegrityProtocol.calculate_checksum(data_json)

        # HMAC 서명 생성
        signature = IntegrityProtocol.generate_signature(data_json, secret_key)

        # 패키지
        package = {
            "data": data,
            "integrity": {
                "checksum": checksum,
                "algorithm": "sha256",
                "signature": signature
            }
        }

        return package, data_json

    @staticmethod
    def verify_package(package: Dict, secret_key: str) -> Tuple[bool, str]:
        """패키지 무결성 검증"""
        import json

        try:
            data = package["data"]
            integrity = package["integrity"]

            # 데이터 재직렬화
            data_json = json.dumps(data, sort_keys=True, ensure_ascii=False)

            # 체크섬 검증
            expected_checksum = IntegrityProtocol.calculate_checksum(data_json)
            if integrity["checksum"] != expected_checksum:
                return False, "체크섬 불일치"

            # 서명 검증
            if not IntegrityProtocol.verify_signature(
                data_json,
                integrity["signature"],
                secret_key
            ):
                return False, "서명 검증 실패"

            return True, "검증 성공"

        except KeyError as e:
            return False, f"필수 필드 누락: {e}"
        except Exception as e:
            return False, f"검증 오류: {e}"

# 사용 예시

# 데이터 패키징

    "@type": "Dialect",
```

```python
# 배치 처리 프로토콜
from typing import List, Dict, Iterator
import json

class BatchProtocol:
    """배치 처리 프로토콜"""

    DEFAULT_BATCH_SIZE = 100
    MAX_BATCH_SIZE = 1000

    @staticmethod
    def create_batch_request(items: List[Dict], batch_size: int = None) -> Iterator[Dict]:
        """배치 요청 생성"""
        if batch_size is None:
            batch_size = BatchProtocol.DEFAULT_BATCH_SIZE

        if batch_size > BatchProtocol.MAX_BATCH_SIZE:
            raise ValueError(f"배치 크기는 {BatchProtocol.MAX_BATCH_SIZE}를 초과할 수 없습니다")

        # 아이템을 배치 크기로 분할
        for i in range(0, len(items), batch_size):
            batch = items[i:i + batch_size]

            yield {
                "@context": "https://wia-lang.org/ns/lang-004/batch",
                "@type": "BatchRequest",
                "batch_number": i // batch_size + 1,
                "total_batches": (len(items) + batch_size - 1) // batch_size,
                "item_count": len(batch),
                "items": batch
            }

    @staticmethod
    def process_batch_response(response: Dict) -> Dict:
        """배치 응답 처리"""
        results = {
            "successful": [],
            "failed": [],
            "total": 0
        }

        for item_result in response.get("results", []):
            results["total"] += 1

            if item_result.get("status") == "success":
                results["successful"].append(item_result)
            else:
                results["failed"].append(item_result)

        return results

    @staticmethod
    def create_multipart_upload(file_path: str, part_size: int = 5 * 1024 * 1024) -> Iterator[bytes]:
        """멀티파트 업로드 (대용량 파일)"""
        with open(file_path, 'rb') as f:
            part_number = 1
            while True:
                chunk = f.read(part_size)
                if not chunk:
                    break

                yield {
                    "part_number": part_number,
                    "size": len(chunk),
                    "data": chunk
                }

                part_number += 1
```

```python
# CDN 캐싱 제어
from typing import Dict
from datetime import datetime, timedelta

class CDNCachingProtocol:
    """CDN 캐싱 프로토콜"""

    @staticmethod
    def generate_cache_headers(resource_type: str, is_immutable: bool = False) -> Dict[str, str]:
        """리소스 유형에 따른 캐시 헤더 생성"""

        headers = {}

        if resource_type == "audio":
            # 음성 파일: 장기 캐싱
            max_age = 86400 * 30  # 30일
            headers["Cache-Control"] = f"public, max-age={max_age}"
            if is_immutable:
                headers["Cache-Control"] += ", immutable"

        elif resource_type == "metadata":
            # 메타데이터: 중기 캐싱
            max_age = 3600  # 1시간
            headers["Cache-Control"] = f"public, max-age={max_age}, must-revalidate"

        elif resource_type == "search":
            # 검색 결과: 단기 캐싱
            max_age = 300  # 5분
            headers["Cache-Control"] = f"private, max-age={max_age}"

        elif resource_type == "user":
            # 사용자 데이터: 캐시 안 함
            headers["Cache-Control"] = "private, no-cache, no-store, must-revalidate"
            headers["Pragma"] = "no-cache"
            headers["Expires"] = "0"

        # CDN 전용 헤더
        headers["CDN-Cache-Control"] = headers["Cache-Control"]

        # Surrogate-Control (Varnish, Fastly 등)
        if resource_type in ["audio", "metadata"]:
            headers["Surrogate-Control"] = f"max-age={max_age * 2}"

        return headers

    @staticmethod
    def generate_cache_key(request_uri: str, headers: Dict[str, str],
                          vary_headers: List[str] = None) -> str:
        """캐시 키 생성"""
        import hashlib

        key_components = [request_uri]

        if vary_headers:
            for header_name in vary_headers:
                if header_name in headers:
                    key_components.append(f"{header_name}:{headers[header_name]}")

        key_string = "|".join(key_components)
        return hashlib.md5(key_string.encode('utf-8')).hexdigest()

    @staticmethod
    def should_cache_bypass(request_headers: Dict[str, str]) -> bool:
        """캐시 우회 여부 판단"""

        # Cache-Control: no-cache
        if "no-cache" in request_headers.get("Cache-Control", ""):
            return True

        # Pragma: no-cache (HTTP/1.0 호환성)
        if request_headers.get("Pragma") == "no-cache":
            return True

        # 인증된 요청
        if "Authorization" in request_headers:
            return True

        return False

# 사용 예시
```

## 6.7 오류 복구 프로토콜

### 6.7.1 재시도 전략

```python
# 재시도 프로토콜
import time
import random
from typing import Callable, Any
from functools import wraps

class RetryProtocol:
    """재시도 프로토콜"""

    @staticmethod
    def exponential_backoff(
        func: Callable,
        max_retries: int = 3,
        base_delay: float = 1.0,
        max_delay: float = 60.0,
        exponential_base: float = 2.0,
        jitter: bool = True
    ) -> Callable:
        """지수 백오프 재시도 데코레이터"""

        @wraps(func)
        def wrapper(*args, **kwargs) -> Any:
            retries = 0

            while retries <= max_retries:
                try:
                    return func(*args, **kwargs)

                except Exception as e:
                    retries += 1

                    if retries > max_retries:
                        print(f"최대 재시도 횟수 초과 ({max_retries}회)")
                        raise

                    # 지수 백오프 계산
                    delay = min(base_delay * (exponential_base ** (retries - 1)), max_delay)

                    # 지터 추가 (선택적)
                    if jitter:
                        delay = delay * (0.5 + random.random() * 0.5)

                    print(f"재시도 {retries}/{max_retries}: {delay:.2f}초 후 재시도")
                    print(f"오류: {e}")

                    time.sleep(delay)

        return wrapper

    @staticmethod
    def circuit_breaker(
        failure_threshold: int = 5,
        recovery_timeout: float = 60.0,
        expected_exception: type = Exception
    ) -> Callable:
        """회로 차단기 패턴 데코레이터"""

        class CircuitBreakerState:
            CLOSED = "closed"  # 정상 작동
            OPEN = "open"      # 차단됨
            HALF_OPEN = "half_open"  # 복구 시도 중

        class CircuitBreaker:
            def __init__(self):
                self.state = CircuitBreakerState.CLOSED
                self.failure_count = 0
                self.last_failure_time = None

            def call(self, func, *args, **kwargs):
                if self.state == CircuitBreakerState.OPEN:
                    # 복구 시간 경과 확인
                    if time.time() - self.last_failure_time >= recovery_timeout:
                        self.state = CircuitBreakerState.HALF_OPEN
                        print("회로 차단기: HALF_OPEN (복구 시도)")
                    else:
                        raise Exception("회로 차단기 OPEN - 서비스 일시적으로 사용 불가")

                try:
                    result = func(*args, **kwargs)

                    # 성공 시 리셋
                    if self.state == CircuitBreakerState.HALF_OPEN:
                        print("회로 차단기: CLOSED (복구 완료)")
                    self.state = CircuitBreakerState.CLOSED
                    self.failure_count = 0

                    return result

                except expected_exception as e:
                    self.failure_count += 1
                    self.last_failure_time = time.time()

                    print(f"실패 카운트: {self.failure_count}/{failure_threshold}")

                    if self.failure_count >= failure_threshold:
                        self.state = CircuitBreakerState.OPEN
                        print(f"회로 차단기: OPEN (임계값 도달)")

                    raise

        breaker = CircuitBreaker()

        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                return breaker.call(func, *args, **kwargs)
            return wrapper

        return decorator
```

- 실시간 통신 (WebSocket, SSE) 지원
- 보안 및 무결성 검증
- 캐싱과 재시도 전략으로 성능 및 안정성 확보
