# 제2장: 방언 보존의 현재 과제

## 들어가며: 위기의 본질

방언 보존은 단순히 "옛날 말을 기록하는 것"이 아닙니다. 이것은 다차원적이고 복잡한 과제로, 기술적, 사회적, 경제적, 정치적 측면이 얽혀 있습니다. 이 장에서는 방언 보존이 직면한 주요 과제들을 체계적으로 분석하고, WIA-LANG-004 표준이 이러한 과제를 어떻게 해결하는지 살펴봅니다.

## 2.1 기술적 과제

### 2.1.1 데이터 수집의 어려움

**1. 화자 접근성 문제**

방언의 유창한 화자는 대부분 고령층에 집중되어 있습니다:

| 방언 | 유창한 화자 연령대 | 추정 화자 수 | 접근성 |
|------|-------------------|--------------|--------|
| 제주 방언 | 70세 이상 | 약 5,000명 | 중간 |
| 육진 방언 (함경도) | 80세 이상 | 약 300명 | 매우 낮음 |
| 경상도 방언 | 60세 이상 | 약 200,000명 | 높음 |
| 전라도 방언 | 60세 이상 | 약 150,000명 | 높음 |
| 충청도 방언 | 65세 이상 | 약 80,000명 | 중간 |

```python
# 화자 접근성 평가 모델
import math
from dataclasses import dataclass
from typing import List, Dict
from datetime import datetime, timedelta

@dataclass
class Speaker:
    """방언 화자 정보"""
    age: int
    proficiency: float  # 0.0 ~ 1.0
    health_status: str  # 'excellent', 'good', 'fair', 'poor'
    location: str
    willingness_to_participate: float  # 0.0 ~ 1.0

class DialectSurvey:
    """방언 조사 계획 및 우선순위 관리"""

    def __init__(self, dialect_name: str):
        self.dialect_name = dialect_name
        self.speakers: List[Speaker] = []

    def add_speaker(self, speaker: Speaker):
        """화자 추가"""
        self.speakers.append(speaker)

    def calculate_urgency_score(self, speaker: Speaker) -> float:
        """화자별 조사 긴급도 점수 계산"""

        # 연령 점수 (나이가 많을수록 긴급)
        age_score = min(speaker.age / 100, 1.0)

        # 숙련도 점수 (숙련도가 높을수록 중요)
        proficiency_score = speaker.proficiency

        # 건강 상태 점수
        health_scores = {
            'excellent': 0.25,
            'good': 0.50,
            'fair': 0.75,
            'poor': 1.00
        }
        health_score = health_scores.get(speaker.health_status, 0.5)

        # 참여 의향 점수 (낮을수록 설득 필요)
        willingness_penalty = 1.0 - (speaker.willingness_to_participate * 0.3)

        # 종합 점수 (가중 평균)
        urgency = (
            age_score * 0.35 +
            proficiency_score * 0.30 +
            health_score * 0.25 +
            willingness_penalty * 0.10
        )

        return urgency

    def prioritize_speakers(self) -> List[tuple]:
        """화자 우선순위 정렬"""
        priority_list = [
            (speaker, self.calculate_urgency_score(speaker))
            for speaker in self.speakers
        ]
        # 긴급도 높은 순으로 정렬
        priority_list.sort(key=lambda x: x[1], reverse=True)
        return priority_list

    def estimate_recording_timeline(self, hours_per_speaker: int = 10) -> Dict:
        """녹음 일정 추정"""
        prioritized = self.prioritize_speakers()

        timeline = []
        cumulative_hours = 0

        for speaker, urgency in prioritized:
            cumulative_hours += hours_per_speaker
            estimated_date = datetime.now() + timedelta(hours=cumulative_hours)

            timeline.append({
                'speaker_age': speaker.age,
                'urgency_score': f"{urgency:.2f}",
                'estimated_date': estimated_date.strftime('%Y-%m-%d'),
                'cumulative_hours': cumulative_hours
            })

        return {
            'total_speakers': len(self.speakers),
            'total_hours': cumulative_hours,
            'estimated_completion': timeline[-1]['estimated_date'] if timeline else None,
            'timeline': timeline
        }

# 제주 방언 조사 시뮬레이션
jeju_survey = DialectSurvey("제주 방언")

# 샘플 화자 추가
jeju_survey.add_speaker(Speaker(
    age=85,
    proficiency=0.98,
    health_status='fair',
    location='제주시 구좌읍',
    willingness_to_participate=0.9
))

jeju_survey.add_speaker(Speaker(
    age=72,
    proficiency=0.85,
    health_status='good',
    location='서귀포시',
    willingness_to_participate=0.7
))

jeju_survey.add_speaker(Speaker(
    age=90,
    proficiency=0.95,
    health_status='poor',
    location='제주시 애월읍',
    willingness_to_participate=0.6
))

# 우선순위 계산
priorities = jeju_survey.prioritize_speakers()
print(f"=== {jeju_survey.dialect_name} 화자 조사 우선순위 ===\n")
for i, (speaker, score) in enumerate(priorities, 1):
    print(f"{i}순위: {speaker.age}세, "
          f"숙련도 {speaker.proficiency:.0%}, "
          f"긴급도 {score:.2f}")

# 일정 추정
timeline = jeju_survey.estimate_recording_timeline(hours_per_speaker=15)
print(f"\n총 소요 시간: {timeline['total_hours']}시간")
print(f"완료 예정일: {timeline['estimated_completion']}")

"""
출력:

1순위: 90세, 숙련도 95%, 긴급도 0.89
2순위: 85세, 숙련도 98%, 긴급도 0.84
3순위: 72세, 숙련도 85%, 긴급도 0.68
```

### 2.1.3 데이터 저장 및 관리

```python
# 방언 데이터 저장소 관리 시스템
from typing import Dict, List
import json
from enum import Enum

class DataType(Enum):
    """데이터 타입 열거"""
    AUDIO = "audio"
    VIDEO = "video"
    TEXT = "text"
    ANNOTATION = "annotation"
    METADATA = "metadata"

class StorageEstimator:
    """저장 용량 추정기"""

    # 파일 크기 추정 (MB 단위)
    AVERAGE_SIZES = {
        DataType.AUDIO: {
            'per_hour_uncompressed': 500,  # 48kHz, 24-bit
            'per_hour_compressed': 50,      # FLAC
        },
        DataType.VIDEO: {
            'per_hour_hd': 4000,           # 1080p
            'per_hour_sd': 1000,           # 720p
        },
        DataType.TEXT: {
            'per_hour': 0.1,               # 전사 텍스트
        },
        DataType.ANNOTATION: {
            'per_hour': 5,                 # 상세 주석
        },
        DataType.METADATA: {
            'per_speaker': 0.01,           # 화자 정보
        }
    }

    def __init__(self, dialect_name: str):
        self.dialect_name = dialect_name
        self.inventory = {
            'speakers': 0,
            'audio_hours': 0,
            'video_hours': 0,
            'annotated_hours': 0
        }

    def add_data(self, speakers: int = 0, audio_hours: float = 0,
                 video_hours: float = 0, annotated_hours: float = 0):
        """데이터 추가"""
        self.inventory['speakers'] += speakers
        self.inventory['audio_hours'] += audio_hours
        self.inventory['video_hours'] += video_hours
        self.inventory['annotated_hours'] += annotated_hours

    def calculate_storage_requirement(self,
                                       use_compression: bool = True,
                                       redundancy_factor: float = 3.0) -> Dict:
        """저장 용량 계산"""

        # 오디오 용량
        audio_key = 'per_hour_compressed' if use_compression else 'per_hour_uncompressed'
        audio_size = (self.inventory['audio_hours'] *
                      self.AVERAGE_SIZES[DataType.AUDIO][audio_key])

        # 비디오 용량 (HD로 가정)
        video_size = (self.inventory['video_hours'] *
                      self.AVERAGE_SIZES[DataType.VIDEO]['per_hour_hd'])

        # 텍스트 및 주석 용량
        text_size = (self.inventory['audio_hours'] *
                     self.AVERAGE_SIZES[DataType.TEXT]['per_hour'])

        annotation_size = (self.inventory['annotated_hours'] *
                          self.AVERAGE_SIZES[DataType.ANNOTATION]['per_hour'])

        # 메타데이터 용량
        metadata_size = (self.inventory['speakers'] *
                        self.AVERAGE_SIZES[DataType.METADATA]['per_speaker'])

        # 총 용량
        total_raw = audio_size + video_size + text_size + annotation_size + metadata_size

        # 백업 및 중복 저장 고려
        total_with_redundancy = total_raw * redundancy_factor

        return {
            'breakdown_mb': {
                'audio': round(audio_size, 2),
                'video': round(video_size, 2),
                'text': round(text_size, 2),
                'annotation': round(annotation_size, 2),
                'metadata': round(metadata_size, 2)
            },
            'total_raw_mb': round(total_raw, 2),
            'total_raw_gb': round(total_raw / 1024, 2),
            'total_with_redundancy_gb': round(total_with_redundancy / 1024, 2),
            'redundancy_factor': redundancy_factor,
            'compression_used': use_compression
        }

    def estimate_costs(self, cost_per_gb_per_month: float = 0.02) -> Dict:
        """저장 비용 추정 (클라우드 기준)"""
        storage = self.calculate_storage_requirement()
        total_gb = storage['total_with_redundancy_gb']

        monthly_cost = total_gb * cost_per_gb_per_month
        yearly_cost = monthly_cost * 12
        five_year_cost = yearly_cost * 5

        return {
            'storage_gb': total_gb,
            'monthly_cost_usd': round(monthly_cost, 2),
            'yearly_cost_usd': round(yearly_cost, 2),
            'five_year_cost_usd': round(five_year_cost, 2)
        }

# 제주 방언 프로젝트 예시
jeju_project = StorageEstimator("제주 방언")
jeju_project.add_data(
    speakers=100,
    audio_hours=500,      # 100명 × 5시간
    video_hours=50,       # 선별된 인터뷰
    annotated_hours=200   # 40% 주석 완료
)

storage = jeju_project.calculate_storage_requirement()
costs = jeju_project.estimate_costs()

print("=== 제주 방언 프로젝트 저장 요구사항 ===\n")
print("데이터 구성:")
for data_type, size in storage['breakdown_mb'].items():
    print(f"  {data_type}: {size} MB")

print(f"\n총 저장 용량: {storage['total_with_redundancy_gb']} GB (백업 포함)")
print(f"\n비용 추정:")
print(f"  월간: ${costs['monthly_cost_usd']}")
print(f"  연간: ${costs['yearly_cost_usd']}")
print(f"  5년: ${costs['five_year_cost_usd']}")

"""
출력:
=== 제주 방언 프로젝트 저장 요구사항 ===

데이터 구성:
  audio: 25000.0 MB
  video: 200000.0 MB
  text: 50.0 MB
  annotation: 1000.0 MB
  metadata: 1.0 MB

총 저장 용량: 662.65 GB (백업 포함)

비용 추정:
  월간: $13.25
  연간: $159.0
  5년: $795.0
"""
```

**스티그마(낙인)의 문제:**

```markdown
## 방언 스티그마의 역사

### 산업화 시대 (1960-1990년대)
- 표준어 = 교육받은 사람
- 방언 = 시골 사람, 교육 부족

   - 방언은 코미디나 촌스러운 캐릭터에만 사용
   - 뉴스, 교양 프로그램에서 방언 배제

3. **직장**
   - 면접에서 표준어 선호
   - "사투리를 고치세요"라는 피드백

- 지역 방언에 대한 자긍심 증가

- 지역 정체성의 재평가

**여전한 과제:**
- 공식 석상에서는 여전히 표준어 우위
- 세대 간 인식 차이 (노년층 vs 청년층)
- 특정 방언에 대한 편견 지속
```

### 2.2.2 세대 간 소통 단절

```python
# 세대 간 방언 사용 시뮬레이션
from typing import Dict, List
from dataclasses import dataclass

@dataclass
class GenerationGroup:
    """세대 그룹"""
    name: str
    age_range: tuple
    dialect_proficiency: float  # 0.0 ~ 1.0
    dialect_usage_frequency: float  # 0.0 ~ 1.0
    attitude_towards_dialect: str  # 'positive', 'neutral', 'negative'

class IntergenerationalCommunication:
    """세대 간 소통 분석"""

    def __init__(self, dialect_name: str):
        self.dialect_name = dialect_name
        self.generations = []

    def add_generation(self, gen: GenerationGroup):
        """세대 추가"""
        self.generations.append(gen)

    def calculate_communication_gap(self, gen1: GenerationGroup,
                                     gen2: GenerationGroup) -> Dict:
        """두 세대 간 소통 격차 계산"""

        # 숙련도 차이
        proficiency_gap = abs(gen1.dialect_proficiency - gen2.dialect_proficiency)

        # 사용 빈도 차이
        usage_gap = abs(gen1.dialect_usage_frequency - gen2.dialect_usage_frequency)

        # 태도 차이
        attitude_scores = {'positive': 1.0, 'neutral': 0.5, 'negative': 0.0}
        attitude_gap = abs(
            attitude_scores[gen1.attitude_towards_dialect] -
            attitude_scores[gen2.attitude_towards_dialect]
        )

        # 종합 격차 (0.0 = 격차 없음, 1.0 = 최대 격차)
        overall_gap = (proficiency_gap * 0.4 +
                       usage_gap * 0.3 +
                       attitude_gap * 0.3)

        return {
            'proficiency_gap': round(proficiency_gap, 2),
            'usage_gap': round(usage_gap, 2),
            'attitude_gap': round(attitude_gap, 2),
            'overall_gap': round(overall_gap, 2),
            'severity': self._interpret_gap(overall_gap)
        }

    def _interpret_gap(self, gap: float) -> str:
        """격차 수준 해석"""
        if gap < 0.2:
            return "최소 (원활한 소통)"
        elif gap < 0.4:
            return "경미 (부분적 어려움)"
        elif gap < 0.6:
            return "중간 (상당한 어려움)"
        elif gap < 0.8:
            return "심각 (의사소통 장애)"
        else:
            return "극심 (사실상 소통 불가)"

    def analyze_all_gaps(self) -> List[Dict]:
        """모든 세대 간 격차 분석"""
        results = []

        for i in range(len(self.generations)):
            for j in range(i + 1, len(self.generations)):
                gen1 = self.generations[i]
                gen2 = self.generations[j]

                gap = self.calculate_communication_gap(gen1, gen2)
                results.append({
                    'generation_1': gen1.name,
                    'generation_2': gen2.name,
                    'gap_analysis': gap
                })

        return results
```

```python
from typing import Dict, List

class DialectPreservationBudget:
    """방언 보존 프로젝트 예산 계획"""

    def __init__(self, project_name: str, duration_years: int):
        self.project_name = project_name
        self.duration_years = duration_years
        self.budget_items = {}

    def add_personnel_cost(self, role: str, count: int,
                           annual_salary: float, years: int = None):
        """인건비 추가"""
        if years is None:
            years = self.duration_years

        total = count * annual_salary * years
        self.budget_items[f"인건비_{role}"] = {
            'category': 'personnel',
            'count': count,
            'annual_cost': annual_salary,
            'years': years,
            'total': total
        }

    def add_equipment_cost(self, item: str, cost: float):
        """장비비 추가"""
        self.budget_items[f"장비_{item}"] = {
            'category': 'equipment',
            'total': cost
        }

    def add_operational_cost(self, item: str, annual_cost: float):
        """운영비 추가"""
        total = annual_cost * self.duration_years
        self.budget_items[f"운영_{item}"] = {
            'category': 'operational',
            'annual_cost': annual_cost,
            'years': self.duration_years,
            'total': total
        }

    def calculate_total_budget(self) -> Dict:
        """총 예산 계산"""
        category_totals = {
            'personnel': 0,
            'equipment': 0,
            'operational': 0
        }

        for item in self.budget_items.values():
            category_totals[item['category']] += item['total']

        grand_total = sum(category_totals.values())

        return {
            'by_category': category_totals,
            'grand_total': grand_total,
            'annual_average': grand_total / self.duration_years
        }

    def generate_budget_report(self) -> str:
        """예산 보고서 생성"""
        totals = self.calculate_total_budget()

        report = f"{'='*60}\n"
        report += f"{self.project_name} 예산 계획 ({self.duration_years}년)\n"
        report += f"{'='*60}\n\n"

        report += "1. 인건비\n"
        for name, item in self.budget_items.items():
            if item['category'] == 'personnel':
                report += f"   - {name}: ${item['total']:,.0f}\n"

        report += f"\n2. 장비비\n"
        for name, item in self.budget_items.items():
            if item['category'] == 'equipment':
                report += f"   - {name}: ${item['total']:,.0f}\n"

        report += f"\n3. 운영비\n"
        for name, item in self.budget_items.items():
            if item['category'] == 'operational':
                report += f"   - {name}: ${item['total']:,.0f}\n"

        report += f"\n{'='*60}\n"
        report += f"총 예산: ${totals['grand_total']:,.0f}\n"
        report += f"연평균: ${totals['annual_average']:,.0f}\n"
        report += f"{'='*60}\n"

        return report
```

```python
# 제주어 데이터 라이선스 예시
```

## 2.5 기술적 표준화 과제

### 2.5.1 상호운용성 부재

현재 방언 데이터는 다양한 형식으로 산재되어 있습니다:

| 기관 | 데이터 형식 | 메타데이터 표준 | API 제공 |
|------|------------|----------------|----------|
| 국립국어원 | 자체 DB | 자체 스키마 | 제한적 |
| 각 대학 연구소 | Excel, CSV, 개별 DB | 불일치 | 없음 |
| 지역 문화원 | 문서, 음성 파일 | 없음 | 없음 |
| 개인 연구자 | 다양 | 없음 | 없음 |

**문제점:**
- 데이터 통합 불가능
- 중복 수집 낭비
- 연구자 간 협력 어려움
- 대규모 분석 불가

**WIA-LANG-004의 솔루션:**
- 통일된 JSON-LD 스키마
- 표준 REST API
- 메타데이터 표준
- 연동 프로토콜
