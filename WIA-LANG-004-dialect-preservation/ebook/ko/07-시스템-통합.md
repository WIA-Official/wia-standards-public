# 제7장: 시스템 통합 - 기존 인프라와의 조화

## 들어가며: 혼자가 아닌 함께

WIA-LANG-004는 고립된 시스템이 아닙니다. 이미 운영 중인 다양한 시스템, 데이터베이스, 연구 도구들과 조화롭게 통합되어야 합니다. 이 장에서는 기존 인프라와 WIA-LANG-004를 통합하는 구체적인 방법을 살펴봅니다.

# SQLAlchemy를 사용한 WIA-LANG-004 데이터베이스 스키마
from sqlalchemy import create_engine, Column, Integer, String, Float, DateTime, ForeignKey, JSON, Text
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import relationship, sessionmaker
from datetime import datetime
from typing import Dict

Base = declarative_base()

class Dialect(Base):
    """방언 테이블"""
    __tablename__ = 'dialects'

    id = Column(String(255), primary_key=True)  # wia-lang://dialects/ko-jeju
    name = Column(String(255), nullable=False)
    alternate_names = Column(JSON)  # 배열로 저장
    iso639_3 = Column(String(3))
    parent_language = Column(String(3), default='kor')
    endangerment_status = Column(String(50))

    # 지리 정보
    primary_region = Column(String(255))
    latitude = Column(Float)
    longitude = Column(Float)

    # 화자 정보
    estimated_speakers = Column(Integer)
    speaker_age_distribution = Column(JSON)

    # 언어학적 특징
    phonetic_features = Column(JSON)
    lexical_features = Column(JSON)
    grammatical_features = Column(JSON)

    # 메타데이터
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 관계
    recordings = relationship("Recording", back_populates="dialect")

    def to_jsonld(self) -> Dict:
        """JSON-LD 형식으로 변환"""
        return {
            "@context": "https://wia-lang.org/ns/lang-004/context.json",
            "@type": "Dialect",
            "@id": self.id,
            "name": self.name,
            "alternateNames": self.alternate_names or [],
            "iso639-3": self.iso639_3,
            "parentLanguage": self.parent_language,
            "endangermentStatus": self.endangerment_status,
            "geographic": {
                "primaryRegion": self.primary_region,
                "geo:lat": self.latitude,
                "geo:long": self.longitude
            },
            "speakers": {
                "estimatedCount": self.estimated_speakers,
                "ageDistribution": self.speaker_age_distribution or {}
            },
            "linguisticFeatures": {
                "phonetic": self.phonetic_features or [],
                "lexical": self.lexical_features or [],
                "grammatical": self.grammatical_features or []
            },
            "created": self.created_at.isoformat() if self.created_at else None,
            "updated": self.updated_at.isoformat() if self.updated_at else None
        }

class Speaker(Base):
    """화자 테이블"""
    __tablename__ = 'speakers'

    id = Column(String(255), primary_key=True)
    birth_year = Column(Integer, nullable=False)
    gender = Column(String(20))
    birthplace = Column(String(255))
    current_residence = Column(String(255))
    residence_history = Column(JSON)

    # 언어 정보
    dialects_spoken = Column(JSON)
    proficiency_levels = Column(JSON)

    # 개인정보 보호
    anonymized = Column(Integer, default=1)  # Boolean
    consent_given = Column(Integer, default=0)

    # 메타데이터
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 관계
    recordings = relationship("Recording", back_populates="speaker")

class Recording(Base):
    """녹음 테이블"""
    __tablename__ = 'recordings'

    id = Column(String(255), primary_key=True)
    dialect_id = Column(String(255), ForeignKey('dialects.id'), nullable=False)
    speaker_id = Column(String(255), ForeignKey('speakers.id'), nullable=False)

    recorded_date = Column(DateTime, nullable=False)
    location = Column(String(255))

    # 음성 파일
    audio_file_url = Column(Text)
    duration_seconds = Column(Float)
    format = Column(String(10))
    sampling_rate = Column(Integer)
    bit_depth = Column(Integer)
    channels = Column(Integer)

    # 녹음 환경
    recording_environment = Column(String(50))
    background_noise_level = Column(Float)
    equipment_used = Column(JSON)

    # 내용
    genre = Column(String(50))
    topics = Column(JSON)

    # 품질
    quality_score = Column(Float)
    transcription_available = Column(Integer, default=0)
    annotation_available = Column(Integer, default=0)

    # 메타데이터
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 관계
    dialect = relationship("Dialect", back_populates="recordings")
    speaker = relationship("Speaker", back_populates="recordings")
    transcriptions = relationship("Transcription", back_populates="recording")

class Transcription(Base):
    """전사 테이블"""
    __tablename__ = 'transcriptions'

    id = Column(Integer, primary_key=True, autoincrement=True)
    recording_id = Column(String(255), ForeignKey('recordings.id'), nullable=False)

    dialect_orthography = Column(Text)
    ipa = Column(Text)
    standard_korean = Column(Text)
    morphemes = Column(JSON)

    # 메타데이터
    transcriber = Column(String(255))
    transcription_method = Column(String(50))
    confidence = Column(Float)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 관계
    recording = relationship("Recording", back_populates="transcriptions")

class Annotation(Base):

    id = Column(Integer, primary_key=True, autoincrement=True)

    annotation_type = Column(String(50))  # 'phonetic', 'lexical', 'grammatical', 'pragmatic'
    content = Column(JSON)

    # 메타데이터
    annotator = Column(String(255))

# MongoDB를 사용한 WIA-LANG-004 데이터 저장
from pymongo import MongoClient
from typing import Dict, List
import json
from datetime import datetime

class WIALangMongoDBAdapter:
    """MongoDB 어댑터"""

    def __init__(self, connection_string: str = "mongodb://localhost:27017/",
                 database: str = "wia_lang_004"):
        self.client = MongoClient(connection_string)
        self.db = self.client[database]

        # 컬렉션
        self.dialects = self.db.dialects
        self.speakers = self.db.speakers
        self.recordings = self.db.recordings
        self.transcriptions = self.db.transcriptions
        self.annotations = self.db.annotations

        # 인덱스 생성
        self._create_indexes()

    def _create_indexes(self):
        """인덱스 생성"""

        # 방언
        self.dialects.create_index("@id", unique=True)
        self.dialects.create_index("name")
        self.dialects.create_index("endangermentStatus")
        self.dialects.create_index([("geographic.geo:lat", "2dsphere")])

        # 녹음
        self.recordings.create_index("@id", unique=True)
        self.recordings.create_index("dialect.@id")
        self.recordings.create_index("speaker.@id")
        self.recordings.create_index("recordedDate")

        # 전사 (전문 검색)
        self.transcriptions.create_index([("dialectOrthography", "text"),
                                          ("standardKorean", "text")])

    def insert_dialect(self, dialect_data: Dict) -> str:
        """방언 데이터 삽입"""
        # JSON-LD 데이터를 그대로 저장
        result = self.dialects.insert_one(dialect_data)
        return str(result.inserted_id)

    def get_dialect(self, dialect_id: str) -> Dict:
        """방언 조회"""
        return self.dialects.find_one({"@id": dialect_id}, {"_id": 0})

    def search_dialects(self, query: Dict) -> List[Dict]:
        """방언 검색"""
        cursor = self.dialects.find(query, {"_id": 0})
        return list(cursor)

    def insert_recording(self, recording_data: Dict) -> str:
        """녹음 데이터 삽입"""
        result = self.recordings.insert_one(recording_data)
        return str(result.inserted_id)

    def get_recordings_by_dialect(self, dialect_id: str, limit: int = 20) -> List[Dict]:
        """방언별 녹음 조회"""
        cursor = self.recordings.find(
            {"dialect.@id": dialect_id},
            {"_id": 0}
        ).limit(limit)
        return list(cursor)

    def full_text_search(self, search_term: str, limit: int = 20) -> List[Dict]:
        """전문 검색"""
        cursor = self.transcriptions.find(
            {"$text": {"$search": search_term}},
            {"score": {"$meta": "textScore"}, "_id": 0}
        ).sort([("score", {"$meta": "textScore"})]).limit(limit)

        return list(cursor)

    def geospatial_query(self, longitude: float, latitude: float,
                        max_distance_km: float = 100) -> List[Dict]:
        """지리공간 검색"""
        cursor = self.dialects.find({
            "geographic.coordinates": {
                "$near": {
                    "$geometry": {
                        "type": "Point",
                        "coordinates": [longitude, latitude]
                    },
                    "$maxDistance": max_distance_km * 1000  # 미터로 변환
                }
            }
        }, {"_id": 0})

        return list(cursor)

# AWS S3를 사용한 음성 파일 저장
import boto3
from botocore.exceptions import ClientError
from typing import BinaryIO, Dict, Optional
import mimetypes
from datetime import datetime, timedelta

class S3AudioStorage:
    """S3 오디오 파일 스토리지"""

    def __init__(self, bucket_name: str, region: str = "ap-northeast-2"):
        self.bucket_name = bucket_name
        self.region = region
        self.s3_client = boto3.client('s3', region_name=region)

    def upload_recording(self, file_obj: BinaryIO, recording_id: str,
                        metadata: Optional[Dict] = None) -> str:
        """녹음 파일 업로드"""

        # 파일 확장자에서 MIME 타입 추정
        file_extension = recording_id.split('.')[-1]
        content_type = mimetypes.types_map.get(f'.{file_extension}', 'audio/mpeg')

        # S3 키 생성 (연도/월/일/recording_id 구조)
        now = datetime.now()
        s3_key = f"recordings/{now.year}/{now.month:02d}/{now.day:02d}/{recording_id}"

        # S3 메타데이터
        s3_metadata = metadata or {}
        s3_metadata.update({
            'uploaded-at': now.isoformat(),
            'wia-lang-version': '1.0'
        })

        try:
            # 업로드
            self.s3_client.upload_fileobj(
                file_obj,
                self.bucket_name,
                s3_key,
                ExtraArgs={
                    'ContentType': content_type,
                    'Metadata': s3_metadata,
                    'ServerSideEncryption': 'AES256',  # 암호화
                    'StorageClass': 'STANDARD_IA'  # 비용 최적화
                }
            )

            # URL 생성
            url = f"https://{self.bucket_name}.s3.{self.region}.amazonaws.com/{s3_key}"
            return url

        except ClientError as e:
            raise Exception(f"S3 업로드 실패: {e}")

    def generate_presigned_url(self, s3_key: str, expiration: int = 3600) -> str:
        """사전 서명된 URL 생성 (임시 다운로드 링크)"""

        try:
            url = self.s3_client.generate_presigned_url(
                'get_object',
                Params={
                    'Bucket': self.bucket_name,
                    'Key': s3_key
                },
                ExpiresIn=expiration
            )
            return url

        except ClientError as e:
            raise Exception(f"URL 생성 실패: {e}")

    def download_recording(self, s3_key: str, local_path: str):
        """녹음 파일 다운로드"""

        try:
            self.s3_client.download_file(
                self.bucket_name,
                s3_key,
                local_path
            )

        except ClientError as e:
            raise Exception(f"S3 다운로드 실패: {e}")

    def delete_recording(self, s3_key: str):
        """녹음 파일 삭제"""

        try:
            self.s3_client.delete_object(
                Bucket=self.bucket_name,
                Key=s3_key
            )

        except ClientError as e:
            raise Exception(f"S3 삭제 실패: {e}")

    def set_lifecycle_policy(self):
        """수명 주기 정책 설정 (자동 아카이브)"""

        lifecycle_policy = {
            'Rules': [
                {
                    'Id': 'Archive old recordings',
                    'Status': 'Enabled',
                    'Filter': {'Prefix': 'recordings/'},
                    'Transitions': [
                        {
                            'Days': 90,
                            'StorageClass': 'GLACIER'  # 90일 후 Glacier로 이동
                        },
                        {
                            'Days': 365,
                            'StorageClass': 'DEEP_ARCHIVE'  # 1년 후 Deep Archive로
                        }
                    ]
                }
            ]
        }

        try:
            self.s3_client.put_bucket_lifecycle_configuration(
                Bucket=self.bucket_name,
                LifecycleConfiguration=lifecycle_policy
            )

        except ClientError as e:
            raise Exception(f"수명 주기 정책 설정 실패: {e}")

from google.cloud import storage
from typing import BinaryIO, Dict

class GCSAudioStorage:
    """Google Cloud Storage 오디오 파일 스토리지"""

    def __init__(self, bucket_name: str, project_id: str):
        self.bucket_name = bucket_name
        self.project_id = project_id
        self.client = storage.Client(project=project_id)
        self.bucket = self.client.bucket(bucket_name)

    def upload_recording(self, file_obj: BinaryIO, blob_name: str,
                        metadata: Optional[Dict] = None) -> str:
        """녹음 파일 업로드"""

        blob = self.bucket.blob(blob_name)

        # 메타데이터 설정
        if metadata:
            blob.metadata = metadata

        # 업로드
        blob.upload_from_file(file_obj, content_type='audio/flac')

        # 공개 URL
        return f"https://storage.googleapis.com/{self.bucket_name}/{blob_name}"

    def generate_signed_url(self, blob_name: str, expiration_hours: int = 1) -> str:
        """서명된 URL 생성"""

        blob = self.bucket.blob(blob_name)
        url = blob.generate_signed_url(
            version="v4",
            expiration=timedelta(hours=expiration_hours),
            method="GET"
        )

        return url

# Elasticsearch를 사용한 전문 검색
from elasticsearch import Elasticsearch
from typing import Dict, List

class ElasticsearchAdapter:
    """Elasticsearch 어댑터"""

    def __init__(self, hosts: List[str] = ["localhost:9200"]):
        self.es = Elasticsearch(hosts)
        self.index_name = "wia_lang_004"

    def create_index(self):
        """인덱스 생성"""

        # 매핑 정의
        mapping = {
            "settings": {
                "analysis": {
                    "analyzer": {
                        "korean": {
                            "type": "custom",
                            "tokenizer": "nori_tokenizer",
                            "filter": ["nori_part_of_speech"]
                        }
                    }
                }
            },
            "mappings": {
                "properties": {
                    "@id": {"type": "keyword"},
                    "@type": {"type": "keyword"},
                    "name": {
                        "type": "text",
                        "analyzer": "korean",
                        "fields": {
                            "keyword": {"type": "keyword"}
                        }
                    },
                    "endangermentStatus": {"type": "keyword"},
                    "geographic": {
                        "properties": {
                            "location": {"type": "geo_point"}
                        }
                    },
                    "transcription": {
                        "properties": {
                            "dialectOrthography": {
                                "type": "text",
                                "analyzer": "korean"
                            },
                            "standardKorean": {
                                "type": "text",
                                "analyzer": "korean"
                            }
                        }
                    }
                }
            }
        }

        self.es.indices.create(index=self.index_name, body=mapping, ignore=400)

    def index_dialect(self, dialect_data: Dict):
        """방언 데이터 색인"""

        self.es.index(
            index=self.index_name,
            id=dialect_data["@id"],
            body=dialect_data
        )

    def search(self, query: str, filters: Dict = None, size: int = 20) -> List[Dict]:
        """검색"""

        # 쿼리 구성
        search_body = {
            "query": {
                "bool": {
                    "must": [
                        {
                            "multi_match": {
                                "query": query,
                                "fields": [
                                    "name^3",  # 이름에 가중치 3
                                    "transcription.dialectOrthography^2",
                                    "transcription.standardKorean"
                                ],
                                "type": "best_fields"
                            }
                        }
                    ]
                }
            },
            "size": size,
            "highlight": {
                "fields": {
                    "transcription.dialectOrthography": {},
                    "transcription.standardKorean": {}
                }
            }
        }

        # 필터 추가
        if filters:
            search_body["query"]["bool"]["filter"] = []

            if "endangerment" in filters:
                search_body["query"]["bool"]["filter"].append({
                    "term": {"endangermentStatus": filters["endangerment"]}
                })

            if "location" in filters:
                # 지리공간 필터
                search_body["query"]["bool"]["filter"].append({
                    "geo_distance": {
                        "distance": filters.get("distance", "100km"),

import subprocess
from typing import Dict, List

class PraatIntegration:
    """Praat 음성 분석 도구 통합"""

    def __init__(self, praat_path: str = "/usr/bin/praat"):
        self.praat_path = praat_path

    def extract_pitch(self, audio_file: str, output_file: str = None) -> Dict:
        """기본 주파수 (F0) 추출"""

        if output_file is None:
            output_file = audio_file.replace('.wav', '.Pitch')

        # Praat 스크립트
        script = f"""
        Read from file... {audio_file}
        To Pitch... 0.0 75 600
        Save as text file... {output_file}
        """

        # 실행
        self._run_script(script)

        # 결과 파싱 (간략화)
        return {
            "pitch_file": output_file,
            "status": "success"
        }

    def extract_formants(self, audio_file: str, num_formants: int = 5) -> Dict:
        """포먼트 추출"""

        output_file = audio_file.replace('.wav', '.Formant')

        script = f"""
        Read from file... {audio_file}
        To Formant (burg)... 0.0 {num_formants} 5500 0.025 50
        Save as text file... {output_file}
        """

        self._run_script(script)

        return {
            "formant_file": output_file,
            "num_formants": num_formants,
            "status": "success"
        }

    def create_textgrid(self, audio_file: str, tiers: List[str] = None) -> str:
        """TextGrid 생성 (주석용)"""

        if tiers is None:
            tiers = ["phonetic", "orthographic", "translation"]

        output_file = audio_file.replace('.wav', '.TextGrid')

        # TextGrid 생성 스크립트
        script = f"""
        Read from file... {audio_file}
        To TextGrid... "{' '.join(tiers)}" ""
        Save as text file... {output_file}
        """

        self._run_script(script)

        return output_file

    def _run_script(self, script: str):
        """Praat 스크립트 실행"""

        # 임시 스크립트 파일 생성
        script_file = "/tmp/praat_script.praat"
        with open(script_file, 'w') as f:
            f.write(script)

        # Praat 실행
        try:
            subprocess.run(
                [self.praat_path, "--run", script_file],
                check=True,
                capture_output=True
            )
        finally:
            # 스크립트 파일 삭제
            if os.path.exists(script_file):
                os.remove(script_file)
