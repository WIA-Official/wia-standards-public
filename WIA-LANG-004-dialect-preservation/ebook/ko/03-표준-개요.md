# 제3장: WIA-LANG-004 표준 개요

## 들어가며: 표준의 철학

WIA-LANG-004 Dialect Preservation 표준은 단순한 기술 명세가 아닙니다. 이것은 **弘益人間**(널리 인간을 이롭게 하라)의 정신을 구현한, 전 세계 방언 공동체를 위한 협력 플랫폼입니다.

이 장에서는 표준의 핵심 구조, 설계 원칙, 그리고 주요 구성요소를 살펴봅니다.

### 3.1.2 공동체 중심 원칙

```python
# 공동체 참여 모델
from enum import Enum
from dataclasses import dataclass
from typing import List, Dict

class ParticipationLevel(Enum):
    """참여 수준"""
    INFORMED = "informed"          # 정보 제공
    CONSULTED = "consulted"        # 협의
    COLLABORATIVE = "collaborative" # 협력
    COMMUNITY_LED = "community_led" # 공동체 주도

@dataclass
class StakeholderRole:
    """이해관계자 역할"""
    name: str
    type: str  # 'speaker', 'community_leader', 'researcher', 'institution'
    participation_level: ParticipationLevel
    rights: List[str]
    responsibilities: List[str]

class CommunityGovernance:
    """공동체 거버넌스 모델"""

    def __init__(self, dialect_name: str):
        self.dialect_name = dialect_name
        self.stakeholders: List[StakeholderRole] = []

    def add_stakeholder(self, stakeholder: StakeholderRole):
        """이해관계자 추가"""
        self.stakeholders.append(stakeholder)

    def define_decision_process(self, decision_type: str) -> Dict:
        """의사결정 프로세스 정의"""

        processes = {
            'data_collection': {
                'requires_approval_from': ['community_leader', 'speaker'],
                'consultation_with': ['researcher'],
                'decision_maker': 'community',
                'veto_power': ['speaker']
            },
            'data_usage': {
                'requires_approval_from': ['community_leader'],
                'consultation_with': ['speaker', 'researcher'],
                'decision_maker': 'community',
                'veto_power': ['community_leader']
            },
            'commercial_use': {
                'requires_approval_from': ['community_leader', 'speaker'],
                'consultation_with': ['institution'],
                'decision_maker': 'community',
                'veto_power': ['community_leader', 'speaker'],
                'benefit_sharing': 'mandatory'
            },
            'research_publication': {
                'requires_approval_from': ['community_leader'],
                'consultation_with': ['speaker'],
                'decision_maker': 'researcher',
                'veto_power': ['community_leader'],
                'attribution': 'mandatory'
            }
        }

        return processes.get(decision_type, {
            'requires_approval_from': ['all'],
            'consultation_with': ['all'],
            'decision_maker': 'consensus',
            'veto_power': ['community']
        })

    def generate_governance_charter(self) -> str:
        """거버넌스 헌장 생성"""

        charter = f"""
{'='*70}
{self.dialect_name} 방언 데이터 거버넌스 헌장
{'='*70}

제1조 (목적)
이 헌장은 {self.dialect_name} 방언 데이터의 수집, 보존, 활용에 있어
공동체의 권리를 보호하고 모든 이해관계자의 책임을 명확히 함을 목적으로 한다.

제2조 (기본 원칙)
1. 공동체 주권: 방언은 공동체의 문화유산이다.
2. 사전 동의: 모든 데이터 수집은 화자의 사전 동의를 받는다.
3. 투명성: 데이터 사용 목적과 방법을 명확히 공개한다.
4. 혜택 공유: 연구 성과는 공동체와 공유한다.
5. 존엄성 존중: 화자와 공동체의 존엄성을 최우선으로 한다.

제3조 (이해관계자 및 역할)
"""

        for i, stakeholder in enumerate(self.stakeholders, 1):
            charter += f"\n{i}. {stakeholder.name} ({stakeholder.type})\n"
            charter += f"   참여 수준: {stakeholder.participation_level.value}\n"
            charter += f"   권리:\n"
            for right in stakeholder.rights:
                charter += f"     - {right}\n"
            charter += f"   책임:\n"
            for resp in stakeholder.responsibilities:
                charter += f"     - {resp}\n"

        charter += f"""
제4조 (의사결정 프로세스)
주요 결정 사항은 다음 프로세스를 따른다:
"""

        decision_types = ['data_collection', 'data_usage', 'commercial_use', 'research_publication']
        for dt in decision_types:
            process = self.define_decision_process(dt)
            charter += f"\n- {dt}:\n"
            charter += f"  승인 필요: {', '.join(process['requires_approval_from'])}\n"
            charter += f"  협의 대상: {', '.join(process['consultation_with'])}\n"
            charter += f"  최종 결정자: {process['decision_maker']}\n"

        charter += f"""
제5조 (분쟁 해결)
이해관계자 간 분쟁은 다음 순서로 해결한다:
1. 당사자 간 대화
2. 중재위원회 중재
3. WIA 표준 위원회 조정
4. 해당 지역 법원 소송

제6조 (헌장의 개정)
이 헌장은 이해관계자 2/3 이상의 동의로 개정할 수 있다.

제정일: 2025년 1월 1일
"""

        charter += f"\n{'='*70}\n"
        charter += "© 2025 " + self.dialect_name + " 공동체\n"
        charter += "WIA-LANG-004 표준 준수\n"
        charter += "弘益人間 (홍익인간) · Benefit All Humanity\n"

        return charter

# 제주어 공동체 거버넌스 예시
jeju_governance = CommunityGovernance("제주어")

# 이해관계자 정의
jeju_governance.add_stakeholder(StakeholderRole(
    name="제주어 화자",
    type="speaker",
    participation_level=ParticipationLevel.COMMUNITY_LED,
    rights=[
        "자신의 음성 데이터에 대한 완전한 통제권",
        "데이터 사용 목적 알 권리",
        "언제든지 동의 철회 권리",
        "연구 결과 열람 권리"
    ],
    responsibilities=[
        "정확한 개인 정보 제공",
        "녹음 일정 준수",
        "자연스러운 방언 사용"
    ]
))

jeju_governance.add_stakeholder(StakeholderRole(
    name="제주도 방언 보존위원회",
    type="community_leader",
    participation_level=ParticipationLevel.COLLABORATIVE,
    rights=[
        "데이터 사용 승인 권한",
        "상업적 활용 거부권",
        "공동체 이익 대변권"
    ],
    responsibilities=[
        "공동체 의견 수렴",
        "투명한 의사결정",
        "혜택의 공정한 분배"
    ]
))

jeju_governance.add_stakeholder(StakeholderRole(
    name="언어학 연구팀",
    type="researcher",
    participation_level=ParticipationLevel.COLLABORATIVE,
    rights=[
        "연구 목적 데이터 접근",
        "학술 출판 권리",
        "연구 방법 선택권"
    ],
    responsibilities=[
        "윤리적 연구 수행",
        "데이터 보안 유지",
        "결과의 공동체 공유",
        "저작권 존중"
    ]
))

print(jeju_governance.generate_governance_charter())
```

```markdown
### Level 0: 공개 메타데이터
- **접근**: 누구나
- **내용**: 방언명, 지역, 화자 수, 샘플 통계
- **목적**: 발견 가능성 제고

- **접근**: 등록 사용자
- **내용**: 5분 미만 샘플, 기본 주석
- **목적**: 연구 타당성 평가
- **예시**: 대표 문장 50개, 음운 특징 샘플

- **접근**: 연구 목적 승인된 사용자
- **내용**: 전체 음성, 전사, 주석
- **목적**: 학술 연구
- **조건**: 공동체 승인, 윤리 서약

### Level 3: 민감 데이터
- **접근**: 공동체 및 지정 연구자만

- **목적**: 보호 필요 자료
- **조건**: 특별 계약, 엄격한 보안

### Level 4: 상업적 사용
- **접근**: 라이선스 계약자
- **내용**: Level 2 데이터
- **목적**: 상업적 응용 (앱, 서비스)
- **조건**: 공동체 합의, 수익 공유 계약
```

**2. API 엔드포인트**

```python
# WIA-LANG-004 API 클라이언트 예시
import requests
from typing import Dict, List, Optional

class WIALANG004Client:
    """WIA-LANG-004 API 클라이언트"""

    def __init__(self, api_key: str, base_url: str = "https://api.wia-lang.org/v1"):
        self.api_key = api_key
        self.base_url = base_url
        self.headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json",
            "Accept": "application/ld+json"
        }

    def search_dialects(self, query: str, filters: Optional[Dict] = None) -> List[Dict]:
        """방언 검색"""
        endpoint = f"{self.base_url}/dialects/search"

        params = {"q": query}
        if filters:
            params.update(filters)

        response = requests.get(endpoint, headers=self.headers, params=params)
        response.raise_for_status()

        return response.json()

    def get_dialect_metadata(self, dialect_id: str) -> Dict:
        """방언 메타데이터 조회"""
        endpoint = f"{self.base_url}/dialects/{dialect_id}"

        response = requests.get(endpoint, headers=self.headers)
        response.raise_for_status()

        return response.json()

    def get_audio_sample(self, dialect_id: str, sample_id: str) -> bytes:
        """음성 샘플 다운로드"""
        endpoint = f"{self.base_url}/dialects/{dialect_id}/audio/{sample_id}"

        response = requests.get(endpoint, headers=self.headers)
        response.raise_for_status()

        return response.content

    def upload_dialect_data(self, data: Dict) -> Dict:
        """방언 데이터 업로드"""
        endpoint = f"{self.base_url}/dialects"

        response = requests.post(endpoint, headers=self.headers, json=data)
        response.raise_for_status()

        return response.json()

    def annotate_utterance(self, dialect_id: str, utterance_id: str,
                          annotation: Dict) -> Dict:
        """발화 주석 추가"""
        endpoint = f"{self.base_url}/dialects/{dialect_id}/utterances/{utterance_id}/annotations"

        response = requests.post(endpoint, headers=self.headers, json=annotation)
        response.raise_for_status()

        return response.json()

    def get_dialect_statistics(self, dialect_id: str) -> Dict:
        """방언 통계 조회"""
        endpoint = f"{self.base_url}/dialects/{dialect_id}/statistics"

        response = requests.get(endpoint, headers=self.headers)
        response.raise_for_status()

        return response.json()

    def compare_dialects(self, dialect_ids: List[str], features: List[str]) -> Dict:
        """방언 비교 분석"""
        endpoint = f"{self.base_url}/analysis/compare"

        payload = {
            "dialect_ids": dialect_ids,
            "features": features
        }

        response = requests.post(endpoint, headers=self.headers, json=payload)
        response.raise_for_status()

        return response.json()

# 사용 예시
```

```python
# 표준 준수 수준 평가
from enum import Enum
from typing import Dict, List
from dataclasses import dataclass

class ComplianceLevel(Enum):
    """준수 수준"""
    CORE = "core"                  # 핵심 (필수)
    EXTENDED = "extended"          # 확장 (권장)
    OPTIONAL = "optional"          # 선택 (부가)

@dataclass
class Requirement:
    """표준 요구사항"""
    id: str
    description: str
    level: ComplianceLevel
    category: str

class ComplianceChecker:
    """표준 준수 검사기"""

    def __init__(self):

    def _define_requirements(self) -> List[Requirement]:

        return [

                level=ComplianceLevel.OPTIONAL,
                category="diachronic"
            ),

    def check_compliance(self, dataset: Dict) -> Dict:
        """준수 여부 검사"""
        results = {
            'core': {'passed': 0, 'failed': 0, 'total': 0},
            'extended': {'passed': 0, 'failed': 0, 'total': 0},
            'optional': {'passed': 0, 'failed': 0, 'total': 0}
        }

        details = []

        for req in self.requirements:
            level_key = req.level.value

            # 실제 검사 로직 (여기서는 시뮬레이션)
            passed = self._check_requirement(req, dataset)

            results[level_key]['total'] += 1
            if passed:
                results[level_key]['passed'] += 1
            else:
                results[level_key]['failed'] += 1

            details.append({
                'id': req.id,
                'description': req.description,
                'level': req.level.value,
                'passed': passed
            })

        # 준수 레벨 결정
        core_compliance = results['core']['passed'] / results['core']['total']
        extended_compliance = results['extended']['passed'] / results['extended']['total']

        if core_compliance == 1.0:
            if extended_compliance >= 0.8:
                certification = "Gold (핵심 + 확장 80%+)"
            elif extended_compliance >= 0.5:
                certification = "Silver (핵심 + 확장 50%+)"
            else:
                certification = "Bronze (핵심만 충족)"
        else:
            certification = "Not Compliant (핵심 요구사항 미충족)"

        return {
            'certification': certification,
            'summary': results,
            'details': details,
            'compliance_score': self._calculate_score(results)
        }

    def _check_requirement(self, req: Requirement, dataset: Dict) -> bool:
        """개별 요구사항 검사 (실제 구현 필요)"""
        # 여기서는 시뮬레이션
        import random
        if req.level == ComplianceLevel.CORE:
            return random.random() > 0.1  # 90% 통과
        elif req.level == ComplianceLevel.EXTENDED:
            return random.random() > 0.3  # 70% 통과
        else:
            return random.random() > 0.5  # 50% 통과

    def _calculate_score(self, results: Dict) -> float:
        """준수 점수 계산"""
        # 가중치: 핵심 60%, 확장 30%, 선택 10%
        core_score = (results['core']['passed'] / results['core']['total']) * 60
        ext_score = (results['extended']['passed'] / results['extended']['total']) * 30
        opt_score = (results['optional']['passed'] / results['optional']['total']) * 10

        return round(core_score + ext_score + opt_score, 1)
```
