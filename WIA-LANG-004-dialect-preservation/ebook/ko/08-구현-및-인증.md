# MVP 범위 정의
from typing import List, Dict
from dataclasses import dataclass

@dataclass
class Feature:
    """기능 정의"""
    name: str
    priority: str  # 'must_have', 'should_have', 'nice_to_have'
    complexity: str  # 'low', 'medium', 'high'
    estimated_days: int

class MVPPlanner:
    """MVP 계획 도구"""

    def __init__(self):
        self.features: List[Feature] = []

    def add_feature(self, feature: Feature):
        """기능 추가"""
        self.features.append(feature)

    def calculate_mvp_scope(self, max_days: int = 90) -> Dict:
        """MVP 범위 계산"""

        # 우선순위별 정렬
        must_have = [f for f in self.features if f.priority == 'must_have']
        should_have = [f for f in self.features if f.priority == 'should_have']
        nice_to_have = [f for f in self.features if f.priority == 'nice_to_have']

        # MVP에 포함될 기능
        mvp_features = []
        total_days = 0

        # 필수 기능 먼저
        for feature in must_have:
            mvp_features.append(feature)
            total_days += feature.estimated_days

        # 여유 있으면 권장 기능 추가
        for feature in should_have:
            if total_days + feature.estimated_days <= max_days:
                mvp_features.append(feature)
                total_days += feature.estimated_days

        return {
            'mvp_features': mvp_features,
            'total_features': len(mvp_features),
            'estimated_days': total_days,
            'within_budget': total_days <= max_days,
            'breakdown': {
                'must_have': len([f for f in mvp_features if f.priority == 'must_have']),
                'should_have': len([f for f in mvp_features if f.priority == 'should_have']),
                'nice_to_have': len([f for f in mvp_features if f.priority == 'nice_to_have'])
            }
        }

    API_KEY = "test_api_key"

    @pytest.fixture

        return {
            "Authorization": f"Bearer {self.API_KEY}",
            "Content-Type": "application/json"
        }

    def test_create_dialect(self, headers):
        """방언 생성 테스트"""
        payload = {
            "@context": "https://wia-lang.org/ns/lang-004/context.json",
            "@type": "Dialect",
            "name": "테스트 방언",
            "endangermentStatus": "safe"
        }

        response = requests.post(
            f"{self.BASE_URL}/dialects",
            json=payload,
            headers=headers
        )

        assert response.status_code == 201
        data = response.json()
        assert "@id" in data
        assert data["name"] == "테스트 방언"

        # 정리
        dialect_id = data["@id"].split("/")[-1]
        requests.delete(f"{self.BASE_URL}/dialects/{dialect_id}", headers=headers)

    def test_get_dialect(self, headers):
        """방언 조회 테스트"""
        response = requests.get(
            f"{self.BASE_URL}/dialects/ko-jeju",
            headers=headers
        )

        assert response.status_code == 200
        data = response.json()
        assert data["@type"] == "Dialect"
        assert "name" in data

    def test_search_dialects(self, headers):
        """방언 검색 테스트"""
        response = requests.post(
            f"{self.BASE_URL}/search",
            json={"query": "제주", "filters": {"language": "kor"}},
            headers=headers
        )

        assert response.status_code == 200
        data = response.json()
        assert "data" in data
        assert "total" in data

### 8.5.2 CI/CD 파이프라인

```yaml
# .github/workflows/deploy.yml
name: WIA-LANG-004 Deploy

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

env:
  PYTHON_VERSION: "3.11"
  NODE_VERSION: "20"

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: ${{ env.PYTHON_VERSION }}

      - name: Install dependencies
        run: |
          pip install -r requirements.txt
          pip install pytest pytest-cov

      - name: Run tests
        run: |
          pytest --cov=. --cov-report=xml

      - name: Upload coverage
        uses: codecov/codecov-action@v3

  lint:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: ${{ env.PYTHON_VERSION }}

      - name: Lint
        run: |
          pip install pylint mypy
          pylint **/*.py
          mypy .

  security:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Security scan
        uses: snyk/actions/python@master
        env:
          SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}

  build:
    needs: [test, lint, security]
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Build Docker image
        run: |
          docker build -t wia-lang-004:${{ github.sha }} .

      - name: Push to registry
        run: |
          echo ${{ secrets.DOCKER_PASSWORD }} | docker login -u ${{ secrets.DOCKER_USERNAME }} --password-stdin
          docker push wia-lang-004:${{ github.sha }}

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'

    steps:
      - name: Deploy to production
        run: |
          # Kubernetes 배포
          kubectl set image deployment/wia-lang-004 \
            app=wia-lang-004:${{ github.sha }}

      - name: Health check
        run: |
          curl -f https://api.wia-lang.org/health || exit 1

      - name: Notify
        uses: 8398a7/action-slack@v3
        with:
          status: ${{ job.status }}
          text: 'WIA-LANG-004 배포 완료!'
          webhook_url: ${{ secrets.SLACK_WEBHOOK }}
```
