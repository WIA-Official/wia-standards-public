import requests
import json

class WIA_OCR_Pipeline:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = 'https://api.example.org/heritage/v1'

    def process_document(self, document_id):
        # 1. 이미지 다운로드
        images = self.get_images(document_id)

        # 2. OCR 엔진 선택 (문서 유형 기반)
        doc_meta = self.get_metadata(document_id)
        engine = self.select_engine(doc_meta)

        results = []
        for i, image_url in enumerate(images):
            # 3. 전처리
            processed = self.preprocess(image_url)

            # 4. OCR 실행
            text = engine.recognize(processed)

            # 5. 후처리 (철자 교정, 레이아웃 분석)
            cleaned = self.postprocess(text, doc_meta['language'])

            # 6. TEI XML 변환
            tei = self.to_tei(cleaned, page_number=i+1)
            results.append(tei)

        # 7. API에 전사 업로드
        self.upload_transcription(document_id, results)
        return results

    def select_engine(self, metadata):
        doc_type = metadata.get('type')
        date = metadata.get('date')

        if doc_type == 'manuscript' and int(date) < 1800:
            return TranskribusEngine()  # 필사본
        elif metadata.get('script') == 'hanja':
            return GoogleVisionEngine()  # 한자
        else:
            return TesseractEngine()  # 기본

    def postprocess(self, text, language):
        # 언어별 사후 교정
        if language == 'ko':
            text = self.correct_korean(text)
        elif language == 'hanja':
            text = self.correct_hanja(text)

        # 레이아웃 복원
        text = self.restore_layout(text)
        return text

pipeline = WIA_OCR_Pipeline(api_key='your-key')

07.2 지식 그래프 통합

역사 문서는 고립된 섬이 아닙니다. 인물, 장소, 사건, 개념이 복잡하게 연결되어 있습니다. 지식 그래프를 통합하면 "정약용과 동시대를 산 조선의 학자는 누구인가?", "종묘에서 거행된 의례를 기록한 모든 문서는?" 같은 복잡한 질문에 답할 수 있습니다.

Neo4j 그래프 데이터베이스

Neo4j는 가장 인기 있는 그래프 데이터베이스입니다. 노드(Node)와 관계(Relationship)로 데이터를 모델링합니다.

// Cypher 쿼리 언어 (Neo4j)

// 1. 문서와 인물 노드 생성
CREATE (doc:Document {
  id: 'joseon-annals-vol127',
  title: '조선왕조실록 제127권',
  date: 1481
})

CREATE (person:Person {
  id: 'seongjong',
  name: '성종',
  birth: 1457,
  death: 1494,
  role: '조선 9대 왕'
})

// 2. 관계 생성
CREATE (doc)-[:MENTIONS {context: '정월 초하루 제사'}]->(person)

// 3. 복잡한 쿼리: 정약용과 2촌 이내 관계
MATCH (dasan:Person {name: '정약용'})-[r:KNOWS*1..2]-(related:Person)
RETURN dasan, r, related

// 4. 특정 장소를 언급한 모든 문서
MATCH (place:Place {name: '종묘'})<-[:MENTIONS]-(doc:Document)
WHERE doc.date >= 1400 AND doc.date <= 1500
RETURN doc.title, doc.date
ORDER BY doc.date

// 5. 공동 저자 네트워크
MATCH (p1:Person)-[:AUTHORED]->(doc:Document)<-[:AUTHORED]-(p2:Person)
WHERE p1.id < p2.id
RETURN p1.name, p2.name, count(doc) as collaborations
ORDER BY collaborations DESC
LIMIT 20

Wikidata 연동

Wikidata는 위키미디어 재단의 오픈 지식 베이스입니다. 9,500만 개 항목, 13억 개 문장을 보유하고 있어 역사 인물, 장소, 사건 정보의 보고입니다.

import requests

def enrich_person_from_wikidata(person_name):
    # SPARQL 쿼리로 Wikidata 검색
    query = f"""
    SELECT ?person ?personLabel ?birth ?death ?occupation WHERE {{
      ?person rdfs:label "{person_name}"@ko .
      ?person wdt:P31 wd:Q5 .  # 인간
      OPTIONAL {{ ?person wdt:P569 ?birth }}  # 생년월일
      OPTIONAL {{ ?person wdt:P570 ?death }}  # 사망일
      OPTIONAL {{ ?person wdt:P106 ?occupation }}  # 직업
      SERVICE wikibase:label {{ bd:serviceParam wikibase:language "ko,en" }}
    }}
    LIMIT 1
    """

    url = 'https://query.wikidata.org/sparql'
    response = requests.get(url, params={'query': query, 'format': 'json'})
    data = response.json()

    if data['results']['bindings']:
        result = data['results']['bindings'][0]
        return {
            'wikidata_id': result['person']['value'],
            'birth': result.get('birth', {}).get('value'),
            'death': result.get('death', {}).get('value'),
            'occupation': result.get('occupationLabel', {}).get('value')
        }
    return None

07.3 블록체인 출처 추적

역사 문서의 진본 여부와 출처는 중요합니다. 블록체인 기술로 문서의 디지털화 과정, 소유권 이전, 전시 이력을 변조 불가능하게 기록할 수 있습니다.

Ethereum 스마트 컨트랙트

// Solidity 스마트 컨트랙트
pragma solidity ^0.8.0;

contract HeritageProvenance {
    struct Document {
        string id;
        string title;
        string iiifManifest;
        address owner;
        uint256 digitizationDate;
        string institution;
    }

    struct ProvenanceEvent {
        string eventType;  // "digitized", "transferred", "exhibited", "conserved"
        uint256 timestamp;
        address actor;
        string details;
        bytes32 hash;  // 문서 상태의 해시
    }

    mapping(string => Document) public documents;
    mapping(string => ProvenanceEvent[]) public provenance;

    event DocumentRegistered(string indexed id, address owner);
    event ProvenanceRecorded(string indexed id, string eventType);

    function registerDocument(
        string memory id,
        string memory title,
        string memory iiifManifest,
        string memory institution
    ) public {
        documents[id] = Document({
            id: id,
            title: title,
            iiifManifest: iiifManifest,
            owner: msg.sender,
            digitizationDate: block.timestamp,
            institution: institution
        });

        provenance[id].push(ProvenanceEvent({
            eventType: "digitized",
            timestamp: block.timestamp,
            actor: msg.sender,
            details: institution,
            hash: keccak256(abi.encodePacked(iiifManifest))
        }));

        emit DocumentRegistered(id, msg.sender);
    }

    function recordEvent(
        string memory id,
        string memory eventType,
        string memory details,
        bytes32 documentHash
    ) public {
        require(documents[id].owner == msg.sender, "Not document owner");

        provenance[id].push(ProvenanceEvent({
            eventType: eventType,
            timestamp: block.timestamp,
            actor: msg.sender,
            details: details,
            hash: documentHash
        }));

        emit ProvenanceRecorded(id, eventType);
    }

    function getProvenance(string memory id)
        public
        view
        returns (ProvenanceEvent[] memory)
    {
        return provenance[id];
    }
}

07.5 AI 자동 번역 통합

역사 문서는 다양한 언어로 작성되어 있어 접근성이 제한됩니다. AI 번역을 통합하면 라틴어, 한문, 아랍어 문서를 영어, 스페인어, 중국어 등 현대 언어로 자동 번역할 수 있습니다.

Google Cloud Translation API v3

from google.cloud import translate_v3

def translate_tei_document(tei_xml, target_language='en'):
    client = translate_v3.TranslationServiceClient()
    parent = f"projects/YOUR-PROJECT/locations/global"

    # TEI에서 텍스트 추출
    text_nodes = extract_text_from_tei(tei_xml)

    translations = []
    for node in text_nodes:
        # 번역 (문맥 보존)
        response = client.translate_text(
            request={
                "parent": parent,
                "contents": [node.text],
                "mime_type": "text/plain",
                "source_language_code": "ko",  # 또는 "hanja", "la"
                "target_language_code": target_language,
                "glossary_config": {
                    "glossary": get_custom_glossary()  # 전문 용어집
                }
            }
        )

        translations.append({
            'original': node.text,
            'translated': response.translations[0].translated_text,
            'element_id': node.id
        })

    # 번역된 텍스트를 TEI에 병합
    bilingual_tei = merge_translations(tei_xml, translations)
    return bilingual_tei

# 커스텀 용어집 (고유명사, 관직명 등)

예상치 못한 도전과 해결

도전 1: 중세 영어 OCR 정확도 70%

중세 영어는 현대 영어와 철자가 크게 달라 일반 OCR이 실패. "knight"가 "knyght", "daughter"가 "doughter"로 쓰임.

해결: 옥스포드 중세영어사전 8만 단어로 커스텀 언어 모델 훈련. 정확도 70% → 92%로 향상.

도전 2: Ethereum 가스비 급등

2023년 여름 가스비가 평균 $50/트랜잭션으로 급등. 15,000건 기록 시 $750,000 필요.

해결: Polygon (Ethereum L2)로 전환. 가스비 $0.01/트랜잭션으로 99.98% 절감.

도전 3: 저작권 복잡성

일부 필사본의 20세기 사진은 저작권 보호 중. 원본은 공공 도메인이지만 사진은 아님.

해결: 2,347건 재촬영. 나머지는 저작권자 협상 (67% 성공, CC BY-SA 동의).

07.7 미래 통합: AR/VR과 메타버스

차세대 통합은 증강현실(AR)과 가상현실(VR)입니다. 사용자가 가상 도서관을 걷고, 3D로 재현된 필사본을 손에 들어 볼 수 있습니다.

07.8 장 요약

✅ 핵심 내용
  1. AI/ML OCR 통합: Tesseract, Transkribus, Google Vision 등 5개 엔진 비교. 파이프라인으로 전처리→OCR→후처리→TEI 변환. 앙상블 OCR로 5-10% 정확도 향상.
  2. 지식 그래프: Neo4j로 문서-인물-장소-사건 관계 모델링. Wikidata SPARQL로 9,500만 항목 연동. SpaCy NER로 자동 개체명 추출.
  3. 블록체인 출처 추적: Ethereum 스마트 컨트랙트로 변조 불가능 이력 기록. IPFS 분산 저장소로 콘텐츠 주소 지정. NFT 인증서 발행.
  4. 디지털 도서관 통합: Europeana (5,800만 객체), DPLA EDM/MAP 매핑. 전 세계 사용자 노출.
  5. AI 자동 번역: Google Translate, DeepL, GPT-4 비교. 병렬 코퍼스로 파인튜닝. 커스텀 용어집으로 고유명사 정확도 향상.
  6. 대영도서관 사례: 15,347건 필사본, £2.3M 예산. OCR 정확도 92.3%, 월 사용자 230% 초과 달성. Polygon L2로 가스비 99.98% 절감.
  7. AR/VR 통합: WebXR로 가상 도서관 구현. 포토그래메트리 3D 스캔 (0.1mm 해상도).

07.9 복습 문제

  1. 5개 OCR 엔진(Tesseract, Transkribus, Google Vision, Kraken, ABBYY)을 비교하고, 16세기 한문 필사본 전사에 가장 적합한 엔진과 그 이유를 설명하세요.
  2. Neo4j Cypher 쿼리에서 "정약용과 2촌 이내 관계"를 찾는 쿼리를 분석하고, 이러한 그래프 쿼리가 전통적 SQL보다 유리한 이유를 논하세요.
  3. Ethereum 스마트 컨트랙트의 registerDocument() 함수를 설명하고, IPFS CID를 블록체인에 기록하는 것이 중앙 서버에 저장하는 것보다 안전한 이유를 기술하세요.
  4. Europeana EDM과 DPLA MAP의 차이점을 설명하고, 왜 두 플랫폼 모두 IIIF 매니페스트를 지원하는지 논하세요.
  5. 대영도서관의 "Illuminated Britain" 프로젝트에서 Ethereum 가스비 문제를 Polygon L2로 해결한 사례를 설명하고, Layer 2 솔루션의 장단점을 분석하세요.
  6. AI 번역에서 병렬 코퍼스와 커스텀 용어집의 역할을 설명하고, GPT-4가 전통적 기계 번역 엔진보다 고문서 번역에 유리한 이유를 기술하세요.
