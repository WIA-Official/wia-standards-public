## 3.3 데이터 모델

### 3.3.1 Translation Resource 구조

```typescript
// WIA-LANG-006 Standard Translation Format
interface WIATranslationResource {
  // 메타데이터
  $schema: 'https://wia.org/schemas/lang-006/v1.0';
  $version: '1.0.0';
  $locale: string;          // BCP 47 language tag (e.g., 'ko-KR')
  $namespace?: string;      // 네임스페이스 (e.g., 'common', 'auth')

  // 번역 데이터
  translations: {
    [key: string]: Translation | TranslationGroup;
  };

  // 메타정보
  metadata: {
    created: string;        // ISO 8601 timestamp
    updated: string;
    author?: string;
    translator?: string;
    reviewer?: string;
    status: 'draft' | 'review' | 'approved' | 'published';
    completeness: number;   // 0-100
  };
}

// Translation 타입
type Translation = string | PluralTranslation | GenderTranslation | ComplexTranslation;

// 복수형 번역
interface PluralTranslation {
  $type: 'plural';
  zero?: string;
  one?: string;
  two?: string;
  few?: string;
  many?: string;
  other: string;           // 필수
}

// 성별 변형 번역
interface GenderTranslation {
  $type: 'gender';
  male: string;
  female: string;
  other?: string;
}

// 복합 번역 (복수형 + 성별)
interface ComplexTranslation {
  $type: 'complex';
  rules: {
    [gender: string]: PluralTranslation;
  };
}

// 번역 그룹 (중첩 구조)
interface TranslationGroup {
  [key: string]: Translation | TranslationGroup;
}
```

// WIA-LANG-006 Initialization
import { WIALANG006 } from '@wia/lang-006';

// 1. 기본 초기화
const i18n = WIALANG006.init({
  defaultLocale: 'en',
  supportedLocales: ['en', 'ko', 'ja', 'zh', 'ar'],
  fallbackLocale: 'en',

  // 리소스 로딩 전략
  loadPath: '/locales/{{locale}}/{{namespace}}.json',

  // 성능 최적화
  preload: ['en', 'ko'],  // 미리 로드할 언어
  lazy: true,              // 지연 로딩

  // 캐싱
  cache: {
    enabled: true,
    maxAge: 3600000,  // 1 hour
    storage: 'localStorage'
  },

  // 개발 옵션
  debug: process.env.NODE_ENV === 'development',
  saveMissing: true,  // 누락된 키 자동 저장
});

// 2. 플러그인 확장
i18n.use(WIALANG006.plugins.validator);
i18n.use(WIALANG006.plugins.analyzer);
i18n.use(WIALANG006.plugins.translationMemory);

// 3. 이벤트 리스너
i18n.on('languageChanged', (locale) => {
  console.log(`Language changed to: ${locale}`);
  document.dir = i18n.getLocaleDirection(locale);
});

i18n.on('missingKey', (locale, key) => {
  console.warn(`Missing translation: ${locale}:${key}`);
});

// 4. 동적 리소스 로더
i18n.setResourceLoader(async (locale, namespace) => {
  const response = await fetch(`/api/translations/${locale}/${namespace}`);
  return response.json();
});

// WIA-LANG-006 Plugin System
interface WIAPlugin {
  name: string;
  version: string;
  type: PluginType;

  // 생명주기 훅
  init?(context: PluginContext): void;
  beforeTranslate?(key: string, locale: string): void;
  afterTranslate?(key: string, value: string, locale: string): string;
  destroy?(): void;
}

enum PluginType {
  LOADER = 'loader',           // 리소스 로더
  FORMATTER = 'formatter',     // 형식 지정자
  VALIDATOR = 'validator',     // 검증기
  ANALYZER = 'analyzer',       // 분석기
  TRANSFORMER = 'transformer'  // 변환기
}

// 플러그인 예제: Translation Memory
class TranslationMemoryPlugin implements WIAPlugin {
  name = 'translation-memory';
  version = '1.0.0';
  type = PluginType.TRANSFORMER;

  private tmService: TranslationMemoryService;

  async beforeTranslate(key: string, locale: string) {
    // TM에서 유사한 번역 검색
    const suggestions = await this.tmService.search(key, locale);
    return suggestions;
  }

  afterTranslate(key: string, value: string, locale: string) {
    // TM에 번역 저장
    this.tmService.save(key, value, locale);
    return value;
  }
}
