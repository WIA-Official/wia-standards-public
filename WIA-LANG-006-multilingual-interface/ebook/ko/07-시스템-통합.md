## 7.2 CMS 통합

### 7.2.1 Headless CMS 통합

```typescript
/**
 * WIA-LANG-006 + Headless CMS Integration
 * 예: Contentful, Strapi, Sanity
 */

// Contentful 통합
import { createClient } from 'contentful';
import { createWIALang } from '@wia/lang-006';

class ContentfulI18nAdapter {
  private contentful: any;
  private i18n: any;

  constructor(config: ContentfulConfig) {
    // Contentful 클라이언트
    this.contentful = createClient({
      space: config.spaceId,
      accessToken: config.accessToken
    });

    // WIA i18n
    this.i18n = createWIALang({
      defaultLocale: config.defaultLocale,
      supportedLocales: config.supportedLocales
    });
  }

  // Contentful에서 번역 가져오기
  async loadTranslationsFromCMS(locale: string): Promise<void> {
    const entries = await this.contentful.getEntries({
      content_type: 'translation',
      locale: locale
    });

    const translations = this.transformEntries(entries.items);

    // WIA i18n에 로드
    await this.i18n.addResourceBundle(locale, 'cms', translations);
  }

  // Contentful 엔트리를 번역 형식으로 변환
  private transformEntries(entries: any[]): Record<string, any> {
    const translations: Record<string, any> = {};

    for (const entry of entries) {
      const { key, value, namespace } = entry.fields;

      if (namespace) {
        if (!translations[namespace]) {
          translations[namespace] = {};
        }
        translations[namespace][key] = value;
      } else {
        translations[key] = value;
      }
    }

    return translations;
  }

  // 번역 내용 가져오기
  async getLocalizedContent(contentType: string, locale: string): Promise<any> {
    const entries = await this.contentful.getEntries({
      content_type: contentType,
      locale: locale
    });

    return entries.items.map(item => ({
      id: item.sys.id,
      ...item.fields
    }));
  }

  // 동적 번역
  async getLocalizedEntry(entryId: string, locale: string): Promise<any> {
    const entry = await this.contentful.getEntry(entryId, { locale });

    return {
      id: entry.sys.id,
      locale: locale,
      fields: entry.fields,
      metadata: {
        createdAt: entry.sys.createdAt,
        updatedAt: entry.sys.updatedAt
      }
    };
  }
}

// 사용 예제
const cmsAdapter = new ContentfulI18nAdapter({
  spaceId: 'your-space-id',
  accessToken: 'your-access-token',
  defaultLocale: 'en',
  supportedLocales: ['en', 'ko', 'ja', 'zh']
});

// CMS에서 번역 로드
await cmsAdapter.loadTranslationsFromCMS('ko');

// 현지화된 콘텐츠 가져오기
const products = await cmsAdapter.getLocalizedContent('product', 'ko');
```

### 7.2.2 WordPress 통합

```php
<?php
/**
 * WIA-LANG-006 WordPress Plugin
 */

class WIA_Lang_006_WordPress {
    private $i18n_service;
    private $default_locale;

    public function __construct() {
        $this->default_locale = get_locale();
        $this->init_hooks();
    }

    private function init_hooks() {
        // 번역 로드
        add_action('init', array($this, 'load_translations'));

        // REST API 엔드포인트
        add_action('rest_api_init', array($this, 'register_rest_routes'));

        // 숏코드
        add_shortcode('wia_translate', array($this, 'translate_shortcode'));

        // 언어 전환기 위젯
        add_action('widgets_init', array($this, 'register_widgets'));
    }

    // 번역 로드
    public function load_translations() {
        $locale = get_locale();

        // WIA 번역 파일 로드
        $translation_file = WP_CONTENT_DIR . "/languages/wia-lang-006/{$locale}.json";

        if (file_exists($translation_file)) {
            $translations = json_decode(file_get_contents($translation_file), true);
            $this->i18n_service->add_resource_bundle($locale, 'wordpress', $translations);
        }
    }

    // REST API 엔드포인트 등록
    public function register_rest_routes() {
        // 번역 조회
        register_rest_route('wia-lang/v1', '/translate', array(
            'methods' => 'POST',
            'callback' => array($this, 'rest_translate'),
            'permission_callback' => '__return_true'
        ));

        // 지원 언어 목록
        register_rest_route('wia-lang/v1', '/locales', array(
            'methods' => 'GET',
            'callback' => array($this, 'rest_get_locales'),
            'permission_callback' => '__return_true'
        ));
    }

    // REST API: 번역
    public function rest_translate(WP_REST_Request $request) {
        $key = $request->get_param('key');
        $locale = $request->get_param('locale') ?: $this->default_locale;
        $params = $request->get_param('params') ?: array();

        $translated = $this->i18n_service->translate($key, $locale, $params);

        return new WP_REST_Response(array(
            'key' => $key,
            'locale' => $locale,
            'translated' => $translated
        ), 200);
    }

    // 숏코드: [wia_translate key="welcome.message" name="User"]
    public function translate_shortcode($atts, $content = null) {
        $atts = shortcode_atts(array(
            'key' => '',
            'locale' => $this->default_locale
        ), $atts);

        // 나머지 속성은 파라미터로
        $params = array_diff_key($atts, array('key' => '', 'locale' => ''));

        return $this->i18n_service->translate($atts['key'], $atts['locale'], $params);
    }

    // 위젯: 언어 전환기
    public function register_widgets() {
        register_widget('WIA_Language_Switcher_Widget');
    }
}

// 플러그인 초기화
new WIA_Lang_006_WordPress();

// 언어 전환기 위젯
class WIA_Language_Switcher_Widget extends WP_Widget {
    public function __construct() {
        parent::__construct(
            'wia_language_switcher',
            __('WIA Language Switcher', 'wia-lang-006'),
            array('description' => __('Language switcher widget', 'wia-lang-006'))
        );
    }

    public function widget($args, $instance) {
        $locales = array('en_US', 'ko_KR', 'ja', 'zh_CN');
        $current_locale = get_locale();

        echo $args['before_widget'];
        echo '<div class="wia-language-switcher">';
        echo '<ul>';

        foreach ($locales as $locale) {
            $locale_name = $this->get_locale_name($locale);
            $url = add_query_arg('lang', $locale, get_permalink());
            $active = ($locale === $current_locale) ? 'active' : '';

            echo sprintf(
                '<li class="%s"><a href="%s">%s</a></li>',
                $active,
                esc_url($url),
                esc_html($locale_name)
            );
        }

        echo '</ul>';
        echo '</div>';
        echo $args['after_widget'];
    }

    private function get_locale_name($locale) {
        $names = array(
            'en_US' => 'English',
            'ko_KR' => '한국어',
            'ja' => '日本語',
            'zh_CN' => '中文'
        );

        return $names[$locale] ?? $locale;
    }
}
```

```sql
-- WIA-LANG-006 Database Schema

-- 언어 테이블
CREATE TABLE locales (
    code VARCHAR(10) PRIMARY KEY,  -- 'ko-KR'
    name VARCHAR(100) NOT NULL,    -- '한국어'
    english_name VARCHAR(100),     -- 'Korean'
    native_name VARCHAR(100),      -- '한국어'
    direction VARCHAR(3) DEFAULT 'ltr',  -- 'ltr' or 'rtl'
    script VARCHAR(4),             -- 'Hang'
    region VARCHAR(2),             -- 'KR'
    status VARCHAR(20) DEFAULT 'active',
    completeness DECIMAL(5,2) DEFAULT 0.00,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 네임스페이스 테이블
CREATE TABLE namespaces (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) UNIQUE NOT NULL,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 번역 테이블
CREATE TABLE translations (
    id SERIAL PRIMARY KEY,
    locale_code VARCHAR(10) REFERENCES locales(code),
    namespace_id INTEGER REFERENCES namespaces(id),
    key VARCHAR(500) NOT NULL,
    value JSONB NOT NULL,  -- 단순 문자열 또는 복잡한 구조
    type VARCHAR(20) DEFAULT 'simple',  -- 'simple', 'plural', 'gender', 'complex'
    context TEXT,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_by VARCHAR(100),
    updated_by VARCHAR(100),
    status VARCHAR(20) DEFAULT 'draft',  -- 'draft', 'review', 'approved', 'published'
    quality_score INTEGER,
    UNIQUE (locale_code, namespace_id, key)
);

-- 인덱스
CREATE INDEX idx_translations_locale ON translations(locale_code);
CREATE INDEX idx_translations_namespace ON translations(namespace_id);
CREATE INDEX idx_translations_key ON translations(key);
CREATE INDEX idx_translations_status ON translations(status);
CREATE INDEX idx_translations_updated ON translations(updated_at DESC);

-- 전문 검색 인덱스 (PostgreSQL)
CREATE INDEX idx_translations_value_gin ON translations USING gin(to_tsvector('simple', value::text));

-- 번역 메타데이터 테이블
CREATE TABLE translation_metadata (
    translation_id INTEGER REFERENCES translations(id) ON DELETE CASCADE,
    key VARCHAR(100),
    value TEXT,
    PRIMARY KEY (translation_id, key)
);

-- 번역 히스토리 테이블
CREATE TABLE translation_history (
    id SERIAL PRIMARY KEY,
    translation_id INTEGER REFERENCES translations(id) ON DELETE CASCADE,
    locale_code VARCHAR(10),
    key VARCHAR(500),
    old_value JSONB,
    new_value JSONB,
    changed_by VARCHAR(100),
    changed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    change_type VARCHAR(20)  -- 'create', 'update', 'delete'
);

-- 트리거: 번역 업데이트 시 히스토리 기록
CREATE OR REPLACE FUNCTION log_translation_change()
RETURNS TRIGGER AS $$
BEGIN
    IF (TG_OP = 'UPDATE') THEN
        INSERT INTO translation_history (
            translation_id,
            locale_code,
            key,
            old_value,
            new_value,
            changed_by,
            change_type
        ) VALUES (
            NEW.id,
            NEW.locale_code,
            NEW.key,
            OLD.value,
            NEW.value,
            NEW.updated_by,
            'update'
        );
    ELSIF (TG_OP = 'DELETE') THEN
        INSERT INTO translation_history (
            translation_id,
            locale_code,
            key,
            old_value,
            changed_by,
            change_type
        ) VALUES (
            OLD.id,
            OLD.locale_code,
            OLD.key,
            OLD.value,
            OLD.updated_by,
            'delete'
        );
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER translation_change_trigger
AFTER UPDATE OR DELETE ON translations
FOR EACH ROW EXECUTE FUNCTION log_translation_change();

-- 뷰: 번역 통계
CREATE VIEW translation_stats AS
SELECT
    l.code AS locale,
    l.name AS locale_name,
    n.name AS namespace,
    COUNT(*) AS total_keys,
    COUNT(CASE WHEN t.status = 'published' THEN 1 END) AS published_keys,
    COUNT(CASE WHEN t.value IS NULL OR t.value::text = '' THEN 1 END) AS missing_keys,
    ROUND(
        (COUNT(CASE WHEN t.status = 'published' THEN 1 END)::DECIMAL /
         NULLIF(COUNT(*), 0)) * 100,
        2
    ) AS completeness
FROM locales l
CROSS JOIN namespaces n
LEFT JOIN translations t ON t.locale_code = l.code AND t.namespace_id = n.id
GROUP BY l.code, l.name, n.id, n.name;
```

### 7.3.2 Node.js Database Adapter

```typescript
import { Pool } from 'pg';

class PostgreSQLTranslationAdapter {
  private pool: Pool;

  constructor(config: DatabaseConfig) {
    this.pool = new Pool({
      host: config.host,
      port: config.port,
      database: config.database,
      user: config.user,
      password: config.password
    });
  }

  // 번역 조회
  async getTranslation(
    locale: string,
    namespace: string,
    key: string
  ): Promise<Translation | null> {
    const query = `
      SELECT t.*, n.name as namespace_name
      FROM translations t
      JOIN namespaces n ON t.namespace_id = n.id
      WHERE t.locale_code = $1
        AND n.name = $2
        AND t.key = $3
        AND t.status = 'published'
    `;

    const result = await this.pool.query(query, [locale, namespace, key]);

    if (result.rows.length === 0) {
      return null;
    }

    return this.mapRowToTranslation(result.rows[0]);
  }

  // 네임스페이스의 모든 번역 조회
  async getTranslations(
    locale: string,
    namespace: string
  ): Promise<Record<string, any>> {
    const query = `
      SELECT t.key, t.value, t.type
      FROM translations t
      JOIN namespaces n ON t.namespace_id = n.id
      WHERE t.locale_code = $1
        AND n.name = $2
        AND t.status = 'published'
    `;

    const result = await this.pool.query(query, [locale, namespace]);

    const translations: Record<string, any> = {};

    for (const row of result.rows) {
      const value = JSON.parse(row.value);

      // 단순 문자열인 경우
      if (row.type === 'simple' && typeof value === 'string') {
        translations[row.key] = value;
      } else {
        translations[row.key] = value;
      }
    }

    return translations;
  }

  // 번역 생성
  async createTranslation(data: CreateTranslationData): Promise<Translation> {
    // 네임스페이스 ID 조회
    const nsQuery = 'SELECT id FROM namespaces WHERE name = $1';
    const nsResult = await this.pool.query(nsQuery, [data.namespace]);

    if (nsResult.rows.length === 0) {
      throw new Error(`Namespace not found: ${data.namespace}`);
    }

    const namespaceId = nsResult.rows[0].id;

    // 번역 삽입
    const query = `
      INSERT INTO translations (
        locale_code, namespace_id, key, value, type,
        context, description, created_by, status
      ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
      RETURNING *
    `;

    const values = [
      data.locale,
      namespaceId,
      data.key,
      JSON.stringify(data.value),
      data.type || 'simple',
      data.context,
      data.description,
      data.createdBy,
      data.status || 'draft'
    ];

    const result = await this.pool.query(query, values);

    return this.mapRowToTranslation(result.rows[0]);
  }

  // 번역 업데이트
  async updateTranslation(
    locale: string,
    namespace: string,
    key: string,
    data: UpdateTranslationData
  ): Promise<Translation> {
    const nsQuery = 'SELECT id FROM namespaces WHERE name = $1';
    const nsResult = await this.pool.query(nsQuery, [namespace]);

    if (nsResult.rows.length === 0) {
      throw new Error(`Namespace not found: ${namespace}`);
    }

    const namespaceId = nsResult.rows[0].id;

    const query = `
      UPDATE translations
      SET value = $1,
          description = $2,
          updated_by = $3,
          updated_at = CURRENT_TIMESTAMP
      WHERE locale_code = $4
        AND namespace_id = $5
        AND key = $6
      RETURNING *
    `;

    const values = [
      JSON.stringify(data.value),
      data.description,
      data.updatedBy,
      locale,
      namespaceId,
      key
    ];

    const result = await this.pool.query(query, values);

    if (result.rows.length === 0) {
      throw new Error('Translation not found');
    }

    return this.mapRowToTranslation(result.rows[0]);
  }

  // 번역 통계
  async getStats(locale: string): Promise<TranslationStats> {
    const query = `
      SELECT * FROM translation_stats
      WHERE locale = $1
    `;

    const result = await this.pool.query(query, [locale]);

    return result.rows.map(row => ({
      locale: row.locale,
      localeName: row.locale_name,
      namespace: row.namespace,
      totalKeys: parseInt(row.total_keys),
      publishedKeys: parseInt(row.published_keys),
      missingKeys: parseInt(row.missing_keys),
      completeness: parseFloat(row.completeness)
    }));
  }

  // 누락된 키 찾기
  async findMissingKeys(
    baseLocale: string,
    targetLocale: string,
    namespace: string
  ): Promise<string[]> {
    const query = `
      SELECT base.key
      FROM translations base
      JOIN namespaces n ON base.namespace_id = n.id
      LEFT JOIN translations target ON
        target.locale_code = $2 AND
        target.namespace_id = base.namespace_id AND
        target.key = base.key
      WHERE base.locale_code = $1
        AND n.name = $3
        AND target.id IS NULL
    `;

    const result = await this.pool.query(query, [
      baseLocale,
      targetLocale,
      namespace
    ]);

    return result.rows.map(row => row.key);
  }

  private mapRowToTranslation(row: any): Translation {
    return {
      id: row.id,
      locale: row.locale_code,
      namespace: row.namespace_name,
      key: row.key,
      value: JSON.parse(row.value),
      type: row.type,
      context: row.context,
      description: row.description,
      status: row.status,
      qualityScore: row.quality_score,
      createdAt: row.created_at,
      updatedAt: row.updated_at,
      createdBy: row.created_by,
      updatedBy: row.updated_by
    };
  }

  async close(): Promise<void> {
    await this.pool.end();
  }
}
```

```typescript
import axios from 'axios';

class CrowdinIntegrationAdapter {
  private apiToken: string;
  private projectId: string;
  private baseURL = 'https://api.crowdin.com/api/v2';

  constructor(config: CrowdinConfig) {
    this.apiToken = config.apiToken;
    this.projectId = config.projectId;
  }

  // Crowdin에 소스 파일 업로드
  async uploadSourceFile(filePath: string, content: any): Promise<void> {
    // 1. 스토리지에 파일 업로드
    const storageId = await this.uploadToStorage(content);

    // 2. 프로젝트에 파일 추가
    await axios.post(
      `${this.baseURL}/projects/${this.projectId}/files`,
      {
        storageId,
        name: filePath,
        title: filePath,
        type: 'json',
        parserVersion: 3
      },
      {
        headers: {
          'Authorization': `Bearer ${this.apiToken}`,
          'Content-Type': 'application/json'
        }
      }
    );
  }

  // 번역 다운로드
  async downloadTranslations(locale: string): Promise<any> {
    // 1. 빌드 시작
    const buildResponse = await axios.post(
      `${this.baseURL}/projects/${this.projectId}/translations/builds`,
      {
        targetLanguageIds: [locale]
      },
      {
        headers: {
          'Authorization': `Bearer ${this.apiToken}`
        }
      }
    );

    const buildId = buildResponse.data.data.id;

    // 2. 빌드 완료 대기
    await this.waitForBuild(buildId);

    // 3. 번역 파일 다운로드
    const downloadResponse = await axios.get(
      `${this.baseURL}/projects/${this.projectId}/translations/builds/${buildId}/download`,
      {
        headers: {
          'Authorization': `Bearer ${this.apiToken}`
        },
        responseType: 'arraybuffer'
      }
    );

    return downloadResponse.data;
  }

  // 번역 진행률 조회
  async getProgress(): Promise<LanguageProgress[]> {
    const response = await axios.get(
      `${this.baseURL}/projects/${this.projectId}/languages/progress`,
      {
        headers: {
          'Authorization': `Bearer ${this.apiToken}`
        }
      }
    );

    return response.data.data.map((item: any) => ({
      languageId: item.data.languageId,
      translationProgress: item.data.translationProgress,
      approvalProgress: item.data.approvalProgress
    }));
  }

  // 웹훅 처리
  async handleWebhook(payload: CrowdinWebhookPayload): Promise<void> {
    switch (payload.event) {
      case 'translation.updated':
        await this.onTranslationUpdated(payload);
        break;

      case 'file.approved':
        await this.onFileApproved(payload);
        break;

      case 'project.translated':
        await this.onProjectTranslated(payload);
        break;
    }
  }

  private async uploadToStorage(content: any): Promise<string> {
    const response = await axios.post(
      `${this.baseURL}/storages`,
      content,
      {
        headers: {
          'Authorization': `Bearer ${this.apiToken}`,
          'Crowdin-API-FileName': 'translations.json',
          'Content-Type': 'application/json'
        }
      }
    );

    return response.data.data.id;
  }

  private async waitForBuild(buildId: string): Promise<void> {
    const maxAttempts = 30;
    let attempts = 0;

    while (attempts < maxAttempts) {
      const response = await axios.get(
        `${this.baseURL}/projects/${this.projectId}/translations/builds/${buildId}`,
        {
          headers: {
            'Authorization': `Bearer ${this.apiToken}`
          }
        }
      );

      if (response.data.data.status === 'finished') {
        return;
      }

      await new Promise(resolve => setTimeout(resolve, 2000));
      attempts++;
    }

    throw new Error('Build timeout');
  }
}
```

```typescript
import { Counter, Histogram, Gauge } from 'prom-client';

class I18nMetrics {
  // 번역 요청 수
  private translationRequests = new Counter({
    name: 'wia_lang_translation_requests_total',
    help: 'Total number of translation requests',
    labelNames: ['locale', 'namespace', 'status']
  });

  // 번역 조회 시간
  private translationDuration = new Histogram({
    name: 'wia_lang_translation_duration_seconds',
    help: 'Translation lookup duration in seconds',
    labelNames: ['locale', 'source'],
    buckets: [0.001, 0.005, 0.01, 0.05, 0.1, 0.5, 1]
  });

  // 캐시 히트율
  private cacheHits = new Counter({
    name: 'wia_lang_cache_hits_total',
    help: 'Total number of cache hits',
    labelNames: ['locale']
  });

  private cacheMisses = new Counter({
    name: 'wia_lang_cache_misses_total',
    help: 'Total number of cache misses',
    labelNames: ['locale']
  });

  // 현재 로드된 언어 수
  private loadedLocales = new Gauge({
    name: 'wia_lang_loaded_locales',
    help: 'Number of currently loaded locales'
  });

  // 번역 완성도
  private completeness = new Gauge({
    name: 'wia_lang_completeness_percent',
    help: 'Translation completeness percentage',
    labelNames: ['locale', 'namespace']
  });

  recordTranslationRequest(locale: string, namespace: string, success: boolean) {
    this.translationRequests.inc({
      locale,
      namespace,
      status: success ? 'success' : 'error'
    });
  }

  recordTranslationDuration(locale: string, source: string, duration: number) {
    this.translationDuration.observe({ locale, source }, duration);
  }

  recordCacheHit(locale: string) {
    this.cacheHits.inc({ locale });
  }

  recordCacheMiss(locale: string) {
    this.cacheMisses.inc({ locale });
  }

  updateLoadedLocales(count: number) {
    this.loadedLocales.set(count);
  }

  updateCompleteness(locale: string, namespace: string, percent: number) {
    this.completeness.set({ locale, namespace }, percent);
  }
}
```
