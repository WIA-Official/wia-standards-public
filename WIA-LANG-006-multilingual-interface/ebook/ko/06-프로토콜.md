## 6.3 동기화 프로토콜

### 6.3.1 클라이언트-서버 동기화

```typescript
/**
 * WIA-LANG-006 Synchronization Protocol
 * 클라이언트와 서버 간 번역 데이터 동기화
 */

interface SyncProtocol {
  // 동기화 전략
  strategy: 'pull' | 'push' | 'bidirectional';

  // 동기화 모드
  mode: 'full' | 'incremental' | 'delta';

  // 충돌 해결
  conflictResolution: 'server-wins' | 'client-wins' | 'manual' | 'merge';
}

// Pull 기반 동기화
class PullSyncStrategy {
  async sync(options: SyncOptions): Promise<SyncResult> {
    // 1. 클라이언트가 마지막 동기화 시간 전송
    const lastSyncTime = await this.getLastSyncTime();

    // 2. 서버에서 변경사항 조회
    const changes = await this.fetchChanges({
      locale: options.locale,
      since: lastSyncTime
    });

    // 3. 로컬에 적용
    await this.applyChanges(changes);

    // 4. 동기화 시간 업데이트
    await this.setLastSyncTime(Date.now());

    return {
      success: true,
      itemsUpdated: changes.length,
      timestamp: Date.now()
    };
  }

  async fetchChanges(params: FetchChangesParams): Promise<Change[]> {
    const response = await fetch('/api/v1/sync/changes', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(params)
    });

    return response.json();
  }
}

// Push 기반 동기화
class PushSyncStrategy {
  async sync(options: SyncOptions): Promise<SyncResult> {
    // 1. 로컬 변경사항 수집
    const localChanges = await this.collectLocalChanges();

    // 2. 서버로 전송
    const result = await this.pushChanges(localChanges);

    // 3. 충돌 처리
    if (result.conflicts.length > 0) {
      await this.resolveConflicts(result.conflicts);
    }

    // 4. 로컬 상태 업데이트
    await this.updateLocalState(result);

    return result;
  }

  async pushChanges(changes: Change[]): Promise<PushResult> {
    const response = await fetch('/api/v1/sync/push', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ changes })
    });

    return response.json();
  }
}

// 양방향 동기화
class BidirectionalSyncStrategy {
  async sync(options: SyncOptions): Promise<SyncResult> {
    // 1. 로컬 변경사항 수집
    const localChanges = await this.collectLocalChanges();

    // 2. 서버와 교환
    const { serverChanges, conflicts } = await this.exchange({
      clientChanges: localChanges,
      lastSyncTime: await this.getLastSyncTime()
    });

    // 3. 충돌 해결
    const resolved = await this.resolveConflicts(conflicts);

    // 4. 변경사항 적용
    await this.applyChanges([...serverChanges, ...resolved]);

    // 5. 동기화 완료
    await this.setLastSyncTime(Date.now());

    return {
      success: true,
      itemsUpdated: serverChanges.length + resolved.length,
      conflicts: conflicts.length
    };
  }
}

// Delta 동기화 (증분 업데이트)
interface DeltaSync {
  // 변경 타입
  operation: 'create' | 'update' | 'delete';

  // 경로
  path: string;  // 예: "translations.common.welcome"

  // 이전 값 (update, delete)
  oldValue?: any;

  // 새 값 (create, update)
  newValue?: any;

  // 타임스탬프
  timestamp: number;

  // 버전
  version: number;

  // 체크섬
  checksum: string;
}

// 변경사항 적용
class DeltaApplier {
  async applyDelta(delta: DeltaSync): Promise<void> {
    // 체크섬 검증
    if (!this.verifyChecksum(delta)) {
      throw new Error('Checksum verification failed');
    }

    // 버전 충돌 확인
    const currentVersion = await this.getVersion(delta.path);
    if (currentVersion >= delta.version) {
      throw new ConflictError('Version conflict');
    }

    // 변경사항 적용
    switch (delta.operation) {
      case 'create':
        await this.create(delta.path, delta.newValue);
        break;

      case 'update':
        await this.update(delta.path, delta.newValue);
        break;

      case 'delete':
        await this.delete(delta.path);
        break;
    }

    // 버전 업데이트
    await this.setVersion(delta.path, delta.version);
  }

  private verifyChecksum(delta: DeltaSync): boolean {
    const computed = this.computeChecksum(delta.newValue);
    return computed === delta.checksum;
  }

  private computeChecksum(data: any): string {
    // SHA-256 해시
    return crypto.createHash('sha256')
      .update(JSON.stringify(data))
      .digest('hex');
  }
}
```

```typescript
/**
 * 오프라인 동기화
 * 네트워크 없이도 작동하고, 연결 시 자동 동기화
 */

class OfflineSyncManager {
  private queue: SyncQueue;
  private storage: OfflineStorage;

  constructor() {
    this.queue = new SyncQueue();
    this.storage = new OfflineStorage();

    // 온라인 상태 감지
    window.addEventListener('online', () => this.onOnline());
    window.addEventListener('offline', () => this.onOffline());
  }

  // 오프라인 번역 사용
  async translate(key: string, locale: string): Promise<string> {
    // 1. 로컬 캐시 확인
    const cached = await this.storage.get(locale, key);
    if (cached) {
      return cached;
    }

    // 2. 온라인인 경우 서버에서 가져오기
    if (navigator.onLine) {
      try {
        const translation = await this.fetchFromServer(key, locale);
        await this.storage.set(locale, key, translation);
        return translation;
      } catch (error) {
        // 네트워크 오류 시 폴백
        return this.getFallback(key, locale);
      }
    }

    // 3. 오프라인인 경우 폴백
    return this.getFallback(key, locale);
  }

  // 오프라인 변경사항 큐에 추가
  async queueChange(change: Change): Promise<void> {
    await this.queue.enqueue(change);

    // 온라인인 경우 즉시 동기화 시도
    if (navigator.onLine) {
      await this.syncQueue();
    }
  }

  // 온라인 전환 시
  private async onOnline(): Promise<void> {
    console.log('Network online - starting sync');

    // 큐에 있는 변경사항 동기화
    await this.syncQueue();

    // 최신 번역 다운로드
    await this.downloadLatestTranslations();
  }

  // 오프라인 전환 시
  private onOffline(): void {
    console.log('Network offline - queuing changes');
  }

  // 큐 동기화
  private async syncQueue(): Promise<void> {
    const changes = await this.queue.getAll();

    for (const change of changes) {
      try {
        await this.syncChange(change);
        await this.queue.remove(change.id);
      } catch (error) {
        console.error('Sync failed:', error);
        // 재시도 로직
        if (change.retryCount < 3) {
          change.retryCount++;
          await this.queue.update(change);
        }
      }
    }
  }
}

class OfflineStorage {
  private db: IDBDatabase;

  async init(): Promise<void> {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open('WIA-LANG-006', 1);

      request.onupgradeneeded = (event) => {
        const db = (event.target as IDBOpenDBRequest).result;

        // 번역 저장소
        if (!db.objectStoreNames.contains('translations')) {
          const store = db.createObjectStore('translations', {
            keyPath: ['locale', 'key']
          });
          store.createIndex('locale', 'locale');
          store.createIndex('timestamp', 'timestamp');
        }

        // 동기화 큐
        if (!db.objectStoreNames.contains('syncQueue')) {
          const queue = db.createObjectStore('syncQueue', {
            keyPath: 'id',
            autoIncrement: true
          });
          queue.createIndex('timestamp', 'timestamp');
        }
      };

      request.onsuccess = () => {
        this.db = request.result;
        resolve();
      };

      request.onerror = () => reject(request.error);
    });
  }

  async get(locale: string, key: string): Promise<string | null> {
    return new Promise((resolve, reject) => {
      const tx = this.db.transaction('translations', 'readonly');
      const store = tx.objectStore('translations');
      const request = store.get([locale, key]);

      request.onsuccess = () => {
        const result = request.result;
        resolve(result ? result.value : null);
      };

      request.onerror = () => reject(request.error);
    });
  }

  async set(locale: string, key: string, value: string): Promise<void> {
    return new Promise((resolve, reject) => {
      const tx = this.db.transaction('translations', 'readwrite');
      const store = tx.objectStore('translations');

      const data = {
        locale,
        key,
        value,
        timestamp: Date.now()
      };

      const request = store.put(data);

      request.onsuccess = () => resolve();
      request.onerror = () => reject(request.error);
    });
  }
}
```

```typescript
/**
 * WIA-LANG-006 WebSocket Protocol
 * 실시간 번역 업데이트
 */

class WIAWebSocketClient {
  private ws: WebSocket;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  private reconnectDelay = 1000;
  private subscriptions: Set<string> = new Set();

  connect(url: string, options: ConnectionOptions): void {
    this.ws = new WebSocket(url);

    this.ws.onopen = () => {
      console.log('WebSocket connected');
      this.reconnectAttempts = 0;

      // 인증
      this.authenticate(options.token);

      // 구독 복원
      this.restoreSubscriptions();
    };

    this.ws.onmessage = (event) => {
      const message = JSON.parse(event.data) as WTPMessage;
      this.handleMessage(message);
    };

    this.ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };

    this.ws.onclose = () => {
      console.log('WebSocket closed');
      this.attemptReconnect(url, options);
    };
  }

  // 메시지 전송
  send(message: WTPMessage): void {
    if (this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(message));
    } else {
      throw new Error('WebSocket is not connected');
    }
  }

  // 채널 구독
  subscribe(channel: string): void {
    this.subscriptions.add(channel);

    this.send({
      version: '1.0',
      id: this.generateId(),
      timestamp: Date.now(),
      type: MessageType.SUBSCRIBE,
      payload: { channel }
    });
  }

  // 구독 취소
  unsubscribe(channel: string): void {
    this.subscriptions.delete(channel);

    this.send({
      version: '1.0',
      id: this.generateId(),
      timestamp: Date.now(),
      type: MessageType.UNSUBSCRIBE,
      payload: { channel }
    });
  }

  // 메시지 처리
  private handleMessage(message: WTPMessage): void {
    switch (message.type) {
      case MessageType.TRANSLATION_UPDATED:
        this.onTranslationUpdated(message.payload);
        break;

      case MessageType.TRANSLATION_CREATED:
        this.onTranslationCreated(message.payload);
        break;

      case MessageType.TRANSLATION_DELETED:
        this.onTranslationDeleted(message.payload);
        break;

      case MessageType.PONG:
        this.onPong(message);
        break;

      case MessageType.ERROR:
        this.onError(message.payload);
        break;
    }
  }

  // 재연결
  private attemptReconnect(url: string, options: ConnectionOptions): void {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      const delay = this.reconnectDelay * Math.pow(2, this.reconnectAttempts - 1);

      console.log(`Reconnecting in ${delay}ms (attempt ${this.reconnectAttempts})`);

      setTimeout(() => {
        this.connect(url, options);
      }, delay);
    } else {
      console.error('Max reconnection attempts reached');
    }
  }

  // Heartbeat
  startHeartbeat(interval: number = 30000): void {
    setInterval(() => {
      if (this.ws.readyState === WebSocket.OPEN) {
        this.send({
          version: '1.0',
          id: this.generateId(),
          timestamp: Date.now(),
          type: MessageType.PING,
          payload: {}
        });
      }
    }, interval);
  }

  private generateId(): string {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }
}

class WIAWebSocketServer {

      this.handleConnection(ws, req);
    });
  }

  private handleConnection(ws: WebSocket, req: http.IncomingMessage): void {
    const clientId = this.generateClientId();

    const client: ClientConnection = {
      id: clientId,
      ws,
      authenticated: false,
      subscriptions: new Set(),
      locale: 'en',
      lastActivity: Date.now()
    };

    this.clients.set(clientId, client);

    ws.on('message', (data) => {
      const message = JSON.parse(data.toString()) as WTPMessage;
      this.handleMessage(clientId, message);
    });

    ws.on('close', () => {
      this.handleDisconnect(clientId);
    });

    ws.on('error', (error) => {
      console.error(`Client ${clientId} error:`, error);
    });
  }

  private handleMessage(clientId: string, message: WTPMessage): void {
    const client = this.clients.get(clientId);
    if (!client) return;

    client.lastActivity = Date.now();

    switch (message.type) {
      case MessageType.SUBSCRIBE:
        this.handleSubscribe(clientId, message.payload.channel);
        break;

      case MessageType.UNSUBSCRIBE:
        this.handleUnsubscribe(clientId, message.payload.channel);
        break;

      case MessageType.PING:
        this.sendPong(clientId, message.id);
        break;

      case MessageType.TRANSLATE_REQUEST:
        this.handleTranslateRequest(clientId, message);
        break;
    }
  }

  // 채널에 메시지 브로드캐스트
  broadcast(channel: string, message: WTPMessage): void {
    const subscribers = this.channels.get(channel);
    if (!subscribers) return;

    for (const clientId of subscribers) {
      const client = this.clients.get(clientId);
      if (client && client.ws.readyState === WebSocket.OPEN) {
        client.ws.send(JSON.stringify(message));
      }
    }
  }

  // 번역 업데이트 브로드캐스트
  broadcastTranslationUpdate(locale: string, key: string, value: string): void {
    const channel = `translations:${locale}`;

    this.broadcast(channel, {
      version: '1.0',
      id: this.generateId(),
      timestamp: Date.now(),
      type: MessageType.TRANSLATION_UPDATED,
      payload: {
        locale,
        key,
        value,
        timestamp: Date.now()
      }
    });
  }

  private handleSubscribe(clientId: string, channel: string): void {
    const client = this.clients.get(clientId);
    if (!client) return;

    // 클라이언트 구독 추가
    client.subscriptions.add(channel);

    // 채널 구독자 추가
    if (!this.channels.has(channel)) {
      this.channels.set(channel, new Set());
    }
    this.channels.get(channel)!.add(clientId);

    console.log(`Client ${clientId} subscribed to ${channel}`);
  }

  private handleUnsubscribe(clientId: string, channel: string): void {
    const client = this.clients.get(clientId);
    if (!client) return;

    client.subscriptions.delete(channel);

    const subscribers = this.channels.get(channel);
    if (subscribers) {
      subscribers.delete(clientId);
    }

    console.log(`Client ${clientId} unsubscribed from ${channel}`);
  }

  private handleDisconnect(clientId: string): void {
    const client = this.clients.get(clientId);
    if (!client) return;

    // 모든 구독 제거
    for (const channel of client.subscriptions) {
      const subscribers = this.channels.get(channel);
      if (subscribers) {
        subscribers.delete(clientId);
      }
    }

    this.clients.delete(clientId);
    console.log(`Client ${clientId} disconnected`);
  }

  private sendPong(clientId: string, requestId: string): void {
    const client = this.clients.get(clientId);
    if (!client) return;

    const pong: WTPMessage = {
      version: '1.0',
      id: requestId,
      timestamp: Date.now(),
      type: MessageType.PONG,
      payload: {}
    };

    client.ws.send(JSON.stringify(pong));
  }

  private generateClientId(): string {
    return `client-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  private generateId(): string {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }
}
```

```typescript
/**
 * Server-Sent Events for Translation Updates
 * 단방향 서버→클라이언트 스트리밍
 */

// 클라이언트
class WIASSEClient {
  private eventSource: EventSource;
  private listeners: Map<string, Set<EventListener>> = new Map();

  connect(url: string, options: SSEOptions): void {
    this.eventSource = new EventSource(url);

    this.eventSource.onopen = () => {
      console.log('SSE connected');
    };

    this.eventSource.onerror = (error) => {
      console.error('SSE error:', error);

      // 자동 재연결 (EventSource가 자동으로 처리)
    };

    // 번역 업데이트 이벤트
    this.eventSource.addEventListener('translation-updated', (event) => {
      const data = JSON.parse(event.data);
      this.emit('translation-updated', data);
    });

    // 번역 생성 이벤트
    this.eventSource.addEventListener('translation-created', (event) => {
      const data = JSON.parse(event.data);
      this.emit('translation-created', data);
    });

    // 번역 삭제 이벤트
    this.eventSource.addEventListener('translation-deleted', (event) => {
      const data = JSON.parse(event.data);
      this.emit('translation-deleted', data);
    });
  }

  on(event: string, listener: EventListener): void {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set());
    }
    this.listeners.get(event)!.add(listener);
  }

  off(event: string, listener: EventListener): void {
    const listeners = this.listeners.get(event);
    if (listeners) {
      listeners.delete(listener);
    }
  }

  private emit(event: string, data: any): void {
    const listeners = this.listeners.get(event);
    if (listeners) {
      listeners.forEach(listener => listener(data));
    }
  }

  disconnect(): void {
    this.eventSource.close();
  }
}

    }
  };

  translationEmitter.on('updated', updateListener);

  req.on('close', () => {
```

```typescript
import jwt from 'jsonwebtoken';

function authenticateToken(
  req: express.Request,
  res: express.Response,
  next: express.NextFunction
) {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  if (!token) {
    return res.status(401).json({ error: 'No token provided' });
  }

  jwt.verify(token, process.env.JWT_PUBLIC_KEY!, (err, payload) => {
    if (err) {
      return res.status(403).json({ error: 'Invalid token' });
    }

    req.user = payload;
    next();
  });
}

// 권한 검사 미들웨어
function requireScope(...requiredScopes: string[]) {
  return (
    req: express.Request,
    res: express.Response,
    next: express.NextFunction
  ) => {
    const userScopes = req.user?.scope || [];

    const hasRequiredScope = requiredScopes.some(scope =>
      userScopes.includes(scope) || userScopes.includes('admin')
    );

    if (!hasRequiredScope) {
      return res.status(403).json({
        error: 'Insufficient permissions',
        required: requiredScopes,
        current: userScopes
      });
    }

    next();
  };
}
```
