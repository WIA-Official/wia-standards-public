### 8.2.1 React 구현 예제

```typescript
// src/i18n/WIALangProvider.tsx
import React, { createContext, useContext, useState, useEffect, useCallback } from 'react';

interface WIALangContextType {
  locale: string;
  setLocale: (locale: string) => void;
  t: (key: string, params?: Record<string, any>) => string;
  formatNumber: (value: number, options?: Intl.NumberFormatOptions) => string;
  formatDate: (date: Date, options?: Intl.DateTimeFormatOptions) => string;
  formatCurrency: (value: number, currency: string) => string;
  dir: 'ltr' | 'rtl';
  isLoading: boolean;
}

const WIALangContext = createContext<WIALangContextType | null>(null);

interface WIALangProviderProps {
  children: React.ReactNode;
  defaultLocale: string;
  supportedLocales: string[];
  loadTranslations: (locale: string) => Promise<Record<string, any>>;
}

export function WIALangProvider({
  children,
  defaultLocale,
  supportedLocales,
  loadTranslations
}: WIALangProviderProps) {
  const [locale, setLocaleState] = useState(defaultLocale);
  const [translations, setTranslations] = useState<Record<string, any>>({});
  const [isLoading, setIsLoading] = useState(true);

  // RTL 언어 목록
  const rtlLocales = ['ar', 'ar-SA', 'he', 'he-IL', 'fa', 'fa-IR', 'ur'];

  // 텍스트 방향 결정
  const dir = rtlLocales.some(rtl => locale.startsWith(rtl)) ? 'rtl' : 'ltr';

  // 번역 로드
  useEffect(() => {
    setIsLoading(true);
    loadTranslations(locale)
      .then(setTranslations)
      .finally(() => setIsLoading(false));
  }, [locale, loadTranslations]);

  // HTML 속성 업데이트
  useEffect(() => {
    document.documentElement.lang = locale;
    document.documentElement.dir = dir;
  }, [locale, dir]);

  // 로케일 변경
  const setLocale = useCallback((newLocale: string) => {
    if (supportedLocales.includes(newLocale)) {
      setLocaleState(newLocale);
      localStorage.setItem('wia-lang-locale', newLocale);
    }
  }, [supportedLocales]);

  // 번역 함수
  const t = useCallback((key: string, params?: Record<string, any>): string => {
    const keys = key.split('.');
    let value: any = translations;

    for (const k of keys) {
      value = value?.[k];
      if (value === undefined) break;
    }

    if (typeof value !== 'string') {
      console.warn(`Missing translation: ${key}`);
      return key;
    }

    // 파라미터 보간
    if (params) {
      return value.replace(/\{\{(\w+)\}\}/g, (_, name) =>
        params[name]?.toString() ?? `{{${name}}}`
      );
    }

    return value;
  }, [translations]);

  // 숫자 포맷
  const formatNumber = useCallback((value: number, options?: Intl.NumberFormatOptions) => {
    return new Intl.NumberFormat(locale, options).format(value);
  }, [locale]);

  // 날짜 포맷
  const formatDate = useCallback((date: Date, options?: Intl.DateTimeFormatOptions) => {
    return new Intl.DateTimeFormat(locale, options).format(date);
  }, [locale]);

  // 통화 포맷
  const formatCurrency = useCallback((value: number, currency: string) => {
    return new Intl.NumberFormat(locale, {
      style: 'currency',
      currency
    }).format(value);
  }, [locale]);

  return (
    <WIALangContext.Provider value={{
      locale,
      setLocale,
      t,
      formatNumber,
      formatDate,
      formatCurrency,
      dir,
      isLoading
    }}>
      {children}
    </WIALangContext.Provider>
  );
}

export function useWIALang() {
  const context = useContext(WIALangContext);
  if (!context) {
    throw new Error('useWIALang must be used within WIALangProvider');
  }
  return context;
}
```

### 8.2.2 언어 선택기 컴포넌트

```tsx
// src/components/LanguageSwitcher.tsx
import React from 'react';
import { useWIALang } from '../i18n/WIALangProvider';

interface LocaleInfo {
  code: string;
  name: string;
  nativeName: string;
  flag: string;
}

const locales: LocaleInfo[] = [
  { code: 'en-US', name: 'English', nativeName: 'English', flag: '🇺🇸' },
  { code: 'ko-KR', name: 'Korean', nativeName: '한국어', flag: '🇰🇷' },
  { code: 'ja-JP', name: 'Japanese', nativeName: '日本語', flag: '🇯🇵' },
  { code: 'zh-CN', name: 'Chinese', nativeName: '中文', flag: '🇨🇳' },
  { code: 'ar-SA', name: 'Arabic', nativeName: 'العربية', flag: '🇸🇦' },
];

export function LanguageSwitcher() {
  const { locale, setLocale, t } = useWIALang();

  return (
    <div className="language-switcher" role="navigation" aria-label={t('accessibility.languageSelector')}>
      <select
        value={locale}
        onChange={(e) => setLocale(e.target.value)}
        aria-label={t('accessibility.selectLanguage')}
      >
        {locales.map((loc) => (
          <option key={loc.code} value={loc.code}>
            {loc.flag} {loc.nativeName}
          </option>
        ))}
      </select>
    </div>
  );
}
```

### 8.4.3 자가 점검 체크리스트

```typescript
interface CertificationChecklist {
  // 필수 기능 (Bronze 이상)
  required: {
    localeDetection: boolean;        // 로케일 자동 감지
    localeSwitching: boolean;        // 로케일 전환
    keyBasedTranslation: boolean;    // 키 기반 번역
    interpolation: boolean;          // 파라미터 보간
    numberFormatting: boolean;       // 숫자 포맷
    dateFormatting: boolean;         // 날짜 포맷
    currencyFormatting: boolean;     // 통화 포맷
    missingKeyHandling: boolean;     // 누락 키 처리
    lazyLoading: boolean;            // 지연 로딩
    langAttribute: boolean;          // HTML lang 속성
  };

  // 권장 기능 (Silver 이상)
  recommended: {
    pluralization: boolean;          // 복수형 처리
    contextSupport: boolean;         // 컨텍스트 지원
    rtlSupport: boolean;             // RTL 지원
    relativeTime: boolean;           // 상대 시간
    listFormatting: boolean;         // 목록 포맷
    bundleSplitting: boolean;        // 번들 분리
    caching: boolean;                // 캐싱
  };

  // 고급 기능 (Gold 이상)
  advanced: {
    genderSupport: boolean;          // 성별 지원
    bidiIsolation: boolean;          // 양방향 격리
    mixedDirection: boolean;         // 혼합 방향
    preloading: boolean;             // 사전 로딩
    ariaLabels: boolean;             // ARIA 레이블
    screenReader: boolean;           // 스크린 리더
    keyboardNav: boolean;            // 키보드 네비게이션
  };

  // 테스트 커버리지
  testing: {
    unitTestCoverage: number;        // 단위 테스트 커버리지 (%)
    e2eTestCoverage: number;         // E2E 테스트 커버리지 (%)
    accessibilityScore: number;      // 접근성 점수 (0-100)
    performanceScore: number;        // 성능 점수 (0-100)
  };

  // 지원 언어
  supportedLocales: string[];
}

// 자가 점검 실행
function runSelfAssessment(checklist: CertificationChecklist): AssessmentResult {
  const requiredScore = Object.values(checklist.required)
    .filter(Boolean).length / Object.keys(checklist.required).length * 100;

  const recommendedScore = Object.values(checklist.recommended)
    .filter(Boolean).length / Object.keys(checklist.recommended).length * 100;

  const advancedScore = Object.values(checklist.advanced)
    .filter(Boolean).length / Object.keys(checklist.advanced).length * 100;

  const localeCount = checklist.supportedLocales.length;

  // 등급 결정
  let grade: 'Bronze' | 'Silver' | 'Gold' | 'Platinum' | 'Not Certified';

  if (requiredScore < 100 || localeCount < 2) {
    grade = 'Not Certified';
  } else if (requiredScore === 100 && recommendedScore >= 80 && advancedScore >= 80 && localeCount >= 20) {
    grade = 'Platinum';
  } else if (requiredScore === 100 && recommendedScore >= 70 && advancedScore >= 70 && localeCount >= 10) {
    grade = 'Gold';
  } else if (requiredScore === 100 && recommendedScore >= 50 && localeCount >= 5) {
    grade = 'Silver';
  } else {
    grade = 'Bronze';
  }

  return {
    grade,
    scores: {
      required: requiredScore,
      recommended: recommendedScore,
      advanced: advancedScore,
      testing: checklist.testing
    },
    localeCount,
    recommendations: generateRecommendations(checklist, grade)
  };
}
```
