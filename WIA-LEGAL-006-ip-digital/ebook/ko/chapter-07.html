07.1 상표 AI 시스템 개요

상표(Trademark)는 기업의 상품이나 서비스를 식별하고 구별하는 핵심적인 지식재산입니다. 전 세계적으로 연간 1,300만 건 이상의 상표가 출원되며, 등록된 상표는 수천만 건에 달합니다. 상표 출원 시 기존 상표와의 유사성을 평가하는 것은 등록 가능성과 분쟁 방지를 위해 필수적입니다. AI 기반 상표 유사도 분석 시스템은 이 과정을 자동화하고 정확도를 향상시킵니다.

상표 유사도 분석은 크게 세 가지 관점에서 이루어집니다: 외관적 유사성(Visual Similarity), 호칭적 유사성(Phonetic Similarity), 관념적 유사성(Conceptual Similarity). 외관적 유사성은 상표의 시각적 형태가 얼마나 비슷한지를, 호칭적 유사성은 발음이 얼마나 비슷한지를, 관념적 유사성은 의미나 인상이 얼마나 비슷한지를 평가합니다. AI 시스템은 컴퓨터 비전, 자연어 처리, 음성 인식 기술을 조합하여 이 세 가지 관점을 종합적으로 분석합니다.

WIA-LEGAL-006 표준은 상표 AI 시스템의 구현을 위한 기술적 프레임워크와 품질 기준을 제시합니다. 弘益人間의 정신에 따라, 이러한 기술이 중소기업과 스타트업도 효과적으로 브랜드를 보호하고 상표 분쟁을 예방할 수 있도록 지원해야 합니다.

// WIA-LEGAL-006 상표 이미지 검색 및 유사도 분석 시스템

interface TrademarkImage {
  id: string;
  applicationNumber: string;
  imageUrl: string;
  niceClasses: number[];
  applicant: string;
  status: 'pending' | 'registered' | 'rejected' | 'abandoned';
  filingDate: Date;
  wordMark?: string;
  viennaCode?: string[];
}

interface SimilarityResult {
  trademark: TrademarkImage;
  visualScore: number;
  phoneticScore: number;
  conceptualScore: number;
  overallScore: number;
  conflictRisk: 'high' | 'medium' | 'low';
  explanation: string;
}

class TrademarkSimilarityAnalyzer {
  private visualEncoder: VisualEncoder;
  private phoneticEncoder: PhoneticEncoder;
  private conceptEncoder: ConceptEncoder;
  private vectorStore: VectorStore;

  constructor(config: {
    visualModelPath: string;
    phoneticModelPath: string;
    conceptModelPath: string;
    vectorDbUrl: string;
  }) {
    this.visualEncoder = new VisualEncoder(config.visualModelPath);
    this.phoneticEncoder = new PhoneticEncoder(config.phoneticModelPath);
    this.conceptEncoder = new ConceptEncoder(config.conceptModelPath);
    this.vectorStore = new VectorStore(config.vectorDbUrl);
  }

  // 종합 유사 상표 검색
  async searchSimilarTrademarks(
    queryImage: Buffer | string,
    queryText?: string,
    options: SearchOptions = {}
  ): Promise {
    const {
      topK = 50,
      niceClasses = [],
      minScore = 0.5,
      includeInactive = false
    } = options;

    // 시각적 특성 추출
    const visualFeatures = await this.visualEncoder.encode(queryImage);

    // 음성적 특성 추출 (텍스트가 있는 경우)
    const phoneticFeatures = queryText
      ? await this.phoneticEncoder.encode(queryText)
      : null;

    // 개념적 특성 추출
    const conceptFeatures = queryText
      ? await this.conceptEncoder.encode(queryText)
      : await this.conceptEncoder.encodeFromImage(queryImage);

    // 벡터 검색
    const filter = this.buildFilter(niceClasses, includeInactive);

    const visualCandidates = await this.vectorStore.search({
      collection: 'trademark_visual',
      vector: visualFeatures,
      topK: topK * 2,
      filter
    });

    // 각 후보에 대해 세부 분석
    const results: SimilarityResult[] = [];

    for (const candidate of visualCandidates) {
      const trademark = await this.fetchTrademarkDetails(candidate.id);

      // 시각적 유사도 (이미 계산됨)
      const visualScore = candidate.score;

      // 음성적 유사도
      const phoneticScore = phoneticFeatures && trademark.wordMark
        ? await this.calculatePhoneticSimilarity(
            phoneticFeatures,
            trademark.wordMark
          )
        : 0;

      // 개념적 유사도
      const conceptScore = await this.calculateConceptualSimilarity(
        conceptFeatures,
        trademark
      );

      // 종합 점수 계산 (가중 평균)
      const overallScore = this.calculateOverallScore(
        visualScore,
        phoneticScore,
        conceptScore,
        trademark
      );

      if (overallScore >= minScore) {
        results.push({
          trademark,
          visualScore,
          phoneticScore,
          conceptualScore: conceptScore,
          overallScore,
          conflictRisk: this.assessConflictRisk(
            overallScore,
            niceClasses,
            trademark.niceClasses
          ),
          explanation: this.generateExplanation(
            visualScore,
            phoneticScore,
            conceptScore
          )
        });
      }
    }

    // 종합 점수로 정렬
    return results.sort((a, b) => b.overallScore - a.overallScore);
  }

  // 시각적 유사도 상세 분석
  async analyzeVisualSimilarity(
    image1: Buffer,
    image2: Buffer
  ): Promise {
    // 특성 추출
    const features1 = await this.visualEncoder.extractDetailedFeatures(image1);
    const features2 = await this.visualEncoder.extractDetailedFeatures(image2);

    // 색상 유사도
    const colorSimilarity = this.compareColors(
      features1.colorHistogram,
      features2.colorHistogram
    );

    // 형태 유사도
    const shapeSimilarity = this.compareShapes(
      features1.shapeDescriptor,
      features2.shapeDescriptor
    );

    // 레이아웃 유사도
    const layoutSimilarity = this.compareLayouts(
      features1.spatialFeatures,
      features2.spatialFeatures
    );

    // 텍스트 요소 유사도 (OCR 기반)
    const textSimilarity = await this.compareTextElements(
      features1.textRegions,
      features2.textRegions
    );

    return {
      overall: (colorSimilarity + shapeSimilarity + layoutSimilarity + textSimilarity) / 4,
      color: colorSimilarity,
      shape: shapeSimilarity,
      layout: layoutSimilarity,
      textElements: textSimilarity,
      heatmap: await this.generateSimilarityHeatmap(features1, features2)
    };
  }

  // 충돌 위험 평가
  private assessConflictRisk(
    overallScore: number,
    queryClasses: number[],
    existingClasses: number[]
  ): 'high' | 'medium' | 'low' {
    // 니스 분류 중복 확인
    const classOverlap = queryClasses.some(c => existingClasses.includes(c));

    if (overallScore >= 0.8 && classOverlap) return 'high';
    if (overallScore >= 0.6 && classOverlap) return 'medium';
    if (overallScore >= 0.8) return 'medium';
    return 'low';
  }

  // 종합 점수 계산
  private calculateOverallScore(
    visual: number,
    phonetic: number,
    conceptual: number,
    trademark: TrademarkImage
  ): number {
    // 상표 유형에 따른 가중치 조정
    const hasImage = trademark.imageUrl && !trademark.wordMark;
    const hasText = !!trademark.wordMark;

    if (hasImage && !hasText) {
      // 도형 상표: 시각적 유사성 강조
      return visual * 0.8 + conceptual * 0.2;
    } else if (hasText && !hasImage) {
      // 문자 상표: 음성적/관념적 유사성 강조
      return phonetic * 0.5 + conceptual * 0.3 + visual * 0.2;
    } else {
      // 결합 상표: 균형 잡힌 평가
      return visual * 0.4 + phonetic * 0.3 + conceptual * 0.3;
    }
  }

  // 설명 생성
  private generateExplanation(
    visual: number,
    phonetic: number,
    conceptual: number
  ): string {
    const parts: string[] = [];

    if (visual >= 0.7) {
      parts.push(`외관상 높은 유사성(${(visual * 100).toFixed(0)}%)`);
    }
    if (phonetic >= 0.7) {
      parts.push(`호칭상 높은 유사성(${(phonetic * 100).toFixed(0)}%)`);
    }
    if (conceptual >= 0.7) {
      parts.push(`관념상 높은 유사성(${(conceptual * 100).toFixed(0)}%)`);
    }

    return parts.length > 0
      ? parts.join(', ') + '이 발견됨'
      : '유의미한 유사성 없음';
  }

  private buildFilter(niceClasses: number[], includeInactive: boolean) {
    const filter: any = {};
    if (niceClasses.length > 0) {
      filter.niceClasses = { $in: niceClasses };
    }
    if (!includeInactive) {
      filter.status = { $in: ['pending', 'registered'] };
    }
    return filter;
  }
}

class PhoneticEncoder {
  private languageModels: Map = new Map();

  constructor(modelPath: string) {
    this.loadModels(modelPath);
  }

  async encode(text: string): Promise {
    const language = await this.detectLanguage(text);

    // 언어별 음성 인코딩
    const phonemes = await this.textToPhonemes(text, language);
    const features = this.phonemesToFeatures(phonemes);

    return features;
  }

  // 발음 유사도 계산
  calculateSimilarity(features1: number[], features2: number[]): number {
    // 편집 거리 기반 유사도
    const editDistance = this.levenshteinDistance(features1, features2);
    const maxLen = Math.max(features1.length, features2.length);
    return 1 - (editDistance / maxLen);
  }

  // 한국어/영어/중국어/일본어 발음 비교
  async compareMultilingual(
    text1: string,
    text2: string
  ): Promise<{ score: number; analysis: string }> {
    const lang1 = await this.detectLanguage(text1);
    const lang2 = await this.detectLanguage(text2);

    // 동일 언어 비교
    if (lang1 === lang2) {
      const f1 = await this.encode(text1);
      const f2 = await this.encode(text2);
      return {
        score: this.calculateSimilarity(f1, f2),
        analysis: `동일 언어(${lang1}) 발음 비교`
      };
    }

    // 크로스 언어 비교 (로마자 표기 기반)
    const roman1 = await this.romanize(text1, lang1);
    const roman2 = await this.romanize(text2, lang2);
    const f1 = await this.encodeRomanized(roman1);
    const f2 = await this.encodeRomanized(roman2);

    return {
      score: this.calculateSimilarity(f1, f2),
      analysis: `크로스 언어(${lang1}↔${lang2}) 로마자 발음 비교`
    };
  }

  private async detectLanguage(text: string): Promise {
    // 언어 감지 로직
    if (/[가-힣]/.test(text)) return 'ko';
    if (/[ぁ-んァ-ン]/.test(text)) return 'ja';
    if (/[\u4e00-\u9fff]/.test(text)) return 'zh';
    return 'en';
  }

07.5 비엔나 분류 기반 도형 상표 분석

// 비엔나 분류 기반 도형 상표 분석

interface ViennaCode {
  category: number;
  division: number;
  section: number;
  description: string;
}

class ViennaClassifier {
  private model: any;
  private codeDescriptions: Map;

  async classifyImage(image: Buffer): Promise {
    // 이미지에서 도형 요소 추출 및 분류
    const predictions = await this.model.predict(image);

    return predictions.map((p: any) => ({
      category: Math.floor(p.code / 10000),
      division: Math.floor((p.code % 10000) / 100),
      section: p.code % 100,
      description: this.codeDescriptions.get(p.code.toString()) || '',
      confidence: p.confidence
    }));
  }

  // 비엔나 코드 기반 유사성 계산
  calculateViennaSimilarity(codes1: ViennaCode[], codes2: ViennaCode[]): number {
    let matchScore = 0;
    let maxPossible = Math.max(codes1.length, codes2.length);

    for (const c1 of codes1) {
      for (const c2 of codes2) {
        if (c1.category === c2.category) {
          matchScore += 0.3;
          if (c1.division === c2.division) {
            matchScore += 0.4;
            if (c1.section === c2.section) {
              matchScore += 0.3;
            }
          }
        }
      }
    }

    return Math.min(1, matchScore / maxPossible);
  }
}
