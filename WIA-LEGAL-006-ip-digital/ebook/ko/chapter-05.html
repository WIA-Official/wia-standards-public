Chapter 05: 디지털 저작권 보호 기술

05.1 디지털 저작권 보호의 개요

디지털 시대에 저작권 보호는 새로운 도전에 직면해 있습니다. 디지털 콘텐츠는 원본과 동일한 품질로 무한 복제가 가능하며, 인터넷을 통해 순식간에 전 세계로 배포될 수 있습니다. 이러한 환경에서 창작자의 권리를 보호하고 적정한 보상을 보장하기 위해서는 기술적 보호 조치(Technological Protection Measures, TPM)가 필수적입니다. WIA-LEGAL-006 표준은 디지털 저작권 보호를 위한 포괄적인 기술 프레임워크를 제시합니다.

디지털 저작권 관리(DRM, Digital Rights Management)는 디지털 콘텐츠의 불법 복제와 무단 사용을 방지하기 위한 기술적 해결책입니다. DRM 시스템은 암호화, 접근 제어, 사용 추적 등의 기술을 조합하여 저작권자가 자신의 콘텐츠가 어떻게 사용되는지 통제할 수 있게 합니다. 그러나 DRM은 사용자 편의성과 상호운용성을 저해할 수 있다는 비판도 받아왔습니다. 따라서 효과적인 DRM 시스템은 저작권 보호와 사용자 경험 사이의 균형을 찾아야 합니다.

디지털 워터마킹은 콘텐츠에 비가시적인 정보를 삽입하여 저작권을 표시하고 불법 배포 경로를 추적하는 기술입니다. 포렌식 워터마킹은 각 사용자에게 고유한 식별자를 삽입하여 유출 시 출처를 파악할 수 있게 합니다. 최근에는 NFT(Non-Fungible Token) 기술이 디지털 자산의 소유권과 진위성을 블록체인에 기록하는 새로운 방식으로 주목받고 있습니다. 이 장에서는 弘益人間의 정신에 따라, 이러한 기술들이 창작자와 소비자 모두에게 공정한 디지털 생태계를 구축하는 데 어떻게 기여할 수 있는지 탐구합니다.

05.2 DRM 기술의 이해

DRM(Digital Rights Management)은 디지털 콘텐츠의 접근과 사용을 제어하는 기술 및 정책의 집합입니다. DRM 시스템의 핵심 구성요소에는 콘텐츠 암호화, 라이선스 관리, 사용자 인증, 그리고 사용 정책 실행이 포함됩니다. 암호화된 콘텐츠는 유효한 라이선스를 보유한 인증된 사용자만 복호화하여 접근할 수 있습니다.

05.2.1 Common Encryption (CENC) 표준

CENC(Common Encryption)는 ISO/IEC 23001-7로 표준화된 암호화 방식으로, 하나의 암호화된 콘텐츠가 여러 DRM 시스템과 호환될 수 있게 합니다. 콘텐츠 제공자는 한 번만 암호화하고, 각 DRM 시스템에 맞는 라이선스만 별도로 발급하면 됩니다. 이는 콘텐츠 배포의 효율성을 크게 향상시키고, 다양한 디바이스와 플랫폼에서의 상호운용성을 보장합니다.

05.3 디지털 워터마킹 기술

디지털 워터마킹은 콘텐츠에 보이지 않는 정보를 삽입하여 저작권을 표시하거나 불법 배포 경로를 추적하는 기술입니다. 효과적인 워터마크는 비가시성(imperceptibility), 견고성(robustness), 용량(capacity)의 세 가지 특성을 균형 있게 갖추어야 합니다. 비가시성은 워터마크가 콘텐츠 품질에 영향을 주지 않아야 함을, 견고성은 다양한 공격에도 워터마크가 유지되어야 함을, 용량은 충분한 정보를 삽입할 수 있어야 함을 의미합니다.

05.3.1 포렌식 워터마킹 구현

포렌식 워터마킹 시스템 구현

다음은 비디오 스트리밍을 위한 포렌식 워터마킹 시스템의 TypeScript 구현입니다:

// WIA-LEGAL-006 포렌식 워터마킹 시스템

interface WatermarkPayload {
  userId: string;
  sessionId: string;
  timestamp: number;
  deviceFingerprint: string;
  contentId: string;
}

interface WatermarkConfig {
  strength: 'low' | 'medium' | 'high';
  embedMethod: 'spatial' | 'frequency' | 'hybrid';
  errorCorrection: 'none' | 'rs' | 'ldpc';
}

class ForensicWatermarkSystem {
  private secretKey: CryptoKey;
  private config: WatermarkConfig;

  constructor(config: WatermarkConfig) {
    this.config = config;
  }

  async initialize(keyMaterial: string): Promise {
    const encoder = new TextEncoder();
    const keyData = encoder.encode(keyMaterial);

    this.secretKey = await crypto.subtle.importKey(
      'raw',
      keyData,
      { name: 'HMAC', hash: 'SHA-256' },
      false,
      ['sign', 'verify']
    );
  }

  // 워터마크 페이로드 생성
  async createPayload(userId: string, contentId: string): Promise {
    const payload: WatermarkPayload = {
      userId,
      sessionId: this.generateSessionId(),
      timestamp: Date.now(),
      deviceFingerprint: await this.getDeviceFingerprint(),
      contentId
    };

    return payload;
  }

  // 페이로드를 바이너리로 인코딩
  encodePayload(payload: WatermarkPayload): Uint8Array {
    const json = JSON.stringify(payload);
    const encoder = new TextEncoder();
    const data = encoder.encode(json);

    // 에러 정정 코드 추가
    if (this.config.errorCorrection === 'rs') {
      return this.addReedSolomon(data);
    }

    return data;
  }

  // 프레임에 워터마크 삽입
  embedWatermark(
    frame: ImageData,
    payload: Uint8Array
  ): ImageData {
    const { width, height, data } = frame;
    const result = new Uint8ClampedArray(data);

    if (this.config.embedMethod === 'spatial') {
      return this.embedSpatial(result, width, height, payload);
    } else if (this.config.embedMethod === 'frequency') {
      return this.embedFrequency(result, width, height, payload);
    } else {
      return this.embedHybrid(result, width, height, payload);
    }
  }

  // 공간 영역 삽입 (LSB 변조)
  private embedSpatial(
    data: Uint8ClampedArray,
    width: number,
    height: number,
    payload: Uint8Array
  ): ImageData {
    const strength = this.getStrengthValue();
    let payloadIndex = 0;
    let bitIndex = 0;

    // 의사 난수 위치 선택
    const positions = this.generateEmbedPositions(
      width * height,
      payload.length * 8
    );

    for (const pos of positions) {
      if (payloadIndex >= payload.length) break;

      const pixelIndex = pos * 4; // RGBA
      const bit = (payload[payloadIndex] >> (7 - bitIndex)) & 1;

      // 청색 채널의 LSB 변조 (시각적 영향 최소화)
      data[pixelIndex + 2] = (data[pixelIndex + 2] & 0xFE) | bit;

      bitIndex++;
      if (bitIndex === 8) {
        bitIndex = 0;
        payloadIndex++;
      }
    }

    return new ImageData(data, width, height);
  }

  // 주파수 영역 삽입 (DCT 기반)
  private embedFrequency(
    data: Uint8ClampedArray,
    width: number,
    height: number,
    payload: Uint8Array
  ): ImageData {
    // 8x8 블록 단위 DCT 변환
    const blockSize = 8;
    const blocksX = Math.floor(width / blockSize);
    const blocksY = Math.floor(height / blockSize);

    let payloadBitIndex = 0;
    const totalBits = payload.length * 8;

    for (let by = 0; by < blocksY && payloadBitIndex < totalBits; by++) {
      for (let bx = 0; bx < blocksX && payloadBitIndex < totalBits; bx++) {
        const block = this.extractBlock(data, width, bx, by, blockSize);
        const dctBlock = this.dct2d(block);

        // 중간 주파수 계수 변조
        const bit = (payload[Math.floor(payloadBitIndex / 8)] >>
                    (7 - (payloadBitIndex % 8))) & 1;

        dctBlock[3][3] = this.quantizeCoefficient(dctBlock[3][3], bit);

        const idctBlock = this.idct2d(dctBlock);
        this.insertBlock(data, width, bx, by, blockSize, idctBlock);

        payloadBitIndex++;
      }
    }

    return new ImageData(data, width, height);
  }

  // 워터마크 추출
  async extractWatermark(frame: ImageData): Promise {
    const { width, height, data } = frame;

    let extractedBits: number[] = [];
    const positions = this.generateEmbedPositions(
      width * height,
      1024 * 8 // 최대 1KB 페이로드
    );

    for (const pos of positions) {
      const pixelIndex = pos * 4;
      const bit = data[pixelIndex + 2] & 1;
      extractedBits.push(bit);
    }

    const bytes = this.bitsToBytes(extractedBits);

    // 에러 정정
    const corrected = this.config.errorCorrection === 'rs'
      ? this.decodeReedSolomon(bytes)
      : bytes;

    try {
      const decoder = new TextDecoder();
      const json = decoder.decode(corrected);
      return JSON.parse(json) as WatermarkPayload;
    } catch {
      return null;
    }
  }

  // 워터마크 검증
  async verifyWatermark(
    payload: WatermarkPayload,
    signature: ArrayBuffer
  ): Promise {
    const encoder = new TextEncoder();
    const data = encoder.encode(JSON.stringify(payload));

    return crypto.subtle.verify(
      'HMAC',
      this.secretKey,
      signature,
      data
    );
  }

  // 유틸리티 메서드들
  private generateSessionId(): string {
    return crypto.randomUUID();
  }

  private async getDeviceFingerprint(): Promise {
    // 디바이스 특성 수집 및 해시
    const canvas = document.createElement('canvas');
    const gl = canvas.getContext('webgl');
    const renderer = gl?.getParameter(gl.RENDERER) || 'unknown';

    const fingerprint = `${navigator.userAgent}|${renderer}|${screen.width}x${screen.height}`;

    const encoder = new TextEncoder();
    const data = encoder.encode(fingerprint);
    const hash = await crypto.subtle.digest('SHA-256', data);

    return Array.from(new Uint8Array(hash))
      .map(b => b.toString(16).padStart(2, '0'))
      .join('');
  }

  private getStrengthValue(): number {
    return { low: 1, medium: 3, high: 5 }[this.config.strength];
  }

  private generateEmbedPositions(totalPixels: number, count: number): number[] {
    // PRNG 기반 위치 생성
    const positions: number[] = [];
    let seed = 12345; // 비밀 시드

    for (let i = 0; i < count; i++) {
      seed = (seed * 1103515245 + 12345) & 0x7fffffff;
      positions.push(seed % totalPixels);
    }

    return positions;
  }

  private addReedSolomon(data: Uint8Array): Uint8Array {
    // Reed-Solomon 에러 정정 코드 추가
    return data; // 실제 구현 필요
  }

  private decodeReedSolomon(data: Uint8Array): Uint8Array {
    return data; // 실제 구현 필요
  }

  private bitsToBytes(bits: number[]): Uint8Array {
    const bytes = new Uint8Array(Math.ceil(bits.length / 8));
    for (let i = 0; i < bits.length; i++) {
      if (bits[i]) {
        bytes[Math.floor(i / 8)] |= (1 << (7 - (i % 8)));
      }
    }
    return bytes;
  }

  private extractBlock(data: Uint8ClampedArray, width: number,
                       bx: number, by: number, size: number): number[][] {
    const block: number[][] = [];
    for (let y = 0; y < size; y++) {
      block[y] = [];
      for (let x = 0; x < size; x++) {
        const px = (by * size + y) * width + (bx * size + x);
        block[y][x] = data[px * 4]; // Red channel
      }
    }
    return block;
  }

  private insertBlock(data: Uint8ClampedArray, width: number,
                      bx: number, by: number, size: number, block: number[][]): void {
    for (let y = 0; y < size; y++) {
      for (let x = 0; x < size; x++) {
        const px = (by * size + y) * width + (bx * size + x);
        data[px * 4] = Math.max(0, Math.min(255, Math.round(block[y][x])));
      }
    }
  }

  private dct2d(block: number[][]): number[][] {
    // 2D DCT 변환
    return block; // 실제 구현 필요
  }

  private idct2d(block: number[][]): number[][] {
    // 2D IDCT 역변환
    return block; // 실제 구현 필요
  }

  private quantizeCoefficient(coeff: number, bit: number): number {
    const q = 10; // 양자화 스텝
    const quantized = Math.round(coeff / q) * q;
    return bit ? quantized + q/2 : quantized - q/2;
  }
}

  strength: 'high',
  embedMethod: 'hybrid',
  errorCorrection: 'rs'
});

await watermark.initialize('secret-key-for-watermarking');

05.4 NFT와 디지털 소유권

NFT(Non-Fungible Token)는 블록체인 기술을 활용하여 디지털 자산의 고유성과 소유권을 증명하는 새로운 방식입니다. 각 NFT는 고유한 식별자를 가지며, 블록체인에 기록된 메타데이터를 통해 창작자, 소유자, 거래 이력 등을 투명하게 추적할 수 있습니다. 이는 디지털 아트, 음악, 비디오 등 다양한 디지털 창작물에 적용되어 창작자 경제의 새로운 가능성을 열고 있습니다.

05.4.1 NFT 저작권 관리 스마트 계약

// WIA-LEGAL-006 NFT 저작권 관리 스마트 계약

import { ethers } from 'ethers';

interface IPRights {
  copyrightHolder: string;
  licenseType: 'exclusive' | 'non-exclusive' | 'creative-commons';
  commercialUse: boolean;
  derivativeWorks: boolean;
  attribution: boolean;
  royaltyPercentage: number;
  territories: string[];
  expirationDate: number | null;
}

interface NFTMetadata {
  name: string;
  description: string;
  image: string;
  contentHash: string;
  ipRights: IPRights;
  createdAt: number;
  creator: string;
}

class IPNFTManager {
  private contract: ethers.Contract;
  private signer: ethers.Signer;

  constructor(contractAddress: string, signer: ethers.Signer) {
    const abi = [
      'function mint(address to, string uri, bytes32 contentHash) returns (uint256)',
      'function setRoyalty(uint256 tokenId, address receiver, uint96 feeNumerator)',
      'function getRoyalty(uint256 tokenId, uint256 salePrice) view returns (address, uint256)',
      'function verifyOwnership(uint256 tokenId, address owner) view returns (bool)',
      'function transferWithRoyalty(uint256 tokenId, address from, address to, uint256 price)',
      'event IPRegistered(uint256 indexed tokenId, address creator, bytes32 contentHash)',
      'event RoyaltyPaid(uint256 indexed tokenId, address creator, uint256 amount)'
    ];

    this.contract = new ethers.Contract(contractAddress, abi, signer);
    this.signer = signer;
  }

  // IP 권리가 포함된 NFT 발행
  async mintIPNFT(
    to: string,
    metadata: NFTMetadata
  ): Promise<{ tokenId: bigint; txHash: string }> {
    // 메타데이터를 IPFS에 업로드
    const metadataUri = await this.uploadToIPFS(metadata);

    // 콘텐츠 해시 생성
    const contentHash = ethers.keccak256(
      ethers.toUtf8Bytes(metadata.contentHash)
    );

    // NFT 발행
    const tx = await this.contract.mint(to, metadataUri, contentHash);
    const receipt = await tx.wait();

    // 이벤트에서 토큰 ID 추출
    const event = receipt.logs.find(
      (log: any) => log.fragment?.name === 'IPRegistered'
    );
    const tokenId = event?.args?.[0] as bigint;

    // 로열티 설정
    if (metadata.ipRights.royaltyPercentage > 0) {
      await this.setRoyalty(
        tokenId,
        metadata.ipRights.copyrightHolder,
        metadata.ipRights.royaltyPercentage * 100 // 백분율을 basis points로
      );
    }

    return { tokenId, txHash: receipt.hash };
  }

  // 로열티 설정
  async setRoyalty(
    tokenId: bigint,
    receiver: string,
    basisPoints: number
  ): Promise {
    const tx = await this.contract.setRoyalty(tokenId, receiver, basisPoints);
    await tx.wait();
  }

  // 로열티 포함 전송
  async transferWithRoyalty(
    tokenId: bigint,
    from: string,
    to: string,
    salePrice: bigint
  ): Promise<{ txHash: string; royaltyPaid: bigint }> {
    // 로열티 금액 조회
    const [receiver, royaltyAmount] = await this.contract.getRoyalty(
      tokenId,
      salePrice
    );

    // 전송 실행 (로열티 자동 지불)
    const tx = await this.contract.transferWithRoyalty(
      tokenId,
      from,
      to,
      salePrice,
      { value: salePrice }
    );
    const receipt = await tx.wait();

    return {
      txHash: receipt.hash,
      royaltyPaid: royaltyAmount
    };
  }

  // 소유권 검증
  async verifyOwnership(tokenId: bigint, owner: string): Promise {
    return this.contract.verifyOwnership(tokenId, owner);
  }

  // 콘텐츠 진위 검증
  async verifyContent(
    tokenId: bigint,
    contentData: Uint8Array
  ): Promise {
    const contentHash = ethers.keccak256(contentData);
    const storedHash = await this.contract.getContentHash(tokenId);
    return contentHash === storedHash;
  }

  // IPFS 업로드 (시뮬레이션)
  private async uploadToIPFS(metadata: NFTMetadata): Promise {
    // 실제 구현에서는 IPFS/Pinata 등에 업로드
    const hash = ethers.keccak256(
      ethers.toUtf8Bytes(JSON.stringify(metadata))
    );
    return `ipfs://${hash.slice(2, 48)}`;
  }
}

// 라이선스 검증기
class LicenseVerifier {
  async verifyLicense(
    tokenId: bigint,
    intendedUse: {
      commercial: boolean;
      derivative: boolean;
      territory: string;
    },
    nftManager: IPNFTManager
  ): Promise<{
    valid: boolean;
    reason?: string;
  }> {
    // NFT 메타데이터 조회
    const metadata = await this.fetchMetadata(tokenId);
    const rights = metadata.ipRights;

    // 상업적 사용 확인
    if (intendedUse.commercial && !rights.commercialUse) {
      return { valid: false, reason: '상업적 사용이 허용되지 않음' };
    }

    // 2차 창작 확인
    if (intendedUse.derivative && !rights.derivativeWorks) {
      return { valid: false, reason: '2차 창작이 허용되지 않음' };
    }

    // 지역 제한 확인
    if (rights.territories.length > 0 &&
        !rights.territories.includes(intendedUse.territory)) {
      return { valid: false, reason: '해당 지역에서 사용이 제한됨' };
    }

    // 만료일 확인
    if (rights.expirationDate && Date.now() > rights.expirationDate) {
      return { valid: false, reason: '라이선스가 만료됨' };
    }

    return { valid: true };
  }

  private async fetchMetadata(tokenId: bigint): Promise {
    // IPFS에서 메타데이터 조회
    return {} as NFTMetadata;
  }
}

05.5 콘텐츠 식별 및 매칭 시스템

콘텐츠 식별 시스템은 디지털 콘텐츠의 고유한 특성을 추출하여 핑거프린트를 생성하고, 이를 데이터베이스와 비교하여 일치하는 콘텐츠를 찾아냅니다. 이 기술은 저작권 침해 탐지, 콘텐츠 추천, 중복 제거 등 다양한 용도로 활용됩니다. 대표적인 예로 YouTube의 Content ID, Shazam의 음악 인식, Google의 이미지 검색 등이 있습니다.
