06.1 특허 AI 시스템 개요

특허 검색과 분석은 지식재산 전략의 핵심입니다. 전 세계적으로 연간 300만 건 이상의 특허가 출원되며, 축적된 특허 문서는 1억 건을 초과합니다. 이러한 방대한 양의 특허 정보를 효율적으로 검색하고 분석하기 위해서는 인공지능 기술의 활용이 필수적입니다. WIA-LEGAL-006 표준은 특허 AI 시스템의 구현을 위한 포괄적인 프레임워크를 제공합니다.

전통적인 특허 검색은 키워드 기반으로 이루어졌으나, 기술 용어의 다양성, 다국어 문서, 복잡한 청구항 구조 등으로 인해 검색 정확도와 재현율에 한계가 있었습니다. AI 기반 특허 검색 시스템은 자연어 처리(NLP), 의미 검색(Semantic Search), 기계 학습을 활용하여 이러한 한계를 극복합니다. 특히 BERT, GPT와 같은 대규모 언어 모델의 발전으로 특허 문서의 의미적 이해와 분석이 가능해졌습니다.

특허 분석 AI는 선행기술 조사(Prior Art Search), 침해 분석(Infringement Analysis), 특허 가치 평가(Patent Valuation), 기술 트렌드 분석(Technology Landscape) 등 다양한 용도로 활용됩니다. 이 장에서는 弘益人間의 정신에 따라, AI 기술이 혁신을 촉진하고 공정한 특허 생태계를 구축하는 데 어떻게 기여할 수 있는지 탐구합니다.

06.3 특허 검색 시스템 구현

특허 시맨틱 검색 시스템

다음은 임베딩 기반 특허 검색 시스템의 TypeScript 구현입니다:

// WIA-LEGAL-006 특허 시맨틱 검색 시스템

interface PatentDocument {
  id: string;
  applicationNumber: string;
  title: string;
  abstract: string;
  claims: string[];
  description: string;
  ipcCodes: string[];
  applicant: string;
  inventors: string[];
  filingDate: Date;
  publicationDate: Date;
  priority: { country: string; number: string; date: Date }[];
  citations: string[];
}

interface SearchResult {
  patent: PatentDocument;
  score: number;
  matchedClaims: number[];
  highlightedAbstract: string;
}

interface PatentEmbedding {
  patentId: string;
  titleEmbedding: number[];
  abstractEmbedding: number[];
  claimsEmbedding: number[];
  compositeEmbedding: number[];
}

class PatentSearchEngine {
  private vectorStore: VectorStore;
  private embedder: PatentEmbedder;
  private reranker: CrossEncoderReranker;

  constructor(config: {
    vectorDbUrl: string;
    modelEndpoint: string;
    rerankerModel: string;
  }) {
    this.vectorStore = new VectorStore(config.vectorDbUrl);
    this.embedder = new PatentEmbedder(config.modelEndpoint);
    this.reranker = new CrossEncoderReranker(config.rerankerModel);
  }

  // 선행기술 검색
  async searchPriorArt(
    query: string | PatentDocument,
    options: SearchOptions = {}
  ): Promise {
    const {
      topK = 100,
      rerank = true,
      finalK = 20,
      filterIPC = [],
      dateRange = null
    } = options;

    // 쿼리 임베딩 생성
    const queryEmbedding = typeof query === 'string'
      ? await this.embedder.embedQuery(query)
      : await this.embedder.embedPatent(query);

    // 벡터 검색
    const candidates = await this.vectorStore.search({
      vector: queryEmbedding.compositeEmbedding,
      topK,
      filter: this.buildFilter(filterIPC, dateRange)
    });

    // 결과 재순위화
    let results = candidates;
    if (rerank) {
      const queryText = typeof query === 'string'
        ? query
        : `${query.title}. ${query.abstract}`;

      results = await this.reranker.rerank(
        queryText,
        candidates,
        finalK
      );
    }

    // 상세 정보 조회 및 하이라이트
    return Promise.all(
      results.slice(0, finalK).map(async (r) => ({
        patent: await this.fetchPatentDetails(r.patentId),
        score: r.score,
        matchedClaims: await this.findMatchedClaims(queryText, r.patentId),
        highlightedAbstract: await this.highlightMatches(r.patentId, queryText)
      }))
    );
  }

  // 침해 분석 검색
  async analyzeInfringement(
    targetProduct: ProductDescription,
    patentClaims: string[]
  ): Promise {
    const analysis: InfringementAnalysis = {
      overallRisk: 'low',
      claimAnalysis: [],
      recommendations: []
    };

    for (let i = 0; i < patentClaims.length; i++) {
      const claim = patentClaims[i];
      const elements = await this.extractClaimElements(claim);

      const elementMatches = await Promise.all(
        elements.map(async (element) => ({
          element,
          match: await this.matchProductFeature(element, targetProduct),
          confidence: 0
        }))
      );

      const matchRatio = elementMatches.filter(m => m.match.matched).length /
                         elements.length;

      analysis.claimAnalysis.push({
        claimNumber: i + 1,
        claimType: this.classifyClaimType(claim),
        elements,
        matchRatio,
        risk: matchRatio > 0.8 ? 'high' : matchRatio > 0.5 ? 'medium' : 'low',
        elementMatches
      });
    }

    // 전체 위험도 계산
    const highRiskClaims = analysis.claimAnalysis.filter(
      c => c.risk === 'high'
    ).length;

    analysis.overallRisk = highRiskClaims > 0 ? 'high' :
      analysis.claimAnalysis.some(c => c.risk === 'medium') ? 'medium' : 'low';

    return analysis;
  }

  // 기술 분류 예측
  async predictClassification(
    patent: Partial
  ): Promise {
    const text = `${patent.title} ${patent.abstract}`;
    const embedding = await this.embedder.embedQuery(text);

    const predictions = await this.classificationModel.predict(embedding);

    return predictions.map(p => ({
      code: p.code,
      description: this.getIPCDescription(p.code),
      confidence: p.confidence,
      level: this.getIPCLevel(p.code)
    })).sort((a, b) => b.confidence - a.confidence);
  }

  // 특허 가치 평가
  async evaluatePatentValue(
    patentId: string
  ): Promise {
    const patent = await this.fetchPatentDetails(patentId);
    const citations = await this.getCitationMetrics(patentId);
    const family = await this.getPatentFamily(patentId);
    const claims = await this.analyzeClaimStrength(patent.claims);

    const features = {
      forwardCitations: citations.forward,
      backwardCitations: citations.backward,
      familySize: family.length,
      claimCount: patent.claims.length,
      independentClaims: claims.independent,
      claimBreadth: claims.breadthScore,
      technologyAge: this.calculateAge(patent.filingDate),
      applicantStrength: await this.getApplicantScore(patent.applicant),
      ipcDiversity: new Set(patent.ipcCodes).size
    };

    const valuationScore = await this.valuationModel.predict(features);

    return {
      patentId,
      overallScore: valuationScore,
      tier: this.scoreToTier(valuationScore),
      factors: {
        citationImpact: citations.forward / (citations.backward + 1),
        familyBreadth: family.length,
        claimStrength: claims.strengthScore,
        technologyRelevance: await this.getTechRelevance(patent.ipcCodes)
      },
      estimatedValue: this.scoreToValue(valuationScore),
      recommendations: this.generateValuationRecommendations(features)
    };
  }

  private buildFilter(ipcCodes: string[], dateRange: DateRange | null) {
    const filter: any = {};
    if (ipcCodes.length > 0) {
      filter.ipcCodes = { $in: ipcCodes };
    }
    if (dateRange) {
      filter.filingDate = {
        $gte: dateRange.start,
        $lte: dateRange.end
      };
    }
    return filter;
  }

  private async extractClaimElements(claim: string): Promise {
    // NLP 기반 청구항 요소 추출
    const response = await this.nlpService.parse(claim, 'claim-elements');
    return response.elements;
  }

  private classifyClaimType(claim: string): 'independent' | 'dependent' {
    return /claim \d+|claims \d+/i.test(claim) ? 'dependent' : 'independent';
  }

  private scoreToTier(score: number): 'S' | 'A' | 'B' | 'C' | 'D' {
    if (score >= 0.9) return 'S';
    if (score >= 0.75) return 'A';
    if (score >= 0.5) return 'B';
    if (score >= 0.25) return 'C';
    return 'D';
  }
}

// 특허 임베딩 생성기
class PatentEmbedder {
  private modelEndpoint: string;
  private cache: Map = new Map();

  constructor(endpoint: string) {
    this.modelEndpoint = endpoint;
  }

  async embedPatent(patent: PatentDocument): Promise {
    if (this.cache.has(patent.id)) {
      return this.cache.get(patent.id)!;
    }

    const [titleEmb, abstractEmb, claimsEmb] = await Promise.all([
      this.embed(patent.title),
      this.embed(patent.abstract),
      this.embedClaims(patent.claims)
    ]);

    const composite = this.weightedAverage([
      { vector: titleEmb, weight: 0.2 },
      { vector: abstractEmb, weight: 0.3 },
      { vector: claimsEmb, weight: 0.5 }
    ]);

    const embedding: PatentEmbedding = {
      patentId: patent.id,
      titleEmbedding: titleEmb,
      abstractEmbedding: abstractEmb,
      claimsEmbedding: claimsEmb,
      compositeEmbedding: composite
    };

    this.cache.set(patent.id, embedding);
    return embedding;
  }

  async embedQuery(query: string): Promise {
    const emb = await this.embed(query);
    return {
      patentId: 'query',
      titleEmbedding: emb,
      abstractEmbedding: emb,
      claimsEmbedding: emb,
      compositeEmbedding: emb
    };
  }

  private async embed(text: string): Promise {
    const response = await fetch(this.modelEndpoint, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ text, model: 'patent-bert-v2' })
    });
    const result = await response.json();
    return result.embedding;
  }

  private async embedClaims(claims: string[]): Promise {
    const embeddings = await Promise.all(claims.map(c => this.embed(c)));
    return this.meanPooling(embeddings);
  }

  private weightedAverage(items: { vector: number[]; weight: number }[]): number[] {
    const dim = items[0].vector.length;
    const result = new Array(dim).fill(0);
    const totalWeight = items.reduce((sum, i) => sum + i.weight, 0);

    for (const { vector, weight } of items) {
      for (let i = 0; i < dim; i++) {
        result[i] += (vector[i] * weight) / totalWeight;
      }
    }
    return result;
  }

  private meanPooling(vectors: number[][]): number[] {
    const dim = vectors[0].length;
    const result = new Array(dim).fill(0);
    for (const v of vectors) {
      for (let i = 0; i < dim; i++) {
        result[i] += v[i] / vectors.length;
      }
    }
    return result;
  }
}

    ipcCode: string,

    // 임베딩 및 클러스터링

      patents.map(p => this.embedder.embedPatent(p))
    );

      domain: ipcCode,
      clusters,
      trends,
      keyPlayers,
      emergingAreas
    };

  async analyzeCitationNetwork(
    patentIds: string[]
  ): Promise {

    const edges: NetworkEdge[] = [];

    for (const id of patentIds) {
      const patent = await this.fetchPatentDetails(id);
      nodes.push({
        id,
        title: patent.title,
        year: patent.filingDate.getFullYear(),
        citations: patent.citations.length

      }
    }

    // 중심성 계산
    const centrality = this.calculateCentrality(nodes, edges);

    return { nodes, edges, centrality };
