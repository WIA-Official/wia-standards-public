Chapter 08: 지식재산 블록체인 등록 시스템

08.1 블록체인과 지식재산권

블록체인 기술은 지식재산권 관리에 혁명적인 변화를 가져오고 있습니다. 블록체인의 불변성(Immutability), 투명성(Transparency), 탈중앙화(Decentralization) 특성은 IP 권리의 등록, 증명, 거래에 있어 신뢰할 수 있는 인프라를 제공합니다. 특히 창작 시점 증명(Proof of Creation), 권리 이전 추적, 라이선스 관리, 로열티 자동 분배 등에서 블록체인의 장점이 두드러집니다.

전통적인 IP 등록 시스템은 중앙 집중화된 특허청, 저작권 등록 기관에 의존합니다. 이러한 시스템은 지역적 한계, 처리 시간, 비용 문제가 있습니다. 블록체인 기반 IP 등록은 글로벌하게 즉시 등록이 가능하고, 제3자 개입 없이도 권리를 증명할 수 있습니다. 물론 블록체인 등록이 법적 권리를 자동으로 부여하지는 않지만, 강력한 증거 수단으로 활용될 수 있습니다.

WIA-LEGAL-006 표준은 지식재산 블록체인 등록을 위한 기술적 프레임워크와 상호운용성 표준을 제시합니다. 弘益人間의 정신에 따라, 이 기술이 개인 창작자와 중소기업도 저렴하고 효과적으로 자신의 지식재산을 보호할 수 있도록 지원해야 합니다.

08.3 IP 블록체인 등록 시스템 구현

IP 블록체인 등록 스마트 계약

다음은 지식재산 등록 및 라이선스 관리를 위한 TypeScript/Solidity 구현입니다:

// WIA-LEGAL-006 IP 블록체인 등록 시스템

import { ethers } from 'ethers';
import * as crypto from 'crypto';

// 지식재산 등록 데이터 구조
interface IPRegistration {
  ipId: string;
  ipType: 'patent' | 'trademark' | 'copyright' | 'design' | 'trade_secret';
  title: string;
  description: string;
  contentHash: string;
  creator: string;
  creationDate: number;
  registrationDate: number;
  metadata: IPMetadata;
  licenses: License[];
}

interface IPMetadata {
  category: string;
  keywords: string[];
  jurisdiction: string;
  externalReferences: ExternalReference[];
  attachments: Attachment[];
}

interface ExternalReference {
  type: 'patent_number' | 'copyright_reg' | 'trademark_reg' | 'doi';
  value: string;
  authority: string;
}

interface License {
  licenseId: string;
  licensee: string;
  licenseType: 'exclusive' | 'non-exclusive' | 'sublicensable';
  scope: LicenseScope;
  royaltyBps: number; // Basis points (10000 = 100%)
  startDate: number;
  endDate: number | null;
  status: 'active' | 'expired' | 'terminated';
}

interface LicenseScope {
  territories: string[];
  fields: string[];
  channels: string[];
  commercialUse: boolean;
  derivativeWorks: boolean;
}

// IP 등록 및 관리 시스템
class IPBlockchainRegistry {
  private contract: ethers.Contract;
  private signer: ethers.Signer;
  private ipfsClient: IPFSClient;

  constructor(config: {
    contractAddress: string;
    signer: ethers.Signer;
    ipfsEndpoint: string;
  }) {
    const abi = [
      'function registerIP(bytes32 contentHash, string metadataUri, uint8 ipType) returns (uint256)',
      'function verifyOwnership(uint256 ipId, address owner) view returns (bool)',
      'function transferIP(uint256 ipId, address to)',
      'function grantLicense(uint256 ipId, address licensee, bytes calldata licenseData) returns (uint256)',
      'function revokeLicense(uint256 ipId, uint256 licenseId)',
      'function recordUsage(uint256 ipId, uint256 licenseId, uint256 amount)',
      'function claimRoyalties(uint256 ipId)',
      'function getIPDetails(uint256 ipId) view returns (tuple(bytes32,string,address,uint256,uint8))',
      'function getLicenses(uint256 ipId) view returns (tuple[])',
      'event IPRegistered(uint256 indexed ipId, address indexed creator, bytes32 contentHash)',
      'event LicenseGranted(uint256 indexed ipId, uint256 indexed licenseId, address licensee)',
      'event RoyaltyPaid(uint256 indexed ipId, address indexed recipient, uint256 amount)'
    ];

    this.contract = new ethers.Contract(config.contractAddress, abi, config.signer);
    this.signer = config.signer;
    this.ipfsClient = new IPFSClient(config.ipfsEndpoint);
  }

  // IP 등록
  async registerIP(registration: Omit): Promise<{
    ipId: string;
    txHash: string;
    contentHash: string;
    metadataUri: string;
  }> {
    // 콘텐츠 해시 생성
    const contentHash = this.generateContentHash(registration);

    // 메타데이터를 IPFS에 업로드
    const metadataUri = await this.ipfsClient.uploadJSON({
      ...registration,
      contentHash,
      registrationDate: Date.now()
    });

    // 블록체인에 등록
    const ipTypeCode = this.ipTypeToCode(registration.ipType);
    const tx = await this.contract.registerIP(
      contentHash,
      metadataUri,
      ipTypeCode
    );
    const receipt = await tx.wait();

    // 이벤트에서 IP ID 추출
    const event = receipt.logs.find(
      (log: any) => log.fragment?.name === 'IPRegistered'
    );
    const ipId = event?.args?.[0].toString();

    return {
      ipId,
      txHash: receipt.hash,
      contentHash,
      metadataUri
    };
  }

  // 라이선스 발급
  async grantLicense(
    ipId: string,
    license: Omit
  ): Promise<{
    licenseId: string;
    txHash: string;
  }> {
    // 소유권 확인
    const isOwner = await this.verifyOwnership(ipId);
    if (!isOwner) {
      throw new Error('권한이 없습니다. IP 소유자만 라이선스를 발급할 수 있습니다.');
    }

    // 라이선스 데이터 인코딩
    const licenseData = ethers.AbiCoder.defaultAbiCoder().encode(
      ['address', 'uint8', 'uint16', 'uint256', 'uint256', 'bytes'],
      [
        license.licensee,
        this.licenseTypeToCode(license.licenseType),
        license.royaltyBps,
        license.startDate,
        license.endDate || 0,
        ethers.toUtf8Bytes(JSON.stringify(license.scope))
      ]
    );

    const tx = await this.contract.grantLicense(ipId, license.licensee, licenseData);
    const receipt = await tx.wait();

    const event = receipt.logs.find(
      (log: any) => log.fragment?.name === 'LicenseGranted'
    );
    const licenseId = event?.args?.[1].toString();

    return { licenseId, txHash: receipt.hash };
  }

  // 사용량 기록 및 로열티 계산
  async recordUsageAndPay(
    ipId: string,
    licenseId: string,
    usageAmount: bigint,
    revenue: bigint
  ): Promise<{
    txHash: string;
    royaltyPaid: bigint;
  }> {
    // 라이선스 정보 조회
    const licenses = await this.contract.getLicenses(ipId);
    const license = licenses.find((l: any) => l.id.toString() === licenseId);

    if (!license) {
      throw new Error('유효하지 않은 라이선스입니다.');
    }

    // 로열티 계산
    const royaltyBps = license.royaltyBps;
    const royaltyAmount = (revenue * BigInt(royaltyBps)) / BigInt(10000);

    // 사용량 기록 및 로열티 지불
    const tx = await this.contract.recordUsage(ipId, licenseId, usageAmount, {
      value: royaltyAmount
    });
    const receipt = await tx.wait();

    return {
      txHash: receipt.hash,
      royaltyPaid: royaltyAmount
    };
  }

  // 소유권 검증
  async verifyOwnership(ipId: string): Promise {
    const ownerAddress = await this.signer.getAddress();
    return this.contract.verifyOwnership(ipId, ownerAddress);
  }

  // 창작 시점 증명 생성
  async createProofOfCreation(
    content: Buffer,
    creator: string,
    additionalData?: Record
  ): Promise {
    // 콘텐츠 해시
    const contentHash = crypto.createHash('sha256').update(content).digest('hex');

    // 타임스탬프 데이터
    const timestampData = {
      contentHash,
      creator,
      timestamp: Date.now(),
      ...additionalData
    };

    // IPFS에 원본 저장 (선택적)
    const contentCid = await this.ipfsClient.uploadBuffer(content);
    const metadataCid = await this.ipfsClient.uploadJSON(timestampData);

    // 블록체인에 해시 앵커링
    const tx = await this.contract.registerIP(
      '0x' + contentHash,
      `ipfs://${metadataCid}`,
      0 // generic proof
    );
    const receipt = await tx.wait();

    return {
      proofId: receipt.logs[0].args[0].toString(),
      contentHash,
      contentCid,
      metadataCid,
      blockNumber: receipt.blockNumber,
      blockTimestamp: (await receipt.getBlock()).timestamp,
      transactionHash: receipt.hash
    };
  }

  // IP 이력 조회
  async getIPHistory(ipId: string): Promise {
    const filter = this.contract.filters.IPRegistered(ipId);
    const events = await this.contract.queryFilter(filter);

    return events.map(event => ({
      eventType: event.fragment?.name || '',
      blockNumber: event.blockNumber,
      transactionHash: event.transactionHash,
      args: event.args
    }));
  }

  // 콘텐츠 해시 생성
  private generateContentHash(registration: any): string {
    const data = JSON.stringify({
      title: registration.title,
      description: registration.description,
      creator: registration.creator,
      creationDate: registration.creationDate,
      metadata: registration.metadata
    });

    return '0x' + crypto.createHash('sha256').update(data).digest('hex');
  }

  private ipTypeToCode(type: string): number {
    const codes: Record = {
      patent: 1, trademark: 2, copyright: 3, design: 4, trade_secret: 5
    };
    return codes[type] || 0;
  }

  private licenseTypeToCode(type: string): number {
    const codes: Record = {
      exclusive: 1, 'non-exclusive': 2, sublicensable: 3
    };
    return codes[type] || 0;
  }
}

// IPFS 클라이언트
class IPFSClient {
  private endpoint: string;

  constructor(endpoint: string) {
    this.endpoint = endpoint;
  }

  async uploadJSON(data: any): Promise {
    const response = await fetch(`${this.endpoint}/api/v0/add`, {
      method: 'POST',
      body: JSON.stringify(data)
    });
    const result = await response.json();
    return result.Hash;
  }

  async uploadBuffer(buffer: Buffer): Promise {
    const formData = new FormData();
    formData.append('file', new Blob([buffer]));

    const response = await fetch(`${this.endpoint}/api/v0/add`, {
      method: 'POST',
      body: formData
    });
    const result = await response.json();
    return result.Hash;
  }
}

interface ProofOfCreation {
  proofId: string;
  contentHash: string;
  contentCid: string;
  metadataCid: string;
  blockNumber: number;
  blockTimestamp: number;
  transactionHash: string;
}

interface IPHistoryEvent {
  eventType: string;
  blockNumber: number;
  transactionHash: string;
  args: any;
}

즉시 분배 거래 시 실시간 분배 투명성, 즉시성 높은 가스비 NFT 마켓플레이스

클레임 기반 수익자가 청구 시 지급 유연성 미청구 위험 대규모 분배
스플리터 계약 사전 정의된 비율로 분할 자동화, 정확성 변경 어려움 공동 저작물 
// 로열티 스플리터 구현

class RoyaltySplitter {
  private shares: Map = new Map();

  constructor(beneficiaries: { address: string; share: number }[]) {
    const totalShare = beneficiaries.reduce((sum, b) => sum + b.share, 0);
    if (totalShare !== 10000) {
      throw new Error('총 지분은 10000 (100%)이어야 합니다');
    }

    for (const b of beneficiaries) {
      this.shares.set(b.address, b.share);
    }
  }

  // 로열티 분배 계산
  calculateDistribution(totalAmount: bigint): Map {
    const distribution = new Map();

    for (const [address, share] of this.shares) {
      const amount = (totalAmount * BigInt(share)) / BigInt(10000);
      distribution.set(address, amount);
    }

    return distribution;
  }

  // 분배 실행 (온체인)
  async executeDistribution(
    contract: ethers.Contract,
    totalAmount: bigint
  ): Promise {
    const distribution = this.calculateDistribution(totalAmount);
    const addresses = Array.from(distribution.keys());
    const amounts = Array.from(distribution.values());

    const tx = await contract.distributeRoyalties(addresses, amounts, {
      value: totalAmount
    });
    return tx.hash;
  }
}
