2.1 Status Types and Definitions
--------------------------------------------------------------------------------

CRYO-LEGAL defines six distinct legal status categories, each with specific
legal implications and requirements.

[ID] Status Type Definitions:

    LegalStatusTypes {
      LIVING: {
        code:        "LIV"
        description: "Standard legal status of a natural person"
        capacity:    "Full legal capacity"
        rights:      "All civil, political, and property rights"
        obligations: "All legal obligations and duties"

        characteristics: {
          canContract:     true
          canVote:         true
          canOwnProperty:  true
          canMarry:        true
          taxLiability:    true
          criminalCapacity: true
        }
      }

      LEGALLY_DECEASED: {
        code:        "DEC"
        description: "Death has been legally declared"
        capacity:    "None (standard death)"
        duration:    "Typically brief - until preservation status"

        consequences: {
          estateOpens:        true
          marriageTerminated: true
          contractsTerminated: "Per terms"
          votingRights:       false
        }

        note: "This is a transitional state. For cryopreserved
               individuals, it leads to PRESERVED_STATUS rather
               than permanent estate distribution."
      }

      PRESERVED_STATUS: {
        code:        "PRE"
        description: "Special status during cryopreservation"
        capacity:    "Suspended, exercised through guardian"
        duration:    "Duration of preservation"

        characteristics: {
          personalRights:    "Suspended"
          propertyRights:    "Held in trust or by guardian"
          privacy:           "Protected"
          dignityProtection: "Guaranteed"
          revivalRight:      "Fundamental right"
        }

        protections: [
          "Cannot be declared permanently dead without cause",
          "Body integrity protected",
          "Advance directives honored",
          "Guardian oversight required"
        ]
      }

      REVIVAL_PENDING: {
        code:        "REV"
        description: "Revival procedure has been initiated"
        capacity:    "Transitioning to full capacity"
        duration:    "Duration of revival and restoration process"

        characteristics: {
          medicalStatus:      "Under active revival"
          legalStatus:        "Restoration proceeding initiated"
          guardianAuthority:  "Continues until restoration"
          anticipatoryRights: "Preparing for full restoration"
        }
      }

      LEGALLY_RESTORED: {
        code:        "RES"
        description: "Full legal personhood restored by court order"
        capacity:    "Full legal capacity"
        rights:      "All rights restored"

        effects: {
          deathCertificate:   "Superseded"
          newIdentityDocs:    "Issued"
          marriageStatus:     "Per court order"
          propertyRights:     "Restored"
          criminalRecord:     "Reviewed and updated"
        }
      }

      PERMANENT_DEATH: {
        code:        "PER"
        description: "Final, irreversible death"
        capacity:    "None"
        finality:    "Absolute"

        triggers: [
          "Revival deemed impossible",
          "Preservation facility failure",
          "Explicit termination request honored",
          "Court determination of permanent death"
        ]

        consequences: {
          estateFinalizes:    true
          guardianshipEnds:   true
          allRightsTerminate: true
        }
      }
    }

2.2 State Transition Model
--------------------------------------------------------------------------------

Legal status transitions must follow defined pathways with proper authorization.

[ID] State Transition Diagram:

    LIVING ---------> LEGALLY_DECEASED
                            |
                            v
                      PRESERVED_STATUS ---------> PERMANENT_DEATH
                            |
                            v
                      REVIVAL_PENDING -----------> PERMANENT_DEATH
                       /    |
                      /     v
        (back to)    /  LEGALLY_RESTORED
    PRESERVED_STATUS

[ID] Valid Transitions:

    TransitionRules {
      fromLiving: {
        validTargets: ["LEGALLY_DECEASED"]
        requirements: {
          deathCertificate: "Required"
          medicalConfirmation: "Required"
          authority: "Medical professional + registrar"
        }
      }

      fromLegallyDeceased: {
        validTargets: ["PRESERVED_STATUS"]
        requirements: {
          preservationInitiated: "Required"
          preservationDeclaration: "Required"
          courtOrder: "Required or automatic per jurisdiction"
          authority: "Court or designated agency"
        }
      }

      fromPreservedStatus: {
        validTargets: ["REVIVAL_PENDING", "PERMANENT_DEATH"]

        toRevivalPending: {
          requirements: {
            revivalInitiation: "Medical confirmation"
            courtNotification: "Required"
            guardianConsent: "Required"
          }
        }

        toPermanentDeath: {
          requirements: {
            cause: "Preservation failure OR termination request"
            courtOrder: "Required"
            medicalCertification: "Required"
          }
        }
      }

      fromRevivalPending: {
        validTargets: ["LEGALLY_RESTORED", "PRESERVED_STATUS", "PERMANENT_DEATH"]

        toLegallyRestored: {
          requirements: {
            revivalSuccess: "Medical confirmation"
            identityVerification: "CRYO-IDENTITY confirmation"
            courtRestoration: "Court order required"
          }
        }

        toPreservedStatus: {
          requirements: {
            revivalPaused: "Medical determination"
            courtNotification: "Required"
          }
        }

        toPermanentDeath: {
          requirements: {
            revivalFailed: "Medical determination"
            courtOrder: "Required"
          }
        }
      }

      fromLegallyRestored: {
        validTargets: []
        note: "Terminal state for successful revival"
      }

      fromPermanentDeath: {
        validTargets: []
        note: "Terminal state - absolutely irreversible"
      }
    }

[!] Important: Status transitions are audited and immutable once recorded.
    Any transition requires proper authorization and documentation.

2.3 Legal Rights by Status
--------------------------------------------------------------------------------

Each status category carries specific rights and limitations.

[ID] Rights Matrix:

    +------------------------------------------------------------------+
    | Right Category      | LIV | DEC | PRE | REV | RES | PER         |
    +------------------------------------------------------------------+
    | Personal Liberty    | Yes | No  | Sus | Sus | Yes | No          |
    | Property Ownership  | Yes | Est | Tst | Tst | Yes | Est         |
    | Contract Capacity   | Yes | No  | Grd | Grd | Yes | No          |
    | Voting Rights       | Yes | No  | No  | No  | Yes | No          |
    | Marriage Status     | Yes | End | Sus | Sus | Rev | End         |
    | Parental Rights     | Yes | End | Sus | Sus | Rev | End         |
    | Privacy Protection  | Yes | Ltd | Yes | Yes | Yes | No          |
    | Dignity Protection  | Yes | Yes | Yes | Yes | Yes | Yes         |
    | Body Autonomy       | Yes | Ltd | Grd | Med | Yes | No          |
    | Revival Right       | N/A | N/A | Yes | Yes | N/A | N/A         |
    +------------------------------------------------------------------+

    Legend:
      Yes = Full right
      No  = No right
      Sus = Suspended (resumable)
      Tst = Held in trust
      Grd = Exercised through guardian
      Est = Estate rules apply
      Ltd = Limited protection
      Rev = Reviewed/restored by court
      Med = Medical authority

[ID] Detailed Rights by Status:

    PreservedStatusRights {

          description: "Right to attempt revival when possible"
          protection:  "Cannot be waived by others"

          description: "Treatment with respect and dignity"
          protection:  "Applies throughout preservation"
        }

        privacy: {
          description: "Protection of personal information"
          protection:  "Limited disclosure rules"
        }
      }

      property: {
        ownership: "Held in trust or by guardian"
        management: "Guardian authority with oversight"
        disposition: "Frozen until restoration or death"
      }

      representation: {
        guardian: "Court-appointed representative"
        advocate: "Right to legal advocacy"
        interests: "Best interests standard applies"
      }
    }

2.4 Documentation Requirements
--------------------------------------------------------------------------------

Each status requires specific documentation.

[ID] Required Documents by Status:

    DocumentationRequirements {
      LIVING: {
        standard: ["Government-issued ID", "Birth certificate"]
        cryoRelated: ["Advance directive (optional)", "Consent documents"]
      }

      LEGALLY_DECEASED: {
        required: [
          "Death certificate",
          "Medical certification of death",
          "Identification of deceased"
        ]
        timeline: "Immediate upon death"
      }

      PRESERVED_STATUS: {
        required: [
          "Death certificate (original)",
          "Preservation declaration",
          "Court order for preservation status",
          "Facility custody documentation",
          "Guardian appointment order"
        ]
        timeline: "Within 30 days of preservation initiation"
      }

      REVIVAL_PENDING: {
        required: [
          "Revival initiation certificate",
          "Medical status reports",
          "Court notification",
          "Guardian acknowledgment"
        ]
        timeline: "Upon revival initiation"
      }

      LEGALLY_RESTORED: {
        required: [
          "Court restoration order",
          "New identity documents",
          "Medical clearance",
          "Identity verification certificate"
        ]
        timeline: "Upon successful restoration"
      }

      PERMANENT_DEATH: {
        required: [
          "Permanent death declaration",
          "Court order (if applicable)",
          "Medical certification",
          "Estate finalization documents"
        ]
        timeline: "Upon determination"
      }
    }

Chapter Summary
--------------------------------------------------------------------------------

Key Concepts:
  [OK] Six legal status categories: LIVING, DECEASED, PRESERVED, REVIVAL,
       RESTORED, PERMANENT_DEATH
  [OK] Status transitions follow defined pathways with authorization
  [OK] Each status has specific rights and limitations
  [OK] Rights may be suspended but not extinguished during preservation
  [OK] Comprehensive documentation required for each status
  [OK] Document retention extends well beyond individual status changes
