Asteroid Resource Catalog Schema

Standard format for asteroid characterization and mining data

{
  "asteroid": {
    "designation": "string (e.g., 101955 Bennu)",
    "type": "C-Type | S-Type | M-Type | X-Type",
    "diameter": "number (meters)",
    "mass": "number (kg)",
    "orbit": {
      "semiMajorAxis": "number (AU)",
      "eccentricity": "number (0-1)",
      "inclination": "number (degrees)",
      "perihelion": "number (AU)"
    },
    "deltaV": "number (km/s, Earth-asteroid-Earth)"
  },
  "composition": {
    "water": "number (mass %)",
    "silicates": "number (mass %)",
    "metals": {
      "iron": "number (mass %)",
      "nickel": "number (mass %)",
      "platinum": "number (ppm)",
      "palladium": "number (ppm)",
      "rhodium": "number (ppm)"
    },
    "organics": "number (mass %)"
  },
  "resources": {
    "estimatedTotalMass": "number (tonnes)",
    "waterMass": "number (tonnes)",
    "metalMass": "number (tonnes)",
    "economicValue": "number (USD)"
  },
  "mission": {
    "status": "Surveyed | Planned | Active | Depleted",
    "operator": "string",
    "extractionStartDate": "ISO8601 (optional)",
    "miningMethod": "Surface | Subsurface | Capture"
  }
}
