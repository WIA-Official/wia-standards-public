4.1 Embodiment State Schema

The core data structure that represents the complete state of an embodied AI system:


{
  "$schema": "https://wiastandards.com/schemas/ai-embodiment/v1.0",
  "type": "object",
  "title": "WIA-AI-EMBODIMENT-STATE",
  "version": "1.0.0",
  "required": ["id", "timestamp", "ai_core", "body_state", "safety"],
  "properties": {
    "id": {
      "type": "string",
      "format": "uuid",
      "description": "Unique identifier for this embodied AI instance"
    },
    "timestamp": {
      "type": "string",
      "format": "date-time",
      "description": "ISO 8601 timestamp"
    },
    "ai_core": {
      "type": "object",
      "properties": {
        "model_id": { "type": "string" },
        "consciousness_level": { "type": "integer", "minimum": 0, "maximum": 5 },
        "active_goals": { "type": "array", "items": { "type": "string" } },
        "emotional_state": { "$ref": "#/definitions/EmotionalState" }
      }
    },
    "body_state": {
      "type": "object",
      "properties": {
        "position": { "$ref": "#/definitions/Position3D" },
        "orientation": { "$ref": "#/definitions/Quaternion" },
        "velocity": { "$ref": "#/definitions/Vector3D" },
        "joint_angles": { "type": "array", "items": { "type": "number" } },
        "power_level": { "type": "number", "minimum": 0, "maximum": 100 }
      }
    },
    "safety": {
      "type": "object",
      "properties": {
        "status": { "enum": ["SAFE", "CAUTION", "WARNING", "DANGER", "EMERGENCY"] },
        "active_constraints": { "type": "array", "items": { "type": "string" } },
        "humans_detected": { "type": "integer" },
        "closest_human_distance": { "type": "number" }
      }
    }
  }
}

      "duration_ms": 500
    },
    {
      "motor_id": "joint_elbow_right",
      "command_type": "VELOCITY",
      "target_value": 10.0,
      "unit": "degrees_per_second",
      "force_limit_n": 30.0
    }
  ],
  "safety_checks": {
    "collision_check": true,
    "force_limit_active": true,
    "human_proximity_check": true
  },
  "fallback_on_error": "HOLD_POSITION"
}

Motor Command Types

Type Description Use Case
POSITION Move to absolute position Reaching, pointing
VELOCITY Move at specified speed Continuous motion
FORCE Apply specified force Gripping, pushing
STOP Immediate halt Emergency, collision
TRAJECTORY Follow path Complex movements

      "type": "object",
