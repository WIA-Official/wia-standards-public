1. 클라우드 데이터 통합 개요

클라우드 데이터 통합은 클라우드 환경에서 다양한 데이터 소스를 연결하고 처리하는 프로세스입니다. 온프레미스 솔루션 대비 확장성, 유연성, 비용 효율성을 제공합니다.

클라우드 데이터 통합의 이점

  • 탄력적 확장성: 수요에 따라 자동으로 리소스 확장/축소
  • 종량제 과금: 사용한 만큼만 비용 지불
  • 관리 부담 감소: 인프라 관리를 클라우드 제공자에게 위임
  • 글로벌 접근성: 전 세계 어디서나 데이터 접근 가능
  • 최신 기술 활용: AI/ML, 서버리스 등 최신 기술 즉시 적용
  • 빠른 구축: 몇 분 안에 리소스 프로비저닝

주요 클라우드 제공자 비교

2. AWS 데이터 통합 서비스

AWS Glue - 서버리스 ETL

AWS Glue는 완전 관리형 ETL 서비스로, 데이터 검색, 카탈로그, 변환, 이동을 자동화합니다.

import sys
from awsglue.transforms import *
from awsglue.utils import getResolvedOptions
from pyspark.context import SparkContext
from awsglue.context import GlueContext
from awsglue.job import Job

# Job 매개변수 가져오기
args = getResolvedOptions(sys.argv, ['JOB_NAME'])

# Glue Context 초기화
sc = SparkContext()
glueContext = GlueContext(sc)
spark = glueContext.spark_session
job = Job(glueContext)
job.init(args['JOB_NAME'], args)

# S3에서 데이터 읽기 (Glue Data Catalog 사용)
datasource = glueContext.create_dynamic_frame.from_catalog(
    database="sales_db",
    table_name="orders"
)

# 데이터 변환
# 1. 불필요한 컬럼 제거
transformed = DropFields.apply(
    frame=datasource,
    paths=["internal_id", "temp_field"]
)

# 2. 데이터 타입 변환 및 정제
from pyspark.sql.functions import col, to_timestamp, when

df = transformed.toDF()
df_cleaned = df \
    .withColumn("order_date", to_timestamp(col("order_date"), "yyyy-MM-dd")) \
    .withColumn("amount", col("amount").cast("decimal(10,2)")) \
    .withColumn("status", when(col("status").isNull(), "PENDING").otherwise(col("status"))) \
    .filter(col("amount") > 0)  # 유효한 금액만

# 3. 집계 및 변환
from pyspark.sql.functions import sum, count, avg
daily_summary = df_cleaned.groupBy("order_date", "customer_id").agg(
    count("order_id").alias("order_count"),
    sum("amount").alias("total_amount"),
    avg("amount").alias("avg_amount")
)

# DynamicFrame으로 변환
output_frame = DynamicFrame.fromDF(daily_summary, glueContext, "output")

# Redshift에 저장
glueContext.write_dynamic_frame.from_jdbc_conf(
    frame=output_frame,
    catalog_connection="redshift-connection",
    connection_options={
        "dbtable": "daily_order_summary",
        "database": "analytics"
    },
    redshift_tmp_dir="s3://my-bucket/temp/"
)

job.commit()

-- Synapse SQL Pool에서 외부 테이블 생성
CREATE EXTERNAL DATA SOURCE AzureDataLakeStore
WITH (
    TYPE = HADOOP,
    LOCATION = 'abfss://container@storageaccount.dfs.core.windows.net',
    CREDENTIAL = AzureStorageCredential
);

CREATE EXTERNAL FILE FORMAT ParquetFormat
WITH (
    FORMAT_TYPE = PARQUET,
    DATA_COMPRESSION = 'org.apache.hadoop.io.compress.SnappyCodec'
);

-- 외부 테이블 정의
CREATE EXTERNAL TABLE dbo.ExternalOrders (
    OrderID INT,
    CustomerID INT,
    OrderDate DATE,
    TotalAmount DECIMAL(10,2),
    Status VARCHAR(50)
)
WITH (
    LOCATION = '/data/orders/',
    DATA_SOURCE = AzureDataLakeStore,
    FILE_FORMAT = ParquetFormat
);

-- CTAS (Create Table As Select)로 최적화된 테이블 생성
CREATE TABLE dbo.Orders
WITH (
    DISTRIBUTION = HASH(CustomerID),
    CLUSTERED COLUMNSTORE INDEX
)
AS
SELECT
    OrderID,
    CustomerID,
    OrderDate,
    TotalAmount,
    Status,
    YEAR(OrderDate) AS OrderYear,
    MONTH(OrderDate) AS OrderMonth
FROM dbo.ExternalOrders
WHERE OrderDate >= '2024-01-01';

-- 통계 생성 (성능 최적화)
CREATE STATISTICS stat_CustomerID ON dbo.Orders(CustomerID);
CREATE STATISTICS stat_OrderDate ON dbo.Orders(OrderDate);

Google Cloud Dataflow - Apache Beam 기반 처리

import apache_beam as beam
from apache_beam.options.pipeline_options import PipelineOptions
from apache_beam.io.gcp.bigquery import WriteToBigQuery
from apache_beam.transforms.window import FixedWindows
import json

class ParseJson(beam.DoFn):
    def process(self, element):
        record = json.loads(element)
        yield {
            'user_id': record.get('user_id'),
            'event_type': record.get('event_type'),
            'timestamp': record.get('timestamp'),
            'amount': float(record.get('amount', 0))
        }

class EnrichData(beam.DoFn):
    def process(self, element):
        # 외부 API나 BigQuery에서 추가 데이터 가져오기
        element['enriched_at'] = beam.pvalue.Timestamp.now()
        element['processed'] = True
        yield element

# Pipeline 옵션 설정
options = PipelineOptions(
    project='my-project',
    region='us-central1',
    runner='DataflowRunner',
    temp_location='gs://my-bucket/temp',
    staging_location='gs://my-bucket/staging',
    num_workers=4,
    max_num_workers=10,
    autoscaling_algorithm='THROUGHPUT_BASED'
)

# Pipeline 정의
with beam.Pipeline(options=options) as pipeline:
    (pipeline
     | 'ReadFromPubSub' >> beam.io.ReadFromPubSub(
         subscription='projects/my-project/subscriptions/events-sub')
     | 'ParseJSON' >> beam.ParDo(ParseJson())
     | 'FilterValidRecords' >> beam.Filter(
         lambda x: x['amount'] > 0 and x['user_id'] is not None)
     | 'WindowInto5Min' >> beam.WindowInto(FixedWindows(5 * 60))
     | 'EnrichData' >> beam.ParDo(EnrichData())
     | 'WriteToBigQuery' >> WriteToBigQuery(
         'my-project:dataset.events_table',
         schema='user_id:INTEGER,event_type:STRING,timestamp:TIMESTAMP,amount:FLOAT,enriched_at:TIMESTAMP,processed:BOOLEAN',
         create_disposition=beam.io.BigQueryDisposition.CREATE_IF_NEEDED,
         write_disposition=beam.io.BigQueryDisposition.WRITE_APPEND
     ))

멀티클라우드 데이터 통합 예제

# Airflow DAG로 멀티클라우드 오케스트레이션
from airflow import DAG
from airflow.providers.amazon.aws.operators.s3 import S3CopyObjectOperator
from airflow.providers.google.cloud.transfers.s3_to_gcs import S3ToGCSOperator
from airflow.providers.google.cloud.operators.bigquery import BigQueryInsertJobOperator
from airflow.providers.microsoft.azure.operators.data_factory import AzureDataFactoryRunPipelineOperator
from datetime import datetime, timedelta

default_args = {
    'owner': 'data-team',
    'depends_on_past': False,
    'start_date': datetime(2025, 1, 1),
    'email_on_failure': True,
    'email_on_retry': False,
    'retries': 3,
    'retry_delay': timedelta(minutes=5)
}

dag = DAG(
    'multicloud_data_sync',
    default_args=default_args,
    description='Sync data across AWS, GCP, and Azure',
    schedule_interval='0 2 * * *',  # 매일 새벽 2시
    catchup=False
)

# Step 1: AWS S3에서 GCS로 데이터 복사
sync_s3_to_gcs = S3ToGCSOperator(
    task_id='sync_s3_to_gcs',
    bucket='aws-source-bucket',
    prefix='data/sales/',
    dest_gcs='gs://gcp-dest-bucket/sales/',
    replace=False,
    gcp_conn_id='google_cloud_default',
    aws_conn_id='aws_default',
    dag=dag
)

# Step 2: BigQuery로 데이터 로드
load_to_bigquery = BigQueryInsertJobOperator(
    task_id='load_to_bigquery',
    configuration={
        "load": {
            "sourceUris": ["gs://gcp-dest-bucket/sales/*.parquet"],
            "destinationTable": {
                "projectId": "my-project",
                "datasetId": "analytics",
                "tableId": "sales_data"
            },
            "sourceFormat": "PARQUET",
            "writeDisposition": "WRITE_APPEND",
            "autodetect": True
        }
    },
    gcp_conn_id='google_cloud_default',
    dag=dag
)

# Step 3: Azure Data Factory 파이프라인 실행
run_azure_pipeline = AzureDataFactoryRunPipelineOperator(
    task_id='run_azure_transform',
    pipeline_name='transform-sales-data',
    resource_group_name='my-resource-group',
    factory_name='my-data-factory',
    azure_data_factory_conn_id='azure_data_factory_default',
    dag=dag
)

# Task 의존성 정의
sync_s3_to_gcs >> load_to_bigquery >> run_azure_pipeline

데이터 암호화 구현 예제

# AWS KMS를 사용한 데이터 암호화
import boto3
import json

kms_client = boto3.client('kms', region_name='us-east-1')
s3_client = boto3.client('s3')

# 1. KMS 키 생성
key_response = kms_client.create_key(
    Description='Data encryption key for sales data',
    KeyUsage='ENCRYPT_DECRYPT',
    Origin='AWS_KMS'
)
key_id = key_response['KeyMetadata']['KeyId']

# 2. S3 버킷에 기본 암호화 설정
s3_client.put_bucket_encryption(
    Bucket='my-sensitive-data-bucket',
    ServerSideEncryptionConfiguration={
        'Rules': [{
            'ApplyServerSideEncryptionByDefault': {
                'SSEAlgorithm': 'aws:kms',
                'KMSMasterKeyID': key_id
            },
            'BucketKeyEnabled': True
        }]
    }
)

# 3. 데이터 업로드 시 암호화
sensitive_data = {'customer_id': 12345, 'ssn': '123-45-6789'}

s3_client.put_object(
    Bucket='my-sensitive-data-bucket',
    Key='customers/sensitive/customer_12345.json',
    Body=json.dumps(sensitive_data),
    ServerSideEncryption='aws:kms',
    SSEKMSKeyId=key_id,
    Metadata={'classification': 'confidential'}
)

print(f"Data encrypted and stored with KMS key: {key_id}")

비용 최적화 전략

전략 설명 예상 절감 구현 난이도
Right-Sizing 실제 사용량에 맞는 리소스 크기 조정 20-30% 낮음
예약 인스턴스 장기 계약으로 할인 (1-3년) 40-60% 중간
스팟 인스턴스 유휴 용량 활용 (중단 가능) 70-90% 높음

쿼리 최적화 스캔하는 데이터 양 최소화 40-60% 중간

AWS 비용 최적화 예제

# Lambda 함수로 S3 객체 라이프사이클 자동화
import boto3
from datetime import datetime, timedelta

s3 = boto3.client('s3')

def lambda_handler(event, context):
    bucket_name = 'my-data-bucket'

    # 라이프사이클 정책 설정
    lifecycle_config = {
        'Rules': [
            {
                'Id': 'Move to IA after 30 days',
                'Status': 'Enabled',
                'Transitions': [
                    {
                        'Days': 30,
                        'StorageClass': 'STANDARD_IA'
                    },
                    {
                        'Days': 90,
                        'StorageClass': 'GLACIER'
                    },
                    {
                        'Days': 365,
                        'StorageClass': 'DEEP_ARCHIVE'
                    }
                ],
                'Expiration': {
                    'Days': 2555  # 7년 후 삭제
                },
                'Filter': {
                    'Prefix': 'logs/'
                }
            },
            {
                'Id': 'Delete incomplete multipart uploads',
                'Status': 'Enabled',
                'AbortIncompleteMultipartUpload': {
                    'DaysAfterInitiation': 7
                },
                'Filter': {}
            }
        ]
    }

    s3.put_bucket_lifecycle_configuration(
        Bucket=bucket_name,
        LifecycleConfiguration=lifecycle_config
    )

    return {'statusCode': 200, 'body': 'Lifecycle policy updated'}
