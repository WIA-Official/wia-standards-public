from ai_pipeline_generator import PipelineGenerator
import openai

# OpenAI GPT를 사용한 파이프라인 생성
class AIDataPipelineGenerator:
    def __init__(self, api_key):
        self.client = openai.OpenAI(api_key=api_key)

    def generate_pipeline(self, requirements: str) -> dict:
        """
        자연어 요구사항을 받아 데이터 파이프라인 코드 생성
        """
        prompt = f"""
        다음 요구사항을 만족하는 데이터 파이프라인을 생성하세요:

        요구사항:
        {requirements}

        생성할 것:
        1. 데이터 소스 연결 코드
        2. 변환 로직 (dbt SQL 모델)
        3. 데이터 품질 검증 (Great Expectations)
        4. Airflow DAG 정의

        JSON 형식으로 반환하세요.
        """

        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are a data engineering expert."},
                {"role": "user", "content": prompt}
            ],
            temperature=0.3
        )

        pipeline_spec = response.choices[0].message.content
        return self.parse_and_validate(pipeline_spec)

    def parse_and_validate(self, spec: str) -> dict:
        """생성된 파이프라인 검증 및 파싱"""
        import json
        import re

        # JSON 추출
        json_match = re.search(r'\{.*\}', spec, re.DOTALL)
        if json_match:
            pipeline = json.loads(json_match.group())

            # 필수 컴포넌트 검증
            required = ['source', 'transformations', 'quality_checks', 'schedule']
            if all(key in pipeline for key in required):
                return pipeline

        raise ValueError("Invalid pipeline specification")

from sklearn.ensemble import RandomForestClassifier
import pandas as pd
import numpy as np

class IntelligentSchemaMapper:
    """
    머신러닝을 활용한 자동 스키마 매핑
    """
    def __init__(self):
        self.model = RandomForestClassifier(n_estimators=100)
        self.is_trained = False

    def extract_column_features(self, df: pd.DataFrame, col: str) -> dict:
        """컬럼의 특성 추출"""
        series = df[col]

        features = {
            'name_length': len(col),
            'has_id': 'id' in col.lower(),
            'has_date': any(word in col.lower() for word in ['date', 'time', 'timestamp']),
            'has_amount': any(word in col.lower() for word in ['amount', 'price', 'total']),
            'null_ratio': series.isnull().sum() / len(series),
            'unique_ratio': series.nunique() / len(series),
            'is_numeric': pd.api.types.is_numeric_dtype(series),
            'is_datetime': pd.api.types.is_datetime64_any_dtype(series),
            'avg_string_length': series.astype(str).str.len().mean() if not pd.api.types.is_numeric_dtype(series) else 0,
            'has_email_pattern': series.astype(str).str.contains('@').any() if not pd.api.types.is_numeric_dtype(series) else False
        }

        return features

    def train(self, training_data: list):
        """
        학습 데이터: [(source_col_features, target_col_name), ...]
        """
        X = []
        y = []

        for features, target_name in training_data:
            feature_vector = list(features.values())
            X.append(feature_vector)
            y.append(target_name)

        self.model.fit(X, y)
        self.is_trained = True

    def map_schemas(self, source_df: pd.DataFrame, target_schema: dict) -> dict:
        """
        소스 DataFrame과 타겟 스키마를 자동으로 매핑
        """
        if not self.is_trained:
            raise ValueError("Model not trained. Call train() first.")

        mapping = {}

        for source_col in source_df.columns:
            features = self.extract_column_features(source_df, source_col)
            feature_vector = [list(features.values())]

            # 가장 가능성 높은 타겟 컬럼 예측
            predicted_target = self.model.predict(feature_vector)[0]
            confidence = self.model.predict_proba(feature_vector)[0].max()

            mapping[source_col] = {
                'target': predicted_target,
                'confidence': confidence,
                'method': 'ml_prediction'
            }

        return mapping

3. 지능형 데이터 품질 관리

AI를 활용하여 데이터 품질 이슈를 자동으로 탐지하고 수정하는 시스템이 등장하고 있습니다.

이상 탐지 및 자동 수정

from sklearn.ensemble import IsolationForest
from sklearn.preprocessing import StandardScaler
import pandas as pd
import numpy as np

class IntelligentDataQualityManager:
    """
    AI 기반 데이터 품질 관리자
    """
    def __init__(self):
        self.anomaly_detectors = {}
        self.scalers = {}

    def detect_anomalies(self, df: pd.DataFrame, column: str) -> pd.Series:
        """
        특정 컬럼의 이상값 탐지
        """
        # 컬럼이 숫자형인 경우에만 처리
        if not pd.api.types.is_numeric_dtype(df[column]):
            return pd.Series([False] * len(df))

        # 데이터 준비
        X = df[[column]].fillna(df[column].median())

        # 스케일링
        if column not in self.scalers:
            self.scalers[column] = StandardScaler()
            X_scaled = self.scalers[column].fit_transform(X)
        else:
            X_scaled = self.scalers[column].transform(X)

        # Isolation Forest로 이상 탐지
        if column not in self.anomaly_detectors:
            self.anomaly_detectors[column] = IsolationForest(
                contamination=0.05,  # 5%가 이상값이라고 가정
                random_state=42
            )
            predictions = self.anomaly_detectors[column].fit_predict(X_scaled)
        else:
            predictions = self.anomaly_detectors[column].predict(X_scaled)

        # -1은 이상값, 1은 정상값
        is_anomaly = predictions == -1

        return pd.Series(is_anomaly, index=df.index)

    def auto_fix(self, df: pd.DataFrame, column: str, method: str = 'median') -> pd.DataFrame:
        """
        이상값 자동 수정
        """
        df_fixed = df.copy()
        anomalies = self.detect_anomalies(df, column)

        if method == 'median':
            # 중앙값으로 대체
            normal_values = df.loc[~anomalies, column]
            median_value = normal_values.median()
            df_fixed.loc[anomalies, column] = median_value

        elif method == 'interpolate':
            # 선형 보간
            df_fixed.loc[anomalies, column] = np.nan
            df_fixed[column] = df_fixed[column].interpolate(method='linear')

        elif method == 'remove':
            # 이상값 제거
            df_fixed = df_fixed[~anomalies]

        return df_fixed

    def generate_quality_report(self, df: pd.DataFrame) -> dict:
        """
        데이터 품질 리포트 생성
        """
        report = {
            'total_rows': len(df),
            'columns': {},
            'overall_quality_score': 0
        }

        quality_scores = []

        for col in df.columns:
            col_report = {
                'dtype': str(df[col].dtype),
                'null_count': df[col].isnull().sum(),
                'null_percentage': df[col].isnull().sum() / len(df) * 100,
                'unique_count': df[col].nunique(),
                'unique_percentage': df[col].nunique() / len(df) * 100
            }

            # 숫자형 컬럼에 대한 추가 분석
            if pd.api.types.is_numeric_dtype(df[col]):
                anomalies = self.detect_anomalies(df, col)
                col_report['anomaly_count'] = anomalies.sum()
                col_report['anomaly_percentage'] = anomalies.sum() / len(df) * 100
                col_report['mean'] = df[col].mean()
                col_report['std'] = df[col].std()

            # 품질 점수 계산 (0-100)
            quality_score = 100
            quality_score -= col_report['null_percentage']
            if 'anomaly_percentage' in col_report:
                quality_score -= col_report['anomaly_percentage']

            col_report['quality_score'] = max(0, quality_score)
            quality_scores.append(quality_score)

            report['columns'][col] = col_report

        report['overall_quality_score'] = np.mean(quality_scores)

        return report

4. 자연어 기반 데이터 파이프라인

자연어로 데이터 파이프라인을 설명하면 자동으로 구현 코드가 생성되는 시대가 도래하고 있습니다.

LLM 기반 SQL 생성

import openai
from typing import List, Dict

class NaturalLanguagePipelineBuilder:
    """
    자연어를 SQL 및 파이프라인 코드로 변환
    """
    def __init__(self, api_key: str, database_schema: Dict):
        self.client = openai.OpenAI(api_key=api_key)
        self.database_schema = database_schema

    def nl_to_sql(self, natural_language_query: str) -> str:
        """
        자연어 쿼리를 SQL로 변환
        """
        schema_description = self._format_schema_for_llm()

        prompt = f"""
        다음 데이터베이스 스키마를 참고하여 자연어 쿼리를 SQL로 변환하세요:

        스키마:
        {schema_description}

        자연어 쿼리:
        {natural_language_query}

        SQL 쿼리만 반환하세요 (설명 없이).
        """

        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are a SQL expert."},
                {"role": "user", "content": prompt}
            ],
            temperature=0
        )

        sql_query = response.choices[0].message.content.strip()

        # SQL 추출 (```sql ... ``` 형식인 경우)
        if "```sql" in sql_query:
            sql_query = sql_query.split("```sql")[1].split("```")[0].strip()

        return sql_query

    def nl_to_dbt_model(self, description: str, model_name: str) -> str:
        """
        자연어 설명을 dbt 모델로 변환
        """
        prompt = f"""
        다음 설명에 맞는 dbt SQL 모델을 생성하세요:

        모델명: {model_name}
        설명: {description}

        스키마:
        {self._format_schema_for_llm()}

        dbt 모델 코드만 반환하세요.
        config, CTE, 주석을 포함하세요.
        """

        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are a dbt expert."},
                {"role": "user", "content": prompt}
            ],
            temperature=0.3
        )

        return response.choices[0].message.content.strip()

    def nl_to_airflow_dag(self, pipeline_description: str) -> str:
        """
        자연어 파이프라인 설명을 Airflow DAG로 변환
        """
        prompt = f"""
        다음 파이프라인 설명을 Airflow DAG Python 코드로 변환하세요:

        {pipeline_description}

        요구사항:
        - PythonOperator, PostgresOperator 등 적절한 Operator 사용
        - 에러 핸들링 포함
        - 리트라이 로직 포함
        - 이메일 알림 설정

        Python 코드만 반환하세요.
        """

        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are an Airflow expert."},
                {"role": "user", "content": prompt}
            ],
            temperature=0.3
        )

        return response.choices[0].message.content.strip()

    def _format_schema_for_llm(self) -> str:
        """스키마를 LLM이 이해하기 쉬운 형식으로 변환"""
        schema_text = []
        for table, columns in self.database_schema.items():
            schema_text.append(f"\nTable: {table}")
            for col_name, col_type in columns.items():
                schema_text.append(f"  - {col_name} ({col_type})")
        return "\n".join(schema_text)

자가 복구 파이프라인 예제

import time
import logging
from typing import Callable, Any
from functools import wraps

class SelfHealingPipeline:
    """
    자가 복구 기능을 가진 데이터 파이프라인
    """
    def __init__(self):
        self.logger = logging.getLogger(__name__)
        self.failure_count = {}
        self.recovery_strategies = {}

    def self_healing(
        self,
        max_retries: int = 3,
        backoff_factor: int = 2,
        fallback: Callable = None
    ):
        """
        자가 복구 데코레이터
        """
        def decorator(func: Callable) -> Callable:
            @wraps(func)
            def wrapper(*args, **kwargs) -> Any:
                func_name = func.__name__

                for attempt in range(max_retries):
                    try:
                        result = func(*args, **kwargs)

                        # 성공 시 실패 카운트 리셋
                        if func_name in self.failure_count:
                            del self.failure_count[func_name]

                        return result

                    except Exception as e:
                        # 실패 카운트 증가
                        self.failure_count[func_name] = self.failure_count.get(func_name, 0) + 1

                        self.logger.error(
                            f"Attempt {attempt + 1}/{max_retries} failed for {func_name}: {str(e)}"
                        )

                        if attempt < max_retries - 1:
                            # 백오프 대기
                            sleep_time = backoff_factor ** attempt
                            self.logger.info(f"Retrying in {sleep_time} seconds...")
                            time.sleep(sleep_time)

                            # 복구 전략 적용
                            if func_name in self.recovery_strategies:
                                self.logger.info(f"Applying recovery strategy for {func_name}")
                                self.recovery_strategies[func_name]()
                        else:
                            # 모든 재시도 실패
                            if fallback:
                                self.logger.warning(f"All retries failed. Using fallback for {func_name}")
                                return fallback(*args, **kwargs)
                            else:
                                raise

            return wrapper
        return decorator

    def register_recovery_strategy(self, func_name: str, strategy: Callable):
        """복구 전략 등록"""
        self.recovery_strategies[func_name] = strategy

6. 데이터 메시 (Data Mesh)

도메인 중심의 분산 데이터 아키텍처인 데이터 메시가 차세대 데이터 플랫폼의 패러다임으로 주목받고 있습니다.

데이터 메시 핵심 원칙

원칙 설명 기존 접근 방식과의 차이
도메인 소유권 각 도메인 팀이 자신의 데이터 제품 소유 중앙 집중식 데이터 팀 → 분산된 도메인 팀

7. 엣지 컴퓨팅 및 IoT 데이터 통합

IoT 기기의 폭발적 증가와 함께 엣지에서의 데이터 처리 및 통합이 중요해지고 있습니다.

엣지 데이터 처리 예제

import asyncio
from typing import Dict, List
import json

class EdgeDataProcessor:
    """
    엣지 디바이스에서 데이터 처리 및 집계
    """
    def __init__(self, batch_size: int = 100, flush_interval: int = 10):
        self.batch_size = batch_size
        self.flush_interval = flush_interval
        self.buffer = []
        self.aggregated_data = {}

    async def process_sensor_data(self, sensor_id: str, reading: float):
        """
        센서 데이터 처리 (엣지에서)
        """
        # 데이터 버퍼에 추가
        self.buffer.append({
            'sensor_id': sensor_id,
            'reading': reading,
            'timestamp': asyncio.get_event_loop().time()
        })

        # 집계 업데이트
        if sensor_id not in self.aggregated_data:
            self.aggregated_data[sensor_id] = {
                'count': 0,
                'sum': 0,
                'min': float('inf'),
                'max': float('-inf')
            }

        agg = self.aggregated_data[sensor_id]
        agg['count'] += 1
        agg['sum'] += reading
        agg['min'] = min(agg['min'], reading)
        agg['max'] = max(agg['max'], reading)

        # 배치가 차면 클라우드로 전송
        if len(self.buffer) >= self.batch_size:
            await self.flush_to_cloud()

    async def flush_to_cloud(self):
        """
        집계된 데이터를 클라우드로 전송 (원본 데이터는 전송하지 않음)
        """
        if not self.buffer:
            return

        # 집계 데이터만 클라우드로 전송 (대역폭 절약)
        summary = {}
        for sensor_id, agg in self.aggregated_data.items():
            summary[sensor_id] = {
                'avg': agg['sum'] / agg['count'],
                'min': agg['min'],
                'max': agg['max'],
                'count': agg['count']
            }

        print(f"Sending aggregated data to cloud: {json.dumps(summary, indent=2)}")

        # 실제로는 HTTP POST, MQTT 등으로 전송
        # await send_to_cloud(summary)

        # 버퍼 및 집계 초기화
        self.buffer.clear()
        self.aggregated_data.clear()

    async def start_periodic_flush(self):
        """주기적으로 데이터 플러시"""
        while True:
            await asyncio.sleep(self.flush_interval)
            await self.flush_to_cloud()

# 사용 예제
async def simulate_iot_sensors():
    processor = EdgeDataProcessor(batch_size=10, flush_interval=5)

    # 주기적 플러시 시작
    asyncio.create_task(processor.start_periodic_flush())

    # 센서 데이터 시뮬레이션
    import random
    for i in range(50):
        sensor_id = f"sensor_{random.randint(1, 5)}"
        reading = random.uniform(20, 30)  # 온도 센서

        await processor.process_sensor_data(sensor_id, reading)
        await asyncio.sleep(0.1)

10. 윤리적 AI 및 책임있는 데이터 통합

AI가 데이터 통합에 더 많이 사용됨에 따라, 윤리적 고려사항이 중요해집니다.

핵심 고려사항

영역 과제 해결 방안
편향성 AI 모델의 편향된 의사결정 공정성 메트릭 모니터링, 다양한 학습 데이터
투명성 AI 의사결정 과정의 불투명성 설명 가능한 AI (XAI), 의사결정 로그
프라이버시 개인정보 보호 차등 프라이버시, 연합 학습
책임성 AI 오류에 대한 책임 소재 인간 검토 프로세스, 감사 추적
보안 적대적 공격에 대한 취약성 강건한 모델 학습, 입력 검증

핵심 요점

  • AI 자동화: 파이프라인 생성, 스키마 매핑, 품질 관리가 AI로 자동화됩니다.
  • 자연어 인터페이스: 자연어로 파이프라인을 정의하고 구현할 수 있습니다.
  • 자율 플랫폼: 자가 관리, 자가 복구, 자가 최적화하는 시스템이 등장합니다.
  • 데이터 메시: 중앙 집중식에서 도메인 분산형 아키텍처로 전환됩니다.
  • 엣지 처리: IoT와 엣지 컴퓨팅으로 데이터 처리가 분산됩니다.
  • 실시간 우선: 배치 처리에서 스트리밍 우선으로 패러다임이 전환됩니다.
  • 블록체인: 데이터 무결성과 계보 추적에 활용됩니다.
  • 윤리성: AI 편향성, 투명성, 프라이버시 보호가 중요해집니다.
  • AutoML 통합: 머신러닝이 데이터 파이프라인에 네이티브로 통합됩니다.
  • 지속 가능성: 에너지 효율적인 데이터 처리가 필수가 됩니다.

요약

이 장에서는 데이터 통합의 미래 트렌드와 AI 자동화 기술을 살펴보았습니다. AI 기반 파이프라인 자동 생성, 지능형 데이터 품질 관리, 자연어 인터페이스, 자율 데이터 플랫폼, 데이터 메시, 엣지 컴퓨팅 등 혁신적인 기술들이 데이터 통합 분야를 변화시키고 있습니다.

다음 5-10년간 데이터 통합은 더욱 자동화되고 지능화될 것입니다. AI와 머신러닝이 데이터 파이프라인의 설계, 구현, 운영 전반에 걸쳐 핵심 역할을 할 것이며, 실시간 처리가 표준이 될 것입니다. 동시에 데이터 윤리, 프라이버시, 지속 가능성에 대한 고려도 더욱 중요해질 것입니다.

이러한 변화에 대비하여 조직은 지속적인 학습과 실험을 통해 새로운 기술을 적극적으로 도입하고, 동시에 윤리적이고 책임있는 데이터 관리를 실천해야 합니다.
