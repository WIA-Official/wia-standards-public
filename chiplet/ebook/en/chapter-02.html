Chapter 2: Chiplet Architecture and Interconnects

This chapter explores the architectural foundations of chiplet systems, focusing on interconnect standards that enable die-to-die communication. We examine UCIe (Universal Chiplet Interconnect Express), Bunch of Wires (BoW), and other protocols, along with 2.5D and 3D packaging architectures that make multi-die integration possible.

The Interconnect Challenge

Die-to-die interconnects must achieve near on-die performance while crossing package boundaries. The industry has converged on several standards to enable chiplet interoperability.

UCIe: Universal Chiplet Interconnect Express

UCIe represents the industry's effort to create a universal standard for chiplet interconnection. Backed by major semiconductor companies including Intel, AMD, ARM, TSMC, Samsung, and many others, UCIe aims to enable an open chiplet ecosystem.

// UCIe Implementation Example
class UCIeController {
    private modules: UCIeModule[];
    private linkStatus: Map<string, LinkState>;

    constructor(config: UCIeConfig) {
        this.modules = config.modules.map(m => new UCIeModule(m));
        this.linkStatus = new Map();
    }

    // Link Training and Initialization
    async initializeLink(targetChiplet: string): Promise<LinkResult> {
        // Phase 1: Reset and detect
        await this.performReset();
        const detected = await this.detectPartner(targetChiplet);

        // Phase 2: Parameter exchange
        const params = await this.exchangeParameters(detected);

        // Phase 3: Speed training
        const trainedSpeed = await this.trainSpeed(params);

        // Phase 4: Link ready
        this.linkStatus.set(targetChiplet, {
            state: 'L0',  // Active state
            speed: trainedSpeed,
            width: params.negotiatedWidth
        });

        return {
            success: true,
            bandwidth: this.calculateBandwidth(trainedSpeed, params.negotiatedWidth)
        };
    }

    // Data Transfer with Flow Control
    async sendData(packet: UCIePacket): Promise<void> {
        // Credit-based flow control
        await this.waitForCredits(packet.size);

        // Apply CRC
        packet.crc = this.calculateCRC(packet.data);

        // Transmit
        await this.transmit(packet);

        // Wait for ACK or handle retry
        const ack = await this.waitForAck(packet.sequenceNumber);
        if (!ack.success) {
            await this.handleRetry(packet);
        }
    }
}

Bunch of Wires (BoW) Protocol

BoW Philosophy

BoW minimizes protocol overhead by extending the on-die interconnect across the die boundary. Ideal when chiplets are designed together and maximum bandwidth efficiency is required.

// Bunch of Wires Interface Definition
interface BoWInterface {
    // Physical Configuration
    physical: {
        wireCount: number;      // Total wires in bundle
        dataWires: number;      // Data-carrying wires
        controlWires: number;   // Clock, valid, credit signals
        bumpPitch: "25-55 μm";  // Typically advanced packaging
    };

    // Signaling
    signaling: {
        type: "Single-ended" | "Differential";
        dataRate: string;       // e.g., "2-8 Gbps"
        clockScheme: "Forwarded" | "Embedded";
    };

    // Flow Control (minimal)
    flowControl: {
        type: "Credit" | "Ready/Valid" | "None";
        credits: number;
        latency: "Minimal";     // Typically 1-2 cycles
    };
}

// BoW Implementation
class BoWLink {
    private config: BoWInterface;

    constructor(config: BoWInterface) {
        this.config = config;
    }

    // Direct wire mapping
    mapSignals(): SignalMapping {
        return {
            txData: range(0, this.config.physical.dataWires),
            txClock: this.config.physical.dataWires,
            txValid: this.config.physical.dataWires + 1,
            txCredit: this.config.physical.dataWires + 2,
            // Reverse direction
            rxData: range(0, this.config.physical.dataWires),
            rxClock: this.config.physical.dataWires,
            rxValid: this.config.physical.dataWires + 1,
            rxCredit: this.config.physical.dataWires + 2
        };
    }

    // Calculate raw bandwidth
    getBandwidth(): number {
        const dataWires = this.config.physical.dataWires;
        const dataRate = parseFloat(this.config.signaling.dataRate);
        return dataWires * dataRate / 8;  // GB/s
    }
}

        "Scalable to large packages",
        "Localized high-density routing",
        "Compatible with standard packaging"

// EMIB System Design
class EMIBSystem {
    private bridges: EMIBBridge[];
    private chiplets: Chiplet[];

    placeChiplets(chiplets: Chiplet[]): PlacementResult {
        // Identify communication-intensive pairs
        const pairs = this.identifyHighBandwidthPairs(chiplets);

        // Place bridges between pairs
        const bridgePlacements = pairs.map(pair => ({
            bridge: new EMIBBridge(),
            chipletA: pair[0],
            chipletB: pair[1],
            bandwidth: this.calculateBandwidth(pair)
        }));

        return {
            chipletPositions: this.optimizePositions(chiplets, bridgePlacements),
            bridges: bridgePlacements
        };
    }
}

3D Packaging Architecture

3D Integration Benefits

Vertical stacking achieves interconnect densities 10-100x greater than 2.5D, enabling memory-on-logic configurations with TB/s bandwidth.

3D Memory Integration (HBM)

// HBM (High Bandwidth Memory) Configuration
interface HBMStack {
    generation: "HBM2" | "HBM2E" | "HBM3" | "HBM3E";
    stackHeight: number;        // DRAM dies (4, 8, 12)
    capacityPerStack: number;   // GB

    interface: {
        channelWidth: 128;      // bits per channel
        channels: number;       // 8 or 16
        dataRate: number;       // Gbps
    };

    // Performance
    getBandwidth(): number {
        return this.interface.channelWidth *
               this.interface.channels *
               this.interface.dataRate / 8 / 1000; // TB/s
    }
}

Topology Design Patterns

// Chiplet Topology Patterns
enum ChipletTopology {
    MESH = "mesh",           // Grid connectivity
    RING = "ring",           // Circular connectivity
    STAR = "star",           // Central hub
    HIERARCHICAL = "hier",   // Tree-like structure
    CUSTOM = "custom"        // Application-specific
}

interface TopologyAnalysis {
    topology: ChipletTopology;
    chipletCount: number;

    analyze(): TopologyMetrics {
        switch(this.topology) {
            case ChipletTopology.MESH:
                return {
                    bisectionBandwidth: "High",
                    averageHops: Math.sqrt(this.chipletCount),
                    scalability: "Excellent",
                    wiringComplexity: "High"
                };
            case ChipletTopology.RING:
                return {
                    bisectionBandwidth: "Low",
                    averageHops: this.chipletCount / 4,
                    scalability: "Good",
                    wiringComplexity: "Low"
                };
            case ChipletTopology.STAR:
                return {
                    bisectionBandwidth: "Medium",
                    averageHops: 2,  // Through central hub
                    scalability: "Limited by hub",
                    wiringComplexity: "Medium"
                };
        }
    }
}
Chapter Summary

Chiplet architecture success depends on choosing appropriate interconnect standards and packaging technologies. UCIe provides industry-standard interoperability, while BoW offers maximum efficiency for custom designs. 2.5D packaging balances performance and manufacturability, while 3D stacking enables the highest integration density. Understanding these options enables optimal chiplet system design.
