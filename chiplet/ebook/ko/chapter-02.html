챕터 2: 칩렛 아키텍처와 인터커넥트

이 챕터에서는 다이-투-다이 통신을 가능하게 하는 인터커넥트 표준에 초점을 맞춰 칩렛 시스템의 아키텍처 기초를 탐구합니다. UCIe(Universal Chiplet Interconnect Express), Bunch of Wires(BoW) 및 기타 프로토콜과 함께 멀티 다이 통합을 가능하게 하는 2.5D 및 3D 패키징 아키텍처를 살펴봅니다.

인터커넥트 표준 개요

칩렛 아키텍처의 성공은 서로 다른 공급업체의 칩렛이 효율적으로 통신할 수 있게 하는 표준화된 인터커넥트에 달려 있습니다. 대역폭, 지연, 전력 및 구현 복잡성 간에 서로 다른 트레이드오프를 가진 여러 표준이 이 요구를 해결하기 위해 등장했습니다.

인터커넥트 과제

다이-투-다이 인터커넥트는 패키지 경계를 넘으면서 거의 온다이 성능을 달성해야 합니다. 이를 위해서는 뛰어난 신호 무결성, 정밀한 타이밍 및 효율적인 프로토콜이 필요합니다. 업계는 칩렛 상호 운용성을 가능하게 하기 위해 여러 표준으로 수렴했습니다.

UCIe: Universal Chiplet Interconnect Express

UCIe는 칩렛 상호 연결을 위한 범용 표준을 만들기 위한 업계의 노력을 나타냅니다. Intel, AMD, ARM, TSMC, Samsung 등 주요 반도체 회사들의 지원을 받는 UCIe는 개방형 칩렛 생태계를 가능하게 하는 것을 목표로 합니다.

UCIe 아키텍처 레이어

// UCIe 프로토콜 스택 정의
interface UCIeStack {
    // 물리 계층 (PHY)
    physical: {
        // 표준 패키지
        standardPackage: {
            bumpPitch: "100-130 마이크로미터";
            dataRate: "4-32 GT/s";
            efficiency: "0.5-2 pJ/bit";
        };
        // 고급 패키지
        advancedPackage: {
            bumpPitch: "25-55 마이크로미터";
            dataRate: "4-32 GT/s";
            efficiency: "0.25-0.5 pJ/bit";
        };
        // 인코딩
        encoding: "64b/68b" | "Raw";
    };

    // 다이-투-다이 어댑터 계층 (D2D)
    d2dAdapter: {
        protocol: "UCIe Streaming" | "CXL" | "PCIe";
        flowControl: "크레딧 기반";
        retryMechanism: boolean;
        crcProtection: boolean;
    };

    // 프로토콜 계층
    protocol: {
        baseProtocol: "Streaming";
        optionalProtocols: ["CXL.io", "CXL.cache", "CXL.mem", "PCIe"];
    };
}

// UCIe 모듈 구성
interface UCIeModule {
    lanes: 16 | 32 | 64;
    speedGrade: "4GT/s" | "8GT/s" | "16GT/s" | "32GT/s";

    // 대역폭 계산
    getBandwidth(): number {
        const rawBits = this.lanes * this.speedGrade;
        const encodingEfficiency = 64/68;  // 64b/68b 인코딩
        return rawBits * encodingEfficiency * 2;  // 양방향
    }
}

// UCIe 구현 예제
class UCIeController {
    private modules: UCIeModule[];
    private linkStatus: Map;

    constructor(config: UCIeConfig) {
        this.modules = config.modules.map(m => new UCIeModule(m));
        this.linkStatus = new Map();
    }

    // 링크 훈련 및 초기화
    async initializeLink(targetChiplet: string): Promise {
        // 1단계: 리셋 및 감지
        await this.performReset();
        const detected = await this.detectPartner(targetChiplet);

        // 2단계: 파라미터 교환
        const params = await this.exchangeParameters(detected);

        // 3단계: 속도 훈련
        const trainedSpeed = await this.trainSpeed(params);

        // 4단계: 링크 준비
        this.linkStatus.set(targetChiplet, {
            state: 'L0',  // 활성 상태
            speed: trainedSpeed,
            width: params.negotiatedWidth
        });

        return {
            success: true,
            bandwidth: this.calculateBandwidth(trainedSpeed, params.negotiatedWidth)
        };
    }
}

Bunch of Wires (BoW) 프로토콜

Bunch of Wires(BoW) 접근 방식은 복잡한 프로토콜에 대한 더 간단하고 낮은 오버헤드 대안을 나타냅니다. BoW는 다이-투-다이 인터페이스를 최소한의 프로토콜 오버헤드로 병렬 와이어로 취급하여 지연에 민감한 애플리케이션에서 이점을 제공합니다.

BoW 철학

BoW는 본질적으로 온다이 인터커넥트를 다이 경계를 넘어 확장하여 프로토콜 오버헤드를 최소화합니다. 이 접근 방식은 두 칩렛이 함께 설계되고 최대 대역폭 효율성이 필요한 경우에 이상적입니다.

// Bunch of Wires 인터페이스 정의
interface BoWInterface {
    // 물리적 구성
    physical: {
        wireCount: number;      // 번들의 총 와이어 수
        dataWires: number;      // 데이터 전달 와이어
        controlWires: number;   // 클록, 유효, 크레딧 신호
        bumpPitch: "25-55 마이크로미터";  // 일반적으로 고급 패키징
    };

    // 시그널링
    signaling: {
        type: "Single-ended" | "Differential";
        dataRate: string;       // 예: "2-8 Gbps"
        clockScheme: "Forwarded" | "Embedded";
    };

    // 흐름 제어 (최소)
    flowControl: {
        type: "Credit" | "Ready/Valid" | "None";
        credits: number;
        latency: "최소";     // 일반적으로 1-2 사이클
    };
}

2.5D 패키징 아키텍처

2.5D 패키징은 실리콘, 유기 또는 유리 인터포저 위에 여러 칩렛을 나란히 배치합니다. 이 접근 방식은 관리 가능한 열 및 제조 복잡성을 유지하면서 고대역폭 다이-투-다이 통신을 가능하게 합니다.

실리콘 인터포저 기술

// 실리콘 인터포저 사양
interface SiliconInterposer {
    // 물리적 치수
    dimensions: {
        size: number;           // mm 제곱 (일반적으로 800-2500)
        thickness: number;      // 마이크로미터 (일반적으로 100-150)
        metalLayers: number;    // 일반적으로 4-8 레이어
    };

    // TSV (Through-Silicon Via) 구성
    tsv: {
        diameter: number;       // 마이크로미터 (일반적으로 5-10)
        pitch: number;          // 마이크로미터 (일반적으로 40-100)
        density: number;        // mm 제곱당 TSV 수
        resistance: number;     // TSV당 mOhm
    };

    // 라우팅 기능
    routing: {
        minLineWidth: number;   // nm (일반적으로 400-2000)
        minSpacing: number;     // nm
        redistribution: boolean;
    };
}

// TSMC CoWoS 구성 예제

3D 패키징 아키텍처

3D 패키징은 칩렛을 수직으로 스태킹하여 가장 높은 인터커넥트 밀도를 제공하지만 상당한 열 및 설계 과제가 있습니다. 이 기술은 전례 없는 통합 밀도를 가능하게 합니다.

3D 통합 이점

수직 스태킹은 2.5D 접근 방식보다 10-100배 더 높은 인터커넥트 밀도를 달성할 수 있어 TB/s 대역폭의 메모리-온-로직 구성을 가능하게 합니다. 그러나 열 관리가 주요 과제가 됩니다.

// 3D 스태킹 아키텍처
interface Stack3DConfiguration {
    // 스택 구성
    layers: StackLayer[];
    bondingTechnology: "MicroBump" | "HybridBond" | "Cu-Cu";

    // 열 설계
    thermal: {
        totalTDP: number;           // 와트
        perLayerBudget: number[];   // 레이어당 와트
        coolingMethod: "TopSide" | "TSV_HeatPath" | "Microfluidic";
    };

    // 인터커넥트 밀도
    interconnect: {
        bondPitch: number;          // 마이크로미터 (일반적으로 0.5-10)
        connectionsPerMm2: number;
        verticalBandwidth: number;  // TB/s
    };
}

// HBM (High Bandwidth Memory) 구성
interface HBMStack {
    generation: "HBM2" | "HBM2E" | "HBM3" | "HBM3E";
    stackHeight: number;        // DRAM 다이 (4, 8, 12)
    capacityPerStack: number;   // GB

    interface: {
        channelWidth: 128;      // 채널당 비트
        channels: number;       // 8 또는 16
        dataRate: number;       // Gbps
    };

    // 성능
    getBandwidth(): number {
        return this.interface.channelWidth *
               this.interface.channels *
               this.interface.dataRate / 8 / 1000; // TB/s
    }
}

토폴로지 설계 패턴

칩렛 시스템은 대역폭, 지연 및 확장성에 대해 서로 다른 특성을 가진 다양한 토폴로지로 구성될 수 있습니다.

// 칩렛 토폴로지 패턴
enum ChipletTopology {
    MESH = "mesh",           // 그리드 연결
    RING = "ring",           // 순환 연결
    STAR = "star",           // 중앙 허브
    HIERARCHICAL = "hier",   // 트리형 구조
    CUSTOM = "custom"        // 애플리케이션 특정
}

interface TopologyAnalysis {
    topology: ChipletTopology;
    chipletCount: number;

    analyze(): TopologyMetrics {
        switch(this.topology) {
            case ChipletTopology.MESH:
                return {
                    bisectionBandwidth: "높음",
                    averageHops: Math.sqrt(this.chipletCount),
                    scalability: "우수",
                    wiringComplexity: "높음"
                };
            case ChipletTopology.RING:
                return {
                    bisectionBandwidth: "낮음",
                    averageHops: this.chipletCount / 4,
                    scalability: "양호",
                    wiringComplexity: "낮음"
                };
            case ChipletTopology.STAR:
                return {
                    bisectionBandwidth: "중간",
                    averageHops: 2,  // 중앙 허브를 통해
                    scalability: "허브에 의해 제한",
                    wiringComplexity: "중간"
                };
        }
    }
}
챕터 요약

칩렛 아키텍처의 성공은 적절한 인터커넥트 표준과 패키징 기술 선택에 달려 있습니다. UCIe는 업계 표준 상호 운용성을 제공하고, BoW는 커스텀 설계에 최대 효율성을 제공합니다. 2.5D 패키징은 성능과 제조 용이성의 균형을 맞추고, 3D 스태킹은 가장 높은 통합 밀도를 가능하게 합니다. 이러한 옵션을 이해하면 최적의 칩렛 시스템 설계가 가능합니다.
