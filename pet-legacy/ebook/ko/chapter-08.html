8.3 데이터베이스 스키마

핵심 테이블
==========

-- 유산 프로필
CREATE TABLE legacy_profiles (
  legacy_id UUID PRIMARY KEY,
  pet_id UUID NOT NULL,
  owner_id UUID NOT NULL,
  pet_info JSONB NOT NULL,
  status VARCHAR(20) DEFAULT 'PLANNING',
  memorial_url TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  activated_at TIMESTAMPTZ
);

-- 생애 사건
CREATE TABLE life_events (
  event_id UUID PRIMARY KEY,
  legacy_id UUID REFERENCES legacy_profiles,
  event_type VARCHAR(30) NOT NULL,
  title VARCHAR(255) NOT NULL,
  event_date DATE NOT NULL,
  significance SMALLINT DEFAULT 3
);

-- 미디어
CREATE TABLE media_attachments (
  media_id UUID PRIMARY KEY,
  legacy_id UUID REFERENCES legacy_profiles,
  media_type VARCHAR(20) NOT NULL,
  original_url TEXT NOT NULL,
  archive_url TEXT,
  visibility VARCHAR(20) DEFAULT 'FAMILY'
);

-- 헌사
CREATE TABLE tributes (
  tribute_id UUID PRIMARY KEY,
  legacy_id UUID REFERENCES legacy_profiles,
  author_name VARCHAR(255) NOT NULL,
  message TEXT NOT NULL,
  status VARCHAR(20) DEFAULT 'PENDING'
);

-- 보관소
CREATE TABLE legacy_archives (
  archive_id UUID PRIMARY KEY,
  legacy_id UUID REFERENCES legacy_profiles,
  storage_location TEXT NOT NULL,
  checksum VARCHAR(64) NOT NULL,
  retention_years INTEGER DEFAULT 100
);

8.4 API 엔드포인트

API 라우트
==========

유산:
  POST   /api/v1/legacy           생성
  GET    /api/v1/legacy           목록
  GET    /api/v1/legacy/{id}      조회
  PUT    /api/v1/legacy/{id}      수정
  POST   /api/v1/legacy/{id}/activate  활성화

타임라인:
  GET    /api/v1/legacy/{id}/timeline  타임라인 조회
  GET    /api/v1/legacy/{id}/events    사건 목록
  POST   /api/v1/legacy/{id}/events    사건 추가

미디어:
  GET    /api/v1/legacy/{id}/media     미디어 목록
  POST   /api/v1/legacy/{id}/media     업로드
  GET    /api/v1/legacy/{id}/media/featured  AI 선택

추모:
  GET    /memorial/{id}           공개 페이지
  GET    /api/v1/legacy/{id}/memorial/config  설정
  PUT    /api/v1/legacy/{id}/memorial/config  수정

헌사:
  GET    /memorial/{id}/tributes  승인된 목록
  POST   /memorial/{id}/tributes  제출
  POST   /api/v1/legacy/{id}/tributes/{tid}/approve

보관:
  POST   /api/v1/legacy/{id}/archive   생성
  GET    /api/v1/legacy/{id}/archive   상태
  POST   /api/v1/legacy/{id}/export    내보내기

8.6 배포

DOCKER COMPOSE
==============

services:
  legacy-api:
    image: wia/legacy-api:latest
    ports:
      - "8080:8080"

      DATABASE_URL: postgres://db:5432/legacy
    depends_on:
      - db

  media-service:
    image: wia/media-service:latest
    environment:
      STORAGE_BUCKET: wia-legacy-media
      ARCHIVE_BUCKET: wia-legacy-archive
    volumes:
      - media_temp:/tmp/media

  memorial-service:
    image: wia/memorial-service:latest

      CDN_BUCKET: wia-memorial-cdn

    environment:
      GLACIER_VAULT: wia-legacy-glacier
