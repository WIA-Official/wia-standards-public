코드 예제 및 패턴

인기 있는 언어의 일반적인 구현 패턴 및 코드 샘플:

Python 예제
from wia_standard import Client, ValidationError

# 구성으로 클라이언트 초기화
client = Client(
    api_key="your-api-key",
    environment="production",
    timeout=30
)

try:
    # 검증과 함께 리소스 생성
    resource = client.create_resource({
        "name": "예제 리소스",
        "type": "standard-compliant",
        "metadata": {
            "version": "1.0",
            "created_by": "user@example.com"
        }
    })

    # 자동 준수 검증
    validation_result = client.validate(resource)

    if validation_result.is_valid:
        print(f"리소스가 성공적으로 생성되었습니다: {resource.id}")
    else:
        print(f"검증 경고: {validation_result.warnings}")

except ValidationError as e:
    print(f"검증 실패: {e.message}")
    print(f"세부사항: {e.details}")
JavaScript/TypeScript 예제
import { WIAClient, Resource } from '@wia/standard-sdk';

// 타입 안전성과 함께 초기화
const client = new WIAClient({
  apiKey: process.env.WIA_API_KEY,
  region: 'us-east-1',
  retryPolicy: {
    maxRetries: 3,
    backoffMultiplier: 2
  }
});

// 완전한 타입 검사와 함께 리소스 생성
const resource: Resource = await client.resources.create({
  name: '예제 리소스',
  type: 'standard-compliant',
  metadata: {
    version: '1.0',
    createdBy: 'user@example.com'
  }
});

// 업데이트 구독
client.resources.watch(resource.id, (update) => {
  console.log('리소스 업데이트:', update);
});
Java 예제
import com.wia.standard.*;

public class WIAExample {
    public static void main(String[] args) {
        // 빌더 패턴을 사용한 구성
        WIAClient client = WIAClient.builder()
            .apiKey(System.getenv("WIA_API_KEY"))
            .environment(Environment.PRODUCTION)
            .connectionTimeout(Duration.ofSeconds(30))
            .build();

        try {
            // 검증과 함께 리소스 생성
            Resource resource = Resource.builder()
                .name("예제 리소스")
                .type("standard-compliant")
                .metadata(Map.of(
                    "version", "1.0",
                    "createdBy", "user@example.com"
                ))
                .build();

            Resource created = client.createResource(resource);
            System.out.println("생성됨: " + created.getId());

        } catch (ValidationException e) {
            System.err.println("검증 실패: " + e.getMessage());
        } finally {
            client.close();
        }
    }
}
