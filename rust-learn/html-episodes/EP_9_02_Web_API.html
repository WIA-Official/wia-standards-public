First Server

use axum::{routing::get, Router};
#[tokio::main]
async fn main() {
let app = Router::new()
.route("/", get(|| async { "Hello, World!" }))
.route("/health", get(health_check));
let listener = tokio::net::TcpListener::bind("0.0.0.0:3000")
.await.unwrap();
println!("Server running on http://localhost:3000");
axum::serve(listener, app).await.unwrap();
}
async fn health_check() -> &'static str { "OK" }

JSON API

JSON API 구현

Implementing JSON API

use axum::{routing::{get, post}, Json, Router};
use serde::{Deserialize, Serialize};
#[derive(Serialize, Deserialize)]
struct User { id: u32, name: String, email: String }
// GET /users - JSON 배열 반환
async fn get_users() -> Json<Vec<User>> {
Json(vec![User { id: 1, name: "Alice".into(), email: "a@b.c".into() }])
}
// POST /users - JSON 수신
async fn create_user(Json(user): Json<User>) -> Json<User> {
Json(user)
}
let app = Router::new()
.route("/users", get(get_users).post(create_user));

Path & Query

경로 파라미터 & 쿼리

Path Parameters & Query

use axum::extract::{Path, Query};
// GET /users/:id
async fn get_user(Path(id): Path<u32>) -> String {
format!("User ID: {}", id)
}
#[derive(Deserialize)]
struct Pagination { page: Option<u32>, limit: Option<u32> }
// GET /items?page=1&limit=10
async fn list_items(Query(pagination): Query<Pagination>) -> String {
format!("Page: {:?}, Limit: {:?}",
pagination.page, pagination.limit)
}

State

애플리케이션 상태

Application State

use axum::extract::State;
use std::sync::Arc;
use tokio::sync::RwLock;
type SharedState = Arc<RwLock<Vec<String>>>;
async fn add_item(
State(state): State<SharedState>,
item: String
) {
state.write().await.push(item);
}
let state: SharedState = Arc::new(RwLock::new(vec![]));
let app = Router::new()
.route("/items", post(add_item))
.with_state(state);

Error Handling

use axum::{http::StatusCode, response::IntoResponse};
enum AppError {
NotFound,
BadRequest(String),
}
impl IntoResponse for AppError {
fn into_response(self) -> axum::response::Response {
match self {
AppError::NotFound => (StatusCode::NOT_FOUND, "Not found"),
AppError::BadRequest(msg) => (StatusCode::BAD_REQUEST, msg),
}.into_response()
}
}
async fn handler() -> Result<String, AppError> {
Err(AppError::NotFound)
}
