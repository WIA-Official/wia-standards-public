# 제7장: Phase 4 - 시스템 통합

## 번역 생태계와의 통합

WIA-LANG-009의 진정한 가치는 기존 번역 워크플로우와의 원활한 통합에서 나옵니다. Phase 4는 다양한 번역 시스템, 도구, 플랫폼과 표준을 통합하는 방법을 정의합니다.

## 통합 아키텍처 개요

// Phrase 워크플로우 확장
class WIAQualityStep extends WorkflowStep {
    async execute(job) {
        // 번역 완료 후 자동 품질 평가
        const assessment = await this.wiaClient.assess({
            sourceLanguage: job.sourceLanguage,
            targetLanguage: job.targetLanguage,
            segments: job.segments,
            domain: job.project.domain
        });

        // 품질 점수에 따라 라우팅
        if (assessment.overallQuality.score >= 80) {
            return this.moveToStep("review");
        } else if (assessment.overallQuality.score >= 60) {
            return this.moveToStep("post_editing");
        } else {
            return this.moveToStep("retranslation");
        }
    }
}

### SDL Trados Studio 플러그인

**플러그인 구조**:

```csharp
using Sdl.FileTypeSupport.Framework.BilingualApi;
using WIA.Lang009.SDK;

namespace WIA.TradosPlugin
{
    public class QualityAssessmentVerifier : AbstractBilingualContentProcessor
    {
        private WIAQualityClient _wiaClient;

        public override void ProcessParagraphUnit(IParagraphUnit paragraphUnit)
        {
            foreach (var segmentPair in paragraphUnit.SegmentPairs)
            {
                // 세그먼트 품질 평가
                var assessment = _wiaClient.AssessSegment(new SegmentRequest
                {
                    SourceLanguage = DocumentProperties.SourceLanguage.CultureInfo.Name,
                    TargetLanguage = DocumentProperties.TargetLanguage.CultureInfo.Name,
                    Source = segmentPair.Source.ToString(),
                    Target = segmentPair.Target.ToString()
                });

                // 품질 점수에 따라 세그먼트 상태 설정
                if (assessment.QualityScore < 70)
                {
                    segmentPair.Properties.ConfirmationLevel = ConfirmationLevel.Draft;

                    // 경고 메시지 추가
                    MessageReporter.ReportMessage(
                        this,
                        paragraphUnit.Properties.FileUri,
                        ErrorLevel.Warning,
                        $"Low quality score: {assessment.QualityScore}",
                        segmentPair.ToString()
                    );
                }
            }

            base.ProcessParagraphUnit(paragraphUnit);
        }
    }
}
```

### Google Cloud Translation 통합

**품질 추정 파이프라인**:

```python
from google.cloud import translate_v3
from wia_lang_009 import WIAQualityClient

# Google Cloud Translation 클라이언트
translate_client = translate_v3.TranslationServiceClient()
project_id = "your-project-id"
location = "global"
parent = f"projects/{project_id}/locations/{location}"

# WIA Quality 클라이언트
wia = WIAQualityClient(api_key="your-wia-key")

def translate_with_quality_check(texts, source_lang, target_lang):
    """번역 및 품질 검증"""

    # Google 번역
    response = translate_client.translate_text(
        request={
            "parent": parent,
            "contents": texts,
            "mime_type": "text/plain",
            "source_language_code": source_lang,
            "target_language_code": target_lang,
        }
    )

    results = []
    for i, translation in enumerate(response.translations):
        # WIA 품질 추정 (참조 번역 없음)
        qe_result = wia.quality_estimation(
            source=texts[i],
            target=translation.translated_text,
            source_language=source_lang,
            target_language=target_lang
        )

        results.append({
            "source": texts[i],
            "target": translation.translated_text,
            "quality_score": qe_result.sentence_level.da_score,
            "needs_review": qe_result.sentence_level.da_score < 70
        })

    return results

# 사용 예시
texts = [
    "Welcome to our application",
    "Click the save button"
]

results = translate_with_quality_check(texts, "en", "ko")

for result in results:
    print(f"Source: {result['source']}")
    print(f"Target: {result['target']}")
    print(f"Quality: {result['quality_score']}")
    print(f"Needs Review: {result['needs_review']}\n")
```

### DeepL Pro 통합

**품질 비교 분석**:

```typescript
import * as deepl from 'deepl-node';
import { WIAQualityClient } from '@wia/lang-009';

const deeplTranslator = new deepl.Translator(process.env.DEEPL_API_KEY);
const wiaClient = new WIAQualityClient({ apiKey: process.env.WIA_API_KEY });

async function compareTranslationQuality(
  text: string,
  sourceLang: string,
  targetLang: string
) {
  // DeepL 번역 (일반 모델)
  const resultStandard = await deeplTranslator.translateText(
    text,
    sourceLang,
    targetLang
  );

  // DeepL 번역 (형식성 조정)
  const resultFormal = await deeplTranslator.translateText(
    text,
    sourceLang,
    targetLang,
    { formality: 'more' }
  );

  // 두 번역의 품질 평가
  const [assessmentStandard, assessmentFormal] = await Promise.all([
    wiaClient.assess({
      sourceLanguage: sourceLang,
      targetLanguage: targetLang,
      segments: [{ id: '1', source: text, target: resultStandard.text }],
      metrics: ['comet', 'bertscore']
    }),
    wiaClient.assess({
      sourceLanguage: sourceLang,
      targetLanguage: targetLang,
      segments: [{ id: '1', source: text, target: resultFormal.text }],
      metrics: ['comet', 'bertscore']
    })
  ]);

  return {
    standard: {
      text: resultStandard.text,
      quality: assessmentStandard.segments[0].quality.overallScore
    },
    formal: {
      text: resultFormal.text,
      quality: assessmentFormal.segments[0].quality.overallScore
    },
    recommendation:
      assessmentFormal.segments[0].quality.overallScore >
      assessmentStandard.segments[0].quality.overallScore
        ? 'formal'
        : 'standard'
  };
}
```

from transformers import pipeline
from wia_lang_009 import WIAQualityClient

# NLLB 번역 모델 로드
translator = pipeline(

### CMS (Content Management System) 통합

**WordPress 플러그인**:

```php
<?php
/**
 * Plugin Name: WIA Translation Quality
 * Description: Integrates WIA-LANG-009 quality assessment
 */

class WIA_Translation_Quality {
    private $api_client;

    public function __construct() {
        $this->api_client = new WIA_API_Client(
            get_option('wia_api_key')
        );

        add_filter('wpml_tm_translation_job_data', array($this, 'assess_quality'), 10, 2);
    }

    public function assess_quality($translation_job, $job_id) {
        // 번역 완료 후 품질 평가
        $assessment = $this->api_client->assess(array(
            'source_language' => $translation_job->source_language,
            'target_language' => $translation_job->target_language,
            'segments' => $this->extract_segments($translation_job)
        ));

        // 품질 점수 메타데이터로 저장
        update_post_meta(
            $translation_job->post_id,
            '_wia_quality_score',
            $assessment->overall_quality->score
        );

        // 저품질인 경우 리뷰 상태로 설정
        if ($assessment->overall_quality->score < 75) {
            wp_set_post_status($translation_job->post_id, 'pending');

            // 편집자에게 알림
            $this->notify_editor($translation_job, $assessment);
        }

        return $translation_job;
    }
}

new WIA_Translation_Quality();
```

### CI/CD 파이프라인 통합

**GitHub Actions 워크플로우**:

```yaml
name: Translation Quality Check

on:
  pull_request:
    paths:
      - 'locales/**/*.json'

jobs:
  quality-check:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Install WIA CLI
        run: npm install -g @wia/lang-009-cli

      - name: Run quality assessment
        env:
          WIA_API_KEY: ${{ secrets.WIA_API_KEY }}
        run: |
          wia-lang-009 assess \
            --source locales/en/ \
            --target locales/ko/ \
            --domain software \
            --format github-check \
            --output quality-report.json

      - name: Post results as comment
        uses: actions/github-script@v6
        with:
          script: |
            const fs = require('fs');
            const report = JSON.parse(fs.readFileSync('quality-report.json'));

            const body = `
            ## Translation Quality Report

            Overall Score: **${report.overall_score}** / 100
            Grade: **${report.grade}**

            ### Details
            - Segments: ${report.segment_count}
            - Issues: ${report.issue_count}

            ${report.issues.map(i => `- ⚠️ ${i.description}`).join('\n')}
            `;

            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: body
            });

      - name: Fail if quality below threshold
        run: |
          SCORE=$(jq '.overall_score' quality-report.json)
          if (( $(echo "$SCORE < 80" | bc -l) )); then
            echo "Quality score $SCORE is below threshold (80)"
            exit 1
          fi
```

const express = require('express');
const { WIAQualityClient } = require('@wia/lang-009');

const app = express();
const wiaClient = new WIAQualityClient({ apiKey: process.env.WIA_API_KEY });

        target: product.title
      },
      {
        id: 'description',
        source: product.description_en,
        target: product.description
      }
    ]
  });

    shopName: process.env.SHOP_NAME,

    key: 'translation_quality',
    value: assessment.overallQuality.score.toString(),
    type: 'number_decimal',
    namespace: 'wia'
  });

  // 저품질인 경우 제품 비공개
  if (assessment.overallQuality.score < 70) {
    await shopifyClient.product.update(product.id, {
      published: false
    });

    // 관리자에게 알림
    await sendQualityAlert(product, assessment);
  }

  res.json({ success: true });
});

app.listen(3000);

### Grafana 통합

**메트릭 내보내기**:

```python
from prometheus_client import Gauge, CollectorRegistry, push_to_gateway
from wia_lang_009 import WIAQualityClient

# Prometheus 메트릭 정의
registry = CollectorRegistry()

quality_score = Gauge(
    'translation_quality_score',
    'Overall translation quality score',
    ['language_pair', 'domain'],
    registry=registry
)

error_rate = Gauge(
    'translation_error_rate',
    'Translation error rate by severity',
    ['language_pair', 'severity'],
    registry=registry
)

# WIA 클라이언트
wia = WIAQualityClient(api_key="your-wia-key")

def export_metrics(assessment):
    """품질 메트릭을 Prometheus로 내보내기"""

    lang_pair = f"{assessment.metadata.source_language}-{assessment.metadata.target_language}"
    domain = assessment.metadata.domain

    # 전체 품질 점수
    quality_score.labels(
        language_pair=lang_pair,
        domain=domain
    ).set(assessment.overall_quality.score)

    # 심각도별 오류율
    error_counts = count_errors_by_severity(assessment)
    for severity, count in error_counts.items():
        error_rate.labels(
            language_pair=lang_pair,
            severity=severity
        ).set(count / assessment.statistics.segment_count)

    # Prometheus Pushgateway로 전송
    push_to_gateway(
        'localhost:9091',
        job='translation_quality',
        registry=registry
    )
```
