# 제6장: Phase 3 - 프로토콜

## 품질 보증 프로세스의 표준화

번역 품질을 일관되게 유지하려면 체계적인 프로세스가 필요합니다. Phase 3은 번역 워크플로우의 각 단계에서 품질을 보장하는 표준화된 프로토콜을 정의합니다.

## 품질 보증 워크플로우 개요

### 4단계 QA 프로세스

```
┌─────────────────────────────────────────────────────────┐
│              번역 품질 보증 워크플로우                    │
└─────────────────────────────────────────────────────────┘

1. Pre-Translation QA (번역 전 품질 검증)
   ↓
2. In-Translation QA (번역 중 품질 모니터링)
   ↓
3. Post-Translation QA (번역 후 품질 검증)
   ↓
4. Continuous QA (지속적 품질 개선)
```

### 각 단계별 목적

| 단계 | 목적 | 주요 활동 |
|------|------|----------|
| **Pre-Translation** | 원문 품질 확보 | 원문 검증, 용어 준비 |
| **In-Translation** | 실시간 품질 유지 | 자동 검사, QE 모니터링 |
| **Post-Translation** | 최종 품질 확인 | 인간 검토, 오류 수정 |
| **Continuous** | 장기 품질 향상 | 피드백 수집, 모델 개선 |

```python
def validate_source_quality(source_text, domain="general"):
    """원문 품질 검증"""
    issues = []

    # 인코딩 확인
    try:
        source_text.encode('utf-8')
    except UnicodeEncodeError:
        issues.append({
            "type": "encoding",
            "severity": "critical",
            "message": "UTF-8 인코딩 오류"
        })

    # 문장 완결성 확인
    if not source_text.strip().endswith(('.', '!', '?', '。', '！', '？')):
        issues.append({
            "type": "completeness",
            "severity": "warning",
            "message": "문장이 완결되지 않음"
        })

    # 비정상적 길이 확인
    if len(source_text) > 1000:
        issues.append({
            "type": "length",
            "severity": "warning",
            "message": f"문장이 매우 김 ({len(source_text)} 문자)"
        })

    # 태그 검증
    if '<' in source_text or '>' in source_text:
        if not validate_html_tags(source_text):
            issues.append({
                "type": "formatting",
                "severity": "critical",
                "message": "HTML 태그 불일치"
            })

    return {
        "valid": len([i for i in issues if i["severity"] == "critical"]) == 0,
        "issues": issues
    }
```

```javascript
// SDL Trados Studio 플러그인
class WIAQualityPlugin {
    onSegmentConfirmed(segment) {
        // 세그먼트 품질 검사
        const issues = this.checkSegment(segment);

        if (issues.critical.length > 0) {
            this.showError(
                "치명적 오류가 발견되었습니다",
                issues.critical
            );
            return false; // 저장 차단
        }

        if (issues.warnings.length > 0) {
            this.showWarning(
                "경고 사항이 있습니다",
                issues.warnings
            );
        }

        return true;
    }

    checkSegment(segment) {
        const issues = {
            critical: [],
            warnings: []
        };

        // 태그 검사
        if (!this.tagsMatch(segment.source, segment.target)) {
            issues.critical.push("태그 불일치");
        }

        // 용어 검사
        const termViolations = this.checkTerminology(segment);
        issues.warnings.push(...termViolations);

        // 품질 추정
        const qeScore = this.estimateQuality(segment);
        if (qeScore < 0.6) {
            issues.warnings.push(`품질 점수 낮음: ${qeScore}`);
        }

        return issues;
    }
}
```

```python
def process_mt_output(segment, qe_model):
    """기계 번역 출력 처리"""

    # 품질 추정
    qe_result = qe_model.predict(
        source=segment.source,
        target=segment.target
    )

    hter = qe_result.sentence_level.hter
    da_score = qe_result.sentence_level.da_score

    # 임계값에 따른 라우팅
    if hter <= 0.10 and da_score >= 85:
        return {
            "action": "auto_approve",
            "status": "ready_for_delivery",
            "qe_score": da_score
        }
    elif 0.10 < hter <= 0.30 and 60 <= da_score < 85:
        return {
            "action": "light_post_edit",
            "assigned_to": "post_editor_pool",
            "qe_score": da_score,
            "estimated_effort": "low"
        }
    else:
        return {
            "action": "full_post_edit",
            "assigned_to": "senior_translator",
            "qe_score": da_score,
            "estimated_effort": "high"
        }
```

**게이트 평가 스크립트**:

```python
def evaluate_quality_gate(assessment, gates_config):
    """품질 게이트 평가"""

    overall_score = assessment.overall_quality.score
    critical_errors = count_errors_by_severity(assessment, "critical")
    major_errors = count_errors_by_severity(assessment, "major")
    word_count = assessment.statistics.word_count_source
    major_errors_per_1k = (major_errors / word_count) * 1000

    for gate_name, gate in gates_config.items():
        criteria = gate["criteria"]

        passes_gate = (
            overall_score >= criteria["overall_score"] and
            critical_errors <= criteria["critical_errors"] and
            major_errors_per_1k <= criteria["major_errors_per_1k_words"]
        )

        if passes_gate:
            return {
                "gate": gate_name,
                "passed": True,
                "action": gate["action_on_pass"],
                "scores": {
                    "overall": overall_score,
                    "critical_errors": critical_errors,
                    "major_errors_per_1k": major_errors_per_1k
                }
            }

    # 모든 게이트 실패
    return {
        "gate": "none",
        "passed": False,
        "action": "reject_and_retranslate",
        "scores": {
            "overall": overall_score,
            "critical_errors": critical_errors,
            "major_errors_per_1k": major_errors_per_1k
        }
    }
```

```python
def should_retrain_model(performance_metrics, threshold_config):
    """모델 재학습 필요 여부 판단"""

    triggers = []

    # 품질 하락
    if performance_metrics.comet_score < threshold_config.min_comet:
        triggers.append({
            "reason": "quality_degradation",
            "metric": "comet",
            "current": performance_metrics.comet_score,
            "threshold": threshold_config.min_comet
        })

    # 새로운 데이터 축적
    if performance_metrics.new_segments > threshold_config.min_new_data:
        triggers.append({
            "reason": "sufficient_new_data",
            "segments": performance_metrics.new_segments,
            "threshold": threshold_config.min_new_data
        })

    # 정기 업데이트
    days_since_last_training = (
        datetime.now() - performance_metrics.last_training_date
    ).days

    if days_since_last_training > threshold_config.max_days:
        triggers.append({
            "reason": "scheduled_update",
            "days": days_since_last_training,
            "threshold": threshold_config.max_days
        })

    return {
        "should_retrain": len(triggers) > 0,
        "triggers": triggers,
        "priority": calculate_priority(triggers)
    }
```
