## 8.2 레퍼런스 구현

### 8.2.1 자동 품질 평가 서비스

```python
# src/services/quality_assessment.py
from comet import download_model, load_from_checkpoint
from sacrebleu import corpus_bleu, corpus_chrf
from typing import List, Dict
import numpy as np

class TranslationQualityService:
    """번역 품질 평가 서비스"""

    def __init__(self):
        # COMET 모델 로드
        model_path = download_model("Unbabel/wmt22-comet-da")
        self.comet_model = load_from_checkpoint(model_path)

        # QE 모델 로드 (참조 번역 불필요)
        qe_model_path = download_model("Unbabel/wmt22-cometkiwi-da")
        self.qe_model = load_from_checkpoint(qe_model_path)

    def evaluate_translation(
        self,
        source: str,
        translation: str,
        reference: str = None,
        source_lang: str = None,
        target_lang: str = None
    ) -> QualityAssessment:
        """단일 번역 품질 평가"""

        scores = {}

        # 참조 번역이 있는 경우
        if reference:
            # COMET 점수
            comet_input = [{
                "src": source,
                "mt": translation,
                "ref": reference
            }]
            comet_output = self.comet_model.predict(comet_input, batch_size=1)
            scores['comet'] = comet_output.scores[0]

            # BLEU 점수
            bleu = corpus_bleu([translation], [[reference]])
            scores['bleu'] = bleu.score

            # chrF 점수
            chrf = corpus_chrf([translation], [[reference]])
            scores['chrf'] = chrf.score

        # 품질 추정 (참조 번역 없이)
        qe_input = [{
            "src": source,
            "mt": translation
        }]
        qe_output = self.qe_model.predict(qe_input, batch_size=1)
        scores['qe'] = qe_output.scores[0]

        # 오류 분석
        errors = self.analyze_errors(source, translation, source_lang, target_lang)

        # 종합 점수 계산
        overall_score = self.calculate_overall_score(scores)

        return QualityAssessment(
            overall_score=overall_score,
            metric_scores=scores,
            errors=errors,
            quality_level=self.get_quality_level(overall_score),
            suggestions=self.generate_suggestions(errors)
        )

    def evaluate_batch(
        self,
        translations: List[Dict],
        include_reference: bool = True
    ) -> BatchQualityReport:
        """배치 번역 품질 평가"""

        results = []
        for item in translations:
            result = self.evaluate_translation(
                source=item['source'],
                translation=item['translation'],
                reference=item.get('reference') if include_reference else None,
                source_lang=item.get('source_lang'),
                target_lang=item.get('target_lang')
            )
            results.append(result)

        # 통계 계산
        scores = [r.overall_score for r in results]

        return BatchQualityReport(
            total_segments=len(results),
            average_score=np.mean(scores),
            std_score=np.std(scores),
            min_score=np.min(scores),
            max_score=np.max(scores),
            quality_distribution=self.get_distribution(scores),
            individual_results=results,
            error_summary=self.summarize_errors(results)
        )

    def analyze_errors(
        self,
        source: str,
        translation: str,
        source_lang: str,
        target_lang: str
    ) -> List[TranslationError]:
        """MQM 기반 오류 분석"""
        errors = []

        # 정확성 오류 검사
        accuracy_errors = self.check_accuracy_errors(source, translation)
        errors.extend(accuracy_errors)

        # 유창성 오류 검사
        fluency_errors = self.check_fluency_errors(translation, target_lang)
        errors.extend(fluency_errors)

        # 용어 오류 검사
        terminology_errors = self.check_terminology_errors(source, translation)
        errors.extend(terminology_errors)

        # 스타일 오류 검사
        style_errors = self.check_style_errors(translation, target_lang)
        errors.extend(style_errors)

        return errors

    def check_accuracy_errors(self, source: str, translation: str) -> List[TranslationError]:
        """정확성 오류 검사"""
        errors = []

        # 누락 감지
        missing = self.detect_omissions(source, translation)
        for m in missing:
            errors.append(TranslationError(
                category='accuracy',
                subcategory='omission',
                severity='major',
                source_span=m['source_span'],
                description=f"원문의 '{m['missing_content']}'가 번역에서 누락됨"
            ))

        # 추가 감지
        additions = self.detect_additions(source, translation)
        for a in additions:
            errors.append(TranslationError(
                category='accuracy',
                subcategory='addition',
                severity='minor',
                target_span=a['target_span'],
                description=f"'{a['added_content']}'가 원문에 없이 추가됨"
            ))

        # 오역 감지
        mistranslations = self.detect_mistranslations(source, translation)
        for mt in mistranslations:
            errors.append(TranslationError(
                category='accuracy',
                subcategory='mistranslation',
                severity='major',
                source_span=mt['source_span'],
                target_span=mt['target_span'],
                description=f"'{mt['source_text']}'가 '{mt['translation']}'로 잘못 번역됨"
            ))

        return errors

    def calculate_overall_score(self, scores: Dict[str, float]) -> float:
        """종합 점수 계산"""
        weights = {
            'comet': 0.4,
            'qe': 0.3,
            'bleu': 0.15,
            'chrf': 0.15
        }

        total_weight = 0
        weighted_sum = 0

        for metric, score in scores.items():
            if metric in weights:
                # 점수 정규화 (0-100 스케일)
                normalized = self.normalize_score(score, metric)
                weighted_sum += normalized * weights[metric]
                total_weight += weights[metric]

        return weighted_sum / total_weight if total_weight > 0 else 0

    def get_quality_level(self, score: float) -> str:
        """품질 등급 결정"""
        if score >= 90:
            return 'excellent'
        elif score >= 80:
            return 'good'
        elif score >= 70:
            return 'acceptable'
        elif score >= 60:
            return 'needs_review'
        else:
            return 'poor'
```

        true_positives = 0
        false_positives = 0
        false_negatives = 0

import pytest
import time
import asyncio
from concurrent.futures import ThreadPoolExecutor

class TestPerformance:
    """성능 테스트"""

    @pytest.fixture
    def service(self):

        source = "This is a test sentence for performance measurement."
        translation = "이것은 성능 측정을 위한 테스트 문장입니다."

        latencies = []
        for _ in range(10):
            start = time.time()
            service.evaluate_translation(source, translation)
            latencies.append(time.time() - start)

        avg_latency = sum(latencies) / len(latencies)
        assert avg_latency < 1.0, f"Average latency {avg_latency}s exceeds 1s"

    def test_batch_evaluation_throughput(self, service):
        """배치 평가 처리량 테스트"""
        batch_size = 100
        translations = [
            {
                'source': f"Test sentence {i}",
                'translation': f"테스트 문장 {i}"
            }
            for i in range(batch_size)
        ]

        start = time.time()
        service.evaluate_batch(translations, include_reference=False)
        duration = time.time() - start

        throughput = batch_size / duration
        assert throughput > 10, f"Throughput {throughput} segments/s is too low"

    def test_concurrent_evaluations(self, service):
        """동시 평가 테스트"""
        num_requests = 50

        async def evaluate_async():
            return service.evaluate_translation(
                "Test sentence",
                "테스트 문장"
            )

        async def run_concurrent():
            tasks = [evaluate_async() for _ in range(num_requests)]
            start = time.time()
            results = await asyncio.gather(*tasks)
            duration = time.time() - start
            return results, duration

        results, duration = asyncio.run(run_concurrent())

        assert len(results) == num_requests
        assert duration < 30, f"Concurrent evaluation took {duration}s"
