8.4.2 OAuth 2.0 인증 구현

// JWT 기반 인증
import jwt from 'jsonwebtoken';

const SECRET_KEY = process.env.JWT_SECRET || 'your-secret-key';

export function generateToken(userId: string, scopes: string[]): string {
    return jwt.sign(
        { userId, scopes },
        SECRET_KEY,
        { expiresIn: '1h' }
    );
}

export function authenticate(req, res, next) {
    const authHeader = req.headers.authorization;

    if (!authHeader || !authHeader.startsWith('Bearer ')) {
        return res.status(401).json({
            status: 'error',
            error: { code: 'UNAUTHORIZED', message: 'Missing or invalid token' }
        });
    }

    const token = authHeader.substring(7);

    try {
        const decoded = jwt.verify(token, SECRET_KEY);
        req.user = decoded;
        next();
    } catch (error) {
        return res.status(401).json({
            status: 'error',
            error: { code: 'TOKEN_EXPIRED', message: 'Token has expired' }
        });
    }
}

export function authorize(...requiredScopes: string[]) {
    return (req, res, next) => {
        const userScopes = req.user.scopes || [];

        const hasPermission = requiredScopes.some(scope =>
            userScopes.includes(scope) || userScopes.includes('admin')
        );

        if (!hasPermission) {
            return res.status(403).json({
                status: 'error',
                error: { code: 'FORBIDDEN', message: 'Insufficient permissions' }
            });
        }

        next();
    };
}

8.5.1 MQTT 클라이언트 구현

// Python paho-mqtt 예제
import paho.mqtt.client as mqtt
import json
import ssl
from datetime import datetime

class WIA007MQTTClient:
    def __init__(self, robot_id, farm_id, broker_host):
        self.robot_id = robot_id
        self.farm_id = farm_id
        self.broker_host = broker_host
        self.client = mqtt.Client(client_id=robot_id)

        # 콜백 설정
        self.client.on_connect = self.on_connect
        self.client.on_message = self.on_message
        self.client.on_disconnect = self.on_disconnect

        # TLS 설정
        self.client.tls_set(
            ca_certs="/path/to/ca.crt",
            tls_version=ssl.PROTOCOL_TLSv1_2
        )

        # Last Will 설정
        will_topic = f"wia-rob-007/{farm_id}/robots/{robot_id}/status"
        will_payload = json.dumps({
            "version": "1.0",
            "timestamp": datetime.utcnow().isoformat() + "Z",
            "robot_id": robot_id,
            "operational_state": "disconnected"
        })
        self.client.will_set(will_topic, will_payload, qos=1, retain=True)

    def on_connect(self, client, userdata, flags, rc):
        if rc == 0:
            print(f"✓ Connected to MQTT broker")
            # 명령 토픽 구독
            command_topic = f"wia-rob-007/{self.farm_id}/robots/{self.robot_id}/command"
            client.subscribe(command_topic, qos=2)
        else:
            print(f"✗ Connection failed with code {rc}")

    def on_message(self, client, userdata, msg):
        try:
            command = json.loads(msg.payload.decode())
            print(f"Received command: {command['command_type']}")
            self.handle_command(command)
        except Exception as e:
            print(f"Error processing message: {e}")

    def on_disconnect(self, client, userdata, rc):
        if rc != 0:
            print("Unexpected disconnect. Reconnecting...")
            self.reconnect_with_backoff()

    def connect(self):
        self.client.connect(self.broker_host, 8883, 60)
        self.client.loop_start()

    def publish_status(self, status_data):
        topic = f"wia-rob-007/{self.farm_id}/robots/{self.robot_id}/status"
        payload = json.dumps({
            "version": "1.0",
            "timestamp": datetime.utcnow().isoformat() + "Z",
            "robot_id": self.robot_id,
            **status_data
        })
        self.client.publish(topic, payload, qos=1, retain=True)

    def publish_telemetry(self, sensor_data):
        topic = f"wia-rob-007/{self.farm_id}/robots/{self.robot_id}/telemetry"
        payload = json.dumps({
            "version": "1.0",
            "timestamp": datetime.utcnow().isoformat() + "Z",
            "robot_id": self.robot_id,
            **sensor_data
        })
        self.client.publish(topic, payload, qos=0)

    def handle_command(self, command):
        # 명령 처리 로직
        if command['command_type'] == 'start_task':
            self.start_task(command['parameters'])
        elif command['command_type'] == 'stop_task':
            self.stop_task()

        # 응답 전송
        self.send_response(command['command_id'], "success")

    def send_response(self, command_id, status):
        topic = f"wia-rob-007/{self.farm_id}/robots/{self.robot_id}/response"
        payload = json.dumps({
            "version": "1.0",
            "command_id": command_id,
            "timestamp": datetime.utcnow().isoformat() + "Z",
            "status": status
        })
        self.client.publish(topic, payload, qos=2)

# 사용 예

robot.connect()

# 상태 발행

# 센서 데이터 발행
robot.publish_telemetry({
    "sensor_type": "soil",

        "temperature_celsius": 18.5,
        "moisture_percent": 65.0
    }
