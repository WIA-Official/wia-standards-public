8장: 안전, 제한 및 통합

8.1 안전 중요 모터 제어

모터 제어 시스템은 기계적 부상, 전기적 위험, 재산 피해를 포함한 심각한 안전 위험을 초래할 수 있습니다. 강력한 안전 메커니즘을 구현하는 것은 선택이 아니라 기본 요구사항입니다. 이 장에서는 안전 아키텍처, 제한 적용, 결함 감지, 신뢰할 수 있는 모터 제어를 위한 시스템 통합을 다룹니다.

안전 최우선 원칙

모터 제어 시스템에서 안전 메커니즘은 주 제어 루프와 독립적이어야 합니다. 하드웨어 인터록, 워치독 타이머, 페일세이프 설계는 단일 소프트웨어 고장이 위험한 상황을 초래하지 않도록 보장합니다. 응용 분야에 적용되는 안전 표준(IEC 61508, ISO 13849, IEC 62443)을 항상 따르십시오.

// 종합 안전 제어기
class SafetyController {
private:
    struct SafetyLimits {
        float max_position;
        float min_position;
        float max_velocity;
        float max_acceleration;
        float max_current;
        float max_torque;
        float max_temperature;
    } limits;

    enum SafetyState {
        NORMAL,
        WARNING,
        SAFE_STOP_1,    // 제어된 감속
        SAFE_STOP_2,    // 즉각적 정지
        EMERGENCY_STOP, // 안전 토크 차단
        FAULT
    } current_state;

    struct FaultFlags {
        bool position_limit;
        bool velocity_limit;
        bool current_limit;
        bool temperature_high;
        bool communication_lost;
        bool encoder_fault;
        bool drive_fault;
        bool external_estop;
        bool watchdog_timeout;
    } faults;

    uint32_t last_heartbeat;
    uint32_t watchdog_timeout_ms;

public:
    SafetyController(SafetyConfig& config) {
        limits = config.limits;
        watchdog_timeout_ms = config.watchdog_timeout;
        current_state = NORMAL;
        reset_faults();
    }

    SafetyState check_and_limit(MotorState& state, float& control_output) {
        check_position_limits(state);
        check_velocity_limits(state);
        check_current_limits(state);
        check_temperature(state);
        check_watchdog();
        check_encoder_validity(state);

        update_safety_state();

        switch (current_state) {
            case NORMAL:
                control_output = apply_soft_limits(control_output, state);
                break;

            case WARNING:
                control_output = apply_reduced_limits(control_output, state);
                log_warning();
                break;

            case SAFE_STOP_1:
                control_output = compute_safe_stop_trajectory(state);
                break;

            case SAFE_STOP_2:
                control_output = compute_emergency_braking(state);
                break;

            case EMERGENCY_STOP:
            case FAULT:
                control_output = 0;
                trigger_sto();
                break;
        }

        return current_state;
    }

    void check_current_limits(MotorState& state) {
        if (fabs(state.current) > limits.max_current) {
            faults.current_limit = true;
        }

        // I²t 열 보호
        static float i2t_accumulator = 0;
        float i2t_limit = limits.max_current * limits.max_current * thermal_time_constant;

        i2t_accumulator += state.current * state.current * dt;
        i2t_accumulator -= i2t_accumulator * dt / thermal_time_constant;

        if (i2t_accumulator > i2t_limit) {
            faults.current_limit = true;
        }
    }

    void heartbeat() {
        last_heartbeat = get_system_time_ms();
    }

private:
    float dt = 0.0001f;
    float thermal_time_constant = 30.0f;
};

8.3 제한 적용

// 다중 수준 제한 적용기
class LimitEnforcer {
private:
    struct PositionLimits {
        float hard_min, hard_max;
        float soft_min, soft_max;
        float warning_min, warning_max;
    } pos_limits;

    struct VelocityLimits {
        float max_velocity;
        float max_acceleration;
        float max_jerk;
    } vel_limits;

    struct TorqueLimits {
        float continuous_current;
        float peak_current;
        float peak_duration_ms;
    } torque_limits;

public:
    // 궤적 명령에 모든 제한 적용
    void enforce_trajectory_limits(TrajectoryPoint& cmd, MotorState& state) {
        enforce_position_limits(cmd, state);

        cmd.velocity = clamp(cmd.velocity, -vel_limits.max_velocity, vel_limits.max_velocity);

        // 가속도 제한
        float accel = (cmd.velocity - state.velocity) / dt;
        if (fabs(accel) > vel_limits.max_acceleration) {
            accel = copysign(vel_limits.max_acceleration, accel);
            cmd.velocity = state.velocity + accel * dt;
        }

        // 저크 제한
        static float last_accel = 0;
        float jerk = (accel - last_accel) / dt;
        if (fabs(jerk) > vel_limits.max_jerk) {
            jerk = copysign(vel_limits.max_jerk, jerk);
            accel = last_accel + jerk * dt;
            cmd.velocity = state.velocity + accel * dt;
        }
        last_accel = accel;
    }

    void enforce_position_limits(TrajectoryPoint& cmd, MotorState& state) {
        cmd.position = clamp(cmd.position, pos_limits.hard_min, pos_limits.hard_max);

        // 한계에 접근할 때 속도 감소
        float margin_to_max = pos_limits.soft_max - state.position;
        float margin_to_min = state.position - pos_limits.soft_min;

        float slow_zone = 0.1f * (pos_limits.soft_max - pos_limits.soft_min);

        if (margin_to_max < slow_zone && cmd.velocity > 0) {
            float scale = margin_to_max / slow_zone;
            cmd.velocity = fmin(cmd.velocity, vel_limits.max_velocity * scale);
        }

        if (margin_to_min < slow_zone && cmd.velocity < 0) {
            float scale = margin_to_min / slow_zone;
            cmd.velocity = fmax(cmd.velocity, -vel_limits.max_velocity * scale);
        }
    }

    // 전류/토크 제한 적용
    float enforce_current_limit(float current_cmd, MotorState& state) {
        float limited = clamp(current_cmd, -torque_limits.peak_current, torque_limits.peak_current);

        // 피크 전류 시간 추적
        static uint32_t peak_start_time = 0;

        if (fabs(limited) > torque_limits.continuous_current) {
            if (peak_start_time == 0) {
                peak_start_time = get_time_ms();
            }

            uint32_t peak_duration = get_time_ms() - peak_start_time;
            if (peak_duration > torque_limits.peak_duration_ms) {
                limited = copysign(torque_limits.continuous_current, limited);
            }
        } else {
            peak_start_time = 0;
        }

        // 온도 기반 디레이팅
        float temp_factor = compute_temperature_derating(state.temperature);
        limited *= temp_factor;

        return limited;
    }

    float compute_temperature_derating(float temperature) {
        float temp_threshold = 80.0f;
        float temp_max = 120.0f;
        float min_derating = 0.3f;

        if (temperature < temp_threshold) {
            return 1.0f;
        } else if (temperature > temp_max) {
            return min_derating;
        } else {
            float ratio = (temperature - temp_threshold) / (temp_max - temp_threshold);
            return 1.0f - ratio * (1.0f - min_derating);
        }
    }

private:
    float dt = 0.0001f;
};

8.4 결함 감지 및 진단

// 결함 감지 시스템
class FaultDetector {
private:
    struct Thresholds {
        float position_error_max;
        float velocity_error_max;
        float encoder_velocity_mismatch;
        float motor_temperature_max;
    } thresh;

    struct FaultCounters {
        int position_error;
        int velocity_error;
        int encoder_fault;
        int overcurrent;
    } counters;

    int fault_threshold;

public:
    FaultDetector(FaultConfig& config) {
        thresh = config.thresholds;
        fault_threshold = config.fault_threshold;
        reset_counters();
    }

    FaultCode detect(MotorState& state, ControlState& ctrl) {
        FaultCode fault = FAULT_NONE;

        // 추종 오차 감지
        if (fabs(ctrl.position_error) > thresh.position_error_max) {
            if (++counters.position_error >= fault_threshold) {
                fault = FAULT_FOLLOWING_ERROR;
            }
        } else {
            counters.position_error = 0;
        }

        // 엔코더 결함 감지
        fault = check_encoder(state);
        if (fault != FAULT_NONE) return fault;

        // 과전류 감지
        if (fabs(state.current) > state.current_limit * 1.2f) {
            if (++counters.overcurrent >= fault_threshold) {
                fault = FAULT_OVERCURRENT;
            }
        } else {
            counters.overcurrent = 0;
        }

        // 모터 정지 감지
        fault = check_stall(state, ctrl);
        if (fault != FAULT_NONE) return fault;

        // 열 결함
        if (state.motor_temperature > thresh.motor_temperature_max) {
            fault = FAULT_OVERTEMPERATURE;
        }

        return fault;
    }

    FaultCode check_encoder(MotorState& state) {
        static float last_position = 0;
        static uint32_t stuck_count = 0;

        // 전류가 흐르는데 위치 변화 없음
        if (fabs(state.position - last_position) < 0.0001f &&
            fabs(state.current) > 0.5f) {
            stuck_count++;
            if (stuck_count > 1000) {
                return FAULT_ENCODER_STUCK;
            }
        } else {
            stuck_count = 0;
        }

        // 엔코더 카운트 점프 검사
        float expected_change = state.velocity * dt;
        float actual_change = state.position - last_position;

        if (fabs(actual_change - expected_change) > thresh.encoder_velocity_mismatch) {
            if (++counters.encoder_fault >= fault_threshold) {
                return FAULT_ENCODER_SIGNAL;
            }
        } else {
            counters.encoder_fault = 0;
        }

        last_position = state.position;
        return FAULT_NONE;
    }

    FaultCode check_stall(MotorState& state, ControlState& ctrl) {
        static uint32_t stall_count = 0;

        bool high_current = fabs(state.current) > 0.8f * state.current_limit;
        bool low_velocity = fabs(state.velocity) < 0.01f * ctrl.velocity_reference;
        bool commanding_motion = fabs(ctrl.velocity_reference) > 0.1f;

        if (high_current && low_velocity && commanding_motion) {
            stall_count++;
            if (stall_count > 5000) {
                return FAULT_MOTOR_STALL;
            }
        } else {
            stall_count = 0;
        }

        return FAULT_NONE;
    }

private:
    float dt = 0.0001f;
};

enum FaultCode {
    FAULT_NONE = 0,
    FAULT_FOLLOWING_ERROR,
    FAULT_OVERCURRENT,
    FAULT_OVERTEMPERATURE,
    FAULT_ENCODER_STUCK,
    FAULT_ENCODER_SIGNAL,
    FAULT_MOTOR_STALL,
    FAULT_COMMUNICATION,
    FAULT_EXTERNAL_ESTOP
};

8.5 안전 토크 차단 (STO) 구현

// 안전 토크 차단 기능
class SafeTorqueOff {
private:
    GPIO_Pin sto_channel_1;
    GPIO_Pin sto_channel_2;

    bool channel_1_active;
    bool channel_2_active;
    bool sto_engaged;

public:
    SafeTorqueOff(STOConfig& config) {
        sto_channel_1 = config.channel_1_pin;
        sto_channel_2 = config.channel_2_pin;
        engage();  // 안전 상태로 초기화
    }

    void engage() {
        // 두 STO 채널 개방 (페일세이프: 개방 = 안전)
        gpio_write(sto_channel_1, LOW);
        gpio_write(sto_channel_2, LOW);

        channel_1_active = false;
        channel_2_active = false;
        sto_engaged = true;
    }

    bool release() {
        if (!can_release()) {
            return false;
        }

        gpio_write(sto_channel_1, HIGH);
        gpio_write(sto_channel_2, HIGH);

        delay_us(100);  // 릴레이 안정화 대기

        channel_1_active = gpio_read(sto_channel_1);
        channel_2_active = gpio_read(sto_channel_2);

        if (channel_1_active && channel_2_active) {
            sto_engaged = false;
            return true;
        } else {
            engage();  // 채널 불일치 - 안전 유지
            log_fault("STO 채널 불일치");
            return false;
        }
    }

    bool can_release() {
        return !has_active_faults() &&
               !estop_pressed() &&
               initialization_complete() &&
               reset_acknowledged();
    }

    void periodic_check() {
        bool ch1_state = gpio_read(sto_channel_1);
        bool ch2_state = gpio_read(sto_channel_2);

        if (sto_engaged) {
            if (ch1_state || ch2_state) {
                log_fault("STO 채널이 닫힌 상태로 고착");
                trigger_fault();
            }
        } else {
            if (!ch1_state || !ch2_state) {
                log_fault("STO 채널이 예기치 않게 열림");
                engage();
            }
        }
    }

    bool is_engaged() { return sto_engaged; }
};

8.7 시스템 통합

// 완전한 모터 제어 시스템 통합
class MotorControlSystem {
private:
    // 핵심 구성요소
    MotorDriver driver;
    EncoderInterface encoder;
    CurrentSensor current_sensor;

    // 제어 구성요소
    PositionController position_ctrl;
    VelocityController velocity_ctrl;
    CurrentController current_ctrl;
    TrajectoryGenerator trajectory;

    // 안전 구성요소
    SafetyController safety;
    LimitEnforcer limits;
    FaultDetector fault_detector;
    SafeTorqueOff sto;

    // 통신
    EtherCATInterface ethercat;

    // 상태
    MotorState state;
    ControlMode mode;
    SystemStatus status;

public:
    MotorControlSystem(SystemConfig& config) :
        driver(config.driver),
        encoder(config.encoder),
        position_ctrl(config.position_ctrl),
        velocity_ctrl(config.velocity_ctrl),
        current_ctrl(config.current_ctrl),
        safety(config.safety),
        limits(config.limits),
        fault_detector(config.fault_detector),
        sto(config.sto)
    {
        mode = MODE_DISABLED;
        status = STATUS_INITIALIZING;
    }

    void initialize() {
        status = STATUS_INITIALIZING;

        driver.initialize();
        encoder.initialize();
        current_sensor.calibrate();

        if (encoder.needs_homing()) {
            perform_homing_sequence();
        }

        validate_parameters();
        safety_system_check();

        status = STATUS_READY;
    }

    void control_loop() {
        // 1. 센서 읽기
        update_state();

        // 2. 안전 검사 (항상 먼저 실행)
        safety.heartbeat();
        FaultCode fault = fault_detector.detect(state, ctrl_state);

        if (fault != FAULT_NONE) {
            handle_fault(fault);
            return;
        }

        SafetyState safety_state = safety.check_and_limit(state, control_output);

        if (safety_state != NORMAL && safety_state != WARNING) {
            apply_output(control_output);
            return;
        }

        // 3. 명령 생성/수신
        TrajectoryPoint cmd;
        if (mode == MODE_TRAJECTORY) {
            cmd = trajectory.get_next_point();
        } else {
            cmd = receive_external_command();
        }

        // 4. 제한 적용
        limits.enforce_trajectory_limits(cmd, state);

        // 5. 제어 계산
        float output;
        switch (mode) {
            case MODE_POSITION:
                output = position_ctrl.compute(cmd.position, state.position, state.velocity);
                output = velocity_ctrl.compute(output, state.velocity);
                output = current_ctrl.compute(output, state.current);
                break;

            case MODE_VELOCITY:
                output = velocity_ctrl.compute(cmd.velocity, state.velocity);
                output = current_ctrl.compute(output, state.current);
                break;

            case MODE_TORQUE:
                output = current_ctrl.compute(cmd.torque / Kt, state.current);
                break;

            case MODE_DISABLED:
                output = 0;
                break;
        }

        // 6. 전류 제한
        output = limits.enforce_current_limit(output, state);

        // 7. 출력 적용
        apply_output(output);

        // 8. 텔레메트리 업데이트
        update_telemetry();
    }

    void handle_fault(FaultCode fault) {
        status = STATUS_FAULT;

        switch (fault) {
            case FAULT_FOLLOWING_ERROR:
            case FAULT_MOTOR_STALL:
                safety.trigger_safe_stop_1();
                break;

            case FAULT_OVERCURRENT:
            case FAULT_OVERTEMPERATURE:
            case FAULT_ENCODER_STUCK:
                safety.trigger_safe_stop_2();
                break;

            case FAULT_ENCODER_SIGNAL:
            case FAULT_EXTERNAL_ESTOP:
                sto.engage();
                break;
        }

        log_fault(fault);
        notify_host(fault);
    }

    bool enable() {
        if (status != STATUS_READY && status != STATUS_DISABLED) {
            return false;
        }

        if (!sto.release()) {
            return false;
        }

        mode = MODE_POSITION;
        status = STATUS_ENABLED;
        return true;
    }

    void disable() {
        mode = MODE_DISABLED;
        sto.engage();
        status = STATUS_DISABLED;
    }

private:
    ControlState ctrl_state;
    float control_output;
    float Kt;
};

// 실시간 모터 제어를 위한 EtherCAT 통합
class EtherCATMotorDrive {
private:
    enum ECState { INIT, PREOP, SAFEOP, OP } ec_state;

    struct TxPDO {
        int32_t actual_position;
        int32_t actual_velocity;
        int16_t actual_torque;
        uint16_t status_word;
    } tx_pdo;

    struct RxPDO {
        int32_t target_position;
        int32_t target_velocity;
        int16_t target_torque;
        uint16_t control_word;
    } rx_pdo;

    enum DriveState {
        NOT_READY_TO_SWITCH_ON,
        SWITCH_ON_DISABLED,
        READY_TO_SWITCH_ON,
        SWITCHED_ON,
        OPERATION_ENABLED,
        QUICK_STOP_ACTIVE,
        FAULT
    } drive_state;

public:
    void process_pdo() {
        tx_pdo.status_word = build_status_word();
        process_control_word(rx_pdo.control_word);

        tx_pdo.actual_position = motor_system.get_position_counts();
        tx_pdo.actual_velocity = motor_system.get_velocity_counts_per_sec();
        tx_pdo.actual_torque = motor_system.get_torque_permille();

        if (drive_state == OPERATION_ENABLED) {
            apply_setpoints();
        }
    }

    void process_control_word(uint16_t cw) {
        // CiA 402 상태 기계 전환
        bool switch_on = cw & 0x0001;
        bool enable_voltage = cw & 0x0002;
        bool quick_stop = cw & 0x0004;
        bool enable_operation = cw & 0x0008;
        bool fault_reset = cw & 0x0080;

        switch (drive_state) {
            case SWITCH_ON_DISABLED:
                if (switch_on && enable_voltage && quick_stop) {
                    drive_state = READY_TO_SWITCH_ON;
                }
                break;

            case READY_TO_SWITCH_ON:
                if (switch_on && enable_voltage && quick_stop) {
                    drive_state = SWITCHED_ON;
                }
                break;

            case SWITCHED_ON:
                if (enable_operation) {
                    drive_state = OPERATION_ENABLED;
                    motor_system.enable();
                }
                break;

            case OPERATION_ENABLED:
                if (!enable_operation) {
                    drive_state = SWITCHED_ON;
                    motor_system.disable();
                }
                break;

            case FAULT:
                if (fault_reset) {
                    motor_system.reset_faults();
                    drive_state = SWITCH_ON_DISABLED;
                }
                break;
        }
    }

private:
    MotorControlSystem motor_system;
};

8.9 테스트 및 검증

// 모터 제어 시스템 테스트 프레임워크
class MotorControlTester {
public:
    struct TestResult {
        string test_name;
        bool passed;
        float measured_value;
        float expected_value;
        string failure_reason;
    };

    vector run_all_tests(MotorControlSystem& system) {
        vector results;

        // 안전 테스트 (먼저 통과해야 함)
        results.push_back(test_sto_functionality(system));
        results.push_back(test_position_limits(system));
        results.push_back(test_velocity_limits(system));
        results.push_back(test_current_limits(system));
        results.push_back(test_emergency_stop(system));

        // 기능 테스트
        results.push_back(test_step_response(system));
        results.push_back(test_following_error(system));
        results.push_back(test_trajectory_tracking(system));

        // 성능 테스트
        results.push_back(test_bandwidth(system));
        results.push_back(test_settling_time(system));

        return results;
    }

    TestResult test_sto_functionality(MotorControlSystem& system) {
        TestResult result;
        result.test_name = "STO 기능";

        system.trigger_sto();
        delay_ms(10);

        float output = system.get_output();
        if (fabs(output) > 0.01f) {
            result.passed = false;
            result.failure_reason = "STO 활성화 시 출력이 0이 아님";
            return result;
        }

        // 결함이 활성화된 상태에서 STO 해제 불가 확인
        system.inject_fault(FAULT_OVERCURRENT);
        bool released = system.release_sto();
        if (released) {
            result.passed = false;
            result.failure_reason = "활성 결함이 있는 상태에서 STO 해제됨";
            return result;
        }

        system.clear_faults();
        released = system.release_sto();
        if (!released) {
            result.passed = false;
            result.failure_reason = "결함 해제 후 STO 해제 안됨";
            return result;
        }

        result.passed = true;
        return result;
    }

    TestResult test_step_response(MotorControlSystem& system) {
        TestResult result;
        result.test_name = "스텝 응답";

        float step_size = 1.0f;
        system.set_position_command(step_size);

        float rise_time = 0, overshoot = 0, settling_time = 0;
        // ... 응답 측정 ...

        result.passed = (rise_time < 0.1f) && (overshoot < 10.0f) && (settling_time < 0.3f);
        result.measured_value = settling_time;
        result.expected_value = 0.3f;

        return result;
    }
};

8.10 모범 사례 요약

  • 다중 방어: 여러 독립적인 안전 계층 구현
  • 페일세이프 설계: 모든 고장 시 안전 상태로 기본값 설정
  • 하드웨어 독립성: 안전 중요 기능은 소프트웨어에만 의존하지 않음
  • 워치독 타이머: 제어 루프 실행 및 통신 모니터링
  • 제한 적용: 여러 수준에서 제한 적용 (궤적, 속도, 전류)
  • 결함 감지: 추종 오차, 엔코더 결함, 과전류, 정지 모니터링
  • 안전 정지 기능: 제어된 감속과 비상 정지 구현
  • 테스트: 배포 전 모든 안전 기능 검증
  • 문서화: 완전한 안전 분석 및 테스트 기록 유지
  • 표준 준수: 적용 가능한 기능 안전 표준 준수

핵심 내용

  • 안전은 모터 제어 시스템에서 가장 높은 우선순위이며 처음부터 설계에 포함되어야 합니다
  • 다중 계층 안전 아키텍처는 고장에 대한 다중 방어를 제공합니다
  • 하드웨어 안전 메커니즘(STO, 워치독, 인터록)은 소프트웨어와 독립적이어야 합니다
  • 제한 적용은 여러 수준에서 작동합니다: 위치, 속도, 가속도, 전류
  • 결함 감지는 추종 오차, 엔코더 유효성, 과전류, 모터 정지를 모니터링합니다
  • 안전 토크 차단(STO)은 페일세이프여야 하는 중요한 안전 기능입니다
  • 기능 안전 표준(IEC 61508, ISO 13849)은 안전 설계를 위한 프레임워크를 제공합니다
  • 시스템 통합에는 제어, 안전, 통신의 신중한 조정이 필요합니다
  • 종합적인 테스트는 안전 기능과 제어 성능을 검증합니다
