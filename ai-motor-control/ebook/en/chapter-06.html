Chapter 6: AI-Enhanced Control Methods

6.1 Introduction to AI in Motor Control

Artificial intelligence transforms motor control by enabling systems to learn, adapt, and optimize beyond what classical control can achieve. While PID and cascaded control work well for known, linear systems, real-world motors face nonlinearities, parameter variations, and disturbances that limit classical controller performance.

AI-enhanced motor control addresses these challenges through:

  • Learning: Automatically discovering optimal control strategies from data
  • Adaptation: Adjusting to changing motor parameters and operating conditions
  • Prediction: Anticipating disturbances and failures before they occur
  • Optimization: Finding globally optimal solutions for complex objectives
  • Generalization: Transferring learned knowledge to similar systems

6.3 Neural Network Controllers

Neural networks can learn complex control mappings that capture nonlinear motor dynamics:

// Neural Network Motor Controller
class NeuralController {
private:
    // Network architecture
    int input_size;
    int hidden_size;
    int output_size;

    // Weight matrices
    Matrix W1, b1;  // Input → Hidden
    Matrix W2, b2;  // Hidden → Output

    // Learning parameters
    float learning_rate;
    bool online_learning;

public:
    NeuralController(int inputs, int hidden, int outputs) {
        input_size = inputs;
        hidden_size = hidden;
        output_size = outputs;

        // Initialize weights (Xavier initialization)
        W1 = Matrix::random(hidden, inputs) * sqrt(2.0f / inputs);
        b1 = Matrix::zeros(hidden, 1);
        W2 = Matrix::random(outputs, hidden) * sqrt(2.0f / hidden);
        b2 = Matrix::zeros(outputs, 1);

        learning_rate = 0.001f;
        online_learning = false;
    }

    // Forward pass: compute control output
    float forward(vector& inputs) {
        Matrix x = Matrix::from_vector(inputs);

        // Hidden layer with ReLU activation
        Matrix h = (W1 * x + b1).apply(relu);

        // Output layer (linear for control)
        Matrix y = W2 * h + b2;

        return y(0, 0);
    }

    // Full forward pass with intermediate values for learning
    void forward_full(vector& inputs, Matrix& hidden_out, float& output) {
        Matrix x = Matrix::from_vector(inputs);

        hidden_out = (W1 * x + b1).apply(relu);
        Matrix y = W2 * hidden_out + b2;

        output = y(0, 0);
    }

    // Online learning with backpropagation
    void learn(vector& inputs, float target, float actual) {
        if (!online_learning) return;

        float error = target - actual;

        // Forward pass
        Matrix hidden;
        float output;
        forward_full(inputs, hidden, output);

        Matrix x = Matrix::from_vector(inputs);

        // Output layer gradient
        Matrix d_output = Matrix(1, 1);
        d_output(0, 0) = error;

        // Backpropagate through output layer
        Matrix d_W2 = d_output * hidden.transpose();
        Matrix d_hidden = W2.transpose() * d_output;

        // Apply ReLU derivative
        for (int i = 0; i < hidden_size; i++) {
            if (hidden(i, 0) <= 0) d_hidden(i, 0) = 0;
        }

        // Backpropagate through hidden layer
        Matrix d_W1 = d_hidden * x.transpose();

        // Update weights
        W2 += learning_rate * d_W2;
        b2 += learning_rate * d_output;
        W1 += learning_rate * d_W1;
        b1 += learning_rate * d_hidden;
    }

private:
    static float relu(float x) { return (x > 0) ? x : 0; }
};

Neural Network as Nonlinearity Compensator

// PID + Neural Network Compensation
class NeuralCompensatedPID {
private:
    PIDController pid;
    NeuralController nn_compensator;

    // Compensation blend factor
    float alpha;  // 0 = pure PID, 1 = pure NN

public:
    NeuralCompensatedPID(PIDParams pid_params, NNParams nn_params)
        : pid(pid_params.kp, pid_params.ki, pid_params.kd, pid_params.dt),
          nn_compensator(nn_params.inputs, nn_params.hidden, 1)
    {
        alpha = 0.3f;  // Start with mostly PID
    }

    float compute(float setpoint, float feedback, vector& state) {
        // Classical PID output
        float pid_output = pid.compute(setpoint, feedback);

        // Neural network compensation
        // Input: [error, error_derivative, velocity, acceleration, temperature, ...]
        float nn_output = nn_compensator.forward(state);

        // Blended output
        float output = (1.0f - alpha) * pid_output + alpha * nn_output;

        // Online learning: use PID as teacher initially
        if (nn_compensator.is_learning()) {
            nn_compensator.learn(state, pid_output, nn_output);
        }

        return output;
    }

    // Gradually increase NN contribution as it learns
    void update_blend(float performance_metric) {
        if (performance_metric < 0.1f) {  // Good performance
            alpha = min(alpha + 0.01f, 0.9f);
        }
    }
};

6.4 Reinforcement Learning for Motor Control

Reinforcement learning (RL) learns optimal control policies through trial and error:

// Deep Q-Network (DQN) for Discrete Motor Control
class DQNMotorController {
private:
    // Q-network (neural network)
    NeuralController q_network;
    NeuralController target_network;

    // Replay buffer
    deque replay_buffer;
    int buffer_size;
    int batch_size;

    // Learning parameters
    float gamma;            // Discount factor
    float epsilon;          // Exploration rate
    float epsilon_decay;
    float epsilon_min;
    int update_target_freq;
    int step_count;

    // Action discretization
    int num_actions;
    float action_min, action_max;

public:
    struct Experience {
        vector state;
        int action;
        float reward;
        vector next_state;
        bool done;
    };

    DQNMotorController(int state_dim, int n_actions) :
        q_network(state_dim, 64, n_actions),
        target_network(state_dim, 64, n_actions)
    {
        num_actions = n_actions;
        buffer_size = 10000;
        batch_size = 32;
        gamma = 0.99f;
        epsilon = 1.0f;
        epsilon_decay = 0.995f;
        epsilon_min = 0.01f;
        update_target_freq = 100;
        step_count = 0;
    }

    int select_action(vector& state) {
        // Epsilon-greedy exploration
        if (random_float() < epsilon) {
            return random_int(0, num_actions);
        }

        // Greedy action from Q-network
        vector q_values = q_network.forward_all(state);
        return argmax(q_values);
    }

    float action_to_control(int action) {
        // Convert discrete action to continuous control signal
        return action_min + (action_max - action_min) * action / (num_actions - 1);
    }

    void store_experience(Experience& exp) {
        replay_buffer.push_back(exp);
        if (replay_buffer.size() > buffer_size) {
            replay_buffer.pop_front();
        }
    }

    void train() {
        if (replay_buffer.size() < batch_size) return;

        // Sample random batch
        vector batch = sample_batch(batch_size);

        for (auto& exp : batch) {
            float target;

            if (exp.done) {
                target = exp.reward;
            } else {
                // Q-learning target: r + gamma * max Q(s', a')
                vector next_q = target_network.forward_all(exp.next_state);
                target = exp.reward + gamma * max_element(next_q);
            }

            // Update Q-network
            q_network.train_single(exp.state, exp.action, target);
        }

        // Decay exploration
        epsilon = max(epsilon * epsilon_decay, epsilon_min);

        // Update target network periodically
        step_count++;
        if (step_count % update_target_freq == 0) {
            target_network.copy_weights_from(q_network);
        }
    }

private:
    vector sample_batch(int size);
    float random_float();
    int random_int(int min, int max);
    int argmax(vector& v);
    float max_element(vector& v);
};

Continuous Action Space with Actor-Critic

// Twin Delayed DDPG (TD3) for Continuous Motor Control
class TD3MotorController {
private:
    // Actor network (policy)
    NeuralController actor;
    NeuralController actor_target;

    // Twin critics
    NeuralController critic1, critic2;
    NeuralController critic1_target, critic2_target;

    // Hyperparameters
    float gamma;
    float tau;              // Soft update rate
    float policy_noise;
    float noise_clip;
    int policy_delay;
    int total_steps;

public:
    TD3MotorController(int state_dim, int action_dim) :
        actor(state_dim, 256, action_dim),
        actor_target(state_dim, 256, action_dim),
        critic1(state_dim + action_dim, 256, 1),
        critic2(state_dim + action_dim, 256, 1),
        critic1_target(state_dim + action_dim, 256, 1),
        critic2_target(state_dim + action_dim, 256, 1)
    {
        gamma = 0.99f;
        tau = 0.005f;
        policy_noise = 0.2f;
        noise_clip = 0.5f;
        policy_delay = 2;
        total_steps = 0;

        // Copy initial weights to targets
        actor_target.copy_weights_from(actor);
        critic1_target.copy_weights_from(critic1);
        critic2_target.copy_weights_from(critic2);
    }

    float select_action(vector& state, bool add_noise = true) {
        float action = actor.forward(state);

        if (add_noise) {
            action += random_gaussian() * 0.1f;
            action = clamp(action, -1.0f, 1.0f);
        }

        return action;
    }

    void update(vector& batch) {
        total_steps++;

        // Compute target actions with noise
        vector target_actions;
        for (auto& exp : batch) {
            float noise = clamp(random_gaussian() * policy_noise,
                               -noise_clip, noise_clip);
            float target_action = actor_target.forward(exp.next_state) + noise;
            target_action = clamp(target_action, -1.0f, 1.0f);
            target_actions.push_back(target_action);
        }

        // Compute target Q values
        vector target_q_values;
        for (int i = 0; i < batch.size(); i++) {
            vector critic_input = concat(batch[i].next_state, {target_actions[i]});
            float q1 = critic1_target.forward(critic_input);
            float q2 = critic2_target.forward(critic_input);
            float min_q = min(q1, q2);

            float target_q = batch[i].reward + gamma * min_q * (1 - batch[i].done);
            target_q_values.push_back(target_q);
        }

        // Update critics
        update_critics(batch, target_q_values);

        // Delayed policy update
        if (total_steps % policy_delay == 0) {
            update_actor(batch);
            soft_update();
        }
    }

private:
    void update_critics(vector& batch, vector& targets);
    void update_actor(vector& batch);
    void soft_update() {
        actor_target.soft_update(actor, tau);
        critic1_target.soft_update(critic1, tau);
        critic2_target.soft_update(critic2, tau);
    }
    float random_gaussian();
    vector concat(vector& a, vector b);
};

6.5 Reward Function Design for Motor Control

// Motor Control Reward Functions
class MotorRewardFunction {
public:
    struct State {
        float position;
        float velocity;
        float current;
        float position_error;
        float velocity_error;
    };

    // Position tracking reward
    static float position_tracking(State& s, float action, float prev_action) {
        // Primary: minimize position error
        float pos_reward = -fabs(s.position_error);

        // Secondary: smooth control (minimize jerk)
        float smoothness_reward = -fabs(action - prev_action) * 0.1f;

        // Tertiary: energy efficiency
        float energy_reward = -s.current * s.current * 0.01f;

        // Bonus for reaching target
        float target_bonus = (fabs(s.position_error) < 0.01f) ? 1.0f : 0.0f;

        return pos_reward + smoothness_reward + energy_reward + target_bonus;
    }

    // Velocity tracking reward
    static float velocity_tracking(State& s, float action) {
        float vel_reward = -fabs(s.velocity_error);
        float current_penalty = -s.current * s.current * 0.01f;

        return vel_reward + current_penalty;
    }

    // Energy-optimal trajectory following
    static float energy_optimal(State& s, float action, float dt) {
        // Track reference with minimum energy
        float tracking_weight = 10.0f;
        float energy_weight = 1.0f;

        float tracking_cost = -tracking_weight * s.position_error * s.position_error;
        float energy_cost = -energy_weight * s.current * s.current * dt;

        return tracking_cost + energy_cost;
    }

    // Safe exploration reward
    static float safe_exploration(State& s, float action, SafetyLimits& limits) {
        float reward = 0;

        // Penalty for approaching limits
        float pos_margin = min(limits.max_pos - s.position,
                               s.position - limits.min_pos);
        if (pos_margin < limits.safety_margin) {
            reward -= (limits.safety_margin - pos_margin) * 10.0f;
        }

        float vel_margin = limits.max_vel - fabs(s.velocity);
        if (vel_margin < limits.safety_margin) {
            reward -= (limits.safety_margin - vel_margin) * 10.0f;
        }

        return reward;
    }
};

6.6 Auto-Tuning with AI

// Bayesian Optimization for PID Tuning
class BayesianPIDTuner {
private:
    // Gaussian Process model
    GaussianProcess gp;

    // Search space bounds
    float kp_min, kp_max;
    float ki_min, ki_max;
    float kd_min, kd_max;

    // History
    vector> evaluated_points;
    vector observed_performances;

    // Acquisition function
    float beta;  // Exploration-exploitation trade-off

public:
    BayesianPIDTuner(TunerConfig config) {
        kp_min = config.kp_range[0]; kp_max = config.kp_range[1];
        ki_min = config.ki_range[0]; ki_max = config.ki_range[1];
        kd_min = config.kd_range[0]; kd_max = config.kd_range[1];

        beta = 2.0f;  // UCB exploration parameter

        // Initialize GP with default kernel
        gp = GaussianProcess(SquaredExponentialKernel(1.0f, 0.1f));
    }

    vector suggest_next() {
        // Optimize acquisition function to find next point
        float best_ucb = -INFINITY;
        vector best_point(3);

        // Grid search over acquisition function (could use more sophisticated optimization)
        for (float kp = kp_min; kp <= kp_max; kp += (kp_max - kp_min) / 20) {
            for (float ki = ki_min; ki <= ki_max; ki += (ki_max - ki_min) / 20) {
                for (float kd = kd_min; kd <= kd_max; kd += (kd_max - kd_min) / 20) {
                    vector point = {kp, ki, kd};

                    // Upper Confidence Bound (UCB) acquisition
                    float mean, std;
                    gp.predict(point, mean, std);
                    float ucb = mean + beta * std;

                    if (ucb > best_ucb) {
                        best_ucb = ucb;
                        best_point = point;
                    }
                }
            }
        }

        return best_point;
    }

    void update(vector& gains, float performance) {
        evaluated_points.push_back(gains);
        observed_performances.push_back(performance);

        // Update GP model
        gp.fit(evaluated_points, observed_performances);
    }

    vector get_best() {
        int best_idx = 0;
        float best_perf = observed_performances[0];
        for (int i = 1; i < observed_performances.size(); i++) {
            if (observed_performances[i] > best_perf) {
                best_perf = observed_performances[i];
                best_idx = i;
            }
        }
        return evaluated_points[best_idx];
    }

    // Evaluate performance of PID gains
    static float evaluate_gains(ServoController& servo, vector& gains) {
        servo.set_pid_gains(gains[0], gains[1], gains[2]);

        // Run step response test
        float total_error = 0;
        float overshoot = 0;
        float settling_time = 0;

        for (int i = 0; i < 1000; i++) {
            servo.step();
            float error = servo.get_position_error();
            total_error += error * error;
            // ... measure overshoot, settling time
        }

        // Convert to performance score (higher is better)
        float ise = total_error / 1000.0f;
        return -ise - 0.1f * overshoot - 0.01f * settling_time;
    }
};

// Motor Health Monitoring with Machine Learning
class MotorHealthMonitor {
private:
    // Feature extraction
    struct Features {
        float rms_current;
        float current_thd;      // Total harmonic distortion
        float temperature;
        float vibration_rms;
        float position_error_std;
        float torque_ripple;
    };

    // Anomaly detection model
    IsolationForest anomaly_detector;

    // Remaining useful life prediction
    NeuralController rul_predictor;

    // Historical data
    deque feature_history;
    int history_size;

    // Thresholds
    float anomaly_threshold;
    float rul_warning_hours;

public:
    MotorHealthMonitor(HealthMonitorConfig config) :
        anomaly_detector(100, 256),  // 100 trees, 256 samples
        rul_predictor(10, 32, 1)     // 10 features, 32 hidden, 1 output (RUL)
    {
        history_size = 1000;
        anomaly_threshold = config.anomaly_threshold;
        rul_warning_hours = config.rul_warning;

        // Load pre-trained models
        anomaly_detector.load(config.anomaly_model_path);
        rul_predictor.load(config.rul_model_path);
    }

    Features extract_features(SensorData& data) {
        Features f;

        // Current analysis
        f.rms_current = compute_rms(data.current_samples);
        f.current_thd = compute_thd(data.current_samples);

        // Thermal
        f.temperature = data.winding_temperature;

        // Vibration
        f.vibration_rms = compute_rms(data.vibration_samples);

        // Control performance
        f.position_error_std = compute_std(data.position_errors);
        f.torque_ripple = compute_ripple(data.torque_estimates);

        return f;
    }

    HealthStatus assess(SensorData& data) {
        Features features = extract_features(data);
        feature_history.push_back(features);
        if (feature_history.size() > history_size) {
            feature_history.pop_front();
        }

        HealthStatus status;

        // Anomaly detection
        vector feature_vector = features_to_vector(features);
        float anomaly_score = anomaly_detector.score(feature_vector);
        status.is_anomaly = (anomaly_score > anomaly_threshold);
        status.anomaly_score = anomaly_score;

        // Remaining useful life prediction
        vector trend_features = compute_trend_features();
        float rul_hours = rul_predictor.forward(trend_features);
        status.remaining_life_hours = rul_hours;
        status.needs_maintenance = (rul_hours < rul_warning_hours);

        // Specific fault diagnosis
        status.fault_type = diagnose_fault(features);

        return status;
    }

    struct HealthStatus {
        bool is_anomaly;
        float anomaly_score;
        float remaining_life_hours;
        bool needs_maintenance;
        string fault_type;
    };

private:
    float compute_rms(vector& samples);
    float compute_thd(vector& samples);
    float compute_std(vector& samples);
    float compute_ripple(vector& samples);
    vector features_to_vector(Features& f);
    vector compute_trend_features();
    string diagnose_fault(Features& f);
};

6.8 Neural Network for Friction Compensation

// Learned Friction Model
class NeuralFrictionCompensator {
private:
    // Network learns friction as function of velocity, position, temperature
    NeuralController friction_nn;

    // Training data collection
    vector training_data;
    bool collecting_data;

    // Physical model baseline (LuGre friction)
    float sigma0, sigma1, sigma2;  // LuGre parameters
    float Fc, Fs;                  // Coulomb and static friction
    float vs;                      // Stribeck velocity

public:
    struct TrainingPoint {
        float velocity;
        float position;
        float temperature;
        float measured_friction;
    };

    NeuralFrictionCompensator() :
        friction_nn(3, 16, 1)  // 3 inputs, 16 hidden, 1 output
    {
        // Initialize physical model parameters (typical values)
        sigma0 = 1e5f;
        sigma1 = 100.0f;
        sigma2 = 0.1f;
        Fc = 0.5f;
        Fs = 1.0f;
        vs = 0.01f;
    }

    // Physics-informed neural network output
    float predict(float velocity, float position, float temperature) {
        // Physical model prediction (LuGre simplified)
        float g_v = Fc + (Fs - Fc) * exp(-pow(velocity / vs, 2));
        float F_physical = g_v * tanh(velocity * 100) + sigma2 * velocity;

        // Neural network residual
        vector inputs = {velocity, position, temperature};
        float F_nn = friction_nn.forward(inputs);

        // Combined prediction
        return F_physical + F_nn;
    }

    // Collect training data during operation
    void collect_data(float velocity, float position, float temperature,
                      float current, float acceleration, float inertia) {
        if (!collecting_data) return;

        // Estimate friction from dynamics: F_friction = I*a - K_t*I
        float motor_torque = current * torque_constant;
        float inertia_torque = inertia * acceleration;
        float friction_estimate = motor_torque - inertia_torque;

        TrainingPoint point = {velocity, position, temperature, friction_estimate};
        training_data.push_back(point);
    }

    // Train neural network on collected data
    void train() {
        for (int epoch = 0; epoch < 100; epoch++) {
            shuffle(training_data);

            for (auto& point : training_data) {
                vector inputs = {point.velocity, point.position, point.temperature};

                // Target: measured friction minus physical model
                float g_v = Fc + (Fs - Fc) * exp(-pow(point.velocity / vs, 2));
                float F_physical = g_v * tanh(point.velocity * 100) + sigma2 * point.velocity;
                float target = point.measured_friction - F_physical;

                // Train NN to predict residual
                friction_nn.train_sample(inputs, target);
            }
        }
    }

private:
    float torque_constant = 0.1f;
    void shuffle(vector& data);
};

6.9 Safe Reinforcement Learning

// Safe RL with Constraint Satisfaction
class SafeRLController {
private:
    // Policy network
    TD3MotorController policy;

    // Safety critic (predicts constraint violations)
    NeuralController safety_critic;

    // Safety constraints
    vector constraints;

    // Fallback controller
    PIDController safe_pid;

    // Safety parameters
    float safety_threshold;
    bool use_shield;

public:
    struct SafetyConstraint {
        string name;
        float (*evaluate)(State& s);
        float limit;
    };

    SafeRLController(int state_dim, int action_dim, vector& safety_constraints) :
        policy(state_dim, action_dim),
        safety_critic(state_dim + action_dim, 64, safety_constraints.size()),
        safe_pid(10.0f, 1.0f, 0.5f, 0.001f)
    {
        constraints = safety_constraints;
        safety_threshold = 0.95f;  // 95% confidence of safety
        use_shield = true;
    }

    float select_safe_action(vector& state) {
        // Get RL action
        float rl_action = policy.select_action(state);

        if (!use_shield) {
            return rl_action;
        }

        // Check safety critic
        vector sa = concat(state, {rl_action});
        vector constraint_values = safety_critic.forward_all(sa);

        bool is_safe = true;
        for (int i = 0; i < constraints.size(); i++) {
            if (constraint_values[i] > constraints[i].limit) {
                is_safe = false;
                break;
            }
        }

        if (is_safe) {
            return rl_action;
        }

        // Shield: project to safe action
        return find_safe_action(state, rl_action);
    }

    // Find closest safe action to RL suggestion
    float find_safe_action(vector& state, float unsafe_action) {
        // Binary search for boundary
        float safe_action = 0;  // Neutral action assumed safe
        float search_action = unsafe_action;
        float step = unsafe_action / 2.0f;

        for (int i = 0; i < 10; i++) {
            vector sa = concat(state, {search_action});
            vector cv = safety_critic.forward_all(sa);

            bool is_safe = true;
            for (int j = 0; j < constraints.size(); j++) {
                if (cv[j] > constraints[j].limit) {
                    is_safe = false;
                    break;
                }
            }

            if (is_safe) {
                safe_action = search_action;
                search_action += step;
            } else {
                search_action -= step;
            }
            step /= 2.0f;
        }

        return safe_action;
    }

    // Train safety critic from safety demonstrations
    void train_safety_critic(vector& safe_data,
                             vector& unsafe_data) {
        // Binary classification: safe (0) vs unsafe (1)
        for (auto& exp : safe_data) {
            vector input = concat(exp.state, {exp.action});
            safety_critic.train_sample(input, 0.0f);
        }
        for (auto& exp : unsafe_data) {
            vector input = concat(exp.state, {exp.action});
            safety_critic.train_sample(input, 1.0f);
        }
    }

private:
    vector concat(vector& a, vector b);
};

// Optimized Neural Network for Embedded Deployment
class EmbeddedNeuralController {
private:
    // Quantized weights (8-bit fixed point)
    int8_t* weights_q8;
    int8_t* biases_q8;

    // Scaling factors
    float input_scale;
    float output_scale;
    float weight_scale;

    // Network dimensions (statically allocated)
    static const int INPUT_SIZE = 6;
    static const int HIDDEN_SIZE = 16;
    static const int OUTPUT_SIZE = 1;

    // Intermediate buffers (avoid dynamic allocation)
    int16_t hidden_buffer[HIDDEN_SIZE];
    int16_t output_buffer[OUTPUT_SIZE];

public:
    EmbeddedNeuralController() {
        // Allocate aligned memory for SIMD
        weights_q8 = (int8_t*)aligned_alloc(16, (INPUT_SIZE * HIDDEN_SIZE + HIDDEN_SIZE * OUTPUT_SIZE));
        biases_q8 = (int8_t*)aligned_alloc(16, HIDDEN_SIZE + OUTPUT_SIZE);
    }

    // Load quantized model
    void load_quantized_model(const char* model_path) {
        // Load from file or flash memory
        FILE* f = fopen(model_path, "rb");
        fread(&input_scale, sizeof(float), 1, f);
        fread(&weight_scale, sizeof(float), 1, f);
        fread(&output_scale, sizeof(float), 1, f);
        fread(weights_q8, 1, INPUT_SIZE * HIDDEN_SIZE + HIDDEN_SIZE * OUTPUT_SIZE, f);
        fread(biases_q8, 1, HIDDEN_SIZE + OUTPUT_SIZE, f);
        fclose(f);
    }

    // Fast inference using integer arithmetic
    float forward(float* inputs) {
        // Quantize inputs
        int8_t inputs_q8[INPUT_SIZE];
        for (int i = 0; i < INPUT_SIZE; i++) {
            int32_t q = (int32_t)(inputs[i] * input_scale);
            inputs_q8[i] = (int8_t)clamp(q, -127, 127);
        }

        // Hidden layer: matrix multiply + ReLU
        int8_t* w1 = weights_q8;
        int8_t* b1 = biases_q8;

        for (int h = 0; h < HIDDEN_SIZE; h++) {
            int32_t acc = b1[h] << 8;  // Scale bias

            // Vectorized dot product (can use SIMD intrinsics)
            for (int i = 0; i < INPUT_SIZE; i++) {
                acc += (int32_t)inputs_q8[i] * (int32_t)w1[h * INPUT_SIZE + i];
            }

            // ReLU and requantize
            hidden_buffer[h] = (acc > 0) ? (int16_t)(acc >> 8) : 0;
        }

        // Output layer
        int8_t* w2 = weights_q8 + INPUT_SIZE * HIDDEN_SIZE;
        int8_t* b2 = biases_q8 + HIDDEN_SIZE;

        int32_t output_acc = b2[0] << 8;
        for (int h = 0; h < HIDDEN_SIZE; h++) {
            output_acc += (int32_t)hidden_buffer[h] * (int32_t)w2[h];
        }

        // Dequantize output
        return (float)output_acc * output_scale / (input_scale * weight_scale * 256.0f);
    }

private:
    int32_t clamp(int32_t x, int32_t min_val, int32_t max_val) {
        if (x < min_val) return min_val;
        if (x > max_val) return max_val;
        return x;
    }
};
