8.1 개발 환경 설정

8.1.2 하드웨어 요구사항

  • EMG 획득 보드: 최소 8채널, 16비트 ADC, 1kHz 샘플링
  • 처리 유닛: ARM Cortex-M4 이상, 또는 Linux SBC
  • 모터 드라이버: PWM 지원, 전류 감지 기능
  • 통신 모듈: BLE 5.0 또는 CAN 트랜시버

8.2 EMG 신호 처리 구현

8.2.1 신호 획득

import { EMGAcquisition, SignalProcessor } from '@wia/prosthetic-sdk';

// EMG 획득 설정
const emg = new EMGAcquisition({
    channels: 8,
    sampleRate: 1000,
    resolution: 16,
    gain: 1000,
    bandpass: { low: 20, high: 450 }
});

// 신호 처리 파이프라인
const processor = new SignalProcessor({
    windowSize: 200,  // ms
    windowIncrement: 50,  // ms
    features: ['MAV', 'RMS', 'WL', 'ZC', 'SSC']
});

// 실시간 처리 루프
emg.onData((samples) => {
    const features = processor.extractFeatures(samples);
    const classification = classifier.predict(features);
    sendCommand(classification);
});

8.2.2 분류기 훈련

import { LDAClassifier, TrainingSession } from '@wia/prosthetic-sdk';

// 훈련 세션 생성
const training = new TrainingSession({
    gestures: [
        'hand_open', 'hand_close', 'wrist_pronation',
        'wrist_supination', 'pinch', 'point', 'rest'
    ],
    repetitions: 5,
    holdTime: 3000,  // ms
    restTime: 2000   // ms
});

// 데이터 수집
await training.start();

// 분류기 훈련
const classifier = new LDAClassifier();
await classifier.train(training.getData());

// 성능 평가
const accuracy = classifier.crossValidate(5);
console.log(`분류 정확도: ${accuracy.mean}% (±${accuracy.std}%)`);

8.3 모터 제어 구현

8.3.1 관절 제어

import { MotorController, JointConfig } from '@wia/prosthetic-sdk';

// 손 관절 구성
const handConfig: JointConfig[] = [
    { id: 'thumb_mcp', motorId: 1, gearRatio: 100, maxTorque: 0.5 },
    { id: 'thumb_ip', motorId: 2, gearRatio: 50, maxTorque: 0.3 },
    { id: 'index_mcp', motorId: 3, gearRatio: 80, maxTorque: 0.4 },
    // ... 추가 관절
];

const motors = new MotorController(handConfig);

// 파지 패턴 실행
async function executeGrip(pattern: GripPattern) {
    const trajectory = pattern.getTrajectory();
    
    for (const waypoint of trajectory) {
        await motors.moveTo(waypoint.positions, {
            velocities: waypoint.velocities,
            forceLimits: waypoint.forces,
            timeout: 1000
        });
    }
}

8.3.2 힘 제어

// 임피던스 제어 구현
class ImpedanceController {
    constructor(
        private stiffness: number,  // N/m
        private damping: number,    // N·s/m
        private inertia: number     // kg
    ) {}
    
    compute(
        targetPos: number,
        currentPos: number,
        currentVel: number,
        externalForce: number
    ): number {
        const posError = targetPos - currentPos;
        const desiredAccel = (
            this.stiffness * posError - 
            this.damping * currentVel + 
            externalForce
        ) / this.inertia;
        
        return desiredAccel;
    }
}

8.4 감각 피드백 구현

import { FeedbackController, VibrationMotor } from '@wia/prosthetic-sdk';

// 피드백 시스템 설정
const feedback = new FeedbackController({
    devices: [
        new VibrationMotor({ id: 1, location: 'forearm_anterior' }),
        new VibrationMotor({ id: 2, location: 'forearm_posterior' })
    ]
});

// 센서-피드백 매핑
function mapGripToFeedback(gripForce: number): FeedbackParams {
    const normalized = Math.min(gripForce / MAX_GRIP_FORCE, 1.0);
    
    return {
        deviceId: 1,
        frequency: 50 + normalized * 200,  // 50-250 Hz
        amplitude: 0.2 + normalized * 0.8,  // 20-100%
        duration: 'continuous'
    };
}

// 실시간 피드백 루프
sensors.onGripForce((force) => {
    const params = mapGripToFeedback(force);
    feedback.apply(params);
});

8.5 시스템 통합

8.5.1 전체 시스템 초기화

import { 
    ProstheticSystem,
    EMGAcquisition,
    SignalProcessor,
    LDAClassifier,
    MotorController,
    FeedbackController,
    SafetyMonitor
} from '@wia/prosthetic-sdk';

async function initializeSystem() {
    // 구성요소 초기화
    const emg = new EMGAcquisition(emgConfig);
    const processor = new SignalProcessor(processorConfig);
    const classifier = await LDAClassifier.load('trained_model.json');
    const motors = new MotorController(handConfig);
    const feedback = new FeedbackController(feedbackConfig);
    const safety = new SafetyMonitor(safetyConfig);
    
    // 시스템 조립
    const system = new ProstheticSystem({
        acquisition: emg,
        processing: processor,
        classification: classifier,
        actuation: motors,
        feedback: feedback,
        safety: safety
    });
    
    // 안전 콜백 등록
    safety.onEmergency(() => {
        motors.emergencyStop();
        feedback.disable();
    });
    
    return system;
}

8.5.2 메인 제어 루프

async function mainControlLoop(system: ProstheticSystem) {
    const LOOP_PERIOD = 20;  // ms (50 Hz)
    
    while (system.isRunning()) {
        const startTime = Date.now();
        
        // 1. EMG 데이터 획득
        const emgData = await system.acquisition.read();
        
        // 2. 특징 추출
        const features = system.processing.extractFeatures(emgData);
        
        // 3. 분류
        const { gesture, confidence } = system.classification.predict(features);
        
        // 4. 명령 생성
        if (confidence > CONFIDENCE_THRESHOLD) {
            const command = generateCommand(gesture, emgData);
            
            // 5. 모터 제어
            await system.actuation.execute(command);
            
            // 6. 감각 피드백
            const sensorData = await system.actuation.readSensors();
            system.feedback.update(sensorData);
        }
        
        // 7. 안전 점검
        system.safety.check();
        
        // 루프 타이밍 유지
        const elapsed = Date.now() - startTime;
        if (elapsed < LOOP_PERIOD) {
            await sleep(LOOP_PERIOD - elapsed);
        }
    }
}
