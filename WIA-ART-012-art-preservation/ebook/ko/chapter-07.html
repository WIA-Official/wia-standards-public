7.2.1 AWS S3 Glacier 통합

// AWS SDK for JavaScript
import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";

const s3 = new S3Client({ region: "us-east-1" });

async function archiveToGlacier(artworkId, filePath) {
  const fileContent = await fs.readFile(filePath);

  const command = new PutObjectCommand({
    Bucket: "wia-archive-deep",
    Key: `artworks/${artworkId}/${path.basename(filePath)}`,
    Body: fileContent,
    StorageClass: "DEEP_ARCHIVE", // 가장 저렴한 클래스
    Metadata: {
      'wia-artwork-id': artworkId,
      'wia-upload-date': new Date().toISOString()
    },
    Tagging: 'wia-standard=1.0&preservation=long-term'
  });

  const result = await s3.send(command);
  console.log(`✅ Archived to Glacier: ${result.ETag}`);

  return {
    bucket: "wia-archive-deep",
    key: `artworks/${artworkId}/${path.basename(filePath)}`,
    storage_class: "DEEP_ARCHIVE",
    url: `s3://wia-archive-deep/artworks/${artworkId}/...`
  };
}

7.3 블록체인 타임스탬프와 출처 증명

블록체인에 작품의 해시를 기록하여 특정 시점에 작품이 존재했음을 증명할 수 있습니다. 이는 법적 분쟁, 저작권 증명, 위조 방지에 유용합니다.

7.3.1 Ethereum 타임스탬프

// Solidity 스마트 컨트랙트
pragma solidity ^0.8.0;

contract WIATimestamp {
    struct ArtworkRecord {
        string artworkId;
        bytes32 checksumSHA256;
        uint256 timestamp;
        address registrar;
    }

    mapping(string => ArtworkRecord) public artworks;

    event ArtworkRegistered(
        string artworkId,
        bytes32 checksum,
        uint256 timestamp
    );

    function registerArtwork(
        string memory _artworkId,
        bytes32 _checksum
    ) public {
        require(
            artworks[_artworkId].timestamp == 0,
            "Artwork already registered"
        );

        artworks[_artworkId] = ArtworkRecord({
            artworkId: _artworkId,
            checksumSHA256: _checksum,
            timestamp: block.timestamp,
            registrar: msg.sender
        });

        emit ArtworkRegistered(_artworkId, _checksum, block.timestamp);
    }

    function verifyArtwork(
        string memory _artworkId,
        bytes32 _checksum
    ) public view returns (bool, uint256) {
        ArtworkRecord memory record = artworks[_artworkId];
        bool isValid = record.checksumSHA256 == _checksum;
        return (isValid, record.timestamp);
    }
}

7.3.2 타임스탬프 검증

// ethers.js로 블록체인 검증
import { ethers } from 'ethers';

const provider = new ethers.JsonRpcProvider('https://mainnet.infura.io/v3/YOUR_KEY');
const contract = new ethers.Contract(CONTRACT_ADDRESS, ABI, provider);

// 작품 등록 (가스 비용 발생)
const checksumBytes32 = ethers.zeroPadValue(ethers.toBeHex(checksum), 32);
const tx = await contract.registerArtwork('art-2024-001', checksumBytes32);
await tx.wait();

console.log(`✅ Registered on blockchain: TX ${tx.hash}`);

// 검증 (무료)
const [isValid, timestamp] = await contract.verifyArtwork('art-2024-001', checksumBytes32);
if (isValid) {
  const date = new Date(timestamp * 1000);
  console.log(`✅ Artwork verified, registered on: ${date.toISOString()}`);
}

addEventListener('fetch', event => {
  event.respondWith(handleRequest(event.request))
})

async function handleRequest(request) {
  const url = new URL(request.url)
  const artworkId = url.pathname.split('/')[2]

  // WIA API에서 메타데이터 조회
  const metadata = await fetch(`https://api.wia.org/v1/artworks/${artworkId}`)

  // 가장 빠른 소스 선택 (IPFS vs S3 vs Arweave)

  return fetch(bestSource, { cf: { cacheTtl: 86400 } })
}

7.6 하이브리드 보존 전략

최고의 보존 전략은 여러 방법을 조합하는 것입니다. 3-2-1 규칙을 따릅니다:

  • 3 copies: 최소 3개의 복사본
  • 2 media types: 최소 2가지 다른 매체 (예: HDD + 클라우드)
  • 1 offsite: 최소 1개는 원격지에 (재해 대비)
계층 위치 기술 비용 접근 속도
Tier 1 (핫) 온프레미스 NAS RAID 6 HDD 높음 즉시
Tier 2 (웜) AWS S3 Standard 클라우드 스토리지 중간 즉시
Tier 3 (콜드) AWS Glacier 아카이브 스토리지 낮음 12시간
Tier 4 (P2P) IPFS/Arweave 탈중앙화 저장소 중간 중간
Tier 5 (공공) Internet Archive 공공 아카이브 무료 중간

7.6.1 자동 백업 스크립트

// 5-tier 백업 자동화
async function backupArtwork(artworkId) {
  const artwork = await client.artworks.get(artworkId);
  const localPath = `/archive/${artworkId}`;

  // Tier 1: 로컬 NAS
  await copyToNAS(localPath, '/nas/archive/');

  // Tier 2: AWS S3
  const s3Result = await uploadToS3(localPath, 'wia-hot-storage');

  // Tier 3: AWS Glacier (수명 주기 정책으로 자동 전환)
  // (S3에서 90일 후 자동으로 Glacier로 이동)

  // Tier 4: IPFS
  const ipfsCID = await uploadToIPFS(localPath);
  await pinata.pinByHash(ipfsCID); // 영구 핀

  // Tier 5: Internet Archive (공개 작품만)
  if (artwork.rights.license === 'CC0' || artwork.rights.license.startsWith('CC BY')) {
    await uploadToInternetArchive(artworkId, localPath);
  }

  // 메타데이터 업데이트
  await client.artworks.update(artworkId, {
    preservation: {
      backup_locations: [
        { tier: 1, type: 'nas', path: '/nas/archive/...' },
        { tier: 2, type: 's3', url: s3Result.Location },
        { tier: 4, type: 'ipfs', cid: ipfsCID }
      ],
      last_backup: new Date().toISOString()
    }
  });

  console.log(`✅ Backed up to 5 tiers: ${artworkId}`);
}

7.8 복습 질문

  1. AWS S3의 스토리지 클래스(Standard, Standard-IA, Glacier, Deep Archive)를 비교하고, 각 클래스의 적절한 사용 사례를 비용과 검색 시간 관점에서 설명하십시오.
  2. S3 수명 주기 정책을 설계하여 1TB 작품 컬렉션의 월 평균 비용을 $23에서 $5로 절감하는 전략을 제시하십시오.
  3. Ethereum 스마트 컨트랙트를 사용한 블록체인 타임스탬프의 원리를 설명하고, 이것이 법적 저작권 증명에 어떻게 활용될 수 있는지 논의하십시오.
  4. Internet Archive에 작품을 업로드하는 과정을 설명하고, 공개 라이선스(CC0, CC BY)와 All Rights Reserved 작품의 처리 방법을 구분하십시오.
  5. 3-2-1 백업 규칙(3 copies, 2 media types, 1 offsite)을 설명하고, 5-tier 하이브리드 전략을 설계하십시오.
  6. CDN(Cloudflare Workers)을 사용하여 IPFS, S3, Arweave 중 가장 빠른 소스를 동적으로 선택하는 라우팅 로직을 구현하십시오.
