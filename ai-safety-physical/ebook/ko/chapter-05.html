5장: 비상 대응 및 안전 장치

5.1 비상 대응의 중요성

예방이 실패할 때, 비상 대응 시스템은 피해에 대한 마지막 방어선이 됩니다. WIA-AI-SAFETY-PHYSICAL 표준은 모든 물리적 AI 시스템이 비상의 성격에 관계없이 정의된 시간 제한 내에 시스템을 안전 상태로 가져올 수 있는 강력한 비상 대응 메커니즘을 포함하도록 의무화합니다.

중요 요구사항: 비상 대응 시스템은 다른 시스템 구성 요소가 고장 났을 때도 올바르게 작동해야 합니다. 이는 비상 회로가 독립적이고, 가능한 경우 하드와이어로 연결되어야 하며, 손상될 수 있는 소프트웨어에 의존하지 않아야 함을 의미합니다.

5.2 정지 카테고리 및 응답 시간

국제 표준은 각각 특정 특성과 사용 사례를 가진 세 가지 정지 카테고리를 정의합니다:

카테고리 방법 특성 사용 사례
카테고리 0 즉시 전원 차단 비제어 정지, 기계적 충격 가능 진정한 비상, 임박한 위험
카테고리 1 제어된 정지 후 전원 차단 감속 후 전원 차단 표준 비상 정지
카테고리 2 제어된 정지, 전원 유지 유지 토크와 함께 정지 보호 정지, 재개 가능

응답 시간 요구사항

// 정지 응답 시간 요구사항
interface StopTimeRequirements {

    emergencyStop: {
        detection: 10,      // ms - 비상 정지 활성화 감지
        processing: 5,      // ms - 안전 컨트롤러 처리
        actuation: 50,      // ms - 브레이크 작동
        deceleration: 500,  // ms - 최대 감속 시간
        totalMax: 565       // ms - 최대 총합
    };

    protectiveStop: {
        detection: 20,      // ms - 센서 처리
        processing: 10,     // ms - 안전 로직
        actuation: 50,      // ms - 제어된 감속 시작
        deceleration: 1000, // ms - 부드러운 감속
        totalMax: 1080      // ms - 최대 총합
    };

    safetyRatedStop: {
        detection: 100,     // ms - 복잡한 조건 처리
        processing: 50,     // ms - 안전 등급 소프트웨어
        actuation: 100,     // ms - 제어된 궤적
        deceleration: 2000, // ms - 점진적 정지
        totalMax: 2250      // ms - 최대 총합
    };

// 최대 정지 거리 계산
function calculateStoppingDistance(
    velocity: number,     // m/s
    deceleration: number, // m/s^2
    reactionTime: number  // s
): number {
    const reactionDistance = velocity * reactionTime;
    const brakingDistance = (velocity * velocity) / (2 * deceleration);
    return reactionDistance + brakingDistance;
}

5.3 비상 정지 시스템 설계

비상 정지 시스템은 신뢰성과 접근성을 보장하기 위해 신중한 설계가 필요합니다:

비상 정지 버튼 요구사항

  • 색상: 노란색 배경에 빨간색 버섯 머리
  • 작동: 직접 작동(소프트웨어 의존 없음)
  • 래칭: 자가 래칭, 의도적인 리셋 필요
  • 접근성: 모든 운영자 위치에서 손이 닿는 범위
  • 수량: 모든 접근 지점을 커버하는 다중 버튼
  • 배선: 상시 폐쇄(NC) 접점, 안전 장치
// 비상 정지 회로 설계
class EmergencyStopCircuit {
    // 이중 채널 안전 릴레이 아키텍처
    private channel1: SafetyRelay;
    private channel2: SafetyRelay;

    // 직렬로 연결된 비상 정지 버튼(모두 NC)
    private estopButtons: EStopButton[] = [];

    constructor(config: EStopConfig) {
        // 이중 중복 채널 초기화
        this.channel1 = new SafetyRelay('CH1', config.relay1);
        this.channel2 = new SafetyRelay('CH2', config.relay2);

        // 모든 비상 정지 버튼 등록
        for (const buttonConfig of config.buttons) {
            const button = new EStopButton(buttonConfig);
            this.estopButtons.push(button);
        }

        // 채널 간 교차 모니터링
        this.setupCrossMonitoring();
    }

    // 비상 정지 활성화 시퀀스
    activateEmergencyStop(source: string): void {
        const activationTime = Date.now();

        // 활성화 소스 로그
        this.logActivation(source, activationTime);

        // 채널 1: 모터에 즉시 전원 차단
        this.channel1.open();

        // 채널 2: 기계식 브레이크 작동
        this.channel2.triggerBrakes();

        // 정지 확인
        setTimeout(() => {
            this.verifyStop(activationTime);
        }, 500); // 500ms 내 확인
    }

    // 리셋 시퀀스(의도적인 동작 필요)
    reset(): ResetResult {
        // 모든 비상 정지 버튼 해제 확인
        for (const button of this.estopButtons) {
            if (button.isPressed()) {
                return {
                    success: false,
                    reason: `비상 정지 버튼 ${button.id}이 여전히 눌림`
                };
            }
        }

        // 안전 조건 확인
        if (!this.checkSafeConditions()) {
            return {
                success: false,
                reason: '안전하지 않은 조건 지속'
            };
        }

        // 수동 리셋 버튼 필요
        if (!this.manualResetPressed) {
            return {
                success: false,
                reason: '수동 리셋 필요'
            };
        }

        // 안전 릴레이 리셋
        this.channel1.close();
        this.channel2.close();

        return { success: true };
    }

    // 불일치 감지를 위한 교차 모니터링
    setupCrossMonitoring(): void {
        setInterval(() => {
            const ch1State = this.channel1.getState();
            const ch2State = this.channel2.getState();

            if (ch1State !== ch2State) {
                // 채널 불일치 - 고장 조건
                this.triggerSafeState('채널 불일치 감지');
            }
        }, 10); // 100Hz 모니터링
    }
}

5.4 안전 장치 설계 원칙

안전 장치 설계는 고장이 위험한 조건이 아닌 안전한 조건으로 이어지도록 보장합니다:

구성 요소 안전 장치 동작 구현
전원 공급 손실 시 브레이크 작동 스프링 적용, 전원 해제 브레이크
통신 손실 시 정지 트리거 워치독 타이머, 하트비트 프로토콜
센서 고장 시 위험 가정 보수적 기본값, 중복성
컨트롤러 정지 시 정지 트리거 하드웨어 워치독, 이중 프로세서
액추에이터 고장 시 위치 유지 기계식 위치 잠금
소프트웨어 예외 시 정지 트리거 방어적 프로그래밍, 안전 기본값 
// 안전 장치 상태 머신
enum SafetyState {
    NORMAL = 'NORMAL',
    WARNING = 'WARNING',
    PROTECTIVE_STOP = 'PROTECTIVE_STOP',
    EMERGENCY_STOP = 'EMERGENCY_STOP',
    FAULT = 'FAULT',
    SAFE = 'SAFE'
}

class FailSafeStateMachine {
    private currentState: SafetyState = SafetyState.SAFE;
    private stateHistory: StateTransition[] = [];

    // 유효한 상태 전이(안전 장치: 항상 더 안전한 상태로 이동 가능)
    private validTransitions: Map<SafetyState, SafetyState[]> = new Map([
        [SafetyState.NORMAL, [SafetyState.WARNING, SafetyState.PROTECTIVE_STOP,
                             SafetyState.EMERGENCY_STOP, SafetyState.FAULT]],
        [SafetyState.WARNING, [SafetyState.NORMAL, SafetyState.PROTECTIVE_STOP,
                              SafetyState.EMERGENCY_STOP, SafetyState.FAULT]],
        [SafetyState.PROTECTIVE_STOP, [SafetyState.NORMAL, SafetyState.WARNING,
                                       SafetyState.EMERGENCY_STOP, SafetyState.FAULT]],
        [SafetyState.EMERGENCY_STOP, [SafetyState.SAFE]], // SAFE로만 이동 가능
        [SafetyState.FAULT, [SafetyState.SAFE]], // SAFE로만 이동 가능
        [SafetyState.SAFE, [SafetyState.NORMAL]] // 의도적인 재시작 필요
    ]);

    transitionTo(newState: SafetyState, reason: string): boolean {
        const validTargets = this.validTransitions.get(this.currentState);

        // 비상 전이는 항상 더 안전한 상태로 허용
        const safetyOrder = [SafetyState.NORMAL, SafetyState.WARNING,
                            SafetyState.PROTECTIVE_STOP, SafetyState.EMERGENCY_STOP,
                            SafetyState.FAULT, SafetyState.SAFE];

        const currentIndex = safetyOrder.indexOf(this.currentState);
        const newIndex = safetyOrder.indexOf(newState);

        // 더 안전한 상태로의 전이는 항상 허용(높은 인덱스)
        if (newIndex > currentIndex || validTargets?.includes(newState)) {
            this.stateHistory.push({
                from: this.currentState,
                to: newState,
                reason,
                timestamp: Date.now()
            });
            this.currentState = newState;
            this.executeStateActions(newState);
            return true;
        }

        return false;
    }

    executeStateActions(state: SafetyState): void {
        switch (state) {
            case SafetyState.PROTECTIVE_STOP:
                this.executeProtectiveStop();
                break;
            case SafetyState.EMERGENCY_STOP:
                this.executeEmergencyStop();
                break;
            case SafetyState.FAULT:
                this.executeFaultResponse();
                break;
            case SafetyState.SAFE:
                this.verifySafeState();
                break;
        }
    }
}

5.5 중복성과 고장 허용

중요한 안전 시스템은 단일 고장을 허용하기 위해 중복성이 필요합니다:

중복성 아키텍처

아키텍처 설명 고장 허용 사용 사례
1oo1 (단일) 단일 채널 0 저위험 응용
1oo2 (이중) 두 채널, 어느 하나가 트리거 1 표준 안전
2oo2 (이중 투표) 둘 다 일치해야 함 0 (높은 가용성) 높은 가용성
2oo3 (TMR) 다수결 투표 1 높은 신뢰성 + 가용성
1oo2D 진단이 있는 이중 1 (감지됨) 로봇공학에서 일반적 
// 삼중 모듈 중복 구현
class TMRVoter<T> {
    private channels: Channel<T>[];
    private comparator: (a: T, b: T) => boolean;

    constructor(
        channel1: Channel<T>,
        channel2: Channel<T>,
        channel3: Channel<T>,
        comparator: (a: T, b: T) => boolean
    ) {
        this.channels = [channel1, channel2, channel3];
        this.comparator = comparator;
    }

    vote(): VoteResult<T> {
        const values = this.channels.map(ch => ch.getValue());

        // 3방향 일치 확인
        if (this.comparator(values[0], values[1]) &&
            this.comparator(values[1], values[2])) {
            return {
                value: values[0],
                confidence: 1.0,
                faultDetected: false
            };
        }

        // 2방향 일치 확인(하나의 결함 채널)
        if (this.comparator(values[0], values[1])) {
            return {
                value: values[0],
                confidence: 0.67,
                faultDetected: true,
                faultyChannel: 2
            };
        }
        if (this.comparator(values[1], values[2])) {
            return {
                value: values[1],
                confidence: 0.67,
                faultDetected: true,
                faultyChannel: 0
            };
        }
        if (this.comparator(values[0], values[2])) {
            return {
                value: values[0],
                confidence: 0.67,
                faultDetected: true,
                faultyChannel: 1
            };
        }

        // 일치 없음 - 모든 채널이 다름
        return {
            value: null,
            confidence: 0,
            faultDetected: true,
            faultyChannel: 'MULTIPLE'
        };
    }
}

// 이중 채널 안전 컨트롤러
class DualChannelController {
    private primary: SafetyProcessor;
    private secondary: SafetyProcessor;

    async executeSafetyFunction(input: SafetyInput): Promise<SafetyOutput> {
        // 두 채널에서 실행
        const [result1, result2] = await Promise.all([
            this.primary.execute(input),
            this.secondary.execute(input)
        ]);

        // 결과 비교
        if (!this.compareResults(result1, result2)) {
            // 불일치 감지 - 안전 상태
            return {
                action: SafetyAction.STOP,
                reason: '채널 불일치',
                diagnosticCode: 'DC_MISMATCH'
            };
        }

        return result1;
    }
}

5.6 점진적 성능 저하

비중요 구성 요소가 고장 났을 때 시스템은 감소된 기능으로 안전하게 계속 작동해야 합니다:

// 점진적 성능 저하 시스템
class GracefulDegradation {
    private capabilities: Map<string, Capability> = new Map();
    private degradationLevels: DegradationLevel[] = [];

    // 대체 옵션과 함께 기능 등록
    registerCapability(
        name: string,
        implementation: () => void,
        fallbacks: Fallback[]
    ): void {
        this.capabilities.set(name, {
            name,
            primary: implementation,
            fallbacks,
            currentLevel: 0
        });
    }

    // 구성 요소 고장 처리
    handleFailure(componentId: string): DegradationResult {
        const affectedCapabilities = this.findAffectedCapabilities(componentId);

        for (const capability of affectedCapabilities) {
            const cap = this.capabilities.get(capability);
            if (!cap) continue;

            // 다음 대체 수준 시도
            if (cap.currentLevel < cap.fallbacks.length) {
                cap.currentLevel++;
                const fallback = cap.fallbacks[cap.currentLevel - 1];

                // 저하 모드 적용
                this.applyDegradedMode(cap.name, fallback);

                this.logDegradation(cap.name, fallback);
            } else {
                // 더 이상 대체 없음 - 기능 비활성화
                this.disableCapability(cap.name);
            }
        }

        return this.generateDegradationReport();
    }

    // 예시: 비전 시스템 저하
    visionDegradationLevels: Fallback[] = [
        {
            name: '해상도 감소',
            description: '절반 해상도로 처리',
            speedReduction: 0.8, // 80% 속도
            implement: () => this.visionSystem.setResolution('HALF')
        },
        {
            name: '단일 카메라',
            description: '기본 카메라만 사용',
            speedReduction: 0.5,
            implement: () => this.visionSystem.useSingleCamera()
        },
        {
            name: '기본 감지만',
            description: '자세 추정 비활성화, 기본 감지만',
            speedReduction: 0.3,
            implement: () => this.visionSystem.setMode('BASIC')
        },
        {
            name: '근접 센서만',
            description: '비비전 센서로 대체',
            speedReduction: 0.2,
            implement: () => this.disableVisionCompletely()
        }
    ];
}

5.7 복구 절차

비상 및 고장 조건에서 안전한 복구는 신중한 시퀀싱이 필요합니다:

// 복구 절차 관리자
class RecoveryManager {
    async executeRecovery(fromState: SafetyState): Promise<RecoveryResult> {
        // 단계 1: 현재 상태 확인
        const currentState = await this.verifyCurrentState();
        if (currentState !== fromState) {
            return { success: false, reason: '복구 중 상태 변경' };
        }

        // 단계 2: 복구 전 진단 실행
        const diagnostics = await this.runDiagnostics();
        if (!diagnostics.allPassed) {
            return {
                success: false,
                reason: '진단 실패',
                failedTests: diagnostics.failed
            };
        }

        // 단계 3: 고장 조건 해제
        const faults = await this.getFaultConditions();
        for (const fault of faults) {
            const cleared = await this.attemptFaultClear(fault);
            if (!cleared) {
                return {
                    success: false,
                    reason: `고장 해제 불가: ${fault.description}`
                };
            }
        }

        // 단계 4: 재시작을 위한 안전 조건 확인
        const safetyCheck = await this.verifySafeConditions();
        if (!safetyCheck.safe) {
            return {
                success: false,
                reason: '안전하지 않은 조건 존재',
                conditions: safetyCheck.unsafeConditions
            };
        }

        // 단계 5: 제어된 재시작 시퀀스 실행
        const restartSequence = [
            () => this.initializeSafetyControllers(),
            () => this.enableSafetySensors(),
            () => this.verifySensorCoverage(),
            () => this.enableMotionController(),
            () => this.enableBrakes(),
            () => this.releaseBrakes(),
            () => this.verifyMotionCapability()
        ];

        for (let i = 0; i < restartSequence.length; i++) {
            try {
                await restartSequence[i]();
            } catch (error) {
                // 어떤 실패에서도 재시작 중단
                await this.abortRestart();
                return {
                    success: false,
                    reason: `단계 ${i}에서 재시작 실패`,
                    error
                };
            }
        }

        // 단계 6: 정상 운영으로 전환
        await this.transitionToNormal();

        return { success: true, newState: SafetyState.NORMAL };
    }

    // 운영자 확인이 있는 수동 재시작
    async manualRestart(operatorId: string): Promise<RecoveryResult> {
        // 운영자 권한 확인
        if (!await this.verifyOperatorAuthorization(operatorId)) {
            return { success: false, reason: '무허가 운영자' };
        }

        // 물리적 리셋 버튼 필요
        console.log('물리적 리셋 버튼 대기 중...');
        const resetPressed = await this.waitForResetButton(30000); // 30초 타임아웃

        if (!resetPressed) {
            return { success: false, reason: '리셋 버튼 타임아웃' };
        }

        // 복구 실행
        return this.executeRecovery(SafetyState.EMERGENCY_STOP);
    }
}

5.8 사고 기록 및 분석

모든 안전 이벤트는 분석 및 지속적인 개선을 위해 기록되어야 합니다:

// 안전 이벤트 기록기
class SafetyEventRecorder {
    private eventLog: SafetyEvent[] = [];
    private blackBox: CircularBuffer<SystemState>;

    constructor() {
        // 100Hz에서 마지막 60초의 시스템 상태 유지
        this.blackBox = new CircularBuffer(6000);
    }

    recordEvent(event: SafetyEvent): void {
        // 현재 시스템 상태 캡처
        const systemState = this.captureSystemState();

        // 문맥 데이터 추가
        event.systemState = systemState;
        event.timestamp = Date.now();
        event.blackBoxSnapshot = this.blackBox.getSnapshot();

        // 이벤트 저장
        this.eventLog.push(event);

        // 비휘발성 스토리지에 유지
        this.persistEvent(event);

        // 안전 관리 시스템에 알림
        this.notifySafetyManagement(event);
    }

    captureSystemState(): SystemState {
        return {
            robotState: {
                jointPositions: this.robot.getJointPositions(),
                jointVelocities: this.robot.getJointVelocities(),
                jointTorques: this.robot.getJointTorques(),
                tcpPose: this.robot.getTCPPose(),
                tcpVelocity: this.robot.getTCPVelocity()
            },
            safetyState: {
                currentState: this.safetyStateMachine.getCurrentState(),
                activeSafetyFunctions: this.getActiveSafetyFunctions(),
                sensorHealth: this.getSensorHealth()
            },
            environmentState: {
                humanPositions: this.humanDetection.getPositions(),
                obstacles: this.obstacleDetection.getObstacles(),
                zoneOccupancy: this.getZoneOccupancy()
            },
            controllerState: {
                activeProgram: this.controller.getActiveProgram(),
                executionPoint: this.controller.getExecutionPoint(),
                commandedVelocity: this.controller.getCommandedVelocity()
            }
        };
    }

    // 사고 보고서 생성
    generateIncidentReport(eventId: string): IncidentReport {
        const event = this.eventLog.find(e => e.id === eventId);
        if (!event) throw new Error('이벤트를 찾을 수 없음');

        return {
            eventId,
            timestamp: event.timestamp,
            type: event.type,
            severity: event.severity,
            description: event.description,
            rootCause: this.analyzeRootCause(event),
            timeline: this.reconstructTimeline(event),
            systemState: event.systemState,
            recommendations: this.generateRecommendations(event)
        };
    }
}

핵심 요점

  • 비상 정지 시스템은 독립적이고, 하드와이어로 연결되며, 안전 장치여야 합니다
  • 세 가지 정지 카테고리(0, 1, 2)는 다른 수준의 응답을 제공합니다
  • 각 안전 기능에 대해 응답 시간을 지정하고 검증해야 합니다
  • 안전 장치 설계는 고장이 안전한 조건으로 이어지도록 보장합니다
  • 중복성 아키텍처(1oo2, 2oo3)는 고장 허용을 제공합니다
  • 점진적 성능 저하는 감소된 기능으로 안전한 운영을 유지합니다
  • 복구 절차에는 검증 단계와 운영자 확인이 포함되어야 합니다
  • 사고 기록은 분석 및 지속적인 개선을 가능하게 합니다
