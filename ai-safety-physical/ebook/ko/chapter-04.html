4장: 모니터링 및 감지 시스템

4.1 안전에서 모니터링의 역할

안전 모니터링 시스템은 물리적 AI 안전을 위한 감각적 기반을 제공합니다. 환경, 로봇 상태, 인간 존재를 지속적으로 관찰하여 사전 위험 회피 및 사후 비상 대응을 가능하게 합니다. 포괄적인 모니터링 전략은 어떤 위험도 감지되지 않고 지나가지 않도록 중복 커버리지를 갖춘 여러 센서 모달리티를 사용합니다.

심층 방어 모니터링: 안전 시스템이 다중 보호 계층을 사용하는 것처럼, 모니터링 시스템도 여러 센서 유형, 중복 커버리지 영역, 이중화된 처리를 사용하여 단일 장애 지점이 안전 인식을 손상시키지 않도록 해야 합니다.

4.2 근접 및 존재 감지

인간 존재 감지는 충돌을 방지하고 안전 구역을 관리하는 첫 번째 방어선입니다:

근접 감지를 위한 센서 기술

기술 범위 해상도 강점 한계
LiDAR 0.1 - 100m mm 수준 높은 정확도, 3D 매핑 비용, 반사 표면
ToF 카메라 0.5 - 10m cm 수준 밀집 깊이 데이터, 적당한 비용 햇빛 간섭
초음파 0.02 - 5m cm 수준 저비용, 먼지/연기에서 작동 넓은 빔, 느린 업데이트
레이더 0.5 - 50m cm 수준 벽 통과, 속도 감지 낮은 공간 해상도
안전 라이트 커튼 0 - 20m mm 수준 안전 인증, 빠른 응답 2D만 가능, 시선 필요
정전용량 0 - 0.3m mm 수준 피부 장착, 접촉 전 감지 매우 짧은 범위 
// 근접 감지 시스템
class ProximityDetectionSystem {
    private sensors: ProximitySensor[] = [];
    private fusedModel: OccupancyGrid;

    constructor(config: ProximityConfig) {
        // 센서 배열 초기화
        this.sensors = [
            new LidarSensor(config.lidar),
            new ToFCamera(config.tofCamera),
            new SafetyLightCurtain(config.lightCurtain),
            new CapacitiveSkin(config.capacitive)
        ];

        this.fusedModel = new OccupancyGrid(config.gridResolution);
    }

    async updateProximityModel(): Promise {
        // 모든 센서에서 병렬로 판독값 가져오기
        const readings = await Promise.all(
            this.sensors.map(s => s.getReading())
        );

        // 센서 데이터 융합
        const fusedData = this.fuseSensorData(readings);

        // 점유 그리드 업데이트
        this.fusedModel.update(fusedData);

        // 가장 가까운 인간 찾기
        const nearestHuman = this.findNearestHuman(fusedData);

        return {
            minDistance: nearestHuman.distance,
            position: nearestHuman.position,
            velocity: nearestHuman.velocity,
            confidence: nearestHuman.confidence,
            zone: this.determineZone(nearestHuman.distance)
        };
    }

    fuseSensorData(readings: SensorReading[]): FusedData {
        const fused = new FusedData();

        for (const reading of readings) {
            // 센서 신뢰도 및 다른 센서와의 일치도에 따른 가중치
            const reliability = this.calculateReliability(reading, readings);

            // 베이지안 센서 융합 적용
            fused.updateBelief(reading.points, reliability);
        }

        return fused;
    }

    // 센서 상태 검증
    validateSensors(): SensorHealthReport {
        const report: SensorHealthReport = {
            timestamp: Date.now(),
            sensors: [],
            overallHealth: true
        };

        for (const sensor of this.sensors) {
            const health = sensor.selfTest();
            report.sensors.push({
                id: sensor.id,
                type: sensor.type,
                healthy: health.passed,
                diagnostics: health.details
            });

            if (!health.passed && sensor.isSafetyCritical) {
                report.overallHealth = false;
            }
        }

        return report;
    }
}

4.3 힘-토크 모니터링

힘-토크 센서는 예상치 못한 접촉 감지를 가능하게 하고 의도적인 상호작용 중 힘 한계가 유지되도록 합니다:

힘 감지 위치

위치 센서 유형 목적 응답 시간
조인트 토크 스트레인 게이지 예상치 못한 저항 감지 < 1ms
손목 F/T 6축 로드셀 엔드이펙터 힘 제어 < 1ms
베이스 다축 로드셀 지면 반력 < 2ms
링크 표면 촉각 스킨 충돌 위치 파악 < 5ms 
// 힘-토크 모니터링 시스템
class ForceTorqueMonitor {
    private jointTorqueSensors: TorqueSensor[];
    private wristFTSensor: FTSensor;
    private expectedTorques: number[];

    constructor(robot: RobotModel) {
        this.jointTorqueSensors = robot.joints.map(j => j.torqueSensor);
        this.wristFTSensor = robot.endEffector.ftSensor;
    }

    async detectCollision(): Promise {
        // 현재 조인트 토크 가져오기
        const measuredTorques = await this.getMeasuredTorques();

        // 동역학 모델에서 예상 토크 가져오기
        const expectedTorques = await this.calculateExpectedTorques();

        // 잔차 계산
        const residuals = measuredTorques.map((m, i) =>
            Math.abs(m - expectedTorques[i])
        );

        // 잔차 임계값을 사용한 충돌 감지
        const collisionThreshold = 5.0; // Nm
        const collisionDetected = residuals.some(r => r > collisionThreshold);

        if (collisionDetected) {
            const collisionJoint = residuals.findIndex(r => r > collisionThreshold);
            return {
                detected: true,
                location: this.estimateCollisionLocation(collisionJoint, residuals),
                force: this.estimateCollisionForce(residuals),
                timestamp: Date.now()
            };
        }

        return { detected: false };
    }

    // 실시간 힘 제한
    enforceForceLimit(
        desiredTorques: number[],
        maxForce: number
    ): number[] {
        const currentForce = this.wristFTSensor.getForce();

        if (currentForce.magnitude() > maxForce * 0.9) {
            // 한계에 근접 - 토크 스케일 다운
            const scale = maxForce * 0.9 / currentForce.magnitude();
            return desiredTorques.map(t => t * scale);
        }

        return desiredTorques;
    }

    // 관측기 기반 충돌 감지
    calculateCollisionResidual(
        jointPosition: number[],
        jointVelocity: number[],
        jointTorque: number[]
    ): number[] {
        // r = M(q)q'' + C(q,q')q' + g(q) - tau
        // 모멘텀 관측기 접근 방식 사용

        const M = this.robot.massMatrix(jointPosition);
        const C = this.robot.coriolisMatrix(jointPosition, jointVelocity);
        const g = this.robot.gravityVector(jointPosition);

        // 관측기 상태 업데이트
        const K = this.observerGain;
        const p = M.multiply(jointVelocity); // 모멘텀

        // 잔차 = 관측기 추정 - 측정값
        return this.observerState.subtract(p);
    }
}

컴퓨터 비전은 안전 응용을 위한 풍부한 환경 이해를 제공합니다:

기능 방법 지연 안전 응용

자세 추정 OpenPose, MediaPipe 30-100ms 인간 동작 예측
제스처 인식 CNN/RNN 50-200ms 정지 제스처, 명령

// 비전 기반 안전 시스템
class VisionSafetySystem {
    private humanDetector: HumanDetector;
    private poseEstimator: PoseEstimator;
    private trajectoryPredictor: TrajectoryPredictor;
    private gestureRecognizer: GestureRecognizer;

    async processFrame(frame: VideoFrame): Promise {
        // 속도를 위한 병렬 처리
        const [humans, anomalies] = await Promise.all([
            this.detectHumans(frame),
            this.detectAnomalies(frame)
        ]);

        const results: VisionSafetyResult = {
            humans: [],
            anomalies,
            emergencyGesture: false
        };

        // 감지된 각 인간 처리
        for (const human of humans) {
            const pose = await this.poseEstimator.estimate(frame, human.bbox);
            const trajectory = this.trajectoryPredictor.predict(pose);
            const gesture = this.gestureRecognizer.recognize(pose);

            results.humans.push({
                id: human.id,
                bbox: human.bbox,
                pose,
                predictedTrajectory: trajectory,
                gesture
            });

            // 비상 정지 제스처 확인
            if (gesture === 'STOP' || gesture === 'EMERGENCY') {
                results.emergencyGesture = true;
            }
        }

        return results;
    }

    // 잠재적 충돌 예측
    predictCollision(
        robotTrajectory: Trajectory,
        humanTrajectories: PredictedTrajectory[]
    ): CollisionPrediction {
        const predictions: CollisionPrediction[] = [];

        for (const humanTraj of humanTrajectories) {
            // 세밀한 시간 해상도로 궤적 샘플링
            for (let t = 0; t < 2.0; t += 0.05) { // 2초 범위, 50ms 간격
                const robotPos = robotTrajectory.positionAt(t);
                const humanPos = humanTraj.positionAt(t);
                const humanUncertainty = humanTraj.uncertaintyAt(t);

                const distance = robotPos.distanceTo(humanPos);
                const safeDistance = this.calculateSafeDistance(humanUncertainty);

                if (distance < safeDistance) {
                    predictions.push({
                        timeToCollision: t,
                        probability: humanTraj.confidenceAt(t),
                        location: humanPos,
                        recommendedAction: this.determineAction(t, distance)
                    });
                }
            }
        }

        return this.aggregatePredictions(predictions);
    }
}

4.5 속도 및 분리 모니터링

로봇과 인간 사이의 보호 분리 거리에 대한 실시간 모니터링:

// 속도 및 분리 모니터링 (ISO/TS 15066)
class SpeedSeparationMonitor {
    // 필요한 분리 거리 계산
    calculateProtectiveSeparation(
        robotSpeed: number,      // m/s
        humanSpeed: number,      // m/s (가정 또는 측정)
        systemReactionTime: number, // 초
        robotStoppingTime: number,  // 초
        intrusion: number        // m (인체 침입)
    ): number {
        // Sp = Sh + Sr + Ss + C + Zd + Zr
        // 여기서:
        // Sh = 분리 거리에 대한 인간 기여
        // Sr = 로봇 반응 거리
        // Ss = 로봇 정지 거리
        // C = 침입 거리
        // Zd = 로봇 위치 불확실성
        // Zr = 인간 감지 위치 불확실성

        const Sh = humanSpeed * (systemReactionTime + robotStoppingTime);
        const Sr = robotSpeed * systemReactionTime;
        const Ss = robotSpeed * robotStoppingTime / 2; // 선형 감속 가정
        const C = intrusion;
        const Zd = 0.05; // 50mm 로봇 위치 불확실성
        const Zr = 0.10; // 100mm 감지 불확실성

        return Sh + Sr + Ss + C + Zd + Zr;
    }

    // 실시간 모니터링 루프
    async monitorSeparation(): Promise {
        while (this.running) {
            const robotState = await this.getRobotState();
            const humanPositions = await this.getHumanPositions();

            for (const human of humanPositions) {
                const currentDistance = this.calculateDistance(
                    robotState.position,
                    human.position
                );

                const requiredSeparation = this.calculateProtectiveSeparation(
                    robotState.speed,
                    human.estimatedSpeed,
                    this.systemReactionTime,
                    this.calculateStoppingTime(robotState.speed),
                    0.1 // 침입 거리
                );

                if (currentDistance < requiredSeparation) {
                    // 위반 - 속도 감소 또는 정지
                    const allowedSpeed = this.calculateAllowedSpeed(
                        currentDistance,
                        human.estimatedSpeed
                    );

                    if (allowedSpeed <= 0) {
                        await this.triggerProtectiveStop('분리 거리 위반');
                    } else {
                        await this.limitSpeed(allowedSpeed);
                    }
                }
            }

            await this.sleep(10); // 100Hz 모니터링 속도
        }
    }

    // 현재 분리에 대한 최대 허용 로봇 속도 계산
    calculateAllowedSpeed(
        currentDistance: number,
        humanSpeed: number
    ): number {
        // S = Sh + Sr + Ss + C + Zd + Zr에서 Vr 계산
        // 단순화: Vr_max = (S - Sh - C - Zd - Zr) / (Tr + Ts/2)

        const S = currentDistance;
        const Sh = humanSpeed * (this.systemReactionTime + this.maxStoppingTime);
        const C = 0.1;
        const Zd = 0.05;
        const Zr = 0.10;

        const numerator = S - Sh - C - Zd - Zr;
        const denominator = this.systemReactionTime + this.maxStoppingTime / 2;

        return Math.max(0, numerator / denominator);
    }
}

4.6 안전을 위한 센서 융합

여러 센서 모달리티를 결합하면 신뢰성과 커버리지가 향상됩니다:

// 다중 모달 센서 융합
class SafetySensorFusion {
    private sensors: SafetySensor[] = [];
    private fusionEngine: BayesianFusion;

    async fuseForSafety(): Promise {
        // 모든 센서 데이터 수집
        const sensorData = await Promise.all(
            this.sensors.map(s => s.getData())
        );

        // 센서 일치 확인
        const agreement = this.checkSensorAgreement(sensorData);

        if (agreement.level < 0.7) {
            // 심각한 불일치 - 보수적으로 처리
            return this.conservativeFusion(sensorData);
        }

        // 베이지안 센서 융합
        const fused = this.fusionEngine.fuse(sensorData);

        // 안전 마진 적용
        return this.applySafetyMargins(fused);
    }

    checkSensorAgreement(data: SensorData[]): Agreement {
        // 다른 센서의 인간 위치 추정 비교
        const humanPositions = data
            .filter(d => d.detectedHumans.length > 0)
            .flatMap(d => d.detectedHumans);

        if (humanPositions.length < 2) {
            return { level: 1.0, note: '비교를 위한 데이터 부족' };
        }

        // 위치 클러스터링 및 이상치 확인
        const clusters = this.clusterPositions(humanPositions);
        const outlierRatio = clusters.outliers.length / humanPositions.length;

        return {
            level: 1 - outlierRatio,
            note: outlierRatio > 0 ? '일부 센서가 불일치' : '좋은 일치'
        };
    }

    conservativeFusion(data: SensorData[]): FusedSafetyState {
        // 모든 센서에서 보고된 최소 거리 사용
        let minDistance = Infinity;
        let nearestHuman = null;

        for (const sensorData of data) {
            for (const human of sensorData.detectedHumans) {
                if (human.distance < minDistance) {
                    minDistance = human.distance;
                    nearestHuman = human;
                }
            }
        }

        // 불확실성에 대한 추가 안전 마진
        const safetyMargin = 0.2; // 200mm 추가

        return {
            minHumanDistance: minDistance - safetyMargin,
            nearestHuman,
            confidence: 0.5, // 불일치로 인한 낮은 신뢰도
            fusionMethod: 'CONSERVATIVE'
        };
    }
}

4.7 실시간 안전 감독

안전 감독자는 모든 모니터링 하위 시스템을 통합하고 실시간 안전 결정을 내립니다:

// 안전 감독 시스템
class SafetySupervisor {
    private proximitySystem: ProximityDetectionSystem;
    private forceMonitor: ForceTorqueMonitor;
    private visionSystem: VisionSafetySystem;
    private separationMonitor: SpeedSeparationMonitor;
    private sensorFusion: SafetySensorFusion;

    private currentSafetyState: SafetyState = SafetyState.NORMAL;

    async supervise(): Promise {
        const cycleTime = 10; // 100Hz 감독 속도

        while (this.running) {
            const cycleStart = Date.now();

            // 모든 안전 입력 수집
            const [proximity, force, vision, separation] = await Promise.all([
                this.proximitySystem.updateProximityModel(),
                this.forceMonitor.detectCollision(),
                this.visionSystem.processLatestFrame(),
                this.separationMonitor.getCurrentStatus()
            ]);

            // 센서 데이터 융합
            const fusedState = await this.sensorFusion.fuseForSafety();

            // 필요한 안전 응답 결정
            const response = this.determineSafetyResponse({
                proximity,
                force,
                vision,
                separation,
                fusedState
            });

            // 안전 응답 실행
            await this.executeSafetyResponse(response);

            // 로그 및 보고
            this.logSafetyState(response);

            // 주기 시간 유지
            const elapsed = Date.now() - cycleStart;
            if (elapsed < cycleTime) {
                await this.sleep(cycleTime - elapsed);
            } else {
                this.logWarning(`안전 주기 초과: ${elapsed}ms`);
            }
        }
    }

    determineSafetyResponse(inputs: SafetyInputs): SafetyResponse {
        // 우선순위 1: 충돌 감지
        if (inputs.force.detected) {
            return {
                action: SafetyAction.EMERGENCY_STOP,
                reason: '충돌 감지',
                priority: 1
            };
        }

        // 우선순위 2: 비상 제스처
        if (inputs.vision.emergencyGesture) {
            return {
                action: SafetyAction.PROTECTIVE_STOP,
                reason: '비상 제스처 감지',
                priority: 2
            };
        }

        // 우선순위 3: 분리 위반
        if (inputs.separation.violated) {
            if (inputs.separation.allowedSpeed <= 0) {
                return {
                    action: SafetyAction.PROTECTIVE_STOP,
                    reason: '분리 거리 위반',
                    priority: 3
                };
            } else {
                return {
                    action: SafetyAction.SPEED_LIMIT,
                    speedLimit: inputs.separation.allowedSpeed,
                    reason: '감소된 분리 거리',
                    priority: 4
                };
            }
        }

        // 우선순위 4: 구역 기반 속도 제한
        const zone = inputs.fusedState.currentZone;
        if (zone.maxSpeed < this.currentSpeed) {
            return {
                action: SafetyAction.SPEED_LIMIT,
                speedLimit: zone.maxSpeed,
                reason: `${zone.name} 구역 속도 제한`,
                priority: 5
            };
        }

        return {
            action: SafetyAction.CONTINUE,
            reason: '안전 문제 없음',
            priority: 10
        };
    }
}

4.8 진단 커버리지 및 자가 모니터링

안전 센서는 자체 올바른 작동을 지속적으로 검증해야 합니다:

진단 방법 커버리지 간격 감지되는 결함
워치독 타이머 90% 연속 프로세서 정지, 루프 실패
참조 측정 95% 주기당 센서 드리프트, 교정 손실
타당성 검사 80% 판독당 총체적 오류, 연결 해제

// 센서 자가 모니터링 시스템
class SensorDiagnostics {
    performDiagnostics(sensor: SafetySensor): DiagnosticResult {
        const results: DiagnosticCheck[] = [];

        // 워치독 확인
        results.push(this.checkWatchdog(sensor));

        // 참조 측정
        if (sensor.hasReference) {
            results.push(this.checkReference(sensor));
        }

        // 타당성 검사
        results.push(this.checkPlausibility(sensor));

        // 중복 센서와 교차 비교
        if (sensor.redundantPair) {
            results.push(this.crossCompare(sensor, sensor.redundantPair));
        }

        // 진단 커버리지 계산
        const coverage = this.calculateDC(results);

        return {
            sensorId: sensor.id,
            healthy: results.every(r => r.passed),
            diagnosticCoverage: coverage,
            checks: results
        };
    }

    checkPlausibility(sensor: SafetySensor): DiagnosticCheck {
        const reading = sensor.getReading();

        // 물리적 한계 내 확인
        if (reading.value < sensor.physicalMin ||
            reading.value > sensor.physicalMax) {
            return {
                name: '타당성',
                passed: false,
                detail: `값 ${reading.value}이 물리적 범위를 벗어남`
            };
        }

        // 변화율 확인
        const previousReading = sensor.getPreviousReading();
        const rateOfChange = Math.abs(reading.value - previousReading.value) /
                             (reading.timestamp - previousReading.timestamp);

        if (rateOfChange > sensor.maxRateOfChange) {
            return {
                name: '타당성',
                passed: false,
                detail: `변화율 ${rateOfChange}이 최대값 초과`
            };
        }

        return { name: '타당성', passed: true };
    }
}

핵심 요점

  • 여러 센서 모달리티(LiDAR, ToF, 초음파, 비전)가 상호 보완적인 안전 커버리지를 제공합니다
  • 힘-토크 모니터링은 동적 모델 잔차를 통해 충돌 감지를 가능하게 합니다
  • 비전 시스템은 인간 감지, 자세 추정, 제스처 인식을 위한 풍부한 이해를 제공합니다
  • 속도 및 분리 모니터링(ISO/TS 15066)은 실시간으로 보호 거리를 계산합니다
  • 센서 융합은 신뢰성을 향상시키지만 불일치를 보수적으로 처리해야 합니다
  • 안전 감독자는 모든 모니터링을 통합하고 100Hz 이상에서 실시간 결정을 내립니다
  • 높은 커버리지의 자가 진단이 센서 신뢰성을 보장합니다
  • 중복성과 교차 비교가 단일 센서 고장을 감지합니다
