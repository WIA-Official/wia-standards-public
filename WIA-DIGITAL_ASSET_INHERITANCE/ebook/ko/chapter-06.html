2. State Machine 기반 상속 프로세스

WIA는 10단계 State Machine으로 상속 프로세스를 관리합니다:

// WIA 상속 프로토콜 State Machine
enum InheritanceState {
  // 1. 초기화
  INITIATED = "initiated",                // 상속 케이스 생성

  // 2. 사망 확인
  DEATH_NOTIFIED = "death_notified",      // 행안부 사망신고 확인

  // 3. 유족 확인
  BENEFICIARY_PENDING = "beneficiary_pending",  // 유족 신원 확인 중
  BENEFICIARY_VERIFIED = "beneficiary_verified", // 유족 확인 완료

  // 4. 자산 검색
  ASSET_DISCOVERY = "asset_discovery",     // 디지털 자산 탐색 중
  ASSET_DISCOVERED = "asset_discovered",   // 자산 탐색 완료

  // 5. 세무 처리
  TAX_CALCULATING = "tax_calculating",     // 상속세 계산 중
  TAX_CALCULATED = "tax_calculated",       // 상속세 산정 완료
  TAX_PENDING = "tax_pending",             // 납세 대기
  TAX_PAID = "tax_paid",                   // 납세 완료

  // 6. 법적 확인
  LEGAL_CHECK = "legal_check",             // 법원 상속 포기/한정승인 확인
  LEGAL_CONFIRMED = "legal_confirmed",     // 법적 확인 완료

  // 7. 자산 이전
  TRANSFER_PENDING = "transfer_pending",   // 이전 대기
  TRANSFER_IN_PROGRESS = "transfer_in_progress", // 이전 진행 중
  TRANSFER_COMPLETED = "transfer_completed",     // 이전 완료

  // 8. 이의 기간
  OBJECTION_PERIOD = "objection_period",   // 30일 이의신청 기간

  // 9. 완료
  COMPLETED = "completed",                 // 상속 절차 완료

  // 10. 예외 상태
  DISPUTED = "disputed",                   // 분쟁 발생
  RENOUNCED = "renounced",                 // 상속 포기
  LIMITED_ACCEPTANCE = "limited_acceptance", // 한정승인
  SUSPENDED = "suspended",                 // 일시 중단 (법원 명령 등)
  FAILED = "failed"                        // 프로세스 실패
}

3. Klaytn 스마트 계약

자산 이전을 자동으로 실행하는 Klaytn 기반 스마트 계약:

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title WIA Digital Asset Inheritance Smart Contract
 * @author WIA (World Certification Industry Association)
 * @notice 디지털 자산 상속을 자동 실행하는 스마트 계약
 */
contract WIAInheritance {
    // 상속 케이스 구조체
    struct InheritanceCase {
        bytes32 caseId;
        bytes32 deceasedHash;      // 고인 주민번호 SHA-256
        uint256 totalAssetValue;    // 총 자산 가치 (KRW)
        uint256 taxAmount;          // 상속세 (KRW)
        uint256 createdAt;
        InheritanceState state;
        bool taxPaid;
        bool disputed;
    }

    // 상속인 구조체
    struct Beneficiary {
        bytes32 identifierHash;     // 주민번호 SHA-256
        string name;
        uint256 shareRatio;         // 지분율 (1000 = 100%)
        bool verified;
        bool approved;
    }

    // 이벤트
    event InheritanceCreated(bytes32 indexed caseId, bytes32 deceasedHash);
    event BeneficiaryAdded(bytes32 indexed caseId, bytes32 beneficiaryHash);
    event StateChanged(bytes32 indexed caseId, InheritanceState newState);
    event TaxPaid(bytes32 indexed caseId, uint256 amount);
    event AssetTransferred(bytes32 indexed caseId, bytes32 beneficiaryHash, uint256 amount);
    event DisputeRaised(bytes32 indexed caseId, string reason);

    // 상속 케이스 저장소
    mapping(bytes32 => InheritanceCase) public inheritanceCases;
    mapping(bytes32 => Beneficiary[]) public beneficiaries;

    // WIA 인증 주소 (행안부, 국세청, 법원 등)
    mapping(address => bool) public authorizedGovernment;
    address public wiaAdmin;

    modifier onlyWIA() {
        require(msg.sender == wiaAdmin, "WIA admin only");
        _;
    }

    modifier onlyAuthorized() {
        require(authorizedGovernment[msg.sender] || msg.sender == wiaAdmin, "Unauthorized");
        _;
    }

    constructor() {
        wiaAdmin = msg.sender;
    }

    /**
     * @notice 상속 케이스 생성
     * @param _caseId 상속 케이스 ID
     * @param _deceasedHash 고인 주민번호 해시
     * @param _totalAssetValue 총 자산 가치
     */
    function createInheritance(
        bytes32 _caseId,
        bytes32 _deceasedHash,
        uint256 _totalAssetValue
    ) external onlyAuthorized {
        require(inheritanceCases[_caseId].caseId == 0, "Case already exists");

        inheritanceCases[_caseId] = InheritanceCase({
            caseId: _caseId,
            deceasedHash: _deceasedHash,
            totalAssetValue: _totalAssetValue,
            taxAmount: 0,
            createdAt: block.timestamp,
            state: InheritanceState.INITIATED,
            taxPaid: false,
            disputed: false
        });

        emit InheritanceCreated(_caseId, _deceasedHash);
    }

    /**
     * @notice 상속인 추가 (행안부 확인 완료 후)
     * @param _caseId 상속 케이스 ID
     * @param _beneficiaryHash 상속인 주민번호 해시
     * @param _name 상속인 이름
     * @param _shareRatio 지분율 (1000 = 100%)
     */
    function addBeneficiary(
        bytes32 _caseId,
        bytes32 _beneficiaryHash,
        string memory _name,
        uint256 _shareRatio
    ) external onlyAuthorized {
        require(inheritanceCases[_caseId].caseId != 0, "Case not found");

        beneficiaries[_caseId].push(Beneficiary({
            identifierHash: _beneficiaryHash,
            name: _name,
            shareRatio: _shareRatio,
            verified: true,
            approved: false
        }));

        emit BeneficiaryAdded(_caseId, _beneficiaryHash);
    }

    /**
     * @notice 상속세 기록 (국세청 연동)
     * @param _caseId 상속 케이스 ID
     * @param _taxAmount 상속세 금액
     */
    function recordTax(bytes32 _caseId, uint256 _taxAmount) external onlyAuthorized {
        inheritanceCases[_caseId].taxAmount = _taxAmount;
        changeState(_caseId, InheritanceState.TAX_CALCULATED);
    }

    /**
     * @notice 상속세 납부 확인
     * @param _caseId 상속 케이스 ID
     */
    function confirmTaxPayment(bytes32 _caseId) external onlyAuthorized {
        inheritanceCases[_caseId].taxPaid = true;
        changeState(_caseId, InheritanceState.TAX_PAID);
        emit TaxPaid(_caseId, inheritanceCases[_caseId].taxAmount);
    }

    /**
     * @notice 자산 이전 실행 (자동)
     * @param _caseId 상속 케이스 ID
     */
    function executeTransfer(bytes32 _caseId) external onlyAuthorized {
        InheritanceCase storage iCase = inheritanceCases[_caseId];
        require(iCase.taxPaid, "Tax not paid");
        require(iCase.state == InheritanceState.TRANSFER_PENDING, "Invalid state");

        Beneficiary[] storage bens = beneficiaries[_caseId];

        // 모든 상속인이 승인했는지 확인
        for (uint i = 0; i < bens.length; i++) {
            require(bens[i].approved, "Not all beneficiaries approved");
        }

        // 자산 이전 (실제 이전은 오프체인에서 WIA API로 실행)
        changeState(_caseId, InheritanceState.TRANSFER_IN_PROGRESS);

        for (uint i = 0; i < bens.length; i++) {
            uint256 amount = (iCase.totalAssetValue * bens[i].shareRatio) / 1000;
            emit AssetTransferred(_caseId, bens[i].identifierHash, amount);
        }

        changeState(_caseId, InheritanceState.TRANSFER_COMPLETED);
        changeState(_caseId, InheritanceState.OBJECTION_PERIOD);
    }

    /**
     * @notice 이의 제기
     * @param _caseId 상속 케이스 ID
     * @param _reason 이의 사유
     */
    function raiseDispute(bytes32 _caseId, string memory _reason) external {
        inheritanceCases[_caseId].disputed = true;
        changeState(_caseId, InheritanceState.DISPUTED);
        emit DisputeRaised(_caseId, _reason);
    }

    /**
     * @notice 상태 변경
     */
    function changeState(bytes32 _caseId, InheritanceState _newState) internal {
        inheritanceCases[_caseId].state = _newState;
        emit StateChanged(_caseId, _newState);
    }

    /**
     * @notice 정부 기관 권한 부여
     */
    function authorizeGovernment(address _gov) external onlyWIA {
        authorizedGovernment[_gov] = true;
    }
}

5. 국세청 홈택스 연동

상속세 자동 계산 및 전자 신고 시스템:

💰 상속세 계산 로직 (민법 및 상속세법 적용)

1단계: 총 상속 재산 평가

  • 금융자산: 사망일 잔액
  • 암호화폐: WIA 기준 3거래소 평균가 (사망일 종가)
  • 게임 아이템: 거래소 시세 또는 감정평가
  • SNS 계정: 수익화 가능 시 시장가치
  • 클라우드 데이터: 개인용은 0원, 상업용은 시장가치

2단계: 상속세 공제

  • 기초공제: 2억원
  • 배우자 공제: 최소 5억원 (법정상속분 한도 내)
  • 자녀 공제: 1인당 5,000만원 (미성년자는 추가)
  • 금융재산 공제: 순금융재산 2억원 한도
  • 부채 및 장례비용 공제

3단계: 세율 적용 (누진세)

1억원 이하10%
1억~5억원20%
5억~10억원30%
10억~30억원40%
30억원 초과50%

실제 사례 계산

  • 총 자산: 3억 5,500만원
  • 배우자 공제: 1억원
  • 자녀 공제: 1억원 (2명 × 5,000만원)
  • 기초공제: 2억원 → 과세표준: 5,500만원
  • 세액: 5,500만원 × 10% = 550만원
  • 신고기한: 사망일로부터 6개월 이내

🚨 상속 포기 및 한정승인 처리

상속 포기 (민법 제1041조)

  • 상속 개시를 안 날로부터 3개월 이내 가정법원 신고
  • WIA 시스템이 법원 전산망에서 자동 확인
  • 포기 확인 시 해당 상속인 자동 제외
  • 차순위 상속인에게 자동 통지

한정승인 (민법 제1019조)

  • 상속재산 한도 내에서만 채무 승계
  • 법원 심판 결과 WIA 시스템 반영
  • 디지털 자산은 통상 채무 없으므로 영향 적음
  • 암호화폐 대출 등 특수 상황 고려

실제 처리 예

  • 홍길동 씨 자녀 중 1명이 상속 포기
  • 법원 신고 후 2시간 내 WIA 시스템 반영
  • 지분율 자동 재계산: 배우자 50% → 60%, 나머지 자녀 25% → 40%
  • 상속세도 자동 재계산
