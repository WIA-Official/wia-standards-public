8.4.2 신한은행 AI 상속 플래너 구현

// 신한은행 AI 상속 플래너 (simplified)
import OpenAI from 'openai';

class AIInheritancePlanner {
  private openai: OpenAI;

  constructor(apiKey: string) {
    this.openai = new OpenAI({ apiKey });
  }

  async generatePlan(customer: CustomerProfile): Promise {
    const prompt = `
고객 정보:
- 나이: ${customer.age}세
- 총 자산: ₩${customer.totalAssets.toLocaleString()}
- 가족 구성: ${customer.family.map(f => f.relationship).join(', ')}
- 부동산: ${customer.realEstate.length}건
- 금융 자산: ₩${customer.financialAssets.toLocaleString()}

한국 상속법과 세법을 고려하여 최적의 상속 계획을 수립하세요.
- 상속세 절감 방안 포함
- 각 상속인별 분배 비율 및 근거
- 법적 주의사항
- 실행 단계별 가이드
`;

    const response = await this.openai.chat.completions.create({
      model: 'gpt-4-turbo',
      messages: [
        { role: 'system', content: '당신은 한국 상속 전문가입니다.' },
        { role: 'user', content: prompt }
      ],
      temperature: 0.3
    });

    const aiSuggestion = response.choices[0].message.content;

    // AI 제안을 구조화된 상속 계획으로 변환
    const plan = this.parseAIPlan(aiSuggestion, customer);

    // 법률 전문가 검토 플래그
    plan.requiresLegalReview = this.needsLegalReview(plan);

    return plan;
  }

  private needsLegalReview(plan: InheritancePlan): boolean {
    // 복잡한 케이스는 법률 전문가 검토 필요
    return plan.totalAssets > 500000000 || // 5억 이상
           plan.beneficiaries.length > 5 ||
           plan.hasRealEstate;
  }
}
