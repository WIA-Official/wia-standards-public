아키텍처

# 연합 학습 기반 의료 영상 분석 시스템

class FederatedMedicalImagingSystem:
    """
    연합 학습 + 차등 프라이버시를 결합한
    병원 연합 의료 영상 분석 시스템
    """

    def __init__(self, hospitals: List[Hospital], central_server: Server):
        self.hospitals = hospitals
        self.server = central_server

        # 프라이버시 설정
        self.dp_epsilon = 1.0  # 라운드당 프라이버시 예산
        self.dp_delta = 1e-5
        self.clip_norm = 1.0

        # 연합 학습 설정
        self.local_epochs = 5
        self.batch_size = 32
        self.num_rounds = 100

    def train_round(self, round_num: int):
        """연합 학습 라운드 실행"""

        # 1. 각 병원에 현재 글로벌 모델 배포
        global_model = self.server.get_global_model()

        # 2. 각 병원에서 로컬 훈련 (병렬 실행)
        local_updates = []
        for hospital in self.hospitals:
            update = hospital.local_train(
                model=global_model,
                epochs=self.local_epochs,
                batch_size=self.batch_size,
                dp_config={
                    'epsilon': self.dp_epsilon,
                    'delta': self.dp_delta,
                    'clip_norm': self.clip_norm
                }
            )
            local_updates.append(update)

        # 3. 안전한 집계 (Secure Aggregation)
        aggregated_update = self.server.secure_aggregate(
            local_updates,
            weights=[h.num_samples for h in self.hospitals]
        )

        # 4. 글로벌 모델 업데이트
        self.server.update_global_model(aggregated_update)

        # 5. 검증 세트에서 성능 평가
        metrics = self.server.evaluate()

        return {
            'round': round_num,
            'auc': metrics['auc'],
            'sensitivity': metrics['sensitivity'],
            'specificity': metrics['specificity']
        }


class Hospital:
    """개별 병원의 로컬 훈련 모듈"""

    def __init__(self, hospital_id: str, local_data_path: str):
        self.hospital_id = hospital_id
        self.data_loader = MedicalImageLoader(local_data_path)
        self.num_samples = len(self.data_loader)

    def local_train(self, model, epochs, batch_size, dp_config):
        """
        로컬 데이터로 모델 훈련 (DP 적용)
        """
        # DP-SGD 옵티마이저 사용
        optimizer = DPAdam(
            model.parameters(),
            lr=0.001,
            noise_multiplier=self._compute_noise_multiplier(dp_config),
            max_grad_norm=dp_config['clip_norm']
        )

        # 로컬 훈련
        for epoch in range(epochs):
            for batch in self.data_loader.get_batches(batch_size):
                # 그래디언트 계산
                loss = model.compute_loss(batch)
                loss.backward()

                # DP-SGD: 클리핑 + 노이즈 추가
                optimizer.step()
                optimizer.zero_grad()

        # 모델 업데이트 (델타) 반환
        return model.get_update_delta()

Cross-Bank Fraud Detection with MPC

배경 및 목표

금융사기는 여러 금융기관에 걸쳐 발생하는 경우가 많습니다. 하지만 고객 정보 보호 때문에 은행 간 데이터 공유가 어렵습니다. MPC를 활용하여 고객 데이터를 공유하지 않으면서 사기 패턴을 탐지합니다.

시스템 구조

# MPC 기반 은행 간 사기 탐지 시스템

class CrossBankFraudDetection:
    """
    안전한 다자간 연산을 이용한
    은행 간 사기 탐지 협력 시스템
    """

    def __init__(self, banks: List[Bank], mpc_engine: MPCEngine):
        self.banks = banks
        self.mpc = mpc_engine

    def private_blacklist_intersection(self):
        """
        Private Set Intersection으로 블랙리스트 교차 검증
        - 각 은행의 블랙리스트는 비공개 유지
        - 공통으로 등장하는 의심 계좌만 식별
        """
        # 각 은행이 블랙리스트를 해시하여 비밀 분산
        hashed_lists = []
        for bank in self.banks:
            hashed = bank.hash_blacklist()
            shares = self.mpc.secret_share(hashed)
            hashed_lists.append(shares)

        # PSI 프로토콜 실행
        intersection = self.mpc.private_set_intersection(hashed_lists)

        return intersection

    def federated_fraud_model_training(self):
        """
        연합 학습으로 사기 탐지 모델 공동 훈련
        """
        # 초기 모델 생성
        global_model = FraudDetectionModel()

        for round in range(self.num_rounds):
            # 각 은행에서 로컬 훈련
            local_models = []
            for bank in self.banks:
                local_model = bank.train_local_model(
                    global_model,
                    differential_privacy=True,
                    epsilon=0.5
                )
                local_models.append(local_model)

            # Secure Aggregation으로 모델 집계
            global_model = self.mpc.secure_model_aggregation(local_models)

        return global_model

    def privacy_preserving_transaction_scoring(self, transaction):
        """
        암호화된 상태로 거래 위험도 점수 계산
        """
        # 거래 정보 암호화 (동형 암호화)
        encrypted_features = self.he_encrypt(transaction.features)

        # 각 은행의 로컬 모델로 점수 계산 (MPC)
        partial_scores = []
        for bank in self.banks:
            # 암호화된 입력에 대해 추론
            partial_score = bank.encrypted_inference(encrypted_features)
            partial_scores.append(partial_score)

        # 점수 집계 (MPC)
        final_score = self.mpc.secure_average(partial_scores)

        return final_score

🏛️
차등 프라이버시 기반 인구통계 시스템
Differentially Private Census Statistics

배경 및 목표

정부 통계청에서 인구조사 데이터를 공개하되, 개인이 식별되지 않도록 합니다. 소규모 지역의 민감한 통계(질병 유병률, 소득 등)도 안전하게 공개합니다. 

# 차등 프라이버시 기반 공공 통계 시스템

class DPCensusSystem:
    """
    차등 프라이버시를 적용한 인구조사 통계 시스템
    """

    def __init__(self, raw_data: DataFrame, total_epsilon: float = 10.0):
        self.raw_data = raw_data
        self.total_epsilon = total_epsilon
        self.spent_epsilon = 0.0

    def publish_count_query(self, query: str, epsilon: float):
        """
        카운트 쿼리 응답 (라플라스 메커니즘)

        예: "서울시 강남구 20대 여성 인구수"
        """
        if self.spent_epsilon + epsilon > self.total_epsilon:
            raise PrivacyBudgetExceeded("프라이버시 예산 초과")

        # 실제 카운트 계산
        true_count = self._execute_query(query)

        # 라플라스 노이즈 추가 (민감도 = 1)
        sensitivity = 1
        scale = sensitivity / epsilon
        noisy_count = true_count + np.random.laplace(0, scale)

        # 음수 방지 및 반올림
        result = max(0, round(noisy_count))

        # 예산 차감
        self.spent_epsilon += epsilon

        return result

    def publish_histogram(self, column: str, bins: List, epsilon: float):
        """
        히스토그램 공개 (병렬 구성 활용)

        예: 연령대별 인구 분포
        """
        # 병렬 구성: 각 빈은 독립적 (총 epsilon = max epsilon)
        per_bin_epsilon = epsilon

        histogram = {}
        for bin_range in bins:
            query = f"{column} BETWEEN {bin_range[0]} AND {bin_range[1]}"
            histogram[bin_range] = self.publish_count_query(query, per_bin_epsilon)

        # 병렬 구성에서는 epsilon 한 번만 차감
        self.spent_epsilon -= (len(bins) - 1) * per_bin_epsilon

        return histogram

    def publish_mean_query(self, column: str, bounds: Tuple, epsilon: float):
        """
        평균 쿼리 응답 (클리핑 + 노이즈)

        예: "평균 가구 소득"
        """
        lower, upper = bounds

        # 데이터 클리핑 (민감도 제한)
        clipped_data = self.raw_data[column].clip(lower, upper)

        # 실제 평균 계산
        true_mean = clipped_data.mean()
        n = len(clipped_data)

        # 평균의 민감도 = (upper - lower) / n
        sensitivity = (upper - lower) / n
        scale = sensitivity / epsilon

        noisy_mean = true_mean + np.random.laplace(0, scale)

        self.spent_epsilon += epsilon

        return noisy_mean

# 프라이버시 보장 테스트 예시

class PrivacyTest:
    """프라이버시 보존 시스템 테스트 프레임워크"""

    def test_differential_privacy_guarantee(self, mechanism, epsilon, num_trials=1000):
        """
        차등 프라이버시 보장 실험적 검증

        인접 데이터셋에서 출력 분포 비교
        """
        # 인접 데이터셋 생성 (한 레코드 차이)
        dataset1 = self.create_dataset(size=1000)
        dataset2 = self.modify_one_record(dataset1)

        # 각 데이터셋에서 출력 분포 수집
        outputs1 = [mechanism(dataset1) for _ in range(num_trials)]
        outputs2 = [mechanism(dataset2) for _ in range(num_trials)]

        # 분포 비율이 exp(epsilon) 이내인지 확인
        for output in set(outputs1 + outputs2):
            prob1 = outputs1.count(output) / num_trials
            prob2 = outputs2.count(output) / num_trials

            if prob2 > 0:
                ratio = prob1 / prob2
                assert ratio <= np.exp(epsilon), \
                    f"Privacy violation: ratio {ratio} > exp({epsilon})"

        print(f"DP test passed with epsilon={epsilon}")

    def test_mpc_correctness(self, mpc_function, plain_function, inputs):
        """
        MPC 프로토콜 정확성 검증

        평문 계산과 MPC 결과 비교
        """
        # 평문 계산
        expected = plain_function(*inputs)

        # MPC 계산
        actual = mpc_function(*inputs)

        assert np.allclose(expected, actual, rtol=1e-5), \
            f"MPC result mismatch: {expected} vs {actual}"

        print("MPC correctness test passed")
