5.1 Introduction to Federated Learning

Federated Learning (FL) is a machine learning paradigm that trains models across decentralized data sources without exchanging raw data. Instead of moving data to a central server, the model travels to the data. Each participant trains the model locally on their own data and shares only model updates (gradients or weights), which are aggregated to improve a global model.

Introduced by Google in 2016 for mobile keyboard predictions, federated learning has since become a cornerstone of privacy-preserving machine learning. It addresses a fundamental tension in modern AI: the need for large datasets to train accurate models versus the privacy concerns and regulatory constraints that prevent data centralization.

Federated Learning
A distributed machine learning approach where:

1. Data stays local: Training data never leaves the device or organization
2. Model comes to data: A global model is distributed to participants
3. Only updates shared: Participants share model updates, not raw data
4. Aggregation server: A central server (or decentralized network) combines updates

5.3 Federated Averaging (FedAvg)

Federated Averaging is the foundational algorithm for federated learning, introduced by McMahan et al. at Google. It balances communication efficiency with convergence by performing multiple local training steps before aggregation.

FedAvg Algorithm
Wt+1 = Σk=1K (nk/n) × Wkt+1

Weighted average based on local dataset sizes

import numpy as np
from typing import List, Dict, Tuple
import copy

class FederatedAveraging:
    """Implementation of the FedAvg algorithm."""

    def __init__(self, model, learning_rate=0.01, local_epochs=5):
        self.global_model = model
        self.learning_rate = learning_rate
        self.local_epochs = local_epochs

    def train_round(self, clients: List['FederatedClient']) -> Dict:
        """Execute one round of federated training."""

        # 1. Distribute global model to selected clients
        global_weights = self.get_model_weights()

        # 2. Local training on each client
        client_updates = []
        total_samples = 0

        for client in clients:
            # Send global model to client
            client.receive_global_model(global_weights)

            # Client trains locally
            local_weights, num_samples = client.local_train(
                epochs=self.local_epochs,
                learning_rate=self.learning_rate
            )

            client_updates.append({
                'weights': local_weights,
                'num_samples': num_samples
            })
            total_samples += num_samples

        # 3. Aggregate client updates (FedAvg)
        new_weights = self.aggregate(client_updates, total_samples)

        # 4. Update global model
        self.set_model_weights(new_weights)

        return {
            'num_clients': len(clients),
            'total_samples': total_samples
        }

    def aggregate(self, client_updates: List[Dict], total_samples: int) -> List[np.ndarray]:
        """Weighted average of client model weights."""

        # Initialize aggregated weights with zeros
        aggregated_weights = [
            np.zeros_like(w) for w in client_updates[0]['weights']
        ]

        # Weighted sum
        for update in client_updates:
            weight_factor = update['num_samples'] / total_samples

            for i, layer_weights in enumerate(update['weights']):
                aggregated_weights[i] += weight_factor * layer_weights

        return aggregated_weights

    def get_model_weights(self) -> List[np.ndarray]:
        """Get current global model weights."""
        return [layer.copy() for layer in self.global_model.get_weights()]

    def set_model_weights(self, weights: List[np.ndarray]):
        """Set global model weights."""
        self.global_model.set_weights(weights)


class FederatedClient:
    """Client in federated learning system."""

    def __init__(self, client_id: str, local_data: Tuple[np.ndarray, np.ndarray], model):
        self.client_id = client_id
        self.X, self.y = local_data
        self.model = model
        self.num_samples = len(self.X)

    def receive_global_model(self, weights: List[np.ndarray]):
        """Receive and set global model weights."""
        self.model.set_weights(weights)

    def local_train(self, epochs: int, learning_rate: float) -> Tuple[List[np.ndarray], int]:
        """Train model locally on client's data."""

        for epoch in range(epochs):
            # Simple gradient descent (in practice, use proper optimizer)
            predictions = self.model.forward(self.X)
            loss = self.model.compute_loss(predictions, self.y)
            gradients = self.model.compute_gradients(self.X, self.y)

            # Update weights
            new_weights = []
            for w, g in zip(self.model.get_weights(), gradients):
                new_weights.append(w - learning_rate * g)
            self.model.set_weights(new_weights)

        return self.model.get_weights(), self.num_samples


# Example: Simple neural network for demonstration
class SimpleNN:
    """Simple 2-layer neural network."""

    def __init__(self, input_dim, hidden_dim, output_dim):
        # Initialize weights
        self.W1 = np.random.randn(input_dim, hidden_dim) * 0.01
        self.b1 = np.zeros(hidden_dim)
        self.W2 = np.random.randn(hidden_dim, output_dim) * 0.01
        self.b2 = np.zeros(output_dim)

    def get_weights(self):
        return [self.W1.copy(), self.b1.copy(), self.W2.copy(), self.b2.copy()]

    def set_weights(self, weights):
        self.W1, self.b1, self.W2, self.b2 = [w.copy() for w in weights]

    def forward(self, X):
        self.z1 = X @ self.W1 + self.b1
        self.a1 = np.maximum(0, self.z1)  # ReLU
        self.z2 = self.a1 @ self.W2 + self.b2
        return self.z2

    def compute_loss(self, predictions, y):
        return np.mean((predictions - y) ** 2)

    def compute_gradients(self, X, y):
        # Simplified gradient computation
        m = X.shape[0]
        predictions = self.forward(X)

        dz2 = (predictions - y) / m
        dW2 = self.a1.T @ dz2
        db2 = np.sum(dz2, axis=0)

        da1 = dz2 @ self.W2.T
        dz1 = da1 * (self.z1 > 0)  # ReLU derivative
        dW1 = X.T @ dz1
        db1 = np.sum(dz1, axis=0)

        return [dW1, db1, dW2, db2]

5.4 Types of Federated Learning

Training across millions of mobile devices or IoT sensors. Characterized by massive scale, unreliable connectivity, and heterogeneous compute capabilities.

Training across a small number of organizations (hospitals, banks, companies). Characterized by reliable participants with substantial data and compute resources.

5.5 Challenges in Federated Learning

5.5.1 Statistical Heterogeneity (Non-IID Data)

Real-world federated data is often non-identically distributed (non-IID). Each client's local data may follow a different distribution, causing model divergence and slow convergence.

import numpy as np

def simulate_non_iid_data(n_clients, n_classes, samples_per_client, skewness=0.8):
    """
    Simulate non-IID data distribution across clients.
    Higher skewness means more non-IID (each client has fewer classes).
    """
    client_data = []

    for client_id in range(n_clients):
        # Each client has preference for certain classes
        # based on Dirichlet distribution
        class_probs = np.random.dirichlet([skewness] * n_classes)

        # Sample class labels according to this distribution
        labels = np.random.choice(
            n_classes,

            'features': features,
            'labels': labels,
            'class_distribution': class_probs
        })

        print(f"Client {client_id}: {np.bincount(labels, minlength=n_classes)}")

5.5.2 Communication Efficiency

Communication is often the bottleneck in federated learning, especially for cross-device settings with limited bandwidth.

Communication Reduction Techniques

  • Gradient Compression: Quantize gradients to fewer bits
  • Sparsification: Send only top-k gradients
  • Local SGD: More local steps, fewer communication rounds
  • Gradient Accumulation: Aggregate updates before sending
  • Model Pruning: Train smaller models
import numpy as np

def topk_sparsification(gradients: np.ndarray, k_fraction: float = 0.1) -> Tuple[np.ndarray, np.ndarray]:
    """
    Top-k gradient sparsification.
    Only send the largest k% of gradients.
    """
    flat_grad = gradients.flatten()
    k = max(1, int(len(flat_grad) * k_fraction))

    # Find top-k indices by absolute value
    top_k_indices = np.argpartition(np.abs(flat_grad), -k)[-k:]

    # Create sparse representation
    values = flat_grad[top_k_indices]

    # Compression ratio
    original_size = flat_grad.nbytes
    compressed_size = values.nbytes + top_k_indices.nbytes
    print(f"Compression ratio: {original_size / compressed_size:.2f}x")

    return top_k_indices, values

def quantize_gradients(gradients: np.ndarray, bits: int = 8) -> np.ndarray:
    """
    Quantize gradients to fewer bits.
    """
    # Normalize to [0, 1]
    g_min, g_max = gradients.min(), gradients.max()
    normalized = (gradients - g_min) / (g_max - g_min + 1e-8)

    # Quantize to n bits
    levels = 2 ** bits
    quantized = np.round(normalized * (levels - 1)).astype(np.uint8)

    # Compression: 32-bit float -> 8-bit int = 4x
    return quantized, g_min, g_max

5.5.3 Privacy Attacks on Federated Learning

While federated learning improves privacy by not sharing raw data, model updates can still leak sensitive information.

Privacy Threats in Federated Learning

  • Gradient Inversion: Reconstruct training data from gradients
  • Membership Inference: Determine if a record was in training data
  • Model Inversion: Infer sensitive attributes from model
  • Property Inference: Learn unintended properties about training data

5.6 Privacy-Enhanced Federated Learning

5.6.1 Differential Privacy in FL

import numpy as np

class DifferentiallyPrivateFedAvg:
    """FedAvg with differential privacy guarantees."""

    def __init__(self, model, learning_rate, noise_multiplier, max_grad_norm, delta=1e-5):
        self.model = model
        self.learning_rate = learning_rate
        self.noise_multiplier = noise_multiplier
        self.max_grad_norm = max_grad_norm
        self.delta = delta

    def clip_gradients(self, gradients: List[np.ndarray]) -> List[np.ndarray]:
        """Clip gradients to bounded L2 norm."""
        # Compute total L2 norm
        total_norm = np.sqrt(sum(np.sum(g ** 2) for g in gradients))

        # Clip if necessary
        clip_factor = min(1.0, self.max_grad_norm / (total_norm + 1e-6))
        clipped = [g * clip_factor for g in gradients]

        return clipped

    def add_noise(self, aggregated_gradients: List[np.ndarray], n_clients: int) -> List[np.ndarray]:
        """Add calibrated Gaussian noise for differential privacy."""
        noisy_gradients = []

        for grad in aggregated_gradients:
            # Noise scale based on sensitivity and privacy budget
            noise_scale = (self.noise_multiplier * self.max_grad_norm) / n_clients
            noise = np.random.normal(0, noise_scale, grad.shape)
            noisy_gradients.append(grad + noise)

        return noisy_gradients

    def private_aggregate(self, client_updates: List[Dict], total_samples: int) -> List[np.ndarray]:
        """Aggregate with differential privacy."""

        # Clip each client's update
        clipped_updates = []
        for update in client_updates:
            clipped = self.clip_gradients(update['gradients'])
            clipped_updates.append({
                'gradients': clipped,
                'num_samples': update['num_samples']
            })

        # Weighted average
        aggregated = [np.zeros_like(g) for g in clipped_updates[0]['gradients']]
        for update in clipped_updates:
            weight = update['num_samples'] / total_samples
            for i, grad in enumerate(update['gradients']):
                aggregated[i] += weight * grad

        # Add noise
        n_clients = len(client_updates)
        noisy_aggregated = self.add_noise(aggregated, n_clients)

        return noisy_aggregated

# Privacy accounting
def compute_epsilon(noise_multiplier, n_rounds, delta=1e-5):
    """
    Compute epsilon using the moments accountant.
    Simplified version - use tensorflow_privacy for production.
    """
    # Approximate using advanced composition
    epsilon = np.sqrt(2 * n_rounds * np.log(1/delta)) / noise_multiplier
    return epsilon

# Example: Privacy budget tracking
noise_multiplier = 1.0
n_rounds = 100
epsilon = compute_epsilon(noise_multiplier, n_rounds)
print(f"Privacy budget after {n_rounds} rounds: epsilon = {epsilon:.2f}")

5.6.2 Secure Aggregation

Secure aggregation ensures the server learns only the aggregate of client updates, not individual contributions.

import numpy as np
from typing import List, Tuple
import hashlib

class SecureAggregation:
    """
    Simplified secure aggregation protocol.
    Real implementations use cryptographic techniques like
    secret sharing or homomorphic encryption.
    """

    def __init__(self, n_clients: int, threshold: int):
        self.n_clients = n_clients
        self.threshold = threshold  # Minimum clients needed

    def generate_pairwise_masks(self, client_id: int, round_id: int) -> List[np.ndarray]:
        """
        Generate pairwise masking values.
        Client i masks its update with sum of masks to all other clients.
        """
        masks = []
        for other_id in range(self.n_clients):
            if other_id != client_id:
                # Derive seed from client pair and round (simplified)
                seed = hash((min(client_id, other_id),
                            max(client_id, other_id),
                            round_id))

                np.random.seed(seed % (2**32))
                mask = np.random.randn(100)  # Size matches model

                # One client adds, other subtracts (masks cancel in sum)
                if client_id < other_id:
                    masks.append(mask)
                else:
                    masks.append(-mask)

        return masks

    def mask_update(self, client_id: int, update: np.ndarray, round_id: int) -> np.ndarray:
        """Mask client update before sending to server."""
        masks = self.generate_pairwise_masks(client_id, round_id)
        total_mask = sum(masks)
        return update + total_mask

    def aggregate_masked_updates(self, masked_updates: List[np.ndarray]) -> np.ndarray:
        """
        Aggregate masked updates.
        If all clients participate, masks cancel and we get true sum.
        """
        # Masks cancel out: sum of all pairwise masks = 0
        return sum(masked_updates)

# Example: Secure aggregation
n_clients = 5
secure_agg = SecureAggregation(n_clients=n_clients, threshold=3)

# Each client has an update
client_updates = [np.random.randn(100) for _ in range(n_clients)]
print(f"True sum: {sum(client_updates).sum():.4f}")

# Mask updates
masked_updates = [
    secure_agg.mask_update(i, update, round_id=1)
    for i, update in enumerate(client_updates)
]

# Server aggregates masked updates (can't see individual updates)
aggregate = secure_agg.aggregate_masked_updates(masked_updates)
print(f"Secure aggregate: {aggregate.sum():.4f}")  # Should match true sum

5.9 Summary

Federated learning enables machine learning on distributed data without centralization, addressing privacy concerns and regulatory constraints. Combined with differential privacy and secure aggregation, it provides strong privacy guarantees while enabling powerful collaborative learning.

Key Takeaways

  • Data stays local; only model updates are shared
  • FedAvg: Local training + weighted averaging of models
  • Cross-device (billions of phones) vs. cross-silo (organizations)
  • Non-IID data is a major challenge requiring specialized algorithms
  • Communication efficiency is critical for scalability
  • Privacy enhanced with DP, secure aggregation, and MPC
  • Production frameworks: TFF, Flower, FATE, PySyft
