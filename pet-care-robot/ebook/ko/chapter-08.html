8.2.2 디바이스 구현

// src/device.ts
import { WIAPetDevice, FeedingEvent } from '@wia/rob-014';

class SmartPetFeeder extends WIAPetDevice {
  private motorPin: number = 17;
  private sensorPin: number = 27;
  private currentFoodLevel: number = 100; // percentage

  constructor(config) {
    super({
      deviceId: config.deviceId,
      type: 'feeder',
      manufacturer: 'My Pet Tech',
      model: 'SPF-001',
      wiaCompliance: {
        version: '1.0',
        level: 'standard'
      }
    });

    this.initializeHardware();
  }

  private initializeHardware() {
    // GPIO 초기화 (Raspberry Pi)
    // const Gpio = require('onoff').Gpio;
    // this.motor = new Gpio(this.motorPin, 'out');
    // this.sensor = new Gpio(this.sensorPin, 'in');
    console.log('하드웨어 초기화 완료');
  }

  async dispense(amount: { value: number; unit: string }): Promise {
    console.log(`급식 시작: ${amount.value}${amount.unit}`);

    // 1. 모터 구동
    await this.runMotor(amount.value);

    // 2. WIA 표준 이벤트 생성
    const event: FeedingEvent = {
      eventId: this.generateEventId(),
      petId: this.assignedPetId,
      type: 'feeding',
      timestamp: new Date().toISOString(),
      amount: amount,
      deviceId: this.deviceId,
      method: 'automatic'
    };

    // 3. 클라우드에 전송 (MQTT)
    await this.publishEvent('activity', event);

    // 4. 사료 잔량 업데이트
    this.currentFoodLevel = await this.measureFoodLevel();
    await this.updateDeviceState({
      foodLevel: this.currentFoodLevel
    });

    // 5. 사료 부족 시 알림
    if (this.currentFoodLevel < 10) {
      await this.sendAlert({
        type: 'low-food',
        severity: 'medium',
        message: '사료가 부족합니다 (10% 이하)'
      });
    }

    console.log('급식 완료');
  }

  private async runMotor(grams: number): Promise {
    // 모터 구동 시간 계산 (예: 1초에 10g)
    const durationMs = (grams / 10) * 1000;

    // 모터 켜기
    // this.motor.writeSync(1);
    await this.sleep(durationMs);
    // 모터 끄기
    // this.motor.writeSync(0);

    console.log(`모터 ${durationMs}ms 동작`);
  }

  private async measureFoodLevel(): Promise {
    // 초음파 센서로 사료 높이 측정
    // const distance = await this.readUltrasonicSensor();
    // const level = this.convertDistanceToPercentage(distance);

    // 시뮬레이션
    return Math.max(0, this.currentFoodLevel - 5);
  }

  private sleep(ms: number): Promise {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// 디바이스 시작
const config = require('../config/device.json');

  console.log('스마트 급식기 준비 완료');

  // 스케줄된 급식
  feeder.schedules.add({
    times: ['08:00', '18:00'],
    amount: { value: 150, unit: 'grams' }
  });
});

// tests/device.test.ts
import { describe, it, expect } from 'vitest';
import { SmartPetFeeder } from '../src/device';
import { validateSchema } from '@wia/rob-014-validator';

describe('SmartPetFeeder', () => {
  let feeder: SmartPetFeeder;

  beforeEach(() => {
    feeder = new SmartPetFeeder({
      deviceId: 'test-feeder-001'
    });
  });

  it('디바이스 메타데이터가 WIA 스키마를 준수해야 함', () => {
    const metadata = feeder.getMetadata();
    const result = validateSchema('device', metadata);

    expect(result.valid).toBe(true);
    expect(result.errors).toHaveLength(0);
  });

  it('급식 이벤트가 올바른 형식이어야 함', async () => {
    const event = await feeder.createFeedingEvent({
      amount: { value: 150, unit: 'grams' }
    });

    const result = validateSchema('activity-feeding', event);

    expect(result.valid).toBe(true);
    expect(event.type).toBe('feeding');
    expect(event.amount.value).toBe(150);
  });

  it('사료 부족 시 알림을 발송해야 함', async () => {
    feeder.currentFoodLevel = 8; // 8%

    const alerts = [];
    feeder.on('alert', (alert) => alerts.push(alert));

    await feeder.dispense({ value: 50, unit: 'grams' });

    expect(alerts).toHaveLength(1);
    expect(alerts[0].type).toBe('low-food');
    expect(alerts[0].severity).toBe('medium');
  });

  it('MQTT 메시지가 올바른 토픽으로 발행되어야 함', async () => {
    const messages = [];
    feeder.mqtt.on('publish', (topic, message) => {
      messages.push({ topic, message });
    });

    await feeder.dispense({ value: 150, unit: 'grams' });

    expect(messages).toHaveLength(1);
    expect(messages[0].topic).toBe(
      `wia/pets/${feeder.assignedPetId}/activity`
    );
  });
});
