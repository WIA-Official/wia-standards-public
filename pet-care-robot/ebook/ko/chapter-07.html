import * as tf from '@tensorflow/tfjs-node';
import { WIARob014AIModel } from '@wia/rob-014-ai';

class PetBehaviorRecognition extends WIARob014AIModel {
  constructor() {
    super({
      modelId: 'pet-behavior-v2.1',
      version: '2.1.0',
      framework: 'tensorflow',
      inputFormat: 'video-frame',
      outputFormat: 'classification'
    });
  }

  async loadModel() {
    this.model = await tf.loadLayersModel(
      'https://models.wia-rob014.org/behavior/v2.1/model.json'
    );
    console.log('모델 로드 완료');
  }

  async predict(videoFrame) {
    // 전처리
    const tensor = tf.browser.fromPixels(videoFrame)
      .resizeBilinear([224, 224])
      .expandDims(0)
      .div(255.0);

    // 예측
    const prediction = await this.model.predict(tensor);
    const scores = await prediction.data();

    // 결과 해석
    const behaviors = [
      'eating', 'playing', 'sleeping', 'walking',
      'grooming', 'barking', 'sitting', 'lying'
    ];

    const results = behaviors.map((behavior, index) => ({
      behavior,
      confidence: scores[index],
      timestamp: new Date().toISOString()
    })).sort((a, b) => b.confidence - a.confidence);

    // WIA-ROB-014 표준 형식으로 반환
    return {
      modelId: this.modelId,
      version: this.version,
      predictions: results.slice(0, 3), // Top 3
      metadata: {
        inferenceTime: performance.now() - startTime,
        deviceId: this.deviceId
      }
    };
  }
}

7.5.2 자동 사료 주문 통합

// 사료 잔량 모니터링 및 자동 주문
class AutoReorderService {
  constructor(wiaClient, ecommerceAPI) {
    this.wia = wiaClient;
    this.ecommerce = ecommerceAPI;
    this.threshold = 10; // 10% 이하 시 주문
  }

  async monitorFoodLevel(deviceId) {
    const device = await this.wia.devices.get(deviceId);

    // 사료 잔량 확인
    const foodLevel = device.state.foodLevel; // percentage

    if (foodLevel <= this.threshold) {
      await this.reorderFood(device);
    }
  }

  async reorderFood(device) {
    // 선호 사료 조회
    const pet = await this.wia.pets.get(device.assignedPetId);
    const preferredFood = pet.preferences.food;

    // 주문 생성
    const order = await this.ecommerce.createOrder({
      items: [
        {
          productId: preferredFood.productId,
          quantity: 1,
          size: preferredFood.size
        }
      ],
      shippingAddress: pet.owner.address,
      priority: 'standard'
    });

    // 사용자 알림
    await this.wia.notifications.send({
      userId: pet.ownerId,
      type: 'auto-reorder',
      message: `${pet.name}의 사료를 자동 주문했습니다. 주문 번호: ${order.id}`,
      data: {
        orderId: order.id,
        estimatedDelivery: order.estimatedDelivery
      }
    });

    console.log(`자동 주문 완료: ${order.id}`);
  }
}

// 웹훅 수신 서버
const express = require('express');
const crypto = require('crypto');

const app = express();
app.use(express.json());

app.post('/webhooks/wia-rob014', (req, res) => {
  // 서명 검증
  const signature = req.headers['x-wia-signature'];
  const payload = JSON.stringify(req.body);
  const expectedSignature = crypto
    .createHmac('sha256', 'webhook-secret-key-xyz')
    .update(payload)
    .digest('hex');

  if (signature !== expectedSignature) {
    return res.status(401).send('Invalid signature');
  }

  // 이벤트 처리
  const event = req.body;

  switch (event.type) {
    case 'pet.activity.feeding':
      console.log('급식 완료:', event.data);
      updateDashboard(event.data);
      break;

    case 'pet.health.alert':
      console.log('건강 알림:', event.data);
      notifyVeterinarian(event.data);
      break;

    case 'device.status.offline':
      console.log('디바이스 오프라인:', event.data);
      sendMaintenanceAlert(event.data);
      break;
  }

  res.status(200).send('OK');
});

// 여러 반려동물의 활동 통합 조회
const getDailyActivity = async (userId) => {
  // 사용자의 모든 반려동물 조회
  const pets = await wiaClient.pets.list({ ownerId: userId });

  // 각 반려동물의 오늘 활동 조회
  const today = new Date().toISOString().split('T')[0];
  const activities = await Promise.all(
    pets.map(pet =>
      wiaClient.activities.list({
        petId: pet.petId,
        startDate: today,
        endDate: today
      })
    )
  );

  // 반려동물별 요약
  return pets.map((pet, index) => ({
    pet: {
      id: pet.petId,
      name: pet.name,
      species: pet.species
    },
    todayActivity: {
      feeding: activities[index].filter(a => a.type === 'feeding').length,
      exercise: activities[index]
        .filter(a => a.type === 'exercise')
        .reduce((sum, a) => sum + a.duration.value, 0),
      sleep: activities[index]
        .filter(a => a.type === 'sleep')
        .reduce((sum, a) => sum + (
          new Date(a.endTime) - new Date(a.startTime)
        ) / 3600000, 0) // hours
    }
  }));
};
