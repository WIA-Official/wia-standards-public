6.2 TLS/SSL 보안

모든 네트워크 통신은 TLS 1.3 이상으로 암호화해야 합니다.

6.2.1 TLS 구성 요구사항

// TLS 1.3 설정 예제 (Node.js)
import https from 'https';
import fs from 'fs';

const tlsOptions = {
  // 인증서
  cert: fs.readFileSync('/path/to/cert.pem'),
  key: fs.readFileSync('/path/to/key.pem'),
  ca: fs.readFileSync('/path/to/ca.pem'),

  // TLS 버전
  minVersion: 'TLSv1.3',
  maxVersion: 'TLSv1.3',

  // 암호화 스위트 (권장)
  ciphers: [
    'TLS_AES_256_GCM_SHA384',
    'TLS_CHACHA20_POLY1305_SHA256',
    'TLS_AES_128_GCM_SHA256'
  ].join(':'),

  // 보안 옵션
  honorCipherOrder: true,
  sessionTimeout: 300,

  // 클라이언트 인증 (선택)
  requestCert: true,
  rejectUnauthorized: true
};

const server = https.createServer(tlsOptions, (req, res) => {
  // 요청 처리
});

6.4 MQTT 프로토콜

IoT 디바이스는 MQTT over TLS로 통신합니다.

6.4.1 MQTT 토픽 구조

토픽 패턴 방향 설명 QoS
wia/pets/{petId}/activity Device → Cloud 활동 데이터 전송 1
wia/pets/{petId}/health Device → Cloud 건강 데이터 전송 2
wia/devices/{deviceId}/control Cloud → Device 제어 명령 전송 2
wia/devices/{deviceId}/status Device → Cloud 디바이스 상태 전송 1
wia/alerts/{petId} Cloud → App 알림 메시지 1

6.4.2 MQTT 메시지 예제

// 디바이스에서 활동 데이터 발행
const mqtt = require('mqtt');

const client = mqtt.connect('mqtts://mqtt.wia-rob014.org:8883', {
  clientId: 'device-feeder-001',
  username: 'device-feeder-001',
  password: 'device-token-xyz',
  ca: fs.readFileSync('/path/to/ca.crt'),
  cert: fs.readFileSync('/path/to/device.crt'),
  key: fs.readFileSync('/path/to/device.key')
});

client.on('connect', () => {
  console.log('MQTT 연결 성공');

  // 활동 데이터 발행
  const activityData = {
    eventId: "event-b2c3d4e5f6a1b2c3d4e5f6a1b2c3d4e5",
    petId: "pet-a1b2c3d4e5f67890a1b2c3d4e5f67890",
    type: "feeding",
    timestamp: new Date().toISOString(),
    amount: { value: 150, unit: "grams" }
  };

  client.publish(
    'wia/pets/pet-a1b2c3d4e5f67890a1b2c3d4e5f67890/activity',
    JSON.stringify(activityData),
    { qos: 1, retain: false }
  );

  // 제어 명령 구독
  client.subscribe('wia/devices/device-feeder-001/control', { qos: 2 });
});

client.on('message', (topic, message) => {
  const command = JSON.parse(message.toString());
  console.log('제어 명령 수신:', command);

  if (command.action === 'dispense') {
    // 급식 실행
    dispensFood(command.parameters.amount);
  }
});

6.4.3 QoS 레벨

QoS 이름 설명 사용 사례
0 At most once 최대 1번 전송, 확인 없음 환경 센서 데이터 (빈번함)
1 At least once 최소 1번 전송, 중복 가능 활동 데이터, 알림
2 Exactly once 정확히 1번 전송, 중복 없음 제어 명령, 건강 데이터

6.5.1 WebSocket 연결

// 클라이언트에서 WebSocket 연결
const ws = new WebSocket('wss://ws.wia-rob014.org/v1/stream', {
  headers: {
    'Authorization': 'Bearer eyJhbGciOiJSUzI1NiIsInR5cCI6IkpXVCJ9...'
  }
});

ws.on('open', () => {
  console.log('WebSocket 연결됨');

  // 반려동물 데이터 구독
  ws.send(JSON.stringify({
    type: 'subscribe',
    channels: [
      'pets:pet-a1b2c3d4e5f67890a1b2c3d4e5f67890:activity',
      'pets:pet-a1b2c3d4e5f67890a1b2c3d4e5f67890:health',
      'alerts:pet-a1b2c3d4e5f67890a1b2c3d4e5f67890'
    ]
  }));
});

ws.on('message', (data) => {
  const message = JSON.parse(data);

  switch (message.type) {
    case 'activity':
      console.log('활동 데이터:', message.data);
      break;
    case 'health':
      console.log('건강 데이터:', message.data);
      break;
    case 'alert':
      console.log('알림:', message.data);
      showNotification(message.data);
      break;
    case 'heartbeat':
      // 연결 유지
      ws.send(JSON.stringify({ type: 'pong' }));
      break;
  }
});

ws.on('error', (error) => {
  console.error('WebSocket 오류:', error);
});

ws.on('close', (code, reason) => {
  console.log(`WebSocket 종료: ${code} - ${reason}`);
  // 재연결 로직
  setTimeout(() => reconnect(), 5000);
});

6.8 재시도 및 에러 복구

네트워크 장애 시 자동 복구 메커니즘입니다.

6.8.1 지수 백오프 재시도

// 재시도 로직 구현
class RetryableRequest {
  async execute(operation, options = {}) {
    const {
      maxRetries = 5,
      initialDelay = 1000,
      maxDelay = 32000,
      backoffMultiplier = 2
    } = options;

    let delay = initialDelay;

    for (let attempt = 0; attempt <= maxRetries; attempt++) {
      try {
        return await operation();
      } catch (error) {
        if (attempt === maxRetries) {
          throw error;
        }

        // 재시도 가능한 오류인지 확인
        if (!this.isRetryable(error)) {
          throw error;
        }

        console.log(`재시도 ${attempt + 1}/${maxRetries}, ${delay}ms 대기`);
        await this.sleep(delay);

        // 지수 백오프
        delay = Math.min(delay * backoffMultiplier, maxDelay);
      }
    }
  }

  isRetryable(error) {
    // 네트워크 오류, 일시적 서버 오류 등
    const retryableErrors = [
      'ECONNRESET', 'ETIMEDOUT', 'ENOTFOUND',
      'ECONNREFUSED', 'ENETUNREACH'
    ];

    const retryableStatuses = [408, 429, 500, 502, 503, 504];

    return (
      retryableErrors.includes(error.code) ||
      retryableStatuses.includes(error.statusCode)
    );
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// 사용 예
const retryable = new RetryableRequest();

await retryable.execute(async () => {
  return await fetch('https://api.wia-rob014.org/v1/pets');
});
