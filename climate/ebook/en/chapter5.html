5.1 Integration Architecture

WIA Climate integrates with diverse systems across the climate technology ecosystem. This chapter covers patterns for connecting IoT sensors, satellite systems, research platforms, visualization tools, and storage backends.

5.1.1 Integration Patterns

Pattern Use Case Transport
Direct Sensor Field IoT devices reporting directly MQTT, WebSocket
Gateway Edge device aggregating multiple sensors HTTP batch, MQTT
Satellite Downlink Remote sensing data from satellites HTTP batch, SFTP+transform
Research Platform Climate model outputs, experiments HTTP API, NetCDF export
Bidirectional Control + monitoring (vertical farms) WebSocket full-duplex

5.2 IoT Sensor Integration

5.2.1 Carbon Capture Facility Integration

Scenario: Climeworks Orca DAC facility in Iceland reporting hourly capture metrics.

Hardware Stack:

  • Raspberry Pi 4 edge gateway
  • Modbus RTU sensors (CO2, temperature, pressure)
  • 4G LTE modem for connectivity
#!/usr/bin/env python3
# Edge gateway code for carbon capture sensor

import json
import time
import requests
from datetime import datetime
from modbus_reader import read_sensors

WIA_API_URL = "https://api.wia.live/climate/v1/data"
API_KEY = "your_api_key_here"

def collect_data():
    sensors = read_sensors()

    message = {
        "version": "1.0.0",
        "type": "carbon_capture",
        "timestamp": {
            "unix_ms": int(time.time() * 1000),
            "iso8601": datetime.utcnow().isoformat() + "Z"
        },
        "location": {
            "latitude": 64.0,
            "longitude": -21.0,
            "altitude_m": 100
        },
        "device": {
            "manufacturer": "Climeworks",
            "model": "Orca DAC",
            "serial": "ORCA-2024-001"
        },
        "data": {
            "technology": "dac",
            "capture_rate_kg_per_hour": sensors['capture_rate'],
            "co2_purity_percentage": sensors['purity'],
            "energy_consumption_kwh": sensors['power']
        },
        "meta": {
            "quality_score": 0.98,
            "source": "sensor"
        }
    }

    return message

def send_to_wia(message):
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }

    response = requests.post(WIA_API_URL,
                            json=message,
                            headers=headers)

    if response.status_code == 201:
        print(f"✓ Data sent: {response.json()['data']['message_id']}")
    else:
        print(f"✗ Error: {response.status_code}")

# Main loop: collect and send every hour
while True:
    try:
        message = collect_data()
        send_to_wia(message)
    except Exception as e:
        print(f"Error: {e}")

    time.sleep(3600)  # 1 hour

5.3 Satellite Integration

5.3.1 Processing Satellite Ocean Cleanup Data

Satellites can detect ocean plastic concentrations using spectral imaging. Transform satellite data into WIA Climate format:

#!/usr/bin/env python3
# Transform satellite ocean monitoring data

import numpy as np
import xarray as xr
from datetime import datetime

def process_satellite_data(netcdf_file):
    """
    Convert satellite NetCDF to WIA Climate ocean_cleanup messages
    """
    ds = xr.open_dataset(netcdf_file)

    messages = []

    for lat_idx in range(ds.dims['lat']):
        for lon_idx in range(ds.dims['lon']):
            plastic_concentration = ds.plastic_density[lat_idx, lon_idx].values

            if plastic_concentration > 0:  # Only report non-zero
                message = {
                    "version": "1.0.0",
                    "type": "ocean_cleanup",
                    "timestamp": {
                        "unix_ms": int(ds.time.values.astype('datetime64[ms]')
                                     .astype('int')),
                        "iso8601": str(ds.time.values)
                    },
                    "location": {
                        "latitude": float(ds.lat[lat_idx].values),
                        "longitude": float(ds.lon[lon_idx].values)
                    },
                    "device": {
                        "manufacturer": "ESA",
                        "model": "Sentinel-2",
                        "serial": ds.attrs['satellite_id']
                    },
                    "data": {
                        "operation_type": "satellite_observation",
                        "collection": {
                            "plastic_density_kg_per_km2":
                                float(plastic_concentration)
                        },
                        "area": {
                            "grid_cell_km2": 1.0
                        }
                    },
                    "meta": {
                        "quality_score": float(ds.quality_flag[lat_idx, lon_idx]),
                        "source": "satellite"
                    }
                }
                messages.append(message)

    return messages

# Process and batch upload
messages = process_satellite_data('sentinel2_ocean_20241214.nc')
batch_upload_to_wia(messages)

5.4 Research Platform Integration

5.4.1 CMIP6 Climate Model Output

Climate models generate massive datasets. Extract key variables and publish to WIA Climate:

#!/usr/bin/env python3
# Extract CMIP6 model data and publish to WIA Climate

import xarray as xr
import cftime
from wia_climate import ClimateClient

def publish_climate_model(model_file, variable='tas'):
    """
    Publish CMIP6 temperature anomaly data
    """
    ds = xr.open_dataset(model_file)
    client = ClimateClient(api_key='your_key')

    # Extract global mean for each time step
    for time_idx in range(len(ds.time)):
        global_mean = float(ds[variable].isel(time=time_idx).mean())

        message = {
            "version": "1.0.0",
            "type": "climate_model",
            "timestamp": {
                "unix_ms": int(cftime.num2date(
                    ds.time[time_idx],
                    ds.time.units
                ).timestamp() * 1000)
            },
            "location": {
                "latitude": 0.0,  # Global average
                "longitude": 0.0
            },
            "device": {
                "manufacturer": ds.attrs['institution_id'],
                "model": ds.attrs['source_id']
            },
            "data": {
                "model": {
                    "source_id": ds.attrs['source_id'],
                    "experiment_id": ds.attrs['experiment_id'],
                    "variant_label": ds.attrs['variant_label']
                },
                "variable": {
                    "name": variable,
                    "long_name": ds[variable].attrs['long_name'],
                    "units": ds[variable].attrs['units']
                },
                "value": {
                    "data": global_mean,
                    "anomaly": global_mean - 287.3  # vs pre-industrial
                },
                "scenario": {
                    "ssp": "SSP2-4.5"
                }
            }
        }

        client.submit_data(message)
        print(f"Published time step {time_idx + 1}/{len(ds.time)}")

publish_climate_model('CESM2_ssp245_tas.nc')

5.6 Storage Backend Integration

5.6.1 InfluxDB Time-Series Storage

#!/usr/bin/env python3
# WIA Climate to InfluxDB adapter

from influxdb_client import InfluxDBClient, Point
from influxdb_client.client.write_api import SYNCHRONOUS

client = InfluxDBClient(
    url="http://localhost:8086",
    token="your_token",
    org="wia-climate"
)

write_api = client.write_api(write_options=SYNCHRONOUS)

def write_to_influx(climate_message):
    point = Point("wia_climate") \
        .tag("type", climate_message['type']) \
        .tag("device_id", climate_message['device']['serial']) \
        .tag("manufacturer", climate_message['device']['manufacturer'])

    # Add measurements based on type
    if climate_message['type'] == 'carbon_capture':
        point.field("capture_rate_kg_per_hour",
                   climate_message['data']['capture_rate_kg_per_hour'])
        point.field("co2_purity_percentage",
                   climate_message['data']['co2_purity_percentage'])

    point.time(climate_message['timestamp']['unix_ms'],
              write_precision='ms')

    write_api.write(bucket="climate-data", record=point)

5.7 Integration Best Practices

✓ Do's

  • Use UTC timestamps consistently
  • Implement exponential backoff for retries
  • Validate data locally before sending
  • Buffer messages during network outages
  • Include quality scores when available
  • Use batch APIs for historical data

✗ Don'ts

  • Don't send duplicate messages
  • Don't hardcode credentials in code
  • Don't ignore API rate limits
  • Don't send data without device registration
  • Don't mix measurement and protocol timestamps
