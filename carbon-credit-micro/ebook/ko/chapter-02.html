아키텍처 개요

WIA-CARBON_CREDIT_MICRO 시스템은 확장성, 보안 및 상호 운용성을 위해 설계된 다층 아키텍처를 기반으로 구축됩니다. 핵심적으로 이 시스템은 불변의 투명한 원장을 제공하기 위해 블록체인 기술을 활용하면서 높은 처리량 운영과 복잡한 계산을 위해 오프체인 구성 요소를 사용합니다.

아키텍처는 마이크로서비스 패턴을 따르며, 구성 요소의 독립적인 확장과 시스템 전체 중단 없이 지속적인 업데이트를 가능하게 합니다.

// 온체인 크레딧 레코드 구조
struct MicroCreditRecord {
    // 식별
    bytes32 creditId;           // 고유 식별자
    bytes32 batchId;            // 집계 배치 참조

    // 크레딧 상세
    uint256 amountGrams;        // CO2 등가 그램
    uint8 actionCategory;       // 운송, 에너지, 소비 등
    uint16 actionType;          // 특정 행동 코드

    // 출처
    address generator;          // 원래 크레딧 생성자
    address currentOwner;       // 현재 소유자
    uint256 generationTime;     // Unix 타임스탬프
    bytes32 locationHash;       // 해시된 지리적 위치

    // 검증
    uint8 verificationTier;     // 1-5 검증 수준
    address verifier;           // 검증 오라클
    bytes32 proofHash;          // 검증 증명의 IPFS 해시

    // 상태
    CreditStatus status;        // 활성, 폐기, 이전됨
    uint256 retirementTime;     // 폐기된 경우, 언제
    address retiredBy;          // 크레딧을 폐기한 사람
}

enum CreditStatus {
    PENDING,     // 검증 대기 중
    ACTIVE,      // 검증되고 거래 가능
    LOCKED,      // 이전 중 또는 폐기 대기 중
    RETIRED,     // 영구 폐기 (상쇄 청구됨)
    REVOKED      // 무효, 유통에서 제거됨
}

스마트 컨트랙트 아키텍처

마이크로 탄소 크레딧 시스템은 이중 토큰 아키텍처를 구현합니다: 프로토콜 결정을 위한 거버넌스 토큰과 실제 탄소 크레딧 표현을 위한 크레딧 토큰.

토큰 설계 철학: 크레딧은 행동 카테고리 내에서 대체 가능한 토큰으로 표현되지만 전체 추적 가능성을 위해 풍부한 메타데이터를 전달합니다. 이것은 유동성 요구와 투명성 요구 사항의 균형을 맞춥니다. 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/token/ERC1155/ERC1155.sol";
import "@openzeppelin/contracts/access/AccessControl.sol";

/**
 * @title MicroCarbonCredit
 * @notice 마이크로 탄소 크레딧을 나타내는 ERC1155 토큰
 * @dev 각 토큰 ID는 고유한 행동 카테고리를 나타냄
 */
contract MicroCarbonCredit is ERC1155, AccessControl {
    bytes32 public constant MINTER_ROLE = keccak256("MINTER_ROLE");
    bytes32 public constant VERIFIER_ROLE = keccak256("VERIFIER_ROLE");

    // 토큰 ID 카테고리
    uint256 public constant TRANSPORT_CREDITS = 1;
    uint256 public constant ENERGY_CREDITS = 2;
    uint256 public constant CONSUMPTION_CREDITS = 3;
    uint256 public constant WASTE_CREDITS = 4;

    // 크레딧 메타데이터 저장
    struct CreditMetadata {
        uint256 totalMinted;
        uint256 totalRetired;
        uint256 averageVerificationScore;
        string metadataURI;
    }

    mapping(uint256 => CreditMetadata) public categoryMetadata;

    // 이벤트
    event CreditMinted(
        address indexed generator,
        uint256 indexed category,
        uint256 amount,
        bytes32 proofHash
    );

    event CreditRetired(
        address indexed owner,
        uint256 indexed category,
        uint256 amount,
        string reason
    );

    function mintCredits(
        address generator,
        uint256 category,
        uint256 amountGrams,
        bytes32 verificationProof
    ) external onlyRole(MINTER_ROLE) {
        require(amountGrams > 0, "금액은 양수여야 함");
        require(category >= 1 && category <= 4, "유효하지 않은 카테고리");

        _mint(generator, category, amountGrams, "");

        categoryMetadata[category].totalMinted += amountGrams;

        emit CreditMinted(generator, category, amountGrams, verificationProof);
    }
}

오라클 네트워크 아키텍처

검증 오라클은 물리적 세계(실제 탄소 절감 행동)와 블록체인을 연결하는 중요한 구성 요소입니다:

// 오라클 네트워크 구현
interface VerificationOracle {
    oracleId: string;
    operatorAddress: string;
    stake: bigint;                 // 슬래싱용 스테이킹된 토큰
    reputation: number;            // 0-100 점수
    specializations: ActionCategory[];
    totalVerifications: number;
    successRate: number;
}

class OracleNetwork {
    private oracles: Map<string, VerificationOracle> = new Map();

    async submitVerificationRequest(
        actionData: ActionData
    ): Promise<string> {
        const request: VerificationRequest = {
            requestId: generateRequestId(),
            actionData,
            requiredConfirmations: this.calculateRequiredConfirmations(actionData),
            currentConfirmations: 0,
            responses: [],
            deadline: new Date(Date.now() + 3600000), // 1시간
            status: 'pending'
        };

        // 검증을 위한 오라클 선택
        const selectedOracles = this.selectOracles(
            actionData.category,
            request.requiredConfirmations
        );

        // 선택된 오라클에 브로드캐스트
        await this.broadcastRequest(request, selectedOracles);

        return request.requestId;
    }

    private selectOracles(
        category: ActionCategory,
        count: number
    ): VerificationOracle[] {
        const eligible = Array.from(this.oracles.values())
            .filter(o => o.specializations.includes(category))
            .filter(o => o.reputation >= 70)
            .sort((a, b) => b.reputation - a.reputation);

        return eligible.slice(0, count);
    }
}

데이터 저장 아키텍처

시스템은 최적의 성능과 비용 효율성을 위해 온체인과 오프체인 저장을 결합한 하이브리드 저장 방식을 사용합니다:

  • 온체인 (블록체인): 크레딧 소유권, 이전, 폐기, 검증 해시
  • IPFS: 검증 증명, 상세 행동 메타데이터, 이미지/미디어
  • PostgreSQL: 사용자 프로필, 분석, 검색 인덱스, 보고 데이터
  • Redis: 실시간 잔액, 세션 데이터, 속도 제한, 캐싱
// 하이브리드 저장 관리자
class HybridStorageManager {
    private blockchain: BlockchainClient;
    private ipfs: IPFSClient;
    private postgres: PostgresPool;
    private redis: RedisClient;

    async storeCredit(credit: MicroCarbonCredit): Promise<StorageResult> {
        // 1. IPFS에 검증 증명 저장
        const proofCid = await this.ipfs.add({
            verification: credit.verification,
            evidence: credit.evidence,
            metadata: credit.metadata
        });

        // 2. 온체인 기록 저장
        const txHash = await this.blockchain.mintCredit({
            generator: credit.generator,
            amount: credit.amount,
            category: credit.action.category,
            proofHash: proofCid
        });

        // 3. 쿼리를 위해 PostgreSQL에 인덱싱
        await this.postgres.query(`
            INSERT INTO credits (id, generator, amount, category, tx_hash, created_at)
            VALUES ($1, $2, $3, $4, $5, NOW())
        `, [credit.id, credit.generator, credit.amount, credit.action.category, txHash]);

        // 4. Redis 캐시 업데이트
        await this.redis.hincrby(`balance:${credit.generator}`, credit.action.category, credit.amount);

        return { creditId: credit.id, txHash, proofCid };
    }
}

요약

WIA-CARBON_CREDIT_MICRO 아키텍처는 수십억 개의 마이크로 탄소 크레딧을 관리하기 위한 강력하고 확장 가능한 기반을 제공합니다. 신뢰와 투명성을 위한 블록체인, 확장성을 위한 레이어 2 솔루션, 검증을 위한 오라클 네트워크, 효율성을 위한 하이브리드 저장을 활용하여 시스템은 탄소 크레딧 시장에 필수적인 무결성을 유지하면서 글로벌 규모의 개인 기후 행동을 수용할 수 있습니다.
