크레딧 생애주기 관리

모든 마이크로 탄소 크레딧은 생성부터 폐기까지 정의된 생애주기를 따릅니다. 이 생애주기를 이해하고 적절히 관리하면 크레딧 무결성과 규제 준수가 보장됩니다.

생애주기 단계

단계 기간 주요 활동
생성 즉시 ~ 72시간 행동 추적, 검증, 발행
활성 최대 5년 거래, 보유, 이전 가능
폐기 대기 24-48시간 잠금, 확인 대기
폐기됨 영구 상쇄 청구됨, 이전 불가
만료됨 5년 이후 상쇄에 더 이상 유효하지 않음 
// 크레딧 생애주기 관리자
class CreditLifecycleManager {
    private stateTransitions: Map<CreditState, CreditState[]> = new Map([
        ['pending', ['active', 'rejected']],
        ['active', ['locked', 'expired']],
        ['locked', ['active', 'retired']],
        ['retired', []],  // 종료 상태
        ['expired', []],  // 종료 상태
        ['rejected', []]  // 종료 상태
    ]);

    async transitionCredit(
        creditId: string,
        targetState: CreditState,
        context: TransitionContext
    ): Promise<TransitionResult> {
        const credit = await this.getCredit(creditId);

        // 전환이 허용되는지 확인
        const allowedTransitions = this.stateTransitions.get(credit.state);
        if (!allowedTransitions?.includes(targetState)) {
            throw new Error(
                `유효하지 않은 전환: ${credit.state} -> ${targetState}`
            );
        }

        // 전환 전 검사 실행
        await this.preTransitionChecks(credit, targetState, context);

        // 크레딧 상태 업데이트
        const previousState = credit.state;
        credit.state = targetState;
        credit.stateHistory.push({
            from: previousState,
            to: targetState,
            timestamp: new Date(),
            actor: context.actor,
            reason: context.reason
        });

        // 전환 후 작업 실행
        await this.postTransitionActions(credit, previousState, context);

        // 변경 사항 저장
        await this.saveCredit(credit);

        return {
            success: true,
            creditId,
            newState: targetState,
            timestamp: new Date()
        };
    }

    async checkExpiration(): Promise<ExpirationReport> {
        const expirationThreshold = new Date();
        expirationThreshold.setFullYear(expirationThreshold.getFullYear() - 5);

        // 만료 예정 크레딧 찾기
        const expiring = await this.findCredits({
            state: 'active',
            expiresWithin: 30 * 24 * 3600 * 1000 // 30일
        });

        // 만료된 크레딧 처리
        const expired = await this.findCredits({
            state: 'active',
            generatedBefore: expirationThreshold
        });

        for (const credit of expired) {
            await this.transitionCredit(credit.id, 'expired', {
                actor: 'system',
                reason: '크레딧 빈티지가 5년을 초과함'
            });
        }

        return {
            expiredCount: expired.length,
            expiringCount: expiring.length
        };
    }
}

유동성 관리

건강한 마켓플레이스는 충분한 유동성을 필요로 합니다. 시스템은 여러 메커니즘을 통해 유동성을 관리합니다:

유동성 풀: 사용자는 자신의 크레딧을 유동성 풀에 기여하여 즉시 거래를 가능하게 하고, 그 대가로 수수료 수익의 일부를 받을 수 있습니다. 
// 유동성 관리 시스템
class LiquidityManager {
    private pools: Map<CreditCategory, LiquidityPool> = new Map();

    async addLiquidity(
        userId: string,
        category: CreditCategory,
        amount: number
    ): Promise<LiquidityPosition> {
        const pool = this.pools.get(category);
        if (!pool) throw new Error('풀을 찾을 수 없음');

        // 사용자 크레딧을 풀로 이전
        await this.transferToPool(userId, pool.id, amount);

        // 유동성 토큰 계산 및 발행
        const lpTokens = this.calculateLPTokens(pool, amount);
        await this.mintLPTokens(userId, pool.id, lpTokens);

        // 풀 상태 업데이트
        pool.totalLiquidity += amount;
        pool.providers.push({
            userId,
            contribution: amount,
            lpTokens,
            timestamp: new Date()
        });

        return {
            poolId: pool.id,
            lpTokens,
            sharePercentage: (amount / pool.totalLiquidity) * 100,
            estimatedAPY: this.calculateEstimatedAPY(pool)
        };
    }

    async removeLiquidity(
        userId: string,
        poolId: string,
        lpTokenAmount: number
    ): Promise<WithdrawalResult> {
        const pool = this.pools.get(poolId);
        const position = await this.getUserPosition(userId, poolId);

        // 출금 가능 크레딧 계산
        const withdrawableCredits = this.calculateWithdrawal(pool, lpTokenAmount);

        // 누적 수수료 계산
        const earnedFees = this.calculateEarnedFees(position, pool);

        // LP 토큰 소각 및 자산 반환
        await this.burnLPTokens(userId, poolId, lpTokenAmount);
        await this.transferFromPool(pool.id, userId, withdrawableCredits + earnedFees);

        return {
            creditsWithdrawn: withdrawableCredits,
            feesEarned: earnedFees,
            remainingPosition: position.lpTokens - lpTokenAmount
        };
    }

    // 시장 조성 알고리즘
    private calculateSpotPrice(pool: LiquidityPool, amount: number): number {
        // 상수 곱 시장 조성 공식
        const k = pool.creditReserve * pool.currencyReserve;
        const newCreditReserve = pool.creditReserve - amount;
        const newCurrencyReserve = k / newCreditReserve;
        return (newCurrencyReserve - pool.currencyReserve) / amount;
    }
}

상응 조정: 국가 인벤토리 보고에 사용되는 크레딧의 경우, 시스템은 관할권 간 이중 계산을 방지하기 위해 파리 협정 제6.2조에서 요구하는 상응 조정을 지원합니다. 
// 규제 준수 관리 시스템
class ComplianceManager {
    private frameworks: Map<string, ComplianceFramework> = new Map();

    async checkCompliance(
        credit: MicroCredit,
        targetFramework: string
    ): Promise<ComplianceCheck> {
        const framework = this.frameworks.get(targetFramework);
        if (!framework) {
            throw new Error(`알 수 없는 프레임워크: ${targetFramework}`);
        }

        const checks: ComplianceCheckItem[] = [];

        // 추가성 검사
        checks.push(await this.checkAdditionality(credit, framework));

        // 영속성 검사 (해당되는 경우)
        if (framework.requiresPermanence) {
            checks.push(await this.checkPermanence(credit, framework));
        }

        // 검증 표준 검사
        checks.push(await this.checkVerificationStandard(credit, framework));

        // 지리적 적격성
        checks.push(await this.checkGeographicEligibility(credit, framework));

        // 빈티지 요구사항
        checks.push(await this.checkVintage(credit, framework));

        // 전체 준수 여부 계산
        const passed = checks.every(c => c.passed);
        const score = checks.filter(c => c.passed).length / checks.length;

        return {
            framework: targetFramework,
            creditId: credit.id,
            compliant: passed,
            score,
            checks,
            recommendations: this.getComplianceRecommendations(checks)
        };
    }

    async requestCorrespondingAdjustment(
        credits: MicroCredit[],
        sourceCountry: string,
        destinationCountry: string
    ): Promise<AdjustmentRequest> {
        // 크레딧이 국제 이전에 적격한지 확인
        for (const credit of credits) {
            const eligible = await this.checkInternationalTransferEligibility(credit);
            if (!eligible) {
                throw new Error(`크레딧 ${credit.id}은 국제 이전에 적격하지 않음`);
            }
        }

        // 조정 요청 제출
        const request: AdjustmentRequest = {
            id: generateRequestId(),
            credits: credits.map(c => c.id),
            totalAmount: credits.reduce((sum, c) => sum + c.amount, 0),
            sourceCountry,
            destinationCountry,
            status: 'pending',
            submittedAt: new Date()
        };

        await this.submitToRegistry(request);

        return request;
    }
}

보고 및 분석

포괄적인 보고는 사용자, 조직 및 규제 기관이 탄소 크레딧 활동과 환경 영향을 추적할 수 있게 합니다:

// 개인 탄소 보고서 생성기
class PersonalReportGenerator {
    async generateMonthlyReport(
        userId: string,
        month: Date
    ): Promise<PersonalCarbonReport> {
        const [activities, credits, retirements, baseline] = await Promise.all([
            this.getActivities(userId, month),
            this.getCredits(userId, month),
            this.getRetirements(userId, month),
            this.getBaseline(userId)
        ]);

        const totalEmissions = this.calculateEmissions(activities);
        const totalReductions = credits.generated.reduce((s, c) => s + c.amount, 0);
        const netEmissions = totalEmissions - totalReductions;

        return {
            period: {
                start: startOfMonth(month),
                end: endOfMonth(month)
            },
            summary: {
                totalEmissions,
                totalReductions,
                netEmissions,
                comparisonToBaseline: ((baseline - netEmissions) / baseline) * 100,
                rankPercentile: await this.calculateRankPercentile(userId, netEmissions)
            },
            breakdown: {
                transport: this.categorizeEmissions(activities, 'transport'),
                energy: this.categorizeEmissions(activities, 'energy'),
                consumption: this.categorizeEmissions(activities, 'consumption'),
                waste: this.categorizeEmissions(activities, 'waste')
            },
            credits: {
                generated: credits.generated.length,
                totalAmount: totalReductions,
                traded: credits.traded,
                retired: retirements
            },
            recommendations: this.generateRecommendations(activities, baseline),
            trends: await this.calculateTrends(userId, 6) // 6개월 추세
        };
    }
}
