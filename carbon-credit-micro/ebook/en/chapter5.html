5.1.2 Smart Meter Integration (Solar/Home Energy)

# Python Integration for Smart Energy Meter
import requests
import time
from datetime import datetime, timezone

class WIASmartMeterClient:
    def __init__(self, meter_id, api_key):
        self.meter_id = meter_id
        self.api_key = api_key
        self.base_url = "https://api.wia.live/v1/carbon-credits"
        self.headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }

    def report_solar_generation(self, generation_kwh, grid_carbon_intensity=0.5):
        """
        Report solar generation and automatically calculate carbon credits

        Args:
            generation_kwh: Energy generated in kWh
            grid_carbon_intensity: kg CO2 per kWh (default: 0.5)
        """
        activity_data = {
            "installation_id": self.meter_id,
            "activity_type": "solar_generation",
            "timestamp_start": datetime.now(timezone.utc).isoformat(),
            "energy_data": {
                "generation_kwh": generation_kwh,
                "grid_offset_kwh": generation_kwh,
                "grid_carbon_intensity_kg_per_kwh": grid_carbon_intensity
            },
            "carbon_calculation": {
                "avoided_emissions_kg": generation_kwh * grid_carbon_intensity
            }
        }

        response = requests.post(
            f"{self.base_url}/activities",
            json=activity_data,
            headers=self.headers
        )

        if response.status_code == 201:
            result = response.json()
            print(f"✓ Activity submitted: {result['activity_record_id']}")
            print(f"  Estimated credits: {result['estimated_credits_tco2e']} tCO2e")
            return result
        else:
            print(f"✗ Error: {response.status_code} - {response.text}")
            return None

    def get_daily_summary(self):
        """Get daily carbon credit summary"""
        response = requests.get(
            f"{self.base_url}/wallets/{self.meter_id}/balance",
            headers=self.headers
        )

        if response.status_code == 200:
            data = response.json()
            print(f"Daily Summary:")
            print(f"  Total Credits: {data['balance']['total_credits_tco2e']} tCO2e")
            print(f"  Value (USD): ${data['total_value_usd']['at_current_market_price']}")
            return data
        return None

// Track Bicycle Commute with Phone GPS
import * as Location from 'expo-location';

async function trackBikeCommute() {
    // Request location permissions
    const { status } = await Location.requestForegroundPermissionsAsync();
    if (status !== 'granted') {
        console.error('Location permission denied');
        return;
    }

    const tripData = {
        startTime: Date.now(),
        waypoints: []
    };

    // Track location every 10 seconds
    const locationSubscription = await Location.watchPositionAsync(
        {
            accuracy: Location.Accuracy.High,
            timeInterval: 10000,
            distanceInterval: 50
        },
        (location) => {
            tripData.waypoints.push({
                lat: location.coords.latitude,
                lon: location.coords.longitude,
                timestamp: new Date(location.timestamp).toISOString()
            });
        }
    );

    // When trip ends
    const endTrip = async () => {
        locationSubscription.remove();
        tripData.endTime = Date.now();

        // Calculate distance
        const distanceKm = calculateDistance(tripData.waypoints);

        // Submit to WIA
        const result = await wiaClient.submitActivity({
            activityType: 'bicycle_commute',
            timestampStart: new Date(tripData.startTime).toISOString(),
            timestampEnd: new Date(tripData.endTime).toISOString(),
            transportData: {
                mode: 'bicycle',
                distanceKm: distanceKm,
                route: {
                    startPoint: tripData.waypoints[0],
                    endPoint: tripData.waypoints[tripData.waypoints.length - 1],
                    waypoints: tripData.waypoints
                }
            }
        });

        console.log(`Earned ${result.estimatedCreditsTco2e} carbon credits!`);
        return result;
    };

    return { endTrip };
}

    func submitActivity() {

// Alexa Skill Lambda Function (Node.js)
const Alexa = require('ask-sdk-core');
const axios = require('axios');

const GetCarbonBalanceIntentHandler = {
    canHandle(handlerInput) {
        return Alexa.getRequestType(handlerInput.requestEnvelope) === 'IntentRequest'
            && Alexa.getIntentName(handlerInput.requestEnvelope) === 'GetCarbonBalanceIntent';
    },
    async handle(handlerInput) {
        const userId = handlerInput.requestEnvelope.context.System.user.userId;

        try {
            const response = await axios.get(
                `https://api.wia.live/v1/carbon-credits/wallets/${userId}/balance`,
                {
                    headers: {
                        'Authorization': `Bearer ${process.env.WIA_API_KEY}`
                    }
                }
            );

            const balance = response.data.balance.total_credits_tco2e;
            const value = response.data.total_value_usd.at_current_market_price;

            const speakOutput = `You have ${balance.toFixed(3)} carbon credits,
                                worth approximately ${value.toFixed(2)} dollars.
                                Great job helping the environment!`;

            return handlerInput.responseBuilder
                .speak(speakOutput)
                .getResponse();
        } catch (error) {
            return handlerInput.responseBuilder
                .speak('Sorry, I had trouble getting your carbon credit balance.')
                .getResponse();
        }
    }
};

// Register bike commute
const LogBikeCommuteIntentHandler = {
    canHandle(handlerInput) {

    async handle(handlerInput) {
        const distanceKm = Alexa.getSlotValue(handlerInput.requestEnvelope, 'distance');

        const activityData = {
            activity_type: 'bicycle_commute',
            timestamp_start: new Date().toISOString(),
            transport_data: {
                mode: 'bicycle',
                distance_km: parseFloat(distanceKm)
            }

        try {
            const response = await axios.post(
                'https://api.wia.live/v1/carbon-credits/activities',
                activityData,
                {
                    headers: {
                        'Authorization': `Bearer ${process.env.WIA_API_KEY}`,
                        'Content-Type': 'application/json'
                    }
                }
            );

            const credits = response.data.estimated_credits_tco2e;
            const speakOutput = `Nice! I logged your ${distanceKm} kilometer bike commute.
                                You earned ${credits.toFixed(4)} carbon credits.`;

            return handlerInput.responseBuilder
                .speak(speakOutput)
                .getResponse();
        } catch (error) {
            return handlerInput.responseBuilder
                .speak('Sorry, I had trouble logging your bike commute.')
                .getResponse();

    }
