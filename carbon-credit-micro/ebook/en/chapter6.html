// Zero-Knowledge Proof: Prove you have enough credits without revealing balance
Contract: WIAPrivateTransfer

function initiatePrivateTransfer(
    address recipient,
    uint256 encryptedAmount,
    bytes calldata zkProof
) external returns (bool) {
    // Verify ZK proof that sender has sufficient balance
    require(
        verifyZKProof(msg.sender, encryptedAmount, zkProof),
        "Insufficient balance or invalid proof"
    );

    // Execute transfer without revealing exact amounts
    _executePrivateTransfer(msg.sender, recipient, encryptedAmount);

    emit PrivateTransferCompleted(msg.sender, recipient, block.timestamp);
    return true;
}

// Verifier uses zk-SNARKs to validate
function verifyZKProof(
    address sender,
    uint256 encryptedAmount,
    bytes memory proof
) internal view returns (bool) {
    // Verify proof that:
    // 1. Sender owns the credits
    // 2. Balance >= transfer amount
    // 3. Amount > 0
    // WITHOUT revealing actual balance or amount
    return zkVerifier.verify(proof);
}

6.2.2 GPS Spoofing Detection Algorithm

def detect_gps_spoofing(gps_samples):
    """
    Detect GPS spoofing attempts using multiple signals

    Returns: (is_spoofed: bool, confidence: float, reasons: list)
    """
    red_flags = []
    confidence_score = 1.0

    # Check 1: Velocity plausibility
    for i in range(1, len(gps_samples)):
        distance_m = haversine_distance(
            gps_samples[i-1]['lat'], gps_samples[i-1]['lon'],
            gps_samples[i]['lat'], gps_samples[i]['lon']
        )
        time_delta_s = (gps_samples[i]['timestamp'] -
                       gps_samples[i-1]['timestamp']).total_seconds()
        speed_mps = distance_m / time_delta_s

        # Bicycle max realistic speed: ~15 m/s (54 km/h)
        if speed_mps > 15:
            red_flags.append(f"Unrealistic speed: {speed_mps:.1f} m/s")
            confidence_score -= 0.3

    # Check 2: Accelerometer correlation
    if 'accelerometer_data' in gps_samples[0]:
        accel_match = correlate_gps_with_accelerometer(gps_samples)
        if accel_match < 0.7:
            red_flags.append(f"GPS-accelerometer mismatch: {accel_match:.2f}")
            confidence_score -= 0.4

    # Check 3: Network location vs GPS
    if 'network_location' in gps_samples[0]:
        network_gps_distance = calculate_average_distance(
            [s['network_location'] for s in gps_samples],
            [(s['lat'], s['lon']) for s in gps_samples]
        )
        if network_gps_distance > 1000:  # 1 km threshold
            red_flags.append(f"GPS-Network location mismatch: {network_gps_distance}m")
            confidence_score -= 0.5

    # Check 4: Satellite signal strength
    avg_satellites = sum(s.get('satellite_count', 0) for s in gps_samples) / len(gps_samples)
    if avg_satellites < 4:
        red_flags.append(f"Low satellite count: {avg_satellites:.1f}")
        confidence_score -= 0.2

    # Check 5: HDOP (Horizontal Dilution of Precision)
    avg_hdop = sum(s.get('hdop', 999) for s in gps_samples) / len(gps_samples)
    if avg_hdop > 5:
        red_flags.append(f"High HDOP (poor accuracy): {avg_hdop:.1f}")
        confidence_score -= 0.2

    is_spoofed = confidence_score < 0.5

    return is_spoofed, max(0, confidence_score), red_flags

6.2.3 Duplicate Activity Detection

// Redis-based duplicate detection
const Redis = require('redis');
const redis = Redis.createClient();

async function checkDuplicateActivity(activityData) {
    // Create activity fingerprint
    const fingerprint = createActivityFingerprint(activityData);

    // Check if similar activity exists within time window
    const timeWindow = 3600; // 1 hour
    const key = `activity:${activityData.user_id}:${fingerprint}`;

    const exists = await redis.exists(key);
    if (exists) {
        return {
            isDuplicate: true,
            message: "Similar activity already submitted within 1 hour"
        };
    }

    // Store fingerprint with TTL
    await redis.setex(key, timeWindow, JSON.stringify(activityData));

    return {
        isDuplicate: false,
        message: "Activity fingerprint is unique"
    };
}

function createActivityFingerprint(activityData) {
    // Create hash of key activity parameters
    const crypto = require('crypto');
    const fingerprintData = {
        userId: activityData.user_id,
        activityType: activityData.activity_type,
        startLat: Math.round(activityData.start_lat * 1000),
        startLon: Math.round(activityData.start_lon * 1000),
        endLat: Math.round(activityData.end_lat * 1000),
        endLon: Math.round(activityData.end_lon * 1000),
        distance: Math.round(activityData.distance_km * 10),
        timeBlock: Math.floor(activityData.timestamp / 300) // 5-min blocks
    };

    return crypto
        .createHash('sha256')
        .update(JSON.stringify(fingerprintData))
        .digest('hex')
        .substring(0, 16);
}

// Example: Store only necessary data
class ActivityDataProcessor {
    processAndMinimize(rawActivityData) {
        // Original data contains precise GPS trace
        const preciseGPS = rawActivityData.gpsTrace; // 420 points

        // Calculate carbon credits
        const credits = this.calculateCredits(preciseGPS);

        // Store only aggregated, pseudonymized data
        const minimizedData = {
            credit_id: generateCreditId(),
            user_pseudonym: hashUserId(rawActivityData.userId),
            activity_type: rawActivityData.activityType,
            timestamp: rawActivityData.timestamp,
            distance_km: this.calculateDistance(preciseGPS),
            carbon_reduction_tco2e: credits,
            route_summary: {
                // Only store rounded coordinates (±100m precision)
                start_grid: this.gridCoordinate(preciseGPS[0], 0.001),
                end_grid: this.gridCoordinate(preciseGPS[preciseGPS.length-1], 0.001),
                city: this.getCityFromCoordinate(preciseGPS[0])
            },
            verification_hash: this.hashVerificationData(preciseGPS)
        };

        // Delete precise GPS data after verification
        delete preciseGPS;

        return minimizedData;
    }

    gridCoordinate(coord, precision) {
        // Round coordinates to grid (reduces precision)
        return {
            lat: Math.round(coord.lat / precision) * precision,
            lon: Math.round(coord.lon / precision) * precision
        };
    }
}

// GDPR Article 17: Right to Erasure
async function processDataDeletionRequest(userId) {
    console.log(`Processing deletion request for user: ${userId}`);

    // 1. Delete personal data from database
    await db.users.delete({ user_id: userId });
    await db.devices.delete({ owner_id: userId });
    await db.activity_records.delete({ user_id: userId });

    // 2. Anonymize wallet on blockchain (cannot delete)
    const wallet = await getWalletByUserId(userId);
    await db.wallets.update(
        { wallet_id: wallet.wallet_id },
        {
            user_id: 'DELETED_USER',
            user_email: null,
            user_name: null,
            deletion_date: new Date()
        }
    );

    // 3. Note: Blockchain transactions cannot be deleted
    // They are already pseudonymized (wallet address only)
    console.log(`Blockchain data: Wallet ${wallet.wallet_address} remains (pseudonymized)`);

    // 4. Delete from analytics systems
    await analyticsDB.deleteUserData(userId);

    // 5. Notify third-party integrations
    await notifyIntegrationPartners(userId, 'USER_DELETED');

    return {
        status: 'COMPLETED',
        deleted_from: [
            'user_database',
            'activity_records',
            'devices',
            'analytics'
        ],
        anonymized: [
            'wallet_records',
            'blockchain_transactions'
        ],
        permanent_records: [
            'blockchain_ledger (pseudonymized)'
        ]
    };
}

6.4.2 Secure Credit Minting

// Solidity: Secure credit minting with multi-sig
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/access/AccessControl.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/security/Pausable.sol";

contract WIACarbonCreditMinter is AccessControl, ReentrancyGuard, Pausable {
    bytes32 public constant VERIFIER_ROLE = keccak256("VERIFIER_ROLE");
    bytes32 public constant AUDITOR_ROLE = keccak256("AUDITOR_ROLE");

    uint256 public constant MIN_CONFIDENCE_SCORE = 85; // 0.85
    uint256 public constant MAX_CREDITS_PER_ACTIVITY = 10e6; // 10 tCO2e

    mapping(string => bool) public processedActivityIds;

    event CreditsMinted(
        address indexed recipient,
        uint256 amount,
        string activityRecordId,
        uint8 confidenceScore
    );

    function mintCredits(
        address recipient,
        uint256 amountMicroCO2e,
        string calldata activityRecordId,
        string calldata verificationId,
        uint8 confidenceScore
    ) external onlyRole(VERIFIER_ROLE) nonReentrant whenNotPaused {
        // Security checks
        require(recipient != address(0), "Invalid recipient");
        require(amountMicroCO2e > 0, "Amount must be positive");
        require(amountMicroCO2e <= MAX_CREDITS_PER_ACTIVITY, "Amount exceeds maximum");
        require(confidenceScore >= MIN_CONFIDENCE_SCORE, "Confidence too low");
        require(!processedActivityIds[activityRecordId], "Activity already processed");

        // Mark activity as processed (prevent double-spending)
        processedActivityIds[activityRecordId] = true;

        // Mint credits
        _mint(recipient, amountMicroCO2e);

        emit CreditsMinted(recipient, amountMicroCO2e, activityRecordId, confidenceScore);
    }

    // Emergency pause function
    function pause() external onlyRole(AUDITOR_ROLE) {
        _pause();
    }

    function unpause() external onlyRole(AUDITOR_ROLE) {
        _unpause();
    }
}
