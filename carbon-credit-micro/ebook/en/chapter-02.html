Ledger Data Structure

Each micro carbon credit entry in the ledger contains comprehensive metadata to ensure traceability and verification:

// On-chain Credit Record Structure
struct MicroCreditRecord {
    // Identification
    bytes32 creditId;           // Unique identifier
    bytes32 batchId;            // Aggregation batch reference

    // Credit Details
    uint256 amountGrams;        // CO2 equivalent in grams
    uint8 actionCategory;       // Transport, Energy, Consumption, etc.
    uint16 actionType;          // Specific action code

    // Provenance
    address generator;          // Original credit generator
    address currentOwner;       // Current owner
    uint256 generationTime;     // Unix timestamp
    bytes32 locationHash;       // Hashed geolocation

    // Verification
    uint8 verificationTier;     // 1-5 verification level
    address verifier;           // Verification oracle
    bytes32 proofHash;          // IPFS hash of verification proof

    // Status
    CreditStatus status;        // Active, Retired, Transferred
    uint256 retirementTime;     // If retired, when
    address retiredBy;          // Who retired the credit
}

enum CreditStatus {
    PENDING,     // Awaiting verification
    ACTIVE,      // Verified and tradeable
    LOCKED,      // In transfer or pending retirement
    RETIRED,     // Permanently retired (offset claimed)
    REVOKED      // Invalid, removed from circulation
}

Token Architecture

The system implements a dual-token architecture: a governance token for protocol decisions and the credit token (ERC1155) for carbon credit representation.

Token Design: Credits are fungible within action categories but carry rich metadata for traceability. 
// MicroCarbonCredit ERC1155 Token
contract MicroCarbonCredit is ERC1155, AccessControl {
    uint256 public constant TRANSPORT_CREDITS = 1;
    uint256 public constant ENERGY_CREDITS = 2;
    uint256 public constant CONSUMPTION_CREDITS = 3;
    uint256 public constant WASTE_CREDITS = 4;

    mapping(uint256 => CreditMetadata) public categoryMetadata;

    function mintCredits(address generator, uint256 category,
        uint256 amountGrams, bytes32 verificationProof
    ) external onlyRole(MINTER_ROLE) {
        _mint(generator, category, amountGrams, "");
        categoryMetadata[category].totalMinted += amountGrams;
        emit CreditMinted(generator, category, amountGrams, verificationProof);
    }

    function retireCredits(uint256 category, uint256 amountGrams,
        string calldata reason) external {
        _burn(msg.sender, category, amountGrams);
        categoryMetadata[category].totalRetired += amountGrams;
        emit CreditRetired(msg.sender, category, amountGrams, reason);
    }
}

Fractional Credit Management

Managing credits at the gram level requires sophisticated systems for aggregation and pooling:

// Credit Pool Management
interface CreditPool {
    poolId: string;
    category: ActionCategory;
    totalCredits: number;
    participants: PoolParticipant[];
    lockPeriod: number;
}

class CreditPoolManager {
    async contributeToPool(poolId: string, address: string, amount: number) {
        const pool = this.pools.get(poolId);
        await this.transferCreditsToPool(address, poolId, amount);
        const newTotal = pool.totalCredits + amount;
        pool.participants.forEach(p => p.sharePercentage = p.contribution / newTotal);
        pool.totalCredits = newTotal;
    }
}

Oracle Network Architecture

Verification oracles bridge the physical world with the blockchain:

// Oracle Network Implementation
class OracleNetwork {
    async submitVerificationRequest(actionData: ActionData): Promise<string> {
        const request = {
            requestId: generateRequestId(),
            actionData,
            requiredConfirmations: this.calculateRequiredConfirmations(actionData),
            status: 'pending'
        };
        const selectedOracles = this.selectOracles(actionData.category);
        await this.broadcastRequest(request, selectedOracles);
        return request.requestId;
    }

    private selectOracles(category: ActionCategory): VerificationOracle[] {
        return Array.from(this.oracles.values())
            .filter(o => o.specializations.includes(category) && o.reputation >= 70)
            .sort((a, b) => b.reputation - a.reputation);
    }
}

Data Storage Architecture

The system uses hybrid storage: on-chain for ownership/transfers, IPFS for proofs, PostgreSQL for analytics, Redis for caching.

// Hybrid Storage Manager
class HybridStorageManager {
    async storeCredit(credit: MicroCarbonCredit): Promise<StorageResult> {
        const proofCid = await this.ipfs.add({ verification: credit.verification });
        const txHash = await this.blockchain.mintCredit({
            generator: credit.generator, amount: credit.amount,
            category: credit.action.category, proofHash: proofCid
        });
        await this.postgres.query(`INSERT INTO credits VALUES ($1, $2, $3, $4, $5, NOW())`,
            [credit.id, credit.generator, credit.amount, credit.action.category, txHash]);
        await this.redis.hincrby(`balance:${credit.generator}`, credit.action.category, credit.amount);
        return { creditId: credit.id, txHash, proofCid };
    }
}
