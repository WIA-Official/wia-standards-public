Understanding Personal Carbon Footprints

A personal carbon footprint represents total greenhouse gas emissions caused by an individual, typically 4-16 tonnes CO2e per year. Tracking enables reduction opportunities and micro credit generation.

Activity Tracking System

The system captures carbon-relevant actions throughout the day:

// Activity Tracking Core
interface TrackedActivity {
    id: string;
    userId: string;
    category: 'TRANSPORT' | 'ENERGY' | 'FOOD' | 'CONSUMPTION' | 'WASTE';
    duration: number;
    distance?: number;
    calculatedEmissions: number;
    potentialCredits: number;
}

class ActivityTracker {
    async startTracking(userId: string, category: string, type: string): Promise<string> {
        const session = { sessionId: generateSessionId(), userId, category, type, startTime: new Date() };
        this.activeTracking.set(session.sessionId, session);
        if (this.requiresLocationTracking(category)) await this.startLocationTracking(session.sessionId);
        return session.sessionId;
    }

    async stopTracking(sessionId: string): Promise<TrackedActivity> {
        const session = this.activeTracking.get(sessionId);
        const metrics = this.calculateMetrics(session);
        const emissions = this.emissionCalculator.calculate(session, metrics);
        return {
            id: generateActivityId(), userId: session.userId,
            category: session.category, duration: metrics.duration,
            calculatedEmissions: emissions.actual,
            potentialCredits: emissions.baseline - emissions.actual
        };
    }
}

Transportation Tracking

Multi-Modal Detection: Combines GPS, accelerometer, speed patterns, and transit APIs for 95%+ accuracy in mode identification. 
// Transportation Mode Detection
class TransportModeDetector {
    private readonly MODE_SIGNATURES = {
        WALKING: { speedRange: [0, 7], confidence: 0.95 },
        CYCLING: { speedRange: [10, 35], confidence: 0.92 },
        DRIVING: { speedRange: [20, 150], confidence: 0.88 },
        PUBLIC_TRANSIT: { speedPattern: 'stop-and-go', confidence: 0.90 }
    };

    async detectMode(sensorData: SensorDataPoint[]): Promise<DetectionResult> {
        const features = this.extractFeatures(sensorData);
        const probabilities = new Map();
        for (const [mode, signature] of Object.entries(this.MODE_SIGNATURES)) {
            probabilities.set(mode, this.calculateModeProbability(features, signature));
        }
        const [detectedMode, highestProb] = [...probabilities.entries()]
            .sort((a, b) => b[1] - a[1])[0];
        return { mode: detectedMode, confidence: highestProb };
    }
}

Energy Consumption Tracking

Home energy tracking integrates with smart home devices and utility APIs:

// Smart Meter Integration
class EnergyTracker {
    async connectMeter(userId: string, provider: string, credentials: MeterCredentials) {
        const connection = await this.establishConnection(provider, credentials);
        this.meterConnections.set(userId, connection);
        await this.baselineService.startBaselineCollection(userId, 'energy');
        return { connected: true, meterId: connection.meterId };
    }

    async getDailyReport(userId: string, date: Date): Promise<DailyEnergyReport> {
        const readings = await this.meterConnections.get(userId).getReadings(date);
        const baseline = await this.baselineService.getBaseline(userId, 'energy', date);
        const emissions = readings.electricity * readings.gridIntensity;
        const savings = Math.max(0, baseline - emissions);
        return { date, emissions, savings, potentialCredits: savings };
    }
}

Food and Consumption Tracking

Food choices significantly impact carbon footprints. Methods include receipt scanning, barcode scanning, meal photo AI, and restaurant APIs.

// Food Carbon Calculator
class FoodCarbonCalculator {
    private readonly EMISSION_FACTORS = {
        beef: 27000, lamb: 24000, cheese: 13500, pork: 12100,
        chicken: 6900, vegetables: 2000, legumes: 900 // grams CO2e per kg
    };

    async calculateMealEmissions(meal: MealData): Promise<MealEmissions> {
        let totalEmissions = 0;
        for (const item of meal.items) {
            let emission = this.getEmissionFactor(item.category) * (item.weight / 1000);
            if (item.isLocal) emission *= 0.85;
            if (item.isSeasonal) emission *= 0.90;
            totalEmissions += emission;
        }
        return { totalEmissions, recommendations: this.generateRecommendations(meal) };
    }

    async trackMeatlessDay(userId: string, date: Date): Promise<CreditGeneration> {
        const avgMeatEmissions = (await this.getUserDietBaseline(userId)).avgDailyMeatEmissions;
        return avgMeatEmissions > 0
            ? { action: 'MEATLESS_DAY', creditsGenerated: avgMeatEmissions }
            : { creditsGenerated: 0 };
    }
}

Baseline Establishment

Baseline Period: New users undergo a 14-day baseline establishment period. Only genuine improvements generate credits. 
// Baseline Service
class BaselineService {
    private readonly BASELINE_PERIOD_DAYS = 14;

    async startBaselineCollection(userId: string, category: string): Promise<BaselineSession> {
        return { userId, category, startDate: new Date(),
            endDate: new Date(Date.now() + this.BASELINE_PERIOD_DAYS * 86400000), dataPoints: [] };
    }

    async finalizeBaseline(session: BaselineSession): Promise<Baseline> {
        const values = session.dataPoints.map(dp => dp.value);
        const sorted = [...values].sort((a, b) => a - b);
        return {
            userId: session.userId, category: session.category,
            median: sorted[Math.floor(sorted.length / 2)],
            validUntil: new Date(Date.now() + 90 * 86400000)
        };
    }

    async getEffectiveBaseline(userId: string, category: string): Promise<number> {
        const personal = await this.getBaseline(userId, category);
        const regional = await this.getRegionalBaseline(category);
        return Math.max(personal?.median || 0, regional.median);
    }
}
