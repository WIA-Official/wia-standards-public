7.1.1 Complete Carbon Credit Client

# wia_carbon_client.py
import requests
import json
from datetime import datetime, timezone
from typing import Dict, List, Optional

class WIACarbonCreditClient:
    """
    WIA Micro Carbon Credit Trading Client
    Complete implementation for submitting activities and managing credits
    """

    def __init__(self, api_key: str, user_id: str, base_url: str = None):
        self.api_key = api_key
        self.user_id = user_id
        self.base_url = base_url or "https://api.wia.live/v1/carbon-credits"
        self.headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }

    def submit_bike_commute(
        self,
        distance_km: float,
        start_lat: float,
        start_lon: float,
        end_lat: float,
        end_lon: float,
        gps_waypoints: Optional[List[Dict]] = None
    ) -> Dict:
        """Submit a bicycle commute activity"""

        activity_data = {
            "user_id": self.user_id,
            "activity_type": "bicycle_commute",
            "timestamp_start": datetime.now(timezone.utc).isoformat(),
            "timestamp_end": datetime.now(timezone.utc).isoformat(),
            "transport_data": {
                "mode": "bicycle",
                "distance_km": distance_km,
                "route": {
                    "start_point": {"lat": start_lat, "lon": start_lon},
                    "end_point": {"lat": end_lat, "lon": end_lon}
                }
            }
        }

        if gps_waypoints:
            activity_data["transport_data"]["route"]["waypoints"] = gps_waypoints

        response = requests.post(
            f"{self.base_url}/activities",
            json=activity_data,
            headers=self.headers
        )

        if response.status_code == 201:
            result = response.json()
            print(f"✓ Activity submitted: {result['activity_record_id']}")
            print(f"  Estimated credits: {result['estimated_credits_tco2e']} tCO2e")
            return result
        else:
            print(f"✗ Error: {response.status_code} - {response.text}")
            return None

    def submit_solar_generation(
        self,
        generation_kwh: float,
        installation_id: str,
        grid_carbon_intensity: float = 0.5
    ) -> Dict:
        """Submit solar energy generation"""

        activity_data = {
            "installation_id": installation_id,
            "activity_type": "solar_generation",
            "timestamp_start": datetime.now(timezone.utc).replace(
                hour=0, minute=0, second=0
            ).isoformat(),
            "timestamp_end": datetime.now(timezone.utc).isoformat(),
            "energy_data": {
                "generation_kwh": generation_kwh,
                "grid_offset_kwh": generation_kwh,
                "grid_carbon_intensity_kg_per_kwh": grid_carbon_intensity
            }
        }

        response = requests.post(
            f"{self.base_url}/activities",
            json=activity_data,
            headers=self.headers
        )

        return response.json() if response.status_code == 201 else None

    def get_wallet_balance(self) -> Dict:
        """Get user's carbon credit wallet balance"""

        response = requests.get(
            f"{self.base_url}/wallets/{self.user_id}/balance",
            headers=self.headers
        )

        if response.status_code == 200:
            data = response.json()
            print(f"\n💰 Wallet Balance:")
            print(f"   Total Credits: {data['balance']['total_credits_tco2e']} tCO2e")
            print(f"   Available: {data['balance']['available_credits_tco2e']} tCO2e")
            print(f"   Value (USD): ${data['total_value_usd']['at_current_market_price']:.2f}")
            return data
        return None

    def sell_credits(
        self,
        amount_tco2e: float,
        price_per_tco2e: float,
        expiration_hours: int = 48
    ) -> Dict:
        """Create a sell order on the marketplace"""

        order_data = {
            "user_id": self.user_id,
            "total_amount_tco2e": amount_tco2e,
            "price_usd_per_tco2e": price_per_tco2e,
            "order_type": "LIMIT",
            "expiration_hours": expiration_hours
        }

        response = requests.post(
            f"{self.base_url}/marketplace/sell",
            json=order_data,
            headers=self.headers
        )

        if response.status_code == 201:
            result = response.json()
            print(f"\n📊 Sell Order Created:")
            print(f"   Order ID: {result['order_id']}")
            print(f"   Amount: {amount_tco2e} tCO2e")
            print(f"   Price: ${price_per_tco2e}/tCO2e")
            print(f"   Total Value: ${result['total_value_usd']:.2f}")
            return result
        return None

    def get_impact_stats(self) -> Dict:
        """Get user's environmental impact statistics"""

        response = requests.get(
            f"{self.base_url}/analytics/users/{self.user_id}/impact",
            headers=self.headers
        )

        if response.status_code == 200:
            data = response.json()
            print(f"\n🌍 Environmental Impact:")
            print(f"   Total Credits Earned: {data['carbon_impact']['total_credits_earned_tco2e']} tCO2e")
            print(f"   CO2 Avoided: {data['carbon_impact']['co2_avoided_kg']} kg")
            print(f"   Equivalent Trees: {data['carbon_impact']['equivalent_trees_planted']}")
            print(f"   Global Rank: #{data['rankings']['global_rank']}")
            return data
        return None

7.2.1 TypeScript SDK with Type Safety

// wia-carbon-client.ts
import axios, { AxiosInstance } from 'axios';

// Type definitions
interface ActivityRecord {
    activity_record_id: string;
    status: string;
    estimated_credits_tco2e: number;
}

interface WalletBalance {
    total_credits_tco2e: number;
    available_credits_tco2e: number;
    pending_credits_tco2e: number;
}

interface BikeCommuteParams {
    distanceKm: number;
    startLat: number;
    startLon: number;
    endLat: number;
    endLon: number;
    gpsWaypoints?: Array<{lat: number; lon: number; timestamp: string}>;
}

interface SolarGenerationParams {
    generationKwh: number;
    installationId: string;
    gridCarbonIntensity?: number;
}

export class WIACarbonCreditClient {
    private api: AxiosInstance;
    private userId: string;

    constructor(apiKey: string, userId: string, baseUrl?: string) {
        this.userId = userId;
        this.api = axios.create({
            baseURL: baseUrl || 'https://api.wia.live/v1/carbon-credits',
            headers: {
                'Authorization': `Bearer ${apiKey}`,
                'Content-Type': 'application/json'
            }
        });
    }

    async submitBikeCommute(params: BikeCommuteParams): Promise {
        const activityData = {
            user_id: this.userId,
            activity_type: 'bicycle_commute',
            timestamp_start: new Date().toISOString(),
            timestamp_end: new Date().toISOString(),
            transport_data: {
                mode: 'bicycle',
                distance_km: params.distanceKm,
                route: {
                    start_point: { lat: params.startLat, lon: params.startLon },
                    end_point: { lat: params.endLat, lon: params.endLon },
                    ...(params.gpsWaypoints && { waypoints: params.gpsWaypoints })
                }
            }
        };

        const response = await this.api.post('/activities', activityData);
        console.log(`✓ Activity submitted: ${response.data.activity_record_id}`);
        console.log(`  Estimated credits: ${response.data.estimated_credits_tco2e} tCO2e`);

        return response.data;
    }

    async submitSolarGeneration(params: SolarGenerationParams): Promise {
        const activityData = {
            installation_id: params.installationId,
            activity_type: 'solar_generation',
            timestamp_start: new Date(new Date().setHours(0,0,0,0)).toISOString(),
            timestamp_end: new Date().toISOString(),
            energy_data: {
                generation_kwh: params.generationKwh,
                grid_offset_kwh: params.generationKwh,
                grid_carbon_intensity_kg_per_kwh: params.gridCarbonIntensity || 0.5
            }
        };

        const response = await this.api.post('/activities', activityData);
        return response.data;
    }

    async getWalletBalance(): Promise {
        const response = await this.api.get(`/wallets/${this.userId}/balance`);
        const data = response.data;

        console.log('\n💰 Wallet Balance:');
        console.log(`   Total Credits: ${data.balance.total_credits_tco2e} tCO2e`);
        console.log(`   Available: ${data.balance.available_credits_tco2e} tCO2e`);
        console.log(`   Value (USD): $${data.total_value_usd.at_current_market_price.toFixed(2)}`);

        return data.balance;
    }

    async sellCredits(
        amountTco2e: number,
        pricePerTco2e: number,
        expirationHours: number = 48
    ): Promise {
        const orderData = {
            user_id: this.userId,
            total_amount_tco2e: amountTco2e,
            price_usd_per_tco2e: pricePerTco2e,
            order_type: 'LIMIT',
            expiration_hours: expirationHours
        };

        const response = await this.api.post('/marketplace/sell', orderData);
        const result = response.data;

        console.log('\n📊 Sell Order Created:');
        console.log(`   Order ID: ${result.order_id}`);
        console.log(`   Amount: ${amountTco2e} tCO2e`);
        console.log(`   Price: $${pricePerTco2e}/tCO2e`);
        console.log(`   Total Value: $${result.total_value_usd.toFixed(2)}`);

        return result;
    }

    async configureAutoTrading(
        minimumBalance: number,
        sellPrice: number,
        autoSellPercentage: number = 80
    ): Promise {
        await this.api.put(`/wallets/${this.userId}/auto-trading`, {
            enabled: true,
            minimum_balance_tco2e: minimumBalance,
            sell_price_usd_per_tco2e: sellPrice,
            auto_sell_enabled: true,
            auto_sell_percentage: autoSellPercentage
        });

        console.log('✓ Auto-trading configured');
    }
}
