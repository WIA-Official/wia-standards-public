8.3 Implementation Example

# Federated Learning with Differential Privacy
import tensorflow as tf
import numpy as np

def add_noise(gradients, epsilon=1.0):
    """Add Laplace noise for differential privacy"""
    sensitivity = 1.0
    scale = sensitivity / epsilon
    noisy_gradients = []
    for grad in gradients:
        noise = np.random.laplace(0, scale, grad.shape)
        noisy_gradients.append(grad + noise)
    return noisy_gradients

# Federated averaging
def federated_averaging(local_models, epsilon=1.0):
    """Aggregate models from multiple institutions"""
    global_weights = []
    num_models = len(local_models)
    
    # Average weights
    for weights_list_tuple in zip(*[m.get_weights() for m in local_models]):
        avg_weights = np.mean(weights_list_tuple, axis=0)
        # Add noise for privacy
        noisy_weights = avg_weights + np.random.laplace(0, 1.0/epsilon, avg_weights.shape)
        global_weights.append(noisy_weights)
    
    return global_weights
