Chapter 6: System Integration

Integration Patterns and API Architecture

6.1 Integration Architecture

Digital executor systems must integrate with numerous external platforms, legacy systems, court filing systems, financial institutions, and third-party services. This chapter covers comprehensive integration patterns, API design, microservices architecture, message queuing, and error handling strategies.

Successful integration requires understanding diverse authentication mechanisms, API rate limits, data formats, error handling patterns, and compliance requirements across platforms. The WIA Digital Executor Standard provides reference architectures for common integration scenarios.

Integration Principle: Design for failure. External systems will have outages, rate limits, and changing APIs. Implement robust error handling, retry logic, circuit breakers, and graceful degradation.

6.3 RESTful API Design

Example 1: Comprehensive REST API Implementation

// Digital Executor REST API Service
const express = require('express');
const rateLimit = require('express-rate-limit');
const helmet = require('helmet');

class DigitalExecutorAPI {
  constructor(config) {
    this.app = express();
    this.config = config;
    this.setupMiddleware();
    this.setupRoutes();
  }

  setupMiddleware() {
    // Security headers
    this.app.use(helmet());
    
    // Rate limiting
    const limiter = rateLimit({
      windowMs: 15 * 60 * 1000, // 15 minutes
      max: 100, // limit each IP to 100 requests per windowMs
      message: 'Too many requests from this IP'
    });
    this.app.use(limiter);
    
    // JSON parsing
    this.app.use(express.json({ limit: '10mb' }));
    
    // CORS
    this.app.use((req, res, next) => {
      res.header('Access-Control-Allow-Origin', this.config.allowedOrigins);
      res.header('Access-Control-Allow-Methods', 'GET,PUT,POST,DELETE,PATCH');
      res.header('Access-Control-Allow-Headers', 'Content-Type,Authorization');
      next();
    });
    
    // Request ID and logging
    this.app.use((req, res, next) => {
      req.id = this.generateRequestId();
      this.logRequest(req);
      next();
    });
    
    // Authentication middleware
    this.app.use(this.authenticate.bind(this));
  }

  setupRoutes() {
    const router = express.Router();

    // Health check
    router.get('/health', (req, res) => {
      res.json({
        status: 'healthy',
        version: this.config.version,
        timestamp: new Date().toISOString()
      });
    });

    // Executor authorization endpoints
    router.post('/v1/executor/authorization/request',
      this.validateRequest('authorizationRequest'),
      this.handleAuthorizationRequest.bind(this)
    );

    router.get('/v1/executor/authorization/status/:requestId',
      this.handleAuthorizationStatus.bind(this)
    );

    // Platform access endpoints
    router.post('/v1/platform/access/request',
      this.requireAuthLevel(2),
      this.validateRequest('platformAccessRequest'),
      this.handlePlatformAccessRequest.bind(this)
    );

    router.get('/v1/platform/:platform/data',
      this.requireAuthLevel(2),
      this.handlePlatformDataRetrieval.bind(this)
    );

    // Asset transfer endpoints
    router.post('/v1/transfer/initiate',
      this.requireAuthLevel(3),
      this.validateRequest('transferRequest'),
      this.handleTransferInitiation.bind(this)
    );

    router.get('/v1/transfer/status/:transferId',
      this.handleTransferStatus.bind(this)
    );

    this.app.use('/api', router);

    // Error handling
    this.app.use(this.errorHandler.bind(this));
  }

  async authenticate(req, res, next) {
    try {
      const authHeader = req.headers.authorization;
      
      if (!authHeader) {
        return res.status(401).json({
          error: 'No authorization header',
          requestId: req.id
        });
      }

      const token = authHeader.split(' ')[1];
      const decoded = await this.verifyToken(token);

      req.executor = decoded.executor;
      req.authLevel = decoded.permissions.level;
      
      next();
    } catch (error) {
      return res.status(401).json({
        error: 'Invalid or expired token',
        requestId: req.id
      });
    }
  }

  requireAuthLevel(minLevel) {
    return (req, res, next) => {
      if (req.authLevel < minLevel) {
        return res.status(403).json({
          error: 'Insufficient authorization level',
          required: minLevel,
          current: req.authLevel,
          requestId: req.id
        });
      }
      next();
    };
  }

  validateRequest(schemaName) {
    return async (req, res, next) => {
      try {
        const schema = await this.getSchema(schemaName);
        const valid = await this.validateAgainstSchema(req.body, schema);
        
        if (!valid) {
          return res.status(400).json({
            error: 'Request validation failed',
            details: this.validationErrors,
            requestId: req.id
          });
        }
        
        next();
      } catch (error) {
        return res.status(500).json({
          error: 'Validation error',
          requestId: req.id
        });
      }
    };
  }

  async handleAuthorizationRequest(req, res) {
    try {
      const result = await this.authorizationService.initiateRequest(req.body);
      
      res.status(201).json({
        success: true,
        requestId: result.requestId,
        status: result.status,
        nextSteps: result.nextSteps,
        estimatedCompletion: result.estimatedCompletion
      });
    } catch (error) {
      this.handleError(error, req, res);
    }
  }

  errorHandler(err, req, res, next) {
    console.error('API Error:', err);
    
    const statusCode = err.statusCode || 500;
    const response = {
      error: err.message || 'Internal server error',
      requestId: req.id,
      timestamp: new Date().toISOString()
    };

    if (this.config.environment === 'development') {
      response.stack = err.stack;
    }

    res.status(statusCode).json(response);
  }

  start(port) {
    this.app.listen(port, () => {
      console.log(\`Digital Executor API listening on port \${port}\`);
    });
  }
}

module.exports = DigitalExecutorAPI;

6.4 Microservices Architecture

Example 2: Microservices Communication

// Microservices Integration Layer
const amqp = require('amqplib');

class MicroservicesOrchestrator {
  constructor(config) {
    this.config = config;
    this.services = new Map();
    this.messageQueue = null;
  }

  async initialize() {
    // Connect to message queue
    const connection = await amqp.connect(this.config.rabbitmqUrl);
    const channel = await connection.createChannel();
    
    this.messageQueue = channel;
    
    // Setup service discovery
    await this.setupServiceDiscovery();
    
    // Setup health checks
    setInterval(() => this.checkServicesHealth(), 30000);
  }

  async registerService(serviceName, serviceConfig) {
    this.services.set(serviceName, {
      name: serviceName,
      endpoint: serviceConfig.endpoint,
      status: 'active',
      lastHealthCheck: new Date(),
      capabilities: serviceConfig.capabilities
    });

    // Setup message queue for service
    await this.setupServiceQueue(serviceName);
  }

  async setupServiceQueue(serviceName) {
    const queueName = \`service.\${serviceName}\`;
    
    await this.messageQueue.assertQueue(queueName, {
      durable: true,
      maxPriority: 10
    });

    await this.messageQueue.consume(queueName, async (msg) => {
      if (msg !== null) {
        try {
          const payload = JSON.parse(msg.content.toString());
          await this.handleServiceMessage(serviceName, payload);
          this.messageQueue.ack(msg);
        } catch (error) {
          console.error(\`Error processing message for \${serviceName}:\`, error);
          this.messageQueue.nack(msg, false, true);
        }
      }
    });
  }

  async sendServiceMessage(targetService, message, priority = 5) {
    const queueName = \`service.\${targetService}\`;
    
    const messageBuffer = Buffer.from(JSON.stringify({
      ...message,
      timestamp: new Date().toISOString(),
      correlationId: this.generateCorrelationId()
    }));

    await this.messageQueue.sendToQueue(queueName, messageBuffer, {
      persistent: true,
      priority: priority
    });
  }

  async orchestrateTransfer(transferRequest) {
    const orchestrationId = this.generateOrchestrationId();
    
    // Step 1: Validate with authorization service
    const authValidation = await this.sendServiceMessage('authorization', {
      action: 'validate',
      data: transferRequest.executorInfo
    });

    if (!authValidation.valid) {
      throw new Error('Authorization validation failed');
    }

    // Step 2: Request platform access
    const platformAccess = await this.sendServiceMessage('platform-access', {
      action: 'requestAccess',
      platforms: transferRequest.platforms
    });

    // Step 3: Initiate transfer
    const transfer = await this.sendServiceMessage('transfer-service', {
      action: 'initiate',
      assets: transferRequest.assets,
      beneficiary: transferRequest.beneficiary
    });

    // Step 4: Notify compliance service
    await this.sendServiceMessage('compliance', {
      action: 'logTransfer',
      transferId: transfer.transferId,
      details: transferRequest
    });

    return {
      orchestrationId: orchestrationId,
      status: 'initiated',
      services: ['authorization', 'platform-access', 'transfer-service', 'compliance']
    };
  }

  async checkServicesHealth() {
    for (const [serviceName, service] of this.services) {
      try {
        const response = await fetch(\`\${service.endpoint}/health\`);
        const health = await response.json();
        
        service.status = health.status;
        service.lastHealthCheck = new Date();
      } catch (error) {
        service.status = 'unhealthy';
        console.error(\`Service \${serviceName} health check failed:\`, error);
      }
    }
  }
}

module.exports = MicroservicesOrchestrator;

6.5 Third-Party Integration

Example 3: Platform Adapter Pattern

// Platform Adapter for External Services
class PlatformAdapter {
  constructor(platform, credentials) {
    this.platform = platform;
    this.credentials = credentials;
    this.client = this.initializeClient();
    this.rateLimiter = this.setupRateLimiter();
  }

  async executeWithRetry(operation, maxRetries = 3) {
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
      try {
        // Check rate limit
        await this.rateLimiter.acquire();
        
        // Execute operation
        const result = await operation();
        
        return result;
      } catch (error) {
        if (this.isRetryable(error) && attempt < maxRetries) {
          const delay = this.calculateBackoff(attempt);
          await this.sleep(delay);
          continue;
        }
        throw error;
      }
    }
  }

  async getData(query) {
    return await this.executeWithRetry(async () => {
      const response = await this.client.get(query);
      return this.transformResponse(response);
    });
  }

  transformResponse(rawResponse) {
    // Normalize platform-specific response to standard format
    return {
      data: rawResponse.data,
      metadata: {
        platform: this.platform,
        retrievedAt: new Date().toISOString(),
        totalItems: rawResponse.count || rawResponse.total
      }
    };
  }

  setupRateLimiter() {
    const limits = this.platform.rateLimits || {
      requestsPerMinute: 60,
      requestsPerHour: 1000
    };

    return new RateLimiter(limits);
  }

  isRetryable(error) {
    const retryableStatusCodes = [408, 429, 500, 502, 503, 504];
    return retryableStatusCodes.includes(error.statusCode);
  }

  calculateBackoff(attempt) {
    return Math.min(1000 * Math.pow(2, attempt), 30000);
  }
}

module.exports = PlatformAdapter;
