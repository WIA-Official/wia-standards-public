Chapter 2: Multi-Platform Access

Comprehensive Platform Integration and Access Management

2.1 Platform Access Overview

Digital executors must manage assets across dozens of platforms, each with unique authentication mechanisms, API structures, and access control policies. This chapter provides comprehensive guidance on accessing email services, cloud storage, social media, financial accounts, blockchain wallets, streaming services, e-commerce platforms, and specialized digital assets.

The WIA Digital Executor Standard defines unified access protocols that work across platforms while respecting each platform's specific security requirements. Executors receive platform-specific credentials and access tokens that enable authorized operations while maintaining comprehensive audit trails.

Platform Integration Principle: Each platform integration must maintain the platform's existing security controls while adding executor-specific authentication and authorization layers. Never bypass platform security mechanisms.

2.4 OAuth 2.0 Integration for Platform Access

Most modern platforms use OAuth 2.0 for authentication and authorization. The Digital Executor Standard provides specialized OAuth flows that incorporate executor verification into the standard OAuth process.

Example 2: Executor OAuth Implementation

// Digital Executor OAuth 2.0 Flow
class ExecutorOAuthManager {
  constructor(config) {
    this.clientId = config.clientId;
    this.clientSecret = config.clientSecret;
    this.redirectUri = config.redirectUri;
    this.platformConfigs = config.platformConfigs;
  }

  // Initiate OAuth flow with executor context
  async initiateExecutorOAuth(platform, executorToken, deceasedAccountInfo) {
    const platformConfig = this.platformConfigs[platform];

    if (!platformConfig) {
      throw new Error(`Platform ${platform} not supported`);
    }

    // Validate executor authorization
    const executorValidation = await this.validateExecutorToken(executorToken);
    if (!executorValidation.valid || executorValidation.level < platformConfig.minimumAuthLevel) {
      throw new Error('Insufficient executor authorization level');
    }

    // Create OAuth state with executor context
    const state = {
      stateId: this.generateSecureState(),
      executorId: executorValidation.executorId,
      deceasedId: deceasedAccountInfo.accountHolderId,
      platform: platform,
      timestamp: new Date().toISOString(),
      expiresAt: new Date(Date.now() + 600000).toISOString(), // 10 minutes
      legalVerification: {
        deathCertificateHash: deceasedAccountInfo.deathCertificateHash,
        probateCaseNumber: executorValidation.probateCaseNumber
      }
    };

    // Store state securely
    await this.storeOAuthState(state.stateId, state);

    // Build authorization URL with executor-specific scopes
    const authUrl = new URL(platformConfig.authorizationEndpoint);
    authUrl.searchParams.append('client_id', this.clientId);
    authUrl.searchParams.append('redirect_uri', this.redirectUri);
    authUrl.searchParams.append('response_type', 'code');
    authUrl.searchParams.append('state', state.stateId);
    authUrl.searchParams.append('scope', platformConfig.executorScopes.join(' '));

    // Add platform-specific executor parameters
    if (platformConfig.supportsExecutorMode) {
      authUrl.searchParams.append('access_type', 'offline');
      authUrl.searchParams.append('prompt', 'consent');
      authUrl.searchParams.append('executor_mode', 'true');
      authUrl.searchParams.append('deceased_account', deceasedAccountInfo.email);
    }

    return {
      authorizationUrl: authUrl.toString(),
      stateId: state.stateId,
      expiresIn: 600
    };
  }

  // Handle OAuth callback with executor verification
  async handleOAuthCallback(code, state, error) {
    if (error) {
      await this.logOAuthError({
        error: error,
        state: state,
        timestamp: new Date().toISOString()
      });
      throw new Error(`OAuth error: ${error}`);
    }

    // Retrieve and validate state
    const storedState = await this.retrieveOAuthState(state);

    if (!storedState) {
      throw new Error('Invalid or expired OAuth state');
    }

    if (new Date() > new Date(storedState.expiresAt)) {
      throw new Error('OAuth state expired');
    }

    const platformConfig = this.platformConfigs[storedState.platform];

    // Exchange authorization code for tokens
    const tokenResponse = await fetch(platformConfig.tokenEndpoint, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/x-www-form-urlencoded',
        'Accept': 'application/json'
      },
      body: new URLSearchParams({
        grant_type: 'authorization_code',
        code: code,
        redirect_uri: this.redirectUri,
        client_id: this.clientId,
        client_secret: this.clientSecret
      })
    });

    if (!tokenResponse.ok) {
      throw new Error('Token exchange failed');
    }

    const tokens = await tokenResponse.json();

    // Create executor-specific access record
    const accessRecord = {
      recordId: this.generateRecordId(),
      executorId: storedState.executorId,
      deceasedId: storedState.deceasedId,
      platform: storedState.platform,
      tokens: {
        accessToken: await this.encryptToken(tokens.access_token),
        refreshToken: tokens.refresh_token ? await this.encryptToken(tokens.refresh_token) : null,
        tokenType: tokens.token_type,
        expiresIn: tokens.expires_in,
        expiresAt: new Date(Date.now() + (tokens.expires_in * 1000)).toISOString(),
        scope: tokens.scope
      },
      grantedAt: new Date().toISOString(),
      legalAuthorization: storedState.legalVerification,
      activityLog: []
    };

    // Store access record
    await this.storeAccessRecord(accessRecord);

    // Clean up OAuth state
    await this.deleteOAuthState(state);

    // Log successful authorization
    await this.logPlatformAccess({
      executorId: storedState.executorId,
      platform: storedState.platform,
      event: 'OAUTH_COMPLETED',
      timestamp: new Date().toISOString(),
      scope: tokens.scope
    });

    return {
      success: true,
      accessRecordId: accessRecord.recordId,
      platform: storedState.platform,
      expiresAt: accessRecord.tokens.expiresAt,
      scope: tokens.scope
    };
  }

  // Refresh expired access token
  async refreshAccessToken(accessRecordId) {
    const record = await this.retrieveAccessRecord(accessRecordId);

    if (!record.tokens.refreshToken) {
      throw new Error('No refresh token available - reauthorization required');
    }

    const platformConfig = this.platformConfigs[record.platform];

    const tokenResponse = await fetch(platformConfig.tokenEndpoint, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/x-www-form-urlencoded'
      },
      body: new URLSearchParams({
        grant_type: 'refresh_token',
        refresh_token: await this.decryptToken(record.tokens.refreshToken),
        client_id: this.clientId,
        client_secret: this.clientSecret
      })
    });

    if (!tokenResponse.ok) {
      throw new Error('Token refresh failed - reauthorization required');
    }

    const newTokens = await tokenResponse.json();

    // Update access record with new tokens
    record.tokens.accessToken = await this.encryptToken(newTokens.access_token);
    record.tokens.expiresIn = newTokens.expires_in;
    record.tokens.expiresAt = new Date(Date.now() + (newTokens.expires_in * 1000)).toISOString();

    if (newTokens.refresh_token) {
      record.tokens.refreshToken = await this.encryptToken(newTokens.refresh_token);
    }

    record.lastRefreshed = new Date().toISOString();

    await this.updateAccessRecord(accessRecordId, record);

    return {
      success: true,
      expiresAt: record.tokens.expiresAt
    };
  }

  // Make authenticated API request to platform
  async makePlatformRequest(accessRecordId, endpoint, method = 'GET', data = null) {
    const record = await this.retrieveAccessRecord(accessRecordId);

    // Check if token needs refresh
    if (new Date() >= new Date(record.tokens.expiresAt)) {
      await this.refreshAccessToken(accessRecordId);
      record = await this.retrieveAccessRecord(accessRecordId);
    }

    const platformConfig = this.platformConfigs[record.platform];
    const accessToken = await this.decryptToken(record.tokens.accessToken);

    const requestConfig = {
      method: method,
      headers: {
        'Authorization': `${record.tokens.tokenType} ${accessToken}`,
        'Content-Type': 'application/json',
        'User-Agent': 'WIA-Digital-Executor/1.0'
      }
    };

    if (data && (method === 'POST' || method === 'PUT' || method === 'PATCH')) {
      requestConfig.body = JSON.stringify(data);
    }

    const response = await fetch(endpoint, requestConfig);

    // Log activity
    await this.logPlatformActivity({
      accessRecordId: accessRecordId,
      executorId: record.executorId,
      platform: record.platform,
      endpoint: endpoint,
      method: method,
      statusCode: response.status,
      timestamp: new Date().toISOString()
    });

    return {
      status: response.status,
      data: await response.json(),
      headers: Object.fromEntries(response.headers.entries())
    };
  }
}

module.exports = ExecutorOAuthManager;

2.5 Platform-Specific Integration Examples

Gmail Integration

// Gmail API Integration for Digital Executors
const { google } = require('googleapis');

class GmailExecutorAccess {
  async getEmailMessages(accessRecordId, query, maxResults = 100) {
    const oauth2Client = await this.getOAuth2Client(accessRecordId);
    const gmail = google.gmail({ version: 'v1', auth: oauth2Client });

    const response = await gmail.users.messages.list({
      userId: 'me',
      q: query,
      maxResults: maxResults
    });

    const messages = [];

    for (const message of response.data.messages || []) {
      const fullMessage = await gmail.users.messages.get({
        userId: 'me',
        id: message.id,
        format: 'full'
      });

      messages.push({
        id: fullMessage.data.id,
        threadId: fullMessage.data.threadId,
        subject: this.getHeader(fullMessage.data.payload.headers, 'Subject'),
        from: this.getHeader(fullMessage.data.payload.headers, 'From'),
        to: this.getHeader(fullMessage.data.payload.headers, 'To'),
        date: this.getHeader(fullMessage.data.payload.headers, 'Date'),
        snippet: fullMessage.data.snippet
      });
    }

    return messages;
  }
}

Dropbox Integration

// Dropbox API Integration for Digital Executors
const { Dropbox } = require('dropbox');

class DropboxExecutorAccess {
  async listFolderContents(accessRecordId, path = '') {
    const accessToken = await this.getDecryptedToken(accessRecordId);
    const dbx = new Dropbox({ accessToken });

    const response = await dbx.filesListFolder({ path });

    return response.result.entries.map(entry => ({
      name: entry.name,
      path: entry.path_display,
      type: entry['.tag'],
      size: entry.size,
      modified: entry.client_modified,
      id: entry.id
    }));
  }

  async downloadFile(accessRecordId, path) {
    const accessToken = await this.getDecryptedToken(accessRecordId);
    const dbx = new Dropbox({ accessToken });

    const response = await dbx.filesDownload({ path });

    return {
      filename: response.result.name,
      content: response.result.fileBinary,
      size: response.result.size,
      contentHash: response.result.content_hash
    };
  }
}
