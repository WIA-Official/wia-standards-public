Chapter 5: Security Architecture

Advanced Security Controls and Cryptographic Systems

5.1 Security Architecture Overview

Digital executor systems handle highly sensitive assets and must implement defense-in-depth security architecture combining encryption, access controls, network security, application security, and operational security. This chapter covers comprehensive security requirements from cryptographic primitives through incident response.

Security is not a feature—it's the foundation. Every component must be designed with security-first principles: least privilege, defense in depth, fail securely, complete mediation, and psychological acceptability.

Security Principle: Assume breach. Design systems to minimize damage when (not if) a component is compromised. Implement layered security controls so single failures don't cascade.

5.3 Cryptographic Implementation

Example 1: Advanced Encryption System

// Enterprise Encryption System
const crypto = require('crypto');

class DigitalExecutorCrypto {
  constructor(config) {
    this.kms = config.keyManagementService;
    this.algorithm = 'aes-256-gcm';
    this.keyDerivationIterations = 100000;
  }

  // Encrypt sensitive data with AES-256-GCM
  async encryptData(plaintext, keyId) {
    const dataKey = await this.kms.getDataKey(keyId);
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv(this.algorithm, dataKey, iv);
    
    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    const authTag = cipher.getAuthTag();

    return {
      ciphertext: encrypted,
      iv: iv.toString('hex'),
      authTag: authTag.toString('hex'),
      keyId: keyId,
      algorithm: this.algorithm,
      timestamp: new Date().toISOString()
    };
  }

  // Decrypt data with integrity verification
  async decryptData(encryptedData) {
    const dataKey = await this.kms.getDataKey(encryptedData.keyId);
    const decipher = crypto.createDecipheriv(
      this.algorithm,
      dataKey,
      Buffer.from(encryptedData.iv, 'hex')
    );
    
    decipher.setAuthTag(Buffer.from(encryptedData.authTag, 'hex'));
    
    let decrypted = decipher.update(encryptedData.ciphertext, 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return decrypted;
  }

  // Envelope encryption for large data
  async envelopeEncrypt(largeData, masterKeyId) {
    const dataKey = crypto.randomBytes(32);
    const encryptedDataKey = await this.kms.encrypt(dataKey, masterKeyId);
    
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv(this.algorithm, dataKey, iv);
    
    const encryptedData = Buffer.concat([
      cipher.update(largeData),
      cipher.final()
    ]);
    
    const authTag = cipher.getAuthTag();
    
    return {
      encryptedData: encryptedData,
      encryptedDataKey: encryptedDataKey,
      iv: iv,
      authTag: authTag,
      masterKeyId: masterKeyId
    };
  }

  // Zero-knowledge proof for authentication
  async generateZKProof(secret, challenge) {
    const commitment = crypto.createHash('sha256')
      .update(secret + challenge)
      .digest('hex');
    
    const proof = {
      commitment: commitment,
      challenge: challenge,
      timestamp: new Date().toISOString(),
      algorithm: 'SHA-256'
    };
    
    return proof;
  }

  // Homomorphic encryption for secure computation
  async homomorphicEncrypt(value, publicKey) {
    // Simplified Paillier-like encryption
    const n = publicKey.n;
    const g = publicKey.g;
    const r = this.getRandomInRange(1, n);
    
    const gm = this.modPow(g, value, n * n);
    const rn = this.modPow(r, n, n * n);
    const ciphertext = (gm * rn) % (n * n);
    
    return ciphertext;
  }

  // Multi-party computation setup
  async initializeMPC(parties, threshold) {
    const shares = [];
    const secret = crypto.randomBytes(32);
    
    // Shamir's Secret Sharing
    for (let i = 1; i <= parties; i++) {
      const share = await this.createSecretShare(secret, i, threshold);
      shares.push(share);
    }
    
    return {
      shares: shares,
      threshold: threshold,
      totalParties: parties
    };
  }

  // Secure key derivation
  async deriveKey(password, salt, iterations = 100000) {
    return new Promise((resolve, reject) => {
      crypto.pbkdf2(
        password,
        salt,
        iterations,
        32,
        'sha512',
        (err, derivedKey) => {
          if (err) reject(err);
          else resolve(derivedKey);
        }
      );
    });
  }

  // Digital signature generation
  async signData(data, privateKey) {
    const sign = crypto.createSign('RSA-SHA256');
    sign.update(data);
    sign.end();
    
    const signature = sign.sign(privateKey, 'hex');
    
    return {
      signature: signature,
      algorithm: 'RSA-SHA256',
      timestamp: new Date().toISOString(),
      dataHash: crypto.createHash('sha256').update(data).digest('hex')
    };
  }

  // Signature verification
  async verifySignature(data, signature, publicKey) {
    const verify = crypto.createVerify('RSA-SHA256');
    verify.update(data);
    verify.end();
    
    return verify.verify(publicKey, signature, 'hex');
  }
}

module.exports = DigitalExecutorCrypto;

5.4 Key Management System

Example 2: Enterprise Key Management

// Key Management Service
class KeyManagementService {
  constructor(hsmConfig) {
    this.hsm = hsmConfig.hardwareSecurityModule;
    this.keyRotationDays = 90;
    this.keyCache = new Map();
  }

  // Generate master encryption key
  async generateMasterKey(keySpec) {
    const masterKey = {
      keyId: this.generateKeyId(),
      keySpec: keySpec,
      algorithm: 'AES-256',
      createdAt: new Date().toISOString(),
      rotationSchedule: this.keyRotationDays,
      state: 'ENABLED',
      origin: 'AWS_KMS' // or 'HSM' or 'EXTERNAL'
    };

    // Store in HSM
    await this.hsm.createKey(masterKey);
    
    // Create key metadata
    await this.storeKeyMetadata(masterKey);
    
    return masterKey;
  }

  // Rotate encryption keys
  async rotateKey(keyId) {
    const oldKey = await this.getKey(keyId);
    
    // Generate new key version
    const newKey = await this.generateMasterKey(oldKey.keySpec);
    
    // Update key alias to point to new version
    await this.updateKeyAlias(keyId, newKey.keyId);
    
    // Schedule re-encryption of data
    await this.scheduleDataReEncryption(oldKey.keyId, newKey.keyId);
    
    // Mark old key for deletion (30-day grace period)
    await this.scheduleKeyDeletion(oldKey.keyId, 30);
    
    return {
      oldKeyId: oldKey.keyId,
      newKeyId: newKey.keyId,
      rotatedAt: new Date().toISOString()
    };
  }

  // Generate data encryption key
  async generateDataKey(masterKeyId) {
    const plaintextKey = crypto.randomBytes(32);
    
    // Encrypt data key with master key
    const encryptedKey = await this.hsm.encrypt(plaintextKey, masterKeyId);
    
    return {
      plaintextKey: plaintextKey,
      encryptedKey: encryptedKey,
      masterKeyId: masterKeyId
    };
  }

  // Secure key backup
  async backupKeys(keyIds) {
    const backup = {
      backupId: this.generateBackupId(),
      timestamp: new Date().toISOString(),
      keys: []
    };

    for (const keyId of keyIds) {
      const key = await this.exportKey(keyId);
      backup.keys.push(key);
    }

    // Encrypt backup with offline master key
    const encryptedBackup = await this.encryptBackup(backup);
    
    // Store in secure offline storage
    await this.storeOfflineBackup(encryptedBackup);
    
    return backup.backupId;
  }
}

module.exports = KeyManagementService;

5.5 Access Control and Authentication

Example 3: Role-Based Access Control

// Advanced RBAC System
class AccessControlSystem {
  constructor() {
    this.roles = new Map();
    this.permissions = new Map();
    this.policies = new Map();
  }

  // Define role with permissions
  async defineRole(roleName, permissions) {
    const role = {
      name: roleName,
      permissions: permissions,
      createdAt: new Date().toISOString(),
      modifiedAt: new Date().toISOString()
    };

    this.roles.set(roleName, role);
    await this.persistRole(role);

    return role;
  }

  // Assign role to user
  async assignRole(userId, roleName) {
    const user = await this.getUser(userId);
    const role = this.roles.get(roleName);

    if (!role) {
      throw new Error('Role not found');
    }

    user.roles = user.roles || [];
    if (!user.roles.includes(roleName)) {
      user.roles.push(roleName);
      await this.updateUser(user);
    }

    return user;
  }

  // Check permission
  async checkPermission(userId, resource, action) {
    const user = await this.getUser(userId);
    
    // Aggregate permissions from all roles
    const userPermissions = new Set();
    for (const roleName of user.roles || []) {
      const role = this.roles.get(roleName);
      for (const perm of role.permissions) {
        userPermissions.add(perm);
      }
    }

    const requiredPermission = `${resource}:${action}`;
    
    // Check wildcard permissions
    if (userPermissions.has('*:*') || 
        userPermissions.has(`${resource}:*`) ||
        userPermissions.has(requiredPermission)) {
      return true;
    }

    // Check policy-based permissions
    return await this.evaluatePolicies(userId, resource, action);
  }

  // Attribute-based access control
  async evaluateABAC(userId, resource, action, context) {
    const policies = await this.getPoliciesForResource(resource);

    for (const policy of policies) {
      const conditions = policy.conditions;
      
      // Evaluate all conditions
      const conditionsMet = await this.evaluateConditions(
        conditions,
        { userId, resource, action, context }
      );

      if (conditionsMet && policy.effect === 'ALLOW') {
        return true;
      } else if (conditionsMet && policy.effect === 'DENY') {
        return false;
      }
    }

    return false; // Default deny
  }

  // Evaluate policy conditions
  async evaluateConditions(conditions, evaluationContext) {
    for (const condition of conditions) {
      switch (condition.type) {
        case 'IP_ADDRESS':
          if (!this.matchIPAddress(evaluationContext.context.ip, condition.value)) {
            return false;
          }
          break;
        case 'TIME_OF_DAY':
          if (!this.matchTimeRange(condition.value)) {
            return false;
          }
          break;
        case 'USER_ATTRIBUTE':
          const userAttr = await this.getUserAttribute(
            evaluationContext.userId,
            condition.attribute
          );
          if (userAttr !== condition.value) {
            return false;
          }
          break;
      }
    }
    return true;
  }
}

module.exports = AccessControlSystem;
