6.1 프로토콜 개요

Phase 3는 시스템 간 실시간 통신을 위한 프로토콜을 정의합니다. 메시지 형식, 전송 메커니즘, 인증, 그리고 신뢰성 보장 방법이 포함됩니다.

  • 효율성: 최소한의 오버헤드
  • 상호운용성: 표준 기술 사용

6.4.2 재시도 전략

class RetryStrategy {
  constructor(config = {}) {
    this.maxRetries = config.maxRetries || 3;
    this.baseDelay = config.baseDelay || 1000;
    this.maxDelay = config.maxDelay || 30000;
  }
  
  async execute(operation) {
    for (let attempt = 0; attempt < this.maxRetries; attempt++) {
      try {
        return await operation();
      } catch (error) {
        if (!this.isRetryable(error)) throw error;
        
        const delay = Math.min(
          this.baseDelay * Math.pow(2, attempt),
          this.maxDelay
        );
        
        console.log(`재시도 ${attempt + 1}/${this.maxRetries}, ${delay}ms 후`);
        await this.sleep(delay);
      }
    }
  }
  
  isRetryable(error) {
    return [408, 429, 500, 502, 503, 504].includes(error.status);
  }
}

// 서명 생성
const signaturePayload = JSON.stringify({
  messageId: message.messageId,
  timestamp: message.timestamp,
  type: message.type,
  payload: message.payload
});

const signature = crypto
  .createHmac('sha256', apiSecret)
  .update(signaturePayload)
  .digest('hex');

message.signature = `sha256:${signature}`;

// 서명 검증
function verifySignature(message, apiSecret) {
  const [algorithm, hash] = message.signature.split(':');
  
  const calculated = crypto
    .createHmac(algorithm, apiSecret)
    .update(signaturePayload)
    .digest('hex');
  
  return crypto.timingSafeEqual(
    Buffer.from(hash),
    Buffer.from(calculated)
  );
}

6.6.3 재생 공격 방지

// 타임스탬프 검증 (5분 오차 허용)
const MAX_CLOCK_SKEW = 300;

function validateTimestamp(timestamp) {
  const messageTime = new Date(timestamp).getTime();
  const currentTime = Date.now();
  const difference = Math.abs(currentTime - messageTime);
  
  if (difference > MAX_CLOCK_SKEW * 1000) {
    throw new Error('요청 타임스탬프가 유효하지 않습니다');
  }
}

// TypeScript 예제: 청크 업로드로 대용량 파일 처리
async function uploadLargeArtwork(file: File) {
  const CHUNK_SIZE = 5 * 1024 * 1024; // 5MB chunks
  const totalChunks = Math.ceil(file.size / CHUNK_SIZE);
  
  for (let i = 0; i < totalChunks; i++) {
    const start = i * CHUNK_SIZE;
    const end = Math.min(start + CHUNK_SIZE, file.size);
    const chunk = file.slice(start, end);
    
    await uploadChunk(chunk, i, totalChunks);
  }
  
  // 모든 청크 업로드 후 병합 요청
  await finalizeUpload(file.name);
}

// 안전한 파일 업로드 검증
function validateUpload(file: File, metadata: ArtworkMetadata): ValidationResult {
  const checks = [
    // 1. MIME 타입 검증
    validateMimeType(file.type, metadata.format),
    
    // 2. 파일 크기 제한
    validateFileSize(file.size, MAX_FILE_SIZE),
    
    // 3. 파일 내용 검사 (매직 넘버)
    validateFileContent(file),
    
    // 4. 바이러스 스캔
    await scanForVirus(file),
    
    // 5. 메타데이터 스키마 검증
    validateSchema(metadata, WIA_ART_SCHEMA)
  ];
  
  return {
    valid: checks.every(c => c.passed),
    errors: checks.filter(c => !c.passed).map(c => c.error)
  };
}
