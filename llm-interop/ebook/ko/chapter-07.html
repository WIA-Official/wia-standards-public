7.1 통합 개요

이 장에서는 WIA LLM 상호운용 표준의 실제 통합을 다룹니다:

  • AI 제공자 통합 (OpenAI, Anthropic, Google, Local)

# WIA 메시지를 OpenAI 형식으로 변환
def translate_to_openai(wia_message):
    return {
        "model": extract_model(wia_message.header.to),
        "messages": [
            {
                "role": "user",
                "content": wia_message.payload.content
            }
        ],
        "max_tokens": wia_message.payload.parameters.maxTokens,
        "temperature": wia_message.payload.parameters.temperature
    }

# OpenAI 응답을 WIA 형식으로 변환
def translate_from_openai(openai_response, correlation_id):
    return {
        "header": {
            "messageType": "RESPONSE",
            "correlationId": correlation_id
        },
        "payload": {
            "status": "SUCCESS",
            "content": openai_response.choices[0].message.content,
            "metadata": {
                "tokensUsed": openai_response.usage.total_tokens,
                "model": openai_response.model
            }
        }
    }

Lambda 함수 구성

# serverless.yml
service: wia-llm-interop

provider:
  name: aws
  runtime: python3.11
  region: ap-northeast-2
  memorySize: 1024
  timeout: 30
  environment:
    WIA_CONFIG_PATH: /opt/config/wia-config.json

    handler: handlers/capability.handler
    events:
      - http:
          path: /capability
          method: post
      - http:
          path: /capability/query
          method: post

  message:
    handler: handlers/message.handler
    events:
      - http:
          path: /message
          method: post

  federation:
    handler: handlers/federation.handler
    events:
      - http:
          path: /federation
          method: post
      - http:
          path: /federation/{id}/task
          method: post

Lambda 핸들러 예시

# handlers/capability.py
import json
from wia_llm import CapabilityService

service = CapabilityService()

def handler(event, context):
    path = event['path']
    method = event['httpMethod']
    body = json.loads(event.get('body', '{}'))

    if path == '/capability' and method == 'POST':
        result = service.register(body)
        return {
            'statusCode': 201,
            'body': json.dumps(result)
        }

    elif path == '/capability/query' and method == 'POST':
        result = service.query(body)
        return {
            'statusCode': 200,
            'body': json.dumps(result)
        }

    return {'statusCode': 404}

7.9 LangChain 통합

LangChain용 WIA LLM 제공자

from langchain.llms.base import LLM
from wia_llm import WIAClient

class WIALLM(LLM):
    """LangChain용 WIA LLM 상호운용 제공자"""

    client: WIAClient
    target_did: str

    def __init__(self, wia_endpoint: str, target_did: str):
        self.client = WIAClient(wia_endpoint)
        self.target_did = target_did

    def _call(self, prompt: str, stop=None) -> str:
        message = {
            "header": {
                "to": self.target_did,
                "messageType": "QUERY"
            },
            "payload": {
                "content": prompt
            }
        }

        response = self.client.send_message(message)
        return response.payload.content

    @property
    def _llm_type(self) -> str:
        return "wia-llm"

# 사용 예시
from langchain.chains import LLMChain
from langchain.prompts import PromptTemplate

llm = WIALLM(
    wia_endpoint="https://api.wia.org/v1/llm",
    target_did="did:wia:llm:anthropic:claude-3-opus"
)

chain = LLMChain(
    llm=llm,
    prompt=PromptTemplate.from_template("분석: {text}")
)

result = chain.run(text="여기에 샘플 코드...")

7.10 LlamaIndex 통합

LlamaIndex용 WIA LLM

from llama_index.llms import CustomLLM
from wia_llm import WIAClient

class WIALlamaIndex(CustomLLM):
    """LlamaIndex용 WIA LLM 제공자"""

    def __init__(self, wia_endpoint: str, did: str):
        self.client = WIAClient(wia_endpoint)
        self.did = did

    def complete(self, prompt: str) -> str:
        response = self.client.query(self.did, prompt)
        return response.content

    def stream_complete(self, prompt: str):
        for chunk in self.client.stream_query(self.did, prompt):
            yield chunk

# RAG와 함께 사용
from llama_index import VectorStoreIndex, SimpleDirectoryReader

llm = WIALlamaIndex(
    wia_endpoint="https://api.wia.org/v1/llm",
    did="did:wia:llm:openai:gpt-4"
)

documents = SimpleDirectoryReader("./data").load_data()
index = VectorStoreIndex.from_documents(documents, llm=llm)
query_engine = index.as_query_engine()

response = query_engine.query("이것은 무엇에 관한 것인가요?")

7.11 SDK 예시

TypeScript SDK

import { WIAClient, Federation, Message } from '@wia/llm-sdk';

// 클라이언트 초기화
const client = new WIAClient({
    endpoint: 'https://api.wia.org/v1/llm',
    auth: { did: 'did:wia:llm:my-service', privateKey: '...' }
});

// 능력 등록
await client.capability.register({
    level: 3,
    domains: [{ code: 'CODE', proficiency: 0.9 }]
});

// 능력 쿼리
const matches = await client.capability.query({
    domains: ['CODE', 'ANALYSIS'],
    minLevel: 2
});

// 메시지 전송
const response = await client.message.send({
    to: 'did:wia:llm:openai:gpt-4',
    content: '이 코드를 분석해주세요...'
});

// 연합 생성
const federation = await client.federation.create({
    name: '보안 팀',
    topology: 'STAR',
    consensus: { algorithm: 'WEIGHTED' }
});

// 작업 제출
const task = await federation.submitTask({
    description: '보안 감사',
    decomposition: { strategy: 'HIERARCHICAL' }
});

Python SDK

from wia_llm import WIAClient, Federation

# 클라이언트 초기화
client = WIAClient(
    endpoint="https://api.wia.org/v1/llm",
    did="did:wia:llm:my-service",
    private_key="..."
)

# 능력 쿼리
matches = client.capability.query(
    domains=["CODE", "ANALYSIS"],
    min_level=3
)

# 스트리밍으로 메시지 전송
async for chunk in client.message.stream(
    to="did:wia:llm:anthropic:claude-3-opus",
    content="양자 컴퓨팅을 설명해주세요"
):
    print(chunk, end="")

# 연합 생성 및 사용
federation = client.federation.create(
    name="연구 팀",
    topology="MESH",
    members=["did:wia:llm:openai:gpt-4", "did:wia:llm:anthropic:claude-3-opus"]
)

result = await federation.submit_task(
    description="최근 AI 연구 요약",
    consensus="DEBATE"
)
