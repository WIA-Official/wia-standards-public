6.3 테스트 전략

// 테스트: 암호화 및 복호화
function testEncryption() {
  const plaintext = Buffer.from('테스트 메시지');
  const key = crypto.randomBytes(32);

  const encrypted = encryptAES(plaintext, key);
  const decrypted = decryptAES(encrypted, key);

  assert(plaintext.equals(decrypted));
  console.log('✓ 암호화 테스트 통과');
}

// 테스트: IV 고유성
function testIVUniqueness() {
  const plaintext = Buffer.from('동일한 메시지');
  const key = crypto.randomBytes(32);

  const enc1 = encryptAES(plaintext, key);
  const enc2 = encryptAES(plaintext, key);

  // 암호문이 달라야 함
  assert(!enc1.ciphertext.equals(enc2.ciphertext));
  console.log('✓ IV 고유성 테스트 통과');
}

// 테스트: 변조 감지
function testTamperDetection() {
  const encrypted = encryptAES('메시지', key);
  encrypted.ciphertext[0] ^= 0xFF; // 변조

  assert.throws(() => decryptAES(encrypted, key));
  console.log('✓ 변조 감지 테스트 통과');
}

6.4 성능 최적화

6.4.1 하드웨어 가속

  • AES-NI: Intel/AMD 명령어 세트 (10배+ 속도 향상)
  • ARM Crypto Extensions: 모바일/임베디드 가속
  • GPU 가속: 대용량 데이터셋 병렬 암호화

6.4.2 모범 사례

  • 대용량 파일에는 스트리밍 암호화 사용
  • 가능한 경우 하드웨어 가속 활성화
  • 적절한 버퍼링 (16KB-64KB 청크)
  • 독립적인 데이터는 병렬 암호화 고려

6.5 보안 코딩 관행

  • 민감한 메모리 정리: 사용 후 키 제로화
  • 상수 시간 연산: 타이밍 공격 방지
  • 입력 유효성 검사: 처리 전 모든 입력 검증
  • 에러 처리: 에러에서 정보 누출 금지
  • 코드 리뷰: 모든 암호화 코드에 보안 중심 리뷰

6.6 에러 처리 모범 사례

// 나쁜 예 - 정보 누출
function badDecrypt(ciphertext, key) {
  try {
    return decrypt(ciphertext, key);
  } catch (e) {
    throw new Error(`복호화 실패: ${e.message}`);
    // 공격자에게 유용한 정보 제공
  }
}

// 좋은 예 - 일반적인 에러
function goodDecrypt(ciphertext, key) {
  try {
    return decrypt(ciphertext, key);
  } catch (e) {
    log.error('복호화 실패', { errorId: generateId() });
    throw new Error('복호화 실패');
    // 최소한의 정보만 노출
  }
}
