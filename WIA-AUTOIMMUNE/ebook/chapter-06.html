Chapter 6: Flare Prediction Algorithms

Disease flares represent sudden worsening of autoimmune disease activity that can cause significant morbidity and reduced quality of life. This chapter presents machine learning algorithms and predictive models designed to identify early warning signs of impending flares, enabling proactive intervention and improved patient outcomes.

Predictive Medicine: WIA-AUTOIMMUNE leverages advanced machine learning techniques to analyze patterns in clinical, laboratory, and patient-reported data, providing clinicians with actionable predictions that can prevent or mitigate disease flares.

Effective flare prediction requires careful feature engineering to capture clinically meaningful patterns from raw data.

Feature Categories

Category Features Clinical Rationale
Laboratory Trends CRP velocity, ESR change, complement trends Rising inflammatory markers precede flares
PRO Patterns Pain trajectory, fatigue trend, morning stiffness duration Symptoms often worsen before clinical flare
Activity Data Step count decline, sleep disruption, activity variability Functional decline indicates disease worsening
Medication Adherence patterns, recent dose changes, steroid use Treatment gaps increase flare risk

// Feature Engineering Pipeline
class FeatureEngineeringPipeline {
    constructor(private config: FeatureConfig) {}

    async extractFeatures(patientData: PatientDataBundle): Promise {
        const features: FeatureVector = {};

        // Laboratory features
        features.lab = await this.extractLabFeatures(patientData.laboratory);

        // Patient-reported outcome features
        features.pro = await this.extractPROFeatures(patientData.surveys);

        // Activity and wearable features
        features.activity = await this.extractActivityFeatures(patientData.wearables);

        // Medication features
        features.medication = await this.extractMedicationFeatures(patientData.medications);

        // Historical features
        features.historical = await this.extractHistoricalFeatures(patientData.history);

        // Interaction features
        features.interactions = await this.computeInteractionFeatures(features);

        return this.normalizeFeatures(features);
    }

    private async extractLabFeatures(labs: LabResult[]): Promise {
        const windows = [7, 14, 30, 90]; // days

        return {
            crp: {
                latest: this.getLatestValue(labs, 'CRP'),
                velocity_7d: this.computeVelocity(labs, 'CRP', 7),
                velocity_30d: this.computeVelocity(labs, 'CRP', 30),
                trend: this.detectTrend(labs, 'CRP', 30),
                above_normal_ratio: this.computeAbnormalRatio(labs, 'CRP', 30)
            },
            esr: {
                latest: this.getLatestValue(labs, 'ESR'),
                velocity_7d: this.computeVelocity(labs, 'ESR', 7),
                trend: this.detectTrend(labs, 'ESR', 30)
            },
            complement: {
                c3_latest: this.getLatestValue(labs, 'C3'),
                c4_latest: this.getLatestValue(labs, 'C4'),
                c3_trend: this.detectTrend(labs, 'C3', 30),
                c4_trend: this.detectTrend(labs, 'C4', 30)
            },
            cbc: {
                wbc_latest: this.getLatestValue(labs, 'WBC'),
                platelet_trend: this.detectTrend(labs, 'Platelets', 30),
                lymphocyte_ratio: this.computeRatio(labs, 'Lymphocytes', 'WBC')
            }
        };
    }

    private async extractPROFeatures(surveys: SurveyResponse[]): Promise {
        return {
            pain: {
                average_7d: this.computeAverage(surveys, 'pain', 7),
                max_7d: this.computeMax(surveys, 'pain', 7),
                trend_14d: this.detectTrend(surveys, 'pain', 14),
                volatility_7d: this.computeStdDev(surveys, 'pain', 7)
            },
            fatigue: {
                average_7d: this.computeAverage(surveys, 'fatigue', 7),
                trend_14d: this.detectTrend(surveys, 'fatigue', 14),
                correlation_with_activity: this.computeCorrelation('fatigue', 'activity')
            },
            stiffness: {
                duration_average_7d: this.computeAverage(surveys, 'morning_stiffness_duration', 7),
                duration_trend: this.detectTrend(surveys, 'morning_stiffness_duration', 14)
            },
            global: {
                patient_global_7d: this.computeAverage(surveys, 'patient_global', 7),
                trend: this.detectTrend(surveys, 'patient_global', 14)
            }
        };
    }
}

// Gradient Boosting Model for Flare Prediction
import { XGBoostClassifier } from '@wia/ml';

class FlareXGBoostModel implements PredictionModel {
    private model: XGBoostClassifier;

    constructor() {
        this.model = new XGBoostClassifier({
            objective: 'binary:logistic',
            max_depth: 6,
            learning_rate: 0.1,
            n_estimators: 200,
            subsample: 0.8,
            colsample_bytree: 0.8,
            min_child_weight: 1,
            reg_alpha: 0.1,
            reg_lambda: 1.0,
            scale_pos_weight: 3.5, // Handle class imbalance
            eval_metric: ['auc', 'aucpr']
        });
    }

    async train(features: FeatureMatrix, labels: number[]): Promise {
        // Split for validation
        const [trainX, valX, trainY, valY] = this.trainTestSplit(features, labels, 0.2);

        // Train with early stopping
        await this.model.fit(trainX, trainY, {
            eval_set: [[valX, valY]],
            early_stopping_rounds: 20,
            verbose: false
        });
    }

    async predict(features: FeatureVector): Promise {
        const probability = await this.model.predictProba(features);
        const explanation = await this.explainPrediction(features);

        return {
            flareRisk: probability[1],
            confidence: this.computeConfidence(probability),
            riskLevel: this.categorizeRisk(probability[1]),
            timeHorizon: '7_days',
            explanation: explanation,
            contributingFactors: this.getTopFactors(features, explanation)
        };
    }

    private async explainPrediction(features: FeatureVector): Promise {
        const shap = await this.model.shapValues(features);
        return {
            baseValue: shap.expectedValue,
            featureContributions: shap.values.map((val, idx) => ({
                feature: this.featureNames[idx],
                contribution: val,
                featureValue: features[idx]
            })).sort((a, b) => Math.abs(b.contribution) - Math.abs(a.contribution))
        };
    }
}

class FlareLSTMModel implements PredictionModel {
    private model: Sequential;

    constructor() {
        this.model = new Sequential([
            new LSTM({
                units: 64,
                returnSequences: true,
                inputShape: [30, 25] // 30 days, 25 features
            }),
            new Dropout(0.3),
            new LSTM({
                units: 32,
                returnSequences: false
            }),
            new Dropout(0.3),
            new Dense({ units: 16, activation: 'relu' }),
            new Dense({ units: 1, activation: 'sigmoid' })
        ]);

        this.model.compile({
            optimizer: new Adam({ learningRate: 0.001 }),
            loss: 'binaryFocalCrossentropy',
            metrics: ['auc', 'precision', 'recall']
        });
    }

    async train(sequences: SequenceData[], labels: number[]): Promise {
        const [trainSeq, valSeq, trainY, valY] = this.trainTestSplit(sequences, labels, 0.2);

        await this.model.fit(trainSeq, trainY, {
            epochs: 100,
            batchSize: 32,
            validationData: [valSeq, valY],
            callbacks: [
                new EarlyStopping({ patience: 10, restoreBestWeights: true }),
                new ReduceLROnPlateau({ factor: 0.5, patience: 5 })
            ]
        });
    }
}

Ensemble Method

The ensemble approach combines predictions from multiple models to improve overall accuracy and reliability.

// Ensemble Prediction Engine
class EnsemblePredictionEngine {
    private models: Map;
    private weights: Map;
    private calibrator: IsotonicRegression;

    constructor() {
        this.models = new Map();
        this.weights = new Map();
        this.calibrator = new IsotonicRegression();
    }

    async addModel(name: string, model: PredictionModel, weight: number): Promise {
        this.models.set(name, model);
        this.weights.set(name, weight);
    }

    async predict(patientData: PatientDataBundle): Promise {
        const features = await this.featurePipeline.extractFeatures(patientData);
        const predictions: ModelPrediction[] = [];

        // Get predictions from all models
        for (const [name, model] of this.models) {
            const pred = await model.predict(features);
            predictions.push({
                modelName: name,
                probability: pred.flareRisk,
                weight: this.weights.get(name)!,
                explanation: pred.explanation
            });
        }

        // Weighted average
        const weightedProb = predictions.reduce((sum, p) =>
            sum + p.probability * p.weight, 0) /
            predictions.reduce((sum, p) => sum + p.weight, 0);

        // Calibrate probability
        const calibratedProb = this.calibrator.predict(weightedProb);

        // Aggregate explanations
        const aggregatedExplanation = this.aggregateExplanations(predictions);

        return {
            flareRisk: calibratedProb,
            riskLevel: this.categorizeRisk(calibratedProb),
            confidence: this.computeEnsembleConfidence(predictions),
            timeHorizon: '7_days',
            individualPredictions: predictions,
            aggregatedExplanation: aggregatedExplanation,
            recommendedActions: this.generateRecommendations(calibratedProb, aggregatedExplanation)
        };
    }

    private categorizeRisk(probability: number): RiskLevel {
        if (probability < 0.2) return 'low';
        if (probability < 0.5) return 'moderate';
        if (probability < 0.8) return 'high';
        return 'critical';
    }
}
