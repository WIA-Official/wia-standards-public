제6장

안전 시스템 및 비상 프로토콜

장 개요

이 장에서는 농업 로봇의 안전한 작동을 보장하는 중요한 시스템을 탐구합니다. 비상 정지 절차, 충돌 회피, 지오펜싱, ISO 18497 규정 준수, 그리고 안전 시스템의 검증 및 테스트를 다룹니다.

6.1 비상 정지(E-STOP) 시스템

비상 정지는 농업 로봇의 가장 중요한 안전 기능입니다. WIA 표준은 100밀리초 이내에 로봇을 정지시킬 수 있는 다층 E-STOP 시스템을 요구합니다.

6.1.1 E-STOP 트리거 조건

조건 감지 방법 응답 시간
5m 안전 구역 내 인간 감지 LiDAR + 카메라 <100ms
장애물 감지 실패 센서 자체 진단 <200ms
수동 비상 버튼 누름 물리적 버튼 즉시
통신 손실 >30초 하트비트 모니터링 30초 후
중요 시스템 고장 시스템 모니터링 <500ms
경로 편차 >2m GPS 추적 <1초

6.1.2 E-STOP 실행 절차

class EmergencyStopSystem {
  executeEmergencyStop(reason, sensorData) {
    const startTime = performance.now();
    
    try {
      // 1단계: 즉시 - 모든 액추에이터 전원 차단 (<100ms)
      this.cutActuatorPower();
      
      // 2단계: 활성화 - 기계식 브레이크 작동
      this.engageMechanicalBrakes();
      
      // 3단계: 브로드캐스트 - 모든 시스템에 E-STOP 경고 전송
      this.broadcastEmergencyStop({
        type: "EMERGENCY_STOP",
        robotId: this.robotId,
        timestamp: new Date().toISOString(),
        reason: reason,
        position: this.getCurrentPosition(),
        severity: "CRITICAL",
        sensorData: sensorData
      });
      
      // 4단계: 기록 - 전체 센서 상태로 이벤트 기록
      this.logEmergencyEvent({
        reason: reason,
        executionTime: performance.now() - startTime,
        sensorSnapshot: this.captureSensorSnapshot(),
        robotState: this.captureRobotState()
      });
      
      // 5단계: 대기 - 수동 검사 및 재설정 필요
      this.enterSafeMode();
      this.requireManualReset();
      
    } catch (error) {
      // E-STOP 실행 실패는 치명적 오류
      this.triggerHardwareFailsafe();
      this.logCriticalError(error);
    }
  }

  cutActuatorPower() {
    // 하드웨어 릴레이를 통해 전원 차단
    this.hardwareRelay.disable();
    this.motorController.emergencyStop();
    this.hydraulics.shutoff();
  }

  engageMechanicalBrakes() {
    // 기계식 브레이크는 전원 손실 시 자동으로 작동 (fail-safe)
    this.brakeSolenoid.deenergize();  // 전원 차단 시 브레이크 작동
  }

  requireManualReset() {
    // E-STOP 후 자동 재개 방지
    this.state = "EMERGENCY_STOPPED";
    this.allowAutoResume = false;
    
    // 운영자 재설정 필요
    this.notifyOperator({
      type: "EMERGENCY_STOP_REQUIRES_RESET",
      message: "수동 검사 및 재설정 필요",
      resetProcedure: [
        "1. 로봇 및 주변 검사",
        "2. E-STOP 원인 해결",
        "3. 물리적 재설정 버튼 누름",
        "4. 시스템 자체 진단 실행",
        "5. 운영자 승인 후 재개"
      ]
    });
  }
}

Fail-Safe 설계 원칙

모든 안전 시스템은 fail-safe 원칙을 따라야 합니다: 전원 손실 또는 시스템 고장 시 로봇은 자동으로 안전한 상태(정지)로 전환되어야 합니다. 예: 브레이크 솔레노이드는 평상시 전원을 소비하여 브레이크를 풀고, 전원 손실 시 스프링력으로 브레이크가 작동합니다.

6.2 충돌 회피 시스템

충돌 회피는 장애물과의 접촉을 방지하는 능동적 안전 시스템입니다. WIA 표준은 다층 감지 및 응답 전략을 정의합니다.

6.2.1 감지 범위 및 응답

class CollisionAvoidanceSystem {
  constructor() {
    this.detectionZones = {
      IMMEDIATE_STOP: { range: [0, 5], action: "EMERGENCY_STOP", responseTime: 100 },
      SLOW_50_PERCENT: { range: [5, 15], action: "SLOW_TO_50", responseTime: 500 },
      SLOW_75_PERCENT: { range: [15, 30], action: "SLOW_TO_75", responseTime: 1000 },
      MONITOR: { range: [30, 50], action: "MONITOR_AND_ALERT", responseTime: null }
    };
  }

  async processObstacles(obstacles) {
    const criticalObstacles = [];
    
    for (const obstacle of obstacles) {
      const distance = this.calculateDistance(obstacle);
      const zone = this.determineZone(distance);
      const timeToCollision = distance / this.currentSpeed;
      
      const response = {
        obstacle: obstacle,
        distance: distance,
        zone: zone,
        timeToCollision: timeToCollision,
        action: zone.action
      };
      
      // 인간 또는 동물은 항상 최우선
      if (obstacle.type === "HUMAN" || obstacle.type === "ANIMAL") {
        response.action = "EMERGENCY_STOP";
        response.priority = "CRITICAL";
      }
      
      criticalObstacles.push(response);
    }
    
    // 가장 위험한 장애물에 따라 행동
    const mostCritical = this.selectMostCritical(criticalObstacles);
    await this.executeResponse(mostCritical);
    
    return criticalObstacles;
  }

  determineZone(distance) {
    for (const [zoneName, zone] of Object.entries(this.detectionZones)) {
      if (distance >= zone.range[0] && distance < zone.range[1]) {
        return zone;
      }
    }
    return this.detectionZones.MONITOR;
  }

  async executeResponse(response) {
    switch (response.action) {
      case "EMERGENCY_STOP":
        this.emergencyStop.execute(
          `Obstacle detected: ${response.obstacle.type}`,
          response
        );
        break;
        
      case "SLOW_TO_50":
        await this.speedController.setSpeed(this.currentSpeed * 0.5, 500);
        this.logger.warn("Slowing to 50%", response);
        break;
        
      case "SLOW_TO_75":
        await this.speedController.setSpeed(this.currentSpeed * 0.75, 1000);
        this.logger.info("Slowing to 75%", response);
        break;
        
      case "MONITOR_AND_ALERT":
        this.alertSystem.notify("Obstacle ahead", response);
        break;
    }
  }
}

6.2.2 회피 경로 생성

class AvoidancePathPlanner {
  generateAvoidancePath(obstacle, currentPath, robotState) {
    // 1. 장애물 주변 안전 버퍼 계산
    const safetyMargin = 2.0;  // 미터
    const obstacleBuffer = this.expandObstacleBoundary(
      obstacle, 
      safetyMargin
    );
    
    // 2. 가능한 회피 옵션 평가
    const options = [
      this.tryLeftAvoidance(obstacleBuffer, currentPath, robotState),
      this.tryRightAvoidance(obstacleBuffer, currentPath, robotState),
      this.tryStopAndWait(obstacle, currentPath)
    ];
    
    // 3. 실행 가능한 옵션 필터링
    const feasibleOptions = options.filter(opt => 
      opt.feasible && opt.safetyScore > 0.7
    );
    
    if (feasibleOptions.length === 0) {
      // 회피 불가능 - 정지
      return {
        action: "STOP_AND_NOTIFY",
        reason: "No feasible avoidance path"
      };
    }
    
    // 4. 최상의 옵션 선택 (안전성, 효율성, 편안함)
    const bestOption = this.selectBestOption(feasibleOptions);
    
    return bestOption.path;
  }

  tryLeftAvoidance(obstacle, currentPath, robotState) {
    const avoidancePath = [];
    const lateralShift = 3.0;  // 3m 왼쪽으로
    
    // 현재 경로에서 벗어나기
    avoidancePath.push(this.createCurvedSegment(
      robotState.position,
      lateralShift,
      "LEFT"
    ));
    
    // 장애물 주변 이동
    avoidancePath.push(this.createStraightSegment(
      obstacle.length + 5.0  // 5m 추가 여유
    ));
    
    // 원래 경로로 복귀
    avoidancePath.push(this.createCurvedSegment(
      lateralShift,
      0,
      "RIGHT"
    ));
    
    // 실행 가능성 및 안전성 평가
    const feasible = this.checkPathFeasibility(avoidancePath, robotState);
    const safetyScore = this.calculateSafetyScore(avoidancePath);
    
    return {
      path: avoidancePath,
      feasible: feasible,
      safetyScore: safetyScore,
      addedDistance: this.calculatePathLength(avoidancePath),
      addedTime: this.estimateTime(avoidancePath)
    };
  }
}

6.3 ISO 18497 규정 준수

ISO 18497은 고도로 자동화된 농업 기계의 안전에 대한 국제 표준입니다. WIA-FARMING_ROBOT 표준은 이 표준과 완전히 호환됩니다.

6.3.1 ISO 18497 핵심 요구 사항

요구 사항 WIA 구현 검증 방법
보호 구역 모니터링 LiDAR + 카메라 융합 침투 테스트 (인간 크기 물체)
비상 정지 <100ms 하드웨어 E-STOP 시스템 응답 시간 측정 (100회)

감시 시스템 (일부 시나리오) 운영자 모니터링 대시보드 모니터링 시뮬레이션

6.4 날씨 응답 프로토콜

안전한 작동은 기상 조건 모니터링 및 적절한 응답이 필요합니다.

6.4.1 기상 한계 및 조치

class WeatherSafetyMonitor {
  constructor() {
    this.limits = {
      windSpeed: 40,        // km/h
      rainfall: 5,          // mm/h
      temperature: [-5, 45], // 섭씨
      visibility: 100,       // 미터
      lightning: 10         // km (10km 이내 번개 감지)
    };
  }

  async monitorWeatherConditions() {
    while (this.isOperating) {
      const weather = await this.fetchCurrentWeather();
      const safety = this.evaluateSafety(weather);
      
      if (!safety.isSafe) {
        await this.handleUnsafeConditions(safety);
      }
      
      await this.sleep(60000);  // 1분마다 확인
    }
  }

  evaluateSafety(weather) {
    const violations = [];
    const actions = [];
    
    // 풍속 확인
    if (weather.windSpeed > this.limits.windSpeed) {
      violations.push("Wind speed too high");
      actions.push("SUSPEND_ALL_OPERATIONS");
    } else if (weather.windSpeed > 25 && this.currentTask === "SPRAYING") {
      violations.push("Wind speed too high for spraying");
      actions.push("PAUSE_SPRAYING");
    }
    
    // 강우량 확인
    if (weather.precipitation > this.limits.rainfall) {
      violations.push("Heavy rain");
      actions.push("RETURN_TO_SHELTER");
    } else if (weather.precipitation > 1) {
      violations.push("Light rain");
      actions.push("CONTINUE_WITH_CAUTION");
    }
    
    // 온도 확인
    if (weather.temperature < this.limits.temperature[0] ||
        weather.temperature > this.limits.temperature[1]) {
      violations.push("Temperature outside safe range");
      actions.push("STOP_AND_PROTECT");
    }
    
    // 번개 확인
    if (weather.lightningDistance < this.limits.lightning) {
      violations.push("Lightning nearby");
      actions.push("IMMEDIATE_SHELTER");
    }
    
    return {
      isSafe: violations.length === 0,
      violations: violations,
      actions: actions,
      weather: weather
    };
  }

  async handleUnsafeConditions(safety) {
    const mostCriticalAction = this.determineMostCriticalAction(safety.actions);
    
    switch (mostCriticalAction) {
      case "IMMEDIATE_SHELTER":
        await this.emergencyReturnToShelter();
        break;
        
      case "SUSPEND_ALL_OPERATIONS":
        await this.suspendOperations("Unsafe weather conditions");
        break;
        
      case "RETURN_TO_SHELTER":
        await this.returnToShelter();
        break;
        
      case "PAUSE_SPRAYING":
        await this.pauseCurrentTask("Wind too strong for spraying");
        break;
        
      case "STOP_AND_PROTECT":
        await this.stopAndProtect();
        break;
    }
    
    this.notifyOperator({
      type: "WEATHER_SAFETY_ALERT",
      safety: safety,
      action: mostCriticalAction
    });
  }
}

6.5.2 안전 테스트 자동화

class SafetyTestSuite {
  async runDailySafetyChecks() {
    const results = {
      timestamp: new Date().toISOString(),
      tests: []
    };
    
    // 1. E-STOP 응답 시간 테스트
    results.tests.push(
      await this.testEmergencyStopResponseTime()
    );
    
    // 2. 장애물 감지 테스트
    results.tests.push(
      await this.testObstacleDetection()
    );
    
    // 3. 센서 자체 진단
    results.tests.push(
      await this.testSensorHealth()
    );
    
    // 4. 브레이크 기능 테스트
    results.tests.push(
      await this.testBrakeSystem()
    );
    
    // 모든 테스트가 통과했는지 확인
    const allPassed = results.tests.every(t => t.passed);
    
    if (!allPassed) {
      this.notifyMaintenance({
        type: "SAFETY_TEST_FAILURE",
        results: results
      });
      this.lockRobot("Safety test failed");
    }
    
    return results;
  }

  async testEmergencyStopResponseTime() {
    const measurements = [];
    
    for (let i = 0; i < 10; i++) {
      const startTime = performance.now();
      await this.triggerTestEmergencyStop();
      const stopTime = performance.now();
      const responseTime = stopTime - startTime;
      
      measurements.push(responseTime);
      await this.resetFromTestStop();
    }
    
    const avgResponseTime = measurements.reduce((a, b) => a + b) / measurements.length;
    const maxResponseTime = Math.max(...measurements);
    
    return {
      testName: "E-STOP Response Time",
      passed: maxResponseTime < 100,
      avgResponseTime: avgResponseTime,
      maxResponseTime: maxResponseTime,
      measurements: measurements
    };
  }
}

핵심 요약

  • 비상 정지 시스템은 100밀리초 이내에 로봇을 정지시켜야 하며, fail-safe 설계 원칙을 따라야 합니다
  • 충돌 회피는 거리 기반 응답 구역을 사용하여 장애물에 대한 점진적 응답을 제공합니다
  • ISO 18497 규정 준수는 보호 구역 모니터링, 비상 정지 성능 및 안전 관련 소프트웨어를 포함합니다
  • 기상 안전 모니터링은 풍속, 강우량, 온도 및 번개에 대한 한계를 시행합니다
  • 회피 경로 계획은 장애물 주변의 안전한 대체 경로를 생성하거나 회피가 불가능한 경우 정지합니다
  • 안전 시스템 검증은 일일, 주간 및 월간 테스트를 포함하여 모든 안전 기능이 작동하는지 확인합니다
  • 수동 재설정 요구 사항은 E-STOP 이벤트 후 운영자가 로봇을 검사하도록 보장합니다
