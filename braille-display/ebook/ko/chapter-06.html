제6장: Phase 3 - 통신 프로토콜

6.1 개요

통신 프로토콜은 점자 디스플레이 하드웨어와 호스트 컴퓨터 간의 데이터 교환을 정의합니다. WIA 점자 디스플레이 표준은 세 가지 주요 통신 방식을 지원합니다:

  • USB HID: 가장 일반적인 연결 방식, 드라이버 불필요
  • Bluetooth LE: 무선 연결, 모바일 기기 지원
  • 시리얼 통신: 레거시 시스템 호환성

이 장에서는 각 프로토콜의 세부사항, 메시지 형식, 보안 고려사항, 성능 최적화 방법을 다룹니다.

6.2 USB HID 프로토콜

USB HID (Human Interface Device)는 점자 디스플레이를 위한 표준 통신 프로토콜입니다. HID를 사용하면 별도의 드라이버 없이도 대부분의 운영체제에서 기기를 인식할 수 있습니다.

6.2.1 HID 디스크립터

점자 디스플레이를 위한 HID 디스크립터는 다음과 같이 정의됩니다:

// USB HID 디스크립터 (C 구조체)
static const uint8_t braille_hid_descriptor[] = {
    // Usage Page (Braille Display)
    0x05, 0x41,

    // Usage (Braille Display)
    0x09, 0x01,

    // Collection (Application)
    0xA1, 0x01,

        // Usage (Braille Row)
        0x09, 0x20,

        // Logical Minimum (0)
        0x15, 0x00,

        // Logical Maximum (255)
        0x26, 0xFF, 0x00,

        // Report Size (8 bits)
        0x75, 0x08,

        // Report Count (40 cells)
        0x95, 0x28,

        // Output (Data, Variable, Absolute)
        0x91, 0x02,

        // Usage (Router Set 1)
        0x09, 0x21,

        // Report Count (40 routing keys)
        0x95, 0x28,

        // Input (Data, Variable, Absolute)
        0x81, 0x02,

    // End Collection
    0xC0
};

6.2.2 HID 리포트 구조

리포트 타입 ID 크기 설명
Output Report 0x01 N+2 바이트 점자 셀 데이터 쓰기 (N=셀 개수)
Input Report 0x02 N+2 바이트 라우팅 키 상태 읽기
Feature Report 0x03 64 바이트 기기 정보 및 설정
Input Report 0x04 4 바이트 버튼 이벤트

6.2.3 Output Report (셀 데이터 쓰기)

/*
 * Output Report 형식 (점자 셀 쓰기)
 *
 * Byte 0: Report ID (0x01)
 * Byte 1: Start Cell Index (0-based)
 * Byte 2-N: Braille Cell Data (각 바이트는 하나의 셀)
 */

struct BrailleOutputReport {
    uint8_t reportId;        // 0x01
    uint8_t startCell;       // 시작 셀 인덱스
    uint8_t cellData[MAX_CELLS];  // 점자 셀 데이터
} __attribute__((packed));

// 예제: "안녕" 쓰기
BrailleOutputReport report = {
    .reportId = 0x01,
    .startCell = 0,
    .cellData = {
        0b00110101,  // 안 (한글 점자)
        0b00011011,  // 녕
        0b00000000,  // 빈 셀
        // ... 나머지 셀
    }
};

// USB를 통해 전송
hid_write(device, (uint8_t*)&report, sizeof(report));

6.2.4 Input Report (라우팅 키)

/*
 * Input Report 형식 (라우팅 키 상태)
 *
 * Byte 0: Report ID (0x02)
 * Byte 1: Key State Flags
 * Byte 2-N: Routing Key Bitmap (비트 단위로 각 키 상태)
 */

struct RoutingKeyReport {
    uint8_t reportId;        // 0x02
    uint8_t flags;           // 플래그 (예약됨)
    uint8_t keyBitmap[(MAX_CELLS + 7) / 8];  // 비트맵
} __attribute__((packed));

// 라우팅 키 읽기
RoutingKeyReport report;
hid_read(device, (uint8_t*)&report, sizeof(report));

// 특정 키가 눌렸는지 확인
bool isKeyPressed(const RoutingKeyReport* report, int keyIndex) {
    int byteIndex = keyIndex / 8;
    int bitIndex = keyIndex % 8;
    return (report->keyBitmap[byteIndex] & (1 << bitIndex)) != 0;
}

6.2.6 USB HID 구현 예제

#include <hidapi.h>

class UsbHidBrailleDisplay {
private:
    hid_device* device;
    uint8_t cellCount;

public:
    // 디바이스 열기
    bool open(uint16_t vendorId, uint16_t productId) {
        device = hid_open(vendorId, productId, NULL);
        if (!device) {
            return false;
        }

        // 논블로킹 모드 설정
        hid_set_nonblocking(device, 1);

        // 기기 정보 읽기
        BrailleDeviceInfo info;
        hid_get_feature_report(device, (uint8_t*)&info, sizeof(info));
        cellCount = info.cellCount;

        return true;
    }

    // 셀 쓰기
    bool writeCells(uint8_t startCell, const uint8_t* data, uint8_t length) {
        BrailleOutputReport report;
        report.reportId = 0x01;
        report.startCell = startCell;
        memcpy(report.cellData, data, length);

        int result = hid_write(device, (uint8_t*)&report, length + 2);
        return result > 0;
    }

    // 라우팅 키 읽기
    bool readRoutingKeys(bool* keys) {
        RoutingKeyReport report;
        int result = hid_read(device, (uint8_t*)&report, sizeof(report));

        if (result > 0 && report.reportId == 0x02) {
            for (int i = 0; i < cellCount; i++) {
                keys[i] = isKeyPressed(&report, i);
            }
            return true;
        }

        return false;
    }

    // 닫기
    void close() {
        if (device) {
            hid_close(device);
            device = nullptr;
        }
    }
};

6.3 Bluetooth LE GATT 서비스

Bluetooth Low Energy (BLE)는 모바일 기기와의 무선 연결을 위한 표준입니다. WIA 점자 디스플레이는 GATT (Generic Attribute Profile)를 사용하여 BLE 통신을 구현합니다.

6.3.5 BLE 연결 예제 (Web Bluetooth)

// Web Bluetooth API를 사용한 점자 디스플레이 연결

class BleBrailleDisplay {
    constructor() {
        this.device = null;
        this.server = null;
        this.service = null;
        this.characteristics = {};
    }

    async connect() {
        // 기기 스캔 및 연결
        this.device = await navigator.bluetooth.requestDevice({
            filters: [{
                services: ['0000FE60-0000-1000-8000-00805F9B34FB']
            }]
        });

        // GATT 서버 연결
        this.server = await this.device.gatt.connect();

        // 서비스 가져오기
        this.service = await this.server.getPrimaryService(
            '0000FE60-0000-1000-8000-00805F9B34FB'
        );

        // 특성 가져오기
        this.characteristics.cellData = await this.service.getCharacteristic(
            '0000FE61-0000-1000-8000-00805F9B34FB'
        );
        this.characteristics.routingKeys = await this.service.getCharacteristic(
            '0000FE62-0000-1000-8000-00805F9B34FB'
        );
        this.characteristics.deviceInfo = await this.service.getCharacteristic(
            '0000FE64-0000-1000-8000-00805F9B34FB'
        );

        // 라우팅 키 알림 설정
        await this.setupNotifications();
    }

    async setupNotifications() {
        await this.characteristics.routingKeys.startNotifications();

        this.characteristics.routingKeys.addEventListener(
            'characteristicvaluechanged',
            this.handleRoutingKey.bind(this)
        );
    }

    handleRoutingKey(event) {
        const value = event.target.value;
        const eventType = value.getUint8(0);
        const cellIndex = value.getUint8(1);

        if (eventType === 1) {  // Press
            this.onRoutingKeyPress(cellIndex);
        }
    }

    async write(text) {
        const brailleCells = this.textToBraille(text);
        await this.writeCells(0, brailleCells);
    }

    async writeCells(startCell, cellData) {
        const data = new Uint8Array(cellData.length + 2);
        data[0] = startCell;
        data[1] = cellData.length;
        data.set(cellData, 2);

        await this.characteristics.cellData.writeValue(data);
    }

    async disconnect() {
        if (this.device && this.device.gatt.connected) {
            await this.device.gatt.disconnect();
        }
    }
}

// 사용 예제
const display = new BleBrailleDisplay();
await display.connect();
await display.write("안녕하세요!");

6.4 직렬 통신 (레거시 지원)

일부 레거시 점자 디스플레이는 RS-232 직렬 통신을 사용합니다. WIA 표준은 하위 호환성을 위해 직렬 통신도 지원합니다.

6.4.1 직렬 통신 설정

매개변수 기본값 설명
Baud Rate 9600 초당 비트 수
Data Bits 8 데이터 비트
Parity None 패리티 검사
Stop Bits 1 정지 비트
Flow Control None 흐름 제어

6.4.2 직렬 프로토콜 메시지 형식

/*
 * 직렬 프로토콜 패킷 형식
 *
 * Byte 0: Start Marker (0xFF)
 * Byte 1: Command ID
 * Byte 2: Data Length
 * Byte 3-N: Data
 * Byte N+1: Checksum (XOR of all bytes)
 * Byte N+2: End Marker (0xFE)
 */

struct SerialPacket {
    uint8_t startMarker;     // 0xFF
    uint8_t commandId;
    uint8_t dataLength;
    uint8_t data[256];
    uint8_t checksum;
    uint8_t endMarker;       // 0xFE
} __attribute__((packed));

// 명령 ID
enum SerialCommand {
    CMD_WRITE_CELLS = 0x01,
    CMD_READ_KEYS = 0x02,
    CMD_GET_INFO = 0x03,
    CMD_RESET = 0x04,
    CMD_ACK = 0x80,
    CMD_NACK = 0x81
};

6.4.3 체크섬 계산

uint8_t calculateChecksum(const uint8_t* data, size_t length) {
    uint8_t checksum = 0;
    for (size_t i = 0; i < length; i++) {
        checksum ^= data[i];
    }
    return checksum;
}

bool verifyChecksum(const SerialPacket* packet) {
    uint8_t calculated = 0;
    calculated ^= packet->startMarker;
    calculated ^= packet->commandId;
    calculated ^= packet->dataLength;

    for (int i = 0; i < packet->dataLength; i++) {
        calculated ^= packet->data[i];
    }

    return calculated == packet->checksum;
}

6.4.4 직렬 통신 구현 예제

#include <termios.h>
#include <fcntl.h>
#include <unistd.h>

class SerialBrailleDisplay {
private:
    int fd;

    bool sendPacket(const SerialPacket* packet) {
        size_t totalSize = 5 + packet->dataLength;
        ssize_t written = write(fd, packet, totalSize);
        return written == totalSize;
    }

    bool receivePacket(SerialPacket* packet, int timeoutMs) {
        // 타임아웃 설정
        fd_set readfds;
        struct timeval timeout;
        timeout.tv_sec = timeoutMs / 1000;
        timeout.tv_usec = (timeoutMs % 1000) * 1000;

        FD_ZERO(&readfds);
        FD_SET(fd, &readfds);

        if (select(fd + 1, &readfds, NULL, NULL, &timeout) <= 0) {
            return false;  // 타임아웃
        }

        // 패킷 읽기
        if (read(fd, packet, sizeof(SerialPacket)) < 5) {
            return false;
        }

        return verifyChecksum(packet);
    }

public:
    bool open(const char* portName, int baudRate = 9600) {
        fd = ::open(portName, O_RDWR | O_NOCTTY);
        if (fd < 0) {
            return false;
        }

        // 직렬 포트 설정
        struct termios tty;
        memset(&tty, 0, sizeof(tty));

        if (tcgetattr(fd, &tty) != 0) {
            return false;
        }

        // Baud rate 설정
        cfsetospeed(&tty, B9600);
        cfsetispeed(&tty, B9600);

        // 8N1 설정
        tty.c_cflag &= ~PARENB;   // 패리티 없음
        tty.c_cflag &= ~CSTOPB;   // 정지 비트 1
        tty.c_cflag &= ~CSIZE;
        tty.c_cflag |= CS8;        // 데이터 비트 8

        // Raw 모드
        tty.c_lflag &= ~(ICANON | ECHO | ECHOE | ISIG);
        tty.c_iflag &= ~(IXON | IXOFF | IXANY);
        tty.c_oflag &= ~OPOST;

        // 적용
        if (tcsetattr(fd, TCSANOW, &tty) != 0) {
            return false;
        }

        return true;
    }

    bool writeCells(uint8_t startCell, const uint8_t* data, uint8_t length) {
        SerialPacket packet;
        packet.startMarker = 0xFF;
        packet.commandId = CMD_WRITE_CELLS;
        packet.dataLength = length + 1;
        packet.data[0] = startCell;
        memcpy(&packet.data[1], data, length);
        packet.checksum = calculateChecksum((uint8_t*)&packet, 3 + packet.dataLength);
        packet.endMarker = 0xFE;

        if (!sendPacket(&packet)) {
            return false;
        }

        // ACK 대기
        SerialPacket ack;
        if (!receivePacket(&ack, 1000)) {
            return false;
        }

        return ack.commandId == CMD_ACK;
    }

    void close() {
        if (fd >= 0) {
            ::close(fd);
            fd = -1;
        }
    }
};

BRLTTY는 Linux 시스템에서 널리 사용되는 점자 디스플레이 드라이버입니다. WIA 표준은 BRLTTY와의 호환성을 보장합니다.

6.5.3 BRLTTY 드라이버 구현 예제

// WIA BRLTTY 드라이버 구현

static int wia_construct(BrailleDisplay *brl, char **parameters, const char *device) {
    // WIA 디바이스 초기화
    WiaDevice *wia = malloc(sizeof(WiaDevice));
    if (!wia) return 0;

    if (!wia_open(wia, device)) {
        free(wia);
        return 0;
    }

    brl->data = wia;
    brl->textColumns = wia->cellCount;
    brl->textRows = 1;

    return 1;
}

static void wia_destruct(BrailleDisplay *brl) {
    WiaDevice *wia = brl->data;
    if (wia) {
        wia_close(wia);
        free(wia);
    }
}

static int wia_writeWindow(BrailleDisplay *brl, const wchar_t *text) {
    WiaDevice *wia = brl->data;
    uint8_t brailleCells[MAX_CELLS];

    // 유니코드 텍스트를 점자로 변환
    int length = textToBraille(text, brailleCells, wia->cellCount);

    // 디스플레이에 쓰기
    return wia_writeCells(wia, 0, brailleCells, length);
}

static int wia_readCommand(BrailleDisplay *brl, KeyTableCommandContext context) {
    WiaDevice *wia = brl->data;
    WiaEvent event;

    // 이벤트 읽기 (논블로킹)
    if (!wia_readEvent(wia, &event, 0)) {
        return EOF;
    }

    // 이벤트를 BRLTTY 명령으로 변환
    switch (event.type) {
        case WIA_EVENT_ROUTING_KEY:
            return BRL_CMD_BLK(ROUTE) | event.data.routingKey.cellIndex;

        case WIA_EVENT_BUTTON:
            return mapButtonToCommand(event.data.button.buttonId);

        default:
            return EOF;
    }
}

0x10 KEY_EVENT Host ← Device 키 이벤트
0x11 ROUTING_EVENT Host ← Device 라우팅 키 이벤트
0x20 GET_INFO Host ↔ Device 기기 정보 요청/응답
0x21 SET_CONFIG Host → Device 설정 변경
0x22 GET_CONFIG Host ↔ Device 설정 조회
0xF0 RESET Host → Device 디바이스 리셋
0xFE ACK 양방향 확인 응답
0xFF NACK 양방향 오류 응답

6.6.2 데이터 인코딩

/*
 * 점자 셀 인코딩
 *
 * 6점식 점자:
 * Bit 0: 점 1
 * Bit 1: 점 2
 * Bit 2: 점 3
 * Bit 3: 점 4
 * Bit 4: 점 5
 * Bit 5: 점 6
 * Bit 6-7: 예약됨 (0)
 *
 * 8점식 점자:
 * Bit 0: 점 1
 * Bit 1: 점 2
 * Bit 2: 점 3
 * Bit 3: 점 4
 * Bit 4: 점 5
 * Bit 5: 점 6
 * Bit 6: 점 7
 * Bit 7: 점 8
 */

// 점자 셀 인코딩 예제
uint8_t encodeBraille6(bool dots[6]) {
    uint8_t cell = 0;
    for (int i = 0; i < 6; i++) {
        if (dots[i]) {
            cell |= (1 << i);
        }
    }
    return cell;
}

uint8_t encodeBraille8(bool dots[8]) {
    uint8_t cell = 0;
    for (int i = 0; i < 8; i++) {
        if (dots[i]) {
            cell |= (1 << i);
        }
    }
    return cell;
}

// 점자 셀 디코딩
void decodeBraille(uint8_t cell, bool dots[8], int dotCount) {
    for (int i = 0; i < dotCount; i++) {
        dots[i] = (cell & (1 << i)) != 0;
    }
}

6.7 보안 고려사항

6.7.1 데이터 암호화

민감한 정보를 다루는 경우, 점자 디스플레이 통신을 암호화해야 합니다:

// AES-128 암호화 예제

#include <openssl/aes.h>
#include <openssl/rand.h>

class SecureBrailleProtocol {
private:
    AES_KEY encryptKey;
    AES_KEY decryptKey;
    uint8_t iv[AES_BLOCK_SIZE];

public:
    bool initialize(const uint8_t* key) {
        // 암호화 키 설정
        if (AES_set_encrypt_key(key, 128, &encryptKey) != 0) {
            return false;
        }

        // 복호화 키 설정
        if (AES_set_decrypt_key(key, 128, &decryptKey) != 0) {
            return false;
        }

        // IV 생성
        RAND_bytes(iv, AES_BLOCK_SIZE);

        return true;
    }

    void encrypt(const uint8_t* plaintext, uint8_t* ciphertext, size_t length) {
        uint8_t ivCopy[AES_BLOCK_SIZE];
        memcpy(ivCopy, iv, AES_BLOCK_SIZE);

        AES_cbc_encrypt(plaintext, ciphertext, length, &encryptKey, ivCopy, AES_ENCRYPT);
    }

    void decrypt(const uint8_t* ciphertext, uint8_t* plaintext, size_t length) {
        uint8_t ivCopy[AES_BLOCK_SIZE];
        memcpy(ivCopy, iv, AES_BLOCK_SIZE);

        AES_cbc_encrypt(ciphertext, plaintext, length, &decryptKey, ivCopy, AES_DECRYPT);
    }
};

6.7.2 기기 인증

/*
 * 기기 인증 프로토콜
 *
 * 1. 호스트가 챌린지 (난수) 전송
 * 2. 디바이스가 개인 키로 서명하여 응답
 * 3. 호스트가 공개 키로 검증
 */

struct AuthenticationChallenge {
    uint8_t commandId;       // 0x30
    uint8_t challenge[32];   // 난수
};

struct AuthenticationResponse {
    uint8_t commandId;       // 0x31
    uint8_t signature[64];   // ECDSA 서명
};

bool authenticateDevice(int fd) {
    // 1. 챌린지 생성
    AuthenticationChallenge challenge;
    challenge.commandId = 0x30;
    RAND_bytes(challenge.challenge, 32);

    // 2. 챌린지 전송
    write(fd, &challenge, sizeof(challenge));

    // 3. 응답 수신
    AuthenticationResponse response;
    read(fd, &response, sizeof(response));

    // 4. 서명 검증
    return verifySignature(challenge.challenge, response.signature);
}

6.8.2 성능 최적화

// 배치 업데이트로 지연 시간 최소화

class OptimizedBrailleDisplay {
private:
    uint8_t cellBuffer[MAX_CELLS];
    bool dirtyFlags[MAX_CELLS];
    uint64_t lastUpdateTime;

    const uint64_t UPDATE_INTERVAL_MS = 16;  // 60 FPS

public:
    void setCell(int index, uint8_t value) {
        if (cellBuffer[index] != value) {
            cellBuffer[index] = value;
            dirtyFlags[index] = true;
        }
    }

    void flush() {
        uint64_t now = getCurrentTimeMs();

        // 프레임 레이트 제한
        if (now - lastUpdateTime < UPDATE_INTERVAL_MS) {
            return;
        }

        // 변경된 셀만 전송
        int startIndex = -1;
        for (int i = 0; i <= MAX_CELLS; i++) {
            if (i < MAX_CELLS && dirtyFlags[i]) {
                if (startIndex == -1) {
                    startIndex = i;
                }
            } else {
                if (startIndex != -1) {
                    // 연속된 더티 셀 전송
                    int length = i - startIndex;
                    writeCells(startIndex, &cellBuffer[startIndex], length);

                    // 플래그 초기화
                    memset(&dirtyFlags[startIndex], 0, length);
                    startIndex = -1;
                }
            }
        }

        lastUpdateTime = now;
    }
};

6.8.3 지연 시간 측정

// 지연 시간 프로파일링

class LatencyProfiler {
private:
    struct LatencyStats {
        uint64_t min;
        uint64_t max;
        uint64_t avg;
        uint64_t count;
    };

    std::map<std::string, LatencyStats> stats;

public:
    void measureOperation(const std::string& name, std::function<void()> operation) {
        auto start = std::chrono::high_resolution_clock::now();

        operation();

        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end - start);

        updateStats(name, duration.count());
    }

    void updateStats(const std::string& name, uint64_t latency) {
        auto& stat = stats[name];

        if (stat.count == 0) {
            stat.min = stat.max = stat.avg = latency;
            stat.count = 1;
        } else {
            stat.min = std::min(stat.min, latency);
            stat.max = std::max(stat.max, latency);
            stat.avg = (stat.avg * stat.count + latency) / (stat.count + 1);
            stat.count++;
        }
    }

    void printStats() {
        for (const auto& [name, stat] : stats) {
            printf("%s:\n", name.c_str());
            printf("  Min: %lu μs\n", stat.min);
            printf("  Max: %lu μs\n", stat.max);
            printf("  Avg: %lu μs\n", stat.avg);
            printf("  Count: %lu\n", stat.count);
        }
    }
};

// 사용 예제
LatencyProfiler profiler;

profiler.measureOperation("writeCells", [&]() {
    display.writeCells(0, brailleCells, 40);
});

profiler.printStats();
