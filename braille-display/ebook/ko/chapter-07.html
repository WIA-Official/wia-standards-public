7.2.1 NVDA 드라이버 구조

# NVDA 점자 드라이버 템플릿
# brailleDisplayDrivers/wia.py

import braille
import inputCore
import hwPortUtils
from logHandler import log

class BrailleDisplayDriver(braille.BrailleDisplayDriver):
    """WIA 점자 디스플레이 드라이버"""

    name = "wia"
    description = "WIA Braille Display"

    @classmethod
    def check(cls):
        """드라이버 사용 가능 여부 확인"""
        return True

    def __init__(self):
        """드라이버 초기화"""
        super(BrailleDisplayDriver, self).__init__()

        # USB를 통한 자동 검색
        for portInfo in hwPortUtils.listComPorts():
            if portInfo.get("bluetoothName", "").startswith("WIA"):
                self._dev = self._openPort(portInfo["port"])
                break
        else:
            raise RuntimeError("WIA 점자 디스플레이를 찾을 수 없습니다")

        # 기기 정보 읽기
        self._getDeviceInfo()

    def _openPort(self, port):
        """포트 열기"""
        import serial
        return serial.Serial(
            port,
            baudrate=9600,
            timeout=0.2,
            writeTimeout=0.2
        )

    def _getDeviceInfo(self):
        """기기 정보 가져오기"""
        # 정보 요청 명령 전송
        self._dev.write(b'\xFF\x20\x00\x20\xFE')

        # 응답 읽기
        response = self._dev.read(64)
        if response:
            # 응답 파싱
            self.numCells = response[35]
            log.info(f"WIA 점자 디스플레이 연결됨: {self.numCells}셀")

    def display(self, cells):
        """점자 셀 표시"""
        # NVDA의 셀 데이터를 WIA 프로토콜로 변환
        cellBytes = bytes(cells)

        # Output Report 구성
        data = b'\x01' + bytes([0, len(cellBytes)]) + cellBytes

        # 전송
        self._dev.write(data)

    def terminate(self):
        """드라이버 종료"""
        super(BrailleDisplayDriver, self).terminate()
        if hasattr(self, '_dev'):
            self._dev.close()

7.2.2 키 입력 처리

class BrailleDisplayDriver(braille.BrailleDisplayDriver):
    # ... (이전 코드)

    def _handleKeyPress(self):
        """키 입력 처리"""
        # 키 이벤트 읽기
        data = self._dev.read(8)
        if not data or len(data) < 2:
            return

        reportId = data[0]

        if reportId == 0x02:  # 라우팅 키
            self._handleRoutingKey(data)
        elif reportId == 0x04:  # 버튼
            self._handleButton(data)

    def _handleRoutingKey(self, data):
        """라우팅 키 처리"""
        # 비트맵에서 눌린 키 찾기
        for cellIndex in range(self.numCells):
            byteIndex = 2 + cellIndex // 8
            bitIndex = cellIndex % 8

            if data[byteIndex] & (1 << bitIndex):
                # 라우팅 키 이벤트 전송
                try:
                    inputCore.manager.executeGesture(
                        InputGesture(self, "routing", cellIndex)
                    )
                except inputCore.NoInputGestureAction:
                    pass

    def _handleButton(self, data):
        """버튼 처리"""
        buttonId = data[1]
        eventType = data[2]

        if eventType == 1:  # Press
            try:
                inputCore.manager.executeGesture(
                    InputGesture(self, "button", buttonId)
                )
            except inputCore.NoInputGestureAction:
                pass

class InputGesture(braille.BrailleDisplayGesture):
    """WIA 점자 디스플레이 제스처"""

    source = BrailleDisplayDriver.name

    def __init__(self, display, gestureType, value):
        super(InputGesture, self).__init__()

        self.display = display
        self.id = f"{gestureType}_{value}"

7.4.1 iOS 외부 액세서리 프레임워크

// WIABrailleDisplay.swift
// iOS용 WIA 점자 디스플레이 통합

import UIKit
import ExternalAccessory

class WIABrailleDisplay: NSObject {
    private var session: EASession?
    private var accessory: EAAccessory?

    // 지원하는 프로토콜
    static let protocolString = "com.wia.braille"

    // 초기화
    override init() {
        super.init()

        // 액세서리 연결 알림 등록
        NotificationCenter.default.addObserver(
            self,
            selector: #selector(accessoryConnected(_:)),
            name: .EAAccessoryDidConnect,
            object: nil
        )

        NotificationCenter.default.addObserver(
            self,
            selector: #selector(accessoryDisconnected(_:)),
            name: .EAAccessoryDidDisconnect,
            object: nil
        )

        // 이미 연결된 액세서리 확인
        connectToAccessory()
    }

    // 액세서리 연결
    private func connectToAccessory() {
        let manager = EAAccessoryManager.shared()

        for accessory in manager.connectedAccessories {
            if accessory.protocolStrings.contains(WIABrailleDisplay.protocolString) {
                self.accessory = accessory
                openSession()
                break
            }
        }
    }

    // 세션 열기
    private func openSession() {
        guard let accessory = self.accessory else { return }

        session = EASession(
            accessory: accessory,
            forProtocol: WIABrailleDisplay.protocolString
        )

        if let session = session {
            // 입력 스트림 설정
            session.inputStream?.delegate = self
            session.inputStream?.schedule(in: .main, forMode: .default)
            session.inputStream?.open()

            // 출력 스트림 설정
            session.outputStream?.schedule(in: .main, forMode: .default)
            session.outputStream?.open()

            print("WIA 점자 디스플레이 연결됨")
        }
    }

    // 점자 쓰기
    func write(brailleCells: [UInt8]) {
        guard let outputStream = session?.outputStream else { return }

        var packet = Data()
        packet.append(0x01)  // Report ID
        packet.append(0x00)  // Start cell
        packet.append(contentsOf: brailleCells)

        let written = packet.withUnsafeBytes { (bytes: UnsafeRawBufferPointer) -> Int in
            return outputStream.write(
                bytes.bindMemory(to: UInt8.self).baseAddress!,
                maxLength: packet.count
            )
        }

        if written != packet.count {
            print("쓰기 오류: \(written)/\(packet.count) 바이트")
        }
    }

    // 액세서리 연결 알림
    @objc private func accessoryConnected(_ notification: Notification) {
        if let accessory = notification.userInfo?[EAAccessoryKey] as? EAAccessory {
            if accessory.protocolStrings.contains(WIABrailleDisplay.protocolString) {
                self.accessory = accessory
                openSession()
            }
        }
    }

    // 액세서리 연결 해제 알림
    @objc private func accessoryDisconnected(_ notification: Notification) {
        session?.inputStream?.close()
        session?.outputStream?.close()
        session = nil
        accessory = nil
        print("WIA 점자 디스플레이 연결 해제됨")
    }
}

extension WIABrailleDisplay: StreamDelegate {

7.5.1 Android Accessibility Service

// WiaBrailleService.kt
// Android TalkBack 통합

package com.wia.braille

import android.accessibilityservice.AccessibilityService
import android.bluetooth.BluetoothAdapter
import android.bluetooth.BluetoothDevice
import android.bluetooth.BluetoothGatt
import android.bluetooth.BluetoothGattCallback
import android.bluetooth.BluetoothGattCharacteristic
import android.view.accessibility.AccessibilityEvent
import java.util.UUID

class WiaBrailleService : AccessibilityService() {

    companion object {
        // GATT 서비스 및 특성 UUID
        val SERVICE_UUID: UUID = UUID.fromString("0000FE60-0000-1000-8000-00805F9B34FB")
        val CELL_DATA_UUID: UUID = UUID.fromString("0000FE61-0000-1000-8000-00805F9B34FB")
        val ROUTING_KEY_UUID: UUID = UUID.fromString("0000FE62-0000-1000-8000-00805F9B34FB")
    }

    private var bluetoothGatt: BluetoothGatt? = null
    private var cellDataCharacteristic: BluetoothGattCharacteristic? = null
    private var routingKeyCharacteristic: BluetoothGattCharacteristic? = null

    override fun onServiceConnected() {
        super.onServiceConnected()

        // 점자 디스플레이 연결
        connectToBrailleDisplay()
    }

    override fun onAccessibilityEvent(event: AccessibilityEvent?) {
        event?.let {
            // 접근성 이벤트를 점자로 변환하여 표시
            val text = getEventText(it)
            displayBraille(text)
        }
    }

    override fun onInterrupt() {
        // 서비스 중단 처리
    }

    private fun connectToBrailleDisplay() {
        val bluetoothAdapter = BluetoothAdapter.getDefaultAdapter()
        val pairedDevices = bluetoothAdapter.bondedDevices

        // WIA 점자 디스플레이 찾기
        val wiaDevice = pairedDevices.find { device ->
            device.name?.startsWith("WIA") == true
        }

        wiaDevice?.let { device ->
            bluetoothGatt = device.connectGatt(
                this,
                false,
                gattCallback
            )
        }
    }

    private val gattCallback = object : BluetoothGattCallback() {
        override fun onConnectionStateChange(
            gatt: BluetoothGatt?,
            status: Int,
            newState: Int
        ) {
            if (newState == BluetoothGatt.STATE_CONNECTED) {
                gatt?.discoverServices()
            }
        }

        override fun onServicesDiscovered(gatt: BluetoothGatt?, status: Int) {
            if (status == BluetoothGatt.GATT_SUCCESS) {
                val service = gatt?.getService(SERVICE_UUID)

                cellDataCharacteristic = service?.getCharacteristic(CELL_DATA_UUID)
                routingKeyCharacteristic = service?.getCharacteristic(ROUTING_KEY_UUID)

                // 라우팅 키 알림 활성화
                routingKeyCharacteristic?.let { characteristic ->
                    gatt?.setCharacteristicNotification(characteristic, true)
                }
            }
        }

        override fun onCharacteristicChanged(
            gatt: BluetoothGatt?,
            characteristic: BluetoothGattCharacteristic?
        ) {
            if (characteristic?.uuid == ROUTING_KEY_UUID) {
                val data = characteristic.value
                handleRoutingKey(data)
            }
        }
    }

    private fun displayBraille(text: String) {
        cellDataCharacteristic?.let { characteristic ->
            // 텍스트를 점자로 변환
            val brailleCells = textToBraille(text)

            // 데이터 패킷 구성
            val packet = ByteArray(brailleCells.size + 2)
            packet[0] = 0  // Start cell
            packet[1] = brailleCells.size.toByte()
            System.arraycopy(brailleCells, 0, packet, 2, brailleCells.size)

            characteristic.value = packet
            bluetoothGatt?.writeCharacteristic(characteristic)
        }
    }

    private fun handleRoutingKey(data: ByteArray) {
        if (data.size >= 2) {
            val eventType = data[0]
            val cellIndex = data[1]

            if (eventType.toInt() == 1) {  // Press
                // TalkBack에 라우팅 이벤트 전달
                performGlobalAction(GLOBAL_ACTION_ACCESSIBILITY_BUTTON)
            }
        }
    }

    private fun getEventText(event: AccessibilityEvent): String {
        return event.text.joinToString(" ")
    }

    private fun textToBraille(text: String): ByteArray {
        // 텍스트를 점자로 변환 (간단한 구현)
        // 실제로는 더 복잡한 변환 로직 필요
        return text.toByteArray()
    }

    override fun onDestroy() {
        super.onDestroy()
        bluetoothGatt?.close()
    }
}

7.7.2 Web Bluetooth API 통합

// web-braille.js
// 웹에서 직접 점자 디스플레이 제어

class WebBrailleDisplay {
    constructor() {
        this.device = null;
        this.server = null;
        this.service = null;
        this.cellDataCharacteristic = null;
    }

    async connect() {
        try {
            // 블루투스 기기 요청
            this.device = await navigator.bluetooth.requestDevice({
                filters: [{
                    services: ['0000fe60-0000-1000-8000-00805f9b34fb']
                }]
            });

            // GATT 서버 연결
            this.server = await this.device.gatt.connect();

            // 서비스 가져오기
            this.service = await this.server.getPrimaryService(
                '0000fe60-0000-1000-8000-00805f9b34fb'
            );

            // 특성 가져오기
            this.cellDataCharacteristic = await this.service.getCharacteristic(
                '0000fe61-0000-1000-8000-00805f9b34fb'
            );

            console.log('WIA 점자 디스플레이 연결됨');
            return true;

        } catch (error) {
            console.error('연결 실패:', error);
            return false;
        }
    }

    async write(text) {
        if (!this.cellDataCharacteristic) {
            throw new Error('디스플레이가 연결되지 않았습니다');
        }

        // 텍스트를 점자로 변환
        const brailleCells = this.textToBraille(text);

        // 데이터 패킷 구성
        const data = new Uint8Array(brailleCells.length + 2);
        data[0] = 0;  // Start cell
        data[1] = brailleCells.length;
        data.set(brailleCells, 2);

        // 쓰기
        await this.cellDataCharacteristic.writeValue(data);
    }

    textToBraille(text) {
        // 간단한 ASCII to Braille 변환
        // 실제로는 더 복잡한 변환 로직 필요
        const encoder = new TextEncoder();
        return encoder.encode(text);
    }

    async disconnect() {
        if (this.device && this.device.gatt.connected) {
            await this.device.gatt.disconnect();
        }
    }
}

// 사용 예제
const brailleDisplay = new WebBrailleDisplay();

document.getElementById('connect-btn').addEventListener('click', async () => {
    await brailleDisplay.connect();
});

document.getElementById('write-btn').addEventListener('click', async () => {
    const text = document.getElementById('text-input').value;
    await brailleDisplay.write(text);
});

7.9 장 요약

이 장에서는 WIA 점자 디스플레이를 다양한 스크린 리더 및 플랫폼과 통합하는 방법을 다루었습니다:

  • NVDA: Python 기반 드라이버 개발, 제스처 매핑, 애드온 패키징
  • JAWS: 스크립트 작성, 설정 파일, DLL 드라이버 구현
  • VoiceOver/iOS: 외부 액세서리 프레임워크, Swift 통합
  • TalkBack/Android: Accessibility Service, Bluetooth GATT 연결
  • Orca/Linux: BRLTTY 통합, 사용자 정의 스크립트
  • 웹 접근성: ARIA 라이브 리전, Web Bluetooth API
  • 弘益人間: 포용적 설계, 오픈소스 기여, 커뮤니티 협력

이러한 통합을 통해 WIA 점자 디스플레이는 모든 플랫폼에서 시각 장애인의 디지털 접근성을 향상시킵니다. 弘益人間의 정신에 따라, 우리는 기술이 모든 사람에게 이로움을 주도록 노력합니다.
