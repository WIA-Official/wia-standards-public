// hello-braille.js
// 간단한 점자 디스플레이 프로그램

const { WiaBrailleDisplay } = require('@wia/braille-display');

async function main() {
    console.log('WIA 점자 디스플레이 시작...');

    // 1. 디스플레이 생성 및 연결
    const display = new WiaBrailleDisplay({
        autoConnect: true
    });

    try {
        await display.connect();
        console.log('연결 성공!');

        // 2. 기기 정보 출력
        const info = display.getInfo();
        console.log(`모델: ${info.manufacturer} ${info.model}`);
        console.log(`셀 개수: ${info.cellCount}`);

        // 3. 환영 메시지 표시
        await display.write('안녕하세요, WIA!');
        console.log('메시지 표시됨');

        // 4. 3초 대기
        await sleep(3000);

        // 5. 카운트다운
        for (let i = 3; i >= 1; i--) {
            await display.write(`카운트다운: ${i}`);
            await sleep(1000);
        }

        // 6. 작별 인사
        await display.write('안녕히 가세요!');
        await sleep(2000);

        // 7. 화면 지우기
        await display.clear();

    } catch (error) {
        console.error('오류:', error.message);
    } finally {
        // 8. 연결 해제
        await display.disconnect();
        console.log('연결 해제됨');
    }
}

// 유틸리티 함수
function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}

// 프로그램 실행
main().catch(console.error);

8.2.4 이벤트 처리 추가 (10분)

// interactive-braille.js
// 대화형 점자 디스플레이 프로그램

const { WiaBrailleDisplay } = require('@wia/braille-display');
const readline = require('readline');

async function main() {
    const display = new WiaBrailleDisplay();
    await display.connect();

    // 라우팅 키 이벤트 리스너
    display.on('routingKey', (event) => {
        console.log(`라우팅 키 ${event.cellIndex} 눌림`);
        display.write(`셀 ${event.cellIndex} 선택됨`);
    });

    // 버튼 이벤트 리스너
    display.on('button', (event) => {
        console.log(`버튼 ${event.buttonName} ${event.type}`);
    });

    // 오류 이벤트 리스너
    display.on('error', (error) => {
        console.error('디스플레이 오류:', error.message);
    });

    // 사용자 입력 처리
    const rl = readline.createInterface({
        input: process.stdin,
        output: process.stdout
    });

    console.log('텍스트를 입력하세요 (종료하려면 "quit" 입력):');

    rl.on('line', async (line) => {
        if (line.toLowerCase() === 'quit') {
            await display.disconnect();
            rl.close();
            process.exit(0);
        } else {
            await display.write(line);
            console.log('표시 완료');
        }
    });
}

main().catch(console.error);

// advanced-braille.js
// 고급 기능 시연

const { WiaBrailleDisplay } = require('@wia/braille-display');

async function demonstrateFeatures() {
    const display = new WiaBrailleDisplay();
    await display.connect();

    // 1. 스크롤 텍스트
    console.log('스크롤 텍스트 시연...');
    await display.scrollText(
        '弘益人間 - 널리 인간을 이롭게 하라. WIA는 모든 사람을 위한 기술을 만듭니다.',
        {
            speed: 300,
            loop: false
        }
    );

    await sleep(2000);

    // 2. 점자 셀 직접 제어
    console.log('점자 패턴 시연...');
    const pattern = new Uint8Array([
        0b00000001,  // ⠁
        0b00000011,  // ⠃
        0b00000111,  // ⠇
        0b00001111,  // ⠏
        0b00011111,  // ⠟
        0b00111111,  // ⠿
    ]);
    await display.setCells(0, pattern);

    await sleep(2000);

    // 3. 애니메이션
    console.log('애니메이션 시연...');
    for (let i = 0; i < display.getInfo().cellCount; i++) {
        await display.setCell(i, 0b11111111);
        await sleep(50);
    }

    await sleep(1000);
    await display.clear();

    await display.disconnect();
}

function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}

// test/braille-display.test.js
// Jest를 사용한 자동화 테스트

const { WiaBrailleDisplay } = require('@wia/braille-display');

describe('WIA Braille Display', () => {
    let display;

    beforeAll(async () => {
        display = new WiaBrailleDisplay();
        await display.connect();
    });

    afterAll(async () => {
        await display.disconnect();
    });

    test('should connect successfully', () => {
        expect(display.isConnected()).toBe(true);
    });

    test('should return device info', () => {
        const info = display.getInfo();
        expect(info).toHaveProperty('manufacturer');
        expect(info).toHaveProperty('model');
        expect(info).toHaveProperty('cellCount');
        expect(info.cellCount).toBeGreaterThan(0);
    });

    test('should write text', async () => {
        await expect(display.write('테스트')).resolves.not.toThrow();
    });

    test('should write braille cells', async () => {
        const cells = new Uint8Array([0x01, 0x03, 0x07]);
        await expect(display.setCells(0, cells)).resolves.not.toThrow();
    });

    test('should clear display', async () => {
        await expect(display.clear()).resolves.not.toThrow();
    });

    test('should handle routing key events', (done) => {
        display.once('routingKey', (event) => {
            expect(event).toHaveProperty('cellIndex');
            expect(event).toHaveProperty('type');
            done();
        });

        // 라우팅 키 시뮬레이션 (테스트 환경에서)
        display._simulateRoutingKey(0);
    });

    test('should handle errors gracefully', async () => {
        // 잘못된 셀 인덱스
        await expect(display.setCell(-1, 0xFF)).rejects.toThrow();
    });

    test('should support multiple languages', async () => {
        await expect(display.write('안녕하세요')).resolves.not.toThrow();
        await expect(display.write('Hello')).resolves.not.toThrow();
        await expect(display.write('こんにちは')).resolves.not.toThrow();
    });
});

8.3.3 성능 테스트

// test/performance.test.js
// 성능 벤치마크

const { WiaBrailleDisplay } = require('@wia/braille-display');

async function benchmarkWriteLatency() {
    const display = new WiaBrailleDisplay();
    await display.connect();

    const iterations = 100;
    const latencies = [];

    for (let i = 0; i < iterations; i++) {
        const start = performance.now();
        await display.write('테스트 메시지');
        const end = performance.now();

        latencies.push(end - start);
    }

    const avg = latencies.reduce((a, b) => a + b, 0) / latencies.length;
    const min = Math.min(...latencies);
    const max = Math.max(...latencies);

    console.log('쓰기 지연 시간 통계:');
    console.log(`  평균: ${avg.toFixed(2)}ms`);
    console.log(`  최소: ${min.toFixed(2)}ms`);
    console.log(`  최대: ${max.toFixed(2)}ms`);

    // 합격 기준: 평균 < 100ms
    expect(avg).toBeLessThan(100);

    await display.disconnect();
}

async function benchmarkThroughput() {
    const display = new WiaBrailleDisplay();
    await display.connect();

    const duration = 10000;  // 10초
    const start = performance.now();
    let updates = 0;

    while (performance.now() - start < duration) {
        await display.write(`Update ${updates}`);
        updates++;
    }

    const elapsed = performance.now() - start;
    const updatesPerSecond = (updates / elapsed) * 1000;

    console.log('처리량 통계:');
    console.log(`  총 업데이트: ${updates}`);
    console.log(`  초당 업데이트: ${updatesPerSecond.toFixed(2)}`);

    // 합격 기준: > 10 updates/sec
    expect(updatesPerSecond).toBeGreaterThan(10);

    await display.disconnect();
}

benchmarkWriteLatency();
benchmarkThroughput();

// debug-tool.js
// WIA 점자 디스플레이 디버깅 도구

const { WiaBrailleDisplay } = require('@wia/braille-display');

async function diagnose() {
    console.log('WIA 점자 디스플레이 진단 도구');
    console.log('━'.repeat(50));

    // 1. 기기 검색
    console.log('\n1. 기기 검색 중...');
    const devices = await WiaBrailleDisplay.discover();
    console.log(`   발견된 기기: ${devices.length}개`);

    devices.forEach((device, i) => {
        console.log(`   ${i + 1}. ${device.manufacturer} ${device.model}`);
        console.log(`      시리얼: ${device.serialNumber}`);
        console.log(`      연결: ${device.connectionType}`);
    });

    if (devices.length === 0) {
        console.log('\n❌ 기기를 찾을 수 없습니다.');
        console.log('   - USB 케이블 확인');
        console.log('   - Bluetooth 활성화 확인');
        console.log('   - 드라이버 설치 확인');
        return;
    }

    // 2. 연결 테스트
    console.log('\n2. 연결 테스트 중...');
    const display = new WiaBrailleDisplay();

    try {
        await display.connect();
        console.log('   ✓ 연결 성공');

        // 3. 기기 정보
        console.log('\n3. 기기 정보:');
        const info = display.getInfo();
        console.log(`   제조사: ${info.manufacturer}`);
        console.log(`   모델: ${info.model}`);
        console.log(`   펌웨어: ${info.firmwareVersion}`);
        console.log(`   셀 개수: ${info.cellCount}`);
        console.log(`   점 설정: ${info.dotConfig}점식`);

        // 4. 기능 테스트
        console.log('\n4. 기능 테스트:');

        console.log('   - 쓰기 테스트...');
        await display.write('WIA 테스트');
        console.log('     ✓ 쓰기 성공');

        console.log('   - 지우기 테스트...');
        await display.clear();
        console.log('     ✓ 지우기 성공');

        console.log('   - 셀 직접 제어 테스트...');
        const testCell = new Uint8Array([0xFF]);
        await display.setCells(0, testCell);
        console.log('     ✓ 셀 제어 성공');

        // 5. 성능 테스트
        console.log('\n5. 성능 테스트:');
        const start = performance.now();
        await display.write('성능 테스트');
        const latency = performance.now() - start;
        console.log(`   지연 시간: ${latency.toFixed(2)}ms`);

        if (latency < 100) {
            console.log('   ✓ 성능 양호');
        } else {
            console.log('   ⚠ 성능 주의 (목표: <100ms)');
        }

        // 6. 연결 해제
        await display.disconnect();
        console.log('\n6. 연결 해제 성공');

        console.log('\n━'.repeat(50));
        console.log('진단 완료: 모든 테스트 통과 ✓');

    } catch (error) {
        console.log(`\n❌ 오류 발생: ${error.message}`);
        console.log(`\n스택 트레이스:\n${error.stack}`);
    }
}

diagnose().catch(console.error);
