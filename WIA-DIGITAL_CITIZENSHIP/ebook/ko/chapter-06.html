2. WebSocket 프로토콜 아키텍처

디지털 시민권 시스템은 실시간 양방향 통신이 필수적입니다. 시민권 상태 변경, 권리 행사 내역, 의무 이행 알림 등이 즉시 반영되어야 합니다.

2.1 WebSocket 연결 수립

// TypeScript WebSocket Client
import WebSocket from 'ws';

interface CitizenshipConnection {
  citizenId: string;
  deviceId: string;
  authToken: string;
}

class CitizenshipWebSocketClient {
  private ws: WebSocket;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;

  constructor(private config: CitizenshipConnection) {
    this.connect();
  }

  private connect(): void {
    const wsUrl = `wss://citizenship.wia.live/v1/realtime`;

    this.ws = new WebSocket(wsUrl, {
      headers: {
        'Authorization': `Bearer ${this.config.authToken}`,
        'X-Citizen-ID': this.config.citizenId,
        'X-Device-ID': this.config.deviceId
      }
    });

    this.ws.on('open', () => {
      console.log('디지털 시민권 실시간 연결 수립');
      this.reconnectAttempts = 0;
      this.subscribe();
    });

    this.ws.on('message', (data: string) => {
      this.handleMessage(JSON.parse(data));
    });

    this.ws.on('close', () => {
      this.handleReconnect();
    });

    this.ws.on('error', (error) => {
      console.error('WebSocket 에러:', error);
    });
  }

  private subscribe(): void {
    const subscribeMessage = {
      type: 'subscribe',
      topics: [
        'citizenship.status',
        'citizenship.rights',
        'citizenship.duties',
        'verification.updates'
      ],
      citizenId: this.config.citizenId
    };

    this.ws.send(JSON.stringify(subscribeMessage));
  }

  private handleMessage(message: any): void {
    switch (message.type) {
      case 'citizenship.status.changed':
        this.handleStatusChange(message.data);
        break;
      case 'citizenship.rights.updated':
        this.handleRightsUpdate(message.data);
        break;
      case 'citizenship.duties.reminder':
        this.handleDutiesReminder(message.data);
        break;
      case 'verification.completed':
        this.handleVerification(message.data);
        break;
    }
  }

  private handleReconnect(): void {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      const delay = Math.min(1000 * Math.pow(2, this.reconnectAttempts), 30000);
      setTimeout(() => this.connect(), delay);
    }
  }
}

3. MQTT 기반 이벤트 아키텍처

MQTT(Message Queuing Telemetry Transport)는 경량 메시지 프로토콜로, IoT 디바이스와 모바일 애플리케이션 간 효율적인 통신을 지원합니다. 디지털 시민권 시스템에서는 다중 디바이스 동기화와 이벤트 브로드캐스팅에 활용됩니다.

3.2 MQTT 클라이언트 구현

import mqtt from 'mqtt';

interface MQTTConfig {
  brokerUrl: string;
  username: string;
  password: string;
  citizenId: string;
}

class CitizenshipMQTTClient {
  private client: mqtt.MqttClient;

  constructor(private config: MQTTConfig) {
    this.connect();
  }

  private connect(): void {
    this.client = mqtt.connect(this.config.brokerUrl, {
      username: this.config.username,
      password: this.config.password,
      clientId: `citizen-${this.config.citizenId}-${Date.now()}`,
      clean: true,
      reconnectPeriod: 1000,
      connectTimeout: 30000,
      will: {
        topic: `citizenship/${this.config.citizenId}/status`,
        payload: JSON.stringify({ status: 'offline', timestamp: Date.now() }),
        qos: 1,
        retain: true
      }
    });

    this.client.on('connect', () => {
      console.log('MQTT 연결 성공');
      this.subscribeToTopics();
      this.publishOnlineStatus();
    });

    this.client.on('message', (topic, message) => {
      this.handleMessage(topic, message);
    });
  }

  private subscribeToTopics(): void {
    const topics = [
      `citizenship/${this.config.citizenId}/status/#`,
      `citizenship/${this.config.citizenId}/rights/#`,
      `citizenship/${this.config.citizenId}/duties/#`,
      `citizenship/${this.config.citizenId}/verification/#`,
      'citizenship/global/#'
    ];

    topics.forEach(topic => {
      this.client.subscribe(topic, { qos: 1 });
    });
  }

  private publishOnlineStatus(): void {
    const topic = `citizenship/${this.config.citizenId}/status/active`;
    const payload = {
      status: 'online',
      timestamp: Date.now(),
      device: navigator.userAgent
    };

    this.client.publish(topic, JSON.stringify(payload), { qos: 1, retain: true });
  }

  public publishRightExercise(rightType: string, details: any): void {
    const topic = `citizenship/${this.config.citizenId}/rights/${rightType}`;
    const payload = {
      action: 'exercise',
      rightType,
      details,
      timestamp: Date.now()
    };

    this.client.publish(topic, JSON.stringify(payload), { qos: 2 });
  }
}

4.2 엔드투엔드 암호화 구현

import { createCipheriv, createDecipheriv, randomBytes } from 'crypto';

class CitizenshipEncryption {
  private algorithm = 'aes-256-gcm';

  // 시민권 데이터 암호화
  encrypt(data: any, citizenKey: Buffer): EncryptedData {
    const iv = randomBytes(16);
    const cipher = createCipheriv(this.algorithm, citizenKey, iv);

    const plaintext = JSON.stringify(data);
    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');

    const authTag = cipher.getAuthTag();

    return {
      encrypted,
      iv: iv.toString('hex'),
      authTag: authTag.toString('hex'),
      timestamp: Date.now()
    };
  }

  // 시민권 데이터 복호화
  decrypt(encryptedData: EncryptedData, citizenKey: Buffer): any {
    const iv = Buffer.from(encryptedData.iv, 'hex');
    const authTag = Buffer.from(encryptedData.authTag, 'hex');

    const decipher = createDecipheriv(this.algorithm, citizenKey, iv);
    decipher.setAuthTag(authTag);

    let decrypted = decipher.update(encryptedData.encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');

    return JSON.parse(decrypted);
  }

  // Pass인증서 연동 암호화
  encryptForPassAuth(citizenshipData: any): string {
    // Pass인증서 표준 암호화 형식
    const payload = {
      type: 'WIA_CITIZENSHIP',
      version: '1.0',
      data: citizenshipData,
      timestamp: Date.now()
    };

    // Pass 공개키로 암호화 (실제로는 Pass SDK 사용)
    return Buffer.from(JSON.stringify(payload)).toString('base64');
  }
}

6.1 정부24 WebSocket 연동

interface Gov24Config {
  apiKey: string;
  citizenRRN: string;  // 주민등록번호 (암호화)
  serviceCode: string;
}

class Gov24RealtimeSync {
  private ws: WebSocket;

  constructor(private config: Gov24Config) {
    this.connectToGov24();
  }

  private connectToGov24(): void {
    const wsUrl = 'wss://api.gov.kr/v2/realtime';

    this.ws = new WebSocket(wsUrl, {
      headers: {
        'X-API-Key': this.config.apiKey,
        'X-Service-Code': this.config.serviceCode
      }
    });

    this.ws.on('open', () => {
      this.requestCitizenshipSync();
    });

    this.ws.on('message', (data) => {
      const message = JSON.parse(data.toString());
      this.handleGov24Message(message);
    });
  }

  private requestCitizenshipSync(): void {
    const request = {
      type: 'CITIZENSHIP_SYNC',
      citizenId: this.config.citizenRRN,
      requestedData: [
        'RESIDENT_REGISTRATION',  // 주민등록
        'FAMILY_RELATIONS',       // 가족관계
        'LAND_REGISTRY',          // 부동산
        'VEHICLE_REGISTRATION',   // 차량등록
        'BUSINESS_LICENSE',       // 사업자등록
        'TAX_RECORDS'            // 세금납부
      ],
      timestamp: Date.now()
    };

    this.ws.send(JSON.stringify(request));
  }

  private handleGov24Message(message: any): void {
    switch (message.type) {
      case 'CITIZENSHIP_DATA':
        this.updateCitizenshipInfo(message.data);
        break;
      case 'STATUS_CHANGE':
        this.handleStatusChange(message.data);
        break;
      case 'RIGHTS_UPDATE':
        this.handleRightsUpdate(message.data);
        break;
    }
  }

  // WIA 시민권 데이터로 변환
  private updateCitizenshipInfo(gov24Data: any): void {
    const wiaData = {
      citizenId: this.hashRRN(this.config.citizenRRN),
      personalInfo: {
        name: gov24Data.name,
        address: gov24Data.address,
        registrationDate: gov24Data.regDate
      },
      rights: this.extractRights(gov24Data),
      duties: this.extractDuties(gov24Data),
      verificationLevel: 'GOVERNMENT_VERIFIED',
      lastSync: Date.now()
    };

    // WIA 시민권 시스템에 업데이트
    this.syncToWIA(wiaData);
  }
}

8.1 Circuit Breaker 패턴

class CitizenshipCircuitBreaker {
  private failureCount = 0;
  private lastFailureTime = 0;
  private state: 'CLOSED' | 'OPEN' | 'HALF_OPEN' = 'CLOSED';

  private readonly failureThreshold = 5;
  private readonly timeout = 60000; // 1분
  private readonly halfOpenRequests = 3;

  async execute(operation: () => Promise): Promise {
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime > this.timeout) {
        this.state = 'HALF_OPEN';
      } else {
        throw new Error('Circuit breaker is OPEN');
      }
    }

    try {
      const result = await operation();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  private onSuccess(): void {
    this.failureCount = 0;
    this.state = 'CLOSED';
  }

  private onFailure(): void {
    this.failureCount++;
    this.lastFailureTime = Date.now();

    if (this.failureCount >= this.failureThreshold) {
      this.state = 'OPEN';
      console.error('Circuit breaker opened due to failures');
    }
  }
}
