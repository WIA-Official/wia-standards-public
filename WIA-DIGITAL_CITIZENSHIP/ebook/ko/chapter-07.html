4단계에서는 WIA-DIGITAL_CITIZENSHIP 시스템을 실제 프로덕션 환경에 배포하고 운영하는 전체 과정을 다룹니다. 클라우드 인프라 구축, 컨테이너 오케스트레이션, CI/CD 파이프라인, 모니터링 및 로깅, 그리고 한국 클라우드 서비스와의 통합을 포함합니다.

  • AWS, Azure, GCP에 디지털 시민권 시스템 배포

FROM node:20-alpine

COPY --from=builder --chown=citizenship:nodejs /app/package.json ./

USER citizenship

EXPOSE 3000

HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD node -e "require('http').get('http://localhost:3000/health', (r) => process.exit(r.statusCode === 200 ? 0 : 1))"

4. CI/CD 파이프라인

GitHub Actions를 사용하여 자동화된 빌드, 테스트, 배포 파이프라인을 구축합니다.

4.1 GitHub Actions 워크플로우

# .github/workflows/deploy.yml
name: Deploy WIA-CITIZENSHIP

on:
  push:
    branches: [main, staging]
  pull_request:
    branches: [main]

env:
  AWS_REGION: ap-northeast-2
  ECR_REPOSITORY: wia-citizenship-api
  EKS_CLUSTER_NAME: wia-citizenship-cluster

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Run tests
        run: npm run test:ci

      - name: Run security audit
        run: npm audit --audit-level=moderate

  build:
    needs: test
    runs-on: ubuntu-latest
    if: github.event_name == 'push'
    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Login to Amazon ECR
        id: login-ecr
        uses: aws-actions/amazon-ecr-login@v2

      - name: Build and push Docker image
        env:
          ECR_REGISTRY: ${{ steps.login-ecr.outputs.registry }}
          IMAGE_TAG: ${{ github.sha }}
        run: |
          docker build -t $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG .
          docker push $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG
          docker tag $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG $ECR_REGISTRY/$ECR_REPOSITORY:latest
          docker push $ECR_REGISTRY/$ECR_REPOSITORY:latest

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Update kubeconfig
        run: |
          aws eks update-kubeconfig --region $AWS_REGION --name $EKS_CLUSTER_NAME

      - name: Deploy to Kubernetes
        run: |
          kubectl set image deployment/citizenship-api \
            citizenship-api=${{ steps.login-ecr.outputs.registry }}/$ECR_REPOSITORY:${{ github.sha }} \
            -n wia-citizenship
          kubectl rollout status deployment/citizenship-api -n wia-citizenship

5.2 네이버 클라우드 배포 스크립트

# scripts/deploy-ncloud.sh
#!/bin/bash

set -e

echo "🚀 네이버 클라우드 배포 시작..."

# 환경 변수 설정
export NCLOUD_ACCESS_KEY="${NCLOUD_ACCESS_KEY}"
export NCLOUD_SECRET_KEY="${NCLOUD_SECRET_KEY}"
export NCLOUD_REGION="KR"

# 서버 인스턴스 생성
echo "📦 서버 인스턴스 생성 중..."
SERVER_INSTANCE_NO=$(ncloud-cli server createServerInstances \
  --serverImageProductCode SW.VSVR.OS.LNX64.UBNTU.SVR2004.B050 \
  --serverProductCode SVR.VSVR.STAND.C002.M008.NET.SSD.B050.G002 \
  --serverName "wia-citizenship-api" \
  --loginKeyName "wia-keypair" \
  --query 'serverInstanceList[0].serverInstanceNo' \
  --output text)

echo "✅ 서버 생성 완료: ${SERVER_INSTANCE_NO}"

# 공인 IP 할당
echo "🌐 공인 IP 할당 중..."
PUBLIC_IP=$(ncloud-cli server associatePublicIpWithServerInstance \
  --serverInstanceNo ${SERVER_INSTANCE_NO} \
  --query 'publicIp' \
  --output text)

echo "✅ 공인 IP 할당 완료: ${PUBLIC_IP}"

# Docker 이미지 배포
echo "🐳 Docker 이미지 배포 중..."
ssh -i ~/.ssh/wia-keypair.pem root@${PUBLIC_IP} << 'EOF'
  # Docker 설치
  curl -fsSL https://get.docker.com | sh
  systemctl start docker
  systemctl enable docker

  # 이미지 Pull 및 실행
  docker pull wia/citizenship-api:latest
  docker run -d \
    --name citizenship-api \
    --restart always \
    -p 80:3000 \
    -e NODE_ENV=production \
    -e NAVER_CLIENT_ID=${NAVER_CLIENT_ID} \
    -e NAVER_CLIENT_SECRET=${NAVER_CLIENT_SECRET} \
    wia/citizenship-api:latest
EOF

echo "✅ 배포 완료!"
echo "📍 서비스 URL: http://${PUBLIC_IP}"

8.1 DR 시나리오 및 대응

# DR Failover Script
#!/bin/bash

# 재해 발생 시 자동 Failover
REGION_PRIMARY="ap-northeast-2"    # 서울
REGION_SECONDARY="ap-northeast-1"  # 도쿄

echo "🚨 재해 감지: Primary Region 장애"
echo "🔄 Secondary Region으로 Failover 시작..."

# Route53 DNS 전환
aws route53 change-resource-record-sets \
  --hosted-zone-id Z1234567890ABC \
  --change-batch '{
    "Changes": [{
      "Action": "UPSERT",
      "ResourceRecordSet": {
        "Name": "api.citizenship.wia.live",
        "Type": "A",
        "SetIdentifier": "Secondary",
        "Region": "'${REGION_SECONDARY}'",
        "AliasTarget": {
          "HostedZoneId": "Z5678901234DEF",
          "DNSName": "secondary-lb.amazonaws.com",
          "EvaluateTargetHealth": true
        }
      }
    }]
  }'

# RDS 복제본 승격
aws rds promote-read-replica \
  --db-instance-identifier citizenship-db-replica-tokyo \
  --region ${REGION_SECONDARY}

# 트래픽 전환 확인
echo "✅ Failover 완료"
echo "📊 새 엔드포인트: ${REGION_SECONDARY}"
