3장: 센서와 페이로드

센서와 페이로드 시스템은 농업용 드론의 감지 능력과 작업 수행 능력을 정의합니다. 이 장에서는 농업 응용 분야에서 사용되는 다양한 센서 기술, 페이로드 통합 고려사항, 캘리브레이션 요구사항을 다룹니다. RGB 카메라부터 멀티스펙트럴, 열화상, LiDAR, 하이퍼스펙트럴 센서까지 각 센서 유형의 특성과 농업 응용을 이해하는 것이 효과적인 데이터 수집에 필수적입니다.

3.1 RGB 카메라 시스템

// RGB 카메라 사양
interface RGBCameraSpecs {
  // 센서 사양
  sensor: {
    type: 'CMOS' | 'CCD';
    size: string;                // 예: '1인치', 'APS-C', '풀프레임'
    megapixels: number;
    pixelSize: number;           // 마이크로미터
  };

  // 렌즈 사양
  lens: {
    focalLength: number;         // mm
    aperture: string;            // 예: 'f/2.8'
    fieldOfView: number;         // 도
    distortion: string;          // 예: '< 2%'
  };

  // 이미지 성능
  imaging: {
    gsdAt100m: number;           // cm/픽셀
    dynamicRange: number;        // 스톱
    iso: { min: number; max: number };
    shutterSpeed: { min: string; max: string };
  };

  // 출력
  output: {
    rawFormat: string;           // 예: 'DNG', 'RAW'
    jpegQuality: string;
    videoCapability: boolean;
    geotagging: boolean;
  };
}

// 예시: 농업용 매핑 카메라
const mappingCamera: RGBCameraSpecs = {
  sensor: {

  },

    focalLength: 8.8,
    aperture: 'f/2.8-f/11',
    fieldOfView: 84,
    distortion: '< 1.5%'
  },
  imaging: {
    gsdAt100m: 2.74,
    dynamicRange: 12.8,
    iso: { min: 100, max: 12800 },
    shutterSpeed: { min: '1/8000', max: '8초' }
  },
  output: {
    rawFormat: 'DNG',
    jpegQuality: '최대',
    videoCapability: true,
    geotagging: true
  }
};

// 멀티스펙트럴 센서 인터페이스
interface MultispectralSensor {
  // 센서 구성
  configuration: {
    bands: number;               // 일반적으로 5-10
    sensorType: 'discrete' | 'pushbroom' | 'snapshot';
    resolution: number;          // 밴드당 메가픽셀
    globalShutter: boolean;
  };

  // 스펙트럴 특성
  spectralCharacteristics: {
    bandDefinitions: {
      name: string;
      centerWavelength: number;  // nm
      bandwidth: number;         // nm FWHM
    }[];
    spectralResponse: string;    // 응답 곡선 파일
  };

  // 캘리브레이션
  calibration: {
    radiometricCalibration: boolean;
    reflectancePanel: boolean;
    dls: boolean;                // 하향 광 센서
    imuIntegration: boolean;     // 기울기 보정용
  };

  // 데이터 출력
  dataOutput: {
    rawFormat: 'TIFF' | 'RAW';
    bitDepth: 12 | 14 | 16;
    metadataFormat: 'EXIF' | 'XMP';
    captureRate: number;         // 초당 프레임
  };
}

3.3 열화상 센서

열화상 응용 분야

  • 수분 스트레스 탐지: 잎 온도와 증발산량 상관관계
  • 관개 관리: 물 분배 패턴 시각화
  • 질병 초기 탐지: 감염된 식물의 열적 징후
  • 서리 피해 평가: 저온 피해 조직 식별
  • 가축 모니터링: 동물 건강 및 위치
// 열화상 카메라 사양
interface ThermalCameraSpecs {
  // 검출기
  detector: {
    type: 'VOx 마이크로볼로미터' | 'InSb' | 'MCT';
    resolution: { width: number; height: number };
    pixelPitch: number;          // 마이크로미터
    netd: number;                // mK (노이즈등가온도차)
  };

  // 열적 성능
  thermal: {
    spectralRange: { min: number; max: number };  // 마이크로미터
    temperatureRange: { min: number; max: number };  // 섭씨
    accuracy: number;            // 도
    frameRate: number;           // Hz
  };

  // 광학
  optics: {
    focalLength: number;         // mm
    fov: { h: number; v: number };  // 도
    fNumber: number;
  };

  // 출력
  output: {
    format: 'RJPEG' | 'TIFF' | 'SEQ';
    radiometric: boolean;        // 온도 데이터 포함
    streaming: boolean;
  };
}

// 예시: 농업용 열화상

  // 스캔 성능
  scanning: {
    pointsPerSecond: number;
    channels: number;
    fov: { horizontal: number; vertical: number };
    angularResolution: number;   // 도
  };

  // 거리 측정

    maxRange: number;            // m
    accuracy: number;            // cm
    precision: number;           // cm
    multipleReturns: number;     // 일반적으로 최대 5
  };

  // 출력
  output: {
    format: 'LAS' | 'LAZ' | 'PCD';
    intensity: boolean;
    classification: boolean;
  };
}

3.6 페이로드 통합

// 페이로드 통합 요구사항
interface PayloadIntegration {
  // 물리적 통합
  physical: {
    mountingType: 'gimbal' | 'fixed' | 'interchangeable';
    vibrationIsolation: boolean;
    quickRelease: boolean;
    weatherSealing: string;      // IP 등급
  };

  // 전기적 통합
  electrical: {
    powerRequirement: number;    // 와트
    voltage: number;             // V
    interface: 'serial' | 'ethernet' | 'USB' | 'CAN';
    triggerMethod: 'time' | 'distance' | 'PWM' | 'GPIO';
  };

  // 데이터 통합
  data: {
    storageLocation: 'onboard' | 'aircraft' | 'ground';
    realTimeLink: boolean;
    bandwidth: number;           // Mbps
    timestamping: 'GPS' | 'FC' | 'sensor';
    geotagging: 'embedded' | 'separate' | 'PPK';
  };

  // 짐벌 요구사항
  gimbal?: {
    axes: 2 | 3;
    range: { pitch: number[]; roll: number[]; yaw: number[] };
    stabilization: number;       // 도
    controlInterface: string;
  };
}

3.7 센서 캘리브레이션

캘리브레이션 프로토콜

  • 기하 캘리브레이션: 렌즈 왜곡, 초점 거리, 주점 보정
  • 방사 캘리브레이션: 센서 응답을 물리적 단위로 변환
  • 스펙트럴 캘리브레이션: 파장 정확도 검증 및 보정
  • 현장 캘리브레이션: 반사 패널을 사용한 실시간 보정
// 캘리브레이션 워크플로우
interface CalibrationWorkflow {
  // 비행 전 캘리브레이션
  preflight: {
    reflectancePanel: {
      material: 'Spectralon' | 'Calibrite';
      reflectance: number;       // 일반적으로 18% 또는 50%
      procedure: '센서 활성화 전 이미지 촬영';
    };
    dlsCheck: {
      required: boolean;
      alignment: '수평 장착 확인';
      validation: '클리어 스카이 레퍼런스';
    };
  };

  // 비행 중 보정
  inflight: {
    irradianceNormalization: boolean;
    attitudeCorrection: boolean;
    vignetteCorrection: boolean;
  };

  // 비행 후 처리
  postflight: {
    empiricalLineCorrection: boolean;
    groundTruthValidation: boolean;
    crossCalibration: boolean;
  };
}
