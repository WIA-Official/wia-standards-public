5.2 식생 지수

지수 공식 응용
NDVI (NIR - Red) / (NIR + Red) 전반적인 식생 건강
NDRE (NIR - RedEdge) / (NIR + RedEdge) 엽록소 함량, 질소 상태
GNDVI (NIR - Green) / (NIR + Green) 엽록소 농도
SAVI ((NIR - Red) / (NIR + Red + L)) * (1 + L) 토양 영향 보정
CWSI (Tc - Ta) / (Twet - Tdry) 작물 수분 스트레스 
// 식생 지수 계산 인터페이스
interface VegetationIndex {
  name: string;
  formula: string;
  bands: string[];
  range: { min: number; max: number };
  interpretation: {
    low: string;
    medium: string;
    high: string;
  };
}

// 식생 지수 계산기
class IndexCalculator {
  // NDVI 계산
  calculateNDVI(nir: number, red: number): number {
    return (nir - red) / (nir + red);
  }

  // NDRE 계산
  calculateNDRE(nir: number, redEdge: number): number {
    return (nir - redEdge) / (nir + redEdge);
  }

  // GNDVI 계산
  calculateGNDVI(nir: number, green: number): number {
    return (nir - green) / (nir + green);
  }

  // SAVI 계산 (L = 0.5가 일반적)
  calculateSAVI(nir: number, red: number, L: number = 0.5): number {
    return ((nir - red) / (nir + red + L)) * (1 + L);
  }

  // 래스터 처리
  processRaster(
    multispectralImage: MultispectralImage,
    indexType: string
  ): GeoTIFF {

  }
}

5.3 AI 기반 분석

머신러닝 응용

  • 작물 분류: CNN을 사용한 작물 종류 및 잡초 식별
  • 병해 탐지: 딥러닝으로 질병 징후 조기 탐지
  • 수확량 예측: 회귀 모델을 통한 수확량 추정
  • 이상 탐지: 비정상 패턴 자동 식별
  • 객체 검출: 개별 식물, 해충, 과일 카운팅
// AI 분석 파이프라인
interface AIAnalyticsPipeline {
  // 이미지 분류
  classification: {
    model: 'ResNet' | 'EfficientNet' | 'custom';
    classes: string[];
    inputSize: { width: number; height: number };

      normalization: 'imagenet' | 'custom';
      augmentation: boolean;
    };
    output: {
      classLabel: string;
      confidence: number;
      probabilities: Record<string, number>;
    };

  // 객체 검출
  objectDetection: {
    model: 'YOLO' | 'FasterRCNN' | 'SSD';
    classes: string[];
    confidenceThreshold: number;
    nmsThreshold: number;
    output: {
      boundingBoxes: BoundingBox[];
      counts: Record<string, number>;

  // 시맨틱 세그멘테이션

    model: 'UNet' | 'DeepLab' | 'SegFormer';

    output: {
      segmentationMask: GeoTIFF;
      areaStatistics: Record<string, number>;
    };
  };

  // 이상 탐지

    method: 'autoencoder' | 'isolation_forest' | 'statistical';
    threshold: number;
    output: {
      anomalyMap: GeoTIFF;
      hotspots: GeoJSON.Point[];
      severity: Record<string, number>;
    };
  };
}

// 시계열 분석 인터페이스

5.7 클라우드 vs 엣지 처리

측면 클라우드 처리 엣지 처리
처리 속도 높은 연산 능력, 병렬 처리 즉각적 결과, 지연 없음
연결성 인터넷 필요 오프라인 가능
비용 사용량 기반 초기 하드웨어 투자
적합한 작업 대규모 정사영상, 딥러닝 실시간 결정, 즉각 살포 
// 하이브리드 처리 아키텍처
interface HybridProcessing {
  // 엣지 처리 (온보드/현장)

    capabilities: [
      '실시간 NDVI 계산',
      '즉각적 이상 경고',

      '기본 품질 검사'
    ];
    hardware: 'Jetson' | 'Intel NCS' | 'Google Coral';

  };

  // 클라우드 처리

    capabilities: [
      '고품질 정사영상 생성',

  };

  // 동기화
  sync: {
    previewOnEdge: true;
    fullProcessingOnCloud: true;
    resultsSyncBack: true;
  };
}
