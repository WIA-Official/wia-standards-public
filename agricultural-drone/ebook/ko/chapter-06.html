6장: 정밀 살포 시스템

정밀 살포는 농약, 비료, 씨앗 등의 농자재를 필요한 곳에 필요한 만큼만 정확하게 살포하는 것을 목표로 합니다. 이 장에서는 드론 살포 시스템 하드웨어, 가변율 기술(VRT), 드리프트 관리, 캘리브레이션 절차, 살포 기록 관리를 다룹니다. 정밀 살포는 투입물 비용을 절감하고 환경 영향을 최소화하면서 작물 보호 효과를 극대화합니다.

6.1 살포 시스템 구성

// 살포 시스템 사양
interface SpraySystem {
  // 탱크 시스템
  tank: {
    capacity: number;            // 리터
    material: 'HDPE' | 'fiberglass' | 'stainless';
    agitation: 'mechanical' | 'hydraulic' | 'none';
    fillerOpening: number;       // mm
    drainPort: boolean;
    levelSensor: 'ultrasonic' | 'capacitive' | 'float';
  };

  // 펌프 시스템
  pump: {
    type: 'diaphragm' | 'centrifugal' | 'peristaltic' | 'piston';
    flowRate: number;            // L/min
    pressure: number;            // bar
    materialCompatibility: string[];
    priming: 'self' | 'manual';
  };

  // 노즐 시스템
  nozzles: {
    count: number;
    type: 'flat_fan' | 'hollow_cone' | 'full_cone' | 'rotary';
    spacing: number;             // mm
    dropletSize: 'fine' | 'medium' | 'coarse' | 'very_coarse';
    flowRate: number;            // L/min @ pressure
    sprayAngle: number;          // 도
    antiDrip: boolean;
  };

  // 붐 시스템
  boom: {
    width: number;               // m
    height: number;              // m (지면 위)
    adjustable: boolean;
    foldable: boolean;
    dampening: boolean;
  };

  // 제어 시스템
  control: {
    flowControl: 'PWM' | 'servo' | 'solenoid';
    pressureRegulation: boolean;
    sectionControl: number;      // 독립 섹션 수
    vrtCapability: boolean;
    gpsIntegration: boolean;
  };
}

6.2 노즐 기술

노즐 유형 패턴 액적 크기 적합한 용도
평판(Flat Fan) 삼각형 미세-중간 일반 농약 살포
공동원추(Hollow Cone) 링 미세 살균제, 침투성 필요
충실원추(Full Cone) 원형 중간-조대 비료, 넓은 커버리지
회전식(Rotary) 조절 가능 균일 정밀 살포, ULV
에어 인젝션 평판 조대 드리프트 저감

6.3 가변율 기술(VRT)

VRT 구현 방법

  • 맵 기반 VRT: 사전 생성된 처방 맵 사용
  • 센서 기반 VRT: 실시간 센서 데이터로 살포율 조정
  • 하이브리드 VRT: 맵 + 실시간 센서 조합
  • 구역 제어: 노즐 그룹 개별 제어
// VRT 시스템 인터페이스
interface VRTSystem {
  // 처방 맵 입력
  prescriptionMap: {
    format: 'shapefile' | 'isoxml' | 'geojson';
    rateField: string;
    unitType: 'L/ha' | 'kg/ha';
    zoneCount: number;
  };

  // 레이트 제어
  rateControl: {
    minRate: number;
    maxRate: number;
    rateResolution: number;      // 최소 조정 단위
    responseTime: number;        // ms
    lookAhead: number;           // m
  };

  // 구역 제어
  sectionControl: {
    sections: number;
    overlapPrevention: boolean;
    boundaryManagement: boolean;
    headlandManagement: boolean;
  };

  // 실시간 조정
  realtimeAdjustment: {
    speedCompensation: boolean;
    heightCompensation: boolean;
    windCompensation: boolean;
    sensorFeedback?: string[];
  };
}

// VRT 살포 계산
class VRTCalculator {
  // 목표 레이트 계산
  calculateTargetRate(
    position: GeoJSON.Point,
    prescriptionMap: PrescriptionMap,
    speed: number
  ): number {
    const zone = prescriptionMap.getZoneAt(position);
    const baseRate = zone.rate;
    // 속도 보정 적용
    return baseRate * this.speedCompensationFactor(speed);
  }

  // 펌프 출력 계산
  calculatePumpOutput(
    targetRate: number,      // L/ha
    speed: number,           // m/s
    boomWidth: number        // m
  ): number {
    // L/min = (L/ha * m/s * m) / 600
    return (targetRate * speed * boomWidth) / 600;
  }

  // PWM 듀티 사이클 계산
  calculatePWM(
    targetFlow: number,
    maxFlow: number
  ): number {
    return Math.min(100, (targetFlow / maxFlow) * 100);
  }
}

    temperatureGradient: boolean;
    realTimeAlerts: boolean;

// 드리프트 리스크 계산
function calculateDriftRisk(
  dropletSize: number,    // VMD (마이크로미터)
  windSpeed: number,      // m/s
  releaseHeight: number,  // m
  temperature: number,    // 섭씨
  humidity: number        // %
): DriftRiskLevel {
  let riskScore = 0;

  // 액적 크기 요인
  if (dropletSize < 150) riskScore += 3;
  else if (dropletSize < 250) riskScore += 2;
  else if (dropletSize < 350) riskScore += 1;

  // 풍속 요인
  if (windSpeed > 4) riskScore += 3;
  else if (windSpeed > 2) riskScore += 1;

  // 고도 요인
  if (releaseHeight > 3) riskScore += 2;
  else if (releaseHeight > 2) riskScore += 1;

  // 위험 수준 반환
  if (riskScore >= 6) return 'HIGH';
  if (riskScore >= 3) return 'MEDIUM';
  return 'LOW';
}

6.6 살포 기록 관리

// 살포 기록 인터페이스
interface ApplicationRecord {
  // 기본 정보
  metadata: {
    recordId: string;
    missionId: string;
    date: ISO8601DateTime;
    operator: string;
    pilot: string;
  };

  // 필드 정보
  field: {
    name: string;
    location: GeoJSON.Polygon;
    area: number;                // ha
    crop: string;
    growthStage: string;
  };

  // 제품 정보
  product: {
    name: string;
    activeIngredient: string;
    concentration: number;
    targetRate: number;          // L/ha 또는 kg/ha
    mixRatio?: string;
    totalApplied: number;
    lotNumber?: string;
  };

  // 살포 조건
  conditions: {
    weather: {
      temperature: number;       // 섭씨
      humidity: number;          // %
      windSpeed: number;         // m/s
      windDirection: string;
    };
    timing: {
      startTime: ISO8601DateTime;
      endTime: ISO8601DateTime;
      duration: number;          // 분
    };
    equipment: {
      droneModel: string;
      nozzleType: string;
      pressure: number;          // bar
      flightHeight: number;      // m
      speed: number;             // m/s
    };
  };

  // As-Applied 데이터
  asApplied: {
    coverageMap: GeoJSON.FeatureCollection;
    actualRate: number;
    efficiency: number;          // %
    missedAreas?: GeoJSON.MultiPolygon;
    overlapAreas?: GeoJSON.MultiPolygon;
  };

  // 규정 준수
  compliance: {
    bufferZonesRespected: boolean;
    labelRequirementsMet: boolean;
    reEntryInterval: number;     // 시간
    preHarvestInterval: number;  // 일
  };
}
