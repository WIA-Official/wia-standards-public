Chapter 4: Flight Operations

Effective flight operations are critical for safe, efficient, and legally compliant agricultural drone missions. This chapter covers mission planning, autonomous navigation, safety protocols, and operational best practices. Understanding these procedures ensures consistent data quality, regulatory compliance, and safe operations in agricultural environments where obstacles, weather, and field conditions present unique challenges.

4.1 Mission Planning Fundamentals

// Mission Planning Data Structure
interface MissionPlan {
  // Mission Identity
  id: string;
  name: string;
  type: 'survey' | 'spray' | 'mapping' | 'inspection';
  createdAt: ISO8601DateTime;
  scheduledFor?: ISO8601DateTime;

  // Field Information
  field: {
    id: string;
    boundary: GeoJSON.Polygon;
    area: number;                // hectares
    obstacles: Obstacle[];
    noFlyZones?: GeoJSON.Polygon[];
  };

  // Flight Parameters
  flightParams: {
    altitude: number;            // meters AGL
    speed: number;               // m/s
    pattern: 'grid' | 'crosshatch' | 'contour' | 'perimeter';
    heading: number | 'auto';    // degrees or sun angle optimization
    overlap: {
      front: number;             // percent
      side: number;              // percent
    };
    turnaroundMode: 'bank' | 'stop_and_turn' | 'curved';
  };

  // Sensor Configuration
  sensorConfig: {
    cameraModel: string;
    triggerMode: 'distance' | 'time' | 'overlap';
    triggerInterval: number;
    exposureSettings: 'auto' | ExposureSettings;
  };

  // Safety Parameters
  safety: {
    minAltitude: number;
    maxAltitude: number;
    geofence: GeoJSON.Polygon;
    rthAltitude: number;
    lowBatteryAction: 'land' | 'rth' | 'continue';
    signalLossAction: 'hover' | 'land' | 'rth' | 'continue';
  };

  // Computed Values
  computed: {
    estimatedDuration: number;   // minutes
    estimatedImages: number;
    batteryFlights: number;
    gsd: number;                 // cm/pixel
  };
}

4.2 Flight Patterns

Pattern Description Best For Considerations
Grid (Lawn Mower) Parallel lines, single direction Flat terrain, basic mapping Efficient, may miss 3D features
Crosshatch (Double Grid) Two perpendicular grids 3D modeling, tall crops Double flight time, better coverage
Contour Follow terrain elevation Hilly terrain Consistent GSD, complex planning
Orbital Circles around point of interest Single tree, structure inspection Limited area coverage
Perimeter Follow field boundary Boundary mapping, fencing Quick overview, no interior coverage

4.3 Autonomous Navigation

// Autonomous Navigation System
interface AutonomousNavigation {
  // Waypoint Navigation
  waypoints: {
    position: { lat: number; lng: number; alt: number };
    speed: number;
    action?: WaypointAction;
    radius: number;              // Acceptance radius
  }[];

  // Path Planning
  pathPlanning: {
    algorithm: 'shortest_path' | 'coverage' | 'terrain_following';
    obstacleAvoidance: boolean;
    terrainFollowing: {
      enabled: boolean;
      source: 'dem' | 'lidar' | 'barometer';
      minClearance: number;      // meters
    };
  };

  // Positioning
  positioning: {
    gnss: {
      mode: 'standalone' | 'dgps' | 'rtk' | 'ppk';
      accuracy: number;          // meters
      constellations: string[];
    };
    visionPositioning: boolean;
    terrainMatching: boolean;
  };

  // State Machine
  flightState: 'idle' | 'preflight' | 'takeoff' | 'mission' |
               'rth' | 'landing' | 'emergency';

  // Mission Execution
  missionExecution: {
    currentWaypoint: number;
    progress: number;            // percent
    distanceTraveled: number;    // meters
    estimatedRemaining: number;  // minutes
  };
}

// Waypoint Actions
type WaypointAction =
  | { type: 'capture'; count: number }
  | { type: 'hover'; duration: number }
  | { type: 'speed_change'; speed: number }
  | { type: 'gimbal'; pitch: number; yaw: number }
  | { type: 'spray'; rate: number }
  | { type: 'custom'; command: string };

4.4 Pre-Flight Procedures

Pre-Flight Checklist

  • Regulatory: Airspace clearance, permits, notifications
  • Weather: Wind speed, precipitation, visibility, temperature
  • Aircraft: Structural integrity, propellers, batteries charged
  • Sensors: Calibration, memory cards, lens cleaning
  • Software: Firmware current, mission uploaded, geofence set
  • Site: Obstacles identified, takeoff/landing zone clear
  • Emergency: RTH point set, emergency procedures reviewed
// Pre-Flight Check API
interface PreFlightCheck {
  // System Checks
  aircraft: {
    batteryLevel: { value: number; status: 'ok' | 'low' | 'critical' };
    batteryHealth: { value: number; status: 'ok' | 'degraded' };
    motorsCheck: { status: 'ok' | 'warning' | 'fail' }[];
    propellerCheck: boolean;
    compassCalibration: { status: 'ok' | 'needed' };
    imuCalibration: { status: 'ok' | 'needed' };
    gnssLock: { satellites: number; accuracy: number };
    storage: { available: number; required: number };
  };

  // Environmental Checks
  environment: {
    windSpeed: { value: number; limit: number; status: 'ok' | 'warning' | 'no_fly' };
    temperature: { value: number; status: 'ok' | 'warning' };
    precipitation: { detected: boolean };
    visibility: { km: number; status: 'ok' | 'reduced' };
    magneticInterference: { status: 'ok' | 'warning' | 'high' };
  };

  // Airspace Checks
  airspace: {
    classification: string;
    restrictions: string[];
    authorizationRequired: boolean;
    authorizationStatus?: 'approved' | 'pending' | 'denied';
    notams: NOTAM[];
  };

  // Mission Validation
  mission: {
    loaded: boolean;
    validated: boolean;
    warnings: string[];
    estimatedDuration: number;
    batteryFlights: number;
  };

  // Overall Status
  overallStatus: 'go' | 'caution' | 'no_go';
  blockers: string[];
}

4.6 Safety Protocols

// Safety Protocol Configuration
interface SafetyProtocols {
  // Geofencing
  geofencing: {
    softGeofence: {
      boundary: GeoJSON.Polygon;
      action: 'alert' | 'brake';
    };
    hardGeofence: {
      boundary: GeoJSON.Polygon;
      action: 'brake' | 'rth' | 'land';
    };
    maxAltitude: number;
    maxDistance: number;
  };

  // Return to Home
  returnToHome: {
    altitude: number;
    speed: number;
    triggers: [
      'low_battery',
      'signal_loss',
      'geofence_breach',
      'pilot_command'
    ];
    lowBatteryThreshold: number;
    signalLossDelay: number;     // seconds before RTH
  };

  // Emergency Procedures
  emergency: {
    motorFailure: {
      detection: 'rpm_monitoring';
      response: 'controlled_descent' | 'emergency_land';
    };
    gpsLoss: {
      detection: 'satellite_count';
      response: 'hold_position' | 'atti_mode' | 'land';
    };
    lowBattery: {
      warningLevel: number;
      criticalLevel: number;
      response: 'rth' | 'land_immediately';
    };
  };

  // Obstacle Avoidance
  obstacleAvoidance: {
    enabled: boolean;
    sensors: ('forward' | 'backward' | 'lateral' | 'upward' | 'downward')[];
    minDistance: number;
    response: 'stop' | 'avoid' | 'climb';
  };
}

4.7 Battery Management

Battery Level Status Recommended Action
100% - 50% Normal Continue mission
50% - 30% Planning Plan RTH, complete current segment
30% - 20% Warning Initiate RTH
20% - 10% Critical Priority RTH, altitude reduction
Below 10% Emergency Immediate landing

4.8 Post-Flight Procedures

Post-Flight Checklist

  • Data Verification: Confirm images captured, check quality samples
  • Data Download: Transfer images, flight logs, telemetry
  • Battery Care: Cool down, storage charge if not flying again
  • Aircraft Inspection: Check for damage, clean sensors and propellers
  • Flight Log: Record flight details, issues, observations
  • Maintenance: Track flight time, schedule maintenance
// Flight Log Record
