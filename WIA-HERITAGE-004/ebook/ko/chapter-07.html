using UnityEngine;
using UnityEngine.XR;
using Photon.Pun;

public class GanggangsullaeVRController : MonoBehaviourPun
{
    public Transform[] participants;  // 50-60명의 참가자
    public AudioSource traditionalSong;
    public GameObject circleGuide;

    void Start()
    {
        if (photonView.IsMine)
        {
            // VR 헤드셋 위치 및 회전 추적
            InputTracking.GetNodeStates(nodeStates);

            // 강강술래 원형 대형 생성
            FormCircle();

            // 전통 노래 재생 (Ambisonic 공간 음향)
            traditionalSong.spatialBlend = 1.0f;
            traditionalSong.Play();
        }
    }

    void FormCircle()
    {
        float radius = 10.0f;
        int participantCount = PhotonNetwork.PlayerList.Length;

        for (int i = 0; i < participantCount; i++)
        {
            float angle = i * (360f / participantCount) * Mathf.Deg2Rad;
            Vector3 position = new Vector3(
                Mathf.Cos(angle) * radius,
                0,
                Mathf.Sin(angle) * radius
            );

            // 각 참가자를 원형으로 배치
            photonView.RPC("PlaceParticipant", RpcTarget.All, i, position);
        }
    }

    [PunRPC]
    void PlaceParticipant(int index, Vector3 position)
    {
        if (index < participants.Length)
        {
            participants[index].position = position;
            participants[index].LookAt(Vector3.zero);  // 원의 중심을 바라봄
        }
    }

    void Update()
    {
        if (photonView.IsMine)
        {
            // 손 모션 캡처로 강강술래 동작 추적
            TrackHandMotions();

            // 노래 박자에 맞춰 원 회전
            RotateCircle();
        }
    }

    void RotateCircle()
    {
        float rotationSpeed = traditionalSong.pitch * 5.0f;
        transform.RotateAround(Vector3.zero, Vector3.up, rotationSpeed * Time.deltaTime);
    }
}

07.3 AI 안무 분석 및 교육

전통 춤은 수십 년의 수련이 필요한 고난도 예술입니다. AI는 명인의 동작을 프레임 단위로 분석하여 초보자도 이해할 수 있는 단계별 교육 커리큘럼을 자동 생성합니다.

모션 캡처 데이터 AI 분석 파이프라인

import tensorflow as tf
import numpy as np
from sklearn.cluster import DBSCAN

class TraditionalDanceAnalyzer:
    def __init__(self, bvh_file):
        self.motion_data = self.load_bvh(bvh_file)
        self.keyframes = []
        self.difficulty_scores = []

    def segment_movements(self):
        """춤 동작을 의미 있는 세그먼트로 분할"""
        # 관절 속도 계산
        velocities = np.diff(self.motion_data['positions'], axis=0)
        speeds = np.linalg.norm(velocities, axis=2)

        # DBSCAN 클러스터링으로 동작 전환점 탐지
        clustering = DBSCAN(eps=0.5, min_samples=10).fit(speeds.T)

        segments = []
        current_segment = []
        for i, label in enumerate(clustering.labels_):
            if label != -1:
                current_segment.append(i)
            else:
                if current_segment:
                    segments.append(current_segment)
                    current_segment = []

        return segments

    def extract_keyframes(self, segment):
        """각 세그먼트에서 핵심 프레임 추출"""
        positions = self.motion_data['positions'][segment]

        # 포즈 변화가 가장 큰 프레임 선택
        pose_changes = np.sum(np.abs(np.diff(positions, axis=0)), axis=(1,2))
        keyframe_indices = self.find_peaks(pose_changes, distance=30)

        return segment[keyframe_indices]

    def calculate_difficulty(self, segment):
        """동작 난이도 자동 계산"""
        positions = self.motion_data['positions'][segment]

        # 난이도 요소들
        speed = np.mean(np.linalg.norm(np.diff(positions, axis=0), axis=2))
        complexity = len(self.extract_keyframes(segment))
        balance_challenge = self.calculate_balance_score(positions)
        coordination = self.calculate_coordination_score(positions)

        # 가중 평균으로 난이도 점수 (0-10)
        difficulty = (
            speed * 0.3 +
            complexity * 0.3 +
            balance_challenge * 0.2 +
            coordination * 0.2
        )

        return min(difficulty, 10.0)

    def generate_curriculum(self):
        """난이도 순으로 교육 커리큘럼 생성"""
        segments = self.segment_movements()
        curriculum = []

        for seg in segments:
            keyframes = self.extract_keyframes(seg)
            difficulty = self.calculate_difficulty(seg)

            curriculum.append({
                'name': self.name_movement(seg),
                'duration': len(seg) / 120.0,  # 120fps
                'keyframes': keyframes.tolist(),
                'difficulty': difficulty,
                'description': self.generate_description(seg),
                'prerequisites': self.find_prerequisites(seg, curriculum)
            })

        # 난이도 순 정렬
        curriculum.sort(key=lambda x: x['difficulty'])
        return curriculum

# 사용 예시
analyzer = TraditionalDanceAnalyzer('pansori-pansori-gestures.bvh')
curriculum = analyzer.generate_curriculum()

print(f"총 {len(curriculum)}개 동작 분석 완료")
for i, movement in enumerate(curriculum):
    print(f"{i+1}. {movement['name']} (난이도: {movement['difficulty']:.1f}/10)")
    print(f"   소요 시간: {movement['duration']:.1f}초")
    print(f"   핵심 프레임: {len(movement['keyframes'])}개")
    print(f"   설명: {movement['description']}")

AI 기반 관계 발견

Graph Neural Network (GNN)으로 숨겨진 패턴 탐지:

  • 전파 경로 추적: 판소리가 실크로드를 통해 중앙아시아 서사시에 영향받았음을 발견

  • 교육 효율화: 한 전통을 배운 사람이 다음으로 배우기 쉬운 전통 추천
🕸️ 놀라운 발견: 판소리와 플라멩코

07.5 소멸 유산 AI 재구성

많은 무형 문화유산은 단편적 기록만 남기고 사라졌습니다. 19세기 사진 3장, 20세기 초 녹음 2분, 구전 증언 몇 줄이 전부인 경우가 많습니다. AI는 이러한 단서들을 종합하여 소멸된 전통을 재구성합니다.

AI 재구성 파이프라인

  1. 데이터 수집: 역사 문헌, 사진, 음성, 구전, 유사 전통 분석
  2. 패턴 학습: GAN (Generative Adversarial Network)으로 유사 전통의 패턴 학습
  3. 생성: 학습된 패턴을 기반으로 누락된 부분 생성
  4. 검증: 원로 전문가와 AI가 협업하여 정확성 검증
  5. 라벨링: 모든 AI 생성 부분에 "AI Reconstructed" 라벨 표시

import torch
import torch.nn as nn

class DanceGenerator(nn.Module):
    """소멸된 춤의 동작 생성"""
    def __init__(self, latent_dim=128, sequence_length=1000):
        super().__init__()
        self.lstm = nn.LSTM(latent_dim, 256, num_layers=3, batch_first=True)
        self.fc = nn.Linear(256, 51)  # 17 joints × 3 (x,y,z)

    def forward(self, noise):
        lstm_out, _ = self.lstm(noise)
        motion = self.fc(lstm_out)
        return motion.view(-1, 17, 3)  # [frames, joints, coordinates]

class DanceDiscriminator(nn.Module):
    """실제 춤과 생성된 춤을 구분"""
    def __init__(self, sequence_length=1000):
        super().__init__()
        self.conv1d = nn.Sequential(
            nn.Conv1d(51, 128, kernel_size=5),
            nn.ReLU(),
            nn.Conv1d(128, 256, kernel_size=5),
            nn.ReLU()
        )
        self.fc = nn.Linear(256 * (sequence_length - 8), 1)
        self.sigmoid = nn.Sigmoid()

    def forward(self, motion):
        motion = motion.view(motion.size(0), -1, motion.size(1))
        conv_out = self.conv1d(motion)
        flattened = conv_out.view(conv_out.size(0), -1)
        return self.sigmoid(self.fc(flattened))

# 훈련
generator = DanceGenerator()
discriminator = DanceDiscriminator()

    g_loss = generator_loss(discriminator(generated_dance))

# 소멸된 춤 재구성
extinct_dance_clues = load_historical_clues()  # 19세기 사진, 증언 등
reconstructed_dance = generator.generate_from_clues(extinct_dance_clues)

AI 재구성은 신중하게 접근해야 합니다:

  • 투명성: 모든 AI 생성 부분을 명확히 표시
  • 공동체 승인: 원래 전통의 후손 공동체가 재구성을 검토하고 승인
  • 학술적 근거: 재구성의 모든 가정과 불확실성을 문서화
  • 대안 제시: 하나의 재구성이 아닌 여러 가능성 제시
  • 살아있는 전통 우선: AI 재구성은 소멸된 전통에만 적용, 현존 전통은 실제 전승자 우선

Unity Timeline으로 시간 여행 구현

using UnityEngine;
using UnityEngine.Timeline;
using UnityEngine.Playables;

public class TimeTravelController : MonoBehaviour
{
    public PlayableDirector timeline;
    public GameObject[] eraScenes;  // 1500년대, 1700년대, 1900년대, 2024년

    public void TravelToEra(int year)
    {
        // 시대별 씬 활성화
        foreach (var scene in eraScenes)
        {
            scene.SetActive(false);
        }

        if (year < 1600)
        {
            eraScenes[0].SetActive(true);  // 조선 초기
            LoadHistoricalPerformance("pansori-early-reconstruction");
        }
        else if (year < 1800)
        {
            eraScenes[1].SetActive(true);  // 조선 중기
            LoadHistoricalPerformance("pansori-middle-era");
        }
        else if (year < 1950)
        {
            eraScenes[2].SetActive(true);  // 조선 말기 ~ 일제강점기
            LoadHistoricalPerformance("pansori-1920s-recording");
        }
        else
        {
            eraScenes[3].SetActive(true);  // 현대
            LoadHistoricalPerformance("pansori-2024-live");
        }

        // 시간 흐름 효과
        StartCoroutine(TimeWarpEffect());
    }

    IEnumerator TimeWarpEffect()
    {
        // 시간 왜곡 시각 효과
        RenderSettings.skybox.SetFloat("_Exposure", 3.0f);

        for (float t = 0; t < 2.0f; t += Time.deltaTime)
        {
            // 화면 왜곡
            Camera.main.fieldOfView = Mathf.Lerp(60, 120, t / 2.0f);
            yield return null;
        }

        Camera.main.fieldOfView = 60;
        RenderSettings.skybox.SetFloat("_Exposure", 1.0f);
    }
}

07.7 다감각 무형유산 재현

  • 진동 플랫폼: 드럼의 저음 진동, 발 구르기

07.9 복습 문제

  1. VR 가상 축제에서 150,000명의 동시 접속을 처리하기 위한 샤딩 전략을 설명하고, 인스턴스 분할과 지역별 서버 배치의 이점을 논의하세요.
  2. AI 안무 분석이 전통 춤 동작의 난이도를 자동 계산하는 4가지 요소(속도, 복잡성, 균형, 협응)를 설명하고, 각 요소가 학습자에게 미치는 영향을 논의하세요.
  3. 문화 지식 그래프가 판소리와 플라멩코의 연결을 발견한 과정을 설명하고, 이러한 발견이 무형 문화유산 연구와 교육에 어떻게 활용될 수 있는지 논의하세요.
  4. AI를 사용한 소멸 유산 재구성의 윤리적 가이드라인 5가지를 설명하고, 왜 투명성과 공동체 승인이 중요한지 논의하세요.
  5. VR 시간 여행에서 600년 전 조선시대 종묘대제를 재현하기 위한 기술적 요소들을 나열하고, 각 요소의 정확도와 한계를 논의하세요.
  6. 5D 다감각 체험이 단순 시청각 체험보다 몰입도가 높은 이유를 설명하고, 촉각, 후각, 온도 같은 추가 감각이 문화유산 교육에 기여하는 바를 논의하세요.
