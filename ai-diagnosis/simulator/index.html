SDK Integration

Python SDK

from wia_ai_diagnosis import DiagnosisClient client = DiagnosisClient( api_key='your_api_key' ) result = client.analyze_image( image_path='chest_xray.dcm', modality='xray', patient_id='P-2025-001' ) print(result.diagnosis)

TypeScript SDK

import { DiagnosisClient } from '@wia/ai-diagnosis'; const client = new DiagnosisClient({ apiKey: 'your_api_key' }); const result = await client.analyzeImage({ imagePath: 'chest_xray.dcm', modality: 'xray', patientId: 'P-2025-001' }); console.log(result.diagnosis);
