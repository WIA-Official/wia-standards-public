# 2장: 수어 디지털화의 현재 과제

## 기술적 장벽의 현실

수어 디지털화는 음성 언어에는 존재하지 않는 독특한 과제에 직면합니다. 수어는 3차원 공간에서 동시에 여러 채널(손, 얼굴, 몸)을 통해 의미를 전달하는 시각-공간 언어입니다. 이러한 복잡성은 기술 시스템에 심각한 도전을 제시합니다.

### 비디오 캡처 및 품질 문제

#### 프레임률 불충분

**문제의 본질:**

수어의 많은 의미 구분은 빠른 손 움직임에 의존합니다. 30fps의 표준 비디오는 이러한 미묘한 차이를 놓칩니다:

```typescript
interface SignDistinction {
  sign1: {
    name: "KNOW",
    handshape: "flat_hand",
    location: "forehead",
    movement: "single_tap",
    duration: 200  // ms - 30fps에서 6프레임만
  },
  sign2: {
    name: "THINK",
    handshape: "flat_hand",
    location: "forehead",
    movement: "circular_rub",
    duration: 500  // ms - 미묘한 원형 움직임
  }
  // 30fps: 움직임이 흐릿함
  // 60fps: 명확한 구분
}
```

class SignLanguageComplexity:
    """
    수어는 여러 독립적이면서도 상호 연결된 채널을 가짐
    """
    def __init__(self):
        self.channels = {
            "dominant_hand": {
                "dimensions": 3,  # x, y, z
                "joints": 21,     # 손가락 관절
                "features": 63    # 21 * 3
            },
            "non_dominant_hand": {
                "dimensions": 3,
                "joints": 21,
                "features": 63
            },
            "face": {
                "landmarks": 68,
                "expressions": 7,  # 감정
                "aus": 17         # Action Units
            },
            "body": {
                "joints": 25,     # 전신 골격
                "dimensions": 3,
                "features": 75
            },
            "head": {
                "orientation": 3,  # yaw, pitch, roll
                "movement": 3      # x, y, z 변위
            }
        }
        self.total_features = 63 + 63 + 68 + 7 + 17 + 75 + 6  # = 299 특징
        self.temporal_frames = 60  # 1초 @ 60fps
        self.input_size = 299 * 60  # = 17,940 차원!

    def challenge(self):
        return """
        음성 인식: 1초 = 80 MFCC 특징
        수어 인식: 1초 = 17,940 특징
        = 224배 더 복잡!
        """
