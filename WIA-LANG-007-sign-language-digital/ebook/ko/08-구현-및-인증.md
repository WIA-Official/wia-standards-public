```typescript
// src/avatar/SignLanguageAvatar.ts
import * as THREE from 'three';
import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader';

interface SignSequence {
  signs: Sign[];
  duration: number;
}

interface Sign {
  hamnosys: string;
  glosses: string[];
  animation: AnimationClip;
}

class SignLanguageAvatar {
  private scene: THREE.Scene;
  private camera: THREE.PerspectiveCamera;
  private renderer: THREE.WebGLRenderer;
  private avatar: THREE.Object3D | null = null;
  private mixer: THREE.AnimationMixer | null = null;
  private animationQueue: AnimationClip[] = [];

  constructor(container: HTMLElement) {
    // Three.js 초기화
    this.scene = new THREE.Scene();
    this.camera = new THREE.PerspectiveCamera(
      45,
      container.clientWidth / container.clientHeight,
      0.1,
      1000
    );
    this.renderer = new THREE.WebGLRenderer({ antialias: true });
    this.renderer.setSize(container.clientWidth, container.clientHeight);
    container.appendChild(this.renderer.domElement);

    // 조명 설정
    const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
    this.scene.add(ambientLight);

    const directionalLight = new THREE.DirectionalLight(0xffffff, 0.8);
    directionalLight.position.set(0, 10, 10);
    this.scene.add(directionalLight);

    // 카메라 위치
    this.camera.position.set(0, 1.5, 3);
    this.camera.lookAt(0, 1.2, 0);

    this.loadAvatar();
    this.animate();
  }

  private async loadAvatar(): Promise<void> {
    const loader = new GLTFLoader();

    try {
      const gltf = await loader.loadAsync('/models/sign_avatar.glb');
      this.avatar = gltf.scene;
      this.scene.add(this.avatar);
      this.mixer = new THREE.AnimationMixer(this.avatar);
    } catch (error) {
      console.error('Failed to load avatar:', error);
    }
  }

  // 수어 시퀀스 재생
  async playSignSequence(sequence: SignSequence): Promise<void> {
    if (!this.mixer || !this.avatar) return;

    for (const sign of sequence.signs) {
      await this.playSign(sign);
    }
  }

  // 개별 수어 재생
  private playSign(sign: Sign): Promise<void> {
    return new Promise((resolve) => {
      if (!this.mixer) {
        resolve();
        return;
      }

      const action = this.mixer.clipAction(sign.animation);
      action.setLoop(THREE.LoopOnce, 1);
      action.clampWhenFinished = true;
      action.play();

      const onFinished = () => {
        this.mixer?.removeEventListener('finished', onFinished);
        resolve();
      };

      this.mixer.addEventListener('finished', onFinished);
    });
  }

  // 텍스트를 수어로 변환하여 재생
  async signText(text: string, targetLanguage: string = 'ASL'): Promise<void> {
    // API를 통해 텍스트를 수어 시퀀스로 변환
    const response = await fetch('/api/translate/text-to-sign', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ text, targetLanguage })
    });

    const sequence: SignSequence = await response.json();
    await this.playSignSequence(sequence);
  }

  // 표정 설정
  setFacialExpression(expression: FacialExpression): void {
    if (!this.avatar) return;

    const head = this.avatar.getObjectByName('Head');
    if (head && head instanceof THREE.Mesh) {
      const morphTargets = head.morphTargetInfluences;
      if (morphTargets) {
        // 표정별 모프 타겟 설정
        morphTargets[0] = expression.eyebrowRaise;
        morphTargets[1] = expression.eyebrowFurrow;
        morphTargets[2] = expression.smile;
        morphTargets[3] = expression.mouthOpen;
      }
    }
  }

  private animate(): void {
    requestAnimationFrame(() => this.animate());

    if (this.mixer) {
      this.mixer.update(0.016); // 60fps
    }

    this.renderer.render(this.scene, this.camera);
  }

  dispose(): void {
    this.renderer.dispose();
    this.scene.clear();
  }
}

interface FacialExpression {
  eyebrowRaise: number;
  eyebrowFurrow: number;
  smile: number;
  mouthOpen: number;
}
```

### 8.2.2 수어 인식 시스템

```python
# src/recognition/sign_recognizer.py
import cv2
import numpy as np
import mediapipe as mp
from tensorflow import keras

class SignLanguageRecognizer:
    """실시간 수어 인식 시스템"""

    def __init__(self, model_path: str, vocabulary_path: str):
        # MediaPipe Holistic 초기화
        self.mp_holistic = mp.solutions.holistic
        self.holistic = self.mp_holistic.Holistic(
            min_detection_confidence=0.5,
            min_tracking_confidence=0.5
        )

        # 수어 인식 모델 로드
        self.model = keras.models.load_model(model_path)

        # 어휘 사전 로드
        with open(vocabulary_path, 'r') as f:
            self.vocabulary = json.load(f)

        # 시퀀스 버퍼
        self.sequence_buffer = []
        self.sequence_length = 30  # 30 프레임

    def extract_landmarks(self, frame: np.ndarray) -> np.ndarray:
        """프레임에서 랜드마크 추출"""
        # BGR to RGB
        image_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)

        # MediaPipe 처리
        results = self.holistic.process(image_rgb)

        # 랜드마크 추출
        landmarks = []

        # 포즈 랜드마크 (33개)
        if results.pose_landmarks:
            for lm in results.pose_landmarks.landmark:
                landmarks.extend([lm.x, lm.y, lm.z, lm.visibility])
        else:
            landmarks.extend([0] * 132)  # 33 * 4

        # 왼손 랜드마크 (21개)
        if results.left_hand_landmarks:
            for lm in results.left_hand_landmarks.landmark:
                landmarks.extend([lm.x, lm.y, lm.z])
        else:
            landmarks.extend([0] * 63)  # 21 * 3

        # 오른손 랜드마크 (21개)
        if results.right_hand_landmarks:
            for lm in results.right_hand_landmarks.landmark:
                landmarks.extend([lm.x, lm.y, lm.z])
        else:
            landmarks.extend([0] * 63)  # 21 * 3

        # 얼굴 랜드마크 (주요 68개)
        if results.face_landmarks:
            # 주요 얼굴 포인트만 선택
            key_face_indices = [33, 133, 362, 263, 1, 61, 291, 199]
            for idx in key_face_indices:
                lm = results.face_landmarks.landmark[idx]
                landmarks.extend([lm.x, lm.y, lm.z])
        else:
            landmarks.extend([0] * 24)  # 8 * 3

        return np.array(landmarks)

    def recognize_sign(self, frame: np.ndarray) -> dict:
        """프레임에서 수어 인식"""
        # 랜드마크 추출
        landmarks = self.extract_landmarks(frame)

        # 시퀀스 버퍼에 추가
        self.sequence_buffer.append(landmarks)

        # 버퍼가 충분히 차면 인식 수행
        if len(self.sequence_buffer) >= self.sequence_length:
            sequence = np.array(self.sequence_buffer[-self.sequence_length:])
            sequence = np.expand_dims(sequence, axis=0)

            # 모델 예측
            predictions = self.model.predict(sequence, verbose=0)[0]

            # 상위 3개 예측
            top_indices = np.argsort(predictions)[-3:][::-1]

            results = []
            for idx in top_indices:
                if predictions[idx] > 0.5:  # 임계값
                    results.append({
                        'sign': self.vocabulary[idx],
                        'confidence': float(predictions[idx])
                    })

            return {
                'recognized': len(results) > 0,
                'predictions': results
            }

        return {
            'recognized': False,
            'predictions': [],
            'message': f'Collecting frames... {len(self.sequence_buffer)}/{self.sequence_length}'
        }

    def recognize_stream(self, video_source: int = 0):
        """실시간 비디오 스트림에서 수어 인식"""
        cap = cv2.VideoCapture(video_source)

        while cap.isOpened():
            ret, frame = cap.read()
            if not ret:
                break

            # 수어 인식
            result = self.recognize_sign(frame)

            # 결과 표시
            if result['recognized']:
                for i, pred in enumerate(result['predictions']):
                    text = f"{pred['sign']}: {pred['confidence']:.2f}"
                    cv2.putText(frame, text, (10, 30 + i * 30),
                               cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)

            cv2.imshow('Sign Language Recognition', frame)

            if cv2.waitKey(1) & 0xFF == ord('q'):
                break

        cap.release()
        cv2.destroyAllWindows()

    def close(self):
        """리소스 정리"""
        self.holistic.close()
```

### 8.2.3 수어 번역 API

```typescript
// src/api/translation.ts
import express from 'express';
import { SignTranslator } from '../translation/SignTranslator';
import { HamNoSysEncoder } from '../encoding/HamNoSysEncoder';

const router = express.Router();
const translator = new SignTranslator();
const encoder = new HamNoSysEncoder();

// 텍스트 → 수어 번역
router.post('/text-to-sign', async (req, res) => {
  try {
    const { text, targetLanguage, outputFormat } = req.body;

    // 텍스트를 수어 글로스로 변환
    const glosses = await translator.textToGloss(text, targetLanguage);

    // 글로스를 수어 시퀀스로 변환
    const signs = await Promise.all(
      glosses.map(async (gloss) => {
        const hamnosys = await encoder.glossToHamNoSys(gloss, targetLanguage);
        const animation = await translator.getAnimation(gloss, targetLanguage);

        return {
          gloss,
          hamnosys,
          animation: animation.id,
          duration: animation.duration
        };
      })
    );

    res.json({
      success: true,
      sourceText: text,
      targetLanguage,
      signs,
      totalDuration: signs.reduce((sum, s) => sum + s.duration, 0)
    });

  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
});

// 수어 → 텍스트 번역
router.post('/sign-to-text', async (req, res) => {
  try {
    const { landmarks, sourceLanguage, targetLanguage } = req.body;

    // 랜드마크에서 수어 인식
    const recognizedSigns = await translator.recognizeSigns(landmarks);

    // 수어를 텍스트로 번역
    const translatedText = await translator.signToText(
      recognizedSigns,
      sourceLanguage,
      targetLanguage
    );

    res.json({
      success: true,
      recognizedSigns,
      translatedText,
      sourceLanguage,
      targetLanguage
    });

  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
});

// 수어 사전 조회
router.get('/dictionary/:language/:term', async (req, res) => {
  try {
    const { language, term } = req.params;

    const entry = await translator.lookupSign(term, language);

    if (!entry) {
      return res.status(404).json({
        success: false,
        error: 'Sign not found'
      });
    }

    res.json({
      success: true,
      entry: {
        term: entry.term,
        gloss: entry.gloss,
        hamnosys: entry.hamnosys,
        signwriting: entry.signwriting,
        videoUrl: entry.videoUrl,
        description: entry.description,
        examples: entry.examples
      }
    });

  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
});

export default router;
```
