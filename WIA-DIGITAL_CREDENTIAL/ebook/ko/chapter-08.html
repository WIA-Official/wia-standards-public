// TypeScript 예제
import { WIAClient } from '@wia/wia-digital_credential-sdk';

// 클라이언트 초기화
const client = new WIAClient({
  apiKey: process.env.WIA_API_KEY,
  baseUrl: 'https://api.wia.org/v1',
  timeout: 30000,
  retryAttempts: 3
});

// 기본 작업 수행
async function performOperation() {
  try {
    const result = await client.executeStandardOperation({
      type: '배포 및 테스트',
      parameters: {
        mode: 'production',
        validation: true
      }
    });
    
    console.log('작업 성공:', result);
    return result;
  } catch (error) {
    console.error('작업 실패:', error.message);
    throw error;
  }
}

// 실행
performOperation()
  .then(result => console.log('완료:', result))
  .catch(error => console.error('오류:', error));

4.1 암호화 구현

// 데이터 암호화 예제
import { encrypt, decrypt } from '@wia/crypto-utils';

// 민감한 데이터 암호화
async function secureData(sensitiveInfo: string) {
  const encrypted = await encrypt({
    data: sensitiveInfo,
    algorithm: 'AES-256-GCM',
    key: process.env.ENCRYPTION_KEY
  });
  
  return {
    ciphertext: encrypted.data,
    iv: encrypted.iv,
    authTag: encrypted.authTag
  };
}

// 데이터 복호화
async function retrieveData(encrypted: EncryptedData) {
  const decrypted = await decrypt({
    ciphertext: encrypted.ciphertext,
    algorithm: 'AES-256-GCM',
    key: process.env.ENCRYPTION_KEY,
    iv: encrypted.iv,
    authTag: encrypted.authTag
  });
  
  return decrypted;
}

7. 실전 예제

7.1 완전한 구현 예제

다음은 배포 및 테스트의 완전한 구현 예제입니다:

// 프로덕션 레벨 구현
class WIADIGITAL_CREDENTIALService {
  private client: WIAClient;
  private cache: CacheManager;
  private logger: Logger;

  constructor(config: ServiceConfig) {
    this.client = new WIAClient(config);
    this.cache = new CacheManager();
    this.logger = new Logger('WIA-DIGITAL_CREDENTIAL');
  }

  async execute(params: ExecutionParams) {
    // 1. 캐시 확인
    const cached = await this.cache.get(params.key);
    if (cached) return cached;

    // 2. 작업 수행
    try {
      this.logger.info('작업 시작', params);
      
      const result = await this.client.execute({
        operation: '배포 및 테스트',
        parameters: params
      });

      // 3. 결과 캐싱
      await this.cache.set(params.key, result, 300);
      
      this.logger.info('작업 완료', result);
      return result;

    } catch (error) {
      this.logger.error('작업 실패', error);
      throw new ServiceError(error.message);
    }
  }

  async healthCheck() {
    return await this.client.ping();
  }
}

// 사용 예제
const service = new WIADIGITAL_CREDENTIALService({
  apiKey: process.env.WIA_API_KEY,
  environment: 'production'
});

await service.execute({ key: 'operation-1' });

// 데이터 모델 예제
interface DataModel {
  // 기본 엔티티
  entity: {
    id: string;
    type: string;
    version: number;
    createdAt: Date;
    updatedAt: Date;
    deletedAt?: Date;
  };
  
  // 관계
  relationships: {
    parent?: Reference;
    children: Reference[];
    related: Reference[];
  };
  
  // 메타데이터
  metadata: {
    tags: string[];
    categories: string[];
    permissions: Permission[];
    audit: AuditLog[];
  };
  
  // 커스텀 속성
  attributes: Record;
}

// 데이터 검증
function validateDataModel(model: DataModel): ValidationResult {
  const errors: ValidationError[] = [];
  
  // ID 검증
  if (!isValidUUID(model.entity.id)) {
    errors.push({ field: 'id', message: 'Invalid UUID format' });
  }
  
  // 타입 검증
  if (!ALLOWED_TYPES.includes(model.entity.type)) {
    errors.push({ field: 'type', message: 'Invalid entity type' });
  }
  
  // 날짜 검증
  if (model.entity.createdAt > model.entity.updatedAt) {
    errors.push({ field: 'dates', message: 'Invalid date sequence' });
  }
  
  return {
    valid: errors.length === 0,
    errors
  };
}

// OpenTelemetry를 사용한 분산 추적
import { trace, context } from '@opentelemetry/api';

const tracer = trace.getTracer('WIA-DIGITAL_CREDENTIAL');

async function tracedOperation(params: OperationParams) {
  // 새 스팬 생성
  const span = tracer.startSpan('operation.execute', {
    attributes: {
      'operation.type': params.type,
      'operation.id': params.id
    }
  });
  
  try {
    // 작업 실행
    const result = await executeOperation(params);
    
    // 성공 기록
    span.setStatus({ code: SpanStatusCode.OK });
    span.setAttribute('result.status', 'success');
    
    return result;
  } catch (error) {
    // 오류 기록
    span.setStatus({
      code: SpanStatusCode.ERROR,
      message: error.message
    });
    span.recordException(error);
    
    throw error;
  } finally {
    // 스팬 종료
    span.end();
  }
}
