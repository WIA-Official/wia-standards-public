8.1 분석의 진화

분석은 기술적 보고에서 예측 및 처방 기능으로 진화했습니다. 다음 프론티어는 증강 분석입니다 - AI와 머신러닝이 인사이트 발견을 자동화하여 비기술 사용자도 분석에 액세스할 수 있게 하고 인사이트까지의 시간을 극적으로 단축시킵니다.

8.1.1 분석의 4단계 발전

단계유형질문기술성숙도
1단계 기술적 분석 무슨 일이 일어났나? 리포팅, 대시보드 보편적
2단계 진단적 분석 왜 일어났나? 드릴다운, 데이터 마이닝 일반적
3단계 예측적 분석 무슨 일이 일어날까? 머신러닝, 통계 모델 성장 중
4단계 처방적 분석 어떻게 해야 하나? 최적화, 시뮬레이션, AI 신흥
5단계 증강 분석 AI가 자동으로 발견 AutoML, NLG, 자동 인사이트 미래

8.2 증강 분석 (Augmented Analytics)

증강 분석은 머신러닝과 AI를 사용하여 데이터 준비, 인사이트 발견 및 인사이트 공유를 자동화합니다. Gartner는 2025년까지 증강 분석이 데이터 과학자뿐만 아니라 비즈니스 사용자의 주요 분석 방법이 될 것으로 예측합니다.

기능설명이점현재 성숙도
자동화된 데이터 준비 AI가 정제 및 변환 제안 수작업 80% 이상 감소 상용화

자동 서술 (NLG) 작성된 요약 생성 자연어로 인사이트 설명 성장 중

# Tableau의 Ask Data를 사용한 자연어 쿼리 시뮬레이션
# Python으로 구현한 간단한 NLQ 엔진

import spacy
import pandas as pd
from dateutil import parser

class NaturalLanguageQueryEngine:
    def __init__(self, dataframe):
        self.df = dataframe
        self.nlp = spacy.load("en_core_web_sm")

    def parse_query(self, query):
        """자연어 쿼리를 SQL로 변환"""
        doc = self.nlp(query.lower())

        # 의도 파악
        intent = self._detect_intent(doc)
        # 엔티티 추출
        entities = self._extract_entities(doc)

        return self._build_query(intent, entities)

    def _detect_intent(self, doc):
        """사용자 의도 탐지"""
        if any(token.text in ["show", "get", "find"] for token in doc):
            return "SELECT"
        elif any(token.text in ["sum", "total", "aggregate"] for token in doc):
            return "AGGREGATE"
        elif any(token.text in ["compare", "versus"] for token in doc):
            return "COMPARE"
        elif any(token.text in ["trend", "over time"] for token in doc):
            return "TREND"
        return "SELECT"

    def _extract_entities(self, doc):
        """엔티티 추출 (메트릭, 차원, 필터)"""
        entities = {
            'metrics': [],
            'dimensions': [],
            'filters': {},
            'time_range': None
        }

        # 메트릭 탐지
        metric_keywords = ["sales", "revenue", "profit", "orders", "customers"]
        for token in doc:
            if token.text in metric_keywords:
                entities['metrics'].append(token.text)

        # 차원 탐지
        dimension_keywords = ["region", "product", "category", "customer"]
        for token in doc:
            if token.text in dimension_keywords:
                entities['dimensions'].append(token.text)

        return entities

    def execute(self, query):
        """쿼리 실행 및 결과 반환"""
        parsed = self.parse_query(query)

        # 간단한 구현
        if "sales by region" in query.lower():
            return self.df.groupby('region')['sales'].sum()
        elif "top products" in query.lower():
            return self.df.groupby('product')['sales'].sum().nlargest(10)

        return self.df.head()

# 사용 예제
df = pd.read_csv('sales_data.csv')
nlq = NaturalLanguageQueryEngine(df)

# 자연어 쿼리
result = nlq.execute("Show me sales by region")
print(result)

result = nlq.execute("What are the top 10 products?")
print(result)

8.2.3 자동 인사이트 생성

# Power BI의 Smart Narratives와 유사한 자동 인사이트 생성

import pandas as pd
import numpy as np
from scipy import stats

class AutoInsightGenerator:
    def __init__(self, dataframe):
        self.df = dataframe

    def generate_insights(self):
        """자동으로 인사이트 발견 및 서술 생성"""
        insights = []

        # 1. 트렌드 탐지
        insights.extend(self._detect_trends())

        # 2. 이상치 탐지
        insights.extend(self._detect_anomalies())

        # 3. 상관관계 발견
        insights.extend(self._find_correlations())

        # 4. 세그먼트 분석
        insights.extend(self._segment_analysis())

        return self._rank_insights(insights)

    def _detect_trends(self):
        """시계열 트렌드 탐지"""
        insights = []

        if 'date' in self.df.columns:
            ts_data = self.df.set_index('date')['sales']

            # 선형 회귀로 트렌드 파악
            x = np.arange(len(ts_data))
            slope, intercept, r_value, p_value, std_err = stats.linregress(x, ts_data)

            if p_value < 0.05:  # 유의미한 트렌드
                direction = "증가" if slope > 0 else "감소"
                change_pct = (slope * len(ts_data) / ts_data.mean()) * 100

                insight = {
                    'type': 'TREND',
                    'importance': abs(change_pct),
                    'narrative': f"판매는 {direction} 추세를 보이고 있으며, "
                                f"기간 동안 약 {abs(change_pct):.1f}% {direction}했습니다.",
                    'confidence': r_value ** 2
                }
                insights.append(insight)

        return insights

    def _detect_anomalies(self):
        """이상치 탐지"""
        insights = []

        for col in self.df.select_dtypes(include=[np.number]).columns:
            mean = self.df[col].mean()
            std = self.df[col].std()
            z_scores = np.abs((self.df[col] - mean) / std)

            anomalies = self.df[z_scores > 3]

            if len(anomalies) > 0:
                insight = {
                    'type': 'ANOMALY',
                    'importance': len(anomalies) / len(self.df) * 100,
                    'narrative': f"{col}에서 {len(anomalies)}개의 이상치가 발견되었습니다. "
                                f"평균 {mean:.2f}에서 크게 벗어난 값들입니다.",
                    'confidence': 0.95
                }
                insights.append(insight)

        return insights

    def _find_correlations(self):
        """강한 상관관계 발견"""
        insights = []
        numeric_cols = self.df.select_dtypes(include=[np.number]).columns

        if len(numeric_cols) >= 2:
            corr_matrix = self.df[numeric_cols].corr()

            # 대각선과 중복 제거
            for i in range(len(corr_matrix.columns)):
                for j in range(i+1, len(corr_matrix.columns)):
                    corr = corr_matrix.iloc[i, j]

                    if abs(corr) > 0.7:  # 강한 상관관계
                        col1, col2 = corr_matrix.columns[i], corr_matrix.columns[j]
                        relationship = "양의" if corr > 0 else "음의"

                        insight = {
                            'type': 'CORRELATION',
                            'importance': abs(corr) * 100,
                            'narrative': f"{col1}과 {col2} 사이에 강한 {relationship} "
                                        f"상관관계가 있습니다 (r={corr:.2f}).",
                            'confidence': abs(corr)
                        }
                        insights.append(insight)

        return insights

    def _segment_analysis(self):
        """세그먼트 성과 분석"""
        insights = []

        if 'category' in self.df.columns and 'sales' in self.df.columns:
            segment_perf = self.df.groupby('category')['sales'].agg(['sum', 'mean', 'count'])
            top_segment = segment_perf['sum'].idxmax()
            top_value = segment_perf.loc[top_segment, 'sum']
            total = segment_perf['sum'].sum()

            insight = {
                'type': 'SEGMENT',
                'importance': (top_value / total) * 100,
                'narrative': f"{top_segment}이 전체 판매의 "
                            f"{(top_value/total)*100:.1f}%를 차지하는 "
                            f"최고 성과 세그먼트입니다.",
                'confidence': 0.90
            }
            insights.append(insight)

        return insights

    def _rank_insights(self, insights):
        """중요도에 따라 인사이트 순위 매기기"""
        return sorted(insights, key=lambda x: x['importance'], reverse=True)

# 사용 예제
df = pd.read_csv('sales_data.csv')
generator = AutoInsightGenerator(df)
insights = generator.generate_insights()

print("🔍 자동 발견된 인사이트:\n")
for i, insight in enumerate(insights[:5], 1):
    print(f"{i}. {insight['narrative']}")
    print(f"   (신뢰도: {insight['confidence']:.0%}, 중요도: {insight['importance']:.1f})\n")

8.3 AutoML (자동화된 머신러닝)

AutoML은 실제 문제에 머신러닝을 적용하는 엔드투엔드 프로세스를 자동화합니다. 데이터 과학자가 아닌 사람도 고품질 ML 모델을 구축할 수 있게 합니다.

AutoML 파이프라인 구성 요소:
  • 자동화된 특징 엔지니어링: 자동으로 특징 생성, 선택, 변환
  • 모델 선택: 여러 알고리즘을 시도하고 최상의 것 선택
  • 하이퍼파라미터 튜닝: 베이지안 최적화로 파라미터 자동 조정
  • 앙상블 학습: 여러 모델을 결합하여 성능 향상
  • 모델 해석: 예측 및 특징 중요도 설명
  • 자동 배포: 프로덕션에 원클릭 모델 배포

8.3.2 AutoML 구현 예제

# H2O AutoML을 사용한 자동 모델 학습

import h2o
from h2o.automl import H2OAutoML
import pandas as pd

# H2O 초기화
h2o.init()

# 데이터 로드
df = h2o.import_file("customer_churn.csv")

# 타겟 및 특징 정의
target = "churn"
features = df.columns
features.remove(target)

# 학습/검증 분할
train, test = df.split_frame(ratios=[0.8])

# AutoML 실행
aml = H2OAutoML(
    max_models=20,              # 최대 모델 수
    max_runtime_secs=3600,      # 최대 실행 시간 (1시간)
    seed=42,
    balance_classes=True,        # 불균형 클래스 처리
    sort_metric="AUC",           # 정렬 메트릭
    exclude_algos=["StackedEnsemble"],  # 제외할 알고리즘
    nfolds=5                     # 교차 검증 폴드
)

# 학습 시작
aml.train(x=features, y=target, training_frame=train)

# 리더보드 - 모든 모델 순위
lb = aml.leaderboard
print(lb.head())

# 최고 모델
best_model = aml.leader
print(f"\n최고 모델: {best_model.model_id}")
print(f"AUC: {best_model.auc():.4f}")

# 테스트 세트 평가
perf = best_model.model_performance(test)
print(f"\n테스트 AUC: {perf.auc():.4f}")

# 특징 중요도
importance = best_model.varimp(use_pandas=True)
print("\n상위 10개 특징:")
print(importance.head(10))

# 예측
predictions = best_model.predict(test)
print("\n예측 샘플:")
print(predictions.head())

# 모델 저장
model_path = h2o.save_model(model=best_model, path="/models", force=True)
print(f"\n모델 저장 위치: {model_path}")

# MOJO로 내보내기 (프로덕션 배포용)
best_model.download_mojo(path="/models", get_genmodel_jar=True)

# 설명
explanation = best_model.explain(test, render=False)

# LangChain과 GPT를 사용한 대화형 데이터 분석

from langchain.agents import create_pandas_dataframe_agent
from langchain.llms import OpenAI
import pandas as pd

# 데이터 로드
df = pd.read_csv("sales_data.csv")

# 대화형 에이전트 생성
agent = create_pandas_dataframe_agent(

# 데이터에서 자연어 보고서 자동 생성

import pandas as pd
from datetime import datetime

class NarrativeReportGenerator:
    def __init__(self, dataframe):
        self.df = dataframe

    def generate_executive_summary(self):
        """경영진 요약 보고서 생성"""
        report = []

        # 제목
        report.append("📊 월간 판매 실적 보고서")
        report.append(f"보고 기간: {datetime.now().strftime('%Y년 %m월')}")
        report.append("\n" + "="*60 + "\n")

        # 주요 메트릭
        total_sales = self.df['sales'].sum()
        total_orders = len(self.df)
        avg_order_value = total_sales / total_orders

        report.append("📈 핵심 성과 지표 (KPI)")
        report.append(f"• 총 판매액: ${total_sales:,.2f}")
        report.append(f"• 주문 건수: {total_orders:,}")
        report.append(f"• 평균 주문 금액: ${avg_order_value:,.2f}")

        # 전월 대비
        if 'previous_month_sales' in self.df.columns:
            prev_total = self.df['previous_month_sales'].sum()
            growth = ((total_sales - prev_total) / prev_total) * 100

            trend = "증가" if growth > 0 else "감소"
            report.append(f"\n전월 대비 {abs(growth):.1f}% {trend}하여 "
                        f"{'긍정적' if growth > 0 else '부정적'}인 추세를 보이고 있습니다.")

        # 지역별 성과
        report.append("\n\n🌍 지역별 성과")
        regional_sales = self.df.groupby('region')['sales'].sum().sort_values(ascending=False)

        top_region = regional_sales.index[0]
        top_value = regional_sales.iloc[0]
        top_pct = (top_value / total_sales) * 100

        report.append(f"{top_region} 지역이 ${top_value:,.2f} ({top_pct:.1f}%)로 "
                     f"가장 높은 판매 실적을 기록했습니다.")

        for region, sales in regional_sales.items():
            pct = (sales / total_sales) * 100
            report.append(f"  • {region}: ${sales:,.2f} ({pct:.1f}%)")

        # 제품 카테고리
        report.append("\n\n📦 제품 카테고리 분석")
        category_sales = self.df.groupby('category')['sales'].sum().sort_values(ascending=False)

        for i, (category, sales) in enumerate(category_sales.head(3).items(), 1):
            pct = (sales / total_sales) * 100
            report.append(f"{i}. {category}: ${sales:,.2f} ({pct:.1f}%)")

        # 인사이트 및 권장 사항
        report.append("\n\n💡 주요 인사이트")

        # 성장 기회
        growth_categories = self.df.groupby('category')['sales'].sum().pct_change().sort_values(ascending=False)
        if not growth_categories.empty:
            fastest_growing = growth_categories.index[0]
            growth_rate = growth_categories.iloc[0] * 100
            report.append(f"• {fastest_growing} 카테고리가 {growth_rate:.0f}% 성장하여 "
                        f"가장 빠른 성장세를 보이고 있습니다.")

        # 주의 사항
        low_performers = regional_sales[regional_sales < regional_sales.mean() * 0.5]
        if not low_performers.empty:
            report.append(f"• {', '.join(low_performers.index)} 지역은 평균 이하의 "
                        f"성과를 보이고 있어 개선이 필요합니다.")

        # 권장 사항
        report.append("\n\n🎯 권장 조치")
        report.append(f"1. {top_region} 지역의 성공 전략을 다른 지역에 적용")
        report.append(f"2. 저성과 지역에 대한 타겟 마케팅 캠페인 실행")
        report.append(f"3. 성장하는 {fastest_growing} 카테고리에 재고 투자 확대")

        return "\n".join(report)
