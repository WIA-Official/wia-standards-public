Chapter 6: API Integration and Federation

Modern organizations rarely operate in isolation. Access control systems must integrate with diverse applications, support single sign-on across platforms, and federate identities across organizational boundaries. This chapter explores the integration capabilities of the WIA-ACCESS_CONTROL standard, covering API security, single sign-on, and identity federation.

WIA-ACCESS_CONTROL API

OAuth 2.0 Integration

The WIA-ACCESS_CONTROL standard provides comprehensive OAuth 2.0 support for delegated authorization. This enables third-party applications to access protected resources on behalf of users without exposing credentials.

OAuth 2.0 Flows Supported:
  • Authorization Code + PKCE: For web and mobile applications
  • Client Credentials: For service-to-service authentication
  • Device Authorization: For devices with limited input capability
  • Token Exchange: For impersonation and delegation scenarios

Authorization Code Flow with PKCE

// OAuth 2.0 Authorization Code Flow with PKCE
class OAuth2Client {
    private config: OAuth2Config;

    async initiateAuthorizationFlow(): Promise<AuthorizationUrl> {
        // Generate PKCE code verifier and challenge
        const codeVerifier = this.generateCodeVerifier();
        const codeChallenge = await this.generateCodeChallenge(codeVerifier);

        // Generate state for CSRF protection
        const state = crypto.randomBytes(32).toString('base64url');

        // Store verifier and state for validation
        await this.sessionStore.save({
            state: state,
            codeVerifier: codeVerifier,
            expiresAt: Date.now() + 10 * 60 * 1000 // 10 minutes
        });

        // Construct authorization URL
        const params = new URLSearchParams({
            response_type: 'code',
            client_id: this.config.clientId,
            redirect_uri: this.config.redirectUri,
            scope: this.config.scopes.join(' '),
            state: state,
            code_challenge: codeChallenge,
            code_challenge_method: 'S256'
        });

        return {
            url: `${this.config.authorizationEndpoint}?${params}`,
            state: state
        };
    }

    async handleCallback(code: string, state: string): Promise<TokenResponse> {
        // Validate state
        const session = await this.sessionStore.get(state);
        if (!session || session.expiresAt < Date.now()) {
            throw new OAuth2Error('invalid_state', 'State mismatch or expired');
        }

        // Exchange code for tokens
        const response = await fetch(this.config.tokenEndpoint, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded'
            },
            body: new URLSearchParams({
                grant_type: 'authorization_code',
                client_id: this.config.clientId,
                client_secret: this.config.clientSecret,
                code: code,
                redirect_uri: this.config.redirectUri,
                code_verifier: session.codeVerifier
            })
        });

        if (!response.ok) {
            const error = await response.json();
            throw new OAuth2Error(error.error, error.error_description);
        }

        const tokens = await response.json();

        // Clear session
        await this.sessionStore.delete(state);

        return tokens;
    }

    private generateCodeVerifier(): string {
        return crypto.randomBytes(32).toString('base64url');
    }

    private async generateCodeChallenge(verifier: string): Promise<string> {
        const hash = crypto.createHash('sha256').update(verifier).digest();
        return hash.toString('base64url');
    }
}

Single Sign-On (SSO)

Single sign-on allows users to authenticate once and access multiple applications without re-entering credentials. The WIA-ACCESS_CONTROL standard supports SSO through OpenID Connect (OIDC) and SAML 2.0.

OpenID Connect Implementation

// OpenID Connect Provider
class OIDCProvider {
    private config: OIDCConfig;
    private keyStore: JWKStore;

    // Discovery endpoint (/.well-known/openid-configuration)
    getDiscoveryDocument(): OIDCDiscoveryDocument {
        return {
            issuer: this.config.issuer,
            authorization_endpoint: `${this.config.baseUrl}/authorize`,
            token_endpoint: `${this.config.baseUrl}/token`,
            userinfo_endpoint: `${this.config.baseUrl}/userinfo`,
            jwks_uri: `${this.config.baseUrl}/.well-known/jwks.json`,
            registration_endpoint: `${this.config.baseUrl}/register`,
            scopes_supported: ['openid', 'profile', 'email', 'roles'],
            response_types_supported: ['code', 'token', 'id_token', 'code token', 'code id_token'],
            grant_types_supported: ['authorization_code', 'refresh_token', 'client_credentials'],
            subject_types_supported: ['public', 'pairwise'],
            id_token_signing_alg_values_supported: ['RS256', 'ES256', 'EdDSA'],
            claims_supported: [
                'sub', 'iss', 'aud', 'exp', 'iat', 'auth_time',
                'name', 'email', 'email_verified', 'roles', 'groups'
            ]
        };
    }

    async generateIdToken(user: User, client: Client, nonce?: string): Promise<string> {
        const now = Math.floor(Date.now() / 1000);

        const claims = {
            // Standard claims
            iss: this.config.issuer,
            sub: this.generateSubject(user, client),
            aud: client.clientId,
            exp: now + 3600,
            iat: now,
            auth_time: user.authTime,

            // Optional claims
            nonce: nonce,
            at_hash: this.generateAtHash(user.accessToken),

            // User claims (based on scopes)
            name: user.name,
            email: user.email,
            email_verified: user.emailVerified,
            roles: user.roles,
            groups: user.groups
        };

        // Sign with current key
        const key = await this.keyStore.getCurrentSigningKey();
        return JWT.sign(claims, key.privateKey, {
            algorithm: key.algorithm,
            keyid: key.kid
        });
    }

    async validateIdToken(token: string, expectedNonce?: string): Promise<IdTokenClaims> {
        // Get public keys
        const jwks = await this.keyStore.getPublicKeys();

        // Verify and decode token
        const decoded = await JWT.verify(token, jwks, {
            issuer: this.config.issuer,
            algorithms: ['RS256', 'ES256', 'EdDSA']
        });

        // Validate nonce if provided
        if (expectedNonce && decoded.nonce !== expectedNonce) {
            throw new OIDCError('invalid_nonce', 'Nonce mismatch');
        }

        return decoded as IdTokenClaims;
    }
}

SAML 2.0 Federation

// SAML 2.0 Service Provider
class SAMLServiceProvider {
    private config: SAMLConfig;
    private metadataCache: Map<string, IdPMetadata>;

    generateAuthnRequest(idpEntityId: string): SAMLAuthnRequest {
        const idpMetadata = this.metadataCache.get(idpEntityId);
        if (!idpMetadata) {
            throw new SAMLError('Unknown identity provider');
        }

        const request = {
            ID: `_${crypto.randomUUID()}`,
            Version: '2.0',
            IssueInstant: new Date().toISOString(),
            Destination: idpMetadata.ssoUrl,
            AssertionConsumerServiceURL: this.config.acsUrl,
            ProtocolBinding: 'urn:oasis:names:tc:SAML:2.0:bindings:HTTP-POST',
            Issuer: this.config.entityId,
            NameIDPolicy: {
                Format: 'urn:oasis:names:tc:SAML:2.0:nameid-format:persistent',
                AllowCreate: true
            },
            RequestedAuthnContext: {
                Comparison: 'minimum',
                AuthnContextClassRef: [
                    'urn:oasis:names:tc:SAML:2.0:ac:classes:PasswordProtectedTransport'
                ]
            }
        };

        return {
            xml: this.buildXML(request),
            relayState: crypto.randomBytes(16).toString('hex'),
            destination: idpMetadata.ssoUrl
        };
    }

    async processResponse(samlResponse: string, relayState: string): Promise<SAMLAssertion> {
        // Decode and parse response
        const xml = Buffer.from(samlResponse, 'base64').toString('utf8');
        const response = this.parseXML(xml);

        // Validate response
        await this.validateResponse(response);

        // Validate signature
        const idpMetadata = this.getIdPFromIssuer(response.Issuer);
        await this.validateSignature(xml, idpMetadata.signingCertificate);

        // Extract and validate assertion
        const assertion = response.Assertion;
        await this.validateAssertion(assertion);

        // Extract user attributes
        return {
            nameId: assertion.Subject.NameID,
            sessionIndex: assertion.AuthnStatement.SessionIndex,
            attributes: this.extractAttributes(assertion.AttributeStatement),
            validUntil: new Date(assertion.Conditions.NotOnOrAfter)
        };
    }
}

Identity Federation

Identity federation enables organizations to trust identities from external identity providers. This is essential for B2B scenarios, partnerships, and multi-organization deployments.

Federation Benefits: Identity federation eliminates the need for users to maintain separate credentials for each organization. It enables seamless collaboration while maintaining security boundaries and audit trails. 
// Identity Federation Manager
class FederationManager {
    private trustStore: TrustStore;
    private claimsMappingEngine: ClaimsMappingEngine;

    async processFederatedIdentity(
        assertion: FederatedAssertion,
        sourceIdP: string
    ): Promise<LocalIdentity> {
        // Verify trust relationship
        const trustConfig = await this.trustStore.getTrustConfig(sourceIdP);
        if (!trustConfig || !trustConfig.active) {
            throw new FederationError('Untrusted identity provider');
        }

        // Validate assertion
        await this.validateFederatedAssertion(assertion, trustConfig);

        // Map claims to local format
        const mappedClaims = await this.claimsMappingEngine.map(
            assertion.claims,
            trustConfig.claimsMapping
        );

        // Provision or link local identity
        let localUser = await this.findLinkedUser(assertion.subject, sourceIdP);

        if (!localUser) {
            if (trustConfig.allowProvisioning) {
                localUser = await this.provisionUser(mappedClaims, sourceIdP);
            } else {
                throw new FederationError('User not found and provisioning disabled');
            }
        }

        // Update user attributes if configured
        if (trustConfig.syncAttributes) {
            await this.syncUserAttributes(localUser, mappedClaims);
        }

        // Create local session
        return {
            userId: localUser.id,
            federatedFrom: sourceIdP,
            originalSubject: assertion.subject,
            claims: mappedClaims,
            sessionExpiry: assertion.validUntil
        };
    }

    async establishTrust(idpMetadata: IdPMetadata, config: TrustConfig): Promise<void> {
        // Validate IdP metadata
        await this.validateIdPMetadata(idpMetadata);

        // Store trust configuration
        await this.trustStore.saveTrustConfig({
            entityId: idpMetadata.entityId,
            displayName: config.displayName,
            metadata: idpMetadata,
            claimsMapping: config.claimsMapping,
            allowProvisioning: config.allowProvisioning,
            syncAttributes: config.syncAttributes,
            roleMapping: config.roleMapping,
            active: true,
            createdAt: new Date()
        });

        // Audit trust establishment
        await this.auditLog.log({
            event: 'federation.trust.established',
            idpEntityId: idpMetadata.entityId,
            config: config
        });
    }
}

API Gateway Integration

API gateways serve as centralized enforcement points for access control. The WIA-ACCESS_CONTROL standard provides integration patterns for popular API gateway platforms.

// API Gateway Policy Enforcement
const gatewayPolicy = {
    // Pre-request authorization
    preRequest: async (request: GatewayRequest): Promise<GatewayDecision> => {
        // Extract authentication token
        const token = extractToken(request.headers.authorization);
        if (!token) {
            return { allow: false, statusCode: 401, message: 'Missing authentication' };
        }

        // Validate token
        const tokenInfo = await accessControl.validateToken(token);
        if (!tokenInfo.valid) {
            return { allow: false, statusCode: 401, message: 'Invalid token' };
        }

        // Check authorization
        const decision = await accessControl.checkAccess({
            subject: { id: tokenInfo.subject, attributes: tokenInfo.claims },
            resource: { path: request.path, method: request.method },
            action: methodToAction(request.method),
            context: {
                ip: request.clientIp,
                headers: request.headers
            }
        });

        if (decision.effect !== 'Permit') {
            return { allow: false, statusCode: 403, message: decision.reason };
        }

        // Add identity headers for downstream services
        return {
            allow: true,
            headers: {
                'X-User-ID': tokenInfo.subject,
                'X-User-Roles': tokenInfo.claims.roles.join(','),
                'X-Request-ID': generateRequestId()
            }
        };
    },

    // Post-response processing
    postResponse: async (request: GatewayRequest, response: GatewayResponse): Promise<void> => {
        // Log access for audit
        await auditLog.log({
            type: 'api.access',
            path: request.path,
            method: request.method,
            subject: request.headers['X-User-ID'],
            statusCode: response.statusCode,
            latency: response.latency
        });
    }
};

Summary

This chapter has explored the integration and federation capabilities of the WIA-ACCESS_CONTROL standard. From the comprehensive REST API for programmatic access control to OAuth 2.0 and OpenID Connect for modern authentication, and SAML 2.0 for enterprise federation, the standard provides the tools needed to integrate access control into any environment.

Key Takeaways:
  • The WIA-ACCESS_CONTROL API enables programmatic access control operations
  • OAuth 2.0 provides delegated authorization for third-party applications
  • OpenID Connect and SAML 2.0 enable single sign-on across platforms
  • Identity federation allows trust relationships across organizations
  • API gateway integration provides centralized policy enforcement
