Chapter 3: Livestreaming Protocols

Standards for Funeral Service Broadcasting

3.1 Livestreaming Protocol Overview

Funeral service livestreaming requires specialized protocols that balance technical excellence with emotional sensitivity. This chapter establishes comprehensive standards for broadcasting funeral and memorial services with professional quality, reliability, and appropriate privacy controls.

The WIA Livestreaming Protocol addresses video/audio encoding, multi-platform distribution, content delivery networks (CDN), privacy protection, recording management, and quality assurance. These systems must operate flawlessly during emotionally sensitive moments when technical failures would be particularly devastating to grieving families.

Critical Requirements

  • Reliability: 99.9% uptime with redundant streaming infrastructure
  • Quality: HD video (1080p minimum) with professional audio
  • Privacy: Encrypted streams with access control and watermarking
  • Multi-platform: Simultaneous distribution to web, mobile, social media
  • Recording: Automatic backup recording with long-term archival
  • Monitoring: Real-time health checks and instant failover capability

3.5 Implementation Examples

Example 1: Creating Livestream (Node.js)

const LivestreamManager = require('./livestream-manager');

class FuneralLivestream {
  constructor(apiKey) {
    this.apiKey = apiKey;
    this.baseUrl = 'https://api.streaming.funeral-services.org';
  }

  async createStream(serviceData) {
    const streamConfig = {
      service_id: serviceData.serviceId,
      title: `Memorial Service for ${serviceData.deceasedName}`,
      streaming_config: {
        video_codec: 'h264',
        audio_codec: 'aac',
        max_resolution: '1080p',
        adaptive_bitrate: true,
        enable_recording: true,
        dvr_window: 14400 // 4 hours
      },
      privacy_settings: {
        access_type: serviceData.isPrivate ? 'password_protected' : 'public',
        password: serviceData.streamPassword,
        watermark_enabled: true
      }
    };

    const response = await fetch(`${this.baseUrl}/api/v1/livestream/create`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(streamConfig)
    });

    return await response.json();
  }

  async startStream(streamId) {
    const response = await fetch(
      `${this.baseUrl}/api/v1/livestream/${streamId}/start`,
      {
        method: 'POST',
        headers: {'Authorization': `Bearer ${this.apiKey}`}
      }
    );

    const result = await response.json();

    // Monitor stream health
    this.monitorStreamHealth(streamId);

    return result;
  }

  monitorStreamHealth(streamId) {
    const healthCheck = setInterval(async () => {
      const health = await this.getStreamHealth(streamId);

      if (!health.is_healthy) {
        console.error('Stream health issue detected:', health);
        await this.triggerFailover(streamId);
      }

      if (health.dropped_frames > 100) {
        console.warn('High frame drops detected');
      }
    }, 5000);

    return healthCheck;
  }

  async getStreamHealth(streamId) {
    const response = await fetch(
      `${this.baseUrl}/api/v1/livestream/${streamId}/health`,
      {headers: {'Authorization': `Bearer ${this.apiKey}`}}
    );
    return await response.json();
  }
}

// Usage
const livestream = new FuneralLivestream('api-key');
const stream = await livestream.createStream({
  serviceId: 'service-123',
  deceasedName: 'John Doe',
  isPrivate: true,
  streamPassword: 'family2024'
});

console.log('Stream created:', stream.urls.hls_playback);

Example 2: Viewer Client (JavaScript)

class LivestreamViewer {
  constructor(streamId, accessToken) {
    this.streamId = streamId;
    this.accessToken = accessToken;
    this.player = null;
  }

  async initialize(videoElement) {
    // Get playback URL
    const playbackUrl = await this.getPlaybackUrl();

    // Initialize HLS player
    if (Hls.isSupported()) {
      this.player = new Hls({
        enableWorker: true,
        lowLatencyMode: true,
        backBufferLength: 90
      });

      this.player.loadSource(playbackUrl);
      this.player.attachMedia(videoElement);

      this.player.on(Hls.Events.MANIFEST_PARSED, () => {
        videoElement.play();
      });

      this.player.on(Hls.Events.ERROR, (event, data) => {
        if (data.fatal) {
          this.handleStreamError(data);
        }
      });
    }

    // Track viewing analytics
    this.trackViewingSession();
  }

  async getPlaybackUrl() {
    const response = await fetch(
      `https://api.streaming.funeral-services.org/api/v1/livestream/${this.streamId}/playback-url`,
      {
        headers: {'Authorization': `Bearer ${this.accessToken}`}
      }
    );
    const data = await response.json();
    return data.hls_playback;
  }

  trackViewingSession() {
    const sessionStart = Date.now();
    let totalWatchTime = 0;

    setInterval(() => {
      if (!this.player.media.paused) {
        totalWatchTime += 30;

        // Send heartbeat
        fetch(
          `https://api.streaming.funeral-services.org/api/v1/livestream/${this.streamId}/analytics/heartbeat`,
          {
            method: 'POST',
            headers: {
              'Authorization': `Bearer ${this.accessToken}`,
              'Content-Type': 'application/json'
            },
            body: JSON.stringify({
              watch_time: totalWatchTime,
              quality_level: this.player.currentLevel
            })
          }
        );
      }
    }, 30000);
  }
}

// Usage
const viewer = new LivestreamViewer('stream-123', 'access-token');
const videoEl = document.getElementById('funeral-stream');
await viewer.initialize(videoEl);

Example 3: Stream Recording Manager (Python)

import requests
from datetime import datetime, timedelta

class StreamRecordingManager:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://api.streaming.funeral-services.org'

    def enable_recording(self, stream_id: str, config: dict):
        """Enable automatic recording for stream"""
        recording_config = {
            'enabled': True,
            'format': config.get('format', 'mp4'),
            'quality': config.get('quality', '1080p'),
            'auto_archive': True,
            'retention_days': config.get('retention_days', 3650)  # 10 years
        }

        response = requests.post(
            f'{self.base_url}/api/v1/livestream/{stream_id}/record',
            headers={'Authorization': f'Bearer {self.api_key}'},
            json=recording_config
        )
        return response.json()

    def get_recording(self, stream_id: str):
        """Retrieve recorded stream"""
        response = requests.get(
            f'{self.base_url}/api/v1/livestream/{stream_id}/recording',
            headers={'Authorization': f'Bearer {self.api_key}'}
        )
        return response.json()

    def download_recording(self, stream_id: str, output_path: str):
        """Download recording file"""
        recording = self.get_recording(stream_id)

        if recording.get('download_url'):
            response = requests.get(recording['download_url'], stream=True)

            with open(output_path, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    f.write(chunk)

            print(f'Recording downloaded to {output_path}')
        else:
            print('Recording not available yet')

# Usage
recorder = StreamRecordingManager('api-key')
recorder.enable_recording('stream-123', {'quality': '1080p', 'retention_days': 3650})

3.7 Best Practices

  • Test stream setup 24 hours before service with full technical rehearsal
  • Use wired internet connection for primary encoder (min 10 Mbps upload)
  • Configure backup encoder on separate network for redundancy
  • Position cameras to capture ceremony while respecting family privacy
  • Use professional microphones for clear audio in large venues
  • Enable automatic recording as backup even for live-only services
  • Provide viewers with instructions 48 hours before service
  • Assign technical operator to monitor stream throughout service
