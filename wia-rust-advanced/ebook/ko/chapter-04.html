🔬 Rust 컴파일러 내부

Rust 코드가 실행 파일이 되기까지 여러 단계를 거칩니다. 각 단계를 이해하면 최적화와 디버깅에 큰 도움이 됩니다.

컴파일 파이프라인

1. Lexing & Parsing

소스 코드를 토큰으로 분해하고 AST(추상 구문 트리)로 변환

2. HIR (High-level IR)

타입 체킹과 트레잇 해결을 위한 고수준 중간 표현

3. MIR (Mid-level IR)

차용 검사와 최적화를 위한 중간 수준 표현

4. LLVM IR

LLVM 백엔드로 전달되는 저수준 표현

5. 머신 코드

타겟 아키텍처의 네이티브 코드 생성

MIR 살펴보기

fn add(a: i32, b: i32) -> i32 {
    a + b
}

// MIR 보기
// rustc --emit mir source.rs

// 생성된 MIR (간소화)
fn add(_1: i32, _2: i32) -> i32 {
    let mut _0: i32;

    bb0: {
        _0 = Add(move _1, move _2);
        return;
    }
}

LLVM IR 확인하기

$ rustc --emit llvm-ir source.rs
$ cat source.ll

; 생성된 LLVM IR
define i32 @add(i32 %a, i32 %b) {
start:
  %0 = add i32 %a, %b
  ret i32 %0
}

🔧 no_std 환경

표준 라이브러리 없이 Rust를 사용하는 환경입니다. 임베디드 시스템이나 운영체제 개발에 필수적입니다.

no_std 프로젝트 시작

#![no_std]
#![no_main]

use core::panic::PanicInfo;

#[panic_handler]
fn panic(_info: &PanicInfo) -> ! {
    loop {}
}

#[no_mangle]
pub extern "C" fn _start() -> ! {
    loop {}
}

core vs std

💡 core 라이브러리
  • 힙 할당 없음 (Box, Vec, String 사용 불가)
  • I/O 기능 없음
  • 스레딩 없음
  • 대신 사용 가능: Option, Result, Iterator, 기본 타입들

alloc 크레이트 사용

#![no_std]

extern crate alloc;

use alloc::vec::Vec;
use alloc::string::String;

#[global_allocator]
static ALLOCATOR: MyAllocator = MyAllocator;

struct MyAllocator;

unsafe impl GlobalAlloc for MyAllocator {
    // 할당자 구현
}

🤖 ARM Cortex-M 개발

가장 인기있는 임베디드 프로세서인 ARM Cortex-M에서 Rust를 사용해봅시다.

프로젝트 설정

# .cargo/config.toml
[build]
target = "thumbv7em-none-eabihf"  # Cortex-M4F

[target.thumbv7em-none-eabihf]
runner = "probe-run --chip STM32F407VGTx"
rustflags = [
  "-C", "link-arg=-Tlink.x",
]

# Cargo.toml
[dependencies]
cortex-m = "0.7"
cortex-m-rt = "0.7"
embedded-hal = "0.2"
panic-halt = "0.2"

LED 깜박이기 (Blinky)

#![no_std]
#![no_main]

use cortex_m_rt::entry;
use panic_halt as _;
use stm32f4xx_hal::{pac, prelude::*};

#[entry]
fn main() -> ! {
    // 주변장치 초기화
    let dp = pac::Peripherals::take().unwrap();
    let gpioa = dp.GPIOA.split();

    // LED 핀 설정
    let mut led = gpioa.pa5.into_push_pull_output();

    loop {
        led.set_high();
        cortex_m::asm::delay(8_000_000);

        led.set_low();
        cortex_m::asm::delay(8_000_000);
    }
}

인터럽트 처리

use cortex_m::interrupt::Mutex;
use core::cell::RefCell;

static COUNTER: Mutexu32>> =
    Mutex::new(RefCell::new(0));

#[interrupt]
fn TIM2() {
    cortex_m::interrupt::free(|cs| {
        let mut counter = COUNTER.borrow(cs).borrow_mut();
        *counter += 1;
    });
}

#[entry]
fn main() -> ! {
    // 타이머 설정
    let dp = pac::Peripherals::take().unwrap();
    let mut timer = dp.TIM2.timer();

    // 인터럽트 활성화
    timer.listen(Event::Update);
    unsafe {
        cortex_m::peripheral::NVIC::unmask(Interrupt::TIM2);
    }

    loop {
        cortex_m::asm::wfi(); // Wait For Interrupt
    }
}

🎛️ 메모리 맵 I/O

하드웨어 레지스터에 직접 접근하여 제어합니다. 

use core::ptr;

// GPIO 포트 A 주소
const GPIOA_BASE: u32 = 0x4002_0000;
const GPIOA_MODER: u32 = GPIOA_BASE + 0x00;
const GPIOA_ODR: u32 = GPIOA_BASE + 0x14;

unsafe fn init_led() {
    // PA5를 출력 모드로 설정
    let moder = GPIOA_MODER as *mut u32;
    ptr::write_volatile(moder, 0x0000_0400);
}

unsafe fn set_led(on: bool) {
    let odr = GPIOA_ODR as *mut u32;
    if on {
        ptr::write_volatile(odr, 0x0020);
    } else {
        ptr::write_volatile(odr, 0x0000);
    }
}

안전한 레지스터 접근

use volatile_register::{RW, RO};

#[repr(C)]
struct GpioRegisters {
    moder: RW<u32>,
    otyper: RW<u32>,
    ospeedr: RW<u32>,
    pupdr: RW<u32>,
    idr: RO<u32>,
    odr: RW<u32>,
}

fn control_gpio() {
    let gpio = unsafe { &(*(0x4002_0000 as *const GpioRegisters)) };

    unsafe {
        gpio.moder.write(0x0000_0400);
        gpio.odr.write(0x0020);
    }
}

🚀 실전 프로젝트: UART 통신

use embedded_hal::serial::Write;

#[entry]
fn main() -> ! {
    let dp = pac::Peripherals::take().unwrap();
    let rcc = dp.RCC.constrain();
    let clocks = rcc.cfgr.freeze();

    let gpioa = dp.GPIOA.split();
    let tx = gpioa.pa2.into_alternate();
    let rx = gpioa.pa3.into_alternate();

    let mut serial = dp.USART2
        .serial((tx, rx), 115200.bps(), &clocks)
        .unwrap();

    loop {
        for byte in b"Hello from Rust!\r\n" {
            block!(serial.write(*byte)).unwrap();
        }

        cortex_m::asm::delay(8_000_000);
    }
}
