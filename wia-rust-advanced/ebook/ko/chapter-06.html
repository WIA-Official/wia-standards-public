🧠 메모리 할당자

메모리 할당자는 프로그램의 성능에 큰 영향을 미칩니다. Rust에서는 GlobalAlloc 트레잇을 구현하여 커스텀 할당자를 만들 수 있습니다.

GlobalAlloc 트레잇

use core::alloc::{GlobalAlloc, Layout};

pub struct MyAllocator;

unsafe impl GlobalAlloc for MyAllocator {
    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
        // 메모리 할당 구현
    }

    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
        // 메모리 해제 구현
    }
}

#[global_allocator]
static ALLOCATOR: MyAllocator = MyAllocator;

Bump Allocator

가장 단순한 할당자입니다. 포인터를 앞으로만 이동시키며 개별 해제를 지원하지 않습니다. 

use core::alloc::{GlobalAlloc, Layout};
use core::ptr;

pub struct BumpAllocator {
    heap_start: usize,
    heap_end: usize,
    next: AtomicUsize,
    allocations: AtomicUsize,
}

impl BumpAllocator {
    pub const fn new() -> Self {
        BumpAllocator {
            heap_start: 0,
            heap_end: 0,
            next: AtomicUsize::new(0),
            allocations: AtomicUsize::new(0),
        }
    }

    pub unsafe fn init(&mut self, heap_start: usize, heap_size: usize) {
        self.heap_start = heap_start;
        self.heap_end = heap_start + heap_size;
        self.next.store(heap_start, Ordering::Relaxed);
    }
}

unsafe impl GlobalAlloc for BumpAllocator {
    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
        let alloc_start = align_up(
            self.next.load(Ordering::Relaxed),
            layout.align()
        );
        let alloc_end = alloc_start + layout.size();

        if alloc_end > self.heap_end {
            ptr::null_mut() // 메모리 부족
        } else {
            self.next.store(alloc_end, Ordering::Relaxed);
            self.allocations.fetch_add(1, Ordering::Relaxed);
            alloc_start as *mut u8
        }
    }

    unsafe fn dealloc(&self, _ptr: *mut u8, _layout: Layout) {
        let allocs = self.allocations.fetch_sub(1, Ordering::Relaxed) - 1;
        if allocs == 0 {
            // 모든 메모리가 해제됨 - 리셋 가능
            self.next.store(self.heap_start, Ordering::Relaxed);
        }
    }
}

Linked List Allocator

해제된 메모리를 연결 리스트로 관리하는 할당자입니다. 

struct ListNode {
    size: usize,
    next: Option<&'static mut ListNode>,
}

impl ListNode {
    const fn new(size: usize) -> Self {
        ListNode { size, next: None }
    }

    fn start_addr(&self) -> usize {
        self as *const Self as usize
    }

    fn end_addr(&self) -> usize {
        self.start_addr() + self.size
    }
}

pub struct LinkedListAllocator {
    head: ListNode,
}

impl LinkedListAllocator {
    pub const fn new() -> Self {
        LinkedListAllocator {
            head: ListNode::new(0),
        }
    }

    pub unsafe fn init(&mut self, heap_start: usize, heap_size: usize) {
        self.add_free_region(heap_start, heap_size);
    }

    unsafe fn add_free_region(&mut self, addr: usize, size: usize) {
        let mut node = ListNode::new(size);
        node.next = self.head.next.take();
        let node_ptr = addr as *mut ListNode;
        node_ptr.write(node);
        self.head.next = Some(&mut *node_ptr);
    }
}

메모리 풀 할당자

고정 크기 블록을 효율적으로 관리하는 할당자입니다. 

pub struct PoolAllocator {
    block_size: usize,
    block_count: usize,
    free_list: Option<*mut u8>,
}

impl PoolAllocator {
    pub unsafe fn new(
        heap_start: *mut u8,
        block_size: usize,
        block_count: usize
    ) -> Self {
        let mut allocator = PoolAllocator {
            block_size,
            block_count,
            free_list: None,
        };

        // 프리 리스트 초기화
        for i in 0..block_count {
            let block = heap_start.add(i * block_size);
            allocator.free_block(block);
        }

        allocator
    }

    unsafe fn free_block(&mut self, block: *mut u8) {
        *(block as *mut Option<*mut u8>) = self.free_list;
        self.free_list = Some(block);
    }

    unsafe fn alloc_block(&mut self) -> Option<*mut u8> {
        self.free_list.map(|block| {
            self.free_list = *(block as *const Option<*mut u8>);
            block
        })
    }
}

🚀 SIMD 프로그래밍

SIMD (Single Instruction Multiple Data)는 하나의 명령으로 여러 데이터를 동시에 처리하는 기술입니다. 벡터 연산에 특히 유용합니다.

기본 SIMD 예제

use std::arch::x86_64::*;

#[target_feature(enable = "avx2")]
unsafe fn add_vectors_simd(a: &[f32], b: &[f32], result: &mut [f32]) {
    assert!(a.len() == b.len() && b.len() == result.len());
    assert!(a.len() % 8 == 0); // AVX2는 8개 float 처리

    for i in (0..a.len()).step_by(8) {
        let va = _mm256_loadu_ps(a.as_ptr().add(i));
        let vb = _mm256_loadu_ps(b.as_ptr().add(i));
        let vr = _mm256_add_ps(va, vb);
        _mm256_storeu_ps(result.as_mut_ptr().add(i), vr);
    }
}

portable_simd 사용

플랫폼 독립적인 SIMD 코드를 작성할 수 있습니다. 

#![feature(portable_simd)]

use std::simd::*;

fn add_vectors(a: &[f32], b: &[f32]) -> Vec<f32> {
    a.chunks_exact(4)
        .zip(b.chunks_exact(4))
        .flat_map(|(a_chunk, b_chunk)| {
            let a_simd = f32x4::from_slice(a_chunk);
            let b_simd = f32x4::from_slice(b_chunk);
            (a_simd + b_simd).to_array()
        })
        .collect()
}

행렬 곱셈 최적화

#[target_feature(enable = "avx2")]
unsafe fn matrix_multiply_simd(
    a: &[f32],
    b: &[f32],
    result: &mut [f32],
    n: usize
) {
    for i in 0..n {
        for j in (0..n).step_by(8) {
            let mut sum = _mm256_setzero_ps();

            for k in 0..n {
                let a_val = _mm256_set1_ps(a[i * n + k]);
                let b_val = _mm256_loadu_ps(b.as_ptr().add(k * n + j));
                sum = _mm256_fmadd_ps(a_val, b_val, sum);
            }

            _mm256_storeu_ps(result.as_mut_ptr().add(i * n + j), sum);
        }
    }
}

🔧 인라인 어셈블리

극한의 성능이 필요할 때는 직접 어셈블리를 작성할 수 있습니다. 

use std::arch::asm;

fn add_numbers(a: u64, b: u64) -> u64 {
    let result: u64;
    unsafe {
        asm!(
            "add {0}, {1}",
            inout(reg) a => result,
            in(reg) b,
        );
    }
    result
}

// CPUID 읽기
fn cpuid(leaf: u32) -> [u32; 4] {
    let mut eax: u32;
    let mut ebx: u32;
    let mut ecx: u32;
    let mut edx: u32;

    unsafe {
        asm!(
            "cpuid",
            inout("eax") leaf => eax,
            out("ebx") ebx,
            out("ecx") ecx,
            out("edx") edx,
        );
    }

    [eax, ebx, ecx, edx]
}

📊 성능 측정

벤치마크

use criterion::{black_box, criterion_group, criterion_main, Criterion};

fn benchmark_add(c: &mut Criterion) {
    let a = vec![1.0f32; 1024];
    let b = vec![2.0f32; 1024];

    c.bench_function("scalar add", |bencher| {
        bencher.iter(|| {
            let mut result = vec![0.0f32; 1024];
            for i in 0..1024 {
                result[i] = black_box(a[i]) + black_box(b[i]);
            }
            result
        });
    });

    c.bench_function("simd add", |bencher| {
        bencher.iter(|| unsafe {
            let mut result = vec![0.0f32; 1024];
            add_vectors_simd(
                black_box(&a),
                black_box(&b),
                &mut result
            );
            result
        });
    });
}

criterion_group!(benches, benchmark_add);
criterion_main!(benches);
