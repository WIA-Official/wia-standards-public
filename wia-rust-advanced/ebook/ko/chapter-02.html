🎯 FFI란 무엇인가?

FFI (Foreign Function Interface)는 서로 다른 프로그래밍 언어로 작성된 코드가 상호작용할 수 있게 해주는 인터페이스입니다. Rust에서는 주로 C/C++ 라이브러리와 통신할 때 사용됩니다.

💡 왜 FFI가 필요한가?
  • 기존 라이브러리 활용: OpenSSL, SQLite 등 검증된 C 라이브러리 사용
  • 시스템 API 접근: 운영체제의 저수준 기능 활용
  • 성능 최적화: 특정 부분을 C/어셈블리로 최적화
  • 상호운용성: 다른 언어로 작성된 프로그램과 통합

🚀 C 함수 호출하기

기본 extern 블록

C 함수를 호출하려면 extern 블록에서 함수 시그니처를 선언합니다. 

// Rust 코드
extern "C" {
    fn abs(input: i32) -> i32;
}

fn main() {
    unsafe {
        println!("Absolute value of -3: {}", abs(-3));
    }
}

표준 C 라이브러리 함수 사용

use std::os::raw::c_char;

extern "C" {
    fn strlen(s: *const c_char) -> usize;
    fn printf(format: *const c_char, ...) -> i32;
}

fn main() {
    let c_string = b"Hello, FFI!\0";

    unsafe {
        let len = strlen(c_string.as_ptr() as *const c_char);
        println!("Length: {}", len);

        printf(b"C printf: %s\n\0".as_ptr() as *const c_char,
               c_string.as_ptr());
    }
}

📦 외부 라이브러리 링크하기

build.rs를 통한 링크

복잡한 C 라이브러리는 build.rs를 사용하여 링크합니다. 

// build.rs
fn main() {
    println!("cargo:rustc-link-lib=mylib");
    println!("cargo:rustc-link-search=/usr/local/lib");
}

// src/main.rs
extern "C" {
    fn my_c_function(x: i32) -> i32;
}

fn main() {
    unsafe {
        let result = my_c_function(42);
        println!("Result: {}", result);
    }
}

🔄 Rust에서 C로: 함수 내보내기

#[no_mangle] 속성

Rust 함수를 C에서 호출하려면 name mangling을 비활성화해야 합니다. 

// lib.rs
#[no_mangle]
pub extern "C" fn rust_add(a: i32, b: i32) -> i32 {
    a + b
}

#[no_mangle]
pub extern "C" fn rust_greet(name: *const c_char) {
    let c_str = unsafe { CStr::from_ptr(name) };
    let rust_str = c_str.to_str().unwrap();
    println!("Hello, {}!", rust_str);
}

/* C 코드에서 사용 */
// example.c
#include <stdio.h>

extern int rust_add(int a, int b);
extern void rust_greet(const char* name);

int main() {
    int result = rust_add(10, 20);
    printf("10 + 20 = %d\n", result);

    rust_greet("World");
    return 0;
}

동적 라이브러리 생성

# Cargo.toml
[lib]
name = "mylib"
crate-type = ["cdylib"]

# 빌드
$ cargo build --release

# C 코드 컴파일
$ gcc example.c -L./target/release -lmylib -o example
$ ./example

🗂️ 데이터 구조 매핑

repr(C) 속성

Rust 구조체를 C와 호환되는 메모리 레이아웃으로 만듭니다. 

#[repr(C)]
pub struct Point {
    pub x: f64,
    pub y: f64,
}

#[repr(C)]
pub struct RustObject {
    pub a: u32,
    pub b: u32,
    pub c: u64,
}

#[no_mangle]
pub extern "C" fn point_distance(p: Point) -> f64 {
    (p.x * p.x + p.y * p.y).sqrt()
}

타입 매핑 테이블

Rust 타입 C 타입 설명
i8, i16, i32, i64 int8_t, int16_t, int32_t, int64_t 부호있는 정수
u8, u16, u32, u64 uint8_t, uint16_t, uint32_t, uint64_t 부호없는 정수
f32, f64 float, double 부동소수점
*const T const T* 불변 포인터
*mut T T* 가변 포인터
bool bool (C99) 1바이트 불리언

📝 문자열 처리

C 문자열 ↔ Rust 문자열

use std::ffi::{CStr, CString};
use std::os::raw::c_char;

// C 문자열을 Rust로
fn c_to_rust(c_str: *const c_char) -> String {
    let cstr = unsafe { CStr::from_ptr(c_str) };
    cstr.to_string_lossy().into_owned()
}

// Rust 문자열을 C로
fn rust_to_c(rust_str: &str) -> *mut c_char {
    let c_string = CString::new(rust_str).unwrap();
    c_string.into_raw()
}

// C 문자열 해제
#[no_mangle]
pub extern "C" fn free_rust_string(s: *mut c_char) {
    if !s.is_null() {
        unsafe {
            let _ = CString::from_raw(s);
        }
    }
}
⚠️ 메모리 관리 주의
  • into_raw()로 생성한 포인터는 반드시 from_raw()로 해제해야 합니다
  • C에서 할당한 메모리는 C의 free()로, Rust에서 할당한 메모리는 Rust로 해제
  • 소유권과 수명에 각별히 주의하세요!

🎭 콜백 함수

C에서 Rust 콜백 호출

// Rust 코드
type Callback = extern "C" fn(i32) -> i32;

#[no_mangle]
pub extern "C" fn register_callback(callback: Callback) {
    let result = callback(42);
    println!("Callback returned: {}", result);
}

// C 콜백 함수 예제
extern "C" fn my_callback(value: i32) -> i32 {
    value * 2
}

fn main() {
    register_callback(my_callback);
}

Rust에서 C 콜백 호출

// C 라이브러리 함수 시그니처
extern "C" {
    fn qsort(
        base: *mut c_void,
        num: size_t,
        size: size_t,
        comparator: extern "C" fn(*const c_void, *const c_void) -> c_int,
    );
}

extern "C" fn compare_i32(a: *const c_void, b: *const c_void) -> c_int {
    unsafe {
        let a = *(a as *const i32);
        let b = *(b as *const i32);
        a - b
    }
}

fn main() {
    let mut numbers = [5, 2, 8, 1, 9];

    unsafe {
        qsort(
            numbers.as_mut_ptr() as *mut c_void,
            numbers.len() as size_t,
            std::mem::size_of::<i32>() as size_t,
            compare_i32,
        );
    }

    println!("{:?}", numbers); // [1, 2, 5, 8, 9]
}

🛠️ 실전 예제: SQLite 래퍼

SQLite C 라이브러리를 Rust에서 사용하는 간단한 래퍼를 만들어봅시다. 

use std::ffi::{CStr, CString};
use std::os::raw::{c_char, c_int, c_void};
use std::ptr;

#[repr(C)]
struct sqlite3;

extern "C" {
    fn sqlite3_open(
        filename: *const c_char,
        ppDb: *mut *mut sqlite3,
    ) -> c_int;

    fn sqlite3_close(db: *mut sqlite3) -> c_int;

    fn sqlite3_exec(
        db: *mut sqlite3,
        sql: *const c_char,
        callback: *const c_void,
        arg: *mut c_void,
        errmsg: *mut *mut c_char,
    ) -> c_int;
}

struct Database {
    db: *mut sqlite3,
}

impl Database {
    fn open(path: &str) -> Result<Self, String> {
        let c_path = CString::new(path).unwrap();
        let mut db = ptr::null_mut();

        let result = unsafe {
            sqlite3_open(c_path.as_ptr(), &mut db)
        };

        if result == 0 {
            Ok(Database { db })
        } else {
            Err("Failed to open database".to_string())
        }
    }

    fn execute(&self, sql: &str) -> Result<(), String> {
        let c_sql = CString::new(sql).unwrap();

        let result = unsafe {
            sqlite3_exec(
                self.db,
                c_sql.as_ptr(),
                ptr::null(),
                ptr::null_mut(),
                ptr::null_mut(),
            )
        };

        if result == 0 {
            Ok(())
        } else {
            Err("Failed to execute SQL".to_string())
        }
    }
}

impl Drop for Database {
    fn drop(&mut self) {
        unsafe {
            sqlite3_close(self.db);
        }
    }
}

✅ FFI 모범 사례

✅ 체크리스트
  1. repr(C) 사용: 구조체 레이아웃 명시
  2. no_mangle 추가: 함수 이름 보존
  3. 메모리 소유권 명확히: 누가 할당하고 해제하는지 문서화
  4. Null 체크: C 포인터는 항상 null 검사
  5. 패닉 방지: extern "C" 함수에서 패닉 금지
  6. 문서화: 안전성 요구사항을 명확히 기술
  7. 테스트: 경계 조건과 에러 케이스 철저히 테스트
⚠️ 피해야 할 것들
  • extern "C" 함수에서 언와인딩(unwinding) 발생시키기
  • repr(C) 없이 구조체를 C로 전달
  • 수명이 끝난 참조를 C로 전달
  • C에서 Rust의 Vec, String 등을 직접 조작
