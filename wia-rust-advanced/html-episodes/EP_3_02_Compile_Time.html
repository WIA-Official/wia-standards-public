const fn 활용 Using const fn

const fn validate_config(size: usize) -> usize {
if size == 0 {
panic!("Size cannot be zero");
}
if size > 1024 {
panic!("Size too large");
}
size
}
// 컴파일 타임에 검증됨!
const BUFFER_SIZE: usize = validate_config(512);

Validate constants at compile-time

타입 레벨 프로그래밍 Type-level Programming

// 상태를 타입으로 인코딩
struct Locked;
struct Unlocked;
struct Door<State> {
state: PhantomData<State>,
}
impl Door<Locked> {
fn unlock(self) -> Door<Unlocked> {
Door { state: PhantomData }
}
}
impl Door<Unlocked> {
fn open(&self) { /* ... */ }
}

Const Generics Const Generics

struct Buffer<const N: usize> {
data: [u8; N],
}
impl<const N: usize> Buffer<N> {
const fn validate() {
if N == 0 {
panic!("Buffer size cannot be 0");
}
if N > 4096 {
panic!("Buffer too large");
}
}
fn new() -> Self {
Self::validate();
Self { data: [0; N] }
}
}

정적 단언 (Static Assertions) Static Assertions

use static_assertions::*;
// 크기 검증
const_assert_eq!(std::mem::size_of::<MyStruct>(), 32);
// 트레이트 구현 검증
assert_impl_all!(MyType: Send, Sync, Clone);
// 타입 동등성 검증
assert_type_eq_all!(i32, impl Into<i64>);
// 필드 존재 검증
assert_fields!(MyStruct: field1, field2);
