use web_sys::{Document, Element};

pub fn run() -> Result<(), JsValue> {
let window = web_sys::window().unwrap();
let document = window.document().unwrap();

document.body().unwrap().append_child(&div)?;

}

이벤트 처리 Event Handling

use wasm_bindgen::JsCast;
#[wasm_bindgen]
pub fn setup_button() {
let document = web_sys::window().unwrap().document().unwrap();
let button = document.get_element_by_id("myButton").unwrap();
let closure = Closure::wrap(Box::new(|| {
web_sys::console::log_1(&"Button clicked!".into());
}) as Box);
button.add_event_listener_with_callback("click", closure.as_ref().unchecked_ref());
closure.forget(); // 메모리 누수 방지
}

Canvas 그래픽 Canvas Graphics

use web_sys::{CanvasRenderingContext2d, HtmlCanvasElement};
#[wasm_bindgen]
pub fn draw_circle() {
let canvas: HtmlCanvasElement = document
.get_element_by_id("canvas").unwrap()
.dyn_into().unwrap();
let ctx: CanvasRenderingContext2d = canvas
.get_context("2d").unwrap().unwrap()
.dyn_into().unwrap();
ctx.begin_path();
ctx.arc(100.0, 100.0, 50.0, 0.0, std::f64::consts::PI * 2.0);
ctx.fill();
}

구조체 내보내기 Exporting Structs

#[wasm_bindgen]
pub struct Counter {
value: i32,
}
#[wasm_bindgen]
impl Counter {
#[wasm_bindgen(constructor)]
pub fn new() -> Counter {
Counter { value: 0 }
}
pub fn increment(&mut self) {
self.value += 1;
}
pub fn get_value(&self) -> i32 {
self.value
}
}
