메모리 맵 I/O Memory-mapped I/O

// GPIO 레지스터 직접 접근
const GPIO_BASE: usize = 0x4002_0000;
#[repr(C)]
struct GpioRegs {
moder: u32, // 모드 레지스터
otyper: u32, // 출력 타입
ospeedr: u32, // 속도
pupdr: u32, // 풀업/다운
idr: u32, // 입력 데이터
odr: u32, // 출력 데이터
}
let gpio = unsafe { &*(GPIO_BASE as *const GpioRegs) };
unsafe { gpio.odr |= 1 << 5; } // Pin 5 HIGH

Volatile 접근 Volatile Access

use core::ptr::{read_volatile, write_volatile};
// 컴파일러 최적화 방지
let reg = 0x4000_0000 as *mut u32;
unsafe {
write_volatile(reg, 0x1234);
let value = read_volatile(reg);
}

⚠️ 중요 ⚠️ Important

하드웨어 레지스터는 volatile 필수!
일반 메모리 접근은 최적화될 수 있음
read/write_volatile로 보장

Hardware registers require volatile!
Normal access can be optimized away
Guaranteed with read/write_volatile

인터럽트 처리 Interrupt Handling

use cortex_m_rt::interrupt;
// 인터럽트 벡터 테이블
#[interrupt]
fn TIM2() {
// 타이머 2 인터럽트 핸들러
unsafe {
// 인터럽트 플래그 클리어
(*TIM2::ptr()).sr.modify(|_, w| w.uif().clear_bit());
}
}
// 인터럽트 활성화
unsafe {
cortex_m::peripheral::NVIC::unmask(Interrupt::TIM2);
}

임계 영역 (Critical Section) Critical Sections

use cortex_m::interrupt;
// 인터럽트 비활성화
interrupt::free(|_cs| {
// 임계 영역: 인터럽트 없음
unsafe {
COUNTER += 1;
}
}); // 인터럽트 자동 재활성화
// Mutex로 안전하게 공유
static DATA: Mutex<Cell<u32>> = Mutex::new(Cell::new(0));
interrupt::free(|cs| {
DATA.borrow(cs).set(42);
});
