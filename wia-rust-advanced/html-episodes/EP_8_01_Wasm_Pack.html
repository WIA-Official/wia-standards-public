설치 및 초기화 Installation & Setup

# wasm-pack 설치
$ curl https://rustwasm.github.io/wasm-pack/installer/init.sh -sSf | sh
# 새 프로젝트 생성
$ cargo new --lib my-wasm-project
$ cd my-wasm-project
# Cargo.toml 수정
[lib]
crate-type = ["cdylib"]
[dependencies]
wasm-bindgen = "0.2"

첫 번째 Wasm 함수 First Wasm Function

// src/lib.rs
use wasm_bindgen::prelude::*;
#[wasm_bindgen]
pub fn greet(name: &str) -> String {
format!("Hello, {}!", name)
}
#[wasm_bindgen]
pub fn add(a: i32, b: i32) -> i32 {
a + b
}

빌드 및 배포 Build & Deploy

# 빌드 (웹 타겟)
$ wasm-pack build --target web
# npm 패키지로 빌드

$ wasm-pack build --target nodejs

# - my_wasm_project.js
# - package.json

웹에서 사용하기 Using in Web

<script type="module">
import init, { greet, add } from './pkg/my_wasm_project.js';
async function run() {
await init();

}
run();
</script>

최적화 옵션 Optimization Options

# Cargo.toml
[profile.release]
opt-level = "z" # 크기 최적화
lto = true # Link-Time Optimization
codegen-units = 1 # 단일 코드 생성 유닛
[dependencies]
wee_alloc = "0.4" # 작은 메모리 할당자
// src/lib.rs
#[global_allocator]
static ALLOC: wee_alloc::WeeAlloc = wee_alloc::WeeAlloc::INIT;
