기본 Lint 구조 Basic Lint Structure

declare_clippy_lint! {
pub MANUAL_UNWRAP,
style,
"manual implementation of unwrap"
}
pub struct ManualUnwrap;
impl LateLintPass for ManualUnwrap {
fn check_expr(&mut self, cx: &LateContext, expr: &Expr) {
if let ExprKind::Match(scrutinee, arms, _) = &expr.kind {
// match Some(x) => x 패턴 감지
if is_manual_unwrap(arms) {
cx.span_lint(MANUAL_UNWRAP, expr.span,
"use .unwrap() instead");
}
}
}
}

HIR 탐색 HIR Navigation

use rustc_hir::{Expr, ExprKind, QPath};
// 메서드 호출 감지
if let ExprKind::MethodCall(path, receiver, args, _) = &expr.kind {
let method_name = path.ident.as_str();
if method_name == "clone" {
// 불필요한 clone 감지
if is_copy_type(cx, receiver) {
cx.span_lint_and_sugg(
UNNECESSARY_CLONE,
expr.span,
"unnecessary clone on Copy type",
"remove this",
format!("{}", snippet(cx, receiver.span, "..")),
);
}
}
}

수정 제안 Suggestions

// 자동 수정 제안
cx.span_lint_and_sugg(
MY_LINT,
expr.span,
"use better pattern",
"try",
"better_code".to_string(),
Applicability::MachineApplicable, // 안전하게 적용 가능
);

🔧 Applicability 레벨 🔧 Applicability Levels

MachineApplicable - 자동 적용 안전
MaybeIncorrect - 검토 필요
HasPlaceholders - 수동 수정 필요

MachineApplicable - Safe to auto-apply
MaybeIncorrect - Needs review
HasPlaceholders - Manual fix needed
