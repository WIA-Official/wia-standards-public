평가기 (Evaluator) Evaluator

type Environment = HashMapf64>;
fn eval(expr: &Expr, env: &mut Environment) -> Result<f64, String> {
match expr {
Expr::Number(n) => Ok(*n),
Expr::Ident(name) => env.get(name)
.copied()
.ok_or_else(|| format!("Undefined: {}", name)),
Expr::BinOp { op, left, right } => {
let l = eval(left, env)?;
let r = eval(right, env)?;
Ok(match op {
BinOp::Add => l + r, BinOp::Sub => l - r,
BinOp::Mul => l * r, BinOp::Div => l / r,
})
}
_ => Err("Not implemented".into()),
}
}

REPL (Read-Eval-Print Loop) REPL (Read-Eval-Print Loop)

fn repl() {
let mut env = Environment::new();
let stdin = io::stdin();
println!("Welcome to MyLang v1.0");
loop {
print!(">> ");
io::stdout().flush().unwrap();
let mut input = String::new();
stdin.read_line(&mut input).unwrap();
let result = run_code(&input, &mut env);
println!("{:?}", result);
}
}
