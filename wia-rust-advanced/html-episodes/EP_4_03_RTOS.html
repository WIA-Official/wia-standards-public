기본 Embassy 앱 Basic Embassy App

#![no_std]
#![no_main]
use embassy_executor::Spawner;
use embassy_time::{Duration, Timer};
#[embassy_executor::main]
async fn main(_spawner: Spawner) {
loop {
// LED 토글
led.toggle();
Timer::after(Duration::from_millis(1000)).await;
}
}

⚡ 핵심 개념 ⚡ Key Concept

async/await로 간결한 비동기 코드
Timer::after()로 논블로킹 대기
자동으로 다른 태스크에 양보

Clean async code with async/await
Non-blocking wait with Timer::after()
Automatically yields to other tasks

태스크 생성 Spawning Tasks

#[embassy_executor::task]
async fn blink_led(mut led: Output) {
loop {
led.toggle();
Timer::after_millis(500).await;
}
}
#[embassy_executor::task]
async fn read_sensor(mut i2c: I2c) {
loop {
let data = i2c.read(&mut buffer).await;
Timer::after_secs(1).await;
}
}
#[embassy_executor::main]
async fn main(spawner: Spawner) {
spawner.spawn(blink_led(led)).unwrap();
spawner.spawn(read_sensor(i2c)).unwrap();
}

채널 (Channels) Channels

use embassy_sync::channel::Channel;
static CHANNEL: Channel<NoopRawMutex, SensorData, 10>
= Channel::new();
#[embassy_executor::task]
async fn producer() {
loop {
let data = read_sensor();
CHANNEL.send(data).await;
}
}
#[embassy_executor::task]
async fn consumer() {
loop {
let data = CHANNEL.receive().await;
process(data);
}
}

시그널 (Signals) Signals

use embassy_sync::signal::Signal;
static BUTTON_SIGNAL: Signal<NoopRawMutex, ()> = Signal::new();
// 인터럽트 핸들러
#[interrupt]
fn EXTI0() {
BUTTON_SIGNAL.signal(());
}
// 태스크에서 대기
#[embassy_executor::task]
async fn button_handler() {
loop {
BUTTON_SIGNAL.wait().await;
// 버튼 처리
}
}
