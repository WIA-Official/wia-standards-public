VGA 텍스트 모드 VGA Text Mode

const VGA_BUFFER: *mut u8 = 0xb8000 as *mut u8;
fn print_hello() {
let hello = b"Hello from Rust Bootloader!";
for (i, &byte) in hello.iter().enumerate() {
unsafe {
*VGA_BUFFER.offset(i as isize * 2) = byte;
*VGA_BUFFER.offset(i as isize * 2 + 1) = 0x0f; // 흰색
}
}
}
💡 0xb8000은 VGA 텍스트 버퍼의 메모리 주소입니다 💡 0xb8000 is the memory address of VGA text buffer

bootloader 크레이트 사용 Using bootloader Crate

# Cargo.toml
[dependencies]
bootloader = "0.9"
// main.rs
use bootloader::{BootInfo, entry_point};
entry_point!(kernel_main);
fn kernel_main(boot_info: &'static BootInfo) -> ! {
println!("Hello from kernel!");
loop {}
}
