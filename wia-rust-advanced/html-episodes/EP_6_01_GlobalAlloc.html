GlobalAlloc 트레잇 GlobalAlloc Trait

use core::alloc::{GlobalAlloc, Layout};
pub unsafe trait GlobalAlloc {
// 메모리 할당
unsafe fn alloc(&self, layout: Layout) -> *mut u8;
// 메모리 해제
unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout);
// 선택적 메서드
unsafe fn realloc(&self, ptr: *mut u8, layout: Layout, new_size: usize) -> *mut u8;
}

alloc 구현 alloc Implementation

unsafe impl GlobalAlloc for Locked {
unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
let mut bump = self.lock();
let alloc_start = align_up(bump.next.load(Ordering::Relaxed), layout.align());
let alloc_end = alloc_start + layout.size();
if alloc_end > bump.heap_end {
return core::ptr::null_mut(); // OOM
}
bump.next.store(alloc_end, Ordering::Relaxed);
bump.allocations.fetch_add(1, Ordering::Relaxed);
alloc_start as *mut u8
}
}

dealloc 구현 dealloc Implementation

unsafe fn dealloc(&self, _ptr: *mut u8, _layout: Layout) {
let mut bump = self.lock();
bump.allocations.fetch_sub(1, Ordering::Relaxed);
// 모든 할당이 해제되면 리셋
if bump.allocations.load(Ordering::Relaxed) == 0 {
bump.next.store(bump.heap_start, Ordering::Relaxed);
}
}
💡 Bump allocator는 개별 해제를 지원하지 않고 전체 리셋만 가능합니다 💡 Bump allocator doesn't support individual deallocation, only full reset

글로벌 할당자 등록 Register Global Allocator

#[global_allocator]
static ALLOCATOR: Locked = Locked::new(BumpAllocator::new());
// 초기화
pub fn init_heap(heap_start: usize, heap_size: usize) {
unsafe {
ALLOCATOR.lock().init(heap_start, heap_size);
}
}
