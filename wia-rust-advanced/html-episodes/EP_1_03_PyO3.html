PyO3 예제PyO3 Example

use pyo3::prelude::*;
#[pyfunction]
fn sum_as_string(a: usize, b: usize) -> String {
(a + b).to_string()
}
#[pymodule]
fn mylib(_py: Python, m: &PyModule) -> PyResult<()> {
m.add_function(wrap_pyfunction!(sum_as_string, m)?)?;
Ok(())
}
# Python usage
import mylib
print(mylib.sum_as_string(5, 20)) # "25"

napi-rs 예제napi-rs Example

use napi_derive::napi;
#[napi]
pub fn fibonacci(n: u32) -> u32 {
match n {
0 | 1 => n,
_ => fibonacci(n - 1) + fibonacci(n - 2),
}
}
// JavaScript/TypeScript usage
const { fibonacci } = require('./index.node');
console.log(fibonacci(10)); // 55
