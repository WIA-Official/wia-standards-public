기본 문법 Basic Syntax

#![feature(asm_sym)]
use std::arch::asm;
fn simple_asm() {
let mut x: u64 = 5;
unsafe {
asm!(
"add {0}, {1}",
inout(reg) x,
in(reg) 10u64,
);
}
println!("Result: {}", x); // 15
}

레지스터 제약 Register Constraints

unsafe {
let result: u64;
asm!(
"mov {0}, {1}",
"add {0}, {2}",
out(reg) result, // 출력 레지스터
in(reg) 10u64, // 입력 레지스터
in(reg) 20u64,
);
println!("Result: {}", result); // 30
}
💡 in, out, inout, lateout 등의 제약을 사용합니다 💡 Use constraints like in, out, inout, lateout

CPUID 예제 CPUID Example

fn get_cpu_info() -> (u32, u32, u32, u32) {
let eax: u32;
let ebx: u32;
let ecx: u32;
let edx: u32;
unsafe {
asm!(
"cpuid",
inout("eax") 0u32 => eax,
out("ebx") ebx,
out("ecx") ecx,
out("edx") edx,
);
}
(eax, ebx, ecx, edx)
}

메모리 배리어 Memory Barrier

fn memory_fence() {
unsafe {
asm!(
"mfence",
options(nostack, preserves_flags),
);
}
}
fn compiler_fence() {
unsafe {
asm!("", options(nomem, nostack, preserves_flags));
}
}
💡 멀티스레드 프로그래밍에서 메모리 순서를 보장합니다 💡 Ensures memory ordering in multithreaded programming

원자적 연산 Atomic Operations

fn atomic_increment(ptr: *mut i64) -> i64 {
let result: i64;
unsafe {
asm!(
"mov {tmp}, 1",
"lock xadd [{ptr}], {tmp}",
ptr = in(reg) ptr,
tmp = out(reg) result,
);
}
result
}
