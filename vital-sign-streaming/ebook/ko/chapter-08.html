8.3.2 펌웨어 구조

// C++ 펌웨어 예제 (ESP32)
#include "WIAMed003.h"

WIAMed003 wia;
ECGSensor ecgSensor(ADC_PIN);

void setup() {
  // WIA-MED-003 초기화
  wia.begin({
    .deviceId = "ECG-ESP32-001",
    .deviceType = WIA_DEVICE_ECG,
    .protocol = WIA_PROTOCOL_BLE,
    .samplingRate = 512  // Hz
  });

  // 센서 초기화
  ecgSensor.begin();
  ecgSensor.setGain(1000);
  ecgSensor.setFilter(0.5, 100);  // 0.5-100Hz bandpass

  // 콜백 등록
  ecgSensor.onData(handleECGData);
}

void loop() {
  // 센서 데이터 읽기
  ecgSensor.update();

  // WIA 프로토콜 처리
  wia.process();

  delay(2);  // 500Hz 루프
}

void handleECGData(float voltage) {
  static float buffer[512];
  static int index = 0;

  // 버퍼에 저장
  buffer[index++] = voltage;

  // 버퍼가 가득 차면 전송
  if (index >= 512) {
    WIAECGData data = {
      .standard = "WIA-MED-003",
      .version = "1.0.0",
      .signalType = "ECG",
      .timestamp = wia.getTimestamp(),
      .samplingRate = 512,
      .data = buffer,
      .dataLength = 512
    };

    // WIA 형식으로 인코딩 및 전송
    wia.sendData(data);

    index = 0;
  }
}

8.4.2 모바일 앱 (React Native)

// React Native + WIA-MED-003
import { VitalSignStream } from 'wia-med-003-mobile';
import { BleManager } from 'react-native-ble-plx';

function VitalSignMonitor() {
  const [device, setDevice] = useState(null);
  const [spo2, setSpo2] = useState(null);

  useEffect(() => {
    const stream = new VitalSignStream({
      bleManager: new BleManager()
    });

    // BLE 기기 검색
    stream.discoverDevices({ signalType: 'SPO2' }).then(devices => {
      if (devices.length > 0) {
        setDevice(devices[0]);
        connectAndStream(stream, devices[0]);
      }
    });
  }, []);

  async function connectAndStream(stream, device) {
    await stream.connect(device.id);
    await stream.startStream(device.id);

    stream.onData(data => {
      setSpo2({
        value: data.measurement.spo2,
        pulseRate: data.measurement.pulseRate,
        quality: data.quality.signalStrength
      });
    });
  }

  return (
    
      산소포화도
      {spo2?.value}%
      맥박: {spo2?.pulseRate} bpm
      신호: {spo2?.quality}
    
  );
}

// 전체 워크플로우 테스트
describe('End-to-End ECG Monitoring', () => {
  test('complete monitoring workflow', async () => {
    // 1. 기기 검색
    const devices = await stream.discoverDevices({ signalType: 'ECG' });
    expect(devices.length).toBeGreaterThan(0);

    // 2. 연결
    const connection = await stream.connect(devices[0].id);
    expect(connection.status).toBe('connected');

    // 3. 스트리밍 시작
    const ecgStream = await stream.startStream(devices[0].id);
    expect(ecgStream.status).toBe('streaming');

    // 4. 데이터 수신
    const receivedData = await new Promise(resolve => {
      stream.onData(data => resolve(data));
    });

    expect(receivedData.signalType).toBe('ECG');
    expect(receivedData.leads).toBeInstanceOf(Array);

    // 5. 데이터 저장
    const saveResult = await stream.saveData(receivedData);
    expect(saveResult.success).toBe(true);

    // 6. 스트리밍 중지
    await stream.stopStream(ecgStream.id);

    // 7. 연결 해제
    await stream.disconnect(devices[0].id);
  });
});

test('streaming performance', async () => {
  const startTime = Date.now();
  let sampleCount = 0;
  let latencies = [];

  stream.onData((data) => {
    const latency = Date.now() - new Date(data.timestamp).getTime();
    latencies.push(latency);
    sampleCount += data.leads[0].data.length;
  });

  await stream.startStream(deviceId);
  await sleep(10000);  // 10초 테스트
  await stream.stopStream(streamId);

  const duration = (Date.now() - startTime) / 1000;
  const avgLatency = latencies.reduce((a, b) => a + b) / latencies.length;
  const throughput = sampleCount / duration;

  expect(avgLatency).toBeLessThan(100);  // <100ms 지연
  expect(throughput).toBeGreaterThan(500);  // >500 samples/sec
});

  • 문서화: 코드 주석, API 문서, README 유지

8.7.2 보안 모범 사례

  • 최소 권한 원칙: 필요한 권한만 요청
  • 시크릿 관리: 환경 변수, Key Vault 사용
  • 정기 업데이트: 의존성 라이브러리 최신 상태 유지

  • 침투 테스트: 연 1회 이상 보안 감사

  • 빠른 연결: 기기 검색 및 연결 10초 이내
  • 명확한 피드백: 로딩 상태, 오류 메시지 표시
  • 오프라인 지원: 네트워크 없어도 기본 기능 작동
  • 배터리 최적화: 백그라운드에서 효율적으로 작동
  • 접근성: 스크린 리더, 고대비 모드 지원
