7.1.3 양방향 동기화

class EHRSync {
  async syncToEHR(vitalSignData) {
    // WIA → EHR
    const observation = convertToFHIR(vitalSignData);
    await ehrClient.create('Observation', observation);

    // 동기화 기록
    await this.recordSync({
      direction: 'to-ehr',
      wiaId: vitalSignData.id,
      fhirId: observation.id,
      timestamp: new Date()
    });
  }

  async syncFromEHR(patientId, startDate) {
    // EHR → WIA
    const observations = await ehrClient.search('Observation', {
      patient: patientId,
      category: 'vital-signs',
      date: `ge${startDate}`
    });

    for (const obs of observations.entry) {
      const wiaData = convertFromFHIR(obs.resource);
      await vitalSignStream.saveData(wiaData);
    }
  }

  async bidirectionalSync() {
    // 양방향 동기화 (충돌 해결 포함)
    const wiaData = await vitalSignStream.loadData({ unsynced: true });
    const ehrData = await ehrClient.search('Observation', { ... });

    const conflicts = detectConflicts(wiaData, ehrData);

    for (const conflict of conflicts) {
      // 최신 데이터 우선 (Last Write Wins)
      const resolved = resolveConflict(conflict, 'latest-wins');
      await this.applyResolution(resolved);
    }
  }
}

7.2.2 원격 환자 모니터링 (RPM)

class RemotePatientMonitoring {
  constructor(patientId) {
    this.patientId = patientId;
    this.alerts = [];
    this.devices = [];
  }

  async startMonitoring() {
    // 환자의 모든 WIA-MED-003 기기 검색
    this.devices = await vitalSignStream.discoverDevices({
      patientId: this.patientId
    });

    // 각 기기에서 데이터 수집
    for (const device of this.devices) {
      const stream = await vitalSignStream.startStream(device.id);

      stream.onData(async (data) => {
        // 클라우드에 저장
        await this.saveToCloud(data);

        // 이상 감지
        const alerts = await this.detectAnomalies(data);

        if (alerts.length > 0) {
          await this.notifyProvider(alerts);
        }
      });
    }
  }

  async notifyProvider(alerts) {
    // 의사에게 알림
    await sendNotification({
      to: 'doctor@example.com',
      subject: `[긴급] 환자 ${this.patientId} 이상 감지`,
      body: alertSummary(alerts),
      sms: true,  // SMS도 함께 전송
      priority: 'high'
    });

    // 대시보드 업데이트
    await updateProviderDashboard(this.patientId, alerts);
  }
}

import HealthKit

class WIAHealthKitBridge {
  let healthStore = HKHealthStore()

  func saveToHealthKit(vitalSignData: VitalSignData) {
    switch vitalSignData.signalType {
    case .ecg:
      let heartRate = HKQuantity(unit: HKUnit.count().unitDivided(by: .minute()),
                                  doubleValue: vitalSignData.analysis.heartRate)
      let sample = HKQuantitySample(type: HKQuantityType.quantityType(forIdentifier: .heartRate)!,
                                    quantity: heartRate,
                                    start: vitalSignData.timestamp,
                                    end: vitalSignData.timestamp)

      healthStore.save(sample) { success, error in
        if success {
          print("HealthKit에 저장됨")
        }
      }

    case .spo2:
      let spo2 = HKQuantity(unit: HKUnit.percent(),
                            doubleValue: vitalSignData.measurement.spo2)
      let sample = HKQuantitySample(type: HKQuantityType.quantityType(forIdentifier: .oxygenSaturation)!,
                                    quantity: spo2,
                                    start: vitalSignData.timestamp,
                                    end: vitalSignData.timestamp)

      healthStore.save(sample) { success, error in ... }

    // 기타 신호 유형...
    }
  }

  func loadFromHealthKit(completion: @escaping ([VitalSignData]) -> Void) {
    let heartRateType = HKQuantityType.quantityType(forIdentifier: .heartRate)!
    let query = HKSampleQuery(sampleType: heartRateType,
                              predicate: nil,
                              limit: HKObjectQueryNoLimit,
                              sortDescriptors: [NSSortDescriptor(key: HKSampleSortIdentifierStartDate, ascending: false)]) {
      query, samples, error in

      let vitalSignData = samples?.compactMap { sample in
        convertFromHealthKit(sample)
      }

      completion(vitalSignData ?? [])
    }

    healthStore.execute(query)
  }
}

7.3.2 Google Fit 통합

// Kotlin (Android)
class WIAGoogleFitBridge(private val context: Context) {
  private val googleSignInAccount = GoogleSignIn.getLastSignedInAccount(context)
  private val fitnessOptions = FitnessOptions.builder()
    .addDataType(DataType.TYPE_HEART_RATE_BPM, FitnessOptions.ACCESS_WRITE)
    .addDataType(DataType.TYPE_OXYGEN_SATURATION, FitnessOptions.ACCESS_WRITE)
    .build()

  fun saveToGoogleFit(vitalSignData: VitalSignData) {
    val dataSource = DataSource.Builder()
      .setAppPackageName(context.packageName)
      .setDataType(when (vitalSignData.signalType) {
        "ECG" -> DataType.TYPE_HEART_RATE_BPM
        "SPO2" -> DataType.TYPE_OXYGEN_SATURATION
        else -> return
      })
      .setType(DataSource.TYPE_RAW)
      .build()

    val dataPoint = DataPoint.builder(dataSource)
      .setTimestamp(vitalSignData.timestamp.time, TimeUnit.MILLISECONDS)
      .setField(when (vitalSignData.signalType) {
        "ECG" -> Field.FIELD_BPM
        "SPO2" -> Field.FIELD_PERCENTAGE
        else -> return
      }, vitalSignData.getValue())
      .build()

    val dataSet = DataSet.builder(dataSource)
      .add(dataPoint)
      .build()

    Fitness.getHistoryClient(context, googleSignInAccount!!)
      .insertData(dataSet)
      .addOnSuccessListener {
        Log.i("WIA", "Google Fit에 저장됨")
      }
  }
}

7.4.1 이상 감지 모델

// TensorFlow.js를 사용한 ECG 이상 감지
import * as tf from '@tensorflow/tfjs';

class ECGAnomalyDetector {
  model: tf.LayersModel;

  async loadModel() {
    this.model = await tf.loadLayersModel('https://models.wia.live/ecg-anomaly/model.json');
  }

  async detectAnomalies(ecgData: ECGData): Promise {
    // 전처리
    const preprocessed = this.preprocess(ecgData);

    // 예측
    const tensor = tf.tensor2d([preprocessed]);
    const prediction = this.model.predict(tensor) as tf.Tensor;
    const probabilities = await prediction.data();

    // 결과 해석
    const anomalies: Anomaly[] = [];
    const classes = ['normal', 'afib', 'vt', 'vf', 'pvc'];

    probabilities.forEach((prob, idx) => {
      if (idx > 0 && prob > 0.7) {  // 정상(idx=0) 제외, 70% 이상 신뢰도
        anomalies.push({
          type: classes[idx],
          probability: prob,
          timestamp: ecgData.timestamp,
          severity: this.calculateSeverity(classes[idx], prob)
        });
      }
    });

    return anomalies;
  }
}

// 사용
const detector = new ECGAnomalyDetector();
await detector.loadModel();

vitalSignStream.onData(async (data) => {
  if (data.signalType === 'ECG') {
    const anomalies = await detector.detectAnomalies(data);

    if (anomalies.length > 0) {
      console.warn('이상 감지:', anomalies);
      await notifyPatient(anomalies);
    }
  }
});

7.4.2 예측 분석

// 혈압 트렌드 예측
class BloodPressurePredictor {
  async predictNextWeek(patientId: string): Promise {
    // 과거 데이터 로드
    const historicalData = await vitalSignStream.loadData({
      signalType: 'BLOOD_PRESSURE',
      patientId: patientId,
      startTime: new Date(Date.now() - 90 * 24 * 60 * 60 * 1000).toISOString() // 90일
    });

    // 시계열 데이터 준비
    const timeSeries = historicalData.map(d => ({
      date: new Date(d.timestamp),
      systolic: d.measurement.systolic,
      diastolic: d.measurement.diastolic
    }));

    // ARIMA 모델로 예측
    const model = new ARIMAModel(timeSeries, { p: 1, d: 1, q: 1 });
    const predictions = model.forecast(7); // 7일 예측

    return predictions.map(pred => ({
      date: pred.date,
      systolic: pred.systolic,
      diastolic: pred.diastolic,
      confidence: pred.confidence,
      risk: this.assessRisk(pred)
    }));
  }

  assessRisk(prediction: any): string {
    if (prediction.systolic > 140 || prediction.diastolic > 90) {
      return 'high';
    } else if (prediction.systolic > 130 || prediction.diastolic > 80) {
      return 'moderate';
    }
    return 'low';
  }
}

7.5.2 Azure Health Data Services

// Azure FHIR API
const { DefaultAzureCredential } = require('@azure/identity');
const axios = require('axios');

class AzureFHIRClient {
  constructor(fhirUrl) {
    this.fhirUrl = fhirUrl;
    this.credential = new DefaultAzureCredential();
  }

  async getAccessToken() {
    const tokenResponse = await this.credential.getToken('https://azurehealthcareapis.com/.default');
    return tokenResponse.token;
  }

  async createObservation(vitalSignData) {
    const token = await this.getAccessToken();
    const observation = convertToFHIR(vitalSignData);

    const response = await axios.post(
      `${this.fhirUrl}/Observation`,
      observation,
      {
        headers: {
          'Authorization': `Bearer ${token}`,
          'Content-Type': 'application/fhir+json'
        }
      }
    );

    return response.data;
  }
}

7.6 완전한 통합 예제

// 종합 헬스케어 플랫폼
class IntegratedHealthPlatform {
  constructor(config) {
    this.vitalSignStream = new VitalSignStream(config.wia);
    this.ehrClient = new FHIRClient(config.ehr);
    this.telemedicine = new TelemedicineClient(config.telemedicine);
    this.ml = new MLAnalytics(config.ml);
    this.cloud = new CloudStorage(config.cloud);
  }

  async startPatientMonitoring(patientId) {
    // 1. 기기 연결
    const devices = await this.vitalSignStream.discoverDevices({ patientId });
    await Promise.all(devices.map(d => this.vitalSignStream.connect(d.id)));

    // 2. 실시간 모니터링 시작
    for (const device of devices) {
      const stream = await this.vitalSignStream.startStream(device.id);

      stream.onData(async (data) => {
        // 병렬 처리
        await Promise.all([
          // EHR에 저장
          this.ehrClient.create('Observation', convertToFHIR(data)),

          // 클라우드 백업
          this.cloud.save(data),

          // ML 분석
          this.analyzeAndAlert(data),

          // 실시간 대시보드 업데이트
          this.updateDashboard(patientId, data)
        ]);
      });
    }

    // 3. 알림 설정
    this.setupAlerts(patientId);

    // 4. 화상 진료 준비
    await this.telemedicine.prepareRoom(patientId);
  }

  async analyzeAndAlert(data) {
    // AI/ML 분석
    const analysis = await this.ml.analyze(data);

    if (analysis.anomalies.length > 0) {
      // 심각도에 따라 다른 조치
      for (const anomaly of analysis.anomalies) {
        if (anomaly.severity === 'critical') {
          // 긴급 연락
          await this.emergencyAlert(anomaly);
        } else if (anomaly.severity === 'high') {
          // 의사에게 알림
          await this.notifyDoctor(anomaly);
        } else {
          // 환자에게 알림
          await this.notifyPatient(anomaly);
        }
      }
    }
  }
}

// 사용

  wia: { apiKey: '...' },
  ehr: { baseUrl: 'https://fhir.epic.com/...' },
  telemedicine: { provider: 'twilio' },
  ml: { modelEndpoint: 'https://ml.wia.live/...' },
  cloud: { provider: 'aws-healthlake' }
});

await platform.startPatientMonitoring('patient-123');
