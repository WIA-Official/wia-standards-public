06.3 기술적 세부 사항: LED 광선 치료 디바이스

LED 광선 치료를 위한 스마트 디바이스 제어 시스템의 구현 예제입니다:

// WIA-IND-004 LED 광선 치료 디바이스 시스템
interface LEDTherapyDevice {
  deviceId: string;
  userId: string;
  model: string;
  ledPanels: LEDPanel[];
  currentSession: TherapySession | null;
  safetyFeatures: SafetyConfig;
  firmwareVersion: string;
}

interface LEDPanel {
  panelId: string;
  position: 'face' | 'neck' | 'body';
  wavelengths: WavelengthConfig[];
  maxIntensity: number;  // mW/cm²
  coverage: number;      // cm²
}

interface WavelengthConfig {
  color: 'red' | 'blue' | 'green' | 'yellow' | 'near_infrared';
  wavelength: number;    // nm (예: 630nm for red)
  targetCondition: SkinCondition[];
  penetrationDepth: 'epidermis' | 'dermis' | 'subcutaneous';
}

interface TherapySession {
  sessionId: string;
  startTime: Date;
  duration: number;      // seconds
  protocol: TherapyProtocol;
  intensityLevel: number;  // 0-100%
  sensorReadings: SensorReading[];
}

type SkinCondition = 'acne' | 'wrinkles' | 'inflammation' | 'hyperpigmentation' | 'wound_healing';

class LEDTherapyController {
  private device: LEDTherapyDevice;
  private safetyMonitor: SafetyMonitor;

  async startSession(protocol: TherapyProtocol, userId: string): Promise<TherapySession> {
    // 안전 체크
    await this.safetyMonitor.preTreatmentCheck(userId);

    // 프로토콜에 따른 파장 및 강도 설정
    const panelConfigs = this.configurePanels(protocol);

    // LED 패널 활성화
    for (const config of panelConfigs) {
      await this.activatePanel(config);
    }

    const session: TherapySession = {
      sessionId: this.generateSessionId(),
      startTime: new Date(),
      duration: protocol.recommendedDuration,
      protocol,
      intensityLevel: protocol.startingIntensity,
      sensorReadings: []
    };

    // 실시간 모니터링 시작
    this.startRealTimeMonitoring(session);

    return session;
  }

  private configurePanels(protocol: TherapyProtocol): PanelConfiguration[] {
    return protocol.targetAreas.map(area => {
      const panel = this.device.ledPanels.find(p => p.position === area);
      const wavelength = panel.wavelengths.find(
        w => w.targetCondition.includes(protocol.targetCondition)
      );
      return {
        panelId: panel.panelId,
        wavelength: wavelength.wavelength,
        intensity: protocol.startingIntensity * panel.maxIntensity / 100
      };
    });
  }

  async adjustIntensity(session: TherapySession, newLevel: number): Promise<void> {
    if (newLevel > this.safetyMonitor.getMaxSafeIntensity()) {
      throw new Error('안전 한계를 초과하는 강도입니다');
    }
    session.intensityLevel = newLevel;
    await this.updatePanelIntensities(newLevel);
  }
}
