신기술

08.3 기술적 세부 사항: 뷰티 제품 블록체인 인증

블록체인 기반 뷰티 제품 진위 확인 및 공급망 추적 시스템의 구현 예제입니다:

// WIA-IND-004 뷰티 제품 블록체인 인증 시스템
interface BeautyProductNFT {
  tokenId: string;
  productId: string;
  brand: string;
  batchNumber: string;
  manufacturingDate: Date;
  expiryDate: Date;
  ingredients: IngredientOrigin[];
  certifications: ProductCertification[];
  supplyChainEvents: SupplyChainEvent[];
}

interface IngredientOrigin {
  ingredient: string;
  source: string;
  harvestDate: Date;
  sustainabilityCert: string | null;
  carbonFootprint: number;  // kg CO2
}

interface SupplyChainEvent {
  eventId: string;
  timestamp: Date;
  location: GeoLocation;
  eventType: 'manufactured' | 'quality_checked' | 'shipped' | 'received' | 'sold';
  actor: string;
  signature: string;
}

interface ProductCertification {
  certType: 'organic' | 'cruelty_free' | 'vegan' | 'fair_trade' | 'eco_cert';
  issuer: string;
  validUntil: Date;
  verificationUrl: string;
}

class BeautyBlockchainVerifier {
  private web3: Web3Provider;
  private contractAddress: string;
  private ipfsClient: IPFSClient;

  async verifyProduct(qrCode: string): Promise<VerificationResult> {
    const tokenId = this.parseQRCode(qrCode);
    const nft = await this.fetchNFTMetadata(tokenId);
    const detailedInfo = await this.ipfsClient.get(nft.metadataUri);

    const chainValid = await this.validateSupplyChain(nft.supplyChainEvents);
    const certsValid = await this.validateCertifications(nft.certifications);

    return {
      isAuthentic: chainValid && certsValid,
      product: nft,
      verificationTimestamp: new Date(),
      trustScore: this.calculateTrustScore(nft, chainValid, certsValid),
      warnings: this.generateWarnings(nft)
    };
  }

  async traceIngredientOrigin(productId: string, ingredient: string): Promise<OriginTrace> {
    const nft = await this.getProductNFT(productId);
    const ingredientInfo = nft.ingredients.find(i => i.ingredient === ingredient);

    return {
      ingredient: ingredientInfo,
      journeyMap: await this.buildJourneyMap(ingredientInfo),
      sustainabilityScore: this.calculateSustainabilityScore(ingredientInfo),
      carbonFootprintBreakdown: await this.analyzeCarbon(ingredientInfo)
    };
  }

  private calculateTrustScore(
    nft: BeautyProductNFT, chainValid: boolean, certsValid: boolean
  ): number {
    let score = 0;
    if (chainValid) score += 40;
    if (certsValid) score += 30;
    score += nft.supplyChainEvents.length * 2;
    score += nft.certifications.length * 5;
    return Math.min(score, 100);
  }
}
