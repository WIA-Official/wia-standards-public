2.2 통신 계층 (Network Layer)

사막 환경의 넓은 면적(10ha 이상)을 커버하기 위해 LoRaWAN 기술을 채택했습니다:

// LoRaWAN 센서 노드 설정 (Arduino/ESP32)
// KAIST 스마트팜 연구실 개발

#include 
#include 
#include 

// LoRa 파라미터
#define FREQUENCY 915E6      // 915MHz (중동 지역)
#define SPREADING_FACTOR 10  // SF10 (통신 거리 최대화)
#define BANDWIDTH 125E3
#define TX_POWER 20          // 20dBm

// 센서 핀 정의
#define DHT_PIN 4
#define SOIL_MOISTURE_PIN A0
#define SOIL_EC_PIN A1

DHT dht(DHT_PIN, DHT22);

struct SensorData {
  uint8_t nodeId;
  float airTemp;
  float airHumidity;
  float soilMoisture;
  float soilEC;
  uint32_t timestamp;
  uint16_t battery;        // mV
};

void setup() {
  Serial.begin(115200);

  // LoRa 초기화
  if (!LoRa.begin(FREQUENCY)) {
    Serial.println("LoRa 초기화 실패!");
    while (1);
  }

  LoRa.setSpreadingFactor(SPREADING_FACTOR);
  LoRa.setSignalBandwidth(BANDWIDTH);
  LoRa.setTxPower(TX_POWER);

  dht.begin();
  Serial.println("센서 노드 준비 완료");
}

void loop() {
  SensorData data;

  // 센서 데이터 수집
  data.nodeId = 1;
  data.airTemp = dht.readTemperature();
  data.airHumidity = dht.readHumidity();
  data.soilMoisture = analogRead(SOIL_MOISTURE_PIN) / 4095.0 * 100;
  data.soilEC = readSoilEC();  // 사용자 정의 함수
  data.timestamp = millis();
  data.battery = readBatteryVoltage();

  // LoRa로 데이터 전송
  LoRa.beginPacket();
  LoRa.write((uint8_t*)&data, sizeof(data));
  LoRa.endPacket();

  Serial.printf("데이터 전송: T=%.1f°C H=%.1f%% SM=%.1f%% EC=%.2f\n",
                data.airTemp, data.airHumidity,
                data.soilMoisture, data.soilEC);

  // 저전력 슬립 (5분)
  delay(300000);
}

float readSoilEC() {
  // EC 센서 읽기 및 보정
  int rawValue = analogRead(SOIL_EC_PIN);
  float voltage = rawValue / 4095.0 * 3.3;
  float ec = voltage * 2.5;  // 보정 계수
  return ec;
}

uint16_t readBatteryVoltage() {
  // 배터리 전압 측정 (리튬 배터리)
  int raw = analogRead(35);
  return (uint16_t)(raw / 4095.0 * 3.3 * 2 * 1000);
}

3. AI 기반 생육 예측 시스템

KAIST 생명화학공학과가 개발한 딥러닝 생육 예측 모델은 94.7%의 정확도를 달성했습니다:

3.1 모델 아키텍처

LSTM (Long Short-Term Memory) 기반 시계열 예측 모델을 사용합니다:

계층 구조 파라미터 출력
입력층 Multivariate Time Series 15개 피처 × 168시간 (batch, 168, 15)
LSTM-1 256 units, return_sequences=True 263,424 (batch, 168, 256)
Dropout-1 rate=0.3 0 (batch, 168, 256)
LSTM-2 128 units, return_sequences=False 197,120 (batch, 128)
Dense-1 64 units, ReLU activation 8,256 (batch, 64)
출력층 3 units (생장량, 품질, 수확일) 195 (batch, 3)

3.2 훈련 데이터셋

  • 데이터 출처: 농촌진흥청 + UAE 실증 농장 3년치 (2020-2023)
  • 샘플 수: 152,000개 작물 생육 기록 (토마토, 상추, 파프리카)
  • 피처: 온도, 습도, CO2, 광량, EC, pH, DO, 급액량, 작물 높이 등 15종
  • 라벨: 7일 후 생장량, 당도/색도, 예상 수확일
  • 전처리: Min-Max Scaling, 결측치 선형 보간, 이상치 제거 (±3σ)

3.3 Python 구현 예제

# KAIST 스마트팜 AI 생육 예측 모델
# TensorFlow 2.x 기반

import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout
import numpy as np

# 데이터 준비
def prepare_data(df, lookback=168):
    """
    과거 168시간(7일) 데이터로 미래 예측
    """
    features = ['temp', 'humidity', 'co2', 'par', 'ec', 'ph', 'do',
                'irrigation', 'height', 'leaf_area', 'stem_diameter',
                'fruit_count', 'days_after_planting', 'vpd', 'dli']

    X, y = [], []
    for i in range(len(df) - lookback - 7):
        X.append(df[features].iloc[i:i+lookback].values)
        # 7일 후 예측값
        y.append([
            df['growth_rate'].iloc[i+lookback+7],  # 생장량
            df['quality_score'].iloc[i+lookback+7], # 품질
            df['harvest_days'].iloc[i+lookback+7]   # 수확일
        ])

    return np.array(X), np.array(y)

# 모델 구축
def build_model(input_shape):
    model = Sequential([
        LSTM(256, return_sequences=True, input_shape=input_shape),
        Dropout(0.3),
        LSTM(128, return_sequences=False),
        Dropout(0.2),
        Dense(64, activation='relu'),
        Dense(3)  # 3개 출력: 생장량, 품질, 수확일
    ])

    model.compile(
        optimizer=tf.keras.optimizers.Adam(learning_rate=0.001),
        loss='mse',
        metrics=['mae']
    )

    return model

# 훈련
def train_model(X_train, y_train, X_val, y_val):
    model = build_model(input_shape=(168, 15))

    callbacks = [
        tf.keras.callbacks.EarlyStopping(
            monitor='val_loss', patience=10, restore_best_weights=True
        ),
        tf.keras.callbacks.ReduceLROnPlateau(
            monitor='val_loss', factor=0.5, patience=5
        ),
        tf.keras.callbacks.ModelCheckpoint(
            'best_model.h5', save_best_only=True
        )
    ]

    history = model.fit(
        X_train, y_train,
        epochs=100,
        batch_size=32,
        validation_data=(X_val, y_val),
        callbacks=callbacks,
        verbose=1
    )

    return model, history

# 예측 실행
def predict_growth(model, current_data):
    """
    현재 데이터로 7일 후 생육 예측
    """
    prediction = model.predict(current_data)

    result = {
        'growth_rate': float(prediction[0][0]),      # mm/day
        'quality_score': float(prediction[0][1]),    # 0-100 점수
        'harvest_days': int(prediction[0][2])        # 예상 수확일
    }

    return result

# 사용 예제
if __name__ == "__main__":
    # 데이터 로드
    import pandas as pd
    df = pd.read_csv('uae_farm_data_2023.csv')

    X, y = prepare_data(df)

    # Train/Validation 분할
    split_idx = int(len(X) * 0.8)
    X_train, X_val = X[:split_idx], X[split_idx:]
    y_train, y_val = y[:split_idx], y[split_idx:]

    # 모델 훈련
    model, history = train_model(X_train, y_train, X_val, y_val)

    # 예측 테스트
    test_sample = X_val[:1]
    prediction = predict_growth(model, test_sample)

    print(f"예측 결과:")
    print(f"  생장 속도: {prediction['growth_rate']:.2f} mm/day")
    print(f"  품질 점수: {prediction['quality_score']:.1f}/100")
    print(f"  예상 수확일: {prediction['harvest_days']}일 후")
