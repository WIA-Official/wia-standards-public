08.2.1 나바호어 TTS 시스템 구축

실시간 텍스트-음성 변환 웹 서비스를 FastAPI로 구현:

# navajo_tts_server.py
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from wia_lang import TTSClient
from pydantic import BaseModel
import io

app = FastAPI(title="Navajo TTS Service")
tts = TTSClient(model="nav-tts-female-v1")

class TTSRequest(BaseModel):
    text: str
    voice: str = "female"
    speed: float = 1.0
    format: str = "wav"

@app.post("/synthesize")
async def synthesize_speech(request: TTSRequest):
    """나바호어 텍스트를 음성으로 변환"""
    try:
        # SSML 태그 지원
        ssml_text = f"""
        <speak>
            <prosody rate="{request.speed}">
                {request.text}
            </prosody>
        </speak>
        """

        # 음성 합성
        audio_data = tts.synthesize(
            text=ssml_text,
            voice=request.voice,
            format=request.format
        )

        # 스트리밍 응답
        return StreamingResponse(
            io.BytesIO(audio_data),
            media_type=f"audio/{request.format}"
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/voices")
async def list_voices():
    """사용 가능한 음성 목록"""
    return {
        "voices": [
            {"id": "female", "name": "여성 화자", "age": "adult"},
            {"id": "male", "name": "남성 화자", "age": "adult"},
            {"id": "elder", "name": "원로 화자", "age": "elder"}
        ]
    }

08.2.2 멀티모달 번역 파이프라인

음성 → 텍스트 → 번역 → 음성 전체 파이프라인:

# multimodal_pipeline.py
from wia_lang import ASRClient, MTClient, TTSClient
from wia_lang.pipeline import TranslationPipeline

class VoiceTranslator:
    def __init__(self, source_lang, target_lang):
        self.asr = ASRClient(language=source_lang)
        self.mt = MTClient(source=source_lang, target=target_lang)
        self.tts = TTSClient(language=target_lang)

    def translate_audio(self, audio_path):
        """음성 → 음성 번역"""
        # 1단계: 음성 인식
        transcription = self.asr.transcribe(audio_path)
        print(f"원문: {transcription['text']}")

        # 2단계: 텍스트 번역
        translation = self.mt.translate(transcription['text'])
        print(f"번역: {translation['text']}")

        # 3단계: 음성 합성
        audio_output = self.tts.synthesize(translation['text'])

        return {
            'source_text': transcription['text'],
            'target_text': translation['text'],
            'audio': audio_output,
            'confidence': {
                'asr': transcription['confidence'],
                'mt': translation['confidence']
            }
        }

# 사용 예시: 나바호어 → 영어 음성 번역
translator = VoiceTranslator(source_lang="nav", target_lang="eng")
result = translator.translate_audio("navajo_greeting.wav")

# 결과 저장
with open("english_translation.wav", "wb") as f:
    f.write(result['audio'])

print(f"ASR 신뢰도: {result['confidence']['asr']:.2%}")
print(f"MT 신뢰도: {result['confidence']['mt']:.2%}")

08.2.3 Flutter 모바일 앱 통합

// lib/services/wia_lang_service.dart
import 'package:wia_lang_flutter/wia_lang_flutter.dart';
import 'package:flutter/services.dart';

class WIALangService {
  late WIALangClient _client;
  bool _isInitialized = false;

  Future<void> initialize() async {
    _client = WIALangClient(
      apiKey: const String.fromEnvironment('WIA_API_KEY'),
      enableOffline: true,
    );

    // 오프라인 모델 다운로드
    await _client.downloadModel(
      language: 'haw',
      task: 'asr',
      modelSize: 'base',
    );

    _isInitialized = true;
  }

  Future<String> transcribeAudio(String audioPath) async {
    if (!_isInitialized) await initialize();

    final result = await _client.asr.transcribe(
      audioPath: audioPath,
      language: 'haw',
    );

    return result.text;
  }

  Stream<String> transcribeStream(Stream<List<int>> audioStream) async* {
    await for (final chunk in audioStream) {
      final result = await _client.asr.transcribeChunk(chunk);
      if (result.isFinal) {
        yield result.text;
      }
    }
  }
}

// lib/screens/voice_recorder_screen.dart
class VoiceRecorderScreen extends StatefulWidget {
  @override
  _VoiceRecorderScreenState createState() => _VoiceRecorderScreenState();
}

class _VoiceRecorderScreenState extends State<VoiceRecorderScreen> {
  final _wiaService = WIALangService();
  String _transcription = '';
  bool _isRecording = false;

  @override
  void initState() {
    super.initState();
    _wiaService.initialize();
  }

  Future<void> _startRecording() async {
    setState(() => _isRecording = true);

    final audioStream = await AudioRecorder.startStream();

    _wiaService.transcribeStream(audioStream).listen((text) {
      setState(() => _transcription = text);
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Column(
        children: [
          Text(_transcription, style: TextStyle(fontSize: 24)),
          ElevatedButton(
            onPressed: _isRecording ? null : _startRecording,
            child: Text(_isRecording ? '녹음 중...' : '녹음 시작'),
          ),
        ],
      ),
    );
  }
}

08.4.1 일반적인 오류와 해결책

❌ 오류: CUDA out of memory

원인: GPU 메모리 부족

해결책:

# 1. 작은 모델 사용
model = load_model("haw-asr-tiny-v1")  # base → tiny

# 2. 배치 크기 감소
result = client.transcribe(audio_path, batch_size=1)

# 3. Mixed precision 사용
import torch
with torch.cuda.amp.autocast():
    result = model.transcribe(audio)

# 4. 모델 양자화
from wia_lang.quantization import quantize_model
quantized_model = quantize_model(model, precision="int8")
❌ 오류: Audio format not supported

원인: 지원하지 않는 오디오 형식

해결책:

# FFmpeg로 변환
ffmpeg -i input.mp3 -ar 16000 -ac 1 output.wav

# Python에서 자동 변환
from wia_lang.audio import convert_audio
audio = convert_audio("input.mp3", sample_rate=16000, channels=1)
❌ 오류: Model download timeout

원인: 네트워크 불안정 또는 방화벽

해결책:

import time

for attempt in range(3):
    try:
        download_model("haw-asr-base-v1", timeout=600)

# 성능 최적화 예시
import asyncio
from wia_lang import ASRClient
from functools import lru_cache

class OptimizedASRService:
    def __init__(self):
        self.client = ASRClient(
            device="cuda",
            compute_type="float16",  # Mixed precision
            num_workers=4,           # 병렬 처리
        )

    @lru_cache(maxsize=1000)
    def _cache_transcription(self, audio_hash):
        """중복 요청 캐싱"""
        pass

    async def batch_transcribe(self, audio_paths):
        """배치 처리"""
        tasks = [
            self.client.transcribe_async(path)
            for path in audio_paths
        ]
        return await asyncio.gather(*tasks)

    def transcribe_with_cache(self, audio_path):
        """캐시 활용"""
        import hashlib
        with open(audio_path, 'rb') as f:
            audio_hash = hashlib.sha256(f.read()).hexdigest()

        cached = self._cache_transcription(audio_hash)
        if cached:
            return cached

        result = self.client.transcribe(audio_path)
        self._cache_transcription.cache_info()  # 캐시 통계
        return result

# 벤치마크
import time
service = OptimizedASRService()

start = time.time()
results = asyncio.run(service.batch_transcribe(["audio1.wav", "audio2.wav", "audio3.wav"]))
print(f"처리 시간: {time.time() - start:.2f}초")
# 출력: 처리 시간: 2.34초 (개별 처리: 7.02초)

# 데이터 소유권 추적
from wia_lang.ocap import OCAPMetadata

class LanguageDataHandler:
    def __init__(self, community_id):
        self.community_id = community_id
        self.metadata = OCAPMetadata()

    def record_audio(self, speaker_id, consent=False):
        """음성 녹음 시 OCAP 준수"""
        if not consent:
            raise PermissionError("Speaker consent required")

        audio_data = self._capture_audio()

        # OCAP 메타데이터 첨부
        self.metadata.add({
            'owner': speaker_id,
            'community': self.community_id,
            'consent_type': 'explicit',
            'usage_restrictions': ['research', 'model_training'],
            'expiry_date': '2026-12-31',
            'attribution_required': True
        })

        return {
            'audio': audio_data,
            'metadata': self.metadata.to_dict()
        }

    def share_data(self, recipient, purpose):
        """데이터 공유 시 권한 확인"""
        if purpose not in self.metadata.get('usage_restrictions'):
            raise PermissionError(f"Purpose '{purpose}' not authorized")

        # 공유 로그 기록
        self.metadata.add_access_log({
            'recipient': recipient,
            'purpose': purpose,
            'timestamp': datetime.now().isoformat()
        })

08.5.3 데이터 암호화

# 저장 데이터 암호화 (at-rest)
from cryptography.fernet import Fernet
import base64

class EncryptedStorage:
    def __init__(self):
        # 키 생성 (안전한 곳에 저장)
        self.key = Fernet.generate_key()
        self.cipher = Fernet(self.key)

    def encrypt_audio(self, audio_bytes):
        """오디오 데이터 암호화"""
        encrypted = self.cipher.encrypt(audio_bytes)
        return encrypted

    def decrypt_audio(self, encrypted_bytes):
        """오디오 데이터 복호화"""
        decrypted = self.cipher.decrypt(encrypted_bytes)
        return decrypted

            'timestamp': datetime.utcnow().isoformat(),
            'level': record.levelname,
            'message': record.getMessage(),
            'module': record.module,
            'function': record.funcName,
        }

        if hasattr(record, 'language'):
            log_data['language'] = record.language
        if hasattr(record, 'duration'):
            log_data['duration_ms'] = record.duration

        return json.dumps(log_data)

# 로거 설정
logger = logging.getLogger('wia-lang')
handler = logging.StreamHandler()
handler.setFormatter(JSONFormatter())
logger.addHandler(handler)
logger.setLevel(logging.INFO)

    })

        duration = (time.time() - start) * 1000

        logger.info(f"Transcription completed", extra={

    except Exception as e:
        logger.error(f"Transcription failed: {str(e)}", extra={

            'error_type': type(e).__name__
        })
        raise

08.6.2 Prometheus 메트릭

# metrics.py
from prometheus_client import Counter, Histogram, Gauge, start_http_server

# 메트릭 정의
transcription_requests = Counter(
    'wia_lang_transcription_requests_total',
    'Total transcription requests',
    ['language', 'status']
)

transcription_duration = Histogram(
    'wia_lang_transcription_duration_seconds',
    'Transcription duration',
    ['language'],
    buckets=[0.1, 0.5, 1.0, 2.0, 5.0, 10.0]
)

active_requests = Gauge(
    'wia_lang_active_requests',
    'Number of active requests'
)

# 계측 데코레이터
def monitor_transcription(func):
    def wrapper(audio_path, language='auto', **kwargs):
        active_requests.inc()

        with transcription_duration.labels(language=language).time():
            try:
                result = func(audio_path, language=language, **kwargs)
                transcription_requests.labels(
                    language=language,
                    status='success'
                ).inc()
                return result
            except Exception as e:
                transcription_requests.labels(
                    language=language,
                    status='error'
                ).inc()
                raise
            finally:
                active_requests.dec()

    return wrapper
