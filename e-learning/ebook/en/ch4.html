4.2 Implementing SCORM Content

SCORM (Sharable Content Object Reference Model) remains the most widely used e-learning standard. Here's how to implement SCORM 2004 4th Edition support.

SCORM API Adapter

// scorm-adapter.ts
export class SCORMAdapter {
private initialized = false;
private dataModel: Map<string, any> = new Map();

Initialize(parameter: string): string {
if (this.initialized) return 'false';
if (parameter !== '') return 'false';

this.initialized = true;
this.setError(0); // No error
return 'true';
}

Terminate(parameter: string): string {
if (!this.initialized) return 'false';
if (parameter !== '') return 'false';

// Send final data to LMS
this.commitData();
this.initialized = false;
return 'true';
}

GetValue(element: string): string {
if (!this.initialized) {
this.setError(301); // Not initialized
return '';
}

const value = this.dataModel.get(element);
this.setError(0);
return value !== undefined ? String(value) : '';
}

SetValue(element: string, value: string): string {
if (!this.initialized) {
this.setError(301);
return 'false';
}

// Validate element and value according to SCORM spec
if (!this.validateElement(element, value)) {
this.setError(406); // Data model element type mismatch
return 'false';
}

this.dataModel.set(element, value);
this.setError(0);
return 'true';
}

Commit(parameter: string): string {
if (!this.initialized) return 'false';
this.commitData();
return 'true';
}

private async commitData() {
const data = {
completion_status: this.dataModel.get('cmi.completion_status'),
success_status: this.dataModel.get('cmi.success_status'),
score: {
raw: this.dataModel.get('cmi.score.raw'),
min: this.dataModel.get('cmi.score.min'),
max: this.dataModel.get('cmi.score.max'),
scaled: this.dataModel.get('cmi.score.scaled')
},
session_time: this.dataModel.get('cmi.session_time'),
location: this.dataModel.get('cmi.location')
};

await fetch('/api/scorm/commit', {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(data)
});
}
}

Implementing the SCORM Runtime

// Initialize SCORM API on window object
window.API_1484_11 = new SCORMAdapter();

// Example usage in SCORM content:
const api = window.API_1484_11;

// Initialize
api.Initialize('');

// Track progress
api.SetValue('cmi.completion_status', 'completed');
api.SetValue('cmi.success_status', 'passed');
api.SetValue('cmi.score.raw', '85');
api.SetValue('cmi.score.scaled', '0.85');

// Commit data
api.Commit('');

// Terminate
api.Terminate('');

Auto-Grading Implementation

async function gradeSubmission(submissionId: string) {
const submission = await client.assessments.getSubmission(submissionId);
const quiz = await client.assessments.get(submission.quizId);

let totalScore = 0;
let maxScore = 0;

for (const response of submission.responses) {
const question = quiz.questions.find(q => q.id === response.questionId);
if (!question) continue;

maxScore += question.points;

switch (question.type) {
case 'multiple-choice':
const selectedOption = question.options.find(
opt => opt.id === response.answer
);
if (selectedOption?.correct) {
totalScore += question.points;
}
break;

case 'multiple-select':
const correctAnswers = question.options
.filter(opt => opt.correct)
.map(opt => opt.id);
const userAnswers = response.answer as string[];

const correctSelections = userAnswers.filter(
ans => correctAnswers.includes(ans)
).length;
const incorrectSelections = userAnswers.filter(
ans => !correctAnswers.includes(ans)
).length;

if (question.partialCredit) {
const score = Math.max(0,
(correctSelections - incorrectSelections) / correctAnswers.length
);
totalScore += question.points * score;
} else if (incorrectSelections === 0 &&
correctSelections === correctAnswers.length) {
totalScore += question.points;
}
break;

case 'essay':
// Queue for manual grading
await client.assessments.queueForGrading(response.id);
break;
}
}

const scaledScore = maxScore > 0 ? totalScore / maxScore : 0;
const passed = scaledScore >= quiz.passingScore;

await client.assessments.updateSubmission(submissionId, {
score: totalScore,
maxScore: maxScore,
scaledScore: scaledScore,
passed: passed,
gradedAt: new Date().toISOString(),
status: 'graded'
});

return { totalScore, maxScore, scaledScore, passed };
}

4.5 Implementing Adaptive Learning

Use learner data to personalize content difficulty and sequencing.

Adaptive Algorithm

interface LearnerProfile {
userId: string;
skillLevels: Map<string, number>; // 0-1 scale
learningVelocity: number; // Concepts per hour
preferredDifficulty: number; // 0-1 scale
}

async function recommendNextContent(
userId: string,
courseId: string
): Promise<LearningObject> {
// Get learner profile and history
const profile = await client.analytics.getLearnerProfile(userId);
const history = await client.analytics.getLearningHistory(userId, courseId);

// Get available content
const availableContent = await client.courses.getContent(courseId, {
excludeCompleted: true,
prerequisitesMet: true
});

// Score each content item
const scores = availableContent.map(content => {
let score = 0;

// Difficulty match
const difficultyDelta = Math.abs(
content.difficulty - profile.preferredDifficulty
);
score += (1 - difficultyDelta) * 40; // 40% weight

// Skill gap
const requiredSkills = content.metadata.skills || [];
const skillGap = requiredSkills
.map(skill => 1 - (profile.skillLevels.get(skill) || 0))
.reduce((a, b) => a + b, 0) / requiredSkills.length;
score += skillGap * 30; // 30% weight

// Learning velocity match
const estimatedTime = content.metadata.estimatedMinutes || 30;
const velocityMatch = estimatedTime / 60 <= profile.learningVelocity;
score += (velocityMatch ? 20 : 0); // 20% weight

// Engagement history
const similarEngagement = history
.filter(h => h.contentType === content.type)
.map(h => h.engagement)
.reduce((a, b) => a + b, 0) / history.length || 0.5;
score += similarEngagement * 10; // 10% weight

return { content, score };
});

// Return highest scoring content
scores.sort((a, b) => b.score - a.score);
return scores[0].content;
}

4.6 LTI Integration

Learning Tools Interoperability (LTI) 1.3 enables secure integration with external tools.

LTI Tool Launch

// LTI 1.3 launch implementation
async function launchLTITool(toolId: string, userId: string, courseId: string) {
// Get tool configuration
const tool = await client.integrations.getLTITool(toolId);

// Generate JWT with LTI claims
const claims = {
iss: 'https://learn.wia.org',
sub: userId,
aud: tool.clientId,
exp: Math.floor(Date.now() / 1000) + 3600,
iat: Math.floor(Date.now() / 1000),
nonce: generateNonce(),
'https://purl.imsglobal.org/spec/lti/claim/message_type': 'LtiResourceLinkRequest',
'https://purl.imsglobal.org/spec/lti/claim/version': '1.3.0',
'https://purl.imsglobal.org/spec/lti/claim/deployment_id': tool.deploymentId,
'https://purl.imsglobal.org/spec/lti/claim/target_link_uri': tool.launchUrl,
'https://purl.imsglobal.org/spec/lti/claim/resource_link': {
id: `course_${courseId}_tool_${toolId}`,
title: tool.title
},
'https://purl.imsglobal.org/spec/lti/claim/roles': [
'http://purl.imsglobal.org/vocab/lis/v2/membership#Learner'
],
'https://purl.imsglobal.org/spec/lti/claim/context': {
id: courseId,
type: ['http://purl.imsglobal.org/vocab/lis/v2/course#CourseOffering']
}
};

const jwt = await signJWT(claims, tool.privateKey);

// Auto-submit form to launch tool
return {
action: tool.launchUrl,
method: 'POST',
fields: {
id_token: jwt,
state: generateState()
}
};
}

4.7 Error Handling and Validation

Robust error handling ensures reliable system operation.

import { WIAError, ErrorCode } from '@wia/e-learning-sdk';

try {
const course = await client.courses.create(courseData);
} catch (error) {
if (error instanceof WIAError) {
switch (error.code) {
case ErrorCode.VALIDATION_ERROR:
console.error('Invalid course data:', error.details);
break;
case ErrorCode.UNAUTHORIZED:
console.error('Authentication failed');
break;
case ErrorCode.RATE_LIMIT_EXCEEDED:
console.error('Too many requests, retry after:', error.retryAfter);
break;
default:
console.error('Unexpected error:', error.message);
}
} else {
console.error('Unknown error:', error);
}
}
