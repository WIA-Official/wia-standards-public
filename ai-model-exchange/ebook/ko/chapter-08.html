8.1 엔드투엔드 모델 교환 워크플로우

강력한 모델 교환 워크플로우는 이전 장의 모든 개념을 학습, 최적화, 버전 관리, 배포, 배포를 아우르는 응집력 있는 파이프라인으로 통합합니다.

완전한 파이프라인 예제

#!/usr/bin/env python3
"""
WIA-AI-008 표준을 따르는 완전한 모델 교환 파이프라인
弘益人間 - 널리 인간을 이롭게 하라
"""

import torch
import mlflow
import onnx
from datetime import datetime

class ModelExchangePipeline:
    def __init__(self, config):
        self.config = config
        mlflow.set_tracking_uri(config['mlflow_uri'])
        mlflow.set_experiment(config['experiment_name'])

    def train(self):
        """단계 1: 모델 학습"""
        print("📚 단계 1: 모델 학습 중...")

        with mlflow.start_run():
            # 모델 학습
            model = self._train_model()

            # 모든 것 로깅
            mlflow.log_params(self.config['hyperparameters'])
            mlflow.log_metrics(self.metrics)
            mlflow.log_artifact("training_logs.txt")

            # PyTorch 모델 저장
            torch.save(model.state_dict(), "model.pth")

            return model

    def optimize(self, model):
        """단계 2: 모델 최적화"""
        print("⚡ 단계 2: 모델 최적화 중...")

        # 양자화
        quantized_model = torch.quantization.quantize_dynamic(
            model, {torch.nn.Linear}, dtype=torch.qint8
        )

        # 프루닝
        # ... 프루닝 코드 ...

        # 정확도 검증
        accuracy_drop = self._validate_optimization(model, quantized_model)
        assert accuracy_drop < 0.02, "정확도 저하가 너무 큽니다"

        return quantized_model

    def export(self, model):
        """단계 3: 여러 형식으로 내보내기"""
        print("📦 단계 3: 여러 형식으로 내보내는 중...")

        # ONNX로 내보내기
        dummy_input = torch.randn(1, *self.config['input_shape'])
        torch.onnx.export(
            model, dummy_input, "model.onnx",
            opset_version=15, export_params=True,
            input_names=['input'], output_names=['output'],
            dynamic_axes={'input': {0: 'batch'}, 'output': {0: 'batch'}}
        )

        # ONNX 검증
        onnx_model = onnx.load("model.onnx")
        onnx.checker.check_model(onnx_model)

        # TorchScript로 내보내기
        scripted_model = torch.jit.script(model)
        scripted_model.save("model.pt")

        print("✓ ONNX 및 TorchScript로 내보냄")

    def create_model_card(self):
        """단계 4: 포괄적인 모델 카드 생성"""
        print("📝 단계 4: 모델 카드 생성 중...")

        model_card = {
            'name': self.config['model_name'],
            'version': self.config['version'],
            'created_at': datetime.now().isoformat(),
            'framework': 'pytorch',
            'architecture': self.config['architecture'],
            'metrics': self.metrics,
            'hyperparameters': self.config['hyperparameters'],
            'intended_use': '이미지 분류',
            'limitations': 'ImageNet 클래스로 제한',
            'ethical_considerations': '학습 데이터의 편향 가능성',
            'philosophy': '弘益人間 - 널리 인간을 이롭게 하라'
        }

        with open("model_card.json", "w") as f:
            json.dump(model_card, f, indent=2)

    def register(self):
        """단계 5: 모델 레지스트리에 등록"""
        print("🏛️ 단계 5: MLflow에 등록 중...")

        mlflow.pytorch.log_model(
            self.model,
            "model",
            registered_model_name=self.config['model_name']
        )

        # 스테이징으로 전환
        client = mlflow.tracking.MlflowClient()
        versions = client.search_model_versions(f"name='{self.config['model_name']}'")
        latest_version = versions[0].version

        client.transition_model_version_stage(
            name=self.config['model_name'],
            version=latest_version,
            stage="Staging"
        )

        print(f"✓ 버전 {latest_version} 등록됨")

    def deploy(self):
        """단계 6: 모델 배포"""
        print("🚀 단계 6: 모델 배포 중...")

        # TorchServe 아카이브 생성
        os.system(f"""
            torch-model-archiver \\
                --model-name {self.config['model_name']} \\
                --version {self.config['version']} \\
                --serialized-file model.pt \\
                --handler image_classifier \\
                --export-path model-store/
        """)

        # TorchServe에 배포
        # ... 배포 코드 ...

        print("✓ 모델 배포됨")

    def run_pipeline(self):
        """전체 파이프라인 실행"""
        print("=" * 60)
        print("WIA-AI-008 모델 교환 파이프라인")
        print("弘益人間 - 널리 인간을 이롭게 하라")
        print("=" * 60)

        model = self.train()
        model = self.optimize(model)
        self.export(model)
        self.create_model_card()
        self.register()
        self.deploy()

        print("\n✅ 파이프라인이 성공적으로 완료되었습니다!")

config = {
    'model_name': 'resnet18-classifier',

    'experiment_name': 'image-classification',
    'mlflow_uri': 'http://localhost:5000',
    'architecture': 'ResNet-18',
    'input_shape': [3, 224, 224],
    'hyperparameters': {
        'learning_rate': 0.001,
        'batch_size': 32,
        'epochs': 10
    }
}

pipeline = ModelExchangePipeline(config)

8.3 사례 연구 2: 크로스 플랫폼 모바일 배포

시나리오: iOS 및 Android에 객체 탐지 배포

한 회사가 PyTorch에서 YOLOv5를 학습하고 일관된 성능을 유지하면서 iOS(CoreML)와 Android(TFLite) 모두에 배포해야 했습니다.

솔루션 아키텍처

# PyTorch (학습) → ONNX → TFLite (Android)
#                            ↓
#                        CoreML (iOS)

# 단계 1: PyTorch를 ONNX로 내보내기
import torch
from models.yolo import YOLOv5

model = YOLOv5.from_pretrained('yolov5s')
model.eval()

dummy_input = torch.randn(1, 3, 640, 640)
torch.onnx.export(
    model, dummy_input, "yolov5s.onnx",
    opset_version=12, input_names=['images'],
    output_names=['output'], dynamic_axes={'images': {0: 'batch'}}
)

# 단계 2: ONNX를 CoreML로 (iOS)
import coremltools as ct
from onnx_coreml import convert

coreml_model = convert(model='yolov5s.onnx')
coreml_model.save('YOLOv5.mlmodel')

# 단계 3: ONNX를 TensorFlow를 거쳐 TFLite로 (Android)
from onnx_tf.backend import prepare
import tensorflow as tf

onnx_model = onnx.load("yolov5s.onnx")
tf_model = prepare(onnx_model)
tf_model.export_graph("yolov5_tf")

converter = tf.lite.TFLiteConverter.from_saved_model("yolov5_tf")
converter.optimizations = [tf.lite.Optimize.DEFAULT]
tflite_model = converter.convert()

with open("yolov5s.tflite", "wb") as f:
    f.write(tflite_model)

# 단계 4: 수치 검증
validate_cross_platform(pytorch_model, coreml_model, tflite_model)

성능 비교

플랫폼 형식 크기 지연 시간 mAP
서버 (PyTorch) .pt 14.4 MB 8ms (GPU) 56.0%
iOS (CoreML) .mlmodel 7.2 MB 45ms (ANE) 55.8%
Android (TFLite) .tflite 7.3 MB 52ms (GPU) 55.7%

교훈

  • 중간 형식으로서의 ONNX는 멀티 플랫폼 배포를 가능하게 했습니다
  • 양자화로 < 1% mAP 손실로 크기를 50% 줄였습니다
  • 플랫폼 간 포괄적인 테스트로 수치 문제를 조기에 발견했습니다
  • 모델 카드는 플랫폼별 최적화를 문서화했습니다

8.4 사례 연구 3: 대규모 MLOps 파이프라인

시나리오: 100개 이상의 모델이 있는 엔터프라이즈 ML 플랫폼

대형 기술 회사가 수십 개 팀의 수백 개 모델을 관리하는 중앙집중식 ML 플랫폼을 구축했습니다.

아키텍처

  • 학습: 분산 학습을 위한 Kubernetes + Kubeflow

  • 서빙: GPU 클러스터의 NVIDIA Triton
  • 모니터링: Prometheus + Grafana
  • CI/CD: 자동화된 테스트가 있는 GitLab CI

자동화된 파이프라인

# .gitlab-ci.yml
stages:
  - train
  - test
  - optimize
  - register
  - deploy

train:
  stage: train
  script:
    - python train.py --config config.yaml
    - python validate.py --threshold 0.90
  artifacts:
    paths:
      - model.pth
      - metrics.json

test:
  stage: test
  script:
    - pytest tests/test_model.py
    - python benchmark.py --latency-threshold 100
  dependencies:
    - train

optimize:
  stage: optimize
  script:
    - python quantize.py --model model.pth
    - python export_onnx.py
    - python validate_onnx.py
  artifacts:
    paths:
      - model_quantized.pth
      - model.onnx

register:
  stage: register
  script:
    - python register_model.py --name $CI_COMMIT_TAG
  only:
    - tags

deploy:
  stage: deploy
  script:
    - python deploy_triton.py --version $CI_COMMIT_TAG
  when: manual
  only:
    - tags

8.8 모델 교환의 미래

신흥 트렌드

  • 연합 학습: 중앙집중화 없이 분산 데이터에서 학습
  • 신경 아키텍처 검색 (NAS): 대상 플랫폼을 위한 자동화된 모델 설계
  • 모델 압축: 고급 기술 (지식 증류, 프루닝, 양자화)
  • 엣지 AI: TinyML을 사용한 온디바이스 추론
  • 모델 마켓플레이스: 상업적 모델 교환 플랫폼
  • AI 규제: 모델 문서화에 대한 규정 준수 요구사항

WIA-AI-008의 역할

WIA-AI-008 표준은 이러한 미래 개발의 기반을 제공하여 상호운용성, 투명성, AI 모델의 윤리적 배포를 보장합니다. 분야가 발전함에 따라 표준은 새로운 모범 사례를 통합하도록 적응하면서 弘益人間(널리 인간을 이롭게 하라)의 핵심 철학을 유지할 것입니다.

8.9 결론

모델 교환은 더 이상 선택 사항이 아닙니다. 현대 ML 운영에 필수적입니다. WIA-AI-008 표준과 이 책에 설명된 모범 사례를 따르면 팀은 다음을 할 수 있습니다:

  • 모델 공유 및 재사용을 통한 개발 가속화
  • 다양한 플랫폼 및 프레임워크에 모델 배포
  • 재현성 및 문서화에 대한 높은 표준 유지
  • 자신감을 가지고 프로덕션 준비가 완료된 ML 시스템 구축
  • 개방형 AI 생태계에 기여

학습에서 프로덕션까지의 여정은 복잡하지만, 올바른 도구, 프로세스, 표준을 사용하면 관리하고 반복할 수 있습니다. 이 책이 모델 교환 요구사항에 대한 포괄적인 가이드 역할을 하기를 바랍니다.

弘益人間 (홍익인간)
"널리 인간을 이롭게 하라" - 이 고대 한국 철학이 WIA-AI-008 표준을 안내합니다. AI 모델을 접근 가능하고, 이식 가능하며, 잘 문서화함으로써 모든 사람에게 이익이 되는 혁신을 가능하게 합니다. 이 여정에 함께해 주셔서 감사합니다.

요약

  • 완전한 엔드투엔드 워크플로우가 모든 모델 교환 개념을 통합합니다
  • 실제 사례 연구가 실용적인 응용 프로그램을 보여줍니다

  • 일반적인 함정과 해결책이 실수를 피하는 데 도움이 됩니다
  • 필수 도구 생태계가 전체 수명주기를 지원합니다
  • 미래 트렌드: 연합 학습, NAS, 엣지 AI, 모델 마켓플레이스
  • WIA-AI-008은 상호운용성과 윤리를 위한 표준화를 제공합니다
  • 弘益人間 철학은 AI가 모든 인류에게 이익이 되도록 보장합니다
