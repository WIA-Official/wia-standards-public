7.1 서빙 인프라 개요

모델 서빙은 학습된 모델을 프로덕션 준비가 완료된 추론 서비스로 변환합니다. 강력한 서빙 인프라는 요청 라우팅, 배칭, 스케일링, 모니터링, 페일오버를 처리합니다. 서빙 프레임워크의 선택은 배포 대상, 성능 요구사항, 운영 제약에 따라 달라집니다.

서빙 패턴

패턴 사용 사례 지연 시간 처리량
실시간 온라인 추론 < 100ms 100-1000 QPS
배치 오프라인 처리 초-시간 100만+ 레코드
스트리밍 이벤트 처리 < 1초 1만-10만 QPS
엣지 온디바이스 추론 < 50ms 로컬 전용

from ts.torch_handler.base_handler import BaseHandler
import torch

class CustomHandler(BaseHandler):
    def preprocess(self, data):
        """
        입력 데이터 전처리
        """
        # 요청에서 입력 추출
        input_data = data[0].get("data") or data[0].get("body")

        # 텐서로 변환
        tensor = torch.FloatTensor(input_data)
        return tensor

    def inference(self, data):
        """
        추론 실행
        """
        with torch.no_grad():
            output = self.model(data)
        return output

    def postprocess(self, data):
        """
        출력 후처리
        """
        # JSON 직렬화를 위해 리스트로 변환
        return data.tolist()

7.5 서버리스 배포

ONNX Runtime을 사용한 AWS Lambda

# lambda_function.py
import json
import onnxruntime as ort
import numpy as np

# 모델 로드 (호출 간 재사용)
session = None

def lambda_handler(event, context):
    global session

    # 콜드 스타트 시 모델 초기화
    if session is None:
        session = ort.InferenceSession("model.onnx")

    # 입력 파싱
    input_data = np.array(json.loads(event['body'])['data'], dtype=np.float32)

    # 추론 실행
    input_name = session.get_inputs()[0].name
    output_name = session.get_outputs()[0].name
    result = session.run([output_name], {input_name: input_data})[0]

    # 응답 반환
    return {
        'statusCode': 200,
        'body': json.dumps({
            'predictions': result.tolist()
        })
    }

7.6 엣지 및 모바일 배포

Android에서 TensorFlow Lite

// Android (Java)
import org.tensorflow.lite.Interpreter;
import java.nio.ByteBuffer;
import java.nio.ByteOrder;

public class ModelInference {
    private Interpreter tflite;

    public ModelInference(Context context) {
        // 자산에서 모델 로드
        try {
            MappedByteBuffer modelBuffer = loadModelFile(context, "model.tflite");
            tflite = new Interpreter(modelBuffer);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public float[][] predict(float[][][] input) {
        // 출력 할당
        float[][] output = new float[1][10];

        // 추론 실행
        tflite.run(input, output);

        return output;
    }

    private MappedByteBuffer loadModelFile(Context context, String modelPath) throws IOException {
        AssetFileDescriptor fileDescriptor = context.getAssets().openFd(modelPath);
        FileInputStream inputStream = new FileInputStream(fileDescriptor.getFileDescriptor());
        FileChannel fileChannel = inputStream.getChannel();
        long startOffset = fileDescriptor.getStartOffset();
        long declaredLength = fileDescriptor.getDeclaredLength();
        return fileChannel.map(FileChannel.MapMode.READ_ONLY, startOffset, declaredLength);
    }
}

iOS에서 CoreML

// Swift
import CoreML
import Vision

class ModelInference {
    var model: VNCoreMLModel?

    init() {
        // CoreML 모델 로드
        guard let mlModel = try? MyModel(configuration: MLModelConfiguration()).model else {
            fatalError("모델 로드 실패")
        }

        model = try? VNCoreMLModel(for: mlModel)
    }

    func predict(image: UIImage, completion: @escaping ([String: Double]?) -> Void) {
        guard let model = model else {
            completion(nil)
            return
        }

        // 요청 생성
        let request = VNCoreMLRequest(model: model) { request, error in
            guard let results = request.results as? [VNClassificationObservation] else {
                completion(nil)
                return
            }

            // 딕셔너리로 변환
            var predictions: [String: Double] = [:]
            for result in results {
                predictions[result.identifier] = Double(result.confidence)
            }

            completion(predictions)
        }

        // 추론 수행
        guard let ciImage = CIImage(image: image) else {
            completion(nil)
            return
        }

        let handler = VNImageRequestHandler(ciImage: ciImage)
        try? handler.perform([request])
    }
}

7.7 배치 추론

Spark 기반 배치 처리

from pyspark.sql import SparkSession
from pyspark.sql.functions import pandas_udf
import pandas as pd
import torch

# Spark 초기화
spark = SparkSession.builder.appName("BatchInference").getOrCreate()

# 데이터 로드
df = spark.read.parquet("s3://data/images.parquet")

# 추론을 위한 UDF 정의
@pandas_udf("array")
def predict_udf(images: pd.Series) -> pd.Series:
    # 모델 로드 (모든 워커에 브로드캐스트)
    model = torch.jit.load("/path/to/model.pt")
    model.eval()

    # 배치 추론
    results = []
    for image_batch in np.array_split(images.values, len(images) // 32):
        with torch.no_grad():
            tensor_batch = torch.FloatTensor(list(image_batch))
            predictions = model(tensor_batch)
            results.extend(predictions.tolist())

    return pd.Series(results)

# UDF 적용
predictions_df = df.withColumn("predictions", predict_udf(df["image"]))

# 결과 저장
predictions_df.write.parquet("s3://results/predictions.parquet")

7.8 모니터링 및 관찰 가능성

메트릭 수집

from prometheus_client import Counter, Histogram, Gauge, start_http_server
import time

# 메트릭 정의
REQUEST_COUNT = Counter('model_requests_total', '총 요청 수', ['model', 'status'])
REQUEST_LATENCY = Histogram('model_request_latency_seconds', '요청 지연 시간', ['model'])
ACTIVE_REQUESTS = Gauge('model_active_requests', '활성 요청 수', ['model'])
MODEL_VERSION = Gauge('model_version', '현재 모델 버전', ['model'])

class MonitoredModel:
    def __init__(self, model, model_name, version):
        self.model = model
        self.model_name = model_name
        MODEL_VERSION.labels(model=model_name).set(version)

    def predict(self, input_data):
        ACTIVE_REQUESTS.labels(model=self.model_name).inc()

        start_time = time.time()
        try:
            result = self.model(input_data)
            REQUEST_COUNT.labels(model=self.model_name, status='success').inc()
            return result
        except Exception as e:
            REQUEST_COUNT.labels(model=self.model_name, status='error').inc()
            raise e
        finally:
            duration = time.time() - start_time
            REQUEST_LATENCY.labels(model=self.model_name).observe(duration)
            ACTIVE_REQUESTS.labels(model=self.model_name).dec()

# Prometheus 엔드포인트 시작
start_http_server(8000)

# 모니터링되는 모델 사용
model = MonitoredModel(torch.jit.load("model.pt"), "resnet18", "1.0.0")
result = model.predict(input_data)

    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

# 추적 구성
trace.set_tracer_provider(TracerProvider())
jaeger_exporter = JaegerExporter(
    agent_host_name="localhost",
    agent_port=6831,
)
trace.get_tracer_provider().add_span_processor(
    BatchSpanProcessor(jaeger_exporter)
)

tracer = trace.get_tracer(__name__)

def predict_with_tracing(model, input_data):
    with tracer.start_as_current_span("model_inference") as span:
        span.set_attribute("model.name", "resnet18")
        span.set_attribute("model.version", "1.0.0")
        span.set_attribute("input.shape", str(input_data.shape))

        # 요청 로깅
        logger.info(json.dumps({
            "event": "inference_request",
            "model": "resnet18",
            "input_shape": list(input_data.shape)
        }))

        # 추론
        start = time.time()
        result = model(input_data)
        latency = time.time() - start

        # 응답 로깅
        logger.info(json.dumps({
            "event": "inference_response",
            "model": "resnet18",
            "latency_ms": latency * 1000,
            "output_shape": list(result.shape)
        }))

        span.set_attribute("inference.latency_ms", latency * 1000)

        return result

7.9 A/B 테스팅 및 카나리 배포

트래픽 분할

import random

class ABTestingRouter:
    def __init__(self, model_a, model_b, traffic_split=0.5):
        self.model_a = model_a
        self.model_b = model_b
        self.traffic_split = traffic_split

    def predict(self, input_data, user_id=None):
        # user_id 기반 결정론적 라우팅
        if user_id:
            route_to_b = hash(user_id) % 100 < (self.traffic_split * 100)
        else:
            route_to_b = random.random() < self.traffic_split

        if route_to_b:
            return self.model_b(input_data), "model_b"
        else:
            return self.model_a(input_data), "model_a"

# 사용법
router = ABTestingRouter(model_v1, model_v2, traffic_split=0.1)  # v2에 10%
result, model_used = router.predict(input_data, user_id="user123")

# 카나리 배포 (점진적 롤아웃)
class CanaryDeployment:
    def __init__(self, stable_model, canary_model):
        self.stable_model = stable_model
        self.canary_model = canary_model
        self.canary_percentage = 0

    def increase_canary(self, step=10):
        self.canary_percentage = min(100, self.canary_percentage + step)

    def predict(self, input_data):
        if random.random() * 100 < self.canary_percentage:
            return self.canary_model(input_data)
        else:
            return self.stable_model(input_data)

# 점진적 롤아웃
canary = CanaryDeployment(model_stable, model_canary)
canary.increase_canary(10)  # 10% 트래픽

7.10 프로덕션 모범 사례

체크리스트

  • 상태 확인: 활성 및 준비 프로브 구현
  • 타임아웃: 적절한 요청 및 추론 타임아웃 설정
  • 속도 제한: 트래픽 급증으로부터 보호
  • 회로 차단기: 다운스트림 서비스가 다운될 때 빠르게 실패
  • 모델 캐싱: 메모리에 모델 캐시, 재로드 방지
  • 입력 검증: 형태, dtype, 범위 검증
  • 오류 처리: 유익한 오류 메시지 반환
  • 버전 관리: API 응답에 모델 버전 포함
  • 우아한 종료: 종료 전 연결 드레인
  • 블루-그린 배포: 무중단 업데이트

요약

  • 모델 서빙 패턴: 실시간, 배치, 스트리밍, 엣지
  • TorchServe는 자동 스케일링을 갖춘 PyTorch 네이티브 서빙을 제공합니다
  • TensorFlow Serving은 고성능 gRPC 및 REST API를 제공합니다
  • NVIDIA Triton은 통합 인프라에서 여러 프레임워크를 지원합니다
  • 서버리스 배포(Lambda, Cloud Functions)는 산발적인 워크로드에 적합합니다
  • 엣지 배포(TFLite, CoreML)는 온디바이스 추론을 위한 것입니다
  • Spark를 사용한 배치 추론으로 대규모 오프라인 처리
  • Prometheus, 로깅, 추적을 통한 포괄적인 모니터링
  • 안전한 롤아웃을 위한 A/B 테스팅 및 카나리 배포
  • 프로덕션 모범 사례: 상태 확인, 속도 제한, 우아한 종료
