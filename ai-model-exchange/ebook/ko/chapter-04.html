4.1 크로스 프레임워크 변환의 과제

현대 AI 개발은 종종 프레임워크 간 모델 변환을 필요로 합니다. 연구팀은 PyTorch에서 학습하지만 TensorFlow Serving으로 배포해야 할 수 있습니다. 모바일 앱은 iOS용 CoreML이나 Android용 TFLite로 TensorFlow 모델을 변환해야 할 수 있습니다. 성공적인 모델 교환을 위해서는 변환 경로와 잠재적 함정을 이해하는 것이 필수적입니다.

일반적인 변환 경로

소스 대상 방법 난이도
PyTorch ONNX torch.onnx.export() 쉬움
TensorFlow ONNX tf2onnx 쉬움
ONNX TensorFlow onnx-tf 중간
PyTorch TFLite PyTorch→ONNX→TF→TFLite 어려움
TensorFlow CoreML coremltools 중간

4.2 PyTorch에서 TensorFlow로 변환

PyTorch에서 TensorFlow로 변환하는 것은 일반적으로 ONNX를 중간 형식으로 사용합니다:

단계별 변환

# 단계 1: PyTorch를 ONNX로
import torch
import torchvision.models as models

model = models.resnet18(pretrained=True)
model.eval()

dummy_input = torch.randn(1, 3, 224, 224)

torch.onnx.export(
    model,
    dummy_input,
    "resnet18.onnx",
    opset_version=13,
    input_names=['input'],
    output_names=['output'],
    dynamic_axes={'input': {0: 'batch'}, 'output': {0: 'batch'}}
)

# 단계 2: ONNX를 TensorFlow로
from onnx_tf.backend import prepare
import onnx

onnx_model = onnx.load("resnet18.onnx")
tf_model = prepare(onnx_model)

# TensorFlow SavedModel로 내보내기
tf_model.export_graph("resnet18_tf")

# 단계 3: TensorFlow에서 로드 및 사용
import tensorflow as tf

loaded_model = tf.saved_model.load("resnet18_tf")
infer = loaded_model.signatures["serving_default"]

# 추론 테스트
import numpy as np
test_input = np.random.randn(1, 3, 224, 224).astype(np.float32)
output = infer(input=tf.constant(test_input))

print(f"출력 형태: {output['output'].shape}")

변환 후 검증

import torch
import tensorflow as tf
import numpy as np

# 테스트 입력 준비
test_input_np = np.random.randn(5, 3, 224, 224).astype(np.float32)

# PyTorch 추론
model_pt = models.resnet18(pretrained=True)
model_pt.eval()
with torch.no_grad():
    output_pt = model_pt(torch.from_numpy(test_input_np)).numpy()

# TensorFlow 추론
model_tf = tf.saved_model.load("resnet18_tf")
infer_tf = model_tf.signatures["serving_default"]
output_tf = infer_tf(input=tf.constant(test_input_np))['output'].numpy()

# 출력 비교
max_diff = np.max(np.abs(output_pt - output_tf))
mean_diff = np.mean(np.abs(output_pt - output_tf))
print(f"최대 차이: {max_diff:.6f}")
print(f"평균 차이: {mean_diff:.6f}")

assert max_diff < 1e-4, "변환 검증 실패!"
print("✓ 변환이 성공적으로 검증되었습니다")

4.3 TensorFlow에서 PyTorch로 변환

역방향 변환도 ONNX를 사용합니다: 

# 단계 1: TensorFlow를 ONNX로
import tensorflow as tf
import tf2onnx

# TensorFlow 모델 로드
model = tf.keras.applications.MobileNetV2(weights='imagenet')

# ONNX로 변환
spec = (tf.TensorSpec((None, 224, 224, 3), tf.float32, name="input"),)
model_proto, _ = tf2onnx.convert.from_keras(model, input_signature=spec, opset=13)

with open("mobilenet_v2.onnx", "wb") as f:
    f.write(model_proto.SerializeToString())

# 단계 2: ONNX를 PyTorch로
import onnx
from onnx2pytorch import ConvertModel

onnx_model = onnx.load("mobilenet_v2.onnx")
pytorch_model = ConvertModel(onnx_model)

# 단계 3: PyTorch에서 사용
import torch

pytorch_model.eval()
test_input = torch.randn(1, 3, 224, 224)

with torch.no_grad():
    output = pytorch_model(test_input)

print(f"출력 형태: {output.shape}")
중요: TensorFlow는 NHWC(batch, height, width, channels) 형식을 사용하는 반면 PyTorch는 NCHW(batch, channels, height, width)를 사용합니다. 변환은 이를 자동으로 처리하지만 디버깅 시 유의하세요.

PyTorch에서 TensorFlow Lite로

Android 및 임베디드 시스템용: 

# 전체 파이프라인: PyTorch → ONNX → TF → TFLite
import torch
import onnx
from onnx_tf.backend import prepare
import tensorflow as tf

# 1. PyTorch를 ONNX로
model_pt = torch.nn.Sequential(
    torch.nn.Conv2d(3, 32, 3, padding=1),
    torch.nn.ReLU(),
    torch.nn.MaxPool2d(2),
    torch.nn.Flatten(),
    torch.nn.Linear(32 * 112 * 112, 10)
)

dummy_input = torch.randn(1, 3, 224, 224)
torch.onnx.export(model_pt, dummy_input, "model.onnx", opset_version=13)

# 2. ONNX를 TensorFlow로
onnx_model = onnx.load("model.onnx")
tf_model = prepare(onnx_model)
tf_model.export_graph("model_tf")

# 3. TensorFlow를 TFLite로
converter = tf.lite.TFLiteConverter.from_saved_model("model_tf")

# 최적화 적용
converter.optimizations = [tf.lite.Optimize.DEFAULT]
converter.target_spec.supported_types = [tf.float16]

tflite_model = converter.convert()

with open("model.tflite", "wb") as f:
    f.write(tflite_model)

print(f"TFLite 모델 크기: {len(tflite_model) / 1024:.2f} KB")

4.5 아키텍처 차이 처리

배치 정규화

PyTorch와 TensorFlow는 BatchNorm을 다르게 구현합니다: 

# PyTorch: momentum = 0.1 (기본값)
# TensorFlow: momentum = 0.99 (기본값)

# PyTorch BatchNorm
bn_pt = torch.nn.BatchNorm2d(64, momentum=0.1)

# 동등한 TensorFlow BatchNorm
bn_tf = tf.keras.layers.BatchNormalization(momentum=0.9)  # 주의: 1 - 0.1 = 0.9

# 변환 시 momentum 값이 조정되는지 확인하세요

패딩 차이

# PyTorch: Conv2d에서 'same' 패딩 사용 불가 (padding 매개변수 사용)
conv_pt = torch.nn.Conv2d(3, 64, kernel_size=3, padding=1)  # 수동 패딩

# TensorFlow: 'same' 패딩 사용 가능
conv_tf = tf.keras.layers.Conv2D(64, 3, padding='same')

# ONNX가 이 변환을 자동으로 처리하지만 수동 변환 시 유의하세요

활성화 함수

# 일부 활성화 함수는 다른 기본 매개변수를 가집니다

# PyTorch LeakyReLU
leaky_pt = torch.nn.LeakyReLU(negative_slope=0.01)

# TensorFlow LeakyReLU
leaky_tf = tf.keras.layers.LeakyReLU(alpha=0.01)  # 다른 매개변수 이름

# GELU 구현이 약간 다를 수 있습니다
gelu_pt = torch.nn.GELU()  # 기본적으로 tanh 근사 사용
gelu_tf = tf.keras.activations.gelu  # 기본적으로 erf(정확한 값) 사용

4.6 고급 변환 기술

사용자 정의 레이어 매핑

from onnx_tf.handlers.backend import BackendHandler
from onnx_tf.handlers.handler import onnx_op
import tensorflow as tf

# 사용자 정의 ONNX 연산자 핸들러 정의
@onnx_op("CustomOp")
class CustomOpHandler(BackendHandler):
    @classmethod
    def version_1(cls, node, **kwargs):
        # 사용자 정의 변환 로직 구현
        x = kwargs["tensor_dict"][node.inputs[0]]
        alpha = node.attrs.get("alpha", 1.0)

        # TensorFlow 연산으로 매핑
        output = tf.multiply(x, alpha)

        return [output]

# 이제 CustomOp이 있는 ONNX 모델을 TensorFlow로 변환할 수 있습니다

4.7 일반적인 변환 문제 해결

문제 1: 지원되지 않는 연산자

# 문제: 대상 프레임워크에서 지원되지 않는 연산자
# 해결: 지원되는 동등한 것으로 교체하거나 사용자 정의 핸들러 구현

# 예: 지원되지 않는 연산 교체
class ModelWithUnsupportedOp(torch.nn.Module):
    def forward(self, x):
        # torch.einsum이 지원되지 않을 수 있습니다
        # 동등한 matmul 연산으로 교체
        return torch.matmul(x, x.transpose(-2, -1))

# 또는 ONNX 폴백 연산자 사용
torch.onnx.export(
    model,
    dummy_input,
    "model.onnx",
    operator_export_type=torch.onnx.OperatorExportTypes.ONNX_FALLTHROUGH
)

문제 3: 수치 정밀도

# 문제: 프레임워크마다 다른 정밀도 기본값
# 해결: dtype을 명시적으로 설정하고 출력 검증

import numpy as np

def validate_conversion(model_original, model_converted, test_inputs, tolerance=1e-5):
    """
    변환이 수치 정확도를 유지하는지 검증
    """
    for i, test_input in enumerate(test_inputs):
        # 원본 출력
        output_original = model_original(test_input).detach().numpy()

        # 변환된 출력
        output_converted = model_converted(test_input).numpy()

        # 차이 계산
        abs_diff = np.abs(output_original - output_converted)
        rel_diff = abs_diff / (np.abs(output_original) + 1e-8)

        max_abs_diff = np.max(abs_diff)
        max_rel_diff = np.max(rel_diff)

        print(f"테스트 {i+1}:")
        print(f"  최대 절대 차이: {max_abs_diff:.2e}")
        print(f"  최대 상대 차이: {max_rel_diff:.2e}")

        if max_abs_diff > tolerance:
            print(f"  ⚠️ 경고: 허용 오차 {tolerance} 초과")
        else:
            print(f"  ✓ 허용 오차 내")

    return True

# 사용법
test_inputs = [torch.randn(1, 3, 224, 224) for _ in range(10)]

4.9 자동화 및 CI/CD 통합

자동화된 변환 파이프라인

#!/usr/bin/env python3
"""
검증이 포함된 자동화된 모델 변환 파이프라인
"""

import torch
import onnx
import tensorflow as tf
from pathlib import Path
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class ModelConverter:
    def __init__(self, source_path, target_format):
        self.source_path = Path(source_path)
        self.target_format = target_format

    def convert(self):
        logger.info(f"{self.source_path}를 {self.target_format}로 변환 중...")

        # 소스 모델 로드
        model = torch.load(self.source_path)
        model.eval()

        # ONNX로 내보내기
        onnx_path = self.source_path.with_suffix('.onnx')
        dummy_input = torch.randn(1, 3, 224, 224)

        torch.onnx.export(
            model, dummy_input, str(onnx_path),
            opset_version=15, export_params=True
        )

        logger.info(f"✓ ONNX로 내보냄: {onnx_path}")

        # ONNX 검증
        onnx_model = onnx.load(str(onnx_path))
        onnx.checker.check_model(onnx_model)
        logger.info("✓ ONNX 모델 검증됨")

        # 대상 형식으로 변환
        if self.target_format == 'tensorflow':
            self._convert_to_tensorflow(onnx_path)
        elif self.target_format == 'tflite':
            self._convert_to_tflite(onnx_path)

        logger.info("✓ 변환 완료")

    def _convert_to_tensorflow(self, onnx_path):
        # 여기에 구현
        pass

    def _convert_to_tflite(self, onnx_path):
        # 여기에 구현
        pass

# CI/CD에서 사용
if __name__ == "__main__":
    converter = ModelConverter("model.pth", "tensorflow")
    converter.convert()

4.10 프레임워크별 최적화

TensorRT 최적화 (NVIDIA GPU)

import tensorrt as trt
import pycuda.driver as cuda

# 최대 GPU 성능을 위해 ONNX를 TensorRT로 변환
TRT_LOGGER = trt.Logger(trt.Logger.WARNING)

def build_engine(onnx_path, engine_path, fp16_mode=True):
    with trt.Builder(TRT_LOGGER) as builder, \
         builder.create_network(1 << int(trt.NetworkDefinitionCreationFlag.EXPLICIT_BATCH)) as network, \
         trt.OnnxParser(network, TRT_LOGGER) as parser:

        # 빌더 구성
        config = builder.create_builder_config()
        config.max_workspace_size = 1 << 30  # 1GB

        if fp16_mode:
            config.set_flag(trt.BuilderFlag.FP16)

        # ONNX 파싱
        with open(onnx_path, 'rb') as model:
            parser.parse(model.read())

        # 엔진 빌드
        engine = builder.build_engine(network, config)

        # 엔진 저장
        with open(engine_path, 'wb') as f:
            f.write(engine.serialize())

        return engine

# TensorRT 엔진 빌드
engine = build_engine("model.onnx", "model.trt", fp16_mode=True)
print("✓ TensorRT 엔진 빌드됨")

요약

  • 크로스 프레임워크 변환으로 다양한 플랫폼에서 모델 배포 가능
  • ONNX는 대부분의 변환을 위한 범용 중간 형식 역할
  • 일반적인 경로: PyTorch ↔ ONNX ↔ TensorFlow, 추가 단계를 통한 모바일 형식
  • 프레임워크 차이(BatchNorm, 패딩, 활성화)는 신중한 처리가 필요
  • 테스트 데이터에 대한 수치 비교로 항상 변환 검증
  • 더 나은 연산자 적용 범위를 위해 최신 ONNX opset 사용
  • 문제를 피하기 위해 변환 전에 모델 단순화
  • 변환 단계 및 검증 결과 문서화
  • CI/CD 통합을 위한 변환 파이프라인 자동화
  • 최상의 성능을 위해 대상별 최적화 적용 (TensorRT, CoreML)
