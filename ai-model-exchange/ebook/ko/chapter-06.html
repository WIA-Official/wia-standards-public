6.1 모델 레지스트리가 필요한 이유

조직이 ML 운영을 확장함에 따라 수십 또는 수백 개의 모델을 관리하는 것이 어려워집니다. 모델 레지스트리는 머신러닝 모델에 대한 중앙집중식 저장소, 버전 관리, 메타데이터 관리, 액세스 제어를 제공합니다. 팀 간 협업, 재현성, 거버넌스를 가능하게 합니다.

주요 이점

  • 중앙집중식 저장소: 모든 모델에 대한 단일 소스
  • 버전 제어: 시간에 따른 모델 진화 추적
  • 메타데이터 관리: 학습 메트릭, 데이터셋, 설정 저장
  • 액세스 제어: 권한 및 감사 로그 관리
  • 배포 추적: 프로덕션에 있는 모델 파악
  • 협업: 팀과 조직 간 모델 공유

from mlflow.server import handlers

def check_permission(user, action, model_name):
    user_role = get_user_role(user)
    permissions = roles.get(user_role, {}).get("permissions", [])

    if "*" in permissions or action in permissions:
        # 모델 액세스 확인
        allowed_models = roles[user_role]["models"]
        if "*" in allowed_models or model_name in allowed_models:
            return True

    return False

모델 서명 및 검증

import hashlib
import hmac

def sign_model(model_path, secret_key):
    """
    모델에 대한 암호화 서명 생성
    """
    with open(model_path, 'rb') as f:
        model_data = f.read()

    signature = hmac.new(
        secret_key.encode(),
        model_data,
        hashlib.sha256
    ).hexdigest()

    return signature

def verify_model(model_path, signature, secret_key):
    """
    모델 무결성 검증
    """
    expected_signature = sign_model(model_path, secret_key)
    return hmac.compare_digest(signature, expected_signature)

# 사용법
secret_key = "my-secret-key"  # 안전하게 저장하세요!

# 모델 서명
signature = sign_model("model.pth", secret_key)
print(f"모델 서명: {signature}")

# 로드 전 검증
if verify_model("model.pth", signature, secret_key):
    model = torch.load("model.pth")
    print("✓ 모델 검증 및 로드됨")
else:
    print("✗ 모델 검증 실패!")

6.8 모델 수명주기 관리

단계 전환

class ModelLifecycle:
    """
    수명주기 단계를 통한 모델 진행 관리
    """
    STAGES = ["Development", "Staging", "Production", "Archived"]

    def __init__(self, registry_client):
        self.client = registry_client

    def promote_to_staging(self, model_name, version, checks=None):
        """
        검증 후 스테이징으로 모델 승격
        """
        if checks:
            # 검증 확인 실행
            if not self.run_checks(model_name, version, checks):
                raise ValueError("모델이 검증 확인에 실패했습니다")

        self.client.transition_model_version_stage(
            name=model_name,
            version=version,
            stage="Staging"
        )

        print(f"✓ {model_name} v{version}이(가) 스테이징으로 승격됨")

    def promote_to_production(self, model_name, version, approval_required=True):
        """
        선택적 승인과 함께 프로덕션으로 승격
        """
        if approval_required:
            approval = input(f"{model_name} v{version}을(를) 프로덕션에 승인하시겠습니까? (yes/no): ")
            if approval.lower() != "yes":
                print("프로덕션 배포가 취소되었습니다")
                return

        # 현재 프로덕션 모델 아카이브
        current_prod = self.get_production_version(model_name)
        if current_prod:
            self.client.transition_model_version_stage(
                name=model_name,
                version=current_prod.version,
                stage="Archived"
            )

        # 새 버전 승격
        self.client.transition_model_version_stage(
            name=model_name,
            version=version,
            stage="Production"
        )

        print(f"✓ {model_name} v{version}이(가) 프로덕션에 배포됨")

    def rollback(self, model_name, to_version):
        """
        이전 버전으로 롤백
        """
        self.client.transition_model_version_stage(
            name=model_name,
            version=to_version,
            stage="Production"
        )

        print(f"✓ {model_name} v{to_version}(으)로 롤백됨")

# 사용법

lifecycle.promote_to_staging("ResNet18", version=3, checks=["accuracy", "latency"])
lifecycle.promote_to_production("ResNet18", version=3)

6.9 레지스트리 연합

다중 레지스트리 아키텍처

class FederatedRegistry:
    """
    여러 레지스트리에서 모델 집계
    """
    def __init__(self):
        self.registries = {
            "mlflow": MLflowRegistry("http://mlflow.company.com"),
            "huggingface": HuggingFaceRegistry("company-org"),
            "s3": S3Registry("s3://company-models")
        }

    def search_models(self, query):
        """
        모든 레지스트리에서 검색
        """
        results = []
        for name, registry in self.registries.items():
            matches = registry.search(query)
            for match in matches:
                match['source'] = name
                results.append(match)

        return results

    def load_model(self, model_uri):
        """
        적절한 레지스트리에서 모델 로드
        """
        # URI 파싱: registry://model-name/version
        registry_name, model_path = self.parse_uri(model_uri)
        registry = self.registries[registry_name]
        return registry.load(model_path)

# 사용법

    """
    모델 등록 전 검증 실행
    """
    tests = {
        "accuracy_threshold": lambda: evaluate_accuracy(model) > 0.90,
        "latency_threshold": lambda: measure_latency(model) < 100,  # ms

    }

    results = {}
    for test_name, test_fn in tests.items():
        try:
            passed = test_fn()
            results[test_name] = "통과" if passed else "실패"
        except Exception as e:
            results[test_name] = f"오류: {str(e)}"

    all_passed = all(v == "통과" for v in results.values())

    return all_passed, results

요약

  • 모델 레지스트리는 중앙집중식 저장소, 버전 관리, 메타데이터 관리를 제공합니다
  • MLflow는 스테이징/프로덕션 워크플로우를 갖춘 오픈 소스 레지스트리를 제공합니다
  • Hugging Face Hub는 포괄적인 모델 카드로 모델 공유에 탁월합니다
  • Git LFS와 DVC는 코드와 함께 대용량 모델의 버전 관리를 가능하게 합니다
  • 포괄적인 메타데이터 표준은 재현성과 거버넌스를 보장합니다
  • RBAC 및 모델 서명은 보안 및 액세스 제어를 제공합니다
  • 수명주기 관리는 모델 승격 및 롤백을 자동화합니다
  • 연합 레지스트리는 여러 소스의 모델을 집계합니다
  • 자동화된 검증은 등록 전 품질을 보장합니다
  • 일관된 명명 및 로깅 관행은 발견 가능성을 향상시킵니다

복습 문제

  1. 모델 레지스트리 사용의 주요 이점은 무엇입니까?
  2. MLflow 모델 버전 관리 및 스테이징 워크플로우를 설명하세요.
  3. Hugging Face Hub에 모델 카드를 생성하고 업로드하는 방법은 무엇입니까?
  4. 모델 버전 관리를 위한 Git LFS와 DVC를 비교하세요. 각각 언제 사용합니까?
  5. 포괄적인 모델 메타데이터에 포함되어야 할 정보는 무엇입니까?
  6. 모델 레지스트리를 위한 역할 기반 액세스 제어(RBAC) 시스템을 설명하세요.
  7. 모델을 암호화 방식으로 서명하고 검증하는 방법은 무엇입니까?
  8. 개발부터 프로덕션까지 단계를 포함하는 모델 수명주기를 설계하세요.
  9. 레지스트리 연합이란 무엇이며 왜 유용합니까?
  10. 새 모델을 등록하기 전에 실행할 검증 체크리스트를 생성하세요.
