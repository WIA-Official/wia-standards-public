3단계: 스마트 계약 프로토콜

스마트 계약 기반 상속

3단계는 블록체인 스마트 계약을 활용하여 신뢰 없는 자동화된 상속 실행을 가능하게 합니다. 이 접근 방식은 중앙화된 제3자에 대한 의존을 제거하고 투명하고 검증 가능한 실행을 보장합니다.

스마트 계약의 장점

  • 신뢰 없음: 제3자 중개자 없이 코드가 규칙을 시행
  • 투명성: 모든 거래와 규칙이 블록체인에 공개적으로 기록
  • 불변성: 배포 후 계약 로직을 변경할 수 없음
  • 자동 실행: 조건이 충족되면 프로그래밍된 대로 실행
  • 비용 효율성: 장기적으로 중개 수수료 제거

다중 서명 지갑

다중 서명(Multi-sig) 지갑은 거래를 승인하기 위해 여러 개인 키가 필요하여 보안을 강화하고 단일 장애 지점을 제거합니다.

기본 Multi-sig 구조

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract InheritanceMultiSig {
    address public owner;
    address[] public executors;
    address[] public beneficiaries;

    uint256 public requiredSignatures;
    uint256 public lastActivity;
    uint256 public inactivityThreshold;

    mapping(address => bool) public isExecutor;
    mapping(bytes32 => mapping(address => bool)) public approvals;

    struct Transaction {
        address to;
        uint256 value;
        bytes data;
        bool executed;
        uint256 approvalCount;
    }

    mapping(bytes32 => Transaction) public transactions;

    event TransactionProposed(bytes32 indexed txId, address indexed proposer);
    event TransactionApproved(bytes32 indexed txId, address indexed approver);
    event TransactionExecuted(bytes32 indexed txId);
    event ActivityRecorded(uint256 timestamp);

    constructor(
        address[] memory _executors,
        address[] memory _beneficiaries,
        uint256 _requiredSignatures,
        uint256 _inactivityThreshold
    ) {
        owner = msg.sender;
        executors = _executors;
        beneficiaries = _beneficiaries;
        requiredSignatures = _requiredSignatures;
        inactivityThreshold = _inactivityThreshold;
        lastActivity = block.timestamp;

        for (uint i = 0; i < _executors.length; i++) {
            isExecutor[_executors[i]] = true;
        }
    }

    function recordActivity() external {
        require(msg.sender == owner, "Only owner can record activity");
        lastActivity = block.timestamp;
        emit ActivityRecorded(block.timestamp);
    }

    function proposeTransaction(
        address _to,
        uint256 _value,
        bytes memory _data
    ) external returns (bytes32) {
        require(isExecutor[msg.sender], "Not an executor");

        bytes32 txId = keccak256(abi.encodePacked(_to, _value, _data, block.timestamp));

        transactions[txId] = Transaction({
            to: _to,
            value: _value,
            data: _data,
            executed: false,
            approvalCount: 0
        });

        emit TransactionProposed(txId, msg.sender);
        return txId;
    }

    function approveTransaction(bytes32 _txId) external {
        require(isExecutor[msg.sender], "Not an executor");
        require(!approvals[_txId][msg.sender], "Already approved");
        require(!transactions[_txId].executed, "Already executed");

        approvals[_txId][msg.sender] = true;
        transactions[_txId].approvalCount++;

        emit TransactionApproved(_txId, msg.sender);

        if (transactions[_txId].approvalCount >= requiredSignatures) {
            executeTransaction(_txId);
        }
    }

    function executeTransaction(bytes32 _txId) internal {
        Transaction storage txn = transactions[_txId];
        require(!txn.executed, "Already executed");

        txn.executed = true;
        (bool success, ) = txn.to.call{value: txn.value}(txn.data);
        require(success, "Transaction execution failed");

        emit TransactionExecuted(_txId);
    }

    function checkInactivity() external view returns (bool) {
        return (block.timestamp - lastActivity) > inactivityThreshold;
    }
}

Deadman Switch 메커니즘

Deadman switch는 소유자가 지정된 기간 동안 활동이 없을 경우 자동으로 트리거되는 메커니즘입니다. 이는 갑작스러운 사망이나 무능력 상태에 대한 안전장치 역할을 합니다.

Deadman Switch 계약

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract DeadmanSwitch {
    address public owner;
    address[] public beneficiaries;
    mapping(address => uint256) public beneficiaryShares; // 백분율 (총 100)

    uint256 public lastCheckIn;
    uint256 public checkInInterval; // 초 단위 (예: 90일 = 7776000초)
    uint256 public warningPeriod; // 경고 기간 (예: 7일)

    bool public triggered;
    bool public executed;

    event CheckIn(uint256 timestamp);
    event WarningIssued(uint256 timestamp, uint256 deadline);
    event Triggered(uint256 timestamp);
    event Executed(uint256 timestamp);
    event FundsDistributed(address indexed beneficiary, uint256 amount);

    constructor(
        address[] memory _beneficiaries,
        uint256[] memory _shares,
        uint256 _checkInInterval,
        uint256 _warningPeriod
    ) {
        require(_beneficiaries.length == _shares.length, "Mismatched arrays");

        owner = msg.sender;
        beneficiaries = _beneficiaries;
        checkInInterval = _checkInInterval;
        warningPeriod = _warningPeriod;
        lastCheckIn = block.timestamp;

        uint256 totalShares = 0;
        for (uint i = 0; i < _beneficiaries.length; i++) {
            beneficiaryShares[_beneficiaries[i]] = _shares[i];
            totalShares += _shares[i];
        }
        require(totalShares == 100, "Shares must total 100");
    }

    function checkIn() external {
        require(msg.sender == owner, "Only owner can check in");
        require(!executed, "Already executed");

        lastCheckIn = block.timestamp;
        triggered = false; // 트리거 리셋

        emit CheckIn(block.timestamp);
    }

    function getTimeUntilTrigger() external view returns (uint256) {
        if (triggered || executed) return 0;

        uint256 deadline = lastCheckIn + checkInInterval;
        if (block.timestamp >= deadline) return 0;

        return deadline - block.timestamp;
    }

    function isInWarningPeriod() public view returns (bool) {
        if (triggered || executed) return false;

        uint256 warningStart = lastCheckIn + checkInInterval - warningPeriod;
        return block.timestamp >= warningStart && block.timestamp < lastCheckIn + checkInInterval;
    }

    function trigger() external {
        require(!triggered, "Already triggered");
        require(!executed, "Already executed");
        require(
            block.timestamp >= lastCheckIn + checkInInterval,
            "Check-in interval not expired"
        );

        triggered = true;
        emit Triggered(block.timestamp);
    }

    function execute() external {
        require(triggered, "Not triggered yet");
        require(!executed, "Already executed");

        // 트리거 후 추가 대기 기간 (예: 30일)
        require(
            block.timestamp >= lastCheckIn + checkInInterval + 30 days,
            "Grace period not expired"
        );

        executed = true;

        uint256 balance = address(this).balance;
        for (uint i = 0; i < beneficiaries.length; i++) {
            address beneficiary = beneficiaries[i];
            uint256 share = beneficiaryShares[beneficiary];
            uint256 amount = (balance * share) / 100;

            (bool success, ) = beneficiary.call{value: amount}("");
            require(success, "Transfer failed");

            emit FundsDistributed(beneficiary, amount);
        }

        emit Executed(block.timestamp);
    }

    receive() external payable {}
}

타임락 계약

타임락 계약은 특정 날짜 또는 블록 번호까지 자산 이전을 지연시킵니다. 이는 미성년 수혜자나 단계적 분배에 유용합니다.

타임락 계약 예제

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract TimeLockInheritance {
    struct Allocation {
        address beneficiary;
        uint256 amount;
        uint256 unlockTime;
        bool claimed;
    }

    address public owner;
    Allocation[] public allocations;

    event AllocationCreated(uint256 indexed id, address beneficiary, uint256 amount, uint256 unlockTime);
    event AllocationClaimed(uint256 indexed id, address beneficiary, uint256 amount);

    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner");
        _;
    }

    constructor() {
        owner = msg.sender;
    }

    function createAllocation(
        address _beneficiary,
        uint256 _amount,
        uint256 _unlockTime
    ) external onlyOwner {
        require(_unlockTime > block.timestamp, "Unlock time must be in future");
        require(_amount > 0, "Amount must be greater than 0");

        allocations.push(Allocation({
            beneficiary: _beneficiary,
            amount: _amount,
            unlockTime: _unlockTime,
            claimed: false
        }));

        emit AllocationCreated(allocations.length - 1, _beneficiary, _amount, _unlockTime);
    }

    function claim(uint256 _allocationId) external {
        require(_allocationId < allocations.length, "Invalid allocation ID");
        Allocation storage allocation = allocations[_allocationId];

        require(msg.sender == allocation.beneficiary, "Not the beneficiary");
        require(block.timestamp >= allocation.unlockTime, "Still locked");
        require(!allocation.claimed, "Already claimed");
        require(address(this).balance >= allocation.amount, "Insufficient balance");

        allocation.claimed = true;

        (bool success, ) = allocation.beneficiary.call{value: allocation.amount}("");
        require(success, "Transfer failed");

        emit AllocationClaimed(_allocationId, allocation.beneficiary, allocation.amount);
    }

    function getUnlockedAllocations(address _beneficiary) external view returns (uint256[] memory) {
        uint256 count = 0;
        for (uint i = 0; i < allocations.length; i++) {
            if (allocations[i].beneficiary == _beneficiary &&
                block.timestamp >= allocations[i].unlockTime &&
                !allocations[i].claimed) {
                count++;
            }
        }

        uint256[] memory result = new uint256[](count);
        uint256 index = 0;
        for (uint i = 0; i < allocations.length; i++) {
            if (allocations[i].beneficiary == _beneficiary &&
                block.timestamp >= allocations[i].unlockTime &&
                !allocations[i].claimed) {
                result[index] = i;
                index++;
            }
        }

        return result;
    }

    receive() external payable {}
}

오라클 통합

오라클은 블록체인 외부의 실제 데이터(예: 사망 증명서)를 스마트 계약에 제공합니다. 이는 법적 이벤트에 기반한 자동 실행을 가능하게 합니다.

Chainlink 오라클 통합

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@chainlink/contracts/src/v0.8/ChainlinkClient.sol";

contract OracleInheritance is ChainlinkClient {
    using Chainlink for Chainlink.Request;

    address public owner;
    address[] public beneficiaries;
    mapping(address => uint256) public shares;

    bytes32 public deathCertificateHash;
    bool public verified;
    bool public executed;

    address private oracle;
    bytes32 private jobId;
    uint256 private fee;

    event VerificationRequested(bytes32 indexed requestId);
    event DeathVerified(bytes32 certificateHash);
    event InheritanceExecuted(uint256 timestamp);

    constructor(
        address _oracle,
        bytes32 _jobId,
        uint256 _fee,
        address[] memory _beneficiaries,
        uint256[] memory _shares
    ) {
        setChainlinkToken(0x...); // Chainlink 토큰 주소
        owner = msg.sender;
        oracle = _oracle;
        jobId = _jobId;
        fee = _fee;

        require(_beneficiaries.length == _shares.length, "Mismatched arrays");
        beneficiaries = _beneficiaries;

        uint256 totalShares = 0;
        for (uint i = 0; i < _beneficiaries.length; i++) {
            shares[_beneficiaries[i]] = _shares[i];
            totalShares += _shares[i];
        }
        require(totalShares == 100, "Shares must total 100");
    }

    function requestDeathVerification(string memory _certificateUrl) external {
        require(!verified, "Already verified");

        Chainlink.Request memory request = buildChainlinkRequest(
            jobId,
            address(this),
            this.fulfillDeathVerification.selector
        );

        request.add("get", _certificateUrl);
        request.add("path", "verified");

        bytes32 requestId = sendChainlinkRequestTo(oracle, request, fee);
        emit VerificationRequested(requestId);
    }

    function fulfillDeathVerification(
        bytes32 _requestId,
        bytes32 _certificateHash
    ) public recordChainlinkFulfillment(_requestId) {
        deathCertificateHash = _certificateHash;
        verified = true;

        emit DeathVerified(_certificateHash);
    }

    function executeInheritance() external {
        require(verified, "Death not verified");
        require(!executed, "Already executed");

        executed = true;
        uint256 balance = address(this).balance;

        for (uint i = 0; i < beneficiaries.length; i++) {
            address beneficiary = beneficiaries[i];
            uint256 amount = (balance * shares[beneficiary]) / 100;

            (bool success, ) = beneficiary.call{value: amount}("");
            require(success, "Transfer failed");
        }

        emit InheritanceExecuted(block.timestamp);
    }

    receive() external payable {}
}

⚠️ 보안 고려사항

  • 감사: 배포 전 모든 스마트 계약은 전문 감사를 받아야 합니다
  • 업그레이드 가능성: 버그 수정을 위한 프록시 패턴 고려
  • 가스 최적화: 높은 가스 비용으로 실행이 차단될 수 있음
  • 재진입 공격: Checks-Effects-Interactions 패턴 사용
  • 오라클 신뢰: 여러 오라클 소스 사용 고려

핵심 요약

  1. 신뢰 없는 실행: 스마트 계약은 중앙 권한 없이 상속 규칙을 자동으로 시행합니다
  2. 다중 서명 보안: Multi-sig 지갑은 단일 장애 지점을 제거하고 분산 제어를 제공합니다
  3. 자동 트리거: Deadman switch는 활동 부족 시 자동으로 상속 프로세스를 시작합니다
  4. 조건부 분배: 타임락은 연령 또는 마일스톤 기반 분배를 가능하게 합니다
  5. 실제 통합: 오라클은 법적 문서를 블록체인 실행과 연결합니다
  6. 감사 필수: 모든 프로덕션 스마트 계약은 전문적인 보안 감사를 받아야 합니다
