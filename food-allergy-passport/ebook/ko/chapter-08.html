코드 예제: 8장 API 사용

// JavaScript/Node.js 예제
const axios = require('axios');

class Chapter8API {
  constructor(apiKey, baseUrl) {
    this.apiKey = apiKey;
    this.baseUrl = baseUrl || 'https://api.wia-foodallergy.org';
  }

  async createResource(data) {
    try {
      const response = await axios.post(
        `${this.baseUrl}/api/v1/chapter8/create`,
        data,
        {
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json',
            'X-API-Version': '1.0.0'
          }
        }
      );
      
      console.log('리소스가 성공적으로 생성되었습니다:', response.data.resourceId);
      return response.data;
    } catch (error) {
      console.error('리소스 생성 오류:', error.message);
      throw error;
    }
  }

  async getResource(resourceId) {
    try {
      const response = await axios.get(
        `${this.baseUrl}/api/v1/chapter8/${resourceId}`,
        {
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'X-API-Version': '1.0.0'
          }
        }
      );
      
      return response.data;
    } catch (error) {
      console.error('리소스 검색 오류:', error.message);
      throw error;
    }
  }

  async validateData(data) {
    try {
      const response = await axios.post(
        `${this.baseUrl}/api/v1/chapter8/validate`,
        data,
        {
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json'
          }
        }
      );
      
      return {
        valid: response.data.valid,
        errors: response.data.errors || [],
        warnings: response.data.warnings || []
      };
    } catch (error) {
      console.error('데이터 검증 오류:', error.message);
      throw error;
    }
  }
}

// 사용법
const api = new Chapter8API('your-api-key-here');

// 리소스 생성
const newResource = await api.createResource({
  name: '샘플 리소스',
  type: 'STANDARD',
  priority: 'HIGH'
});

// 리소스 검색
const resource = await api.getResource(newResource.resourceId);
console.log('검색된 리소스:', resource);

// 데이터 검증
const validation = await api.validateData({
  field1: 'value1',
  field2: 'value2'
});
console.log('검증 결과:', validation);
