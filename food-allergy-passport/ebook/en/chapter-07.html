Chapter 7: Healthcare Integration

The WIA Food Allergy Passport Standard integrates seamlessly with Electronic Health Records (EHR), allergist practices, prescription tracking systems, immunotherapy monitoring platforms, and laboratory result management. This chapter defines standards for bidirectional data exchange with healthcare providers using FHIR (Fast Healthcare Interoperability Resources) and HL7 protocols.

Healthcare Integration Standards

7.3 Healthcare Integration API Endpoints

POST/api/v1/healthcare/ehr/sync

Synchronize allergy profile with EHR system using FHIR R4.

GET/api/v1/healthcare/lab-results

Retrieve allergy test results from connected laboratories.

POST/api/v1/healthcare/prescription/verify

Verify active EpiPen prescription with pharmacy network.

GET/api/v1/healthcare/allergist/appointments

Get upcoming and past allergist appointments.

POST/api/v1/healthcare/immunotherapy/log

Log immunotherapy injection with date, dosage, and reaction.

GET/api/v1/healthcare/treatment-plan

Retrieve current treatment plan from allergist.

POST/api/v1/healthcare/insurance/verify

Verify insurance coverage for allergy treatments.

GET/api/v1/healthcare/providers/search

Find allergists and immunologists by location and specialty.

POST/api/v1/healthcare/referral/request

Request referral to specialist through primary care physician.

GET/api/v1/healthcare/clinical-trials

Find relevant clinical trials for food allergy treatments.

Code Example: EHR Integration

// Python - FHIR EHR Integration
from fhirclient import client
import requests

class EHRIntegration:
    def __init__(self, fhir_base_url, client_id):
        self.settings = {
            'app_id': client_id,
            'api_base': fhir_base_url
        }
        self.smart = client.FHIRClient(settings=self.settings)
    
    def import_allergies_from_ehr(self, patient_id):
        """Import allergy data from EHR to WIA system"""
        search = {
            'patient': patient_id,
            'category': 'food'
        }
        
        allergies = self.smart.server.request_json(
            'AllergyIntolerance',
            params=search
        )
        
        wia_allergies = []
        for allergy in allergies.get('entry', []):
            resource = allergy['resource']
            
            wia_allergy = {
                'allergenId': self.map_snomed_to_wia(resource['code']),
                'severityLevel': self.map_fhir_criticality(resource.get('criticality')),
                'diagnosisDate': resource.get('onsetDateTime'),
                'lastVerified': resource.get('recordedDate'),
                'verifiedBy': self.extract_practitioner(resource.get('asserter')),
                'reactionHistory': self.extract_reactions(resource.get('reaction', []))
            }
            wia_allergies.append(wia_allergy)
        
        return wia_allergies
    
    def export_to_ehr(self, wia_profile):
        """Export WIA profile to EHR in FHIR format"""
        fhir_bundle = {
            'resourceType': 'Bundle',
            'type': 'transaction',
            'entry': []
        }
        
        for allergy in wia_profile['allergies']:
            fhir_allergy = {
                'resource': {
                    'resourceType': 'AllergyIntolerance',
                    'clinicalStatus': {'coding': [{'code': 'active'}]},
                    'verificationStatus': {'coding': [{'code': 'confirmed'}]},
                    'type': 'allergy',
                    'category': ['food'],
                    'criticality': self.map_wia_to_fhir_criticality(allergy['severityLevel']),
                    'code': self.map_wia_to_snomed(allergy['allergenId']),
                    'patient': {'reference': f"Patient/{wia_profile['userId']}"}
                },
                'request': {
                    'method': 'POST',
                    'url': 'AllergyIntolerance'
                }
            }
            fhir_bundle['entry'].append(fhir_allergy)
        
        return self.smart.server.post_json('/', fhir_bundle)
    
    def map_snomed_to_wia(self, code_concept):
        """Map SNOMED CT codes to WIA allergen IDs"""
        snomed_map = {
            '256349002': 'ALLERGEN_PEANUT',
            '91935009': 'ALLERGEN_MILK',
            '91930004': 'ALLERGEN_EGG',
            '227037002': 'ALLERGEN_SHELLFISH'
        }
        snomed_code = code_concept['coding'][0]['code']
        return snomed_map.get(snomed_code, 'ALLERGEN_UNKNOWN')

# Usage
ehr = EHRIntegration('https://ehr.hospital.org/fhir', 'client-id')
allergies = ehr.import_allergies_from_ehr('patient-123')
