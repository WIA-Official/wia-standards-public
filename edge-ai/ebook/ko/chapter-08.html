프로덕션 배포

엣지 AI 배포 파이프라인

프로덕션 환경에서 엣지 AI를 배포하려면 훈련, 최적화, 테스트, 배포, 모니터링의 전체 파이프라인이 필요합니다. 수백만 대의 디바이스에 안정적으로 모델을 배포하고 업데이트하는 것은 엣지 AI의 가장 큰 도전 과제 중 하나입니다.

배포 단계

  1. 모델 훈련: 클라우드/서버에서 FP32로 훈련
  2. 모델 최적화: 양자화, 프루닝, 컴파일
  3. 디바이스 테스트: 타겟 하드웨어에서 검증
  4. 점진적 롤아웃: 소규모 → 전체 배포
  5. 모니터링: 성능, 오류, 사용자 피드백
  6. 반복 개선: A/B 테스트, 재훈련

from PIL import Image

interpreter.allocate_tensors()

input_details = interpreter.get_input_details()
output_details = interpreter.get_output_details()

@app.route('/predict', methods=['POST'])
def predict():
    """이미지 추론 엔드포인트"""
    # 이미지 전처리
    image = Image.open(request.files['image'])
    image = image.resize((224, 224))
    input_data = np.array(image, dtype=np.float32) / 255.0
    input_data = np.expand_dims(input_data, 0)

    # 추론
    interpreter.set_tensor(input_details[0]['index'], input_data)
    interpreter.invoke()
    output_data = interpreter.get_tensor(output_details[0]['index'])

    # 결과 반환
    prediction = int(np.argmax(output_data))
    confidence = float(np.max(output_data))

    return jsonify({
        'prediction': prediction,
        'confidence': confidence
    })

A/B 테스팅

모델 A/B 테스트 프레임워크

import random
import time
from dataclasses import dataclass
from typing import Dict, List

@dataclass
class ModelVariant:
    """모델 변형"""
    name: str
    model: object
    traffic_ratio: float
    metrics: Dict[str, List[float]]

class ABTestingFramework:
    """엣지 AI A/B 테스팅"""

    def __init__(self):
        self.variants = {}
        self.total_requests = 0

    def add_variant(self, name: str, model: object, traffic_ratio: float):
        """모델 변형 추가"""
        self.variants[name] = ModelVariant(
            name=name,
            model=model,
            traffic_ratio=traffic_ratio,
            metrics={
                'latency': [],
                'accuracy': [],
                'errors': []
            }
        )

    def select_variant(self) -> ModelVariant:
        """트래픽 비율에 따라 변형 선택"""
        r = random.random()
        cumulative = 0.0

        for variant in self.variants.values():
            cumulative += variant.traffic_ratio
            if r <= cumulative:
                return variant

        # 폴백
        return list(self.variants.values())[0]

    def predict(self, input_data):
        """A/B 테스트 추론"""
        variant = self.select_variant()

        # 성능 측정
        start = time.perf_counter()
        try:
            result = variant.model.predict(input_data)
            latency = (time.perf_counter() - start) * 1000

            # 메트릭 기록
            variant.metrics['latency'].append(latency)

            return {
                'variant': variant.name,
                'result': result,
                'latency': latency
            }

        except Exception as e:
            variant.metrics['errors'].append(str(e))
            raise

    def get_statistics(self):
        """변형별 통계"""
        stats = {}

        for name, variant in self.variants.items():
            latencies = variant.metrics['latency']
            errors = variant.metrics['errors']

            stats[name] = {
                'requests': len(latencies),
                'avg_latency': np.mean(latencies) if latencies else 0,
                'p95_latency': np.percentile(latencies, 95) if latencies else 0,
                'error_rate': len(errors) / max(len(latencies), 1),
                'traffic_ratio': variant.traffic_ratio
            }

        return stats

    def compare_variants(self):
        """변형 비교 및 승자 선택"""
        stats = self.get_statistics()

        print("=== A/B 테스트 결과 ===")
        for name, metrics in stats.items():
            print(f"\n{name}:")
            print(f"  요청 수: {metrics['requests']}")
            print(f"  평균 지연: {metrics['avg_latency']:.2f}ms")
            print(f"  P95 지연: {metrics['p95_latency']:.2f}ms")
            print(f"  오류율: {metrics['error_rate']:.2%}")

        # 승자 판단 (낮은 지연 + 낮은 오류)
        best = min(
            stats.items(),
            key=lambda x: x[1]['avg_latency'] * (1 + x[1]['error_rate'])
        )

        print(f"\n✓ 승자: {best[0]}")
        return best[0]

통계적 유의성 검증

from scipy import stats

def statistical_significance_test(variant_a, variant_b, alpha=0.05):
    """
    두 변형의 통계적 유의성 검증
    (Welch's t-test)
    """
    latencies_a = variant_a.metrics['latency']
    latencies_b = variant_b.metrics['latency']

    # t-검정
    t_stat, p_value = stats.ttest_ind(
        latencies_a,
        latencies_b,
        equal_var=False  # Welch's test
    )

    is_significant = p_value < alpha

    print(f"t-통계량: {t_stat:.4f}")
    print(f"p-값: {p_value:.4f}")
    print(f"유의수준 {alpha}에서 유의함: {is_significant}")

    if is_significant:
        better = 'A' if np.mean(latencies_a) < np.mean(latencies_b) else 'B'
        print(f"→ 변형 {better}가 통계적으로 우수합니다")

    return is_significant, p_value

점진적 롤아웃

Phased Rollout 전략

단계 대상 트래픽 기간 롤백 기준
Canary 내부 테스터 1% 1-2일 오류율 >1%
Alpha 얼리어답터 10% 3-5일 오류율 >0.5%
Beta 일반 사용자 25% 1주 성능 저하 >5%
Production 전체 사용자 100% 지속 심각한 버그

자동 롤아웃 시스템

import time
from enum import Enum

class RolloutStage(Enum):
    CANARY = 0.01
    ALPHA = 0.10
    BETA = 0.25
    PRODUCTION = 1.0

class ProgressiveRollout:
    """점진적 롤아웃 관리"""

    def __init__(self, model_v1, model_v2):
        self.stable_model = model_v1
        self.new_model = model_v2
        self.current_stage = RolloutStage.CANARY
        self.metrics = {'stable': [], 'new': []}
        self.rollout_start = time.time()

    def get_model(self, user_id: str):
        """사용자별 모델 선택"""
        # 사용자 ID 해시로 일관된 할당
        hash_value = hash(user_id) % 100 / 100.0

        if hash_value < self.current_stage.value:
            return self.new_model, 'new'
        else:
            return self.stable_model, 'stable'

    def record_metrics(self, variant: str, latency: float, error: bool):
        """메트릭 기록"""
        self.metrics[variant].append({
            'latency': latency,
            'error': error,
            'timestamp': time.time()
        })

    def evaluate_stage(self) -> bool:
        """현재 단계 평가"""
        new_metrics = self.metrics['new']
        stable_metrics = self.metrics['stable']

        if len(new_metrics) < 100:
            print("데이터 부족 - 대기 중")
            return True

        # 오류율 계산
        new_errors = sum(1 for m in new_metrics if m['error'])
        new_error_rate = new_errors / len(new_metrics)

        stable_errors = sum(1 for m in stable_metrics if m['error'])
        stable_error_rate = stable_errors / len(stable_metrics) if stable_metrics else 0

        # 평균 지연시간
        new_latency = np.mean([m['latency'] for m in new_metrics])
        stable_latency = np.mean([m['latency'] for m in stable_metrics]) if stable_metrics else 0

        print(f"\n=== {self.current_stage.name} 단계 평가 ===")
        print(f"새 모델 - 오류율: {new_error_rate:.2%}, 지연: {new_latency:.1f}ms")
        print(f"안정 모델 - 오류율: {stable_error_rate:.2%}, 지연: {stable_latency:.1f}ms")

        # 롤백 기준
        if new_error_rate > 0.01:  # 1% 오류율
            print("⚠️ 오류율 초과 - 롤백!")
            return False

        if new_latency > stable_latency * 1.1:  # 10% 지연 증가
            print("⚠️ 성능 저하 - 롤백!")
            return False

        print("✓ 단계 통과")
        return True

    def advance_stage(self):
        """다음 단계로 진행"""
        stages = list(RolloutStage)
        current_idx = stages.index(self.current_stage)

        if current_idx < len(stages) - 1:
            self.current_stage = stages[current_idx + 1]
            print(f"→ {self.current_stage.name} 단계로 진행 ({self.current_stage.value:.0%})")
            # 메트릭 초기화
            self.metrics = {'stable': [], 'new': []}
        else:
            print("🎉 전체 롤아웃 완료!")

    def rollback(self):
        """롤백"""
        print(f"🔄 {self.current_stage.name}에서 롤백")
        self.current_stage = RolloutStage.CANARY
        self.metrics = {'stable': [], 'new': []}

# 사용 예
rollout = ProgressiveRollout(model_v1, model_v2)

while rollout.current_stage != RolloutStage.PRODUCTION:
    # 요청 처리
    for user_id in users:
        model, variant = rollout.get_model(user_id)
        latency, error = process_request(model, user_id)
        rollout.record_metrics(variant, latency, error)

    # 주기적 평가
    if rollout.evaluate_stage():
        rollout.advance_stage()
    else:
        rollout.rollback()
        break

import psutil
import json
import requests
from datetime import datetime

class EdgeDeviceMonitor:
    """엣지 디바이스 모니터링"""

    def __init__(self, device_id: str, api_endpoint: str):
        self.device_id = device_id
        self.api_endpoint = api_endpoint
        self.metrics_buffer = []

    def collect_system_metrics(self):
        """시스템 메트릭 수집"""
        return {
            'timestamp': datetime.now().isoformat(),
            'device_id': self.device_id,
            'cpu_percent': psutil.cpu_percent(interval=1),
            'memory_percent': psutil.virtual_memory().percent,
            'disk_percent': psutil.disk_usage('/').percent,
            'temperature': self.get_temperature()
        }

    def get_temperature(self):
        """디바이스 온도 (Raspberry Pi 예)"""
        try:
            with open('/sys/class/thermal/thermal_zone0/temp', 'r') as f:
                temp = int(f.read()) / 1000.0
            return temp
        except:
            return None

    def collect_model_metrics(self, latency: float, prediction: int,
                             confidence: float):
        """모델 메트릭 수집"""
        metric = {
            'timestamp': datetime.now().isoformat(),
            'device_id': self.device_id,
            'latency_ms': latency,
            'prediction': prediction,
            'confidence': confidence
        }
        self.metrics_buffer.append(metric)

        # 버퍼가 가득 차면 전송
        if len(self.metrics_buffer) >= 100:
            self.flush_metrics()

    def flush_metrics(self):
        """메트릭 서버로 전송"""
        if not self.metrics_buffer:
            return

        try:
            response = requests.post(
                f"{self.api_endpoint}/metrics",
                json=self.metrics_buffer,
                timeout=5
            )
            if response.status_code == 200:
                print(f"메트릭 {len(self.metrics_buffer)}개 전송 완료")
                self.metrics_buffer = []
        except Exception as e:
            print(f"메트릭 전송 실패: {e}")

    def check_health(self):
        """헬스체크"""
        metrics = self.collect_system_metrics()

        # 알림 조건
        if metrics['cpu_percent'] > 90:
            self.send_alert('HIGH_CPU', metrics['cpu_percent'])

        if metrics['memory_percent'] > 90:
            self.send_alert('HIGH_MEMORY', metrics['memory_percent'])

        if metrics['temperature'] and metrics['temperature'] > 80:
            self.send_alert('HIGH_TEMP', metrics['temperature'])

    def send_alert(self, alert_type: str, value: float):
        """알림 전송"""
        alert = {
            'device_id': self.device_id,
            'type': alert_type,
            'value': value,
            'timestamp': datetime.now().isoformat()
        }
        requests.post(f"{self.api_endpoint}/alerts", json=alert)

# 사용 예

while True:

Fallback 전략

class ResilientEdgeAI:
    """장애 대응 엣지 AI"""

    def __init__(self, primary_model, fallback_model=None):
        self.primary_model = primary_model
        self.fallback_model = fallback_model
        self.error_count = 0
        self.error_threshold = 3

    def predict(self, input_data, timeout=1.0):
        """복원력 있는 추론"""
        try:
            # 1차: 주 모델
            result = self.predict_with_timeout(
                self.primary_model,
                input_data,
                timeout
            )
            self.error_count = 0  # 성공 시 초기화
            return result

        except TimeoutError:
            print("⚠️ 주 모델 타임아웃")
            self.error_count += 1

        except Exception as e:
            print(f"⚠️ 주 모델 오류: {e}")
            self.error_count += 1

        # 2차: Fallback 모델
        if self.fallback_model:
            try:
                print("→ Fallback 모델 사용")
                return self.predict_with_timeout(
                    self.fallback_model,
                    input_data,
                    timeout
                )
            except Exception as e:
                print(f"⚠️ Fallback 모델도 실패: {e}")

        # 3차: 기본값 반환
        print("→ 기본 예측 반환")
        return self.default_prediction()

    def predict_with_timeout(self, model, input_data, timeout):
        """타임아웃 적용 추론"""
        import signal

        def timeout_handler(signum, frame):
            raise TimeoutError("추론 타임아웃")

        # 타임아웃 설정
        signal.signal(signal.SIGALRM, timeout_handler)
        signal.alarm(int(timeout))

        try:
            result = model.predict(input_data)
        finally:
            signal.alarm(0)  # 타임아웃 해제

        return result

    def default_prediction(self):
        """기본 예측 (안전한 기본값)"""
        return {
            'prediction': 0,
            'confidence': 0.5,
            'fallback': True
        }

    def health_check(self):
        """헬스체크"""
        if self.error_count >= self.error_threshold:
            print(f"🔴 오류 {self.error_count}회 - 모델 재시작 필요")
            return False
        return True

# 사용 예

result = robust_ai.predict(input_data)

import subprocess

class EdgeAIWatchdog:
    """엣지 AI 워치독"""

    def __init__(self, process_cmd: str, check_interval: int = 60):
        self.process_cmd = process_cmd
        self.check_interval = check_interval
        self.process = None
        self.restart_count = 0
        self.max_restarts = 5

    def start_process(self):
        """프로세스 시작"""
        self.process = subprocess.Popen(
            self.process_cmd.split(),
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE
        )
        print(f"프로세스 시작: PID {self.process.pid}")

    def is_healthy(self) -> bool:
        """헬스체크"""
        try:
            response = requests.get('http://localhost:8080/health', timeout=5)
            return response.status_code == 200
        except:
            return False

    def restart_process(self):
        """프로세스 재시작"""
        print("프로세스 재시작 중...")

        # 기존 프로세스 종료
        if self.process:
            self.process.terminate()
            self.process.wait(timeout=10)

        # 재시작
        self.start_process()
        self.restart_count += 1

        # 재시작 횟수 제한
        if self.restart_count >= self.max_restarts:
            print(f"⚠️ 최대 재시작 횟수({self.max_restarts}) 초과")
            return False

        return True

    def run(self):
        """워치독 실행"""
        self.start_process()

        while True:
            time.sleep(self.check_interval)

            # 프로세스 상태 확인
            if self.process.poll() is not None:
                print("프로세스 종료 감지")
                if not self.restart_process():
                    break

            # 헬스체크
            elif not self.is_healthy():
                print("헬스체크 실패")
                if not self.restart_process():
                    break
            else:
                print("✓ 정상 작동 중")

# 사용 예
watchdog = EdgeAIWatchdog('python inference.py')
watchdog.run()

요약

프로덕션 엣지 AI 배포는 체계적인 파이프라인이 필요합니다:

  • CI/CD: GitHub Actions, Docker, 자동화된 테스트
  • A/B 테스팅: 모델 변형 비교, 통계적 유의성 검증
  • 점진적 롤아웃: Canary → Alpha → Beta → Production
  • 모니터링: 시스템 메트릭, 모델 성능, Prometheus + Grafana
  • 오류 처리: Fallback 모델, 타임아웃, 자동 재시작
  • 사례: Tesla Autopilot, Amazon Alexa, Google Photos
  • 체크리스트: 정확도, 지연, 메모리, 보안, 모니터링

안정적인 배포와 모니터링은 엣지 AI 성공의 핵심입니다.
