연합 엣지 학습

연합 학습이란?

연합 학습(Federated Learning)은 데이터를 중앙 서버로 전송하지 않고 여러 엣지 디바이스에서 협업하여 모델을 훈련하는 분산 머신러닝 기법입니다. Google이 2016년 제안했으며, 프라이버시 보호가 핵심 목표입니다.

핵심 개념

  • 로컬 훈련: 각 디바이스가 자신의 데이터로 모델 훈련
  • 모델 집계: 서버가 업데이트를 수집하고 평균화
  • 데이터 프라이버시: 원본 데이터는 디바이스에 유지
  • 통신 효율: 모델 가중치만 전송 (데이터 대비 작음)

전통적 학습 vs 연합 학습

특성 전통적 학습 연합 학습
데이터 위치 중앙 서버 엣지 디바이스
프라이버시 낮음 (데이터 전송) 높음 (데이터 유지)
통신량 매우 높음 중간 (모델만)
훈련 속도 빠름 느림 (비동기)
데이터 다양성 균일 비균일 (Non-IID)

FedAvg 알고리즘

기본 원리

Federated Averaging(FedAvg)는 가장 기본적인 연합 학습 알고리즘입니다:

# FedAvg 알고리즘 의사 코드

# 서버측
global_model = 초기_모델()

for round in range(총_라운드):
    # 1. 클라이언트 샘플링
    selected_clients = random.sample(전체_클라이언트, K개)

    # 2. 글로벌 모델 배포
    for client in selected_clients:
        client.receive_model(global_model)

    # 3. 로컬 훈련 (병렬)
    local_updates = []
    for client in selected_clients:
        update = client.local_train(E개_에포크)
        local_updates.append(update)

    # 4. 집계 (가중 평균)
    global_model = aggregate(local_updates)

# 클라이언트측
def local_train(global_model, E개_에포크):
    local_model = copy(global_model)

    for epoch in range(E개_에포크):
        for batch in 로컬_데이터:
            loss = compute_loss(local_model, batch)
            gradients = compute_gradients(loss)
            local_model = apply_gradients(local_model, gradients)

    # 업데이트만 전송 (전체 모델 또는 델타)
    update = local_model - global_model
    return update

PyTorch 구현

import torch
import torch.nn as nn
import copy

class FederatedServer:
    def __init__(self, model, num_clients=100, fraction=0.1):
        self.global_model = model
        self.num_clients = num_clients
        self.clients_per_round = int(num_clients * fraction)

    def aggregate(self, client_models, client_weights):
        """가중 평균으로 모델 집계"""
        global_dict = self.global_model.state_dict()

        for key in global_dict.keys():
            global_dict[key] = torch.zeros_like(global_dict[key])

            # 각 클라이언트 모델의 가중 합
            for client_model, weight in zip(client_models, client_weights):
                client_dict = client_model.state_dict()
                global_dict[key] += client_dict[key] * weight

        self.global_model.load_state_dict(global_dict)
        return self.global_model

    def train_round(self, clients):
        """한 라운드 훈련"""
        # 1. 클라이언트 선택
        selected = random.sample(clients, self.clients_per_round)

        # 2. 로컬 훈련
        client_models = []
        client_weights = []

        for client in selected:
            local_model = client.train(copy.deepcopy(self.global_model))
            client_models.append(local_model)
            client_weights.append(len(client.dataset))

        # 3. 정규화
        total_weight = sum(client_weights)
        client_weights = [w / total_weight for w in client_weights]

        # 4. 집계
        self.aggregate(client_models, client_weights)

class FederatedClient:
    def __init__(self, dataset, client_id):
        self.dataset = dataset
        self.client_id = client_id

    def train(self, model, epochs=5, lr=0.01):
        """로컬 훈련"""
        model.train()
        optimizer = torch.optim.SGD(model.parameters(), lr=lr)
        criterion = nn.CrossEntropyLoss()

        for epoch in range(epochs):
            for data, target in self.dataset:
                optimizer.zero_grad()
                output = model(data)
                loss = criterion(output, target)
                loss.backward()
                optimizer.step()

        return model

# 사용 예
model = MyModel()
server = FederatedServer(model, num_clients=100)

for round_num in range(100):
    print(f"라운드 {round_num}")
    server.train_round(clients)

    # 평가
    accuracy = evaluate(server.global_model, test_data)
    print(f"정확도: {accuracy:.2%}")

차등 프라이버시

개념

차등 프라이버시(Differential Privacy)는 개별 데이터를 보호하면서 전체 통계를 공개할 수 있게 합니다:

  • ε-차등 프라이버시: 프라이버시 손실 한계 (ε이 작을수록 강한 보호)
  • 노이즈 추가: Gaussian 또는 Laplace 노이즈
  • Gradient Clipping: 민감도 제한
  • Privacy Budget: 누적 프라이버시 손실 관리

DP-SGD 구현

import torch
import torch.nn as nn
import numpy as np

class DPOptimizer:
    """차등 프라이버시가 적용된 옵티마이저"""

    def __init__(self, optimizer, noise_multiplier=1.0, max_grad_norm=1.0):
        self.optimizer = optimizer
        self.noise_multiplier = noise_multiplier
        self.max_grad_norm = max_grad_norm

    def clip_gradients(self, model):
        """그래디언트 클리핑"""
        total_norm = 0.0
        for p in model.parameters():
            if p.grad is not None:
                param_norm = p.grad.data.norm(2)
                total_norm += param_norm.item() ** 2
        total_norm = total_norm ** 0.5

        # 클리핑
        clip_coef = self.max_grad_norm / (total_norm + 1e-6)
        if clip_coef < 1:
            for p in model.parameters():
                if p.grad is not None:
                    p.grad.data.mul_(clip_coef)

        return total_norm

    def add_noise(self, model):
        """그래디언트에 Gaussian 노이즈 추가"""
        for p in model.parameters():
            if p.grad is not None:
                noise = torch.normal(
                    mean=0,
                    std=self.noise_multiplier * self.max_grad_norm,
                    size=p.grad.shape,
                    device=p.grad.device
                )
                p.grad.data.add_(noise)

    def step(self, model):
        """DP 최적화 스텝"""
        # 1. 그래디언트 클리핑
        self.clip_gradients(model)

        # 2. 노이즈 추가
        self.add_noise(model)

        # 3. 옵티마이저 스텝
        self.optimizer.step()

# 사용 예
model = MyModel()
optimizer = torch.optim.SGD(model.parameters(), lr=0.01)
dp_optimizer = DPOptimizer(optimizer, noise_multiplier=1.0, max_grad_norm=1.0)

for data, target in dataloader:
    optimizer.zero_grad()
    output = model(data)
    loss = criterion(output, target)
    loss.backward()

    # DP 적용된 업데이트
    dp_optimizer.step(model)

프라이버시 예산 계산

import math

def compute_privacy_budget(steps, batch_size, total_data,
                          noise_multiplier, delta=1e-5):
    """
    RDP(Rényi Differential Privacy) 기반 ε 계산
    """
    q = batch_size / total_data  # 샘플링 비율

    # RDP 계산 (간소화된 버전)
    orders = [1 + x / 10.0 for x in range(1, 100)]
    rdp = []

    for alpha in orders:
        rdp_alpha = (q ** 2 * alpha) / (2 * noise_multiplier ** 2)
        rdp.append(rdp_alpha * steps)

    # RDP → DP 변환
    eps_values = []
    for alpha, rdp_val in zip(orders, rdp):
        eps = rdp_val + math.log(1/delta) / (alpha - 1)
        eps_values.append(eps)

    epsilon = min(eps_values)

    print(f"프라이버시 보장: ({epsilon:.2f}, {delta})-DP")
    return epsilon

# 예: 1000 스텝, 배치 64, 총 10000개 데이터
epsilon = compute_privacy_budget(
    steps=1000,
    batch_size=64,
    total_data=10000,
    noise_multiplier=1.1,
    delta=1e-5
)
# 출력: 프라이버시 보장: (2.43, 1e-05)-DP

비동기 연합 학습

동기 vs 비동기

연합 학습은 동기 또는 비동기 방식으로 작동할 수 있습니다:

방식 장점 단점
동기 (FedAvg) 안정적, 이해하기 쉬움 느린 클라이언트 대기
비동기 (FedAsync) 빠른 수렴, Straggler 무시 모델 불일치, 복잡함

FedAsync 구현

import threading
import queue
import time

class AsyncFederatedServer:
    def __init__(self, model, mixing_parameter=0.9):
        self.global_model = model
        self.mixing_param = mixing_parameter
        self.update_queue = queue.Queue()
        self.version = 0
        self.lock = threading.Lock()

    def apply_update(self, client_model, client_version):
        """비동기 업데이트 적용"""
        with self.lock:
            # Staleness 계산
            staleness = self.version - client_version

            # Staleness에 따른 가중치 조정
            alpha = self.mixing_param * (0.9 ** staleness)

            # 모델 업데이트
            global_dict = self.global_model.state_dict()
            client_dict = client_model.state_dict()

            for key in global_dict.keys():
                # 혼합: new = (1-α)*global + α*client
                global_dict[key] = (
                    (1 - alpha) * global_dict[key] +
                    alpha * client_dict[key]
                )

            self.global_model.load_state_dict(global_dict)
            self.version += 1

            print(f"업데이트 적용 (staleness={staleness}, α={alpha:.3f})")

    def get_model(self):
        """현재 글로벌 모델 반환"""
        with self.lock:
            return copy.deepcopy(self.global_model), self.version

class AsyncFederatedClient:
    def __init__(self, dataset, server):
        self.dataset = dataset
        self.server = server

    def run(self):
        """비동기로 계속 훈련"""
        while True:
            # 1. 현재 모델 가져오기
            local_model, version = self.server.get_model()

            # 2. 로컬 훈련
            trained_model = self.train(local_model)

            # 3. 업데이트 전송
            self.server.apply_update(trained_model, version)

# 사용 예
server = AsyncFederatedServer(model)

# 클라이언트를 별도 스레드로 실행
clients = [AsyncFederatedClient(dataset_i, server) for i in range(10)]
threads = [threading.Thread(target=c.run) for c in clients]

for t in threads:
    t.start()

Non-IID 데이터 처리

Non-IID 문제

실제 엣지 환경에서 데이터는 Independent and Identically Distributed(IID)가 아닙니다:

  • Label Skew: 각 디바이스가 특정 클래스 편향
  • Feature Skew: 센서, 지역, 사용자 차이
  • Quantity Skew: 데이터량 불균형
  • Temporal Skew: 시간에 따른 분포 변화

FedProx 알고리즘

Non-IID 데이터를 위한 개선된 연합 학습:

def fedprox_local_train(global_model, local_data, mu=0.01):
    """
    FedProx: 글로벌 모델과의 차이를 제한
    Loss = Data_Loss + (μ/2) * ||w - w_global||²
    """
    local_model = copy.deepcopy(global_model)
    optimizer = torch.optim.SGD(local_model.parameters(), lr=0.01)

    # 글로벌 모델 파라미터 저장
    global_params = {
        name: param.clone().detach()
        for name, param in global_model.named_parameters()
    }

    for epoch in range(epochs):
        for data, target in local_data:
            optimizer.zero_grad()
            output = local_model(data)

            # 데이터 손실
            data_loss = criterion(output, target)

            # Proximal 항 추가
            proximal_term = 0.0
            for name, param in local_model.named_parameters():
                proximal_term += torch.norm(
                    param - global_params[name]
                ) ** 2

            # 총 손실
            loss = data_loss + (mu / 2) * proximal_term

            loss.backward()
            optimizer.step()

    return local_model

import torch

def quantize_model(model, bits=8):
    """모델 가중치 양자화"""
    quantized = {}

    for name, param in model.state_dict().items():
        # FP32 → INT8 변환
        param_flat = param.flatten()
        min_val = param_flat.min()
        max_val = param_flat.max()

        # 양자화
        scale = (max_val - min_val) / (2 ** bits - 1)
        quantized_param = torch.round(
            (param - min_val) / scale
        ).to(torch.int8)

        quantized[name] = {
            'data': quantized_param,
            'min': min_val,
            'scale': scale
        }

    return quantized

def dequantize_model(quantized_dict):
    """역양자화"""
    model_dict = {}

    for name, qdata in quantized_dict.items():
        # INT8 → FP32 변환
        dequantized = (
            qdata['data'].float() * qdata['scale'] + qdata['min']
        )
        model_dict[name] = dequantized.reshape(qdata['shape'])

    return model_dict

# Gradient Sparsification
def sparsify_gradients(gradients, top_k_percent=0.1):
    """상위 K% 그래디언트만 전송"""
    flat_grads = torch.cat([g.flatten() for g in gradients])

    # 상위 K% 인덱스 찾기
    k = int(len(flat_grads) * top_k_percent)
    threshold = torch.topk(flat_grads.abs(), k).values.min()

    # 마스크 생성
    sparse_grads = []
    for grad in gradients:
        mask = grad.abs() >= threshold
        sparse_grads.append(grad * mask)

    return sparse_grads

弘益人間 연합 학습 원칙:

연합 학습은 프라이버시를 보호하면서도 집단 지식의 힘을 활용합니다. 개인 의료 기록을 공유하지 않고도 더 나은 진단 모델을, 타이핑 데이터를 서버로 보내지 않고도 더 스마트한 키보드를 만들어 모든 인류에게 이익을 가져다줍니다.

요약

연합 엣지 학습은 프라이버시를 보호하면서 협업 AI를 실현합니다:

  • 핵심: 데이터를 이동하지 않고 모델을 이동
  • FedAvg: 로컬 훈련 + 서버 집계 (가중 평균)
  • 차등 프라이버시: DP-SGD, 노이즈 추가, 그래디언트 클리핑
  • Non-IID: FedProx로 데이터 불균형 처리
  • 통신 최적화: 양자화, 스파시피케이션, 압축
  • 응용: Gboard, Siri, 의료 진단, 금융
  • 프레임워크: TensorFlow Federated, PySyft, Flower

연합 학습은 엣지 AI의 미래이며 프라이버시 보호 AI의 핵심 기술입니다.
