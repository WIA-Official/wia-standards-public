하드웨어 가속기

엣지 AI 가속기 개요

엣지 AI 추론을 위한 하드웨어 가속기는 CPU보다 10-100배 빠른 성능과 5-50배 향상된 에너지 효율을 제공합니다. 모바일 기기, IoT 디바이스, 자율주행차, 로봇 등 다양한 엣지 환경에서 실시간 AI를 가능하게 합니다.

주요 가속기 유형

  • NPU (Neural Processing Unit): 신경망 연산 전용 칩
  • GPU (Graphics Processing Unit): 병렬 연산에 최적화
  • DSP (Digital Signal Processor): 신호 처리 및 저전력 추론
  • FPGA (Field-Programmable Gate Array): 커스터마이징 가능한 하드웨어
  • ASIC (Application-Specific IC): 특정 용도 전용 칩

아키텍처

Apple Neural Engine은 iPhone, iPad, Mac의 A/M 시리즈 칩에 통합된 NPU입니다. 주요 특징:

  • A17 Pro: 16코어 Neural Engine, 35 TOPS
  • M3: 16코어 Neural Engine, 18 TOPS
  • M3 Max: 40코어 Neural Engine, 38 TOPS
  • 최적화: CoreML 프레임워크 통합

CoreML 활용

Apple Neural Engine을 활용하려면 CoreML 프레임워크를 사용합니다:

// Swift로 CoreML 모델 사용
import CoreML
import Vision

// 1. 모델 로드
guard let model = try? VNCoreMLModel(for: MobileNetV3().model) else {
    fatalError("모델을 로드할 수 없습니다")
}

// 2. 요청 생성
let request = VNCoreMLRequest(model: model) { request, error in
    guard let results = request.results as? [VNClassificationObservation] else {
        return
    }

    // 3. 결과 처리
    if let topResult = results.first {
        print("예측: \(topResult.identifier)")
        print("신뢰도: \(topResult.confidence)")
    }
}

// 4. 이미지 처리 (Neural Engine에서 자동 실행)
let handler = VNImageRequestHandler(ciImage: inputImage)
try? handler.perform([request])

Google Edge TPU

아키텍처

Edge TPU는 Google이 개발한 초저전력 AI 가속기로, TensorFlow Lite 모델 전용입니다:

  • 성능: 4 TOPS @ 2W
  • 인터페이스: USB, PCIe, M.2
  • 지원: INT8 양자화 모델만
  • 용도: IoT, 임베디드, 산업용

모델 컴파일

Edge TPU용으로 모델을 컴파일하는 과정:

# 1. TensorFlow 모델을 TFLite로 변환 (INT8 양자화 필수)
import tensorflow as tf

converter = tf.lite.TFLiteConverter.from_keras_model(model)
converter.optimizations = [tf.lite.Optimize.DEFAULT]

def representative_dataset():
    for _ in range(100):
        data = np.random.rand(1, 224, 224, 3).astype(np.float32)
        yield [data]

converter.representative_dataset = representative_dataset
converter.target_spec.supported_ops = [tf.lite.OpsSet.TFLITE_BUILTINS_INT8]
converter.inference_input_type = tf.uint8
converter.inference_output_type = tf.uint8

tflite_model = converter.convert()

# 2. Edge TPU Compiler로 컴파일
# edgetpu_compiler model.tflite

# 3. Python에서 추론
from pycoral.utils import edgetpu
from pycoral.adapters import common
from pycoral.adapters import classify
from PIL import Image

# Edge TPU 초기화
interpreter = edgetpu.make_interpreter('model_edgetpu.tflite')
interpreter.allocate_tensors()

# 입력 준비
size = common.input_size(interpreter)
image = Image.open('test.jpg').resize(size, Image.LANCZOS)

# 추론 실행
common.set_input(interpreter, image)
interpreter.invoke()

# 결과 추출
classes = classify.get_classes(interpreter, top_k=3)
for c in classes:
    print(f'{c.id}: {c.score:.4f}')

SNPE (Snapdragon Neural Processing Engine)

Qualcomm의 딥러닝 프레임워크:

# 1. 모델을 SNPE DLC로 변환
# TensorFlow → DLC
snpe-tensorflow-to-dlc --input_network model.pb \
    --input_dim input "1,224,224,3" \
    --out_node output \
    --output_path model.dlc

# 2. 양자화 (선택적, 성능 향상)
snpe-dlc-quantize --input_dlc model.dlc \
    --input_list images.txt \
    --output_dlc model_quantized.dlc

# 3. Python으로 추론
import numpy as np
from snpe import modeltools

# DLC 로드
container = modeltools.SNPE_DLC_IO()
container.open('model_quantized.dlc')

# 런타임 생성 (DSP 사용)
runtime = modeltools.SNPE_RUNTIME_DSP

# 입력 준비
input_data = np.random.rand(1, 224, 224, 3).astype(np.float32)

# 추론
output = container.execute(input_data, runtime)

print(f"예측 결과: {np.argmax(output)}")

TensorRT 최적화

NVIDIA TensorRT로 모델을 최적화합니다:

# Python으로 TensorRT 엔진 생성
import tensorrt as trt
import pycuda.driver as cuda
import pycuda.autoinit

# 1. TensorRT 빌더 생성
logger = trt.Logger(trt.Logger.WARNING)
builder = trt.Builder(logger)
network = builder.create_network(
    1 << int(trt.NetworkDefinitionCreationFlag.EXPLICIT_BATCH)
)

# 2. ONNX 파서로 모델 파싱
parser = trt.OnnxParser(network, logger)
with open('model.onnx', 'rb') as f:
    parser.parse(f.read())

# 3. 빌더 설정
config = builder.create_builder_config()
config.set_memory_pool_limit(trt.MemoryPoolType.WORKSPACE, 1 << 30)  # 1GB

# FP16 모드 활성화 (성능 2배 향상)
if builder.platform_has_fast_fp16:
    config.set_flag(trt.BuilderFlag.FP16)

# INT8 양자화 (성능 4배 향상, 캘리브레이션 필요)
# config.set_flag(trt.BuilderFlag.INT8)

# 4. 엔진 빌드
serialized_engine = builder.build_serialized_network(network, config)

# 5. 엔진 저장
with open('model.trt', 'wb') as f:
    f.write(serialized_engine)

# 6. 추론 실행
runtime = trt.Runtime(logger)
engine = runtime.deserialize_cuda_engine(serialized_engine)
context = engine.create_execution_context()

# GPU 메모리 할당
d_input = cuda.mem_alloc(1 * input_size)
d_output = cuda.mem_alloc(1 * output_size)

# 추론
cuda.memcpy_htod(d_input, input_data)
context.execute_v2([int(d_input), int(d_output)])
cuda.memcpy_dtoh(output_data, d_output)

print(f"예측: {np.argmax(output_data)}")

# 5. 입력 설정

infer_request.infer({0: input_tensor})

# 6. 결과 추출
output = infer_request.get_output_tensor(0).data
print(f"예측: {np.argmax(output)}")

성능 측정 코드

# Python으로 추론 성능 측정
import time
import numpy as np

def benchmark_model(model, input_shape, num_runs=100):
    """모델 추론 성능 측정"""
    # 워밍업
    dummy_input = np.random.rand(*input_shape).astype(np.float32)
    for _ in range(10):
        _ = model.predict(dummy_input)

    # 측정
    latencies = []
    for _ in range(num_runs):
        start = time.perf_counter()
        _ = model.predict(dummy_input)
        end = time.perf_counter()
        latencies.append((end - start) * 1000)  # ms

    # 통계
    latencies = np.array(latencies)
    print(f"평균 지연시간: {latencies.mean():.2f} ms")
    print(f"중앙값: {np.median(latencies):.2f} ms")
    print(f"P95: {np.percentile(latencies, 95):.2f} ms")
    print(f"P99: {np.percentile(latencies, 99):.2f} ms")
    print(f"처리량: {1000 / latencies.mean():.1f} FPS")

# 사용 예
benchmark_model(tflite_model, (1, 224, 224, 3))
