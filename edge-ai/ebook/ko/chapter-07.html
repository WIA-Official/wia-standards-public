엣지 AI의 프라이버시 이점

엣지 AI는 데이터를 클라우드로 전송하지 않고 로컬에서 처리하여 프라이버시를 근본적으로 향상시킵니다. 사용자 데이터가 디바이스를 떠나지 않으므로 데이터 유출, 중간자 공격, 서버 침해의 위험이 크게 감소합니다.

보안 엔클레이브

Trusted Execution Environment (TEE)

TEE는 메인 프로세서에서 격리된 보안 영역으로, 민감한 데이터와 코드를 보호합니다:

  • ARM TrustZone: 모바일/임베디드 플랫폼
  • Intel SGX: PC 및 서버
  • Apple Secure Enclave: iOS, macOS
  • Google Titan M: Pixel 스마트폰

// Swift로 Secure Enclave에서 키 생성 및 암호화
import Security
import CryptoKit

func createSecureEnclaveKey() throws -> SecKey {
    // Secure Enclave에 키 생성
    let access = SecAccessControlCreateWithFlags(
        kCFAllocatorDefault,
        kSecAttrAccessibleWhenUnlockedThisDeviceOnly,
        .privateKeyUsage,
        nil
    )!

    let attributes: [String: Any] = [
        kSecAttrKeyType as String: kSecAttrKeyTypeECSECPrimeRandom,
        kSecAttrKeySizeInBits as String: 256,
        kSecAttrTokenID as String: kSecAttrTokenIDSecureEnclave,
        kSecPrivateKeyAttrs as String: [
            kSecAttrIsPermanent as String: true,
            kSecAttrApplicationTag as String: "com.example.key",
            kSecAttrAccessControl as String: access
        ]
    ]

    var error: Unmanaged?
    guard let privateKey = SecKeyCreateRandomKey(attributes as CFDictionary, &error) else {
        throw error!.takeRetainedValue() as Error
    }

    return privateKey
}

func encryptData(data: Data, publicKey: SecKey) throws -> Data {
    // Secure Enclave 키로 암호화
    var error: Unmanaged?
    guard let encrypted = SecKeyCreateEncryptedData(
        publicKey,
        .eciesEncryptionStandardX963SHA256AESGCM,
        data as CFData,
        &error
    ) else {
        throw error!.takeRetainedValue() as Error
    }

    return encrypted as Data
}

#include "tee_internal_api.h"

// Trusted Application (TA) 진입점
TEE_Result TA_InvokeCommandEntryPoint(
    void *session,
    uint32_t cmd_id,
    uint32_t param_types,
    TEE_Param params[4])
{
    switch (cmd_id) {
    case CMD_SECURE_INFERENCE:
        return secure_ml_inference(params);

    case CMD_ENCRYPT_MODEL:
        return encrypt_model_weights(params);

    default:
        return TEE_ERROR_BAD_PARAMETERS;
    }
}

TEE_Result secure_ml_inference(TEE_Param params[4]) {
    // Secure World에서 모델 추론 실행
    uint8_t *input = params[0].memref.buffer;
    uint32_t input_size = params[0].memref.size;

    // 암호화된 모델 로드 (Secure Storage)
    TEE_ObjectHandle model_obj;
    TEE_OpenPersistentObject(
        TEE_STORAGE_PRIVATE,
        "encrypted_model",
        TEE_DATA_FLAG_ACCESS_READ,
        &model_obj
    );

    // 복호화 및 추론
    float *output = run_inference(input, input_size);

    // 결과 반환 (Normal World로)
    memcpy(params[1].memref.buffer, output, sizeof(float) * 10);

    return TEE_SUCCESS;
}

모델 암호화

모델 가중치를 암호화하여 추출을 방지합니다:

import numpy as np
from cryptography.fernet import Fernet
import pickle

class EncryptedModel:
    """암호화된 모델 래퍼"""

    def __init__(self, model, key=None):
        self.model = model
        self.key = key if key else Fernet.generate_key()
        self.cipher = Fernet(self.key)

    def encrypt_weights(self):
        """모델 가중치 암호화"""
        weights = self.model.get_weights()

        # NumPy 배열을 바이트로 직렬화
        weights_bytes = pickle.dumps(weights)

        # 암호화
        encrypted = self.cipher.encrypt(weights_bytes)

        return encrypted

    def decrypt_weights(self, encrypted_weights):
        """가중치 복호화"""
        # 복호화
        weights_bytes = self.cipher.decrypt(encrypted_weights)

        # 역직렬화
        weights = pickle.loads(weights_bytes)

        return weights

    def save_encrypted(self, filepath):
        """암호화된 모델 저장"""
        encrypted = self.encrypt_weights()

        with open(filepath, 'wb') as f:
            f.write(encrypted)

        # 키는 별도로 안전하게 저장
        with open(filepath + '.key', 'wb') as f:
            f.write(self.key)

    def load_encrypted(self, filepath, key_path):
        """암호화된 모델 로드"""
        # 키 로드
        with open(key_path, 'rb') as f:
            self.key = f.read()
            self.cipher = Fernet(self.key)

        # 암호화된 가중치 로드
        with open(filepath, 'rb') as f:
            encrypted = f.read()

        # 복호화 및 가중치 설정
        weights = self.decrypt_weights(encrypted)
        self.model.set_weights(weights)

    def predict(self, x):
        """일반 추론 (복호화된 상태)"""
        return self.model.predict(x)

# 사용 예
encrypted_model = EncryptedModel(keras_model)
encrypted_model.save_encrypted('model_encrypted.bin')

# 배포된 디바이스에서

적대적 견고성

적대적 예제

입력에 미세한 변조를 가해 모델을 속이는 공격:

import torch
import torch.nn as nn

def fgsm_attack(model, images, labels, epsilon=0.03):
    """
    Fast Gradient Sign Method (FGSM) 공격
    """
    images.requires_grad = True

    # 순전파
    outputs = model(images)
    loss = nn.CrossEntropyLoss()(outputs, labels)

    # 역전파로 그래디언트 계산
    model.zero_grad()
    loss.backward()

    # 그래디언트 부호 방향으로 입력 변조
    sign_grad = images.grad.sign()
    adversarial_images = images + epsilon * sign_grad

    # 픽셀 값 범위 유지 [0, 1]
    adversarial_images = torch.clamp(adversarial_images, 0, 1)

    return adversarial_images

# 예제: 정상 이미지 vs 적대적 이미지
original_pred = model(images).argmax(dim=1)
print(f"원본 예측: {original_pred}")

adv_images = fgsm_attack(model, images, labels)
adv_pred = model(adv_images).argmax(dim=1)
print(f"적대적 예측: {adv_pred}")

# 이미지 변화량
diff = (adv_images - images).abs().mean()
print(f"평균 변화량: {diff:.6f}")  # 매우 작음

방어: Adversarial Training

def adversarial_training(model, train_loader, epochs=10, epsilon=0.03):
    """
    적대적 훈련으로 견고성 향상
    """
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    criterion = nn.CrossEntropyLoss()

    for epoch in range(epochs):
        for images, labels in train_loader:
            # 1. 정상 데이터로 훈련
            optimizer.zero_grad()
            outputs = model(images)
            loss_clean = criterion(outputs, labels)

            # 2. 적대적 예제 생성
            adv_images = fgsm_attack(model, images, labels, epsilon)

            # 3. 적대적 예제로도 훈련
            outputs_adv = model(adv_images)
            loss_adv = criterion(outputs_adv, labels)

            # 4. 두 손실 결합
            loss = 0.5 * loss_clean + 0.5 * loss_adv

            loss.backward()
            optimizer.step()

        print(f"Epoch {epoch}: Loss={loss:.4f}")

    return model

# 견고한 모델 훈련
robust_model = adversarial_training(model, train_loader)

PGD (Projected Gradient Descent) 공격

FGSM보다 강력한 반복적 공격:

def pgd_attack(model, images, labels, epsilon=0.03, alpha=0.01, iters=40):
    """
    PGD 공격: FGSM을 여러 번 반복
    """
    adv_images = images.clone().detach()

    for i in range(iters):
        adv_images.requires_grad = True

        outputs = model(adv_images)
        loss = nn.CrossEntropyLoss()(outputs, labels)

        model.zero_grad()
        loss.backward()

        # 작은 스텝으로 이동
        adv_images = adv_images + alpha * adv_images.grad.sign()

        # epsilon 범위 내로 제한
        eta = torch.clamp(adv_images - images, min=-epsilon, max=epsilon)
        adv_images = torch.clamp(images + eta, min=0, max=1).detach()

    return adv_images

안전한 모델 업데이트

OTA (Over-The-Air) 업데이트 보안

엣지 디바이스의 모델을 안전하게 업데이트하는 메커니즘:

import hashlib
import hmac
import requests
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import padding, rsa

class SecureModelUpdater:
    """안전한 모델 OTA 업데이트"""

    def __init__(self, server_public_key):
        self.server_public_key = server_public_key

    def download_update(self, url):
        """모델 업데이트 다운로드"""
        response = requests.get(url)

        return {
            'model': response.json()['model'],
            'signature': response.json()['signature'],
            'hash': response.json()['hash']
        }

    def verify_signature(self, data, signature):
        """디지털 서명 검증"""
        try:
            self.server_public_key.verify(
                bytes.fromhex(signature),
                data.encode(),
                padding.PSS(
                    mgf=padding.MGF1(hashes.SHA256()),
                    salt_length=padding.PSS.MAX_LENGTH
                ),
                hashes.SHA256()
            )
            return True
        except:
            return False

    def verify_hash(self, data, expected_hash):
        """해시 검증"""
        actual_hash = hashlib.sha256(data.encode()).hexdigest()
        return actual_hash == expected_hash

    def apply_update(self, url):
        """안전하게 업데이트 적용"""
        # 1. 다운로드
        update_package = self.download_update(url)

        # 2. 해시 검증
        if not self.verify_hash(update_package['model'],
                               update_package['hash']):
            raise ValueError("해시 불일치 - 변조 감지")

        # 3. 서명 검증
        if not self.verify_signature(update_package['model'],
                                     update_package['signature']):
            raise ValueError("서명 불일치 - 신뢰할 수 없는 출처")

        # 4. 업데이트 적용
        model_data = update_package['model']
        self.load_model(model_data)

        print("모델 업데이트 성공 (검증 완료)")

    def rollback(self):
        """업데이트 실패 시 이전 버전으로 복구"""
        # 이전 모델 복원
        pass

# 사용 예

try:
    updater.apply_update('https://api.example.com/model/v2')
except ValueError as e:

점진적 롤아웃 (Canary Deployment)

class CanaryDeployment:
    """안전한 점진적 모델 배포"""

    def __init__(self, stable_model, canary_model, canary_ratio=0.1):
        self.stable_model = stable_model
        self.canary_model = canary_model
        self.canary_ratio = canary_ratio
        self.metrics = {'stable': [], 'canary': []}

    def predict(self, x):
        """일부 트래픽만 새 모델로"""
        if np.random.random() < self.canary_ratio:
            # Canary 모델 사용
            result = self.canary_model.predict(x)
            self.metrics['canary'].append(result)
            return result
        else:
            # 안정 모델 사용
            result = self.stable_model.predict(x)
            self.metrics['stable'].append(result)
            return result

    def evaluate_canary(self):
        """Canary 성능 평가"""
        stable_acc = np.mean(self.metrics['stable'])
        canary_acc = np.mean(self.metrics['canary'])

        print(f"Stable: {stable_acc:.2%}")
        print(f"Canary: {canary_acc:.2%}")

        # Canary가 2% 이상 나쁘면 롤백
        if canary_acc < stable_acc - 0.02:
            print("⚠️ Canary 성능 저하 - 롤백")
            return False

        # Canary가 좋으면 비율 증가
        if canary_acc >= stable_acc:
            self.canary_ratio = min(1.0, self.canary_ratio + 0.1)
            print(f"✓ Canary 비율 증가: {self.canary_ratio:.0%}")

        return True

# 사용 예
deployment = CanaryDeployment(model_v1, model_v2, canary_ratio=0.1)

for batch in data_stream:
    prediction = deployment.predict(batch)

    # 주기적 평가
    if len(deployment.metrics['canary']) > 100:
        deployment.evaluate_canary()

GDPR 및 규정 준수

GDPR 요구사항

EU 일반 데이터 보호 규정(GDPR) 준수:

  • 데이터 최소화: 필요한 최소한의 데이터만 수집
  • 목적 제한: 명시된 목적으로만 사용
  • 삭제권: 사용자 요청 시 데이터 삭제
  • 이동권: 데이터를 다른 서비스로 이동
  • 투명성: 데이터 사용 방법 설명

온디바이스 데이터 관리

import sqlite3
import json

class PrivacyCompliantStorage:
    """GDPR 준수 로컬 데이터 저장소"""

    def __init__(self, db_path='user_data.db'):
        self.conn = sqlite3.connect(db_path)
        self.create_tables()

    def create_tables(self):
        """테이블 생성"""
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS user_data (
                user_id TEXT PRIMARY KEY,
                data TEXT,
                purpose TEXT,
                collected_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                consent BOOLEAN DEFAULT 0
            )
        ''')

        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS data_processing_log (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id TEXT,
                action TEXT,
                timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        ''')

    def store_data(self, user_id, data, purpose, consent=False):
        """사용자 동의 하에 데이터 저장"""
        if not consent:
            raise ValueError("사용자 동의 필요")

        self.conn.execute(
            'INSERT OR REPLACE INTO user_data VALUES (?, ?, ?, CURRENT_TIMESTAMP, ?)',
            (user_id, json.dumps(data), purpose, consent)
        )

        self.log_action(user_id, f"데이터 저장 (목적: {purpose})")
        self.conn.commit()

    def delete_user_data(self, user_id):
        """사용자 데이터 완전 삭제 (삭제권)"""
        self.conn.execute('DELETE FROM user_data WHERE user_id = ?', (user_id,))
        self.log_action(user_id, "데이터 삭제")
        self.conn.commit()
        print(f"사용자 {user_id} 데이터 삭제 완료")

    def export_user_data(self, user_id):
        """사용자 데이터 내보내기 (이동권)"""
        cursor = self.conn.execute(
            'SELECT * FROM user_data WHERE user_id = ?',
            (user_id,)
        )

        data = cursor.fetchone()
        if data:
            return {
                'user_id': data[0],
                'data': json.loads(data[1]),
                'purpose': data[2],
                'collected_at': data[3]
            }
        return None

    def get_processing_log(self, user_id):
        """데이터 처리 기록 조회 (투명성)"""
        cursor = self.conn.execute(
            'SELECT action, timestamp FROM data_processing_log WHERE user_id = ?',
            (user_id,)
        )

        return cursor.fetchall()

    def log_action(self, user_id, action):
        """처리 기록"""
        self.conn.execute(
            'INSERT INTO data_processing_log (user_id, action) VALUES (?, ?)',
            (user_id, action)
        )

프라이버시 보호 분석

온디바이스 분석

서버로 전송하지 않고 디바이스에서 분석:

class OnDeviceAnalytics:
    """프라이버시 보호 분석"""

    def __init__(self):
        self.local_metrics = {}

    def track_event(self, event_name, properties=None):
        """이벤트 로컬 기록"""
        if event_name not in self.local_metrics:
            self.local_metrics[event_name] = []

        self.local_metrics[event_name].append({
            'timestamp': time.time(),
            'properties': properties or {}
        })

    def aggregate_locally(self):
        """로컬에서 집계 (개인 정보 제거)"""
        aggregated = {}

        for event, records in self.local_metrics.items():
            aggregated[event] = {
                'count': len(records),
                'first_seen': min(r['timestamp'] for r in records),
                'last_seen': max(r['timestamp'] for r in records)
            }

        return aggregated

    def send_differential_private_stats(self, epsilon=1.0):
        """차등 프라이버시 적용하여 전송"""
        stats = self.aggregate_locally()

        # 노이즈 추가
        for event in stats:
            noise = np.random.laplace(0, 1/epsilon)
            stats[event]['count'] += int(noise)
            stats[event]['count'] = max(0, stats[event]['count'])

        # 집계만 전송, 원본 이벤트는 전송하지 않음
        return stats

# 사용 예
analytics = OnDeviceAnalytics()
analytics.track_event('model_inference', {'latency_ms': 15})
analytics.track_event('model_inference', {'latency_ms': 18})

# 차등 프라이버시 적용하여 전송
dp_stats = analytics.send_differential_private_stats(epsilon=1.0)
