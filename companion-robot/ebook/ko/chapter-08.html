// src/models/CompanionProfile.ts
import { v7 as uuidv7 } from 'uuid';

interface CompanionProfile {
  id: string;
  version: string;
  createdAt: string;
  lastUpdated: string;
  name: string;
  avatar?: string;
  description: string;
  personality: PersonalityTraits;
  primaryPurpose: CompanionPurpose;
  targetAgeRange: AgeRange;
  languagePrimary: string;
  languagesSupported: string[];
  capabilities: string[];
  limitations: string[];
}

class CompanionProfileService {
  create(data: Partial): CompanionProfile {
    const now = new Date().toISOString();

    return {
      id: uuidv7(),
      version: '1.0.0',
      createdAt: now,
      lastUpdated: now,
      ...data,
      // 기본값 설정
      personality: data.personality || this.getDefaultPersonality(),
      languagesSupported: data.languagesSupported || [data.languagePrimary]
    } as CompanionProfile;
  }

  validate(profile: CompanionProfile): boolean {
    // UUID v7 검증
    if (!this.isValidUUIDv7(profile.id)) return false;

    // 타임스탬프 검증
    if (!this.isValidISO8601(profile.createdAt)) return false;

    // 성격 특성 검증
    if (!this.validatePersonality(profile.personality)) return false;

    return true;
  }
}

8.5.1 위기 대응 프로토콜

// src/protocols/crisis.ts
class CrisisProtocol {
  async handle(message: string): Promise {
    // 1. 위기 감지
    const detection = await this.detectCrisis(message);

    if (!detection.isCrisis) {
      return { isCrisis: false };
    }

    // 2. 심각도 평가
    const severity = this.assessSeverity(detection);

    // 3. 즉각 돌봄 제공
    const immediateResponse = this.getImmediateCare(severity);

    // 4. 자원 제공
    const resources = this.getResources(detection.type);

    // 5. 전문가 연결 필요 여부
    const escalate = severity === 'critical' || severity === 'high';

    // 6. 후속 조치
    const followUp = this.scheduleFollowUp(severity);

    return {
      isCrisis: true,
      severity,
      immediateResponse,
      resources,
      escalateToHuman: escalate,
      followUp
    };
  }

  private async detectCrisis(message: string): Promise {
    const indicators = [
      { pattern: /죽고\s*싶|자살|끝내고\s*싶/, type: 'suicidal_ideation', weight: 1.0 },
      { pattern: /자해|해치고/, type: 'self_harm', weight: 0.9 },
      { pattern: /희망.*없|의미.*없/, type: 'severe_distress', weight: 0.7 }
    ];

    let maxWeight = 0;
    let detectedType = null;
    const foundIndicators = [];

    for (const indicator of indicators) {
      if (indicator.pattern.test(message)) {
        foundIndicators.push(indicator.pattern.toString());
        if (indicator.weight > maxWeight) {
          maxWeight = indicator.weight;
          detectedType = indicator.type;
        }
      }
    }

    return {
      isCrisis: maxWeight > 0.5,
      type: detectedType,
      confidence: maxWeight,
      indicators: foundIndicators
    };
  }
}
