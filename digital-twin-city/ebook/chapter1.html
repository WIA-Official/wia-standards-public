1.1 Introduction to Digital Twin City Modeling

Digital Twin City Modeling is the foundational technology for creating virtual replicas of physical urban environments. This chapter covers comprehensive 3D modeling techniques, Building Information Modeling (BIM) and City Information Modeling (CIM) integration, geometric representation standards, spatial data structures, and real-time synchronization protocols essential for building accurate and functional digital twin cities.

The WIA Digital Twin City Standard defines a unified framework for representing urban assets in digital space, enabling seamless integration between physical infrastructure and virtual models. This approach supports real-time monitoring, simulation, analysis, and decision-making across all urban systems.

Key Components of Digital Twin Modeling

  • Geometric Modeling: 3D representation of buildings, infrastructure, and terrain
  • Semantic Information: Metadata and attributes describing objects and relationships
  • Spatial Relationships: Topological connections and geographic positioning
  • Temporal Data: Historical changes and real-time updates
  • Multi-scale Representation: Level of Detail (LOD) for different use cases

1.3 Code Examples

Example 1: CityGML Building Model Creation

import citygml
from citygml import Building, BuildingSurface, Solid
from geometries import Polygon, LinearRing

class CityModelBuilder:
    def __init__(self, city_name, srs="EPSG:4326"):
        self.city_model = citygml.CityModel(city_name, srs)

    def create_building(self, building_id, coordinates, height, building_type):
        """Create a LOD2 building with roof and walls"""
        building = Building(gml_id=building_id)
        building.building_type = building_type
        building.measured_height = height

        # Create building geometry
        footprint = self._create_footprint(coordinates)
        walls = self._create_walls(coordinates, height)
        roof = self._create_roof(coordinates, height)

        # Combine into solid geometry
        solid = Solid()
        solid.add_surface(footprint, "GroundSurface")
        for wall in walls:
            solid.add_surface(wall, "WallSurface")
        solid.add_surface(roof, "RoofSurface")

        building.lod2_solid = solid
        self.city_model.add_city_object(building)

        return building

    def _create_footprint(self, coordinates):
        """Create ground surface polygon"""
        ring = LinearRing(coordinates)
        return Polygon(exterior_ring=ring)

    def _create_walls(self, coordinates, height):
        """Create wall surfaces from footprint"""
        walls = []
        for i in range(len(coordinates) - 1):
            wall_coords = [
                coordinates[i],
                coordinates[i+1],
                (coordinates[i+1][0], coordinates[i+1][1], height),
                (coordinates[i][0], coordinates[i][1], height),
                coordinates[i]
            ]
            ring = LinearRing(wall_coords)
            walls.append(Polygon(exterior_ring=ring))
        return walls

    def export_to_gml(self, output_path):
        """Export city model to CityGML file"""
        self.city_model.write_xml(output_path)

# Usage example
builder = CityModelBuilder("Downtown District", "EPSG:4326")
building_coords = [
    (127.0276, 37.4979, 0),
    (127.0280, 37.4979, 0),
    (127.0280, 37.4982, 0),
    (127.0276, 37.4982, 0),
    (127.0276, 37.4979, 0)
]
building = builder.create_building(
    "BLDG_001",
    building_coords,
    45.5,
    "Commercial"
)
builder.export_to_gml("city_model.gml")

Example 2: BIM/CIM Integration with IFC

import ifcopenshell
from ifcopenshell.api import run
import numpy as np

class BIMCIMIntegrator:
    def __init__(self, ifc_schema="IFC4X3"):
        self.ifc_file = ifcopenshell.file(schema=ifc_schema)
        self._setup_project()

    def _setup_project(self):
        """Initialize IFC project structure"""
        # Create project
        self.project = run("root.create_entity", self.ifc_file,
                          ifc_class="IfcProject", name="Digital Twin City")

        # Create site
        self.site = run("root.create_entity", self.ifc_file,
                       ifc_class="IfcSite", name="City Site")
        run("aggregate.assign_object", self.ifc_file,
            relating_object=self.project, product=self.site)

        # Create spatial structure
        self.context = run("context.add_context", self.ifc_file,
                          context_type="Model")
        self.geometric_context = run("context.add_context", self.ifc_file,
                                     context_type="Model",
                                     context_identifier="Body",
                                     parent=self.context)

    def add_building_from_citygml(self, citygml_building):
        """Convert CityGML building to IFC"""
        # Create IFC building
        building = run("root.create_entity", self.ifc_file,
                      ifc_class="IfcBuilding",
                      name=citygml_building.gml_id)

        run("aggregate.assign_object", self.ifc_file,
            relating_object=self.site, product=building)

        # Create building storey
        elevation = citygml_building.get_elevation()
        storey = run("root.create_entity", self.ifc_file,
                    ifc_class="IfcBuildingStorey",
                    name=f"{citygml_building.gml_id}_Floor1")
        storey.Elevation = elevation

        run("aggregate.assign_object", self.ifc_file,
            relating_object=building, product=storey)

        # Convert geometry
        self._convert_geometry(citygml_building, storey)

        return building

    def _convert_geometry(self, citygml_obj, ifc_parent):
        """Convert CityGML geometry to IFC representation"""
        if hasattr(citygml_obj, 'lod2_solid'):
            solid = citygml_obj.lod2_solid

            # Create IFC shape representation
            vertices, faces = self._extract_mesh(solid)

            ifc_vertices = [
                self.ifc_file.createIfcCartesianPoint(v.tolist())
                for v in vertices
            ]

            # Create mesh representation
            representation = run("geometry.add_mesh_representation",
                               self.ifc_file,
                               context=self.geometric_context,
                               vertices=vertices.tolist(),
                               faces=faces.tolist())

            return representation

    def _extract_mesh(self, solid):
        """Extract vertices and faces from CityGML solid"""
        vertices = []
        faces = []
        vertex_index = 0

        for surface in solid.surfaces:
            surface_vertices = surface.exterior_ring.points
            vertex_count = len(surface_vertices)

            # Add vertices
            for v in surface_vertices:
                vertices.append(np.array(v))

            # Add face indices
            face = list(range(vertex_index, vertex_index + vertex_count))
            faces.append(face)
            vertex_index += vertex_count

        return np.array(vertices), faces

    def export_ifc(self, output_path):
        """Export integrated model to IFC"""
        self.ifc_file.write(output_path)

# Usage
integrator = BIMCIMIntegrator()
# Assuming citygml_building from previous example
integrator.add_building_from_citygml(building)
integrator.export_ifc("integrated_city_model.ifc")

Example 3: Real-time 3D Tiles Streaming

import json
import numpy as np
from pyproj import Transformer
import gzip

class TilesGenerator:
    def __init__(self, root_transform):
        self.root_transform = root_transform
        self.transformer = Transformer.from_crs("EPSG:4326", "EPSG:4978")

    def create_tileset(self, buildings, output_dir):
        """Generate 3D Tiles tileset from building collection"""
        # Create root tileset.json
        tileset = {
            "asset": {
                "version": "1.1",
                "tilesetVersion": "1.0.0"
            },
            "geometricError": 500.0,
            "root": self._create_root_tile(buildings)
        }

        with open(f"{output_dir}/tileset.json", 'w') as f:
            json.dump(tileset, f, indent=2)

        # Generate tile hierarchy
        self._generate_tile_hierarchy(buildings, output_dir)

        return tileset

    def _create_root_tile(self, buildings):
        """Create root tile definition"""
        # Calculate bounding volume
        bounds = self._calculate_bounds(buildings)

        root_tile = {
            "boundingVolume": {
                "region": bounds
            },
            "geometricError": 250.0,
            "refine": "REPLACE",
            "content": {
                "uri": "content/root.b3dm"
            },
            "children": []
        }

        # Create quadtree subdivision
        quadrants = self._subdivide_buildings(buildings)

        for quadrant_id, quadrant_buildings in quadrants.items():
            if len(quadrant_buildings) > 0:
                child_tile = self._create_child_tile(
                    quadrant_buildings,
                    quadrant_id
                )
                root_tile["children"].append(child_tile)

        return root_tile

    def _create_child_tile(self, buildings, tile_id):
        """Create child tile for building subset"""
        bounds = self._calculate_bounds(buildings)

        return {
            "boundingVolume": {
                "region": bounds
            },
            "geometricError": 50.0,
            "content": {
                "uri": f"content/{tile_id}.b3dm"
            }
        }

    def _calculate_bounds(self, buildings):
        """Calculate geographic bounding region"""
        all_coords = []
        for building in buildings:
            coords = building.get_coordinates()
            all_coords.extend(coords)

        coords_array = np.array(all_coords)
        min_lon, min_lat = coords_array.min(axis=0)[:2]
        max_lon, max_lat = coords_array.max(axis=0)[:2]

        # Calculate height bounds
        heights = [b.measured_height for b in buildings]
        min_height = 0
        max_height = max(heights) if heights else 100

        return [
            np.radians(min_lon),
            np.radians(min_lat),
            np.radians(max_lon),
            np.radians(max_lat),
            min_height,
            max_height
        ]

    def _subdivide_buildings(self, buildings):
        """Subdivide buildings into quadrants"""
        if len(buildings) == 0:
            return {}

        # Calculate center point
        all_coords = []
        for b in buildings:
            all_coords.extend(b.get_coordinates())

        coords_array = np.array(all_coords)
        center_lon = coords_array[:, 0].mean()
        center_lat = coords_array[:, 1].mean()

        quadrants = {"NE": [], "NW": [], "SE": [], "SW": []}

        for building in buildings:
            center = building.get_center()
            lon, lat = center[0], center[1]

            if lon >= center_lon and lat >= center_lat:
                quadrants["NE"].append(building)
            elif lon < center_lon and lat >= center_lat:
                quadrants["NW"].append(building)
            elif lon >= center_lon and lat < center_lat:
                quadrants["SE"].append(building)
            else:
                quadrants["SW"].append(building)

        return quadrants

    def create_b3dm_tile(self, buildings, output_path):
        """Create Batched 3D Model tile"""
        # Generate glTF from buildings
        gltf_data = self._buildings_to_gltf(buildings)

        # Create feature table
        feature_table = {
            "BATCH_LENGTH": len(buildings)
        }

        feature_table_json = json.dumps(feature_table).encode('utf-8')
        feature_table_binary = b''

        # Create batch table with metadata
        batch_table = {
            "building_id": [b.gml_id for b in buildings],
            "height": [b.measured_height for b in buildings],
            "type": [b.building_type for b in buildings]
        }

        batch_table_json = json.dumps(batch_table).encode('utf-8')

        # Combine into b3dm format
        b3dm_data = self._create_b3dm_file(
            feature_table_json,
            feature_table_binary,
            batch_table_json,
            gltf_data
        )

        # Write compressed tile
        with gzip.open(output_path, 'wb') as f:
            f.write(b3dm_data)

    def _buildings_to_gltf(self, buildings):
        """Convert buildings to glTF format"""
        # Simplified glTF generation
        gltf = {
            "asset": {"version": "2.0"},
            "scene": 0,
            "scenes": [{"nodes": []}],
            "nodes": [],
            "meshes": [],
            "accessors": [],
            "bufferViews": [],
            "buffers": []
        }

        # Add each building as a mesh
        for idx, building in enumerate(buildings):
            mesh_data = self._building_to_mesh(building)
            gltf["meshes"].append(mesh_data)
            gltf["nodes"].append({
                "mesh": idx,
                "name": building.gml_id
            })
            gltf["scenes"][0]["nodes"].append(idx)

        return json.dumps(gltf).encode('utf-8')

    def _building_to_mesh(self, building):
        """Convert building to glTF mesh"""
        # Simplified mesh generation
        return {
            "primitives": [{
                "attributes": {
                    "POSITION": 0
                },
                "mode": 4  # TRIANGLES
            }]
        }

    def _create_b3dm_file(self, feature_table_json, feature_table_binary,
                          batch_table_json, gltf_data):
        """Assemble b3dm binary file"""
        import struct

        # b3dm header
        magic = b'b3dm'
        version = 1

        # Calculate lengths
        header_length = 28
        feature_table_json_length = len(feature_table_json)
        feature_table_binary_length = len(feature_table_binary)
        batch_table_json_length = len(batch_table_json)
        gltf_length = len(gltf_data)

        byte_length = (header_length +
                      feature_table_json_length +
                      feature_table_binary_length +
                      batch_table_json_length +
                      gltf_length)

        # Pack header
        header = struct.pack('<4sIIIIII',
                            magic,
                            version,
                            byte_length,
                            feature_table_json_length,
                            feature_table_binary_length,
                            batch_table_json_length,
                            0)  # batch_table_binary_length

        # Combine all parts
        return (header +
                feature_table_json +
                feature_table_binary +
                batch_table_json +
                gltf_data)

# Usage
generator = TilesGenerator(root_transform=np.eye(4))
# Assuming buildings is a list of Building objects
tileset = generator.create_tileset(buildings, "./output/tileset")
