3.1 Traffic Simulation Framework

Advanced traffic simulation is essential for digital twin cities to model, predict, and optimize urban mobility. This chapter covers microscopic and macroscopic traffic simulation models, real-time traffic prediction using AI/ML, autonomous vehicle integration, signal optimization, and smart mobility solutions.

3.2 Code Examples

Example 1: Microscopic Traffic Simulation Engine

import numpy as np
from dataclasses import dataclass
from typing import List, Dict, Tuple
from collections import deque

@dataclass
class Vehicle:
    id: str
    position: float  # meters from start
    speed: float  # m/s
    acceleration: float  # m/s^2
    length: float  # meters
    max_speed: float  # m/s
    max_accel: float  # m/s^2
    max_decel: float  # m/s^2
    lane_id: int
    vehicle_type: str  # 'car', 'truck', 'bus', 'motorcycle'

class MicroscopicTrafficSimulator:
    def __init__(self, road_length=1000, num_lanes=3, time_step=0.1):
        self.road_length = road_length
        self.num_lanes = num_lanes
        self.time_step = time_step
        self.vehicles: Dict[str, Vehicle] = {}
        self.current_time = 0.0

        # Intelligent Driver Model (IDM) parameters
        self.idm_params = {
            'desired_time_gap': 1.5,  # seconds
            'min_spacing': 2.0,  # meters
            'accel_exponent': 4
        }

    def add_vehicle(self, vehicle: Vehicle):
        """Add vehicle to simulation"""
        self.vehicles[vehicle.id] = vehicle

    def step(self):
        """Execute one simulation time step"""
        # Update all vehicles
        for vid, vehicle in self.vehicles.items():
            # Find leading vehicle
            leader = self._find_leader(vehicle)

            # Calculate acceleration using IDM
            accel = self._calculate_idm_acceleration(vehicle, leader)

            # Update speed and position
            new_speed = max(0, vehicle.speed + accel * self.time_step)
            new_speed = min(new_speed, vehicle.max_speed)

            new_position = vehicle.position + new_speed * self.time_step

            # Update vehicle state
            vehicle.speed = new_speed
            vehicle.position = new_position
            vehicle.acceleration = accel

            # Lane changing logic
            if self._should_change_lane(vehicle, leader):
                new_lane = self._select_target_lane(vehicle)
                if self._is_lane_change_safe(vehicle, new_lane):
                    vehicle.lane_id = new_lane

        # Remove vehicles that exited
        self.vehicles = {
            vid: v for vid, v in self.vehicles.items()
            if v.position < self.road_length
        }

        self.current_time += self.time_step

    def _find_leader(self, vehicle: Vehicle) -> Vehicle:
        """Find the leading vehicle in the same lane"""
        same_lane = [
            v for v in self.vehicles.values()
            if v.lane_id == vehicle.lane_id and v.id != vehicle.id
        ]

        # Find closest vehicle ahead
        ahead = [v for v in same_lane if v.position > vehicle.position]

        if ahead:
            return min(ahead, key=lambda v: v.position)

        return None

    def _calculate_idm_acceleration(self, vehicle: Vehicle,
                                   leader: Vehicle = None) -> float:
        """Intelligent Driver Model acceleration calculation"""
        # Free-flow acceleration
        free_accel = vehicle.max_accel * (
            1 - (vehicle.speed / vehicle.max_speed) **
            self.idm_params['accel_exponent']
        )

        if leader is None:
            return free_accel

        # Calculate spacing
        spacing = leader.position - vehicle.position - leader.length

        # Desired spacing
        speed_diff = vehicle.speed - leader.speed
        desired_spacing = (
            self.idm_params['min_spacing'] +
            vehicle.speed * self.idm_params['desired_time_gap'] +
            (vehicle.speed * speed_diff) /
            (2 * np.sqrt(vehicle.max_accel * vehicle.max_decel))
        )

        # Interaction acceleration
        interaction_accel = -vehicle.max_accel * (
            desired_spacing / max(spacing, 0.1)
        ) ** 2

        return free_accel + interaction_accel

    def _should_change_lane(self, vehicle: Vehicle,
                           leader: Vehicle) -> bool:
        """Determine if vehicle should attempt lane change"""
        if leader is None:
            return False

        # Check if following too closely
        spacing = leader.position - vehicle.position - leader.length
        desired_spacing = (
            self.idm_params['min_spacing'] +
            vehicle.speed * self.idm_params['desired_time_gap']
        )

        return spacing < desired_spacing

    def _select_target_lane(self, vehicle: Vehicle) -> int:
        """Select best lane for lane change"""
        current_lane = vehicle.lane_id

        # Consider adjacent lanes
        candidate_lanes = []
        if current_lane > 0:
            candidate_lanes.append(current_lane - 1)
        if current_lane < self.num_lanes - 1:
            candidate_lanes.append(current_lane + 1)

        # Evaluate lanes
        best_lane = current_lane
        best_score = self._evaluate_lane(vehicle, current_lane)

        for lane in candidate_lanes:
            score = self._evaluate_lane(vehicle, lane)
            if score > best_score:
                best_score = score
                best_lane = lane

        return best_lane

    def _evaluate_lane(self, vehicle: Vehicle, lane: int) -> float:
        """Evaluate desirability of a lane"""
        # Find leader in target lane
        same_lane = [
            v for v in self.vehicles.values()
            if v.lane_id == lane and v.position > vehicle.position
        ]

        if same_lane:
            leader = min(same_lane, key=lambda v: v.position)
            gap = leader.position - vehicle.position - leader.length
            return gap  # Higher gap = better lane

        return float('inf')  # Empty lane is best

    def _is_lane_change_safe(self, vehicle: Vehicle,
                            target_lane: int) -> bool:
        """Check if lane change is safe"""
        # Find vehicles in target lane
        target_vehicles = [
            v for v in self.vehicles.values()
            if v.lane_id == target_lane
        ]

        for other in target_vehicles:
            # Check front gap
            if other.position > vehicle.position:
                gap = other.position - vehicle.position - vehicle.length
                if gap < self.idm_params['min_spacing'] * 2:
                    return False

            # Check rear gap
            else:
                gap = vehicle.position - other.position - other.length
                if gap < self.idm_params['min_spacing'] * 2:
                    return False

        return True

    def get_traffic_metrics(self) -> Dict:
        """Calculate traffic performance metrics"""
        if not self.vehicles:
            return {
                'flow': 0,
                'density': 0,
                'avg_speed': 0,
                'total_vehicles': 0
            }

        speeds = [v.speed for v in self.vehicles.values()]
        positions = [v.position for v in self.vehicles.values()]

        return {
            'flow': len(self.vehicles) / (self.current_time + 0.001),
            'density': len(self.vehicles) / (self.road_length / 1000),
            'avg_speed': np.mean(speeds),
            'total_vehicles': len(self.vehicles),
            'congestion_level': self._calculate_congestion()
        }

    def _calculate_congestion(self) -> str:
        """Calculate congestion level"""
        if not self.vehicles:
            return 'free'

        avg_speed = np.mean([v.speed for v in self.vehicles.values()])
        avg_max_speed = np.mean([v.max_speed for v in self.vehicles.values()])

        speed_ratio = avg_speed / avg_max_speed

        if speed_ratio > 0.8:
            return 'free'
        elif speed_ratio > 0.6:
            return 'light'
        elif speed_ratio > 0.4:
            return 'moderate'
        elif speed_ratio > 0.2:
            return 'heavy'
        else:
            return 'severe'

# Usage example
simulator = MicroscopicTrafficSimulator(road_length=5000, num_lanes=3)

# Add vehicles
for i in range(20):
    vehicle = Vehicle(
        id=f"VEH_{i:03d}",
        position=i * 50.0,
        speed=20.0 + np.random.randn() * 2,
        acceleration=0.0,
        length=4.5,
        max_speed=30.0,
        max_accel=2.0,
        max_decel=4.0,
        lane_id=np.random.randint(0, 3),
        vehicle_type='car'
    )
    simulator.add_vehicle(vehicle)

# Run simulation
for step in range(1000):
    simulator.step()
    if step % 100 == 0:
        metrics = simulator.get_traffic_metrics()
        print(f"Time: {simulator.current_time:.1f}s - {metrics}")

Example 2: AI-based Traffic Prediction

import torch
import torch.nn as nn
import numpy as np
from typing import List, Tuple

class TrafficPredictionLSTM(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size):
        super(TrafficPredictionLSTM, self).__init__()
        self.hidden_size = hidden_size
        self.num_layers = num_layers

        self.lstm = nn.LSTM(
            input_size,
            hidden_size,
            num_layers,
            batch_first=True,
            dropout=0.2
        )

        self.fc = nn.Sequential(
            nn.Linear(hidden_size, hidden_size // 2),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_size // 2, output_size)
        )

    def forward(self, x):
        # Initialize hidden state
        h0 = torch.zeros(self.num_layers, x.size(0),
                        self.hidden_size).to(x.device)
        c0 = torch.zeros(self.num_layers, x.size(0),
                        self.hidden_size).to(x.device)

        # LSTM forward pass
        out, _ = self.lstm(x, (h0, c0))

        # Take output from last time step
        out = self.fc(out[:, -1, :])

        return out

class TrafficPredictor:
    def __init__(self, sequence_length=12, prediction_horizon=6):
        self.sequence_length = sequence_length
        self.prediction_horizon = prediction_horizon

        # Model parameters
        self.input_size = 10  # Features per time step
        self.hidden_size = 128
        self.num_layers = 3
        self.output_size = prediction_horizon

        # Initialize model
        self.model = TrafficPredictionLSTM(
            self.input_size,
            self.hidden_size,
            self.num_layers,
            self.output_size
        )

        self.device = torch.device('cuda' if torch.cuda.is_available()
                                  else 'cpu')
        self.model.to(self.device)

    def prepare_features(self, traffic_data: List[Dict]) -> np.ndarray:
        """Prepare features from traffic data"""
        features = []

        for data_point in traffic_data:
            feature_vector = [
                data_point['speed'],
                data_point['volume'],
                data_point['occupancy'],
                data_point['hour_of_day'] / 24.0,
                data_point['day_of_week'] / 7.0,
                data_point['is_weekend'],
                data_point['weather_temp'] / 40.0,
                data_point['weather_rain'],
                data_point['incident_count'],
                data_point['special_event']
            ]
            features.append(feature_vector)

        return np.array(features)

    def predict(self, historical_data: List[Dict]) -> np.ndarray:
        """Predict future traffic conditions"""
        # Prepare input features
        features = self.prepare_features(historical_data)

        # Ensure we have enough data
        if len(features) < self.sequence_length:
            raise ValueError(f"Need at least {self.sequence_length} data points")

        # Take last sequence_length points
        input_sequence = features[-self.sequence_length:]

        # Convert to tensor
        x = torch.FloatTensor(input_sequence).unsqueeze(0).to(self.device)

        # Predict
        self.model.eval()
        with torch.no_grad():
            predictions = self.model(x)

        return predictions.cpu().numpy()[0]

    def train_model(self, train_data, train_labels, epochs=100):
        """Train the prediction model"""
        optimizer = torch.optim.Adam(self.model.parameters(), lr=0.001)
        criterion = nn.MSELoss()

        self.model.train()

        for epoch in range(epochs):
            # Forward pass
            outputs = self.model(train_data)
            loss = criterion(outputs, train_labels)

            # Backward pass
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            if (epoch + 1) % 10 == 0:
                print(f'Epoch [{epoch+1}/{epochs}], Loss: {loss.item():.4f}')

# Usage example
predictor = TrafficPredictor(sequence_length=12, prediction_horizon=6)

# Historical traffic data (last 12 time periods)
historical_data = [
    {
        'speed': 45.5, 'volume': 850, 'occupancy': 0.35,
        'hour_of_day': 8, 'day_of_week': 1, 'is_weekend': 0,
        'weather_temp': 22, 'weather_rain': 0,
        'incident_count': 0, 'special_event': 0
    }
    for _ in range(12)
]

# Predict next 6 time periods
predictions = predictor.predict(historical_data)
print(f"Predicted traffic speeds: {predictions}")

Example 3: Traffic Signal Optimization

import numpy as np
from scipy.optimize import minimize
from typing import Dict, List

class AdaptiveSignalController:
    def __init__(self, intersection_id, num_phases=4):
        self.intersection_id = intersection_id
        self.num_phases = num_phases
        self.min_green_time = 10  # seconds
        self.max_green_time = 90  # seconds
        self.yellow_time = 3  # seconds
        self.all_red_time = 2  # seconds
        self.cycle_length = 120  # seconds

    def optimize_signal_timing(self, traffic_demands: Dict) -> Dict:
        """Optimize signal timing based on traffic demand"""
        # Extract demands for each phase
        demands = [
            traffic_demands.get(f'phase_{i}', 0)
            for i in range(self.num_phases)
        ]

        # Initial guess: equal split
        initial_green_times = np.ones(self.num_phases) * 20

        # Constraints
        constraints = [
            # Sum of green times + lost times = cycle length
            {
                'type': 'eq',
                'fun': lambda x: (
                    np.sum(x) +
                    self.num_phases * (self.yellow_time + self.all_red_time) -
                    self.cycle_length
                )
            }
        ]

        # Bounds for each phase
        bounds = [
            (self.min_green_time, self.max_green_time)
            for _ in range(self.num_phases)
        ]

        # Optimize
        result = minimize(
            fun=lambda x: self._objective_function(x, demands),
            x0=initial_green_times,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )

        optimized_times = result.x

        # Generate signal plan
        signal_plan = self._generate_signal_plan(optimized_times)

        return signal_plan

    def _objective_function(self, green_times: np.ndarray,
                           demands: List[float]) -> float:
        """Calculate total delay (objective to minimize)"""
        total_delay = 0

        for i, (green_time, demand) in enumerate(zip(green_times, demands)):
            # Webster's delay formula
            red_time = self.cycle_length - green_time

            # Arrival rate (vehicles per second)
            arrival_rate = demand / 3600.0

            # Saturation flow rate (vehicles per second)
            saturation_flow = 1800 / 3600.0

            # Capacity
            capacity = saturation_flow * green_time / self.cycle_length

            # Degree of saturation
            x = arrival_rate / capacity if capacity > 0 else 1

            if x < 1.0:
                # Webster's delay formula
                delay = (
                    0.5 * self.cycle_length * (1 - green_time/self.cycle_length)**2 /
                    (1 - min(x, 0.99) * green_time/self.cycle_length)
                )
            else:
                # Oversaturated: very high delay
                delay = 999999

            total_delay += delay * demand

        return total_delay

    def _generate_signal_plan(self, green_times: np.ndarray) -> Dict:
        """Generate detailed signal timing plan"""
        signal_plan = {
            'intersection_id': self.intersection_id,
            'cycle_length': self.cycle_length,
            'phases': []
        }

        current_time = 0

        for i, green_time in enumerate(green_times):
            phase = {
                'phase_id': i,
                'green_start': current_time,
                'green_duration': int(green_time),
                'yellow_start': current_time + int(green_time),
                'yellow_duration': self.yellow_time,
                'all_red_start': (current_time + int(green_time) +
                                 self.yellow_time),
                'all_red_duration': self.all_red_time
            }

            signal_plan['phases'].append(phase)

            current_time += (int(green_time) + self.yellow_time +
                           self.all_red_time)

        return signal_plan

# Usage
controller = AdaptiveSignalController("INT_001", num_phases=4)

traffic_demands = {
    'phase_0': 1200,  # vehicles/hour
    'phase_1': 800,
    'phase_2': 1500,
    'phase_3': 600
}

optimal_plan = controller.optimize_signal_timing(traffic_demands)
print(f"Optimized Signal Plan: {optimal_plan}")
