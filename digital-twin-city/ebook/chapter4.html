4.1 AI-Powered Urban Planning Framework

Modern urban planning requires sophisticated tools that leverage AI, machine learning, and big data analytics to create sustainable, livable cities. This chapter covers AI-powered planning algorithms, zoning optimization, environmental impact assessment, demographic analysis, and sustainable development frameworks for digital twin cities.

4.2 Code Examples

Example 1: AI-Based Land Use Optimization

import numpy as np
from typing import List, Dict, Tuple
from dataclasses import dataclass

@dataclass
class LandParcel:
    id: str
    area: float  # m²
    location: Tuple[float, float]
    current_use: str
    allowed_uses: List[str]
    value: float

class GeneticAlgorithmPlanner:
    def __init__(self, parcels: List[LandParcel], population_size=100):
        self.parcels = parcels
        self.population_size = population_size
        self.num_parcels = len(parcels)

        # Land use types
        self.use_types = ['residential', 'commercial', 'industrial',
                         'park', 'institutional', 'mixed']

        # Objectives weights
        self.weights = {
            'economic_value': 0.3,
            'environmental': 0.25,
            'social_equity': 0.25,
            'accessibility': 0.2
        }

    def optimize(self, generations=200):
        """Run genetic algorithm optimization"""
        # Initialize population
        population = self._initialize_population()

        best_solution = None
        best_fitness = float('-inf')

        for gen in range(generations):
            # Evaluate fitness
            fitness_scores = [
                self._evaluate_fitness(individual)
                for individual in population
            ]

            # Track best solution
            max_fitness_idx = np.argmax(fitness_scores)
            if fitness_scores[max_fitness_idx] > best_fitness:
                best_fitness = fitness_scores[max_fitness_idx]
                best_solution = population[max_fitness_idx].copy()

            # Selection
            selected = self._tournament_selection(
                population, fitness_scores
            )

            # Crossover
            offspring = self._crossover(selected)

            # Mutation
            mutated = self._mutate(offspring)

            # New population
            population = mutated

            if gen % 20 == 0:
                print(f"Generation {gen}: Best fitness = {best_fitness:.4f}")

        return self._decode_solution(best_solution)

    def _initialize_population(self):
        """Create initial random population"""
        population = []

        for _ in range(self.population_size):
            individual = []
            for parcel in self.parcels:
                # Random use from allowed uses
                use = np.random.choice(parcel.allowed_uses)
                use_idx = self.use_types.index(use)
                individual.append(use_idx)
            population.append(np.array(individual))

        return population

    def _evaluate_fitness(self, individual):
        """Multi-objective fitness evaluation"""
        # Decode land uses
        land_uses = [self.use_types[idx] for idx in individual]

        # Calculate objectives
        economic = self._calculate_economic_value(land_uses)
        environmental = self._calculate_environmental_score(land_uses)
        social = self._calculate_social_equity(land_uses)
        accessibility = self._calculate_accessibility(land_uses)

        # Weighted sum
        fitness = (
            self.weights['economic_value'] * economic +
            self.weights['environmental'] * environmental +
            self.weights['social_equity'] * social +
            self.weights['accessibility'] * accessibility
        )

        return fitness

    def _calculate_economic_value(self, land_uses):
        """Calculate total economic value"""
        value_multipliers = {
            'residential': 1.0,
            'commercial': 2.5,
            'industrial': 1.8,
            'park': 0.3,
            'institutional': 0.8,
            'mixed': 2.0
        }

        total_value = sum(
            parcel.value * value_multipliers[use]
            for parcel, use in zip(self.parcels, land_uses)
        )

        # Normalize to 0-1
        max_possible = sum(
            parcel.value * 2.5 for parcel in self.parcels
        )

        return total_value / max_possible if max_possible > 0 else 0

    def _calculate_environmental_score(self, land_uses):
        """Calculate environmental sustainability score"""
        env_scores = {
            'residential': 0.6,
            'commercial': 0.4,
            'industrial': 0.2,
            'park': 1.0,
            'institutional': 0.7,
            'mixed': 0.5
        }

        # Green space requirement: at least 15% of total area
        total_area = sum(p.area for p in self.parcels)
        green_area = sum(
            p.area for p, use in zip(self.parcels, land_uses)
            if use == 'park'
        )

        green_ratio = green_area / total_area
        green_penalty = 1.0 if green_ratio >= 0.15 else green_ratio / 0.15

        # Average environmental score
        avg_score = np.mean([
            env_scores[use] for use in land_uses
        ])

        return avg_score * green_penalty

    def _calculate_social_equity(self, land_uses):
        """Calculate social equity score"""
        # Distribution of different use types
        use_counts = {use_type: 0 for use_type in self.use_types}
        for use in land_uses:
            use_counts[use] += 1

        # Diversity score (Shannon entropy)
        total = len(land_uses)
        proportions = [
            count / total for count in use_counts.values()
            if count > 0
        ]

        if not proportions:
            return 0

        entropy = -sum(p * np.log(p) for p in proportions)
        max_entropy = np.log(len(self.use_types))

        diversity_score = entropy / max_entropy

        # Check for institutional facilities (schools, hospitals)
        institutional_ratio = use_counts['institutional'] / total
        institutional_score = min(institutional_ratio / 0.05, 1.0)

        return 0.7 * diversity_score + 0.3 * institutional_score

    def _calculate_accessibility(self, land_uses):
        """Calculate accessibility score"""
        # Simplified: Check mixing of uses
        score = 0

        for i, (parcel, use) in enumerate(zip(self.parcels, land_uses)):
            # Find neighbors within 500m
            neighbors = []
            for j, other_parcel in enumerate(self.parcels):
                if i != j:
                    dist = self._distance(
                        parcel.location,
                        other_parcel.location
                    )
                    if dist <= 500:
                        neighbors.append(land_uses[j])

            # Score based on use diversity in neighborhood
            if neighbors:
                unique_uses = len(set(neighbors))
                score += unique_uses / len(self.use_types)

        return score / len(self.parcels) if self.parcels else 0

    def _distance(self, loc1, loc2):
        """Calculate Euclidean distance"""
        return np.sqrt(
            (loc1[0] - loc2[0])**2 + (loc1[1] - loc2[1])**2
        )

    def _tournament_selection(self, population, fitness_scores,
                              tournament_size=3):
        """Tournament selection"""
        selected = []

        for _ in range(self.population_size):
            # Random tournament
            contestants = np.random.choice(
                len(population),
                tournament_size,
                replace=False
            )
            winner = contestants[
                np.argmax([fitness_scores[i] for i in contestants])
            ]
            selected.append(population[winner].copy())

        return selected

    def _crossover(self, population, crossover_rate=0.8):
        """Single-point crossover"""
        offspring = []

        for i in range(0, len(population), 2):
            parent1 = population[i]
            parent2 = population[i+1] if i+1 < len(population) else population[0]

            if np.random.random() < crossover_rate:
                point = np.random.randint(1, self.num_parcels)
                child1 = np.concatenate([parent1[:point], parent2[point:]])
                child2 = np.concatenate([parent2[:point], parent1[point:]])
                offspring.extend([child1, child2])
            else:
                offspring.extend([parent1.copy(), parent2.copy()])

        return offspring[:self.population_size]

    def _mutate(self, population, mutation_rate=0.1):
        """Random mutation"""
        for individual in population:
            for i in range(len(individual)):
                if np.random.random() < mutation_rate:
                    # Mutate to random allowed use
                    allowed = self.parcels[i].allowed_uses
                    new_use = np.random.choice(allowed)
                    individual[i] = self.use_types.index(new_use)

        return population

    def _decode_solution(self, individual):
        """Decode solution to readable format"""
        solution = {
            'land_use_plan': [],
            'statistics': {}
        }

        for parcel, use_idx in zip(self.parcels, individual):
            use = self.use_types[use_idx]
            solution['land_use_plan'].append({
                'parcel_id': parcel.id,
                'area': parcel.area,
                'location': parcel.location,
                'assigned_use': use,
                'current_use': parcel.current_use
            })

        # Calculate statistics
        land_uses = [self.use_types[idx] for idx in individual]
        use_counts = {use_type: 0 for use_type in self.use_types}
        use_areas = {use_type: 0 for use_type in self.use_types}

        for parcel, use in zip(self.parcels, land_uses):
            use_counts[use] += 1
            use_areas[use] += parcel.area

        solution['statistics'] = {
            'use_counts': use_counts,
            'use_areas': use_areas,
            'total_area': sum(p.area for p in self.parcels)
        }

        return solution

# Usage
parcels = [
    LandParcel(
        f"P{i:03d}",
        area=np.random.uniform(1000, 10000),
        location=(np.random.uniform(0, 5000), np.random.uniform(0, 5000)),
        current_use='vacant',
        allowed_uses=['residential', 'commercial', 'park', 'mixed'],
        value=np.random.uniform(100, 1000)
    )
    for i in range(50)
]

planner = GeneticAlgorithmPlanner(parcels)
optimal_plan = planner.optimize(generations=100)

Example 2: Environmental Impact Assessment

import numpy as np
from typing import Dict, List

class EnvironmentalImpactAnalyzer:
    def __init__(self, project_area):
        self.project_area = project_area
        self.baseline_data = {}

    def assess_impact(self, development_plan: Dict) -> Dict:
        """Comprehensive environmental impact assessment"""
        assessment = {
            'air_quality': self._assess_air_quality(development_plan),
            'carbon_emissions': self._assess_carbon(development_plan),
            'energy_consumption': self._assess_energy(development_plan),
            'water_impact': self._assess_water(development_plan),
            'biodiversity': self._assess_biodiversity(development_plan),
            'noise_pollution': self._assess_noise(development_plan),
            'heat_island': self._assess_heat_island(development_plan),
            'overall_score': 0
        }

        # Calculate overall environmental score
        weights = {
            'air_quality': 0.20,
            'carbon_emissions': 0.20,
            'energy_consumption': 0.15,
            'water_impact': 0.15,
            'biodiversity': 0.15,
            'noise_pollution': 0.10,
            'heat_island': 0.05
        }

        overall = sum(
            assessment[key]['score'] * weights[key]
            for key in weights.keys()
        )

        assessment['overall_score'] = overall
        assessment['rating'] = self._get_rating(overall)

        return assessment

    def _assess_air_quality(self, plan):
        """Air quality impact assessment"""
        # Building density effect
        building_coverage = plan.get('building_coverage', 0.3)
        green_space_ratio = plan.get('green_space_ratio', 0.2)

        # Traffic generation
        traffic_volume = plan.get('daily_traffic', 10000)

        # Emissions from buildings
        building_area = plan.get('total_building_area', 100000)

        # Calculate pollutant concentrations
        pm25_concentration = (
            20 * building_coverage +
            0.002 * traffic_volume +
            0.0001 * building_area -
            10 * green_space_ratio
        )

        nox_concentration = (
            15 * building_coverage +
            0.003 * traffic_volume
        )

        # Score (0-100, higher is better)
        pm25_score = max(0, 100 - pm25_concentration)
        nox_score = max(0, 100 - nox_concentration)

        air_score = (pm25_score + nox_score) / 2

        return {
            'score': air_score / 100,
            'pm25': pm25_concentration,
            'nox': nox_concentration,
            'recommendation': self._air_quality_recommendation(air_score)
        }

    def _assess_carbon(self, plan):
        """Carbon emissions assessment"""
        # Building emissions (kg CO2/year)
        building_area = plan.get('total_building_area', 100000)
        building_efficiency = plan.get('energy_efficiency_rating', 0.6)

        building_emissions = (
            building_area * 50 * (1 - building_efficiency * 0.5)
        )

        # Transport emissions
        traffic_volume = plan.get('daily_traffic', 10000)
        public_transport_ratio = plan.get('public_transport_ratio', 0.3)

        transport_emissions = (
            traffic_volume * 365 * 0.12 *  # kg CO2 per vehicle-day
            (1 - public_transport_ratio * 0.7)
        )

        total_emissions = building_emissions + transport_emissions

        # Per capita (assume population)
        population = plan.get('estimated_population', 5000)
        per_capita = total_emissions / population if population > 0 else 0

        # Score based on targets (< 2000 kg/capita/year)
        score = max(0, min(100, 100 - (per_capita - 2000) / 50))

        return {
            'score': score / 100,
            'total_emissions': total_emissions,
            'per_capita': per_capita,
            'building_emissions': building_emissions,
            'transport_emissions': transport_emissions,
            'target': 2000
        }

    def _assess_energy(self, plan):
        """Energy consumption assessment"""
        building_area = plan.get('total_building_area', 100000)
        efficiency_rating = plan.get('energy_efficiency_rating', 0.6)
        renewable_ratio = plan.get('renewable_energy_ratio', 0.1)

        # Annual energy consumption (kWh)
        base_consumption = building_area * 150  # kWh/m²/year

        # Apply efficiency
        actual_consumption = base_consumption * (1 - efficiency_rating * 0.5)

        # Renewable offset
        renewable_generation = actual_consumption * renewable_ratio
        net_consumption = actual_consumption - renewable_generation

        # Score
        consumption_per_m2 = net_consumption / building_area
        score = max(0, min(100, 100 - (consumption_per_m2 - 50)))

        return {
            'score': score / 100,
            'total_consumption': actual_consumption,
            'renewable_generation': renewable_generation,
            'net_consumption': net_consumption,
            'consumption_per_m2': consumption_per_m2
        }

    def _assess_water(self, plan):
        """Water impact assessment"""
        building_area = plan.get('total_building_area', 100000)
        green_space_area = plan.get('green_space_area', 20000)
        permeable_surface = plan.get('permeable_surface_ratio', 0.3)

        # Water consumption
        population = plan.get('estimated_population', 5000)
        daily_consumption = population * 150  # liters per person per day

        # Runoff calculation
        total_area = plan.get('total_area', 200000)
        impermeable_area = total_area * (1 - permeable_surface)
        runoff_volume = impermeable_area * 0.001 * 800  # m³/year

        # Score
        consumption_score = max(0, 100 - daily_consumption / 1000)
        runoff_score = max(0, 100 - runoff_volume / 100)

        water_score = (consumption_score + runoff_score) / 2

        return {
            'score': water_score / 100,
            'daily_consumption': daily_consumption,
            'annual_runoff': runoff_volume,
            'permeable_ratio': permeable_surface
        }

    def _assess_biodiversity(self, plan):
        """Biodiversity impact assessment"""
        green_space_ratio = plan.get('green_space_ratio', 0.2)
        native_vegetation_ratio = plan.get('native_vegetation_ratio', 0.5)
        habitat_connectivity = plan.get('habitat_connectivity_score', 0.3)

        score = (
            green_space_ratio * 40 +
            native_vegetation_ratio * 30 +
            habitat_connectivity * 30
        ) * 100

        return {
            'score': min(score / 100, 1.0),
            'green_space_ratio': green_space_ratio,
            'native_vegetation': native_vegetation_ratio,
            'connectivity': habitat_connectivity
        }

    def _assess_noise(self, plan):
        """Noise pollution assessment"""
        traffic_volume = plan.get('daily_traffic', 10000)
        industrial_area = plan.get('industrial_area', 0)
        noise_barriers = plan.get('noise_barriers', False)

        # Noise level estimation (dB)
        traffic_noise = 50 + 10 * np.log10(traffic_volume / 1000)
        industrial_noise = industrial_area * 0.01

        total_noise = traffic_noise + industrial_noise

        if noise_barriers:
            total_noise -= 10  # 10 dB reduction

        # Score (target < 55 dB)
        score = max(0, 100 - (total_noise - 55) * 5)

        return {
            'score': score / 100,
            'estimated_noise_level': total_noise,
            'target': 55
        }

    def _assess_heat_island(self, plan):
        """Urban heat island assessment"""
        building_coverage = plan.get('building_coverage', 0.3)
        green_space_ratio = plan.get('green_space_ratio', 0.2)
        tree_canopy_ratio = plan.get('tree_canopy_ratio', 0.15)
        cool_roof_ratio = plan.get('cool_roof_ratio', 0.1)

        # Temperature increase estimation
        temp_increase = (
            building_coverage * 5 -
            green_space_ratio * 3 -
            tree_canopy_ratio * 4 -
            cool_roof_ratio * 2
        )

        score = max(0, 100 - temp_increase * 20)

        return {
            'score': score / 100,
            'temp_increase': temp_increase,
            'mitigation_score': (tree_canopy_ratio + cool_roof_ratio) / 2
        }

    def _air_quality_recommendation(self, score):
        if score >= 80:
            return "Excellent air quality expected"
        elif score >= 60:
            return "Good air quality with minor concerns"
        elif score >= 40:
            return "Moderate - consider more green space"
        else:
            return "Poor - significant mitigation required"

    def _get_rating(self, score):
        if score >= 0.9:
            return "Excellent"
        elif score >= 0.75:
            return "Good"
        elif score >= 0.6:
            return "Satisfactory"
        elif score >= 0.4:
            return "Poor"
        else:
            return "Unacceptable"

Example 3: Population Growth Prediction

import numpy as np
from scipy.optimize import curve_fit

class PopulationGrowthPredictor:
    def __init__(self):
        self.models = {}

    def fit_logistic_model(self, years, population):
        """Fit logistic growth model"""
        def logistic(t, K, P0, r):
            """K: carrying capacity, P0: initial pop, r: growth rate"""
            return K / (1 + ((K - P0) / P0) * np.exp(-r * (t - years[0])))

        # Initial parameter guess
        K_init = max(population) * 1.5
        P0_init = population[0]
        r_init = 0.02

        try:
            params, _ = curve_fit(
                logistic,
                years,
                population,
                p0=[K_init, P0_init, r_init],
                maxfev=10000
            )

            self.models['logistic'] = {
                'params': params,
                'func': logistic
            }

            return params
        except:
            return None

    def predict(self, future_years, model='logistic'):
        """Predict future population"""
        if model not in self.models:
            raise ValueError(f"Model {model} not fitted")

        model_data = self.models[model]
        predictions = model_data['func'](
            future_years,
            *model_data['params']
        )

        return predictions

# Usage
years = np.array([2010, 2012, 2014, 2016, 2018, 2020, 2022, 2024])
population = np.array([100000, 115000, 132000, 148000, 165000,
                       180000, 195000, 208000])

predictor = PopulationGrowthPredictor()
predictor.fit_logistic_model(years, population)

# Predict 2025-2035
future_years = np.arange(2025, 2036)
predictions = predictor.predict(future_years)
