2.1 IoT Sensor Network Architecture

IoT sensor integration is critical for creating a responsive, data-driven digital twin city. This chapter covers comprehensive sensor network design, device protocols, real-time data streaming, edge computing frameworks, and device lifecycle management for city-wide IoT deployments.

The WIA Digital Twin City Standard defines a unified IoT architecture that supports heterogeneous sensor types, protocols, and data formats while ensuring interoperability, security, and scalability across the entire urban infrastructure.

Sensor Network Components

  • Sensor Devices: Environmental, traffic, structural, energy monitoring sensors
  • Edge Gateways: Local data processing and protocol translation
  • Network Infrastructure: LoRaWAN, NB-IoT, 5G, WiFi mesh networks
  • Data Platform: Cloud/edge data ingestion and processing
  • Analytics Engine: Real-time analytics and anomaly detection

Example 1: MQTT Sensor Data Publisher

import paho.mqtt.client as mqtt
import json
import time
from datetime import datetime
import random

class IoTSensorPublisher:
    def __init__(self, broker_host, broker_port=1883, client_id=None):
        self.broker_host = broker_host
        self.broker_port = broker_port
        self.client_id = client_id or f"sensor_{random.randint(1000, 9999)}"
        self.client = mqtt.Client(client_id=self.client_id)
        self.client.on_connect = self.on_connect
        self.client.on_publish = self.on_publish
        self.connected = False

    def on_connect(self, client, userdata, flags, rc):
        if rc == 0:
            print(f"Connected to MQTT broker: {self.broker_host}")
            self.connected = True
        else:
            print(f"Connection failed with code {rc}")

    def on_publish(self, client, userdata, mid):
        print(f"Message {mid} published successfully")

    def connect(self, username=None, password=None):
        """Connect to MQTT broker"""
        if username and password:
            self.client.username_pw_set(username, password)

        self.client.connect(self.broker_host, self.broker_port, 60)
        self.client.loop_start()

        # Wait for connection
        timeout = 10
        start_time = time.time()
        while not self.connected and (time.time() - start_time) < timeout:
            time.sleep(0.1)

        return self.connected

    def publish_sensor_data(self, sensor_id, sensor_type, data, location=None):
        """Publish sensor data to MQTT topic"""
        topic = f"city/sensors/{sensor_type}/{sensor_id}"

        payload = {
            "sensorId": sensor_id,
            "sensorType": sensor_type,
            "timestamp": datetime.utcnow().isoformat(),
            "data": data,
            "location": location or {"lat": 0, "lon": 0, "alt": 0},
            "quality": self._calculate_data_quality(data),
            "batteryLevel": random.uniform(20, 100),
            "signalStrength": random.randint(-90, -40)
        }

        message = json.dumps(payload)
        result = self.client.publish(topic, message, qos=1, retain=False)

        return result.rc == mqtt.MQTT_ERR_SUCCESS

    def _calculate_data_quality(self, data):
        """Calculate data quality score"""
        # Simplified quality calculation
        if all(v is not None for v in data.values()):
            return "high"
        elif any(v is not None for v in data.values()):
            return "medium"
        else:
            return "low"

    def publish_environmental_data(self, sensor_id, location):
        """Publish environmental sensor data"""
        data = {
            "temperature": round(random.uniform(-10, 40), 2),
            "humidity": round(random.uniform(20, 90), 2),
            "pressure": round(random.uniform(980, 1030), 2),
            "pm25": round(random.uniform(0, 150), 2),
            "pm10": round(random.uniform(0, 200), 2),
            "co2": random.randint(400, 2000),
            "noise": round(random.uniform(30, 90), 1)
        }

        return self.publish_sensor_data(
            sensor_id, "environmental", data, location
        )

    def publish_traffic_data(self, sensor_id, location):
        """Publish traffic sensor data"""
        data = {
            "vehicleCount": random.randint(0, 100),
            "averageSpeed": round(random.uniform(0, 80), 1),
            "congestionLevel": random.choice(["low", "medium", "high"]),
            "occupancy": round(random.uniform(0, 100), 1),
            "incidents": random.randint(0, 3)
        }

        return self.publish_sensor_data(
            sensor_id, "traffic", data, location
        )

    def disconnect(self):
        """Disconnect from MQTT broker"""
        self.client.loop_stop()
        self.client.disconnect()
        self.connected = False

# Usage example
publisher = IoTSensorPublisher("mqtt.digitaltwin.city", 1883)

Example 2: Edge Computing Data Processing

import asyncio
import json
from collections import deque
from datetime import datetime, timedelta
import numpy as np

class EdgeComputingGateway:
    def __init__(self, gateway_id, buffer_size=1000):
        self.gateway_id = gateway_id
        self.buffer_size = buffer_size
        self.data_buffer = deque(maxlen=buffer_size)
        self.aggregated_data = {}
        self.anomaly_threshold = 3.0  # Standard deviations

    async def process_sensor_stream(self, sensor_data_queue):
        """Process incoming sensor data stream"""
        while True:
            try:
                # Get sensor data from queue
                sensor_data = await asyncio.wait_for(
                    sensor_data_queue.get(),
                    timeout=1.0
                )

                # Add to buffer
                self.data_buffer.append(sensor_data)

                # Process data
                processed_data = self._process_data_point(sensor_data)

                # Detect anomalies
                if self._is_anomaly(processed_data):
                    await self._handle_anomaly(processed_data)

                # Aggregate data
                self._aggregate_data(processed_data)

                # Send to cloud if threshold met
                if self._should_send_to_cloud():
                    await self._send_to_cloud()

            except asyncio.TimeoutError:
                # No data received, continue
                continue
            except Exception as e:
                print(f"Error processing sensor data: {e}")

    def _process_data_point(self, sensor_data):
        """Process individual sensor data point"""
        processed = {
            "sensorId": sensor_data["sensorId"],
            "timestamp": sensor_data["timestamp"],
            "processedAt": datetime.utcnow().isoformat(),
            "gatewayId": self.gateway_id,
            "data": {}
        }

        # Apply calibration and filtering
        for key, value in sensor_data["data"].items():
            if value is not None:
                calibrated = self._calibrate_value(key, value)
                filtered = self._apply_filter(key, calibrated)
                processed["data"][key] = filtered

        return processed

    def _calibrate_value(self, sensor_type, value):
        """Apply sensor calibration"""
        calibration_factors = {
            "temperature": {"offset": -0.5, "scale": 1.02},
            "humidity": {"offset": 2.0, "scale": 0.98},
            "pm25": {"offset": 0, "scale": 1.05}
        }

        if sensor_type in calibration_factors:
            factor = calibration_factors[sensor_type]
            return (value + factor["offset"]) * factor["scale"]

        return value

    def _apply_filter(self, sensor_type, value):
        """Apply moving average filter"""
        # Get recent values for this sensor type
        recent_values = [
            data["data"].get(sensor_type)
            for data in list(self.data_buffer)[-10:]
            if sensor_type in data.get("data", {})
        ]

        if len(recent_values) >= 3:
            # Simple moving average
            avg = np.mean(recent_values)
            # Weighted average: 70% new value, 30% historical
            return 0.7 * value + 0.3 * avg

        return value

    def _is_anomaly(self, processed_data):
        """Detect anomalies using statistical methods"""
        for key, value in processed_data["data"].items():
            # Get historical values
            historical = [
                data["data"].get(key)
                for data in list(self.data_buffer)[-100:]
                if key in data.get("data", {})
            ]

            if len(historical) >= 30:
                mean = np.mean(historical)
                std = np.std(historical)

                # Check if value is beyond threshold
                if abs(value - mean) > (self.anomaly_threshold * std):
                    return True

        return False

    async def _handle_anomaly(self, data):
        """Handle detected anomaly"""
        anomaly_event = {
            "type": "anomaly_detected",
            "gatewayId": self.gateway_id,
            "sensorId": data["sensorId"],
            "timestamp": datetime.utcnow().isoformat(),
            "data": data["data"],
            "severity": self._calculate_severity(data)
        }

        # Send immediate alert to cloud
        print(f"ANOMALY DETECTED: {json.dumps(anomaly_event, indent=2)}")
        # In production: await self._send_alert(anomaly_event)

    def _calculate_severity(self, data):
        """Calculate anomaly severity"""
        # Simplified severity calculation
        deviation_scores = []

        for key, value in data["data"].items():
            historical = [
                d["data"].get(key)
                for d in list(self.data_buffer)[-100:]
                if key in d.get("data", {})
            ]

            if len(historical) >= 30:
                mean = np.mean(historical)
                std = np.std(historical)
                if std > 0:
                    z_score = abs(value - mean) / std
                    deviation_scores.append(z_score)

        if deviation_scores:
            max_deviation = max(deviation_scores)
            if max_deviation > 5:
                return "critical"
            elif max_deviation > 3:
                return "high"
            elif max_deviation > 2:
                return "medium"

        return "low"

    def _aggregate_data(self, processed_data):
        """Aggregate sensor data for efficient transmission"""
        sensor_id = processed_data["sensorId"]

        if sensor_id not in self.aggregated_data:
            self.aggregated_data[sensor_id] = {
                "sensorId": sensor_id,
                "readings": [],
                "statistics": {}
            }

        self.aggregated_data[sensor_id]["readings"].append(processed_data)

        # Calculate statistics
        self._calculate_statistics(sensor_id)

    def _calculate_statistics(self, sensor_id):
        """Calculate statistical summaries"""
        readings = self.aggregated_data[sensor_id]["readings"]

        if len(readings) > 0:
            # Get all data keys
            data_keys = set()
            for reading in readings:
                data_keys.update(reading["data"].keys())

            stats = {}
            for key in data_keys:
                values = [
                    r["data"].get(key)
                    for r in readings
                    if key in r["data"]
                ]

                if values:
                    stats[key] = {
                        "min": min(values),
                        "max": max(values),
                        "avg": np.mean(values),
                        "std": np.std(values),
                        "count": len(values)
                    }

            self.aggregated_data[sensor_id]["statistics"] = stats

    def _should_send_to_cloud(self):
        """Determine if aggregated data should be sent to cloud"""
        # Send every 5 minutes or when buffer is 80% full
        total_readings = sum(
            len(data["readings"])
            for data in self.aggregated_data.values()
        )

        return total_readings >= (self.buffer_size * 0.8)

    async def _send_to_cloud(self):
        """Send aggregated data to cloud platform"""
        cloud_payload = {
            "gatewayId": self.gateway_id,
            "timestamp": datetime.utcnow().isoformat(),
            "aggregatedData": self.aggregated_data,
            "dataPoints": sum(
                len(data["readings"])
                for data in self.aggregated_data.values()
            )
        }

        print(f"Sending to cloud: {cloud_payload['dataPoints']} data points")
        # In production: await self._http_post_to_cloud(cloud_payload)

        # Clear aggregated data
        self.aggregated_data = {}
