5.1 Emergency Management Framework

Effective emergency response requires real-time coordination, predictive analytics, and seamless integration between multiple agencies. This chapter covers disaster prediction systems, emergency coordination protocols, evacuation planning, resource allocation algorithms, and multi-agency communication standards for digital twin cities.

Emergency Types and Response Levels

Emergency Type Response Level Response Time Agencies Involved Priority
Fire Emergency Level 1-3 < 4 minutes Fire, Police, Medical Critical
Natural Disaster Level 2-5 Pre-event prep All agencies, Military Critical
Medical Emergency Level 1-2 < 8 minutes EMS, Hospitals Critical
Hazmat Incident Level 2-4 < 10 minutes Fire, Hazmat, EPA High
Infrastructure Failure Level 1-3 < 30 minutes Utilities, Engineering Medium-High
Security Threat Level 1-5 < 2 minutes Police, Counter-terrorism Critical

import asyncio

from datetime import datetime
from typing import Dict, List
from enum import Enum

class EmergencyType(Enum):
    FIRE = "fire"
    MEDICAL = "medical"
    HAZMAT = "hazmat"
    NATURAL_DISASTER = "natural_disaster"
    SECURITY = "security"
    INFRASTRUCTURE = "infrastructure"

class Severity(Enum):
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    CRITICAL = 4
    CATASTROPHIC = 5

class EmergencyDetectionSystem:
    def __init__(self):
        self.active_emergencies = {}
        self.sensor_network = {}
        self.responders = {}

    async def monitor_sensors(self):
        """Continuous sensor monitoring for emergency detection"""
        while True:
            for sensor_id, sensor_data in self.sensor_network.items():
                # Analyze sensor data for anomalies
                emergency = await self._detect_emergency(sensor_data)

                if emergency:
                    await self.trigger_emergency_response(emergency)

            await asyncio.sleep(1)  # Check every second

    async def _detect_emergency(self, sensor_data: Dict):
        """AI-based emergency detection"""
        emergency_type = None
        severity = Severity.LOW

        # Fire detection
        if sensor_data.get('temperature', 0) > 60:  # Celsius
            emergency_type = EmergencyType.FIRE
            severity = Severity.CRITICAL if sensor_data['temperature'] > 80 else Severity.HIGH

        # Gas leak detection
        elif sensor_data.get('gas_concentration', 0) > 100:  # ppm
            emergency_type = EmergencyType.HAZMAT
            severity = Severity.CRITICAL

        # Structural integrity
        elif sensor_data.get('vibration_level', 0) > 5.0:
            emergency_type = EmergencyType.INFRASTRUCTURE
            severity = Severity.HIGH

        # Security breach
        elif sensor_data.get('intrusion_detected', False):
            emergency_type = EmergencyType.SECURITY
            severity = Severity.HIGH

        if emergency_type:
            return {
                'emergency_id': self._generate_emergency_id(),
                'type': emergency_type.value,
                'severity': severity.value,
                'location': sensor_data['location'],
                'sensor_id': sensor_data['sensor_id'],
                'timestamp': datetime.utcnow().isoformat(),
                'sensor_readings': sensor_data
            }

        return None

    async def trigger_emergency_response(self, emergency: Dict):
        """Trigger multi-agency emergency response"""
        emergency_id = emergency['emergency_id']

        # Store active emergency
        self.active_emergencies[emergency_id] = emergency

        # Dispatch to appropriate agencies
        agencies = self._determine_responding_agencies(emergency)

        # Send alerts
        alert_tasks = [
            self._alert_agency(agency, emergency)
            for agency in agencies
        ]

        await asyncio.gather(*alert_tasks)

        # Optimize resource allocation
        allocation = await self._optimize_resource_allocation(emergency, agencies)

        # Update emergency with allocation
        emergency['resource_allocation'] = allocation

        # Real-time coordination
        await self._coordinate_response(emergency_id)

    def _determine_responding_agencies(self, emergency: Dict) -> List[str]:
        """Determine which agencies should respond"""
        emergency_type = emergency['type']
        severity = emergency['severity']

        agency_map = {
            'fire': ['fire_department', 'police', 'ems'],
            'medical': ['ems', 'hospitals'],
            'hazmat': ['fire_department', 'hazmat_team', 'epa'],
            'natural_disaster': ['all_agencies', 'emergency_management', 'military'],
            'security': ['police', 'counter_terrorism'],
            'infrastructure': ['utilities', 'engineering', 'fire_department']
        }

        agencies = agency_map.get(emergency_type, ['emergency_management'])

        # Add additional agencies for high severity
        if severity >= Severity.CRITICAL.value:
            agencies.extend(['emergency_management', 'hospitals', 'military'])

        return list(set(agencies))  # Remove duplicates

    async def _alert_agency(self, agency: str, emergency: Dict):
        """Send emergency alert to agency"""
        alert = {
            'alert_id': f"ALERT_{datetime.utcnow().timestamp()}",
            'agency': agency,
            'emergency': emergency,
            'priority': 'CRITICAL' if emergency['severity'] >= 4 else 'HIGH',
            'timestamp': datetime.utcnow().isoformat()
        }

        print(f"🚨 ALERT to {agency}: {emergency['type']} - Severity {emergency['severity']}")
        # In production: await self._send_to_agency_system(agency, alert)

    async def _optimize_resource_allocation(self, emergency: Dict,
                                           agencies: List[str]) -> Dict:
        """Optimize allocation of emergency resources"""
        location = emergency['location']
        emergency_type = emergency['type']

        allocation = {
            'vehicles': [],
            'personnel': [],
            'equipment': []
        }

        # Find nearest available resources
        for agency in agencies:
            available_resources = await self._get_available_resources(agency)

            # Calculate optimal deployment
            for resource in available_resources:
                distance = self._calculate_distance(
                    resource['location'],
                    location
                )

                estimated_arrival = distance / resource['avg_speed']

                allocation['vehicles'].append({
                    'resource_id': resource['id'],
                    'agency': agency,
                    'type': resource['type'],
                    'eta_minutes': estimated_arrival,
                    'assigned_route': self._calculate_optimal_route(
                        resource['location'],
                        location
                    )
                })

        return allocation

    async def _get_available_resources(self, agency: str) -> List[Dict]:
        """Get available emergency resources for agency"""

            {
                'id': f"{agency}_UNIT_001",

                'location': (37.5665, 126.9780),
                'avg_speed': 60,  # km/h
                'available': True
            }
        ]

    def _calculate_distance(self, loc1, loc2):
        """Calculate distance between two locations"""
        import math
        lat1, lon1 = loc1
        lat2, lon2 = loc2

        R = 6371  # Earth radius in km

        dlat = math.radians(lat2 - lat1)
        dlon = math.radians(lon2 - lon1)

        a = (math.sin(dlat/2)**2 +
             math.cos(math.radians(lat1)) * math.cos(math.radians(lat2)) *
             math.sin(dlon/2)**2)

        c = 2 * math.asin(math.sqrt(a))

        return R * c

    def _calculate_optimal_route(self, start, end):
        """Calculate optimal route using traffic data"""
        # Simplified - use traffic simulation for real-time routing
        return {
            'waypoints': [start, end],
            'distance_km': self._calculate_distance(start, end),
            'traffic_delay': 0
        }

    async def _coordinate_response(self, emergency_id: str):
        """Coordinate multi-agency response in real-time"""
        emergency = self.active_emergencies.get(emergency_id)

        if not emergency:
            return

        coordination_channel = {
            'emergency_id': emergency_id,
            'incident_commander': 'determine_based_on_type',
            'communication_protocol': 'P25',
            'shared_resources': emergency.get('resource_allocation', {}),
            'real_time_updates': []
        }

        # Set up real-time coordination
        print(f"Coordination established for {emergency_id}")

    def _generate_emergency_id(self):
        """Generate unique emergency ID"""
        return f"EMG_{datetime.utcnow().strftime('%Y%m%d_%H%M%S_%f')}"

Example 2: Evacuation Planning Algorithm

import numpy as np
from typing import List, Dict, Tuple
import networkx as nx

class EvacuationPlanner:
    def __init__(self, city_graph: nx.Graph):
        self.city_graph = city_graph
        self.safe_zones = []
        self.building_populations = {}

    def plan_evacuation(self, affected_area: Dict,
                       population_data: Dict) -> Dict:
        """Generate optimal evacuation plan"""
        # Identify affected buildings
        affected_buildings = self._identify_affected_buildings(affected_area)

        # Calculate total population to evacuate
        total_population = sum(
            population_data.get(bldg, 0)
            for bldg in affected_buildings
        )

        # Find nearest safe zones
        safe_zones = self._identify_safe_zones(affected_area)

        # Calculate evacuation routes
        evacuation_routes = self._calculate_evacuation_routes(
            affected_buildings,
            safe_zones
        )

        # Estimate evacuation time
        evacuation_time = self._estimate_evacuation_time(
            evacuation_routes,
            population_data
        )

        # Generate evacuation phases
        phases = self._generate_evacuation_phases(
            affected_buildings,
            population_data,
            evacuation_routes
        )

        return {
            'total_population': total_population,
            'affected_buildings': len(affected_buildings),
            'safe_zones': safe_zones,
            'evacuation_routes': evacuation_routes,
            'estimated_time_minutes': evacuation_time,
            'evacuation_phases': phases,
            'resource_requirements': self._calculate_resource_requirements(
                total_population
            )
        }

    def _identify_affected_buildings(self, affected_area: Dict) -> List[str]:
        """Identify buildings in affected area"""
        center = affected_area['center']
        radius = affected_area['radius']

        affected = []
        for node in self.city_graph.nodes():
            if self.city_graph.nodes[node].get('type') == 'building':
                location = self.city_graph.nodes[node]['location']
                distance = self._calculate_distance(center, location)

                if distance <= radius:
                    affected.append(node)

        return affected

    def _identify_safe_zones(self, affected_area: Dict) -> List[Dict]:
        """Identify safe zones outside affected area"""
        safe_zones = []

        for node in self.city_graph.nodes():
            if self.city_graph.nodes[node].get('type') == 'safe_zone':
                location = self.city_graph.nodes[node]['location']
                distance = self._calculate_distance(
                    affected_area['center'],
                    location
                )

                if distance > affected_area['radius'] * 1.5:
                    safe_zones.append({
                        'id': node,
                        'location': location,
                        'capacity': self.city_graph.nodes[node].get('capacity', 1000),
                        'distance_from_event': distance
                    })

        return sorted(safe_zones, key=lambda x: x['distance_from_event'])

    def _calculate_evacuation_routes(self, buildings: List[str],
                                    safe_zones: List[Dict]) -> List[Dict]:
        """Calculate optimal evacuation routes"""
        routes = []

        for building in buildings:
            building_loc = self.city_graph.nodes[building]['location']

            # Find best safe zone
            best_zone = None
            best_path = None
            min_time = float('inf')

            for zone in safe_zones[:3]:  # Consider top 3 closest
                try:
                    path = nx.shortest_path(
                        self.city_graph,
                        building,
                        zone['id'],
                        weight='travel_time'
                    )

                    path_time = nx.shortest_path_length(
                        self.city_graph,
                        building,
                        zone['id'],
                        weight='travel_time'
                    )

                    if path_time < min_time:
                        min_time = path_time
                        best_path = path
                        best_zone = zone

                except nx.NetworkXNoPath:
                    continue

            if best_path:
                routes.append({
                    'building': building,
                    'safe_zone': best_zone['id'],
                    'path': best_path,
                    'distance_km': self._calculate_path_distance(best_path),
                    'estimated_time_min': min_time
                })

        return routes

    def _estimate_evacuation_time(self, routes: List[Dict],
                                 population_data: Dict) -> float:
        """Estimate total evacuation time"""
        if not routes:
            return 0

        # Consider bottlenecks and congestion
        max_time = max(route['estimated_time_min'] for route in routes)

        # Add overhead for coordination and congestion
        congestion_factor = 1.5
        coordination_overhead = 10  # minutes

        return max_time * congestion_factor + coordination_overhead

    def _generate_evacuation_phases(self, buildings: List[str],
                                   population_data: Dict,
                                   routes: List[Dict]) -> List[Dict]:
        """Generate phased evacuation plan"""
        # Prioritize vulnerable populations
        phases = [
            {
                'phase': 1,
                'priority': 'Critical',
                'description': 'Hospitals, nursing homes, schools',
                'buildings': [],
                'estimated_duration_min': 15
            },
            {
                'phase': 2,
                'priority': 'High',
                'description': 'High-occupancy buildings',
                'buildings': [],
                'estimated_duration_min': 30
            },
            {
                'phase': 3,
                'priority': 'Medium',
                'description': 'Residential areas',
                'buildings': [],
                'estimated_duration_min': 45
            }
        ]

        for building in buildings:
            bldg_type = self.city_graph.nodes[building].get('building_type', 'residential')

            if bldg_type in ['hospital', 'school', 'nursing_home']:
                phases[0]['buildings'].append(building)
            elif population_data.get(building, 0) > 500:
                phases[1]['buildings'].append(building)
            else:
                phases[2]['buildings'].append(building)

        return phases

    def _calculate_resource_requirements(self, population: int) -> Dict:
        """Calculate emergency resources needed"""
        return {
            'buses': int(np.ceil(population / 50)),
            'emergency_personnel': int(np.ceil(population / 100)),
            'medical_units': int(np.ceil(population / 500)),
            'communication_units': 5
        }

    def _calculate_distance(self, loc1: Tuple, loc2: Tuple) -> float:
        """Calculate distance between locations"""
        return np.sqrt((loc1[0] - loc2[0])**2 + (loc1[1] - loc2[1])**2)

    def _calculate_path_distance(self, path: List[str]) -> float:
        """Calculate total path distance"""
        total_distance = 0
        for i in range(len(path) - 1):
            loc1 = self.city_graph.nodes[path[i]]['location']
            loc2 = self.city_graph.nodes[path[i+1]]['location']
            total_distance += self._calculate_distance(loc1, loc2)
        return total_distance

import asyncio
import json
from typing import Dict, List
from datetime import datetime

class MultiAgencyCoordination:
    def __init__(self):
        self.agencies = {}
        self.communication_channels = {}
        self.shared_situation_awareness = {}

    def register_agency(self, agency_id: str, agency_info: Dict):
        """Register emergency response agency"""
        self.agencies[agency_id] = {
            'info': agency_info,
            'status': 'available',
            'current_incidents': [],
            'resources': agency_info.get('resources', [])
        }

    async def establish_coordination(self, incident_id: str,
                                    participating_agencies: List[str]):
        """Establish multi-agency coordination for incident"""
        # Create shared communication channel
        channel = {
            'incident_id': incident_id,
            'agencies': participating_agencies,
            'incident_commander': self._designate_commander(participating_agencies),
            'created_at': datetime.utcnow().isoformat(),
            'messages': [],
            'shared_resources': [],
            'situation_updates': []
        }

        self.communication_channels[incident_id] = channel

        # Initialize Common Operating Picture (COP)
        await self._initialize_cop(incident_id)

        return channel

    def _designate_commander(self, agencies: List[str]) -> str:
        """Designate incident commander based on incident type"""
        # Priority order for incident command
        priority = [
            'fire_department',
            'police',
            'emergency_management',
            'ems'
        ]

        for agency_type in priority:
            matching = [a for a in agencies if agency_type in a]
            if matching:
                return matching[0]

        return agencies[0] if agencies else None

    async def _initialize_cop(self, incident_id: str):
        """Initialize Common Operating Picture"""
        self.shared_situation_awareness[incident_id] = {
            'incident_location': None,
            'incident_type': None,
            'severity': None,
            'affected_area': None,
            'deployed_resources': [],
            'casualties': 0,
            'evacuations': 0,
            'hazards': [],
            'weather_conditions': None,
            'timeline': [],
            'last_updated': datetime.utcnow().isoformat()
        }

    async def update_situation_awareness(self, incident_id: str,
                                        update: Dict):
        """Update shared situation awareness"""
        if incident_id not in self.shared_situation_awareness:
            return

        cop = self.shared_situation_awareness[incident_id]
        cop.update(update)
        cop['last_updated'] = datetime.utcnow().isoformat()

        # Broadcast update to all agencies
        await self._broadcast_update(incident_id, update)

    async def _broadcast_update(self, incident_id: str, update: Dict):
        """Broadcast update to all participating agencies"""
        channel = self.communication_channels.get(incident_id)

        if not channel:
            return

        message = {
            'type': 'situation_update',
            'incident_id': incident_id,
            'timestamp': datetime.utcnow().isoformat(),
            'update': update
        }

        # Send to all agencies
        for agency in channel['agencies']:
            await self._send_to_agency(agency, message)

    async def request_resource(self, incident_id: str,
                              requesting_agency: str,
                              resource_type: str,
                              quantity: int):
        """Request resource from other agencies"""
        request = {
            'request_id': f"REQ_{datetime.utcnow().timestamp()}",
            'incident_id': incident_id,
            'requesting_agency': requesting_agency,
            'resource_type': resource_type,
            'quantity': quantity,
            'status': 'pending'
        }

        # Find agencies with available resources
        available_agencies = self._find_agencies_with_resource(
            resource_type,
            quantity,
            incident_id
        )

        if available_agencies:
            # Allocate from nearest agency
            request['allocated_from'] = available_agencies[0]

                                    quantity: int,
                                    incident_id: str) -> List[str]:
        """Find agencies with requested resource"""
        available = []

        for agency_id, agency_data in self.agencies.items():
            resources = agency_data.get('resources', [])

            matching = [
                r for r in resources
                if r['type'] == resource_type and r['available'] >= quantity
            ]

            if matching:
                available.append(agency_id)

        return available
