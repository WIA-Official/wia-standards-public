8.2 Quick Start Guides

8.2.1 AWS Quick Start

# 1. Create S3 bucket
aws s3 mb s3://my-datalake --region us-east-1

# 2. Set up Glue catalog
aws glue create-database --database-input '{
  "Name": "datalake",
  "Description": "WIA Data Lake"
}'

# 3. Launch EMR cluster with Delta Lake
aws emr create-cluster \
  --name "DataLake-Cluster" \
  --release-label emr-6.10.0 \
  --applications Name=Spark Name=Hive \
  --ec2-attributes KeyName=my-key \
  --instance-type m5.xlarge \
  --instance-count 3 \
  --configurations file://delta-config.json

# 4. Create first Delta table
spark-sql --packages io.delta:delta-core_2.12:2.4.0
> CREATE TABLE datalake.events
  USING delta
  LOCATION 's3://my-datalake/curated/events/';

8.2.2 Azure Quick Start

# 1. Create ADLS Gen2 account
az storage account create \
  --name mydatalake \
  --resource-group my-rg \
  --location eastus \
  --sku Standard_LRS \
  --kind StorageV2 \
  --hierarchical-namespace true

# 2. Create Databricks workspace
az databricks workspace create \
  --resource-group my-rg \
  --name my-workspace \
  --location eastus \
  --sku premium

# 3. Create Delta Lake table
%sql
CREATE DATABASE IF NOT EXISTS datalake;
CREATE TABLE datalake.events
USING delta
LOCATION 'abfss://curated@mydatalake.dfs.core.windows.net/events/';

8.2.3 GCP Quick Start

# 1. Create GCS bucket
gcloud storage buckets create gs://my-datalake --location=us-central1

# 2. Create Dataproc cluster
gcloud dataproc clusters create datalake-cluster \
  --region us-central1 \
  --zone us-central1-a \
  --master-machine-type n1-standard-4 \
  --worker-machine-type n1-standard-4 \
  --num-workers 2 \
  --image-version 2.1-debian11

# 3. Create Iceberg table
spark-sql --packages org.apache.iceberg:iceberg-spark-runtime-3.3_2.12:1.4.0
> CREATE TABLE datalake.events
  USING iceberg
  LOCATION 'gs://my-datalake/curated/events/';

8.3 Performance Tuning

8.3.1 File Size Optimization

Problem Symptom Solution
Small Files Slow queries, high listing overhead OPTIMIZE table (compact files to 128MB-1GB)
Large Files Low parallelism, OOM errors Repartition data before writing 
# Compact small files
OPTIMIZE events WHERE date = '2025-01-15';

# Repartition before writing
df.repartition(100).write.format("delta").save(...)

8.3.2 Partitioning Strategy

# Good: Balanced partitions (1-10 GB each)
year=2025/month=01/day=15/

# Bad: Over-partitioned (thousands of tiny partitions)
year=2025/month=01/day=15/hour=10/minute=30/

# Bad: Under-partitioned (100+ GB per partition)
year=2025/

8.3.3 Z-Ordering

# Multi-dimensional clustering for better data skipping
OPTIMIZE events ZORDER BY (user_id, event_type);

# Typical improvements:
# - 50-90% reduction in data scanned
# - 2-10x faster query performance
# - Works best with 2-4 columns

8.3.4 Caching Strategies

# Cache frequently accessed tables
CACHE TABLE events;

# Partitioned caching (Delta Cache in Databricks)
SELECT * FROM events
OPTIONS (cachingEnabled = true)
WHERE date >= '2025-01-01';

8.4 Security & Governance

8.4.1 Encryption

# Encryption at rest (S3)
aws s3api put-bucket-encryption \
  --bucket my-datalake \
  --server-side-encryption-configuration '{
    "Rules": [{
      "ApplyServerSideEncryptionByDefault": {
        "SSEAlgorithm": "aws:kms",
        "KMSMasterKeyID": "arn:aws:kms:..."
      }
    }]
  }'

# Encryption in transit
spark.hadoop.fs.s3a.connection.ssl.enabled=true

8.4.2 Access Control

# Table-level permissions (Unity Catalog)
GRANT SELECT ON TABLE curated.events TO analytics_team;
GRANT MODIFY ON TABLE curated.events TO data_engineers;
REVOKE ALL ON TABLE curated.sensitive_data FROM public;

# Column-level masking
CREATE FUNCTION mask_email(email STRING)
  RETURN CASE WHEN is_member('pii_viewers')
              THEN email
              ELSE 'REDACTED'
         END;

ALTER TABLE users ALTER COLUMN email SET MASK mask_email;

8.4.3 Data Quality Checks

from great_expectations import DataContext

context = DataContext()

# Define expectations
expectation_suite = context.create_expectation_suite("events_suite")
batch = context.get_batch(...)

batch.expect_column_values_to_not_be_null("user_id")
batch.expect_column_values_to_be_unique("id")
batch.expect_column_values_to_be_in_set("event_type", ["click", "purchase", "view"])
batch.expect_column_min_to_be_between("timestamp", min_value=datetime(2020,1,1))

# Validate
results = batch.validate()
if not results["success"]:
    raise DataQualityException("Quality checks failed")

8.5 Cost Optimization

8.5.1 Storage Tiering

Data Age Access Pattern Storage Class Cost Savings
0-30 days Frequent S3 Standard Baseline
30-90 days Infrequent S3 Intelligent-Tiering ~40%
90-365 days Rare S3 Glacier Instant ~70%
1+ years Archive S3 Glacier Deep Archive ~95%

8.5.2 Compute Optimization

# Auto-scaling EMR clusters
aws emr create-cluster \
  --auto-scaling-role EMR_AutoScaling_DefaultRole \
  --instance-groups \
    InstanceGroupType=MASTER,InstanceType=m5.xlarge,InstanceCount=1 \
    InstanceGroupType=CORE,InstanceType=m5.xlarge,InstanceCount=2,AutoScalingPolicy=file://scaling-policy.json

# Spot instances for cost savings (50-90% cheaper)
--instance-groups \
  InstanceGroupType=TASK,InstanceType=m5.xlarge,BidPrice=0.10,InstanceCount=10

8.5.3 Data Lifecycle Policies

# Automatic data archival
CREATE TABLE events (...)
TBLPROPERTIES (
  'delta.deletedFileRetentionDuration' = '7 days',
  'delta.logRetentionDuration' = '30 days'
);

# Vacuum old files regularly
VACUUM events RETAIN 168 HOURS;  -- Keep 7 days of history

8.7 Migration Strategies

8.7.1 From Data Warehouse to Data Lake

Phase 1: Parallel Run (Months 1-2)
- Keep existing warehouse
- Replicate data to lake
- Validate data consistency
- Build confidence

Phase 2: Hybrid Mode (Months 3-4)
- New workloads on lake
- Legacy reports on warehouse
- Gradual migration of queries

Phase 3: Full Migration (Months 5-6)
- All workloads on lake
- Decommission warehouse
- Cost savings realized

8.7.2 From Hive to Iceberg/Delta

# In-place conversion (Iceberg)
ALTER TABLE hive_table
SET TBLPROPERTIES (
  'table_type'='ICEBERG',
  'format-version'='2'
);

# Or: Create new table + migrate
CREATE TABLE events_iceberg USING iceberg AS
SELECT * FROM hive_events;

8.8 Troubleshooting Guide

Problem Cause Solution
Slow Queries Small files, no partitioning OPTIMIZE + Z-ORDER, add partitions
OOM Errors Large files, skewed data Repartition, increase executor memory
Data Corruption Concurrent writes without ACID Use Delta/Iceberg/Hudi, never Parquet directly
High Costs No lifecycle policies Implement tiered storage, vacuum old data
Schema Conflicts Incompatible schema changes Use schema evolution, test migrations
