Chapter 5: Phase 2 - API Interface Standard

A consistent API is worth a thousand documentation pages.

The WIA Data Lake API provides a unified interface across all implementations.

5.1 Core API Design Philosophy

The WIA Data Lake API follows these principles:

  • Simplicity: Common operations should be one-liners
  • Consistency: Same patterns across languages
  • Composability: Methods chain naturally
  • Extensibility: Easy to add custom functionality

5.2 WiaDataLake Class

5.2.1 Initialization

# Python
from wia_datalake import WiaDataLake

lake = WiaDataLake(
    storage_path="s3://my-datalake/",
    catalog_uri="glue://us-east-1/my-catalog",
    table_format="delta"  # or "iceberg", "hudi"
)

# TypeScript
import { WiaDataLake } from '@wia/datalake';

const lake = new WiaDataLake({
  storagePath: 's3://my-datalake/',
  catalogUri: 'glue://us-east-1/my-catalog',
  tableFormat: 'delta'
});

5.2.2 Core Methods

Method Purpose Returns
ingest() Load data into lake IngestJob
query() Execute SQL query DataFrame
get_metadata() Retrieve table metadata TableMetadata
optimize() Compact files, z-order OptimizeResult
vacuum() Clean up old files VacuumResult

5.3 Data Ingestion API

5.3.1 Basic Ingestion

# Ingest JSON files to raw zone
job = lake.ingest(
    source="s3://source-bucket/events/*.json",
    target_zone="raw",
    target_table="events",
    format="parquet",
    mode="append"
)

# Monitor job
print(f"Status: {job.status}")
print(f"Records processed: {job.records_processed}")
print(f"Duration: {job.duration_ms} ms")

5.3.2 Advanced Ingestion Options

# With partitioning, schema, and transformation
job = lake.ingest(
    source="s3://source/data/",
    target_zone="curated",
    target_table="events_processed",
    format="delta",
    partition_by=["year", "month", "day"],
    schema={
        "user_id": "bigint",
        "event_type": "string",
        "timestamp": "timestamp",
        "properties": "map"
    },
    transform=lambda df: df.filter("event_type != 'test'"),
    mode="overwrite",
    compression="zstd"
)

5.3.3 Streaming Ingestion

# Continuous ingestion from Kafka
stream_job = lake.ingest_stream(
    source="kafka://localhost:9092/events",
    target_zone="curated",
    target_table="real_time_events",
    format="delta",
    checkpoint_location="s3://checkpoints/events/",
    trigger="processingTime='10 seconds'"
)

5.4 Query Interface

5.4.1 SQL Query API

# Execute SQL query
df = lake.query("""
    SELECT
        user_id,
        COUNT(*) as event_count,
        MAX(timestamp) as last_event
    FROM curated.events
    WHERE date = '2025-01-15'
    GROUP BY user_id
    ORDER BY event_count DESC
    LIMIT 100
""")

# Results as DataFrame
for row in df.collect():
    print(f"User {row.user_id}: {row.event_count} events")

5.4.2 DataFrame API

# Programmatic query building
df = lake.table("curated.events") \
    .filter("date = '2025-01-15'") \
    .groupBy("user_id") \
    .agg({"event_type": "count", "timestamp": "max"}) \
    .orderBy("count", ascending=False) \
    .limit(100)

5.4.3 Time Travel Queries

# Query historical version (Delta Lake)
df = lake.query("""
    SELECT * FROM curated.events
    VERSION AS OF 10
""")

# Query as of timestamp
df = lake.query("""
    SELECT * FROM curated.events
    TIMESTAMP AS OF '2025-01-15 10:00:00'
""")

5.5 Metadata Operations

5.5.1 Get Table Metadata

metadata = lake.get_metadata("curated.events")

print(f"Table: {metadata.table_name}")
print(f"Location: {metadata.location}")
print(f"Format: {metadata.format}")
print(f"Schema: {metadata.schema}")
print(f"Partitions: {metadata.partitions}")
print(f"Row count: {metadata.num_rows}")
print(f"Size: {metadata.size_bytes / 1e9:.2f} GB")

5.5.2 List Tables

# List all tables in a zone
tables = lake.list_tables(zone="curated")
for table in tables:
    print(f"{table.name}: {table.num_rows} rows, {table.size_gb:.2f} GB")

5.5.3 Schema Evolution

# Add new column
lake.add_column(
    table="curated.events",
    column_name="session_id",
    column_type="string",
    nullable=True
)

# Rename column
lake.rename_column(
    table="curated.events",
    old_name="user_id",
    new_name="customer_id"
)

5.6 Optimization Operations

5.6.1 File Compaction

# Compact small files
result = lake.optimize(
    table="curated.events",
    strategy="compact",
    target_file_size_mb=128
)

print(f"Files before: {result.files_before}")
print(f"Files after: {result.files_after}")
print(f"Space saved: {result.space_saved_mb:.2f} MB")

5.6.2 Z-Ordering

# Z-order for better query performance
result = lake.optimize(
    table="curated.events",
    strategy="zorder",
    zorder_by=["user_id", "date"]
)

print(f"Data skipping improvement: {result.skipping_improvement}%")

5.6.3 Vacuum Old Files

# Clean up files older than 7 days
result = lake.vacuum(
    table="curated.events",
    retention_hours=168  # 7 days
)

print(f"Files deleted: {result.files_deleted}")
print(f"Space freed: {result.space_freed_gb:.2f} GB")

5.7 Error Handling

from wia_datalake import (
    WiaDataLake,
    TableNotFoundException,
    SchemaValidationException,
    IngestionException
)

try:
    lake.ingest(
        source="s3://data/",
        target_table="invalid_table",
        schema={"id": "bigint"}
    )
except TableNotFoundException as e:
    print(f"Table not found: {e.table_name}")
except SchemaValidationException as e:
    print(f"Schema mismatch: {e.message}")
except IngestionException as e:
    print(f"Ingestion failed: {e.message}")
    print(f"Failed records: {e.failed_records}")

5.8 Python SDK Reference

class WiaDataLake:
    """WIA Data Lake Standard API"""

    def __init__(self,
                 storage_path: str,
                 catalog_uri: str,
                 table_format: str = "delta"):
        """Initialize data lake connection"""

    def ingest(self,
               source: str,
               target_zone: str,
               target_table: str,
               format: str = "parquet",
               partition_by: List[str] = None,
               mode: str = "append") -> IngestJob:
        """Ingest data into the lake"""

    def query(self, sql: str) -> DataFrame:
        """Execute SQL query"""

    def table(self, table_name: str) -> Table:
        """Get table handle"""

    def get_metadata(self, table_name: str) -> TableMetadata:
        """Get table metadata"""

    def optimize(self,
                 table: str,
                 strategy: str = "compact",
                 **kwargs) -> OptimizeResult:
        """Optimize table storage"""

    def vacuum(self,
               table: str,
               retention_hours: int = 168) -> VacuumResult:
        """Remove old data files"""
