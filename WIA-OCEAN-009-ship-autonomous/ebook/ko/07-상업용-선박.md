# 7장: 상업용 자율운항선박

## 자동화의 비즈니스 사례

자율운항선박은 상업 해운 운영을 변혁할 것을 약속합니다. 기술 시연이 실현 가능성을 입증했지만, 광범위한 채택은 경제적 실행 가능성, 운영 신뢰성 및 경쟁 우위에 달려 있습니다.

### 컨테이너선

컨테이너 해운은 전 세계 해상 무역의 60%를 가치 기준으로 운송합니다. 자동화는 이 부문에 상당한 이점을 제공합니다.

**경제 분석: 14,000 TEU 컨테이너선**

```typescript
interface ContainerShipEconomics {
  vessel: {
    type: "컨테이너선";
    capacity: "14,000 TEU";
    length: 366;  // 미터
    serviceSpeed: 22;  // 노트
  };

  conventionalOperation: {
    crewSize: 22;
    annualCrewCost: 35억원;  // 원화
    operatingCost: {
      crew: 35억원;
      fuel: 80억원;     // 200톤/일 × 300일 × 톤당 $800
      maintenance: 20억원;
      insurance: 15억원;
      port: 10억원;
      other: 10억원;
      total: 170억원
    };
    utilization: 0.85;  // 85% (선원 휴식, 항만 시간)
  };

  autonomousOperation: {
    crewSize: 0;  // 육상 기반 모니터링만
    annualCrewCost: 5억원;  // 육상 운영자 (3명 × 1.6억원)
    operatingCost: {
      crew: 5억원;
      fuel: 64억원;     // 최적화를 통한 20% 감소
      maintenance: 18억원;  // 10% 감소 (선원 숙소 없음)
      insurance: 12억원;    // 20% 감소 (사고 위험 감소)
      technology: 8억원;    // 자율 시스템 라이선스
      satellite: 6억원;     // 고대역폭 통신
      shoreControl: 4억원;  // 육상 제어 센터 할당
      other: 10억원;
      total: 127억원
    };
    utilization: 0.95;  // 95% (24/7 운영, 더 빠른 항만 처리)
  };

  savings: {
    annualOperatingCost: 43억원;  // 연간 43억원 절감
    utilizationGain: 0.10;         // 10% 더 많은 생산적 시간
    additionalRevenue: 20억원;     // 증가된 활용도에서
    totalBenefit: 63억원;          // 연간 63억원

    paybackPeriod: {
      autonomousSystemCost: 150억원;  // 150억원 개조 또는 250억원 신조 프리미엄
      payback: 2.4  // 년 (150억원 / 63억원)
    };
  };
}
```
