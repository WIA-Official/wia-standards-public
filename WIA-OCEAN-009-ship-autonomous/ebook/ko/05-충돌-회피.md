# 5장: 충돌 회피 시스템

## 상상할 수 없는 일의 예방

해상 충돌은 가장 흔하고 가장 두려운 해양 사고입니다. 자율 충돌 회피 시스템은 인간 항해사의 성능과 일치하거나 초과하여 더 빠르고 신뢰성 있게 위협에 대응하고 국제 규정을 완전히 준수해야 합니다.

### 충돌 회피 파이프라인

충돌 회피는 연속적인 주기로 작동합니다:

```typescript
interface CollisionAvoidanceSystem {

  detectThreats(environment: FusedEnvironmentModel): Threat[];

  // 2. 평가

  // 3. 계획

  executeManeuver(maneuver: AvoidanceManeuver): void;

  // 5. 모니터링

  // 6. 침로 복귀
  planReturnToCourse(clearOfThreat: boolean): CourseResumption;
}
```

### 위협 감지 및 분류

**위협 유형:**
```typescript
enum ThreatType {
  VESSEL_UNDERWAY = "vessel_underway",        // 항행 중인 선박
  VESSEL_STATIONARY = "vessel_stationary",    // 정지 선박
  SMALL_CRAFT = "small_craft",                // 소형 선박
  FISHING_VESSEL = "fishing_vessel",          // 어선
  BUOY = "buoy",                              // 부표
  DEBRIS = "debris",                          // 잔해
  LANDMASS = "landmass",                      // 육지
  SHALLOW_WATER = "shallow_water",            // 천해
  RESTRICTED_AREA = "restricted_area"         // 제한 구역
}

interface Threat {
  id: string;
  type: ThreatType;
  position: GeoPosition;
  velocity?: Velocity;  // 정지 물체의 경우 null

  // 감지 소스
  sources: {
    radar?: RadarTrack;
    lidar?: LidarObject;
    ais?: AISTarget;
    chart?: ChartFeature;
  };

  // 위협 특성
  characteristics: {
    size: { length: number; width: number };
    maneuverability: "high" | "medium" | "low" | "stationary";
    predictability: number;  // 0-1 (행동의 예측 가능성)

  };

  // 시간 데이터

  lastUpdated: Date;
```

**다단계 감지:**
```typescript
class ThreatDetectionSystem {
  private readonly DETECTION_RANGES = {
    longRange: 20,    // 해리 - 레이더 감시
    mediumRange: 5,   // 해리 - 세부 추적
    shortRange: 1,    // 해리 - 즉각적인 위협
    closeRange: 0.25  // 해리 - 충돌 임박
  };

  detectThreats(environment: FusedEnvironmentModel): Threat[] {
    const threats: Threat[] = [];

    // 1. 장거리 감시 (레이더)
    const longRangeContacts = environment.vessels.filter(
      v => v.distance <= this.DETECTION_RANGES.longRange
    );

    for (const contact of longRangeContacts) {
      const threat = this.createThreat(contact);
      threats.push(threat);
    }

    // 2. 해도 기반 위험
    const chartHazards = this.detectChartHazards(
      environment.ownShip.position,
      environment.ownShip.course
    );
    threats.push(...chartHazards);

    // 3. 동적 장애물 (라이다/비전)
    const dynamicObstacles = this.detectDynamicObstacles(
      environment.obstacles,
      environment.ownShip
    );
    threats.push(...dynamicObstacles);

    return threats;
  }
}
```

### 위험 평가

**충돌 위험 계산:**
```typescript
interface RiskAssessment {
  threat: Threat;
  risk: {
    cpa: CPACalculation;
    timeToClosestApproach: number;  // 분
    collisionProbability: number;   // 0-1
    severity: number;                // 0-1 (충돌 발생 시 영향 심각도)
    overallRisk: number;             // 확률 × 심각도
  };

  colregs: {
    applicable: boolean;
    encounterType: "head_on" | "crossing" | "overtaking" | "none";
    ownShipRole: "give_way" | "stand_on" | "both_give_way" | "unclear";
  };

  actionRequired: {
    urgency: "routine" | "caution" | "urgent" | "emergency";
    recommendedAction: string;
    timeToAction: number;  // 행동을 시작해야 할 때까지 초
  };
}

class CollisionRiskAssessor {
  private readonly RISK_THRESHOLDS = {
    safe: 0.1,
    caution: 0.3,
    warning: 0.6,
    critical: 0.8
  };

  assessRisk(threat: Threat, ownShip: ShipState): RiskAssessment {
    // 1. CPA 계산
    const cpa = this.calculateCPA(threat, ownShip);

    // 2. 충돌 확률 평가
    const collisionProb = this.calculateCollisionProbability(threat, ownShip, cpa);

    // 3. 심각도 추정
    const severity = this.estimateSeverity(threat, ownShip);

    // 4. 전체 위험
    const overallRisk = collisionProb * severity;

    // 5. COLREGs 분석
    const colregs = this.analyzeColregs(threat, ownShip);

    // 6. 필요한 행동 결정
    const actionRequired = this.determineActionRequired(
      overallRisk,
      cpa,
      colregs
    );

    return {
      threat,
      risk: {
        cpa,
        timeToClosestApproach: cpa.time,
        collisionProbability: collisionProb,
        severity,
        overallRisk
      },
      colregs,
      actionRequired
    };
  }

  private calculateCollisionProbability(
    threat: Threat,
    ownShip: ShipState,
    cpa: CPACalculation
  ): number {
    let probability = 0;

    // CPA 거리에서 기본 확률
    if (cpa.distance < 0.1) {
      probability = 0.9;  // 매우 높은 위험
    } else if (cpa.distance < 0.5) {
      probability = 0.7;
    } else if (cpa.distance < 1.0) {
      probability = 0.4;
    } else if (cpa.distance < 2.0) {
      probability = 0.2;
    } else {
      probability = 0.05;
    }

    // 목표물 예측 가능성에 따라 조정
    const unpredictability = 1 - threat.characteristics.predictability;
    probability += unpredictability * 0.2;

    // 시정 조건에 따라 조정
    const visibility = this.getVisibility();
    if (visibility < 2) {  // 제한된 시정
      probability *= 1.5;
    }

    // 목표물 기동성에 따라 조정 (어선은 예측 가능성이 낮음)
    if (threat.type === ThreatType.FISHING_VESSEL) {
      probability *= 1.3;
    }

    // CPA까지의 시간에 따라 조정 (시간이 많을수록 확률 낮음)
    if (cpa.time < 5) {
      probability *= 1.2;
    } else if (cpa.time > 30) {
      probability *= 0.8;
    }

    return Math.min(probability, 1.0);
  }
}
```

### 충돌 회피 의사결정

**의사결정 트리:**
```typescript
class CollisionAvoidanceDecisionMaker {
  decideAction(assessment: RiskAssessment, ownShip: ShipState): AvoidanceAction {
    const { risk, colregs, actionRequired } = assessment;

    // 비상: COLREGs에 관계없이 즉각적인 행동
    if (actionRequired.urgency === "emergency") {
      return this.emergencyAvoidance(assessment, ownShip);
    }

    // 적용 가능한 경우 COLREGs 적용
    if (colregs.applicable) {
      return this.colregsAvoidance(assessment, ownShip);
    }

    // 일반 좋은 선원정신
    if (actionRequired.urgency === "urgent" || actionRequired.urgency === "caution") {
      return this.prudentAvoidance(assessment, ownShip);
    }

    // 상황 모니터링
    return {
      type: "monitor",
      reason: "허용 가능한 한도 내 위험"
    };
  }

  private colregsAvoidance(
    assessment: RiskAssessment,
    ownShip: ShipState
  ): AvoidanceAction {
    const { colregs, threat } = assessment;

    if (colregs.ownShipRole === "give_way") {
      // 조기에 상당한 행동을 취해야 함

      switch (colregs.encounterType) {
        case "head_on":
          // 규칙 14: 양측 모두 우현으로 침로 변경
          return {
            type: "alter_course",
            direction: "starboard",
            amount: 30,  // 각도
            timing: "immediate",
            reason: "COLREGs 규칙 14 - 정면 조우"
          };

        case "crossing":
          // 규칙 15: 선미를 통과하도록 침로 변경
          const bearing = this.calculateBearing(ownShip, threat);

          if (bearing < 180) {
            // 목표물이 전방, 우현으로 침로 변경하여 선미 통과
            return {
              type: "alter_course",
              direction: "starboard",
              amount: 45,
              timing: "immediate",
              reason: "COLREGs 규칙 15 - 횡단, 피항"
            };
          } else {
            // 목표물이 후방, 속도 감소
            return {
              type: "reduce_speed",
              amount: 5,  // 노트
              timing: "immediate",
              reason: "COLREGs 규칙 15 - 횡단, 피항"
            };
          }

        case "overtaking":
          // 규칙 13: 피항, 어느 쪽으로든 통과
          const clearSide = this.determineClearSide(threat, ownShip);

          return {
            type: "alter_course",
            direction: clearSide,
            amount: 30,
            timing: "immediate",
            reason: "COLREGs 규칙 13 - 추월"
          };

        default:
          return this.prudentAvoidance(assessment, ownShip);
      }

    } else if (colregs.ownShipRole === "stand_on") {
      // 침로와 속도 유지, 하지만 행동 준비

      const cpa = assessment.risk.cpa;

      if (cpa.distance < 0.5 && cpa.time < 5) {
        // 충돌이 임박하면 유지선도 행동 가능
        return {
          type: "alter_course",
          direction: "starboard",  // 유지선은 절대 좌현으로
          amount: 45,
          timing: "immediate",
          reason: "COLREGs 규칙 17 - 유지선 충돌 회피 행동"
        };
      }

      return {
        type: "maintain_course",
        reason: "COLREGs - 유지선"
      };
    }

    return this.prudentAvoidance(assessment, ownShip);
  }
}
```
