def generate_train_command(config: dict) -> str:
    """설정에서 accelerate launch 명령어 생성"""
    script = "train_text_to_image_lora_sdxl.py"  # diffusers 예제 스크립트
    
    args = [
        f"accelerate launch --mixed_precision=fp16 {script}",
        f"--pretrained_model_name_or_path={config['pretrained_model_name_or_path']}",
        f"--dataset_name={config['dataset_name']}",
        f"--caption_column={config['caption_column']}",
        f"--resolution={config['resolution']}",
        f"--rank={config['rank']}",
        f"--learning_rate={config['learning_rate']}",
        f"--lr_scheduler={config['lr_scheduler']}",
        f"--lr_warmup_steps={config['lr_warmup_steps']}",
        f"--max_train_steps={config['max_train_steps']}",
        f"--train_batch_size={config['train_batch_size']}",
        f"--gradient_accumulation_steps={config['gradient_accumulation_steps']}",
        f"--gradient_checkpointing",
        f"--enable_xformers_memory_efficient_attention",
        f"--output_dir={config['output_dir']}",
        f"--checkpointing_steps={config['checkpointing_steps']}",
        f"--validation_prompt=\"{config['validation_prompt']}\"",
        f"--report_to={config['report_to']}",
    ]
    
    return " \\\n    ".join(args)

def validate_trained_lora(
    base_model_id: str,
    lora_dir: str,
    test_prompts: list,
    output_dir: str = "./validation_results",
):
    """
    학습 완료된 LoRA로 이미지 생성 + CLIP Score 평가
    """
    import torch
    from diffusers import DiffusionPipeline
    from peft import PeftModel
    
    os.makedirs(output_dir, exist_ok=True)
    
    print("검증용 파이프라인 로드...")
    pipe = DiffusionPipeline.from_pretrained(
        base_model_id,
        torch_dtype=torch.float16,
    ).to("cuda")
    
    pipe.load_lora_weights(lora_dir)
    pipe.set_progress_bar_config(disable=True)
    
    print("검증 이미지 생성:")
    results = []
    for i, prompt in enumerate(test_prompts):
        result = pipe(
            prompt=prompt,
            num_inference_steps=20,
            guidance_scale=7.5,
        ).images[0]
        
        out_path = os.path.join(output_dir, f"val_{i:02d}.png")
        result.save(out_path)
        print(f"  {i+1}. {prompt[:50]}... → {out_path}")
        results.append(out_path)
    
    print(f"\n검증 완료: {len(results)}개 이미지 → {output_dir}")
    return results

## 함정 카드 — 자주 실수하는 5가지

1. **"처음부터 완벽한 데이터셋을 만들려 한다"** → 100장으로 시작. 결과를 보고 문제점을 찾아 데이터를 개선하는 것이 훨씬 효율적이다. 나는 처음에 1,000장 데이터를 준비하느라 3일을 썼지만, 나중에 100장의 고품질 이미지가 훨씬 나은 결과를 냈다.

2. **"학습이 끝날 때까지 기다린다"** → 100 스텝 지점에서 반드시 검증 이미지를 확인. 방향이 잘못됐다면 200 스텝에서 알아채야지, 1,000 스텝 후에 알면 $1.5를 날린다. wandb 실시간 모니터링을 항상 켜놓는다.

3. **"모델 업로드 전 README를 소홀히 한다"** → Hugging Face Hub에 올린 모델의 가치는 README에 달려 있다. 사용 예시 코드, 학습 데이터 출처, 라이선스를 명확히 써야 다운로드 수가 올라간다. 중간심사 심사위원이 Hugging Face 링크를 클릭했을 때 첫인상이 결정된다.

4. **"GPU 인스턴스를 종료하지 않고 잊는다"** → 나는 학습 완료 후 인스턴스를 종료하지 않아 $47을 날린 경험이 있다. 학습 완료 → 파일 다운로드 → 즉시 인스턴스 종료 순서를 철칙으로. 학습 스크립트 마지막에 알림 메시지 추가 또는 `--notify_email` 옵션 활용.

5. **"첫 결과가 나쁘면 포기한다"** → 처음 100 스텝 학습한 LoRA는 항상 어색하다. 학습 곡선이 수렴하고 있으면 계속 진행. 300~500 스텝부터 스타일이 잡히기 시작. CLIP Score가 조금씩 올라가고 있다면 올바른 방향이다.

## 연습문제

**문제 1.** 조선 전통 문양 LoRA 학습에 적합한 캡션 스타일을 설계하라. 영어와 한국어 혼합이 좋은가, 한국어만이 좋은가? SDXL의 CLIP 인코더 특성을 고려하여 설명하라.

> **해설:** **영한 혼합 권장**. SDXL의 CLIP 인코더는 영어 텍스트에 더 강하게 반응한다. 한국어만 쓰면 CLIP이 제대로 처리 못 하는 경우가 있다. 권장 포맷: `"조선 전통 연꽃 문양, joseon traditional lotus pattern, traditional Korean painting, 청자 청색, celadon blue, high quality"` — 한국어 키워드 + 영어 번역 병기. T5 인코더(SD3.5)는 한국어를 더 잘 처리하므로, 향후 SD3.5로 전환 시 한국어 위주 캡션 재실험 권장.

**문제 2.** 1,000 스텝 학습 후 CLIP Score가 28로 낮게 나왔다. 가능한 원인 3가지와 각각의 해결 방법을 제시하라.

> **해설:** ① **캡션-이미지 불일치**: 캡션이 이미지와 맞지 않음. 해결: 각 이미지를 Claude Vision으로 자동 캡션 생성하여 일관성 확보. ② **과적합(Overfitting)**: 100장 데이터로 rank 64 이상 LoRA 학습. 해결: rank를 8~16으로 낮추거나 데이터를 200장 이상으로 증가. ③ **학습률 너무 높음**: 기반 모델 표현이 LoRA로 덮어씌워짐. 해결: lr을 5e-5로 낮추고 warmup steps 100으로 설정하여 재학습.
