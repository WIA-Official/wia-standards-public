Chapter 5: Supply Chain Integration

Agricultural supply chains connect farms to consumers through complex networks of aggregators, processors, distributors, and retailers. This chapter covers traceability systems, quality and certification data, market connectivity, and the logistics integration that enable transparent, efficient food supply chains. Data exchange standards are essential for farm-to-fork visibility.

5.1 Traceability Fundamentals

Why Traceability Matters

  • Food Safety: Rapid recall response, contamination tracking
  • Consumer Trust: Origin verification, production practices
  • Regulatory Compliance: FSMA, EU regulations, export requirements
  • Premium Markets: Organic, non-GMO, sustainable certifications
  • Supply Chain Efficiency: Inventory management, waste reduction
// Traceability Event Model (based on GS1 EPCIS)
interface TraceabilityEvent {
  eventId: string;
  eventType: 'ObjectEvent' | 'AggregationEvent' |
             'TransactionEvent' | 'TransformationEvent';
  eventTime: ISO8601DateTime;
  recordTime: ISO8601DateTime;
  eventTimeZoneOffset: string;

  // What
  epcList?: string[];            // Electronic Product Codes
  quantityList?: {
    epcClass: string;
    quantity: number;
    uom: string;
  }[];

  // When
  eventTime: ISO8601DateTime;

  // Where
  readPoint: string;             // URN of location
  bizLocation?: string;          // Business location

  // Why
  bizStep: string;               // e.g., 'shipping', 'receiving', 'storing'
  disposition?: string;          // e.g., 'in_transit', 'in_progress'
  bizTransactionList?: {
    type: string;                // 'po', 'inv', 'bol'
    bizTransaction: string;
  }[];

  // Extensions
  sourceList?: Source[];
  destinationList?: Destination[];
  sensorElementList?: SensorElement[];  // Temperature, humidity
}

5.2 Product Identification

// GS1 Identification Standards for Agriculture
interface ProductIdentification {
  // Global Trade Item Number
  gtin: {
    format: 'GTIN-14' | 'GTIN-13' | 'GTIN-12';
    value: string;
    description: 'Unique product identifier';
  };

  // Global Location Number
  gln: {
    format: '13-digit';
    value: string;
    description: 'Identifies parties and locations';
  };

  // Serial Shipping Container Code
  sscc: {
    format: '18-digit';
    value: string;
    description: 'Unique logistics unit identifier';
  };

  // Lot/Batch Number
  lot: {
    ai: '10';                    // GS1 Application Identifier
    value: string;
    description: 'Production batch identifier';
  };

  // Harvest Date
  harvestDate: {
    ai: '13';
    format: 'YYMMDD';
    value: string;
  };

  // Best Before Date
  bestBefore: {
    ai: '15';
    format: 'YYMMDD';
    value: string;
  };

  // Country of Origin
  countryOfOrigin: {
    ai: '422';
    format: 'ISO 3166';
    value: string;
  };
}

// Certification Record Schema
interface CertificationRecord {
  id: string;
  certificationBody: {
    name: string;
    accreditation: string;
    gln?: string;
  };

  // Certificate Details
  certificate: {
    number: string;
    type: string;
    scope: string[];             // Products/processes covered
    standard: string;            // e.g., 'USDA NOP', 'GlobalG.A.P.'
    version: string;
  };

  // Validity
  validity: {
    issueDate: ISO8601Date;
    expiryDate: ISO8601Date;
    status: 'active' | 'suspended' | 'withdrawn' | 'expired';
  };

  // Certified Entity
  certifiedEntity: {
    name: string;
    gln: string;
    locations: string[];
  };

  // Verification
  verification: {
    url?: string;                // Online verification portal
    qrCode?: string;
    blockchain?: {
      network: string;
      transactionId: string;
    };
  };
}

5.4 Quality Data Exchange

// Quality Assessment Record
interface QualityRecord {
  id: string;
  lotId: string;
  productGtin: string;

  // Sample Information
  sample: {
    id: string;
    type: 'grab' | 'composite' | 'probe';
    size: { value: number; unit: string };
    collectedAt: ISO8601DateTime;
    collectedBy: string;
    location: string;
  };

  // Test Results
  tests: {
    parameter: string;
    method: string;              // Reference method
    value: number;
    unit: string;
    specification: {
      min?: number;
      max?: number;
      target?: number;
    };
    result: 'pass' | 'fail' | 'marginal';
  }[];

  // Grading
  grade?: {
    system: string;              // e.g., 'USDA', 'EU'
    value: string;               // e.g., 'US No. 1', 'Grade A'
    factors: {
      factor: string;
      score: number;
    }[];
  };

  // Laboratory
  laboratory?: {
    name: string;
    accreditation: string;
    reportNumber: string;
  };

  // Overall Assessment
  disposition: 'accepted' | 'rejected' | 'hold' | 'rework';
  notes?: string;
}

5.5 Logistics Integration

Transportation Data Exchange

Agricultural logistics involves specialized requirements including temperature control, regulatory compliance for perishables, and coordination across multiple transport modes. Standard data formats enable seamless handoffs between carriers and visibility across the supply chain.

// Shipment Data Model
interface Shipment {
  id: string;
  type: 'FTL' | 'LTL' | 'rail' | 'container' | 'bulk_vessel';

  // Parties
  shipper: PartyInfo;
  consignee: PartyInfo;
  carrier: PartyInfo;
  broker?: PartyInfo;

  // Cargo
  cargo: {
    description: string;
    commodityCode: string;       // Harmonized System code
    items: {
      sscc?: string;
      gtin: string;
      lot: string;
      quantity: number;
      unit: string;
      weight: { value: number; unit: 'kg' };
    }[];
    totalWeight: { value: number; unit: 'kg' };
    totalVolume?: { value: number; unit: 'm3' };
  };

  // Schedule
  schedule: {
    pickupRequested: ISO8601DateTime;
    pickupActual?: ISO8601DateTime;
    deliveryRequested: ISO8601DateTime;
    deliveryActual?: ISO8601DateTime;
    transitTime: number;         // Hours
  };

  // Conditions
  conditions?: {
    temperature: { min: number; max: number; unit: 'C' };
    humidity?: { min: number; max: number };
    specialHandling?: string[];
  };

  // Documents
  documents: {
    type: 'BOL' | 'POD' | 'invoice' | 'phyto' | 'customs';
    number: string;
    url?: string;
  }[];

  // Tracking
  tracking: {
    currentStatus: string;
    currentLocation?: GeoJSON.Point;
    events: TrackingEvent[];
  };
}

// Tracking Event
interface TrackingEvent {
  timestamp: ISO8601DateTime;
  status: string;
  location: string;
  details?: string;
  sensorData?: {
    temperature?: number;
    humidity?: number;
    shock?: boolean;
  };
}

// Market Data Integration
interface MarketIntegration {

      '850': 'Purchase Order',

      '810': 'Invoice',

5.7 Recall Management

// Recall Event Structure
interface RecallEvent {
  id: string;
  classification: 'Class_I' | 'Class_II' | 'Class_III';
  status: 'initiated' | 'ongoing' | 'completed' | 'terminated';

  // Product Information
  product: {
    name: string;
    gtin: string;
    lots: string[];
    productionDates: { from: ISO8601Date; to: ISO8601Date };
    quantity: { value: number; unit: string };
  };

  // Reason
  reason: {
    hazard: string;
    description: string;
    discoverySource: string;
    discoveryDate: ISO8601Date;
  };

  // Distribution
  distribution: {
    channels: string[];
    geography: string[];
    retailers: string[];
  };

  // Actions
  actions: {
    consumerNotice: string;
    retailerNotice: string;
    returnInstructions: string;
    disposalMethod: string;
  };

  // Traceability Query
  traceQuery: {
    upstream: TraceabilityEvent[];   // Back to farm
    downstream: TraceabilityEvent[]; // To consumers
    affectedLots: string[];
    affectedLocations: string[];
  };

  // Timeline
  timeline: {
    initiated: ISO8601DateTime;
    notified: ISO8601DateTime;
    completed?: ISO8601DateTime;
  };
}

5.8 Blockchain Integration

Distributed Ledger for Traceability

Blockchain technology provides immutable, transparent records for agricultural supply chains. While not required, blockchain integration offers enhanced trust and auditability for high-value or regulated supply chains.

// Blockchain Integration Model
interface BlockchainTraceability {
  network: 'Hyperledger' | 'Ethereum' | 'IBM_Food_Trust' | 'private';

  // On-chain Data (minimal, immutable)
  onChain: {
    transactionId: string;
    timestamp: ISO8601DateTime;
    eventHash: string;           // Hash of full event data
    participants: string[];      // Public keys
    signatures: string[];
  };

  // Off-chain Data (detailed, linked)
  offChain: {
    storageLocation: 'IPFS' | 'cloud' | 'enterprise';
    eventData: TraceabilityEvent;
    documents: string[];         // Hashes of supporting docs
    sensorData: string[];        // Hashes of IoT data
  };

  // Smart Contract
  smartContract: {
    address: string;
    functions: [
      'recordEvent(bytes32 eventHash)',
      'verifyChain(bytes32[] eventHashes)',
      'transferOwnership(address newOwner)',
      'certify(bytes32 certHash)'
    ];
  };
}
