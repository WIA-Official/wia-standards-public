4장: 농장 관리 데이터

농장 관리 정보 시스템(FMIS)은 현대 농업 기업의 운영 중추입니다. 이 장에서는 농장, 필드, 작물을 나타내는 핵심 데이터 구조와 계획부터 수확까지의 농업 활동을 추적하는 운영 기록을 다룹니다. 이러한 데이터 모델은 포괄적인 농장 기록 관리 및 의사결정 지원을 가능하게 합니다.

4.1 농장 엔티티 모델

// 핵심 농장 데이터 구조
interface Farm {
  // 식별
  id: string;                    // UUID
  externalIds?: {
    system: string;              // 예: 'mafra', 'regional_registry'
    id: string;
  }[];

  // 기본 정보
  name: string;
  legalName?: string;
  businessType: 'sole_proprietor' | 'partnership' | 'corporation' | 'cooperative';

  // 위치
  address: {
    street?: string;
    city: string;
    province: string;
    postalCode: string;
    country: string;
  };
  headquarters?: GeoJSON.Point;  // 농장 본부 위치

  // 연락처
  contacts: {
    type: 'primary' | 'manager' | 'agronomist';
    name: string;
    email?: string;
    phone?: string;
  }[];

  // 집계
  totalArea: { value: number; unit: 'hectare' | 'pyeong' };
  fieldCount: number;

  // 메타데이터
  created: ISO8601DateTime;
  modified: ISO8601DateTime;
  status: 'active' | 'inactive';
}

4.2 필드 데이터 모델

// 농업 필드 표현
interface Field {
  id: string;
  farmId: string;
  name: string;
  commonName?: string;           // 지역/역사적 이름

  // 공간 데이터
  boundary: GeoJSON.Polygon;
  area: { value: number; unit: 'hectare' | 'pyeong' };
  perimeter: { value: number; unit: 'meter' };
  centroid: GeoJSON.Point;

  // 토양 정보
  soilType?: string;             // 토양 분류
  soilSeries?: string;
  avgSlope?: number;             // 퍼센트
  drainageClass?: string;

  // 인프라
  irrigationType?: 'none' | 'center_pivot' | 'drip' | 'furrow' | 'flood';
  irrigationSource?: string;
  drainageStatus?: 'none' | 'partial' | 'full';

  // 현재 시즌
  currentCrop?: {
    cropCode: string;            // WIA-HCC 코드
    variety: string;
    plantingDate: ISO8601Date;
    expectedHarvest: ISO8601Date;
  };

  // 이력
  cropHistory: CropYear[];

  // 관리 구역
  zones?: ManagementZone[];

  // 메타데이터
  created: ISO8601DateTime;
  modified: ISO8601DateTime;
  status: 'active' | 'fallow' | 'retired';
}

// 가변율 적용을 위한 관리 구역
interface ManagementZone {
  id: string;
  fieldId: string;
  name: string;
  geometry: GeoJSON.Polygon;
  area: { value: number; unit: 'hectare' };
  attributes: {
    yieldPotential: 'high' | 'medium' | 'low';
    soilOM?: number;             // 유기물 %
    pH?: number;
    ec?: number;                 // 전기전도도
  };
}

4.4 파종 기록

// 파종 작업 데이터
interface PlantingRecord {
  id: string;
  fieldId: string;
  operationType: 'planting';

  // 시간
  startTime: ISO8601DateTime;
  endTime: ISO8601DateTime;
  duration: number;              // 분

  // 작물 정보
  crop: {
    code: string;                // WIA-HCC
    name: string;
    variety: string;
    seedLot?: string;
  };

  // 파종 매개변수
  parameters: {
    seedingRate: {
      target: number;
      actual: number;
      unit: 'seeds/ha' | 'kg/ha';
    };
    rowSpacing: { value: number; unit: 'cm' };
    plantingDepth: { value: number; unit: 'cm' };
  };

  // 커버리지
  coverage: {
    planned: GeoJSON.Polygon;
    actual: GeoJSON.MultiPolygon;
    areaPlanted: { value: number; unit: 'hectare' };
    completionPercent: number;
  };

  // 장비
  equipment: {
    tractorId: string;
    implementId: string;
    implementType: 'planter';
    rowCount: number;
    width: { value: number; unit: 'meter' };
  };

  // 조건
  conditions?: {
    soilTemp: { value: number; unit: 'C' };
    soilMoisture: string;        // 'optimal', 'dry', 'wet'
    airTemp: { value: number; unit: 'C' };
  };
}

4.5 적용 기록

// 화학/비료 적용 기록
interface ApplicationRecord {
  id: string;
  fieldId: string;
  operationType: 'application';

  // 시간
  startTime: ISO8601DateTime;
  endTime: ISO8601DateTime;

  // 적용 유형
  applicationType: 'fertilizer' | 'pesticide' | 'herbicide' |
                   'fungicide' | 'seed_treatment' | 'growth_regulator';

  // 적용된 제품
  products: {
    productId: string;
    name: string;
    registrationNumber?: string; // 농약 등록번호
    activeIngredients: {
      name: string;
      concentration: number;
      unit: string;
    }[];
    rate: {
      target: number;
      actual: number;
      unit: 'L/ha' | 'kg/ha';
    };
    totalApplied: { value: number; unit: string };
  }[];

  // 적용 방법
  method: {
    type: 'broadcast' | 'band' | 'in-furrow' | 'foliar' | 'injection';
    equipment: 'sprayer' | 'spreader' | 'fertigation' | 'aerial';
    carrierVolume?: { value: number; unit: 'L/ha' };
    nozzleType?: string;
    pressure?: { value: number; unit: 'bar' };
  };

  // 가변율
  prescription?: {
    id: string;
    type: 'variable_rate' | 'uniform';
    source: 'soil_sample' | 'yield_map' | 'satellite' | 'manual';
  };

  // 날씨 조건
  weatherConditions: {
    temperature: { value: number; unit: 'C' };
    windSpeed: { value: number; unit: 'km/h' };
    windDirection: number;       // 도
    humidity: number;            // 퍼센트
    precipitation: boolean;
  };

  // 규정 준수
  applicatorId?: string;
  licenseNumber?: string;
  rei?: number;                  // 재입장 간격 (시간)
  phi?: number;                  // 수확 전 간격 (일)
}

4.6 수확 기록

수확량 데이터의 중요성

수확 수확량 데이터는 가장 가치 있는 농장 데이터 중 하나로, 필드 성능 분석, 품종 비교, 투입재 최적화를 가능하게 합니다. 현대 콤바인 수확량 모니터는 정밀 농업 분석의 기초를 형성하는 고해상도 공간 수확량 지도를 생성합니다.

// 수확 작업 기록
interface HarvestRecord {
  id: string;
  fieldId: string;
  operationType: 'harvest';

  // 시간
  startTime: ISO8601DateTime;
  endTime: ISO8601DateTime;

  // 수확된 작물
  crop: {
    code: string;
    name: string;
    variety: string;
    plantingRecordId: string;    // 파종과 연결
  };

  // 수확량 요약
  yield: {
    total: { value: number; unit: 'kg' | 't' };
    perArea: { value: number; unit: 't/ha' | 'kg/pyeong' };
    dryMatter?: { value: number; unit: 't/ha' };
  };

  // 품질
  quality: {
    moisture: number;            // 수확 시 수분 퍼센트
    testWeight?: { value: number; unit: 'kg/hL' };
    protein?: number;            // 퍼센트
    damage?: number;             // 퍼센트
    foreignMaterial?: number;    // 퍼센트
  };

  // 공간 수확량 데이터
  yieldMap?: {
    format: 'GeoJSON' | 'shapefile' | 'ISOXML';
    resolution: { value: number; unit: 'meter' };
    pointCount: number;
    coverage: GeoJSON.FeatureCollection;
  };

  // 장비
  equipment: {
    combineId: string;
    headerType: string;
    headerWidth: { value: number; unit: 'meter' };
    yieldMonitor: {
      model: string;
      calibrationDate: ISO8601Date;
    };
  };

  // 목적지
  destination: {
    type: 'on_farm' | 'rpc' | 'processor' | 'direct_sale';
    locationId?: string;
    loads: {
      loadId: string;
      weight: { value: number; unit: 'kg' };
      moisture: number;
      timestamp: ISO8601DateTime;
    }[];
  };
}

4.7 필드 경계

// 필드 경계 관리
interface FieldBoundary {
  fieldId: string;
  version: number;
  effectiveDate: ISO8601Date;

  // 지오메트리
  boundary: {
    type: 'Polygon';
    coordinates: number[][][];   // [[[lng, lat], ...]]
    crs: 'EPSG:4326';           // WGS84
  };

  // 파생 값
  area: {
    gross: { value: number; unit: 'hectare' };
    net: { value: number; unit: 'hectare' };  // 비경작지 제외
  };

  // 제외 영역 (경계 내에 있지만 경작하지 않음)
  exclusions?: {
    type: 'waterway' | 'wetland' | 'structure' | 'buffer' | 'other';
    geometry: GeoJSON.Polygon;
    area: { value: number; unit: 'hectare' };
  }[];

  // 출처
  source: {
    method: 'gps_survey' | 'satellite' | 'digitized' | 'cadastral' | 'imported';
    accuracy?: { value: number; unit: 'meter' };
    collectedBy?: string;
    collectedDate: ISO8601Date;
  };

  // 법적 참조
  legalDescription?: {
    type: 'cadastral' | 'parcel_id';
    description: string;
  };
}

4.8 작물 연도 구조

// 작물 연도 기록
interface CropYear {
  fieldId: string;
  year: number;

  // 작물 정보
  crop: {
    primary: CropInfo;
    cover?: CropInfo;
    double?: CropInfo;           // 이모작
  };

  // 시즌 날짜
  dates: {
    prePlant?: ISO8601Date;
    planting: ISO8601Date;
    emergence?: ISO8601Date;
    harvest: ISO8601Date;
    postHarvest?: ISO8601Date;
  };

  // 작업 요약
  operations: {
    plantings: string[];         // 기록 ID
    applications: string[];
    irrigations?: string[];
    harvests: string[];
  };

  // 성과
  performance: {
    yieldActual: { value: number; unit: 't/ha' };
    yieldExpected?: { value: number; unit: 't/ha' };
    inputCosts?: { value: number; currency: string; unit: '/ha' };
    revenue?: { value: number; currency: string; unit: '/ha' };
    margin?: { value: number; currency: string; unit: '/ha' };
  };

  // 메모
  notes?: string;
  status: 'planned' | 'in_progress' | 'completed';
}
