7장: 보안 및 프라이버시

농업 데이터에는 농업 운영, 토지 자산, 비즈니스 관행에 대한 민감한 정보가 포함되어 있어 강력한 보호가 필요합니다. 이 장에서는 데이터 소유권 원칙, 동의 관리 프레임워크, 접근 제어 메커니즘, 프라이버시 보존 기술, 규정 준수 요구사항을 다룹니다. 강력한 보안 및 프라이버시 제어를 통한 신뢰 구축은 농업 데이터 교환의 광범위한 채택에 필수적입니다.

7.1 데이터 소유권 원칙

핵심 소유권 원칙

  • 농부 중심: 농부가 자신의 운영에서 생성된 데이터를 소유
  • 투명성: 데이터 수집 및 사용에 대한 명확한 이해
  • 통제: 접근 권한 부여, 철회, 감사 능력
  • 이동성: 시스템 간 데이터 내보내기 및 이동 권리
  • 가치 공유: 데이터가 가치를 창출할 때 공정한 보상
  • 보안: 무단 접근 및 오용으로부터 보호
// 데이터 소유권 모델
interface DataOwnership {

7.3 접근 제어

모델 설명 용도 예시
RBAC 역할 기반 접근 제어 조직 계층 농장 관리자, 농학자, 열람자
ABAC 속성 기반 접근 제어 세분화된 정책 필드, 시간, 목적에 따른 접근
ReBAC 관계 기반 접근 제어 데이터 공유 관계 자문가가 자문 농장 접근 가능
PBAC 목적 기반 접근 제어 동의 시행 분석 vs. 마케팅 사용 
// 접근 제어 구현
interface AccessControl {
  // 역할
  roles: {
    farm_owner: {
      permissions: ['*'];        // 자신의 데이터에 전체 접근
      scope: 'owned_farms';
    };
    farm_manager: {
      permissions: ['read', 'write', 'share'];
      scope: 'managed_farms';
    };
    agronomist: {
      permissions: ['read', 'write:recommendations'];
      scope: 'advised_farms';
      expiry: 'contract_end';
    };
    equipment_provider: {
      permissions: ['read:telematics', 'write:diagnostics'];
      scope: 'contracted_equipment';
    };
    researcher: {
      permissions: ['read:anonymized'];
      scope: 'consented_data';
      purpose: 'research_only';
    };
  };

  // 정책 언어 (OPA/Rego 스타일 예시)
  policies: {
    allowFieldAccess: `
      allow {
        input.user.role == "agronomist"
        input.resource.type == "field_data"
        input.resource.farmId in input.user.advisedFarms
        input.action == "read"
      }
    `;
  };

  // 접근 결정
  accessDecision: {
    request: {
      subject: { id: string; roles: string[]; attributes: Record };
      resource: { id: string; type: string; owner: string };
      action: 'read' | 'write' | 'delete' | 'share';
      context: { purpose: string; timestamp: ISO8601DateTime };
    };
    response: {
      decision: 'permit' | 'deny';
      obligations?: string[];    // 필수 조치 (예: 로그, 알림)
      advice?: string[];         // 권장사항
    };
  };
}

7.5 프라이버시 보존 기술

프라이버시를 보호하면서 데이터 사용 가능

농업 데이터는 종종 연구, 벤치마킹, 시장 인사이트를 위해 여러 농장에 걸쳐 집계되거나 분석되어야 합니다. 프라이버시 보존 기술은 개별 농장의 기밀성을 보호하면서 이러한 사용 사례를 가능하게 합니다.

// 프라이버시 보존 기술

  // 익명화

      masking: '값의 부분 삭제';
      pseudonymization: '식별자를 토큰으로 대체';

    kAnonymity: {
      description: '각 레코드가 k-1개의 다른 레코드와 구별 불가능';
      threshold: 5;
    };
    lDiversity: {
      description: '민감한 속성이 l개의 고유 값 보유';

  };
