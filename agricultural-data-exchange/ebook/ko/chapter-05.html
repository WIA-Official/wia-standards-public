5장: 공급망 통합

농업 공급망은 수집업자, 가공업체, 유통업체, 소매업체의 복잡한 네트워크를 통해 농장과 소비자를 연결합니다. 이 장에서는 추적성 시스템, 품질 및 인증 데이터, 시장 연결성, 투명하고 효율적인 식품 공급망을 가능하게 하는 물류 통합을 다룹니다. 데이터 교환 표준은 농장에서 식탁까지의 가시성을 위해 필수적입니다.

5.1 추적성 기초

추적성의 중요성

  • 식품 안전: 신속한 리콜 대응, 오염 추적
  • 소비자 신뢰: 원산지 확인, 생산 관행
  • 규제 준수: 식품 안전법, 수출 요건
  • 프리미엄 시장: 유기농, 비GMO, 지속가능성 인증
  • 공급망 효율성: 재고 관리, 폐기물 감소
// 추적성 이벤트 모델 (GS1 EPCIS 기반)
interface TraceabilityEvent {
  eventId: string;
  eventType: 'ObjectEvent' | 'AggregationEvent' |
             'TransactionEvent' | 'TransformationEvent';
  eventTime: ISO8601DateTime;
  recordTime: ISO8601DateTime;
  eventTimeZoneOffset: string;

  // 무엇을
  epcList?: string[];            // 전자제품 코드
  quantityList?: {
    epcClass: string;
    quantity: number;
    uom: string;
  }[];

  // 언제
  eventTime: ISO8601DateTime;

  // 어디서
  readPoint: string;             // 위치의 URN
  bizLocation?: string;          // 비즈니스 위치

  // 왜
  bizStep: string;               // 예: 'shipping', 'receiving', 'storing'
  disposition?: string;          // 예: 'in_transit', 'in_progress'
  bizTransactionList?: {
    type: string;                // 'po', 'inv', 'bol'
    bizTransaction: string;
  }[];

  // 확장
  sourceList?: Source[];
  destinationList?: Destination[];
  sensorElementList?: SensorElement[];  // 온도, 습도
}

const shippingEvent: TraceabilityEvent = {
  eventId: 'evt-2025-ship-001',
  eventType: 'ObjectEvent',
  eventTime: '2025-06-15T08:00:00+09:00',

  eventTimeZoneOffset: '+09:00',
  epcList: ['urn:epc:id:sscc:8801234.1234567890'],

5.3 인증 데이터

인증 범위 데이터 요구사항 검증
친환경 인증 생산 관행 투입재 기록, 토지 이력 연간 심사
GAP 인증 우수농산물 관리 포괄적 농장 기록 제3자 감사
HACCP 식품 안전 위해요소 관리 기록 정기 점검
공정무역 사회적/경제적 노동, 가격, 커뮤니티 인증기관
탄소중립 GHG 배출 범위 1-3 배출 데이터 LCA, 검증 
// 인증 기록 스키마
interface CertificationRecord {
  id: string;
  certificationBody: {
    name: string;
    accreditation: string;
    gln?: string;
  };

  // 인증서 세부정보
  certificate: {
    number: string;
    type: string;
    scope: string[];             // 적용 제품/프로세스
    standard: string;            // 예: '친환경', 'GAP'
    version: string;
  };

  // 유효성
  validity: {
    issueDate: ISO8601Date;
    expiryDate: ISO8601Date;
    status: 'active' | 'suspended' | 'withdrawn' | 'expired';
  };

  // 인증된 엔티티
  certifiedEntity: {
    name: string;
    gln: string;
    locations: string[];
  };

  // 검증
  verification: {
    url?: string;                // 온라인 검증 포털
    qrCode?: string;
    blockchain?: {
      network: string;
      transactionId: string;
    };
  };
}

5.4 품질 데이터 교환

// 품질 평가 기록
interface QualityRecord {
  id: string;

  // 샘플 정보
  sample: {
    id: string;

    collectedAt: ISO8601DateTime;
    collectedBy: string;
    location: string;
  };

  // 검사 결과
  tests: {
    parameter: string;
    method: string;              // 참조 방법
    value: number;
    unit: string;
    specification: {
      min?: number;
      max?: number;
      target?: number;
    };
    result: 'pass' | 'fail' | 'marginal';
  }[];

  // 등급
  grade?: {
    system: string;              // 예: '농관원', '한국'

      factor: string;
      score: number;

    name: string;
    accreditation: string;
    reportNumber: string;
  };

  // 전체 평가
  disposition: 'accepted' | 'rejected' | 'hold' | 'rework';

}

5.5 물류 통합

운송 데이터 교환

농업 물류는 온도 제어, 부패성 물품 규제 준수, 여러 운송 모드에 걸친 조정을 포함한 특수 요구사항을 포함합니다. 표준 데이터 형식은 운송업체 간의 원활한 인수인계와 공급망 전반의 가시성을 가능하게 합니다.

// 배송 데이터 모델
interface Shipment {
  id: string;
  type: 'FTL' | 'LTL' | 'rail' | 'container' | 'bulk_vessel';

  // 당사자
  shipper: PartyInfo;
  consignee: PartyInfo;
  carrier: PartyInfo;
  broker?: PartyInfo;

  // 화물
  cargo: {
    description: string;
    commodityCode: string;       // HS 코드
    items: {
      sscc?: string;
      gtin: string;
      lot: string;
      quantity: number;
      unit: string;
      weight: { value: number; unit: 'kg' };
    }[];
    totalWeight: { value: number; unit: 'kg' };
    totalVolume?: { value: number; unit: 'm3' };
  };

  // 일정
  schedule: {
    pickupRequested: ISO8601DateTime;
    pickupActual?: ISO8601DateTime;
    deliveryRequested: ISO8601DateTime;
    deliveryActual?: ISO8601DateTime;
    transitTime: number;         // 시간
  };

  // 조건
  conditions?: {
    temperature: { min: number; max: number; unit: 'C' };
    humidity?: { min: number; max: number };
    specialHandling?: string[];
  };

  // 서류
  documents: {
    type: 'BOL' | 'POD' | 'invoice' | 'phyto' | 'customs';
    number: string;
    url?: string;
  }[];

  // 추적
  tracking: {
    currentStatus: string;
    currentLocation?: GeoJSON.Point;
    events: TrackingEvent[];
  };
}

5.8 블록체인 통합

추적성을 위한 분산원장

블록체인 기술은 농업 공급망에 불변하고 투명한 기록을 제공합니다. 필수는 아니지만 블록체인 통합은 고부가가치 또는 규제 대상 공급망에 대해 향상된 신뢰와 감사 가능성을 제공합니다.

// 블록체인 통합 모델
interface BlockchainTraceability {
  network: 'Hyperledger' | 'Ethereum' | 'private';

  // 온체인 데이터 (최소, 불변)
  onChain: {
    transactionId: string;
    timestamp: ISO8601DateTime;
    eventHash: string;           // 전체 이벤트 데이터의 해시
    participants: string[];      // 공개키
    signatures: string[];
  };

  // 오프체인 데이터 (상세, 연결)
  offChain: {
    storageLocation: 'IPFS' | 'cloud' | 'enterprise';
    eventData: TraceabilityEvent;
    documents: string[];         // 지원 문서의 해시
    sensorData: string[];        // IoT 데이터의 해시
  };

  // 스마트 계약
  smartContract: {
    address: string;
    functions: [
      'recordEvent(bytes32 eventHash)',
      'verifyChain(bytes32[] eventHashes)',
      'transferOwnership(address newOwner)',
      'certify(bytes32 certHash)'
    ];
  };
}
