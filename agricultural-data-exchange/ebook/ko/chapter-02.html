2장: 데이터 표준 및 형식

표준화된 데이터 형식은 농업 데이터 상호운용성의 기초입니다. 이 장에서는 공간 정보, 작물 및 관측 데이터, 장비 텔레매틱스, 그리고 시스템 간 일관된 해석을 보장하는 의미론적 프레임워크를 위한 핵심 데이터 표준을 다룹니다. 이러한 표준은 다양한 농업 플랫폼 간의 의미 있는 데이터 교환을 가능하게 합니다.

// 작물 분류 시스템
interface CropCodeSystems {

  ICC: {
    example: '01111',
    structure: 'group.class.subclass',
    crops: {
      '01111': '밀',

    };
  };

  // 한국 농산물 코드
  KFDA: {
    example: '1101',
    structure: 'category.subcategory.item',
    region: 'Korea'
  };

  // GS1 제품 식별
  GS1: {
    example: '01234567890123',
    structure: 'GTIN-13/14',
    purpose: '공급망 식별'
  };

  // WIA 조화 작물 코드
  WIA_HCC: {
    structure: '{system}:{code}',

      'KFDA:1101',      // 한국 쌀
      'UNSPSC:10151500' // UN 밀
    ],
    mapping: '크로스워크 테이블 제공'
  };
}

2.4 관측 데이터 모델

// 농업 관측 스키마 (OGC Observations)
interface AgriculturalObservation {
  // 핵심 관측 속성
  id: string;
  type: 'Observation';
  phenomenonTime: ISO8601DateTime;
  resultTime: ISO8601DateTime;

  // 무엇을 관측했는가
  observedProperty: {
    id: string;
    name: string;
    definition: URI;    // 어휘에 대한 링크
    unit: string;
  };

  // 어디서 관측했는가
  featureOfInterest: {
    id: string;
    type: 'Field' | 'Zone' | 'Point' | 'Plant';
    geometry?: GeoJSON.Geometry;
  };

  // 관측 값
  result: {
    value: number | string | object;
    quality?: QualityCode;
  };

  // 어떻게 관측했는가
  procedure: {
    type: 'Sensor' | 'LabAnalysis' | 'VisualAssessment' | 'Model';
    sensor?: SensorInfo;
    method?: string;
  };
}

// 예시: 토양 수분 관측
const soilMoistureObs: AgriculturalObservation = {

2.5 장비 및 텔레매틱스

ISOBUS (ISO 11783)

  • 태스크 컨트롤러: 작업 지시 및 적용 데이터
  • 데이터 사전: 표준화된 매개변수 정의
  • 장치 설명: 작업기 기능
  • 공간 데이터: 처방 지도 및 커버리지
// 기계 텔레매틱스 데이터
interface MachineTelematics {
  device: {
    id: string;
    type: 'tractor' | 'combine' | 'sprayer' | 'planter';
    manufacturer: string;
    model: string;
    serialNumber: string;
  };

  position: {
    timestamp: ISO8601DateTime;
    latitude: number;
    longitude: number;
    altitude?: number;
    heading: number;        // 도
    speed: number;          // km/h
    accuracy: number;       // 미터
  };

  engine: {
    rpm: number;
    fuelRate: number;       // L/h
    fuelLevel: number;      // 퍼센트
    coolantTemp: number;    // 섭씨
    oilPressure: number;    // kPa
    engineHours: number;
  };

  implement?: {
    type: string;
    width: number;          // 미터
    status: 'raised' | 'lowered' | 'working';
    sectionStates?: boolean[];
    applicationRate?: number;
  };

  alerts?: Alert[];
}

2.6 측정 단위

측정 SI 단위 일반 대안 변환 필요
면적 m² (헥타르) 에이커, 평 1 ha = 3,025 평
질량 kg (톤) 근, 가마 1 t = 1,000 kg
부피 L (m³) 말, 되 1 L = 0.001 m³
적용량 kg/ha, L/ha kg/평 제품별 다름
수확량 t/ha kg/평 작물별 다름

2.7 시간 표준

// 날짜/시간 표준
interface TemporalStandards {
  // 타임스탬프용 ISO 8601
  instant: '2025-06-15T10:30:00Z';       // UTC
  localTime: '2025-06-15T19:30:00+09:00'; // 한국 시간

  // 재배 시즌 참조
  growingSeason: {
    year: 2025,
    hemisphere: 'northern',
    startDate: '2025-04-15',
    endDate: '2025-10-31'
  };

  // 생육 단계 (BBCH 척도)
  cropStage: {
    code: '65',
    description: '만개기',
    crop: 'wheat',
    scale: 'BBCH'
  };

  // 상대적 기간
  period: {
    type: 'pre-plant' | 'in-season' | 'post-harvest';
    daysRelativeTo: 'planting' | 'emergence' | 'harvest';
  };
}

2.8 의미론적 어휘

농업 어휘

  • AGROVOC: FAO의 농업 다국어 시소러스
  • Crop Ontology: 작물의 특성, 방법, 척도
  • ENVO: 생태계를 위한 환경 온톨로지
  • QUDT: 양, 단위, 차원, 유형
  • Schema.org: 농업을 위한 웹 어휘 확장
