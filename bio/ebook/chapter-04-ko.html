제4장: 임상시험 데이터 관리

임상 데이터 수집, 품질 및 규제 제출 표준

4.1 임상 데이터 표준 소개

임상시험은 실험실 발견에서 환자 혜택으로 가는 중요한 경로입니다. WIA-BIO 임상시험 데이터 관리 표준은 현대 생명공학 역량을 활용하면서 글로벌 규제 기대에 부합하는 데이터 수집, 품질 보증, 이상반응 보고 및 규제 제출에 대한 포괄적인 요구사항을 수립합니다.

효과적인 임상 데이터 관리는 유전자 치료에서 맞춤의학 접근법에 이르기까지 생명공학 제품의 안전성과 유효성을 입증하는 데 필수적입니다. WIA-BIO 표준은 생명공학 응용에 대한 구체적인 지침을 제공하면서 CDISC, ICH E6 (GCP) 및 HL7 FHIR을 포함한 확립된 프레임워크와 통합됩니다.

4.2 임상시험 단계 및 데이터 요구사항

임상 개발의 각 단계는 적절한 데이터 관리 전략을 통해 해결해야 하는 고유한 데이터 요구사항을 가집니다:

단계 주요 목적 참가자 주요 데이터 유형
1상 안전성, 용량, PK/PD 20-100명 건강한 지원자 안전성, PK, 바이오마커
2상 유효성, 용량-반응 100-500명 환자 유효성 평가변수, 안전성
3상 유효성 확인, AE 모니터링 1,000-5,000명 환자 1차/2차 평가변수
4상 시판 후 조사 수천 명의 환자 장기 안전성, RWE

4.3 CDISC 표준 구현

WIA-BIO는 규제 제출을 위해 임상 데이터 교환 표준 컨소시엄(CDISC) 표준 준수를 의무화합니다: 

// CDISC SDTM 도메인 구조
interface SDTMDomain {
    domainCode: string;          // 예: "DM", "AE", "LB"
    datasetName: string;         // 데이터셋 이름
    description: string;         // 설명
    class: 'General Observations' | 'Events' | 'Findings' | 'Special Purpose';
    structure: '피험자당 1레코드' | '이벤트당 1레코드' | '결과당 1레코드';
}

// 인구통계 도메인 (DM)
interface DMDomain {
    STUDYID: string;             // 연구 식별자
    DOMAIN: 'DM';                // 도메인 약어
    USUBJID: string;             // 고유 피험자 식별자
    SUBJID: string;              // 연구 피험자 식별자
    RFSTDTC: string;             // 피험자 참조 시작 날짜/시간
    RFENDTC: string;             // 피험자 참조 종료 날짜/시간
    SITEID: string;              // 연구 기관 식별자
    BRTHDTC?: string;            // 생년월일
    AGE: number;                 // 나이
    AGEU: 'YEARS' | 'MONTHS';    // 나이 단위
    SEX: 'M' | 'F' | 'U';        // 성별
    RACE: string;                // 인종
    ETHNIC: string;              // 민족
    ARMCD: string;               // 계획된 군 코드
    ARM: string;                 // 계획된 군 설명
    COUNTRY: string;             // 국가
}

// 이상반응 도메인 (AE)
interface AEDomain {
    STUDYID: string;
    DOMAIN: 'AE';
    USUBJID: string;
    AESEQ: number;               // 순서 번호
    AETERM: string;              // 보고된 AE 용어
    AEDECOD: string;             // 사전 파생 용어
    AEBODSYS: string;            // 신체 계통 또는 기관 분류
    AESEV: 'MILD' | 'MODERATE' | 'SEVERE';  // 중증도
    AESER: 'Y' | 'N';            // 중대한 이벤트
    AEACN: string;               // 시험 치료와 관련된 조치
    AEREL: string;               // 인과관계
    AEOUT: string;               // 이상반응 결과
    AESTDTC: string;             // AE 시작 날짜/시간
    AEENDTC?: string;            // AE 종료 날짜/시간
}

4.4 전자 데이터 수집(EDC) 요구사항

WIA-BIO는 임상시험에 사용되는 전자 데이터 수집 시스템에 대한 요구사항을 명시합니다:

EDC 시스템 요구사항

  • 21 CFR Part 11 준수: 전자서명, 감사추적, 접근 통제
  • 데이터 검증: 실시간 편집 검사, 범위 검증, 일관성 검사
  • 감사 추적: 사용자, 타임스탬프, 이유와 함께 모든 데이터 변경의 완전한 이력
  • 보안: 역할 기반 접근, 저장 및 전송 시 암호화
  • 통합: 데이터 교환을 위한 HL7 FHIR, CDISC ODM 지원
  • 백업: 검증된 복구 절차가 있는 자동 백업
// EDC 시스템 구성 인터페이스
interface EDCConfiguration {
    studyId: string;             // 연구 ID
    protocolVersion: string;     // 프로토콜 버전

    // 양식 정의
    forms: CRFForm[];

    // 검증 규칙
    editChecks: EditCheck[];

    // 사용자 관리
    roles: UserRole[];

    // 통합 엔드포인트
    integrations: {
        labSystems: LabIntegration[];      // 검사 시스템
        eProSystems: eProIntegration[];    // ePRO 시스템
        safetyDatabases: SafetyIntegration[]; // 안전성 데이터베이스
    };

    // 규정 준수 설정
    compliance: {
        cfr21Part11: boolean;
        gdpr: boolean;
        hipaa: boolean;
    };
}

interface EditCheck {
    checkId: string;             // 검사 ID
    description: string;         // 설명
    formId: string;              // 양식 ID
    fieldId: string;             // 필드 ID
    checkType: 'RANGE' | 'CONSISTENCY' | 'CONDITIONAL' | 'CROSS_FORM';
    expression: string;          // 표현식
    severity: 'ERROR' | 'WARNING' | 'INFO';
    message: string;             // 메시지
}

// 편집 검사 구현 예시
const creatinineLevelCheck: EditCheck = {
    checkId: "EC001",
    description: "크레아티닌 수준 범위 검사",
    formId: "LAB_RESULTS",
    fieldId: "CREATININE",
    checkType: "RANGE",
    expression: "CREATININE >= 0.5 AND CREATININE <= 15.0",
    severity: "WARNING",
    message: "크레아티닌 값이 예상 범위(0.5-15.0 mg/dL)를 벗어남"
};

4.5 이상반응 보고

시기적절하고 정확한 이상반응 보고는 환자 안전에 중요합니다. WIA-BIO는 AE 수집 및 보고에 대한 엄격한 요구사항을 수립합니다:

이벤트 유형 보고 기한 보고 대상 형식
치명적/생명 위협 SAE 7일 (역일) 규제 기관, IRB CIOMS-I, MedWatch
기타 중대한 AE 15일 (역일) 규제 기관, IRB CIOMS-I, MedWatch
비중대 AE 프로토콜 일정에 따라 안전성 데이터베이스 EDC 입력
SUSAR (예상) 15일 (역일) IND 안전성 보고서 서술형 + CIOMS 
// 이상반응 처리 워크플로우
interface AEProcessingWorkflow {
    stages: {
        intake: {
            source: 'EDC' | 'SITE_CALL' | 'PATIENT_REPORT';
            initialAssessment: boolean;      // 초기 평가
            medicalCoding: 'MedDRA';          // 의학 코딩
            priorityTriage: boolean;         // 우선순위 분류
        };

        medicalReview: {
            reviewerQualification: 'MD' | 'PharmD'; // 검토자 자격
            seriousnessAssessment: boolean;  // 중대성 평가
            causalityAssessment: boolean;    // 인과관계 평가
            expectednessAssessment: boolean; // 예상 여부 평가
        };

        regulatory: {
            reportabilityDetermination: boolean; // 보고 가능성 결정
            narrativeGeneration: boolean;    // 서술형 생성
            regulatorySubmission: boolean;   // 규제 제출
            trackingNumber: string;          // 추적 번호
        };

        followUp: {
            resolutionTracking: boolean;     // 해결 추적
            sequelaeDocumentation: boolean;  // 후유증 문서화
            outcomeRecording: boolean;       // 결과 기록
        };
    };
}

// MedDRA 코딩 인터페이스
interface MedDRACoding {
    version: string;               // 예: "25.1"
    primarySOC: string;            // 기관계 대분류
    hlgt: string;                  // 상위그룹용어
    hlt: string;                   // 상위용어
    pt: string;                    // 대표용어
    llt: string;                   // 하위용어
}

4.6 데이터 품질 관리

데이터 품질 보장은 규제 승인 및 과학적 타당성에 필수적입니다. WIA-BIO는 포괄적인 품질 관리 프로세스를 의무화합니다: 

// 데이터 품질 메트릭 인터페이스
interface DataQualityMetrics {
    studyId: string;
    reportDate: Date;

    // 완전성 메트릭
    completeness: {
        overallCompletionRate: number;    // 목표: > 95%
        criticalFieldCompletion: number;  // 목표: > 99%
        missingDataRate: number;          // 목표: < 2%
    };

    // 정확성 메트릭
    accuracy: {
        queryRate: number;                // 목표: < 5%
        errorRate: number;                // 목표: < 1%
        discrepancyRate: number;          // 목표: < 3%
    };

    // 적시성 메트릭
    timeliness: {
        dataEntryLag: number;             // 방문 후 일수
        queryResolutionTime: number;      // 평균 일수
        lockToDBLockTime: number;         // 일수
    };

    // 일관성 메트릭
    consistency: {
        protocolDeviationRate: number;    // 프로토콜 일탈률
        editCheckFailureRate: number;     // 편집 검사 실패율
        crossFormConsistency: number;     // 목표: > 98%
    };
}

// 쿼리 관리 시스템
interface QueryManagement {
    queryId: string;             // 쿼리 ID
    siteId: string;              // 기관 ID
    subjectId: string;           // 피험자 ID
    formName: string;            // 양식명
    fieldName: string;           // 필드명
    queryType: 'AUTO' | 'MANUAL'; // 쿼리 유형
    status: 'OPEN' | 'ANSWERED' | 'CLOSED' | 'CANCELLED';
    priority: 'HIGH' | 'MEDIUM' | 'LOW';
    queryText: string;           // 쿼리 텍스트
    response?: string;           // 응답
    openedDate: Date;            // 개시일
    closedDate?: Date;           // 종료일
    daysOpen: number;            // 개시 일수
}

4.7 생명공학 특화 데이터 요소

생명공학 임상시험은 표준 CDISC 도메인에서 다루지 않는 전문화된 데이터 요소를 필요로 합니다. WIA-BIO는 확장된 데이터 구조를 정의합니다: 

// 유전자 치료 임상시험 데이터 확장
interface GeneTherapyData {
    // 벡터 정보
    vector: {
        type: 'AAV' | 'Lentivirus' | 'Adenovirus' | 'mRNA-LNP';
        serotype?: string;           // 혈청형
        titer: number;               // 역가
        tiferUnit: string;           // 역가 단위
        lotNumber: string;           // 로트 번호
        manufacturingDate: Date;     // 제조일
    };

    // 투약 정보
    dosing: {
        route: 'IV' | 'IT' | 'IM' | 'SC' | 'Intravitreal' | 'Other';
        doseLevel: number;           // 용량 수준
        doseUnit: string;            // 예: "vg/kg"
        administrationDate: Date;    // 투여일
        premedication: string[];     // 전처치
    };

    // 바이오마커 추적
    biomarkers: {
        transductionEfficiency?: number;  // 형질도입 효율
        transgeneExpression: {
            tissue: string;          // 조직
            level: number;           // 수준
            unit: string;            // 단위
            timepoint: string;       // 시점
        }[];
        immunogenicity: {
            aabPresence: boolean;    // AAb 존재
            neutralizingAb: boolean; // 중화 항체
            tCellResponse: boolean;  // T세포 반응
        };
    };

    // 장기 추적조사
    ltfu: {
        followUpYears: number;       // 일반적으로 5-15년
        malignancyScreening: boolean; // 악성종양 스크리닝
        insertionalMutagenesis: boolean; // 삽입 돌연변이
        durabilityAssessment: boolean;   // 지속성 평가
    };
}

4.8 규제 제출 표준

WIA-BIO는 주요 규제 기관에 대한 규제 제출에 필요한 데이터 형식 및 구조를 명시합니다:

핵심 요약

  • CDISC 준수: FDA 제출을 위해 원시 데이터용 SDTM, 분석 데이터셋용 ADaM 필수
  • EDC 요구사항: 포괄적인 감사 추적이 있는 21 CFR Part 11 준수 시스템
  • AE 보고: 규제 기관에 대한 중대한 이상반응 보고의 엄격한 기한
  • 데이터 품질: 정의된 메트릭 및 임계값으로 지속적인 모니터링
  • 생명공학 확장: 유전자 치료, 세포 치료 및 기타 생명공학 방식을 위한 전문화된 데이터 구조
  • 글로벌 정렬: FDA, EMA, PMDA 및 기타 규제 요구사항을 충족하도록 설계된 표준
