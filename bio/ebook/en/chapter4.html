4.2 FASTA Compatibility Layer

Bidirectional conversion between WIA JSON format and FASTA files.

FASTA Import

Python - FASTA to WIA Conversion
def convert_fasta_to_wia(fasta_file):
    """Convert FASTA file to WIA Biotech JSON format"""
    records = parse_fasta(fasta_file)

    sequences = []
    for record in records:
        seq = {
            "sequence_id": generate_id(),
            "sequence_info": {
                "name": record.description,
                "description": record.description
            },
            "sequence_type": infer_type(record.sequence),
            "length_bp": len(record.sequence),
            "gc_content": calculate_gc(record.sequence) if is_dna(record.sequence) else None,
            "files": {
                "fasta": fasta_file
            },
            "checksum": {
                "algorithm": "sha256",
                "value": hashlib.sha256(record.sequence.encode()).hexdigest()
            }
        }
        sequences.append(seq)

    return sequences

FASTA Export

Python - WIA to FASTA Conversion
def convert_wia_to_fasta(sequence_json, fasta_file):
    """Convert WIA Biotech JSON to FASTA format"""
    seq = load_json(sequence_json)

    header = f">{seq['sequence_id']} {seq['sequence_info']['name']}"
    if 'organism' in seq['sequence_info']:
        header += f" | {seq['sequence_info']['organism']}"
    header += f" | {seq['length_bp']}bp"

    # Read actual sequence from linked FASTA file
    sequence = read_sequence_from_file(seq['files']['fasta'])

    with open(fasta_file, 'w') as f:
        f.write(header + '\n')
        # Write sequence in 80-character lines
        for i in range(0, len(sequence), 80):
            f.write(sequence[i:i+80] + '\n')

    return fasta_file

def convert_wia_to_sbol(part_json, sbol_file):
    """Convert WIA Biotech part to SBOL format"""
    import sbol2

    part = load_json(part_json)
    doc = sbol2.Document()

    # Create component definition
    component = doc.componentDefinitions.create(part['part_id'])
    component.name = part['part_info']['name']
    component.description = part['part_info'].get('description', '')
    component.roles = map_part_type_to_sbol(part['part_info']['part_type'])

    # Add sequence
    seq = doc.sequences.create(f"{part['part_id']}_seq")
    seq.elements = part['sequence']['dna']
    seq.encoding = sbol2.IUPAC_DNA
    component.sequences = seq

    # Add assembly metadata
    if 'assembly' in part:
        component.wasDerivedFrom = part['registry'].get('url', '')

    doc.write(sbol_file)
    return sbol_file

def import_alphafold_structure(uniprot_id):
    """Import AlphaFold prediction to WIA format"""
    # Fetch from AlphaFold DB
    pdb_url = f"https://alphafold.ebi.ac.uk/files/AF-{uniprot_id}-F1-model_v4.pdb"
    confidence_url = f"https://alphafold.ebi.ac.uk/files/AF-{uniprot_id}-F1-confidence_v4.json"

    pdb_data = download_file(pdb_url)
    confidence_data = download_json(confidence_url)

    structure = {
        "structure_id": f"struct-{uniprot_id}",
        "structure_info": {
            "name": get_protein_name(uniprot_id),
            "uniprot_id": uniprot_id
        },
        "prediction": {
            "method": "alphafold",
            "version": "AlphaFold 2.3",
            "source": "alphafold_db",
            "date": datetime.now().isoformat()
        },
        "quality": {
            "mean_plddt": calculate_mean_plddt(confidence_data['plddt']),
            "confident_residues_pct": calculate_confident_pct(confidence_data['plddt'])
        },
        "files": {
            "pdb": save_pdb(pdb_data),
            "confidence": save_json(confidence_data)
        }
    }

    return structure

Import/export with GenBank annotated sequence format.

GenBank Import

Python - GenBank to WIA Conversion
from Bio import SeqIO

def import_genbank(gb_file):
    """Import GenBank file to WIA format"""
    record = SeqIO.read(gb_file, "genbank")

    sequence = {
        "sequence_id": generate_id(),
        "sequence_info": {
            "name": record.name,
            "description": record.description,
            "organism": record.annotations.get('organism', ''),
            "gene_symbol": extract_gene_symbol(record)
        },
        "sequence_type": "dna",
        "length_bp": len(record.seq),
        "gc_content": calculate_gc(str(record.seq)),
        "source": {
            "database": "NCBI GenBank",
            "accession": record.id,
            "version": record.annotations.get('sequence_version', '')
        },
        "annotations": []
    }

    # Convert features to WIA annotations
    for feature in record.features:
        annotation = {
            "feature": feature.type,
            "start": int(feature.location.start) + 1,  # 1-based
            "end": int(feature.location.end),
            "strand": "+" if feature.location.strand == 1 else "-",
            "name": feature.qualifiers.get('gene', [''])[0],
            "qualifiers": dict(feature.qualifiers)
        }
        sequence['annotations'].append(annotation)

    return sequence

4.6 Data Validation Protocol

Schema validation ensures data integrity across the ecosystem.

JSON Schema Validation

Python - Schema Validation
import jsonschema

def validate_wia_data(data, schema_type):
    """Validate WIA Biotech data against JSON schema"""
    schema_url = f"https://wia.live/schemas/bio/{schema_type}.schema.json"
    schema = download_json(schema_url)

    try:
        jsonschema.validate(instance=data, schema=schema)
        return {"valid": True}
    except jsonschema.ValidationError as e:
        return {
            "valid": False,
            "error": str(e.message),
            "path": list(e.path)
        }

# Example usage
sequence_data = load_json("sequence.json")
result = validate_wia_data(sequence_data, "sequence")
if not result['valid']:
    print(f"Validation error: {result['error']}")
