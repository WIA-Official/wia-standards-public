Encryption Standards

Python - Data Encryption
from cryptography.fernet import Fernet
import hashlib

class GeneticDataProtection:
    """Encryption for sensitive genetic data"""

    def __init__(self, encryption_key: bytes):
        self.cipher = Fernet(encryption_key)

    def encrypt_sequence(self, sequence_data: dict) -> dict:
        """Encrypt sensitive sequence fields"""
        # Generate de-identified ID
        original_id = sequence_data['sequence_id']
        deidentified_id = hashlib.sha256(original_id.encode()).hexdigest()[:16]

        # Encrypt PII fields
        encrypted = sequence_data.copy()
        if 'sequence_info' in encrypted:
            if 'patient_id' in encrypted['sequence_info']:
                encrypted['sequence_info']['patient_id'] = self.cipher.encrypt(
                    encrypted['sequence_info']['patient_id'].encode()
                ).decode()

        encrypted['sequence_id'] = deidentified_id
        encrypted['encryption_applied'] = True

        return encrypted

    def decrypt_sequence(self, encrypted_data: dict) -> dict:
        """Decrypt sequence data for authorized users"""
        decrypted = encrypted_data.copy()

        if 'patient_id' in decrypted.get('sequence_info', {}):
            decrypted['sequence_info']['patient_id'] = self.cipher.decrypt(
                decrypted['sequence_info']['patient_id'].encode()
            ).decode()

        return decrypted

6.4 Access Control & Authorization

Role-Based Access Control (RBAC)

Python - RBAC Implementation
from enum import Enum
from typing import Set

class Role(Enum):
    PI = "principal_investigator"
    RESEARCHER = "researcher"
    TECHNICIAN = "technician"
    VIEWER = "viewer"

class Permission(Enum):
    READ = "read"
    WRITE = "write"
    DELETE = "delete"
    SHARE = "share"
    ADMIN = "admin"

ROLE_PERMISSIONS = {
    Role.PI: {Permission.READ, Permission.WRITE, Permission.DELETE, Permission.SHARE, Permission.ADMIN},
    Role.RESEARCHER: {Permission.READ, Permission.WRITE, Permission.SHARE},
    Role.TECHNICIAN: {Permission.READ, Permission.WRITE},
    Role.VIEWER: {Permission.READ}
}

class AccessControl:
    """Manage project access permissions"""

    def __init__(self, project_id: str):
        self.project_id = project_id
        self.user_roles = {}  # user_id -> Role

    def grant_access(self, user_id: str, role: Role):
        """Grant user access to project"""
        self.user_roles[user_id] = role

    def check_permission(self, user_id: str, permission: Permission) -> bool:
        """Check if user has permission"""
        role = self.user_roles.get(user_id)
        if not role:
            return False
        return permission in ROLE_PERMISSIONS[role]

    def audit_log(self, user_id: str, action: str, resource: str):
        """Log access for audit trail"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "user_id": user_id,
            "project_id": self.project_id,
            "action": action,
            "resource": resource
        }
        # Write to audit log
        write_audit_log(log_entry)

Compliance Checklist

  • Data encryption at rest (AES-256) and in transit (TLS 1.3)
  • De-identification of human genetic data
  • IRB approval for human subjects research
  • Biosafety committee approval for GMO work
  • Access control with role-based permissions
  • Comprehensive audit logging (5-year retention)
  • Incident response plan documented
  • Regular security assessments
  • Data retention and deletion policies
  • Third-party data sharing agreements
