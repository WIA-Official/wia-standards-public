import json

# Initialize client
client = WIABiotechClient(api_key='your_api_key')

        "category": "gene_therapy"
    },
    "organization": {
        "name": "Research Institute",
        "principal_investigator": "Dr. Kim"
    },
    "ethics": {
        "irb_approval": "IRB-2025-0001",
        "biosafety_level": "BSL-2"
    }
})
project_id = project['project_id']

# Step 2: Upload target sequence
with open('target_gene.fasta', 'r') as f:
    sequence = client.sequences.create(
        project_id=project_id,
        sequence_info={
            "name": "Target Gene ABC",
            "organism": "Homo sapiens",
            "gene_symbol": "ABC1"
        },
        sequence_type="dna",
        fasta_file=f
    )
sequence_id = sequence['sequence_id']

# Step 3: Design gRNAs
from wia_biotech.crispr import GRNADesigner

designer = GRNADesigner(pam="NGG")
grnas = designer.design(
    sequence_file='target_gene.fasta',
    target_region=(500, 1500),
    count=3
)

# Step 4: Create CRISPR experiment
experiment = client.crispr_experiments.create(
    project_id=project_id,
    experiment_info={
        "name": "Gene ABC Knockout Study",
        "experimenter": "Dr. Lee"
    },
    target={
        "gene_symbol": "ABC1",
        "cell_line": "HEK293T",
        "sequence_ref": sequence_id
    },
    editing_system={
        "type": "crispr_cas9",
        "cas_variant": "SpCas9",
        "delivery_method": "lipofection"
    },
    guide_rnas=grnas,
    editing_type="knockout"
)
experiment_id = experiment['experiment_id']

# Step 5: Upload results (after wet-lab work)
with open('ngs_results.fastq.gz', 'rb') as f1, \
     open('ngs_results_R2.fastq.gz', 'rb') as f2:
    results = client.crispr_experiments.upload_results(
        experiment_id=experiment_id,
        analysis_method="NGS",
        fastq_files=[f1, f2]
    )

print(f"Editing efficiency: {results['editing_efficiency']['overall_efficiency']:.2%}")
print(f"Off-target sites detected: {len(results['off_target_analysis']['sites'])}")

from wia_biotech import WIABiotechClient
from Bio import SeqIO

# Upload protein sequence

    project_id=project_id,
    sequence_info={
        "name": "ABC1 Protein",
        "organism": "Homo sapiens"
    },
    sequence_type="protein",
    fasta_file=open('protein.fasta', 'r')
)

# Request AlphaFold prediction (async)
prediction_job = client.structures.predict(
    sequence_id=sequence['sequence_id'],
    method="alphafold",
    version="2.3"
)
job_id = prediction_job['job_id']

# Poll for completion
import time
while True:
    status = client.structures.get_prediction_status(job_id)
    if status['status'] == 'completed':
        break
    print(f"Status: {status['progress']}%")
    time.sleep(30)

# Download structure
structure = client.structures.get(status['structure_id'])
pdb_file = client.structures.download_pdb(structure['structure_id'])

print(f"Mean pLDDT: {structure['quality']['mean_plddt']:.1f}")
print(f"Confident residues: {structure['quality']['confident_residues_pct']:.1%}")

// Create Golden Gate assembly
async function createGoldenGateAssembly(projectId: string, parts: any) {
  const assembly = await client.assemblies.create({
    projectId,
    assemblyInfo: {
      name: 'GFP Expression Cassette'
    },
    assemblyMethod: AssemblyStandard.GOLDEN_GATE,
    parts: [
      { position: 1, partRef: parts.promoter.part_id, type: PartType.PROMOTER },
      { position: 2, partRef: parts.rbs.part_id, type: PartType.RBS },
      { position: 3, partRef: parts.gfp.part_id, type: PartType.CDS },
      { position: 4, partRef: parts.terminator.part_id, type: PartType.TERMINATOR }
    ]
  });

  // Export as SBOL
  const sbol = await client.parts.exportSBOL(assembly.assembly_id);
  return sbol;
}

7.3 Bash/CLI Examples

Sequence Upload Pipeline

Bash - Batch Sequence Upload
#!/bin/bash
# Upload multiple FASTA files to WIA Biotech

WIA_API_KEY="your_api_key"
PROJECT_ID="proj-001"
API_URL="https://api.wia.live/v1"

# Process all FASTA files in directory
for fasta_file in *.fasta; do
    echo "Processing $fasta_file..."

    # Extract sequence info from FASTA header
    name=$(head -n 1 "$fasta_file" | sed 's/>//' | cut -d' ' -f1)

    # Upload sequence
    response=$(curl -s -X POST \
        -H "Authorization: Bearer $WIA_API_KEY" \
        -F "sequence_info={\"name\":\"$name\",\"organism\":\"Homo sapiens\"}" \
        -F "sequence_type=dna" \
        -F "fasta_file=@$fasta_file" \
        "$API_URL/projects/$PROJECT_ID/sequences")

    sequence_id=$(echo "$response" | jq -r '.sequence_id')
    echo "  Created sequence: $sequence_id"
done

echo "Upload complete!"
