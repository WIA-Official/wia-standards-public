5.1 Laboratory Information Management Systems (LIMS)

Integrate WIA Biotech standard into existing LIMS platforms for seamless data management.

LIMS Integration Architecture

Python - LIMS Connector
class WIABiotechLIMSIntegration:
    """Integration layer for LIMS systems"""

    def __init__(self, lims_api_url, wia_api_url):
        self.lims = LIMSClient(lims_api_url)
        self.wia = WIABiotechClient(wia_api_url)

    def sync_sequence_data(self, lims_sample_id):
        """Sync sequence data from LIMS to WIA"""
        # Fetch from LIMS
        sample = self.lims.get_sample(lims_sample_id)

        # Convert to WIA format
        wia_sequence = {
            "sequence_info": {
                "name": sample['name'],
                "organism": sample['organism']
            },
            "sequence_type": "dna",
            "source": {
                "database": "Internal LIMS",
                "accession": lims_sample_id
            }
        }

        # Upload to WIA
        response = self.wia.create_sequence(wia_sequence)
        return response['sequence_id']

    def sync_crispr_results(self, experiment_id):
        """Sync CRISPR results back to LIMS"""
        results = self.wia.get_crispr_results(experiment_id)

        # Update LIMS with editing efficiency
        self.lims.update_experiment_results(
            experiment_id,
            editing_efficiency=results['editing_efficiency']['overall_efficiency'],
            indel_frequency=results['editing_efficiency']['indel_frequency']
        )

        return True

from opentrons import protocol_api

def wia_golden_gate_assembly(protocol: protocol_api.ProtocolContext):
    """Opentrons protocol for Golden Gate assembly with WIA parts"""

    # Load WIA assembly design
    assembly = wia_client.get_assembly('asm-001')

    # Configure deck
    tiprack = protocol.load_labware('opentrons_96_tiprack_300ul', 1)
    plate = protocol.load_labware('nest_96_wellplate_200ul_flat', 2)
    pipette = protocol.load_instrument('p300_single', 'left', tip_racks=[tiprack])

    # Fetch part concentrations from WIA
    for part in assembly['parts']:
        part_data = wia_client.get_part(part['part_ref'])
        volume = calculate_volume(part_data['sequence']['length_bp'])

        # Pipette part into assembly well
        well = plate.wells()[part['position'] - 1]
        pipette.transfer(volume, part_source[part['part_ref']], well)

    # Add enzyme mix
    pipette.transfer(10, enzyme_mix, plate.wells()[0])

    protocol.comment("Golden Gate assembly complete")

5.3 Research Platform Integration

Benchling Integration

TypeScript - Benchling Connector
import { BenchlingAPI } from '@benchling/api';
import { WIABiotechClient } from '@wia/biotech-sdk';

class BenchlingWIASync {
  constructor(
    private benchling: BenchlingAPI,
    private wia: WIABiotechClient
  ) {}

  async syncDNASequence(benchlingId: string): Promise<string> {
    // Fetch from Benchling
    const dnaSequence = await this.benchling.dnaSequences.get(benchlingId);

    // Convert to WIA format
    const wiaSequence = {
      sequence_info: {
        name: dnaSequence.name,
        description: dnaSequence.description
      },
      sequence_type: 'dna',
      length_bp: dnaSequence.bases.length,
      gc_content: this.calculateGC(dnaSequence.bases),
      annotations: dnaSequence.annotations.map(ann => ({
        feature: ann.type,
        start: ann.start,
        end: ann.end,
        strand: ann.strand,
        name: ann.name
      }))
    };

    // Upload to WIA
    const response = await this.wia.sequences.create(wiaSequence);
    return response.sequence_id;
  }

  async exportToSBOL(partId: string): Promise<Blob> {
    const part = await this.wia.parts.get(partId);
    const sbol = await this.wia.parts.exportSBOL(partId);
    return sbol;
  }
}

Geneious Integration

Java - Geneious Plugin
public class WIABiotechGeneiousPlugin extends DocumentOperation {

    @Override
    public List<AnnotatedPluginDocument> performOperation(
        List<AnnotatedPluginDocument> documents,
        ProgressListener progress,
        Options options
    ) {
        WIABiotechClient wia = new WIABiotechClient(apiKey);

        for (AnnotatedPluginDocument doc : documents) {
            NucleotideSequenceDocument seq = (NucleotideSequenceDocument) doc.getDocument();

            // Convert to WIA format
            JSONObject wiaSequence = new JSONObject();
            wiaSequence.put("sequence_info", new JSONObject()
                .put("name", seq.getName())
                .put("organism", seq.getOrganism())
            );
            wiaSequence.put("sequence_type", "dna");
            wiaSequence.put("length_bp", seq.getCharSequence().length());

            // Upload to WIA
            JSONObject response = wia.sequences.create(wiaSequence);
            progress.setMessage("Uploaded: " + response.getString("sequence_id"));
        }

        return documents;
    }
}

import boto3
import json

wia = WIABiotechClient(api_key=os.environ['WIA_API_KEY'])

def lambda_handler(event, context):
    """Process FASTA files uploaded to S3 and sync to WIA"""

    bucket = event['Records'][0]['s3']['bucket']['name']
    key = event['Records'][0]['s3']['object']['key']

    # Download FASTA from S3
    fasta_content = s3.get_object(Bucket=bucket, Key=key)['Body'].read()

    # Convert to WIA format
    sequences = convert_fasta_to_wia(fasta_content)

    # Upload to WIA
    for seq in sequences:
        response = wia.sequences.create(seq)
        print(f"Created sequence: {response['sequence_id']}")

    return {
        'statusCode': 200,
        'body': json.dumps(f'Processed {len(sequences)} sequences')
    }

from google.cloud import storage
import functions_framework

@functions_framework.cloud_event
def process_genbank_upload(cloud_event):
    """Process GenBank files in GCS and sync to WIA"""

    data = cloud_event.data
    bucket_name = data["bucket"]
    file_name = data["name"]

    # Download from GCS
    storage_client = storage.Client()
    bucket = storage_client.bucket(bucket_name)
    blob = bucket.blob(file_name)
    genbank_content = blob.download_as_text()

    # Convert to WIA
    sequence = import_genbank_string(genbank_content)

    # Upload to WIA
    wia = WIABiotechClient(api_key=os.environ['WIA_API_KEY'])
    response = wia.sequences.create(sequence)

    print(f"Created sequence: {response['sequence_id']}")
    return response

Nextflow - CRISPR Analysis Workflow
#!/usr/bin/env nextflow

params.wia_project = 'proj-001'
params.wia_api_key = env.WIA_API_KEY

process downloadSequence {
    input:
    val sequence_id

    output:
    path 'sequence.fasta'

    script:
    """
    curl -H "Authorization: Bearer ${params.wia_api_key}" \\
         https://api.wia.live/v1/sequences/${sequence_id}/fasta \\
         -o sequence.fasta
    """
}

process designGRNAs {
    input:
    path fasta

    output:
    path 'grnas.json'

    script:
    """
    python design_grnas.py ${fasta} > grnas.json
    """
}

process uploadCRISPRExperiment {
    input:
    path grnas

    output:
    val experiment_id

    script:
    """
    curl -X POST -H "Authorization: Bearer ${params.wia_api_key}" \\
         -H "Content-Type: application/json" \\
         -d @${grnas} \\
         https://api.wia.live/v1/projects/${params.wia_project}/crispr-experiments
    """
}

workflow {
    sequence_id = Channel.of('seq-001')
    downloadSequence(sequence_id)
        | designGRNAs
        | uploadCRISPRExperiment
}
