제2장: CRISPR 및 유전자 편집 프로토콜

정밀 유전체 수정을 위한 표준화된 절차

2.1 유전자 편집 기술 개요

유전자 편집은 현대 생명공학에서 가장 혁신적인 능력 중 하나로, 생물체의 유전 코드를 정밀하게 수정할 수 있게 합니다. WIA-BIO 유전자 편집 표준은 CRISPR-Cas 시스템, 염기 편집기, 프라임 편집기 및 관련 기술에 대한 종합적인 프로토콜을 수립하여, 모든 응용에서 안전성, 재현성 및 윤리적 준수를 보장합니다.

CRISPR-Cas9 기술의 개발은 유전자 편집을 대중화하여 전 세계 연구실에서 접근할 수 있게 만들었습니다. 그러나 이러한 접근성은 실험의 재현성, 결과의 신뢰성, 그리고 연구자와 더 넓은 환경 모두에 대한 응용의 안전성을 보장하기 위한 엄격한 표준화를 요구합니다.

WIA-BIO는 각각 특정 응용에 최적화된 여러 CRISPR-Cas 시스템을 지원합니다. 각 시스템의 특성을 이해하는 것은 적절한 도구 선택에 필수적입니다:

2.3 가이드 RNA 설계 표준

효과적인 가이드 RNA (gRNA) 설계는 성공적인 유전자 편집에 중요합니다. WIA-BIO는 오프타겟 효과를 최소화하기 위해 여러 알고리즘을 사용한 계산 분석을 의무화합니다: 

// WIA-BIO gRNA 설계 인터페이스
interface GuideRNADesign {
    targetSequence: string;      // 20-nt 표적 시퀀스
    pamSequence: string;         // PAM 시퀀스 (예: "NGG")
    casSystem: CasSystemType;

    // 점수 메트릭
    onTargetScore: number;       // 최소: 0.5 (50%)
    offTargetScore: number;      // 최대: 0.1 (10%)
    gcContent: number;           // 최적: 40-70%

    // 오프타겟 분석
    offTargets: OffTargetSite[];
    maxMismatches: number;       // 일반적으로 0-4

    // 전달 매개변수
    deliveryMethod: DeliveryType;
    modification: RNAModification;
}

interface OffTargetSite {
    chromosome: string;          // 염색체
    position: number;            // 위치
    sequence: string;            // 시퀀스
    mismatches: number;          // 불일치 수
    mismatchPositions: number[]; // 불일치 위치
    score: number;               // 점수
    gene?: string;               // 중첩 유전자 (있는 경우)
    criticalRegion: boolean;     // 엑손, 프로모터 등
}

// gRNA 점수 계산 알고리즘
function calculateGRNAScore(grna: GuideRNADesign): GRNAScore {
    // 위치 가중 불일치 점수
    const positionWeights = [
        0.0, 0.0, 0.014, 0.0, 0.0,       // 위치 1-5
        0.395, 0.317, 0.0, 0.389, 0.079, // 위치 6-10
        0.445, 0.508, 0.613, 0.851, 0.732, // 위치 11-15
        0.828, 0.615, 0.804, 0.685, 0.583  // 위치 16-20
    ];

    let offTargetRisk = 0;
    for (const site of grna.offTargets) {
        let siteRisk = 1.0;
        for (const pos of site.mismatchPositions) {
            siteRisk *= (1 - positionWeights[pos]);
        }
        offTargetRisk += siteRisk;
    }

    return {
        onTarget: grna.onTargetScore,
        offTarget: Math.max(0, 1 - offTargetRisk),
        overall: grna.onTargetScore * Math.max(0, 1 - offTargetRisk)
    };
}

2.4 실험 프로토콜 표준

WIA-BIO는 실험실 간 재현성을 보장하기 위해 유전자 편집 실험을 위한 표준화된 프로토콜을 수립합니다:

2.4.1 세포주 준비

편집 전 요구사항

  • 세포 인증: 6개월 이내 STR 프로파일링
  • 마이코플라즈마 검사: 30일 이내 음성 결과
  • 계대 번호: 문서화 및 권장 범위 제한
  • 생존율: 형질전환 전 최소 90%
  • 밀도: 전달 시점에 60-80%
// 세포 준비 프로토콜
interface CellPreparation {
    cellLine: string;            // 세포주
    strProfile: STRProfile;      // STR 프로파일
    mycoplasmaTest: {            // 마이코플라즈마 검사
        result: 'negative' | 'positive';
        date: Date;
        method: 'PCR' | 'culture' | 'luminescence';
    };
    passageNumber: number;       // 계대 번호
    viability: number;           // 생존율 (%)
    confluence: number;          // 밀도 (%)
    cultureConditions: {         // 배양 조건
        medium: string;          // 배지
        serum: string;           // 혈청
        serumPercentage: number; // 혈청 비율
        temperature: number;     // 온도 (섭씨)
        co2Percentage: number;   // CO2 비율
        humidity: number;        // 습도 (%)
    };
}

// 형질전환 프로토콜
interface TransfectionProtocol {
    method: 'lipofection' | 'electroporation' | 'nucleofection' | 'viral';
    reagent?: string;            // 시약
    dnaAmount: number;           // 마이크로그램
    rnaAmount?: number;          // 피코몰 (RNP용)
    proteinAmount?: number;      // 마이크로그램 (RNP용)
    cellNumber: number;          // 세포 수
    plateFormat: '6-well' | '12-well' | '24-well' | '96-well';
    incubationTime: number;      // 형질전환 후 시간
    mediaChangeTime: number;     // 배지 교체 시간
}

2.6 염기 편집 프로토콜

염기 편집기는 이중가닥 절단 없이 정밀한 단일 뉴클레오타이드 변화를 가능하게 합니다. WIA-BIO는 시토신 염기 편집기 (CBE)와 아데닌 염기 편집기 (ABE) 모두를 지원합니다: 

// 염기 편집기 구성
interface BaseEditorConfig {
    editorType: 'CBE' | 'ABE' | 'dual';
    version: string;            // 예: "BE4max", "ABE8e"

    // 편집 창 (PAM 원위 말단에서의 위치)
    editWindowStart: number;    // 일반적으로 4
    editWindowEnd: number;      // 일반적으로 8

    // 예상 전환
    conversion: {
        from: 'C' | 'A';        // 원래 염기
        to: 'T' | 'G';          // 변환된 염기
    };

    // 품질 메트릭
    editingEfficiency: number;  // 최소 20%
    bystanderEditing: number;   // 최대 30%
    indelFrequency: number;     // 최대 5%
}

// CBE 활성 계산
function calculateCBEActivity(
    targetSequence: string,
    editWindow: [number, number]
): CBEPrediction {
    const cytosines: number[] = [];
    for (let i = editWindow[0]; i <= editWindow[1]; i++) {
        if (targetSequence[i - 1] === 'C') {
            cytosines.push(i);
        }
    }

    return {
        editablePositions: cytosines,      // 편집 가능 위치
        predictedProducts: generateProducts(targetSequence, cytosines),
        bystanderRisk: cytosines.length > 1 ? 'HIGH' : 'LOW',
        recommendation: cytosines.length === 1 ? 'PROCEED' : 'REVIEW'
    };
}

2.7 프라임 편집 표준

프라임 편집은 이중가닥 절단 없이 모든 유형의 정밀 편집을 가능하게 합니다. WIA-BIO는 pegRNA (프라임 편집 가이드 RNA)에 대한 설계 표준을 수립합니다: 

// 프라임 편집 가이드 RNA 설계
interface PegRNADesign {
    spacerSequence: string;     // 20-nt 스페이서
    scaffoldSequence: string;   // tracrRNA 스캐폴드
    pbs: {                      // 프라이머 결합 부위
        sequence: string;
        length: number;         // 최적: 10-16 nt
        tm: number;             // 용융 온도
    };
    rtt: {                      // 역전사효소 템플릿
        sequence: string;
        length: number;         // 최적: 10-30 nt
        editType: EditType;     // 편집 유형
        editPosition: number;   // 편집 위치
    };
    nickingGuide?: {            // PE3/PE5 전략용
        sequence: string;
        distance: number;       // 주요 닉으로부터 거리
        strand: '+' | '-';
    };
}

enum EditType {
    SUBSTITUTION = "substitution",  // 치환
    INSERTION = "insertion",        // 삽입
    DELETION = "deletion",          // 결실
    COMPLEX = "complex"             // 복합
}

// PE 효율 예측
function predictPEEfficiency(pegRNA: PegRNADesign): PEPrediction {
    // PBS 점수 계산
    const pbsScore = calculatePBSScore(pegRNA.pbs);

    // RTT 점수 계산
    const rttScore = calculateRTTScore(pegRNA.rtt);

    // 전체 예측
    return {
        efficiency: pbsScore * rttScore * 0.8,
        pbsOptimal: pegRNA.pbs.length >= 12 && pegRNA.pbs.length <= 14,
        rttOptimal: pegRNA.rtt.length >= 12 && pegRNA.rtt.length <= 20,
        recommendation: generateRecommendation(pbsScore, rttScore)
    };
}

2.8 검증 및 문서화

모든 유전자 편집 실험은 포괄적인 검증 및 문서화를 포함해야 합니다:

필수 문서화 요구사항

  • 생거 시퀀싱: 클론 집단에서 편집 확인
  • NGS 검증: 표적 부위에서 편집 효율 정량화
  • 오프타겟 스크리닝: 상위 50개 예측 부위 분석
  • 기능적 검증: 표현형 변화 확인
  • 핵형 분석: 치료 응용에 필수
  • 감사 추적: 모든 단계의 완전한 문서화

핵심 요약

  • 시스템 선택: PAM 가용성, 전달 제약 및 편집 유형에 따라 적절한 CRISPR 시스템 선택
  • gRNA 설계: 여러 알고리즘 사용 및 최소 온타겟 점수 0.5 요구
  • 오프타겟 분석: 계산적 예측 필수; 치료 응용에는 전장 유전체 방법 필요
  • 염기 편집: 점 돌연변이에 선호됨; 삽입결실 형성 최소화
  • 프라임 편집: 가장 다목적; 신중한 pegRNA 설계 최적화 필요
  • 검증: 다운스트림 응용 전 다단계 검증 필수
