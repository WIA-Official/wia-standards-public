2.3 Carbon Footprint Declaration

[EU] Performance Classes

  • Class A: < 50 kg CO2eq/kWh
  • Class B: 50-65 kg CO2eq/kWh
  • Class C: 65-80 kg CO2eq/kWh
  • Class D: > 80 kg CO2eq/kWh

2.4 Rust Type Definitions

types.rs
use serde::{Deserialize, Serialize};
use chrono::{DateTime, NaiveDate, Utc};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct BatteryPassport {
    pub id: String,
    pub battery_id: String,
    pub category: BatteryCategory,
    pub manufacturer: Manufacturer,
    pub model: String,
    pub chemistry: Chemistry,
    pub capacity: Capacity,
    pub carbon_footprint: CarbonFootprint,
    pub recycled_content: RecycledContent,
    pub state_of_health: Option<f32>,
    pub manufacturing_date: NaiveDate,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
pub enum BatteryCategory {
    EvBattery,
    IndustrialBattery,
    LmtBattery,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum Chemistry {
    NMC811,
    NMC622,
    LFP,
    NCA,
    SolidState,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct CarbonFootprint {
    pub value: f32,
    pub unit: String,
    pub performance_class: PerformanceClass,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum PerformanceClass { A, B, C, D }
