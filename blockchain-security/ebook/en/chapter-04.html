Chapter 4: DeFi Security

Chapter Overview: Decentralized Finance (DeFi) introduces unique security challenges beyond traditional smart contract vulnerabilities. This chapter covers flash loan attacks, MEV mitigation, oracle security, and liquidity pool protection strategies.

Flash Loan Attack Vectors

Flash loans enable borrowing large amounts of capital without collateral, provided the loan is repaid within the same transaction. While legitimate use cases exist, flash loans have been exploited in numerous attacks totaling billions in losses.

Flash Loan Attack Anatomy

Flash Loan Attack Flow:

1. Attacker initiates flash loan
   └── Borrows $100M from lending protocol

2. Manipulate market conditions
   ├── Swap large amounts on DEX A (price impact)
   ├── Trigger liquidations on lending protocol
   └── Exploit price discrepancy on DEX B

3. Extract profit
   ├── Sell tokens at manipulated prices
   └── Collect liquidation bonuses

4. Repay flash loan + fee
   └── Return $100M + 0.09% fee

5. Keep profit
   └── Attacker nets $5M in profit

All steps executed in single atomic transaction!

Flash Loan Protection

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

/**
 * @title FlashLoanGuard
 * @notice Protection mechanisms against flash loan attacks
 */
abstract contract FlashLoanGuard {
    // Track block numbers for each address
    mapping(address => uint256) private _lastActionBlock;

    // Minimum blocks between sensitive actions
    uint256 public constant MIN_BLOCK_DELAY = 1;

    /**
     * @notice Prevents same-block interactions
     * @dev Flash loans must complete in one block, so this breaks attack atomicity
     */
    modifier noFlashLoan() {
        require(
            _lastActionBlock[msg.sender] < block.number,
            "FlashLoanGuard: same-block interaction"
        );
        _lastActionBlock[msg.sender] = block.number;
        _;
    }

    /**
     * @notice Require minimum blocks between actions
     */
    modifier delayedAction(uint256 minBlocks) {
        require(
            block.number >= _lastActionBlock[msg.sender] + minBlocks,
            "FlashLoanGuard: action too soon"
        );
        _lastActionBlock[msg.sender] = block.number;
        _;
    }
}

/**
 * @title SecureLending
 * @notice Lending protocol with flash loan protections
 */
contract SecureLending is FlashLoanGuard {
    mapping(address => uint256) public deposits;
    mapping(address => uint256) public borrows;

    // Collateralization ratio (150%)
    uint256 public constant COLLATERAL_RATIO = 150;

    function deposit() external payable noFlashLoan {
        deposits[msg.sender] += msg.value;
    }

    function borrow(uint256 amount) external noFlashLoan {
        uint256 maxBorrow = (deposits[msg.sender] * 100) / COLLATERAL_RATIO;
        require(borrows[msg.sender] + amount <= maxBorrow, "Insufficient collateral");

        borrows[msg.sender] += amount;
        payable(msg.sender).transfer(amount);
    }

    function liquidate(address user) external delayedAction(MIN_BLOCK_DELAY) {
        // Use TWAP price instead of spot price
        uint256 twapPrice = _getTWAPPrice();

        uint256 collateralValue = deposits[user] * twapPrice / 1e18;
        uint256 borrowValue = borrows[user];

        require(
            collateralValue * 100 < borrowValue * COLLATERAL_RATIO,
            "Position is healthy"
        );

        // Perform liquidation with delay-verified price
        _executeLiquidation(user);
    }
}

MEV (Maximal Extractable Value) Mitigation

MEV refers to the profit that miners/validators can extract by reordering, inserting, or censoring transactions. Front-running, sandwich attacks, and back-running are common MEV strategies.

MEV Impact: MEV extraction costs DeFi users an estimated $500-700M annually. Effective mitigation strategies are essential for protecting user value.

Common MEV Attack Types

Attack TypeDescriptionVictim Impact
Front-runningAttacker places transaction before victimWorse execution price
Sandwich AttackAttacker surrounds victim transactionMaximum price impact
Back-runningAttacker places transaction after victimLost arbitrage opportunity
Liquidation SnipingCompete to execute liquidationsHigher gas costs
Time-bandit AttackReorg to steal profitable transactionsTransaction reversal

MEV Protection Implementation

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

/**
 * @title MEVProtectedSwap
 * @notice DEX swap with MEV protection mechanisms
 */
contract MEVProtectedSwap {
    // Commit-reveal scheme storage
    struct SwapCommitment {
        bytes32 commitHash;
        uint256 commitBlock;
        bool revealed;
    }

    mapping(address => SwapCommitment) public commitments;
    uint256 public constant REVEAL_DELAY = 2; // blocks
    uint256 public constant COMMIT_EXPIRY = 20; // blocks

    /**
     * @notice Phase 1: Commit to a swap without revealing parameters
     * @param commitHash Hash of (tokenIn, tokenOut, amountIn, minAmountOut, salt)
     */
    function commitSwap(bytes32 commitHash) external {
        require(
            commitments[msg.sender].commitBlock == 0 ||
            block.number > commitments[msg.sender].commitBlock + COMMIT_EXPIRY,
            "Existing commitment"
        );

        commitments[msg.sender] = SwapCommitment({
            commitHash: commitHash,
            commitBlock: block.number,
            revealed: false
        });
    }

    /**
     * @notice Phase 2: Reveal and execute swap after delay
     */
    function revealSwap(
        address tokenIn,
        address tokenOut,
        uint256 amountIn,
        uint256 minAmountOut,
        bytes32 salt
    ) external returns (uint256 amountOut) {
        SwapCommitment storage commitment = commitments[msg.sender];

        // Verify commitment exists and delay passed
        require(commitment.commitBlock > 0, "No commitment");
        require(!commitment.revealed, "Already revealed");
        require(
            block.number >= commitment.commitBlock + REVEAL_DELAY,
            "Reveal too early"
        );
        require(
            block.number <= commitment.commitBlock + COMMIT_EXPIRY,
            "Commitment expired"
        );

        // Verify commitment hash
        bytes32 expectedHash = keccak256(
            abi.encodePacked(tokenIn, tokenOut, amountIn, minAmountOut, salt)
        );
        require(commitment.commitHash == expectedHash, "Invalid reveal");

        commitment.revealed = true;

        // Execute swap with slippage protection
        amountOut = _executeSwap(tokenIn, tokenOut, amountIn, minAmountOut);

        return amountOut;
    }

    /**
     * @notice Calculate maximum slippage allowed based on pool state
     */
    function calculateSafeSlippage(
        address pool,
        uint256 amountIn
    ) public view returns (uint256 maxSlippageBps) {
        uint256 poolLiquidity = _getPoolLiquidity(pool);
        uint256 impactRatio = (amountIn * 10000) / poolLiquidity;

        // Dynamic slippage based on trade size relative to pool
        if (impactRatio < 10) { // < 0.1% of pool
            return 50; // 0.5% slippage
        } else if (impactRatio < 100) { // < 1% of pool
            return 100; // 1% slippage
        } else {
            return 300; // 3% slippage for large trades
        }
    }
}

Oracle Security

Oracles provide external data to smart contracts. Compromised or manipulated oracles have been the root cause of many DeFi exploits.

Oracle Manipulation Vectors

Oracle Attack Scenarios:

1. Spot Price Manipulation
   └── Attacker moves DEX price with large swap
   └── Oracle reads manipulated spot price
   └── Attacker exploits mispriced collateral

2. Low Liquidity Oracle
   └── Oracle sources from thin liquidity pool
   └── Small capital moves price significantly
   └── Cascade effect on dependent protocols

3. Delayed Oracle Updates
   └── Oracle price lags real market
   └── Attacker exploits stale prices
   └── Arbitrage between oracle and market

4. Centralized Oracle Compromise
   └── Single oracle operator compromised
   └── False data submitted to blockchain
   └── Protocols act on incorrect data

Secure Oracle Implementation

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

import "@chainlink/contracts/src/v0.8/interfaces/AggregatorV3Interface.sol";

contract SecureOracle {
    mapping(address => AggregatorV3Interface[]) public priceFeeds;
    uint256 public constant MIN_ORACLE_SOURCES = 3;

    function getSecurePrice(address token) external view returns (uint256 price) {
        AggregatorV3Interface[] storage feeds = priceFeeds[token];
        require(feeds.length >= MIN_ORACLE_SOURCES, "Insufficient sources");

        uint256[] memory prices = new uint256[](feeds.length);
        uint256 validPrices = 0;

        for (uint256 i = 0; i < feeds.length; i++) {
            (, int256 answer,, uint256 updatedAt,) = feeds[i].latestRoundData();
            if (block.timestamp - updatedAt < 1 hours && answer > 0) {
                prices[validPrices++] = uint256(answer);
            }
        }
        require(validPrices >= 2, "Insufficient valid prices");
        return _calculateMedian(prices, validPrices);
    }

    function getTWAP(address pool, uint32 period) external view returns (uint256) {
        (int56 tickStart, int56 tickEnd) = _getTickCumulatives(pool, period);
        int24 avgTick = int24((tickEnd - tickStart) / int56(uint56(period)));
        return _tickToPrice(avgTick);
    }
}

Liquidity Pool Security

Liquidity pools are central to DeFi but present unique security challenges including impermanent loss, rug pulls, and pool manipulation.

Rug Pull Prevention

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

/**
 * @title SecureLiquidityPool
 * @notice LP with rug-pull prevention mechanisms
 */
contract SecureLiquidityPool {
    address public owner;
    uint256 public constant LOCK_PERIOD = 180 days;
    uint256 public poolCreatedAt;

    mapping(address => LPPosition) public positions;

    struct LPPosition {
        uint256 liquidity;
        uint256 depositedAt;
        uint256 withdrawableAt;
    }

    // Maximum single withdrawal as percentage of pool
    uint256 public constant MAX_WITHDRAWAL_PERCENT = 10;

    // Cooldown between large withdrawals
    uint256 public lastLargeWithdrawal;
    uint256 public constant WITHDRAWAL_COOLDOWN = 24 hours;

    modifier liquidityLocked() {
        require(
            block.timestamp >= poolCreatedAt + LOCK_PERIOD,
            "Liquidity locked"
        );
        _;
    }

    function addLiquidity(uint256 amount) external {
        // Transfer tokens and mint LP
        uint256 lpAmount = _calculateLPTokens(amount);

        positions[msg.sender] = LPPosition({
            liquidity: positions[msg.sender].liquidity + lpAmount,
            depositedAt: block.timestamp,
            withdrawableAt: block.timestamp + 7 days // Min lock
        });
    }

    function removeLiquidity(
        uint256 lpAmount
    ) external liquidityLocked returns (uint256 amountOut) {
        LPPosition storage position = positions[msg.sender];

        require(position.liquidity >= lpAmount, "Insufficient LP");
        require(
            block.timestamp >= position.withdrawableAt,
            "Position locked"
        );

        uint256 totalLiquidity = _getTotalLiquidity();
        uint256 withdrawPercent = (lpAmount * 100) / totalLiquidity;

        // Large withdrawal checks
        if (withdrawPercent > MAX_WITHDRAWAL_PERCENT) {
            require(
                block.timestamp >= lastLargeWithdrawal + WITHDRAWAL_COOLDOWN,
                "Withdrawal cooldown active"
            );
            lastLargeWithdrawal = block.timestamp;

            // Cap at maximum
            lpAmount = (totalLiquidity * MAX_WITHDRAWAL_PERCENT) / 100;
        }

        position.liquidity -= lpAmount;
        amountOut = _calculateTokensOut(lpAmount);

        // Transfer tokens
        _transferOut(msg.sender, amountOut);

        return amountOut;
    }
}
