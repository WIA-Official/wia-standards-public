Chapter 3: Vulnerability Detection and Auditing

Chapter Overview: Proactive security requires systematic vulnerability detection and thorough auditing processes. This chapter covers methodologies, tools, and best practices for identifying security issues before they can be exploited.

Common Smart Contract Vulnerabilities

Understanding common vulnerability patterns is the first step in effective detection. Smart contracts have unique vulnerability classes that differ from traditional software.

Reentrancy Attacks

Reentrancy remains one of the most dangerous vulnerabilities in smart contracts. It occurs when an external call allows the called contract to re-enter the calling function before state updates are complete.

// VULNERABLE: Reentrancy Attack Example
pragma solidity ^0.8.19;

contract VulnerableBank {
    mapping(address => uint256) public balances;

    function deposit() external payable {
        balances[msg.sender] += msg.value;
    }

    // VULNERABLE: State update after external call
    function withdraw(uint256 amount) external {
        require(balances[msg.sender] >= amount, "Insufficient balance");

        // External call before state update - VULNERABLE
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");

        // State update after external call - TOO LATE
        balances[msg.sender] -= amount;
    }
}

// ATTACK CONTRACT
contract ReentrancyAttacker {
    VulnerableBank public target;
    uint256 public attackCount;

    constructor(address _target) {
        target = VulnerableBank(_target);
    }

    function attack() external payable {
        target.deposit{value: msg.value}();
        target.withdraw(msg.value);
    }

    receive() external payable {
        if (address(target).balance >= msg.value && attackCount < 10) {
            attackCount++;
            target.withdraw(msg.value);
        }
    }
}

Secure Implementation

// SECURE: Reentrancy-Safe Implementation
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/security/ReentrancyGuard.sol";

contract SecureBank is ReentrancyGuard {
    mapping(address => uint256) public balances;

    function deposit() external payable {
        balances[msg.sender] += msg.value;
    }

    // SECURE: Uses checks-effects-interactions pattern
    function withdraw(uint256 amount) external nonReentrant {
        // Checks
        require(balances[msg.sender] >= amount, "Insufficient balance");

        // Effects - State update BEFORE external call
        balances[msg.sender] -= amount;

        // Interactions - External call LAST
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");
    }
}

Vulnerability Scanner Implementation

// Automated Vulnerability Scanner
import { ethers } from 'ethers';

interface VulnerabilityScan {
    contract: string;
    findings: Finding[];
    riskScore: number;
    timestamp: Date;
}

interface Finding {
    id: string;
    severity: 'critical' | 'high' | 'medium' | 'low' | 'info';
    title: string;
    description: string;
    location: CodeLocation;
    recommendation: string;
}

class VulnerabilityScanner {
    private patterns: VulnerabilityPattern[];

    constructor() {
        this.patterns = this.loadPatterns();
    }

    async scanContract(bytecode: string, abi: any[]): Promise {
        const findings: Finding[] = [];

        // Check for known vulnerable patterns
        findings.push(...this.checkReentrancy(bytecode, abi));
        findings.push(...this.checkAccessControl(abi));
        findings.push(...this.checkDelegatecall(bytecode));
        findings.push(...this.checkSelfDestruct(bytecode));
        findings.push(...this.checkTxOrigin(bytecode));

        // Calculate risk score
        const riskScore = this.calculateRiskScore(findings);

        return {
            contract: ethers.utils.keccak256(bytecode),
            findings,
            riskScore,
            timestamp: new Date()
        };
    }

    private checkReentrancy(bytecode: string, abi: any[]): Finding[] {
        const findings: Finding[] = [];

        // Check for CALL followed by SSTORE pattern
        const opcodes = this.disassemble(bytecode);

        for (let i = 0; i < opcodes.length - 1; i++) {
            if (opcodes[i].opcode === 'CALL') {
                // Look for SSTORE after CALL in same function
                for (let j = i + 1; j < Math.min(i + 50, opcodes.length); j++) {
                    if (opcodes[j].opcode === 'SSTORE') {
                        findings.push({
                            id: `REEN-${i}`,
                            severity: 'critical',
                            title: 'Potential Reentrancy Vulnerability',
                            description: 'State modification detected after external call',
                            location: { offset: opcodes[i].offset },
                            recommendation: 'Apply checks-effects-interactions pattern'
                        });
                        break;
                    }
                    if (opcodes[j].opcode === 'RETURN' || opcodes[j].opcode === 'STOP') {
                        break;
                    }
                }
            }
        }

        return findings;
    }

    private checkAccessControl(abi: any[]): Finding[] {
        const findings: Finding[] = [];
        const sensitivePatterns = ['withdraw', 'transfer', 'mint', 'burn', 'pause', 'upgrade'];

        for (const item of abi) {
            if (item.type !== 'function') continue;

            const isSensitive = sensitivePatterns.some(p =>
                item.name.toLowerCase().includes(p)
            );

            if (isSensitive && !this.hasAccessModifier(item)) {
                findings.push({
                    id: `AC-${item.name}`,
                    severity: 'high',
                    title: 'Missing Access Control',
                    description: `Sensitive function ${item.name} may lack access control`,
                    location: { functionName: item.name },
                    recommendation: 'Add onlyOwner or role-based access control'
                });
            }
        }

        return findings;
    }

    private calculateRiskScore(findings: Finding[]): number {
        const weights = { critical: 100, high: 50, medium: 20, low: 5, info: 1 };
        return findings.reduce((score, f) => score + weights[f.severity], 0);
    }
}

Security Audit Process

A comprehensive security audit follows a structured methodology to ensure thorough coverage of all potential vulnerabilities.

Audit Phases: A complete security audit consists of multiple phases, each with specific objectives and deliverables. Following this structured approach ensures comprehensive coverage and actionable results.

Audit Workflow

Security Audit Process Flow:

Phase 1: Preparation (2-3 days)
├── Receive codebase and documentation
├── Define scope and objectives
├── Set up development environment
├── Review previous audits (if any)
└── Establish communication channels

Phase 2: Automated Analysis (1-2 days)
├── Run static analysis tools (Slither, Mythril)
├── Execute symbolic execution (Manticore)
├── Perform fuzzing campaigns
├── Generate initial findings list
└── Prioritize areas for manual review

Phase 3: Manual Review (5-10 days)
├── Architecture review
├── Business logic analysis
├── Access control verification
├── State machine validation
├── External dependency review
├── Gas optimization analysis
└── Code quality assessment

Phase 4: Attack Simulation (2-3 days)
├── Develop proof-of-concept exploits
├── Test on forked mainnet
├── Verify attack scenarios
├── Document attack paths
└── Estimate potential impact

Phase 5: Reporting (2-3 days)
├── Compile findings with severity ratings
├── Provide remediation recommendations
├── Create executive summary
├── Review with development team
└── Deliver final report

Phase 6: Remediation Verification (2-3 days)
├── Review implemented fixes
├── Verify no new issues introduced
├── Re-run automated tools
├── Issue remediation report
└── Final sign-off

Penetration Testing Techniques

Penetration testing goes beyond static analysis to actively attempt exploitation of vulnerabilities in a controlled environment.

Fuzzing Framework

// Smart Contract Fuzzing Framework
interface FuzzConfig {
    iterations: number;
    seed: number;
    invariants: InvariantCheck[];
}

class ContractFuzzer {
    private contract: ethers.Contract;
    private config: FuzzConfig;
    private violations: InvariantViolation[] = [];

    async fuzz(): Promise {
        const rng = this.createRNG(this.config.seed);

        for (let i = 0; i < this.config.iterations; i++) {
            const call = this.generateRandomCall(rng);

            try {
                await this.executeCall(call);
                const state = await this.captureState();

                for (const invariant of this.config.invariants) {
                    if (!invariant.check(state)) {
                        this.violations.push({ iteration: i, call, invariant: invariant.name });
                    }
                }
            } catch (error) {
                if (!this.isExpectedError(error)) {
                    this.violations.push({ iteration: i, call, error: error.message });
                }
            }
        }
        return { iterations: this.config.iterations, violations: this.violations };
    }
}

// Run fuzzing with invariants
const fuzzer = new ContractFuzzer(tokenContract, {
    iterations: 10000,
    seed: Date.now(),

        { name: 'Total supply equals sum of balances', check: (s) => s.totalSupply.eq(sumBalances(s)) },
        { name: 'No balance exceeds total supply', check: (s) => allBalancesValid(s) }
    ]
});
