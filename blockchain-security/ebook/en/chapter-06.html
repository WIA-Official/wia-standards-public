Chapter Overview: This chapter covers the integration of security tools and APIs into blockchain applications. Learn how to design security monitoring interfaces, implement incident response automation, and integrate with existing security infrastructure.

Security API Architecture

Well-designed security APIs provide programmatic access to security functions while maintaining strict access controls and comprehensive audit logging.

API Design Principles

Security API Design Principles:

1. Authentication & Authorization
   ├── API key authentication for services
   ├── OAuth 2.0 for user-facing applications
   ├── mTLS for service-to-service communication
   └── Role-based access control (RBAC)

2. Input Validation
   ├── Schema validation (JSON Schema, OpenAPI)
   ├── Parameter sanitization
   ├── Request size limits
   └── Rate limiting by client/endpoint

3. Secure Communication
   ├── TLS 1.3 mandatory
   ├── Certificate pinning for mobile clients
   ├── Request signing for critical operations
   └── Replay attack prevention (nonces)

4. Audit & Monitoring
   ├── Log all API requests
   ├── Track request origin and authentication
   ├── Monitor for anomalous patterns
   └── Alert on security-relevant events

5. Error Handling
   ├── Generic error messages (no info leakage)
   ├── Detailed logging internally
   ├── Consistent error format
   └── Graceful degradation

Security Monitoring API

// Security Monitoring API Implementation
import express from 'express';
import { authenticateApiKey, authorize } from './middleware/auth';
import { validateRequest } from './middleware/validation';
import { rateLimiter } from './middleware/rateLimit';

const router = express.Router();

// OpenAPI Schema for transaction monitoring
const monitorTransactionSchema = {
    type: 'object',
    required: ['transactionHash', 'chainId'],
    properties: {
        transactionHash: {
            type: 'string',
            pattern: '^0x[a-fA-F0-9]{64}$'
        },
        chainId: {
            type: 'integer',
            minimum: 1
        },
        riskThreshold: {
            type: 'number',
            minimum: 0,
            maximum: 100,
            default: 50
        }
    }
};

/**
 * @route POST /api/v1/monitor/transaction
 * @description Analyze transaction for security risks
 */
router.post(
    '/monitor/transaction',
    authenticateApiKey,
    authorize(['security:read']),
    rateLimiter({ windowMs: 60000, max: 100 }),
    validateRequest(monitorTransactionSchema),
    async (req, res, next) => {
        try {
            const { transactionHash, chainId, riskThreshold } = req.body;

            // Analyze transaction
            const analysis = await transactionAnalyzer.analyze({
                hash: transactionHash,
                chainId,
                options: {
                    checkKnownExploits: true,
                    checkSanctionedAddresses: true,
                    checkFlashLoan: true,
                    checkReentrancy: true
                }
            });

            // Generate risk score
            const riskScore = calculateRiskScore(analysis);

            // Log analysis for audit
            await auditLog.record({
                action: 'TRANSACTION_ANALYSIS',
                requestId: req.id,
                clientId: req.apiKey.clientId,
                transactionHash,
                riskScore,
                timestamp: Date.now()
            });

            // Alert if above threshold
            if (riskScore >= riskThreshold) {
                await alertService.trigger({
                    type: 'HIGH_RISK_TRANSACTION',
                    severity: riskScore >= 80 ? 'critical' : 'warning',
                    data: { transactionHash, riskScore, analysis }
                });
            }

            res.json({
                success: true,
                data: {
                    transactionHash,
                    riskScore,
                    riskLevel: getRiskLevel(riskScore),
                    findings: analysis.findings,
                    recommendations: analysis.recommendations
                }
            });
        } catch (error) {
            next(error);
        }
    }
);

/**
 * @route POST /api/v1/monitor/address
 * @description Screen address against security databases
 */
router.post(
    '/monitor/address',
    authenticateApiKey,
    authorize(['security:read']),
    rateLimiter({ windowMs: 60000, max: 200 }),
    async (req, res, next) => {
        const { address, chainId } = req.body;

        const screening = await addressScreener.screen(address, {
            checkSanctions: true,
            checkMixers: true,
            checkDarknet: true,
            checkScamDb: true,
            chainId
        });

        res.json({
            success: true,
            data: {
                address,
                risk: screening.riskLevel,
                flags: screening.flags,
                labels: screening.labels,
                lastActivity: screening.lastActivity
            }
        });
    }
);

export default router;

Incident Response Automation

Automated incident response enables rapid reaction to security threats, reducing the window of opportunity for attackers.

Response Time Matters: Studies show that automated incident response can reduce breach impact by up to 95% compared to manual response. Every second counts when dealing with blockchain exploits where transactions are irreversible.

Incident Response Framework

// Automated Incident Response System
interface SecurityIncident {
    id: string;
    type: IncidentType;
    severity: 'critical' | 'high' | 'medium' | 'low';
    source: string;
    timestamp: number;
    data: any;
    status: 'detected' | 'analyzing' | 'responding' | 'resolved';
}

type IncidentType =
    | 'EXPLOIT_DETECTED'
    | 'ANOMALOUS_TRANSACTION'
    | 'KEY_COMPROMISE'
    | 'DOS_ATTACK'
    | 'ORACLE_MANIPULATION'
    | 'BRIDGE_ATTACK';

class IncidentResponseEngine {
    private handlers: Map = new Map();
    private escalationRules: EscalationRule[];

    constructor(config: ResponseConfig) {
        this.initializeHandlers(config);
        this.escalationRules = config.escalationRules;
    }

    async handleIncident(incident: SecurityIncident): Promise {
        // Log incident detection
        await this.logIncident(incident);

        // Determine response handlers
        const handlers = this.handlers.get(incident.type) || [];

        // Execute automated responses
        const responses: ResponseAction[] = [];

        for (const handler of handlers) {
            if (this.shouldExecute(handler, incident)) {
                try {
                    const action = await handler.execute(incident);
                    responses.push(action);

                    // Log action
                    await this.logAction(incident.id, action);
                } catch (error) {
                    await this.logError(incident.id, handler.name, error);
                }
            }
        }

        // Check escalation rules
        await this.checkEscalation(incident, responses);

        return {
            incidentId: incident.id,
            responses,
            status: 'handled',
            timestamp: Date.now()
        };
    }

    private initializeHandlers(config: ResponseConfig): void {
        // Exploit detection handlers
        this.handlers.set('EXPLOIT_DETECTED', [
            new PauseContractHandler(),
            new AlertSecurityTeamHandler(),
            new BlockAttackerAddressHandler(),
            new PreserveEvidenceHandler()
        ]);

        // Key compromise handlers
        this.handlers.set('KEY_COMPROMISE', [
            new RevokeKeyHandler(),
            new RotateSecretsHandler(),
            new NotifyAffectedUsersHandler(),
            new ForensicAnalysisHandler()
        ]);

        // Oracle manipulation handlers
        this.handlers.set('ORACLE_MANIPULATION', [
            new SwitchToBackupOracleHandler(),
            new PausePriceUpdatesHandler(),
            new AlertProtocolOwnersHandler()
        ]);
    }
}

// Example: Pause Contract Handler
class PauseContractHandler implements IncidentHandler {
    name = 'PauseContract';

    async execute(incident: SecurityIncident): Promise {
        const contract = await this.getContract(incident.data.contractAddress);

        // Check if pausable
        if (!await this.isPausable(contract)) {
            return {
                handler: this.name,
                success: false,
                reason: 'Contract not pausable'
            };
        }

        // Execute pause via multi-sig or emergency key
        const tx = await this.emergencyPause(contract);

        return {
            handler: this.name,
            success: true,
            transactionHash: tx.hash,
            action: 'Contract paused',
            timestamp: Date.now()
        };
    }
}

Threat Intelligence Integration

Integrating threat intelligence feeds enables proactive defense by identifying known malicious addresses, attack patterns, and emerging threats.

Threat Intelligence Pipeline

// Threat Intelligence Integration
interface ThreatIntelFeed {
    name: string;
    type: 'sanctions' | 'exploits' | 'malware' | 'phishing' | 'mixer';
    endpoint: string;
}

class ThreatIntelligenceService {
    private feeds: ThreatIntelFeed[];
    private cache: Map = new Map();

    async checkAddress(address: string): Promise {
        const findings: ThreatFinding[] = [];

        for (const feed of this.feeds) {
            const feedData = this.cache.get(feed.name);
            const match = feedData?.addresses.get(address.toLowerCase());
            if (match) {
                findings.push({ source: feed.name, type: feed.type, confidence: match.confidence });
            }
        }

        return {
            address,
            riskScore: this.calculateRiskScore(findings),
            findings,
            checkedAt: Date.now()
        };
    }

    private calculateRiskScore(findings: ThreatFinding[]): number {
        const weights = { sanctions: 100, exploits: 90, malware: 85, mixer: 70, phishing: 60 };
        if (findings.length === 0) return 0;
        let total = findings.reduce((sum, f) => sum + (weights[f.type] || 50) * f.confidence, 0);
        return Math.min(100, total / findings.length);
    }
}

// Security Dashboard WebSocket API
import { WebSocketServer } from 'ws';

class SecurityDashboardServer {
    private wss: WebSocketServer;
    private clients: Map = new Map();

    constructor(port: number) {
        this.wss = new WebSocketServer({ port });
        this.wss.on('connection', (ws, req) => {
            const clientId = this.authenticateClient(req);
            if (!clientId) { ws.close(4001, 'Auth failed'); return; }

            this.clients.set(clientId, { id: clientId, ws, subscriptions: new Set() });
            ws.on('close', () => this.clients.delete(clientId));
        });

        securityEventEmitter.on('incident', (incident) => this.broadcast('incident', incident));
        securityEventEmitter.on('alert', (alert) => this.broadcast('alert', alert));
    }

    private broadcast(event: string, data: any): void {
        for (const [_, client] of this.clients) {
            client.ws.send(JSON.stringify({ event, data, timestamp: Date.now() }));
        }
    }
}
