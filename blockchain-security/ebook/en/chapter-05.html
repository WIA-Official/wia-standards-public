Secure Multi-Sig Implementation

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

contract SecureMultiSig {
    event TransactionProposed(uint256 indexed txId, address indexed proposer);
    event TransactionExecuted(uint256 indexed txId);

    uint256 public constant EXECUTION_DELAY = 24 hours;
    address[] public signers;
    mapping(address => bool) public isSigner;
    uint256 public threshold;
    uint256 public nonce;

    struct Transaction {
        address to;
        uint256 value;
        bytes data;
        uint256 proposedAt;
        uint256 signatureCount;
        bool executed;
    }

    mapping(uint256 => Transaction) public transactions;

    modifier onlySigner() { require(isSigner[msg.sender], "Not a signer"); _; }

    constructor(address[] memory _signers, uint256 _threshold) {
        require(_signers.length >= _threshold && _threshold > 0);
        for (uint256 i = 0; i < _signers.length; i++) {
            isSigner[_signers[i]] = true;
            signers.push(_signers[i]);
        }
        threshold = _threshold;
    }

    function proposeTransaction(address to, uint256 value, bytes calldata data)
        external onlySigner returns (uint256 txId) {
        txId = nonce++;
        transactions[txId] = Transaction(to, value, data, block.timestamp, 1, false);
        emit TransactionProposed(txId, msg.sender);
    }

    function executeTransaction(uint256 txId) external onlySigner {
        Transaction storage txn = transactions[txId];
        require(!txn.executed && txn.signatureCount >= threshold);
        require(block.timestamp >= txn.proposedAt + EXECUTION_DELAY);
        txn.executed = true;
        (bool success, ) = txn.to.call{value: txn.value}(txn.data);
        require(success);
        emit TransactionExecuted(txId);
    }

    receive() external payable {}
}

Cold Storage Architecture

Cold storage keeps private keys completely offline, providing maximum security against remote attacks. Proper cold storage implementation requires careful protocol design.

Cold Storage Principle: Private keys in cold storage should never be exposed to internet-connected devices. All transaction signing occurs on air-gapped systems, with signed transactions transferred via secure methods.

Cold Storage Protocol

Cold Storage Transaction Protocol:

Step 1: Transaction Preparation (Online)
├── Create unsigned transaction on hot system
├── Verify destination and amount
├── Generate transaction hash for signing
└── Export to QR code or USB device

Step 2: Air-Gap Transfer
├── Verify device has never been connected to network
├── Transfer unsigned transaction via:
│   ├── QR code scanning (preferred)
│   ├── USB device (single-use, formatted after)
│   └── Manual entry (for simple transactions)
└── Verify transaction details on cold device

Step 3: Offline Signing
├── Review transaction on cold storage device
├── Verify all transaction parameters
├── Enter PIN/passphrase to unlock signing key
├── Sign transaction
└── Export signed transaction

Step 4: Broadcast (Online)
├── Transfer signed transaction to hot system
├── Final verification of signed transaction
├── Broadcast to blockchain network
└── Monitor for confirmation

Security Measures:
├── Cold storage room with physical access control
├── Faraday cage to prevent wireless signals
├── Multiple witness requirement for signing
├── Video recording of signing ceremonies
└── Hardware wallet with tamper-evident seals

Hardware Security Module (HSM) Integration

HSMs provide the highest level of key protection for institutional use, offering tamper resistance, audit logging, and cryptographic acceleration.

HSM Configuration

// HSM Integration for Blockchain Signing
import { HSMClient, KeyType, SignatureAlgorithm } from '@hsm/client';

interface HSMConfig {
    endpoint: string;
    partition: string;
    credentials: { type: 'certificate' | 'password'; value: string; };
    keyPolicy: { algorithm: string; quorumSize: number; };
}

class BlockchainHSMSigner {
    private client: HSMClient;

    constructor(config: HSMConfig) {
        this.client = new HSMClient(config);
    }

    async initialize(): Promise {
        await this.client.connect();
        const status = await this.client.getStatus();
        if (status.tamperStatus !== 'CLEAR') throw new Error('HSM tamper detected');
    }

    async generateKey(keyId: string): Promise {
        const keyHandle = await this.client.generateKey({
            id: keyId, type: KeyType.ECDSA, curve: 'secp256k1'
        });
        return await this.client.getPublicKey(keyHandle);
    }

    async signTransaction(keyId: string, txHash: Buffer, auth: AuthToken[]): Promise {
        if (auth.length < this.config.keyPolicy.quorumSize) {
            throw new Error('Insufficient authorization');
        }
        const sig = await this.client.sign({ keyId, data: txHash, authorization: auth });
        return { r: sig.r, s: sig.s, v: sig.recoveryParam };
    }
}

Key Lifecycle Management

Proper key lifecycle management ensures keys remain secure throughout their useful life and are properly retired when no longer needed.

Key Lifecycle Stages:

1. Generation
   ├── Use cryptographically secure RNG
   ├── Generate in secure environment (HSM/air-gapped)
   ├── Implement key ceremony with witnesses
   └── Document generation parameters

2. Distribution
   ├── Secure key share distribution (Shamir)
   ├── Verify share integrity
   ├── Establish custody chain
   └── Document share holders

3. Storage
   ├── Encrypt keys at rest
   ├── Implement access controls
   ├── Regular backup verification
   └── Geographic distribution

4. Usage
   ├── Audit all key usage
   ├── Implement rate limiting
   ├── Monitor for anomalies
   └── Enforce authorization policies

5. Rotation
   ├── Regular rotation schedule
   ├── Emergency rotation capability
   ├── Overlap period for migrations
   └── Update all dependent systems

6. Retirement
   ├── Secure deletion from all systems
   ├── Revoke any remaining access
   ├── Archive audit logs
   └── Document destruction
Summary: Key storage and wallet security are fundamental to protecting blockchain assets. Multi-signature wallets eliminate single points of failure, cold storage provides maximum protection for long-term holdings, and HSMs offer institutional-grade security. Implementing proper key lifecycle management ensures security throughout the key's useful life.
