Chapter 2: Executor Designation

2.1 Executor Framework Overview

The WIA Digital Will Standard defines a comprehensive executor designation framework that enables secure, verifiable, and legally compliant administration of digital wills. Executors serve as trusted intermediaries who facilitate the transfer of digital assets according to the testator's wishes while maintaining the highest standards of security and accountability.

Unlike traditional estate executors who primarily handle physical assets and legal documentation, digital will executors must possess technical expertise in blockchain technology, cryptography, and digital asset management. The WIA standard provides role-based access control, multi-signature authorization, and comprehensive audit trails to ensure proper executor governance.

2.1.1 Executor Role Classification

The WIA standard defines multiple executor roles with distinct responsibilities and authorization levels:

Executor Role Responsibilities Authorization Level Required Certifications
Primary Executor Overall will administration, asset inventory management, beneficiary communication Full Access WIA-EXEC-L3, Legal Certification
Technical Executor Smart contract deployment, blockchain transaction execution, key management Technical Operations WIA-TECH-L2, Blockchain Certification
Legal Executor Jurisdictional compliance verification, legal document preparation, court representation Legal Operations WIA-LEGAL-L2, Bar Association License
Financial Executor Asset valuation, tax calculation, financial reporting, distribution coordination Financial Operations WIA-FIN-L2, CPA/CFP Certification
Custodial Executor Secure storage of encrypted keys, emergency access provision, backup management Custody Operations WIA-CUST-L3, SOC 2 Type II
Verification Executor Identity verification, trigger event confirmation, death certificate validation Verification Only WIA-VERIFY-L1, KYC Certification
Backup Executor Successor to primary executor if unavailable or incapacitated Conditional Full Access WIA-EXEC-L2, Same as Primary

2.3 Multi-Signature Authorization Framework

Critical operations in digital will execution require multi-signature authorization to prevent unauthorized access and ensure proper oversight. The WIA standard implements flexible threshold signature schemes that balance security with operational efficiency.

Asset Transfer Execution 3 of 5 Primary, Technical, Legal Executors 72 hours Court Order Only
Beneficiary Modification 2 of 3 Primary, Legal Executors 168 hours (7 days) Testator Only
Emergency Access 4 of 7 All Executor Roles 24 hours No Override
Key Recovery 3 of 5 Technical, Custodial Executors 120 hours (5 days) Court Order Only

2.3.2 Multi-Signature Implementation

// Example 1: Multi-Signature Executor Authorization System
const { ethers } = require('ethers');
const { WIAExecutor } = require('@wia/digital-will-sdk');

class MultiSigExecutorFramework {
  constructor(contractAddress, provider) {
    this.contract = new ethers.Contract(
      contractAddress,
      this.getABI(),
      provider
    );
    this.pendingOperations = new Map();
  }

  async initiateOperation(operation) {
    const {
      operationType,
      targetAsset,
      parameters,
      initiatorDID,
      initiatorSignature
    } = operation;

    // Validate initiator authorization
    const isAuthorized = await this.verifyExecutorRole(
      initiatorDID,
      operationType
    );

    if (!isAuthorized) {
      throw new Error('Initiator not authorized for this operation type');
    }

    // Get signature requirements for this operation type
    const sigRequirements = this.getSignatureRequirements(operationType);

    // Create operation proposal
    const operationId = ethers.utils.keccak256(
      ethers.utils.defaultAbiCoder.encode(
        ['string', 'string', 'bytes', 'uint256'],
        [operationType, targetAsset, parameters, Date.now()]
      )
    );

    // Store operation proposal
    const tx = await this.contract.proposeOperation(
      operationId,
      operationType,
      targetAsset,
      parameters,
      sigRequirements.threshold,
      sigRequirements.timeDelay
    );

    await tx.wait();

    // Add initiator's signature
    await this.addSignature(operationId, initiatorDID, initiatorSignature);

    console.log(`Operation ${operationId} initiated by ${initiatorDID}`);
    console.log(`Requires ${sigRequirements.threshold} signatures`);

    return {
      operationId: operationId,
      status: 'PENDING',
      signaturesRequired: sigRequirements.threshold,
      signaturesReceived: 1,
      timeDelay: sigRequirements.timeDelay,
      executionTime: Date.now() + sigRequirements.timeDelay * 1000
    };
  }

  async addSignature(operationId, executorDID, signature) {
    // Verify executor authorization
    const operation = await this.contract.getOperation(operationId);
    const isAuthorized = await this.verifyExecutorRole(
      executorDID,
      operation.operationType
    );

    if (!isAuthorized) {
      throw new Error('Executor not authorized for this operation');
    }

    // Verify signature
    const messageHash = this.getOperationHash(operation);
    const recoveredAddress = ethers.utils.verifyMessage(messageHash, signature);
    const executorAddress = await this.resolveExecutorAddress(executorDID);

    if (recoveredAddress !== executorAddress) {
      throw new Error('Invalid signature');
    }

    // Add signature to contract
    const tx = await this.contract.addSignature(operationId, executorDID, signature);
    await tx.wait();

    // Check if threshold reached
    const signatureCount = await this.contract.getSignatureCount(operationId);
    const required = operation.signaturesRequired;

    console.log(`Signature added: ${signatureCount}/${required}`);

    if (signatureCount >= required) {
      console.log('Threshold reached. Operation ready for execution.');
      return { status: 'READY_FOR_EXECUTION', operationId: operationId };
    }

    return {
      status: 'PENDING',
      operationId: operationId,
      signaturesReceived: signatureCount,
      signaturesRequired: required
    };
  }

  async executeOperation(operationId) {
    const operation = await this.contract.getOperation(operationId);

    // Verify threshold reached
    const signatureCount = await this.contract.getSignatureCount(operationId);
    if (signatureCount < operation.signaturesRequired) {
      throw new Error('Insufficient signatures');
    }

    // Verify time delay passed
    const currentTime = Date.now();
    if (currentTime < operation.executionTime) {
      throw new Error(`Time delay not passed. Execution available at ${new Date(operation.executionTime)}`);
    }

    // Execute operation
    const tx = await this.contract.executeOperation(operationId);
    const receipt = await tx.wait();

    console.log(`Operation ${operationId} executed successfully`);
    console.log(`Transaction hash: ${receipt.transactionHash}`);

    // Emit audit event
    await this.emitAuditEvent({
      operationId: operationId,
      operationType: operation.operationType,
      executionTime: currentTime,
      txHash: receipt.transactionHash,
      signatures: await this.contract.getSignatures(operationId)
    });

    return {
      success: true,
      operationId: operationId,
      txHash: receipt.transactionHash
    };
  }

  getSignatureRequirements(operationType) {
    const requirements = {
      'ASSET_TRANSFER': { threshold: 3, timeDelay: 259200 }, // 72 hours
      'BENEFICIARY_MODIFICATION': { threshold: 2, timeDelay: 604800 }, // 7 days
      'EMERGENCY_ACCESS': { threshold: 4, timeDelay: 86400 }, // 24 hours
      'KEY_RECOVERY': { threshold: 3, timeDelay: 432000 }, // 5 days
      'CONTRACT_DEPLOYMENT': { threshold: 2, timeDelay: 172800 }, // 48 hours
      'REPORT_GENERATION': { threshold: 1, timeDelay: 0 },
      'VERIFICATION_APPROVAL': { threshold: 2, timeDelay: 172800 } // 48 hours
    };

    return requirements[operationType] || { threshold: 2, timeDelay: 86400 };
  }

  async verifyExecutorRole(executorDID, operationType) {
    const executor = await this.contract.getExecutor(executorDID);
    const authorizedRoles = this.getAuthorizedRoles(operationType);
    return authorizedRoles.includes(executor.role);
  }

  getAuthorizedRoles(operationType) {
    const roleMap = {
      'ASSET_TRANSFER': ['PRIMARY_EXECUTOR', 'TECHNICAL_EXECUTOR', 'LEGAL_EXECUTOR'],
      'BENEFICIARY_MODIFICATION': ['PRIMARY_EXECUTOR', 'LEGAL_EXECUTOR'],
      'EMERGENCY_ACCESS': ['PRIMARY_EXECUTOR', 'TECHNICAL_EXECUTOR', 'LEGAL_EXECUTOR', 'CUSTODIAL_EXECUTOR'],
      'KEY_RECOVERY': ['TECHNICAL_EXECUTOR', 'CUSTODIAL_EXECUTOR'],
      'CONTRACT_DEPLOYMENT': ['PRIMARY_EXECUTOR', 'TECHNICAL_EXECUTOR'],
      'REPORT_GENERATION': ['PRIMARY_EXECUTOR', 'FINANCIAL_EXECUTOR'],
      'VERIFICATION_APPROVAL': ['VERIFICATION_EXECUTOR', 'LEGAL_EXECUTOR']
    };

    return roleMap[operationType] || ['PRIMARY_EXECUTOR'];
  }

  getABI() {
    return [
      'function proposeOperation(bytes32 operationId, string operationType, string targetAsset, bytes parameters, uint256 threshold, uint256 timeDelay) returns (bool)',
      'function addSignature(bytes32 operationId, string executorDID, bytes signature) returns (bool)',
      'function executeOperation(bytes32 operationId) returns (bool)',
      'function getOperation(bytes32 operationId) view returns (tuple(string operationType, string targetAsset, bytes parameters, uint256 signaturesRequired, uint256 executionTime, bool executed))',
      'function getSignatureCount(bytes32 operationId) view returns (uint256)',
      'function getExecutor(string executorDID) view returns (tuple(string role, address addr, bool active))',
      'function getSignatures(bytes32 operationId) view returns (bytes[])'
    ];
  }
}

2.5.1 Role-Based Access Control Implementation

// Example 2: Comprehensive Role-Based Access Control for Executors
const AccessControl = require('@wia/access-control');

class ExecutorAccessManager {
  constructor(blockchain) {
    this.blockchain = blockchain;
    this.rolePermissions = this.initializeRolePermissions();
  }

  initializeRolePermissions() {
    return {
      PRIMARY_EXECUTOR: [
        'READ_ASSETS', 'MODIFY_ASSETS', 'EXECUTE_TRANSFERS',
        'MANAGE_BENEFICIARIES', 'DEPLOY_CONTRACTS', 'SIGN_TRANSACTIONS',
        'GENERATE_REPORTS', 'COMMUNICATE_BENEFICIARIES', 'ASSIGN_EXECUTORS'
      ],
      TECHNICAL_EXECUTOR: [
        'READ_ASSETS', 'EXECUTE_TRANSFERS', 'DEPLOY_CONTRACTS',
        'SIGN_TRANSACTIONS', 'ACCESS_KEYS', 'MANAGE_SMART_CONTRACTS'
      ],
      LEGAL_EXECUTOR: [
        'READ_ASSETS', 'VERIFY_IDENTITY', 'MANAGE_BENEFICIARIES',
        'GENERATE_REPORTS', 'LEGAL_COMPLIANCE', 'COURT_REPRESENTATION'
      ],
      FINANCIAL_EXECUTOR: [
        'READ_ASSETS', 'GENERATE_REPORTS', 'CALCULATE_TAXES',
        'VALUATION', 'FINANCIAL_REPORTING'
      ],
      CUSTODIAL_EXECUTOR: [
        'ACCESS_KEYS', 'BACKUP_MANAGEMENT', 'EMERGENCY_RECOVERY',
        'SECURE_STORAGE'
      ],
      VERIFICATION_EXECUTOR: [
        'VERIFY_IDENTITY', 'VALIDATE_DOCUMENTS', 'CONFIRM_EVENTS'
      ],
      BACKUP_EXECUTOR: [
        'READ_ASSETS', 'CONDITIONAL_FULL_ACCESS'
      ]
    };
  }

  async checkPermission(executorDID, permission, resourceId) {
    // Get executor details
    const executor = await this.blockchain.getExecutor(executorDID);

    if (!executor || !executor.active) {
      return {
        allowed: false,
        reason: 'Executor not found or inactive'
      };
    }

    // Check if executor role has required permission
    const rolePermissions = this.rolePermissions[executor.role];

    if (!rolePermissions || !rolePermissions.includes(permission)) {
      return {
        allowed: false,
        reason: `Role ${executor.role} does not have permission ${permission}`
      };
    }

    // Check resource-specific permissions
    const resourcePermission = await this.checkResourcePermission(
      executorDID,
      permission,
      resourceId
    );

    if (!resourcePermission.allowed) {
      return resourcePermission;
    }

    // Check time-based restrictions
    const timeRestriction = await this.checkTimeRestriction(
      executorDID,
      permission
    );

    if (!timeRestriction.allowed) {
      return timeRestriction;
    }

    // Log access attempt
    await this.logAccessAttempt({
      executorDID: executorDID,
      permission: permission,
      resourceId: resourceId,
      timestamp: Date.now(),
      result: 'GRANTED'
    });

    return {
      allowed: true,
      reason: 'Access granted'
    };
  }

  async checkResourcePermission(executorDID, permission, resourceId) {
    // Get resource (will) details
    const will = await this.blockchain.getWill(resourceId);

    if (!will) {
      return { allowed: false, reason: 'Resource not found' };
    }

    // Check if executor is assigned to this will
    const assignment = will.executors.find(e => e.executorDID === executorDID);

    if (!assignment) {
      return {
        allowed: false,
        reason: 'Executor not assigned to this will'
      };
    }

    // Check if assignment is active
    const currentTime = Date.now();
    if (assignment.startDate > currentTime) {
      return {
        allowed: false,
        reason: 'Assignment not yet active'
      };
    }

    if (assignment.endDate && assignment.endDate < currentTime) {
      return {
        allowed: false,
        reason: 'Assignment expired'
      };
    }

    // Check specific permissions for this assignment
    if (!assignment.permissions.includes(permission)) {
      return {
        allowed: false,
        reason: 'Permission not granted for this assignment'
      };
    }

    return { allowed: true };
  }

  async checkTimeRestriction(executorDID, permission) {
    // Some operations may have time restrictions (e.g., business hours only)
    const restrictions = await this.blockchain.getTimeRestrictions(executorDID);

    if (!restrictions || restrictions.length === 0) {
      return { allowed: true };
    }

    const currentTime = new Date();
    const currentHour = currentTime.getUTCHours();
    const currentDay = currentTime.getUTCDay();

    for (const restriction of restrictions) {
      if (restriction.permissions.includes(permission)) {
        if (restriction.allowedHours &&
            !restriction.allowedHours.includes(currentHour)) {
          return {
            allowed: false,
            reason: 'Operation not allowed at this time'
          };
        }

        if (restriction.allowedDays &&
            !restriction.allowedDays.includes(currentDay)) {
          return {
            allowed: false,
            reason: 'Operation not allowed on this day'
          };
        }
      }
    }

    return { allowed: true };
  }

  async logAccessAttempt(accessLog) {
    // Store access log on blockchain for audit trail
    await this.blockchain.storeAccessLog(accessLog);

    // Emit event for monitoring
    this.blockchain.emit('AccessAttempt', accessLog);
  }
}

const { WIAMonitoring } = require('@wia/monitoring');
const EventEmitter = require('events');

class ExecutorDashboard extends EventEmitter {

      this.handleExecutionResult(result);

      assignedWills,
      pendingOperations,

      performanceMetrics,

  async getPendingOperations() {

      executorDID: this.executorDID,

    });

    return operations.map(op => ({
      operationId: op.id,
      operationType: op.type,
      willId: op.willId,
      initiatedBy: op.initiator,
      initiatedAt: op.timestamp,

  }

  async signOperation(operationId, privateKey) {
    // Retrieve operation details
    const operation = await this.monitoring.getOperation(operationId);

    // Verify authorization
    const isAuthorized = await this.verifyAuthorization(operation);
    if (!isAuthorized) {
      throw new Error('Not authorized to sign this operation');
    }

    // Review operation details
    const review = await this.reviewOperation(operation);
    if (!review.approved) {
      throw new Error(`Operation review failed: ${review.reason}`);
    }

    // Generate signature
    const messageHash = this.generateMessageHash(operation);
    const signature = await this.signMessage(messageHash, privateKey);

    // Submit signature
    const result = await this.monitoring.submitSignature({
      operationId: operationId,
      executorDID: this.executorDID,
      signature: signature,
      timestamp: Date.now()
    });

    // Log action
    await this.logAction({
      action: 'OPERATION_SIGNED',
      operationId: operationId,
      timestamp: Date.now()
    });

    this.emit('operation_signed', {
      operationId: operationId,
      result: result
    });

    return result;
  }

  async reviewOperation(operation) {
    // Automated checks
    const checks = {
      validAssets: await this.validateAssets(operation.targetAssets),
      validBeneficiaries: await this.validateBeneficiaries(operation.beneficiaries),
      legalCompliance: await this.checkLegalCompliance(operation),
      riskAssessment: await this.assessRisk(operation)
    };

    // Calculate approval score
    const approvalScore = Object.values(checks).filter(Boolean).length / Object.keys(checks).length;

    return {
      approved: approvalScore >= 0.75,
      score: approvalScore,
      checks: checks,
      reason: approvalScore < 0.75 ? 'Failed one or more validation checks' : 'All checks passed'
    };
  }

  calculatePriority(operation) {
    let priority = 0;

    // Time sensitivity
    const hoursUntilDeadline = (operation.executionTime - Date.now()) / (1000 * 60 * 60);
    if (hoursUntilDeadline < 24) priority += 3;
    else if (hoursUntilDeadline < 72) priority += 2;
    else if (hoursUntilDeadline < 168) priority += 1;

    // Asset value
    if (operation.assetValue > 1000000) priority += 3;
    else if (operation.assetValue > 100000) priority += 2;
    else if (operation.assetValue > 10000) priority += 1;

    // Operation type
    if (operation.type === 'EMERGENCY_ACCESS') priority += 3;
    else if (operation.type === 'ASSET_TRANSFER') priority += 2;

    return Math.min(priority, 10); // Max priority of 10
  }

  async getPerformanceMetrics() {
    const metrics = await this.monitoring.getExecutorMetrics(this.executorDID);

    return {
      completedWills: metrics.completedWills,
      averageExecutionTime: metrics.avgExecutionTime,
      completionRate: metrics.completionRate,
      customerSatisfaction: metrics.customerSatisfaction,
      disputeCount: metrics.disputeCount,
      onTimeDelivery: metrics.onTimeDelivery,
      responseTime: metrics.avgResponseTime,
      earnings: metrics.totalEarnings
    };
  }

  handleAlert(alert) {
    console.log(`[ALERT] ${alert.severity}: ${alert.message}`);

    // Send notification based on severity
    if (alert.severity === 'CRITICAL') {
      this.sendUrgentNotification(alert);
    }

    this.emit('alert', alert);
  }
}
