Chapter 6: Multi-Signature Authorization

6.1 Multi-Signature Architecture for Digital Wills

Multi-signature (multi-sig) authorization represents a critical security layer in digital will execution, requiring multiple independent parties to approve asset transfers before execution. This distributed trust model prevents single points of failure, unauthorized access, and ensures compliance with legal oversight requirements.

The WIA Digital Will Standard implements advanced multi-signature protocols across multiple blockchain networks, supporting various threshold schemes (m-of-n), time-locked releases, hierarchical key structures, and emergency override mechanisms. This chapter provides comprehensive technical specifications for implementing robust multi-signature systems.

6.3.1 Advanced Multi-Sig Wallet Contract

// Example 1: Comprehensive Multi-Signature Wallet for Digital Wills
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/access/AccessControl.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/utils/cryptography/ECDSA.sol";

contract WIAMultiSigWallet is AccessControl, ReentrancyGuard {
    using ECDSA for bytes32;

    bytes32 public constant SIGNER_ROLE = keccak256("SIGNER_ROLE");
    bytes32 public constant EXECUTOR_ROLE = keccak256("EXECUTOR_ROLE");

    struct Transaction {
        uint256 txId;
        address destination;
        uint256 value;
        bytes data;
        bool executed;
        uint256 confirmations;
        uint256 submissionTime;
        uint256 expirationTime;
        mapping(address => bool) isConfirmed;
        TransactionType txType;
        string willReference;
    }

    struct SignerInfo {
        address signerAddress;
        uint256 weight;
        bool isActive;
        uint256 addedAt;
        uint256 expiresAt;
        string role;
    }

    enum TransactionType { ASSET_TRANSFER, ADD_SIGNER, REMOVE_SIGNER, CHANGE_THRESHOLD, EMERGENCY }

    uint256 public requiredSignatures;
    uint256 public requiredWeight;
    uint256 public timelockDuration;
    uint256 public transactionCount;

    mapping(uint256 => Transaction) public transactions;
    mapping(address => SignerInfo) public signers;
    address[] public signerList;

    event TransactionSubmitted(uint256 indexed txId, address indexed submitter, address destination, uint256 value);
    event TransactionConfirmed(uint256 indexed txId, address indexed signer, uint256 currentConfirmations);
    event TransactionExecuted(uint256 indexed txId, address indexed executor);
    event TransactionCanceled(uint256 indexed txId, address indexed canceler);
    event SignerAdded(address indexed signer, uint256 weight, string role);
    event SignerRemoved(address indexed signer);
    event ThresholdChanged(uint256 newThreshold, uint256 newWeight);

    modifier onlyValidSigner() {
        require(hasRole(SIGNER_ROLE, msg.sender), "Not authorized signer");
        require(signers[msg.sender].isActive, "Signer inactive");
        require(block.timestamp < signers[msg.sender].expiresAt, "Signer expired");
        _;
    }

    modifier txExists(uint256 txId) {
        require(txId < transactionCount, "Transaction does not exist");
        _;
    }

    modifier notExecuted(uint256 txId) {
        require(!transactions[txId].executed, "Transaction already executed");
        _;
    }

    modifier notExpired(uint256 txId) {
        require(
            transactions[txId].expirationTime == 0 ||
            block.timestamp < transactions[txId].expirationTime,
            "Transaction expired"
        );
        _;
    }

    constructor(
        address[] memory _signers,
        uint256[] memory _weights,
        string[] memory _roles,
        uint256 _requiredSignatures,
        uint256 _requiredWeight,
        uint256 _timelockDuration
    ) {
        require(_signers.length > 0, "Signers required");
        require(_signers.length == _weights.length, "Length mismatch");
        require(_signers.length == _roles.length, "Roles length mismatch");
        require(_requiredSignatures <= _signers.length, "Invalid threshold");

        _setupRole(DEFAULT_ADMIN_ROLE, msg.sender);

        for (uint256 i = 0; i < _signers.length; i++) {
            address signer = _signers[i];
            require(signer != address(0), "Invalid signer");
            require(!signers[signer].isActive, "Duplicate signer");

            _setupRole(SIGNER_ROLE, signer);

            signers[signer] = SignerInfo({
                signerAddress: signer,
                weight: _weights[i],
                isActive: true,
                addedAt: block.timestamp,
                expiresAt: block.timestamp + 365 days,
                role: _roles[i]
            });

            signerList.push(signer);
            emit SignerAdded(signer, _weights[i], _roles[i]);
        }

        requiredSignatures = _requiredSignatures;
        requiredWeight = _requiredWeight;
        timelockDuration = _timelockDuration;
    }

    function submitTransaction(
        address destination,
        uint256 value,
        bytes memory data,
        TransactionType txType,
        string memory willReference,
        uint256 expirationDuration
    ) public onlyValidSigner returns (uint256) {
        uint256 txId = transactionCount;

        Transaction storage newTx = transactions[txId];
        newTx.txId = txId;
        newTx.destination = destination;
        newTx.value = value;
        newTx.data = data;
        newTx.executed = false;
        newTx.confirmations = 0;
        newTx.submissionTime = block.timestamp;
        newTx.expirationTime = expirationDuration > 0 ? block.timestamp + expirationDuration : 0;
        newTx.txType = txType;
        newTx.willReference = willReference;

        transactionCount++;

        emit TransactionSubmitted(txId, msg.sender, destination, value);

        // Auto-confirm by submitter
        confirmTransaction(txId);

        return txId;
    }

    function confirmTransaction(uint256 txId)
        public
        onlyValidSigner
        txExists(txId)
        notExecuted(txId)
        notExpired(txId)
    {
        Transaction storage txn = transactions[txId];
        require(!txn.isConfirmed[msg.sender], "Already confirmed");

        txn.isConfirmed[msg.sender] = true;
        txn.confirmations++;

        emit TransactionConfirmed(txId, msg.sender, txn.confirmations);

        // Auto-execute if threshold met and timelock passed
        if (isExecutable(txId)) {
            executeTransaction(txId);
        }
    }

    function executeTransaction(uint256 txId)
        public
        onlyValidSigner
        txExists(txId)
        notExecuted(txId)
        nonReentrant
    {
        require(isExecutable(txId), "Cannot execute transaction");

        Transaction storage txn = transactions[txId];
        txn.executed = true;

        (bool success, ) = txn.destination.call{value: txn.value}(txn.data);
        require(success, "Transaction execution failed");

        emit TransactionExecuted(txId, msg.sender);
    }

    function isExecutable(uint256 txId) public view returns (bool) {
        Transaction storage txn = transactions[txId];

        if (txn.executed) return false;
        if (txn.expirationTime > 0 && block.timestamp >= txn.expirationTime) return false;
        if (txn.confirmations < requiredSignatures) return false;
        if (block.timestamp < txn.submissionTime + timelockDuration) return false;

        // Check weighted threshold
        uint256 totalWeight = 0;
        for (uint256 i = 0; i < signerList.length; i++) {
            address signer = signerList[i];
            if (txn.isConfirmed[signer]) {
                totalWeight += signers[signer].weight;
            }
        }

        return totalWeight >= requiredWeight;
    }

    function revokeConfirmation(uint256 txId)
        public
        onlyValidSigner
        txExists(txId)
        notExecuted(txId)
    {
        Transaction storage txn = transactions[txId];
        require(txn.isConfirmed[msg.sender], "Not confirmed");

        txn.isConfirmed[msg.sender] = false;
        txn.confirmations--;
    }

    function addSigner(
        address newSigner,
        uint256 weight,
        string memory role,
        uint256 expirationDuration
    ) external onlyRole(DEFAULT_ADMIN_ROLE) {
        require(newSigner != address(0), "Invalid address");
        require(!signers[newSigner].isActive, "Signer already exists");

        _setupRole(SIGNER_ROLE, newSigner);

        signers[newSigner] = SignerInfo({
            signerAddress: newSigner,
            weight: weight,
            isActive: true,
            addedAt: block.timestamp,
            expiresAt: block.timestamp + expirationDuration,
            role: role
        });

        signerList.push(newSigner);
        emit SignerAdded(newSigner, weight, role);
    }

    function removeSigner(address signer) external onlyRole(DEFAULT_ADMIN_ROLE) {
        require(signers[signer].isActive, "Signer not active");

        signers[signer].isActive = false;
        _revokeRole(SIGNER_ROLE, signer);

        emit SignerRemoved(signer);
    }

    function changeThreshold(uint256 newRequiredSignatures, uint256 newRequiredWeight)
        external
        onlyRole(DEFAULT_ADMIN_ROLE)
    {
        require(newRequiredSignatures > 0, "Invalid threshold");
        require(newRequiredSignatures <= signerList.length, "Threshold too high");

        requiredSignatures = newRequiredSignatures;
        requiredWeight = newRequiredWeight;

        emit ThresholdChanged(newRequiredSignatures, newRequiredWeight);
    }

    function getTransactionConfirmations(uint256 txId)
        public
        view
        txExists(txId)
        returns (address[] memory confirmedSigners)
    {
        Transaction storage txn = transactions[txId];
        address[] memory confirmations = new address[](txn.confirmations);
        uint256 count = 0;

        for (uint256 i = 0; i < signerList.length; i++) {
            if (txn.isConfirmed[signerList[i]]) {
                confirmations[count] = signerList[i];
                count++;
            }
        }

        return confirmations;
    }

    receive() external payable {}
}

6.3.2 Threshold Signature Implementation

// Example 2: Threshold Signature Scheme (TSS) for Digital Wills
const { TSS } = require('@wia/threshold-signatures');
const { ECDSA } = require('elliptic').ec;

class ThresholdSignatureWallet {
  constructor(threshold, totalParties) {
    this.threshold = threshold;
    this.totalParties = totalParties;
    this.tss = new TSS('secp256k1');
    this.shares = new Map();
    this.publicKey = null;
  }

  async initialize() {
    // Generate distributed key shares
    const keyGenResult = await this.tss.distributedKeyGeneration({
      threshold: this.threshold,
      parties: this.totalParties
    });

    this.publicKey = keyGenResult.publicKey;

    return {
      publicKey: this.publicKey,
      shareDistribution: keyGenResult.shares.map((share, index) => ({
        partyIndex: index + 1,
        encryptedShare: share
      }))
    };
  }

  async signTransaction(transaction, participantShares) {
    // Verify we have enough shares
    if (participantShares.length < this.threshold) {
      throw new Error(`Insufficient participants: ${participantShares.length} < ${this.threshold}`);
    }

    // Prepare message to sign
    const messageHash = this.hashTransaction(transaction);

    // Each participant generates signature share
    const signatureShares = [];

    for (const participant of participantShares) {
      const share = await this.tss.generateSignatureShare({
        message: messageHash,
        keyShare: participant.keyShare,
        partyIndex: participant.partyIndex
      });

      signatureShares.push({
        partyIndex: participant.partyIndex,
        signatureShare: share,
        proof: participant.proof
      });
    }

    // Verify all signature shares
    for (const share of signatureShares) {
      const isValid = await this.tss.verifySignatureShare({
        signatureShare: share.signatureShare,
        partyIndex: share.partyIndex,
        publicKey: this.publicKey,
        message: messageHash
      });

      if (!isValid) {
        throw new Error(`Invalid signature share from party ${share.partyIndex}`);
      }
    }

    // Combine signature shares into final signature
    const signature = await this.tss.combineSignatureShares({
      shares: signatureShares.slice(0, this.threshold),
      message: messageHash
    });

    // Verify combined signature
    const isValidSignature = await this.tss.verifySignature({
      signature: signature,
      message: messageHash,
      publicKey: this.publicKey
    });

    if (!isValidSignature) {
      throw new Error('Combined signature verification failed');
    }

    return {
      signature: signature,
      publicKey: this.publicKey,
      signers: signatureShares.map(s => s.partyIndex),
      timestamp: Date.now()
    };
  }

  async refreshShares(currentShares, rotationEpoch) {
    // Proactive secret sharing - refresh shares without changing public key
    const refreshedShares = await this.tss.refreshShares({
      currentShares: currentShares,
      threshold: this.threshold,
      parties: this.totalParties,
      epoch: rotationEpoch
    });

    return refreshedShares;
  }

  async addParticipant(newPartyIndex, existingShares) {
    // Add new participant without regenerating all shares
    const newShare = await this.tss.generateShareForNewParty({
      existingShares: existingShares,
      newPartyIndex: newPartyIndex,
      threshold: this.threshold
    });

    this.totalParties++;

    return newShare;
  }

  async removeParticipant(removedPartyIndex, remainingShares) {
    // Remove participant and redistribute shares
    const updatedShares = await this.tss.redistributeShares({
      remainingShares: remainingShares,
      removedPartyIndex: removedPartyIndex,
      threshold: this.threshold
    });

    this.totalParties--;

    return updatedShares;
  }

  hashTransaction(transaction) {
    const crypto = require('crypto');
    return crypto
      .createHash('sha256')
      .update(JSON.stringify(transaction))
      .digest();
  }
}

6.3.3 Multi-Chain Multi-Sig Coordinator

// Example 3: Cross-Chain Multi-Signature Coordination
const { ethers } = require('ethers');

class MultiChainMultiSigCoordinator {
  constructor(config) {
    this.chains = {};
    this.signatures = new Map();

    // Initialize connections to multiple chains
    for (const [chainName, rpcUrl] of Object.entries(config.chains)) {
      this.chains[chainName] = {
        provider: new ethers.providers.JsonRpcProvider(rpcUrl),
        multiSigAddress: config.multiSigAddresses[chainName]
      };
    }

    this.threshold = config.threshold;
    this.signers = config.signers;
  }

  async proposeTransaction(proposal) {
    const proposalId = this.generateProposalId(proposal);

    // Broadcast proposal to all chains
    const broadcasts = [];

    for (const [chainName, chain] of Object.entries(this.chains)) {
      const contract = new ethers.Contract(
        chain.multiSigAddress,
        ['function proposeTransaction(bytes32 proposalId, bytes calldata data) returns (bool)'],
        chain.provider.getSigner()
      );

      const encodedProposal = this.encodeProposal(proposal, chainName);
      const tx = await contract.proposeTransaction(proposalId, encodedProposal);
      broadcasts.push({ chain: chainName, txHash: tx.hash });
    }

    // Wait for all broadcasts to confirm
    await Promise.all(broadcasts.map(b => this.waitForConfirmation(b.chain, b.txHash)));

    this.signatures.set(proposalId, {
      proposal: proposal,
      collected: [],
      status: 'PENDING',
      createdAt: Date.now()
    });

    return {
      proposalId: proposalId,
      broadcasts: broadcasts,
      requiredSignatures: this.threshold
    };
  }

  async collectSignature(proposalId, signerDID, signature, chainName) {
    const proposal = this.signatures.get(proposalId);

    if (!proposal) {
      throw new Error('Proposal not found');
    }

    // Verify signature
    const messageHash = ethers.utils.hashMessage(proposalId);
    const recoveredAddress = ethers.utils.verifyMessage(proposalId, signature);

    const signer = this.signers.find(s => s.did === signerDID);
    if (!signer || signer.address !== recoveredAddress) {
      throw new Error('Invalid signature');
    }

    // Store signature
    proposal.collected.push({
      signerDID: signerDID,
      signature: signature,
      chainName: chainName,
      timestamp: Date.now()
    });

    // Check if threshold met
    if (proposal.collected.length >= this.threshold) {
      proposal.status = 'READY';
      await this.executeAcrossChains(proposalId);
    }

    return {
      proposalId: proposalId,
      collectedSignatures: proposal.collected.length,
      requiredSignatures: this.threshold,
      status: proposal.status
    };
  }

  async executeAcrossChains(proposalId) {
    const proposal = this.signatures.get(proposalId);

    if (proposal.status !== 'READY') {
      throw new Error('Proposal not ready for execution');
    }

    const executions = [];

    for (const [chainName, chain] of Object.entries(this.chains)) {
      try {
        const contract = new ethers.Contract(
          chain.multiSigAddress,
          ['function executeTransaction(bytes32 proposalId, bytes[] calldata signatures) returns (bool)'],
          chain.provider.getSigner()
        );

        // Collect signatures for this chain
        const signatures = proposal.collected
          .filter(sig => sig.chainName === chainName)
          .map(sig => sig.signature);

        const tx = await contract.executeTransaction(proposalId, signatures);
        const receipt = await tx.wait();

        executions.push({
          chain: chainName,
          txHash: receipt.transactionHash,
          status: 'SUCCESS',
          gasUsed: receipt.gasUsed.toString()
        });
      } catch (error) {
        executions.push({
          chain: chainName,
          status: 'FAILED',
          error: error.message
        });
      }
    }

    proposal.status = 'EXECUTED';
    proposal.executions = executions;

    return executions;
  }

  async monitorProposal(proposalId) {
    const proposal = this.signatures.get(proposalId);

    if (!proposal) {
      throw new Error('Proposal not found');
    }

    return {
      proposalId: proposalId,
      status: proposal.status,
      collectedSignatures: proposal.collected.length,
      requiredSignatures: this.threshold,
      timeElapsed: Date.now() - proposal.createdAt,
      executions: proposal.executions || []
    };
  }

  generateProposalId(proposal) {
    return ethers.utils.keccak256(
      ethers.utils.toUtf8Bytes(JSON.stringify(proposal))
    );
  }

  encodeProposal(proposal, chainName) {
    return ethers.utils.defaultAbiCoder.encode(
      ['string', 'string', 'uint256', 'address', 'string'],
      [
        proposal.willId,
        proposal.assetId,
        proposal.amount,
        proposal.beneficiary,
        chainName
      ]
    );
  }

  async waitForConfirmation(chainName, txHash, confirmations = 3) {
    const provider = this.chains[chainName].provider;
    const receipt = await provider.waitForTransaction(txHash, confirmations);
    return receipt.status === 1;
  }
}

6.4 Multi-Signature API Endpoints

POST/api/v1/multisig/create

Description: Create new multi-signature wallet configuration

Request Body: Signers, threshold, weights, security policies

Response: Multi-sig ID, wallet address, configuration details

POST/api/v1/multisig/propose

Description: Propose new transaction for multi-signature approval

Request Body: Transaction details, will reference, expiration

POST/api/v1/multisig/sign

Description: Submit signature for pending transaction

Request Body: Transaction ID, signature, signer DID

GET/api/v1/multisig/transaction/{txId}

Description: Retrieve transaction status and signatures

Response: Transaction details, collected signatures, execution status

POST/api/v1/multisig/execute

Description: Execute transaction after threshold met

Request Body: Transaction ID, final signatures

DELETE/api/v1/multisig/revoke/{txId}

Description: Revoke signature from pending transaction

Parameters: Transaction ID, signer authorization

POST/api/v1/multisig/signers/add

Description: Add new signer to multi-sig wallet

Request Body: Signer DID, public key, role, weight, expiration

DELETE/api/v1/multisig/signers/remove

Description: Remove signer from wallet (requires multi-sig approval)

Request Body: Signer DID, removal reason, approval signatures

PUT/api/v1/multisig/threshold

Description: Update signature threshold requirements

Request Body: New required signatures, new weight threshold

GET/api/v1/multisig/pending

Description: List all pending transactions for signer

Query Parameters: Signer DID, status filter, pagination

POST/api/v1/multisig/tss/keygen

Description: Initialize threshold signature key generation

Request Body: Threshold, total parties, protocol type

POST/api/v1/multisig/tss/sign

Description: Generate threshold signature share

Request Body: Message hash, key share, party index

POST/api/v1/multisig/tss/combine

Description: Combine threshold signature shares

Request Body: Signature shares, message, threshold

GET/api/v1/multisig/audit/{walletId}

Description: Retrieve complete audit trail for wallet
