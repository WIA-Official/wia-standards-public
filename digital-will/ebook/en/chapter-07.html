Chapter 7: Digital Testament Verification

7.1 Verification Framework for Digital Wills

Digital testament verification establishes authenticity, validity, and legal enforceability of blockchain-based wills through cryptographic proofs, notarization protocols, witness attestation systems, and regulatory compliance mechanisms. The WIA Digital Will Standard provides comprehensive verification protocols ensuring digital wills meet the highest standards of legal recognition across jurisdictions.

Verification encompasses multiple dimensions: cryptographic integrity validation, testator identity authentication, witness verification, temporal authenticity (creation and modification timestamps), legal compliance certification, and cross-jurisdictional recognition. This multi-layered approach ensures digital wills withstand legal challenges and maintain evidentiary value.

7.3.1 Comprehensive Verification System

// Example 1: Digital Will Verification Engine
const crypto = require('crypto');
const { VerifiableCredential } = require('@wia/did-verification');
const { BlockchainTimestamp } = require('@wia/blockchain-timestamp');

class DigitalWillVerificationEngine {
  constructor(config) {
    this.blockchainTimestamp = new BlockchainTimestamp(config.blockchain);
    this.vcVerifier = new VerifiableCredential();
    this.notaryRegistry = config.notaryRegistry;
    this.verificationResults = new Map();
  }

  async verifyDigitalWill(will, verificationOptions = {}) {
    const verificationId = this.generateVerificationId();

    const results = {
      verificationId: verificationId,
      willId: will.id,
      overallStatus: 'PENDING',
      verifications: [],
      score: 0,
      timestamp: new Date().toISOString()
    };

    try {
      // 1. Verify document integrity
      const integrityCheck = await this.verifyDocumentIntegrity(will);
      results.verifications.push(integrityCheck);
      if (integrityCheck.passed) results.score += 15;

      // 2. Verify testator identity
      const identityCheck = await this.verifyTestatorIdentity(will.testator);
      results.verifications.push(identityCheck);
      if (identityCheck.passed) results.score += 20;

      // 3. Verify testator capacity
      const capacityCheck = await this.verifyTestamentaryCapacity(
        will.testator,
        will.createdAt
      );
      results.verifications.push(capacityCheck);
      if (capacityCheck.passed) results.score += 20;

      // 4. Verify witnesses
      const witnessCheck = await this.verifyWitnesses(will.witnesses);
      results.verifications.push(witnessCheck);
      if (witnessCheck.passed) results.score += 15;

      // 5. Verify notarization (if present)
      if (will.notaryAttestation) {
        const notaryCheck = await this.verifyNotaryAttestation(
          will.notaryAttestation
        );
        results.verifications.push(notaryCheck);
        if (notaryCheck.passed) results.score += 15;
      }

      // 6. Verify blockchain timestamp
      const timestampCheck = await this.verifyBlockchainTimestamp(will);
      results.verifications.push(timestampCheck);
      if (timestampCheck.passed) results.score += 10;

      // 7. Verify legal compliance
      const complianceCheck = await this.verifyLegalCompliance(
        will,
        verificationOptions.jurisdiction
      );
      results.verifications.push(complianceCheck);
      if (complianceCheck.passed) results.score += 15;

      // 8. Verify signatures
      const signatureCheck = await this.verifyAllSignatures(will);
      results.verifications.push(signatureCheck);
      if (signatureCheck.passed) results.score += 10;

      // Calculate overall status
      results.overallStatus = this.calculateOverallStatus(results.score);

      // Store verification results on blockchain
      await this.recordVerificationOnChain(results);

    } catch (error) {
      results.overallStatus = 'FAILED';
      results.error = error.message;
    }

    this.verificationResults.set(verificationId, results);
    return results;
  }

  async verifyDocumentIntegrity(will) {
    const currentHash = crypto
      .createHash('sha3-256')
      .update(JSON.stringify(will.content))
      .digest('hex');

    const originalHash = will.documentHash;

    return {
      verificationType: 'DOCUMENT_INTEGRITY',
      passed: currentHash === originalHash,
      details: {
        originalHash: originalHash,
        currentHash: currentHash,
        algorithm: 'SHA3-256'
      },
      timestamp: new Date().toISOString()
    };
  }

  async verifyTestatorIdentity(testator) {
    try {
      // Verify DID and associated credentials
      const didVerification = await this.vcVerifier.verifyDID(testator.did);

      // Verify identity credentials
      const credentialVerification = await this.vcVerifier.verifyCredential(
        testator.identityCredential
      );

      return {
        verificationType: 'TESTATOR_IDENTITY',
        passed: didVerification.valid && credentialVerification.valid,
        details: {
          didValid: didVerification.valid,
          credentialValid: credentialVerification.valid,
          identityProvider: credentialVerification.issuer,
          confidenceLevel: 'HIGH'
        },
        timestamp: new Date().toISOString()
      };
    } catch (error) {
      return {
        verificationType: 'TESTATOR_IDENTITY',
        passed: false,
        error: error.message,
        timestamp: new Date().toISOString()
      };
    }
  }

  async verifyTestamentaryCapacity(testator, willCreationDate) {
    // Verify testator had legal capacity at time of will creation
    const capacityAssessment = testator.capacityAssessment;

    if (!capacityAssessment) {
      return {
        verificationType: 'TESTAMENTARY_CAPACITY',
        passed: false,
        details: { reason: 'No capacity assessment provided' },
        timestamp: new Date().toISOString()
      };
    }

    // Check assessment date is close to will creation
    const assessmentDate = new Date(capacityAssessment.date);
    const creationDate = new Date(willCreationDate);
    const daysDifference = Math.abs(
      (assessmentDate - creationDate) / (1000 * 60 * 60 * 24)
    );

    const timely = daysDifference <= 30; // Within 30 days

    // Verify assessor credentials
    const assessorValid = await this.verifyAssessorCredentials(
      capacityAssessment.assessorDID
    );

    return {
      verificationType: 'TESTAMENTARY_CAPACITY',
      passed: capacityAssessment.capacity === 'FULL_CAPACITY' &&
              timely &&
              assessorValid,
      details: {
        capacity: capacityAssessment.capacity,
        assessmentDate: capacityAssessment.date,
        assessor: capacityAssessment.assessorDID,
        timeliness: timely,
        assessorValid: assessorValid
      },
      timestamp: new Date().toISOString()
    };
  }

  async verifyWitnesses(witnesses) {
    if (!witnesses || witnesses.length < 2) {
      return {
        verificationType: 'WITNESS_VERIFICATION',
        passed: false,
        details: { reason: 'Insufficient witnesses (minimum 2 required)' },
        timestamp: new Date().toISOString()
      };
    }

    const witnessVerifications = [];

    for (const witness of witnesses) {
      // Verify witness identity
      const didValid = await this.vcVerifier.verifyDID(witness.did);

      // Verify witness signature
      const signatureValid = await this.verifySignature(
        witness.attestation,
        witness.signature,
        witness.publicKey
      );

      // Check witness independence (not a beneficiary)
      const independent = await this.checkWitnessIndependence(witness);

      witnessVerifications.push({
        witnessDID: witness.did,
        didValid: didValid.valid,
        signatureValid: signatureValid,
        independent: independent,
        overall: didValid.valid && signatureValid && independent
      });
    }

    const allValid = witnessVerifications.every(w => w.overall);

    return {
      verificationType: 'WITNESS_VERIFICATION',
      passed: allValid,
      details: {
        totalWitnesses: witnesses.length,
        verifications: witnessVerifications
      },
      timestamp: new Date().toISOString()
    };
  }

  async verifyNotaryAttestation(notaryAttestation) {
    // Verify notary is registered and in good standing
    const notaryValid = await this.notaryRegistry.verifyNotary({
      notaryDID: notaryAttestation.notaryDID,
      jurisdiction: notaryAttestation.jurisdiction,
      licenseNumber: notaryAttestation.licenseNumber
    });

    // Verify notary signature and seal
    const signatureValid = await this.verifySignature(
      notaryAttestation.attestation,
      notaryAttestation.signature,
      notaryAttestation.publicKey
    );

    return {
      verificationType: 'NOTARY_ATTESTATION',
      passed: notaryValid && signatureValid,
      details: {
        notaryDID: notaryAttestation.notaryDID,
        jurisdiction: notaryAttestation.jurisdiction,
        notaryValid: notaryValid,
        signatureValid: signatureValid,
        certificationNumber: notaryAttestation.certificationNumber
      },
      timestamp: new Date().toISOString()
    };
  }

  async verifyBlockchainTimestamp(will) {
    try {
      const timestampProof = await this.blockchainTimestamp.verify({
        documentHash: will.documentHash,
        transactionHash: will.blockchainTimestamp.txHash,
        blockchain: will.blockchainTimestamp.blockchain
      });

      return {
        verificationType: 'BLOCKCHAIN_TIMESTAMP',
        passed: timestampProof.valid,
        details: {
          blockchain: will.blockchainTimestamp.blockchain,
          transactionHash: will.blockchainTimestamp.txHash,
          blockNumber: timestampProof.blockNumber,
          timestamp: timestampProof.timestamp,
          confirmations: timestampProof.confirmations
        },
        timestamp: new Date().toISOString()
      };
    } catch (error) {
      return {
        verificationType: 'BLOCKCHAIN_TIMESTAMP',
        passed: false,
        error: error.message,
        timestamp: new Date().toISOString()
      };
    }
  }

  async verifyLegalCompliance(will, jurisdiction) {
    // Check compliance with jurisdictional requirements
    const requirements = await this.getLegalRequirements(jurisdiction);

    const complianceChecks = {
      minimumWitnesses: will.witnesses.length >= requirements.minimumWitnesses,
      testatorAge: will.testator.age >= requirements.minimumAge,
      notarizationRequired: requirements.notarizationRequired ?
        !!will.notaryAttestation : true,
      formCompliance: true, // Additional checks would go here
      contentCompliance: true
    };

    const allCompliant = Object.values(complianceChecks).every(c => c);

    return {
      verificationType: 'LEGAL_COMPLIANCE',
      passed: allCompliant,
      details: {
        jurisdiction: jurisdiction,
        requirements: requirements,
        complianceChecks: complianceChecks
      },
      timestamp: new Date().toISOString()
    };
  }

  async verifyAllSignatures(will) {
    const signatures = [
      { type: 'testator', ...will.testatorSignature },
      ...will.witnesses.map(w => ({ type: 'witness', ...w })),
    ];

    if (will.notaryAttestation) {
      signatures.push({ type: 'notary', ...will.notaryAttestation });
    }

    const verifications = [];

    for (const sig of signatures) {
      const valid = await this.verifySignature(
        sig.signedData || sig.attestation,
        sig.signature,
        sig.publicKey
      );

      verifications.push({
        type: sig.type,
        signer: sig.did || sig.notaryDID,
        valid: valid
      });
    }

    const allValid = verifications.every(v => v.valid);

    return {
      verificationType: 'SIGNATURE_VERIFICATION',
      passed: allValid,
      details: { verifications: verifications },
      timestamp: new Date().toISOString()
    };
  }

  async verifySignature(message, signature, publicKey) {
    try {
      const verify = crypto.createVerify('SHA256');
      verify.update(message);
      return verify.verify(publicKey, signature, 'hex');
    } catch (error) {
      return false;
    }
  }

  calculateOverallStatus(score) {
    if (score >= 90) return 'FULLY_VERIFIED';
    if (score >= 70) return 'VERIFIED';
    if (score >= 50) return 'PARTIALLY_VERIFIED';
    return 'VERIFICATION_FAILED';
  }

  async recordVerificationOnChain(results) {
    const verificationHash = crypto
      .createHash('sha3-256')
      .update(JSON.stringify(results))
      .digest('hex');

    await this.blockchainTimestamp.record({
      data: verificationHash,
      metadata: {
        verificationId: results.verificationId,
        willId: results.willId,
        status: results.overallStatus,
        score: results.score
      }
    });
  }

  generateVerificationId() {
    const timestamp = Date.now().toString().slice(-8);
    const random = crypto.randomBytes(8).toString('hex').toUpperCase();
    return `VERIFY-${timestamp}-${random}`;
  }

// Example 2: Smart Contract for Will Attestation
pragma solidity ^0.8.19;

contract WIAWillAttestation {
    struct Attestation {
        bytes32 willHash;
        address testator;
        uint256 timestamp;
        AttestationType attestationType;
        address[] witnesses;
        address notary;
        bool isValid;
        string jurisdiction;
        bytes32 verificationProof;
    }

    enum AttestationType {
        WITNESS,
        NOTARY,
        LEGAL_COUNSEL,
        MEDICAL_CAPACITY,
        BLOCKCHAIN_TIMESTAMP
    }

    mapping(bytes32 => Attestation) public attestations;
    mapping(bytes32 => mapping(address => bool)) public hasAttested;
    mapping(address => bool) public authorizedNotaries;
    mapping(address => bool) public authorizedVerifiers;

    event AttestationCreated(bytes32 indexed willHash, address indexed testator, uint256 timestamp);
    event WitnessAttested(bytes32 indexed willHash, address indexed witness, uint256 timestamp);
    event NotaryAttested(bytes32 indexed willHash, address indexed notary, uint256 timestamp);
    event AttestationVerified(bytes32 indexed willHash, address indexed verifier, bool isValid);

    modifier onlyAuthorizedNotary() {
        require(authorizedNotaries[msg.sender], "Not authorized notary");
        _;
    }

    modifier onlyAuthorizedVerifier() {
        require(authorizedVerifiers[msg.sender], "Not authorized verifier");
        _;
    }

    function createAttestation(
        bytes32 willHash,
        string memory jurisdiction
    ) external returns (bool) {
        require(attestations[willHash].timestamp == 0, "Attestation exists");

        attestations[willHash] = Attestation({
            willHash: willHash,
            testator: msg.sender,
            timestamp: block.timestamp,
            attestationType: AttestationType.BLOCKCHAIN_TIMESTAMP,
            witnesses: new address[](0),
            notary: address(0),
            isValid: false,
            jurisdiction: jurisdiction,
            verificationProof: bytes32(0)
        });

        emit AttestationCreated(willHash, msg.sender, block.timestamp);
        return true;
    }

    function addWitnessAttestation(
        bytes32 willHash
    ) external returns (bool) {
        Attestation storage attestation = attestations[willHash];
        require(attestation.timestamp > 0, "Attestation does not exist");
        require(!hasAttested[willHash][msg.sender], "Already attested");
        require(msg.sender != attestation.testator, "Testator cannot be witness");

        attestation.witnesses.push(msg.sender);
        hasAttested[willHash][msg.sender] = true;

        emit WitnessAttested(willHash, msg.sender, block.timestamp);

        // Auto-verify if minimum requirements met
        if (attestation.witnesses.length >= 2) {
            attestation.isValid = true;
        }

        return true;
    }

    function addNotaryAttestation(
        bytes32 willHash,
        bytes32 verificationProof
    ) external onlyAuthorizedNotary returns (bool) {
        Attestation storage attestation = attestations[willHash];
        require(attestation.timestamp > 0, "Attestation does not exist");
        require(attestation.notary == address(0), "Notary already attested");

        attestation.notary = msg.sender;
        attestation.verificationProof = verificationProof;
        attestation.isValid = true;

        emit NotaryAttested(willHash, msg.sender, block.timestamp);
        return true;
    }

    function verifyAttestation(
        bytes32 willHash,
        bool isValid
    ) external onlyAuthorizedVerifier returns (bool) {
        Attestation storage attestation = attestations[willHash];
        require(attestation.timestamp > 0, "Attestation does not exist");

        attestation.isValid = isValid;

        emit AttestationVerified(willHash, msg.sender, isValid);
        return true;
    }

    function getAttestation(
        bytes32 willHash
    ) external view returns (
        address testator,
        uint256 timestamp,
        address[] memory witnesses,
        address notary,
        bool isValid,
        string memory jurisdiction
    ) {
        Attestation storage attestation = attestations[willHash];
        return (
            attestation.testator,
            attestation.timestamp,
            attestation.witnesses,
            attestation.notary,
            attestation.isValid,
            attestation.jurisdiction
        );
    }

    function addAuthorizedNotary(address notary) external {
        authorizedNotaries[notary] = true;
    }

    function addAuthorizedVerifier(address verifier) external {
        authorizedVerifiers[verifier] = true;
    }
}

7.3.3 Notarization Integration System

// Example 3: Digital Notarization Service Integration
const { DigitalNotary } = require('@wia/digital-notary');

class DigitalWillNotarizationService {
  constructor(config) {
    this.notary = new DigitalNotary(config);
    this.notaryDID = config.notaryDID;
    this.jurisdiction = config.jurisdiction;
  }

  async notarizeDigitalWill(will, testatorPresent = true) {
    // Step 1: Verify notary credentials
    const notaryValid = await this.verifyNotaryCredentials();

    if (!notaryValid) {
      throw new Error('Invalid notary credentials');
    }

    // Step 2: Verify testator identity
    const identityVerification = await this.verifyTestatorIdentity(
      will.testator,
      testatorPresent
    );

    // Step 3: Assess testamentary capacity
    const capacityAssessment = await this.assessTestamentaryCapacity(
      will.testator
    );

    // Step 4: Verify document integrity
    const documentHash = this.hashDocument(will);

    // Step 5: Create notarization certificate
    const certificate = await this.createNotarizationCertificate({
      will: will,
      documentHash: documentHash,
      identityVerification: identityVerification,
      capacityAssessment: capacityAssessment,
      testatorPresent: testatorPresent
    });

    // Step 6: Sign certificate with notary's private key
    const signedCertificate = await this.signCertificate(certificate);

    // Step 7: Record on blockchain
    const blockchainRecord = await this.recordOnBlockchain(signedCertificate);

    // Step 8: Generate notary seal
    const notarySeal = await this.generateNotarySeal(signedCertificate);

    return {
      certificate: signedCertificate,
      blockchainTxHash: blockchainRecord.txHash,
      notarySeal: notarySeal,
      certificationNumber: this.generateCertificationNumber(),
      timestamp: new Date().toISOString()
    };
  }

  async verifyNotaryCredentials() {
    return await this.notary.verifyCredentials({
      notaryDID: this.notaryDID,
      jurisdiction: this.jurisdiction
    });
  }

  async verifyTestatorIdentity(testator, inPerson) {
    if (inPerson) {
      // In-person identity verification
      return {
        method: 'IN_PERSON_VERIFICATION',
        verified: true,
        documentType: 'GOVERNMENT_ID',
        timestamp: new Date().toISOString()
      };
    } else {
      // Remote identity verification
      return await this.notary.verifyRemoteIdentity({
        did: testator.did,
        biometricData: testator.biometricData,
        videoConference: true
      });
    }
  }

  async assessTestamentaryCapacity(testator) {
    return {
      assessmentPerformed: true,
      capacity: 'FULL_CAPACITY',
      assessmentDate: new Date().toISOString(),
      assessor: this.notaryDID,
      notes: 'Testator demonstrated clear understanding of will provisions'
    };
  }

  hashDocument(will) {
    return crypto
      .createHash('sha3-256')
      .update(JSON.stringify(will))
      .digest('hex');
  }

  async createNotarizationCertificate(data) {
    return {
      certificateId: this.generateCertificateId(),
      willId: data.will.id,
      testator: data.will.testator.did,
      documentHash: data.documentHash,
      notaryDID: this.notaryDID,
      jurisdiction: this.jurisdiction,
      identityVerification: data.identityVerification,
      capacityAssessment: data.capacityAssessment,
      testatorPresent: data.testatorPresent,
      notarizationDate: new Date().toISOString(),
      expirationDate: new Date(Date.now() + 365 * 24 * 60 * 60 * 1000).toISOString()
    };
  }

  async signCertificate(certificate) {
    const signature = await this.notary.sign(
      JSON.stringify(certificate),
      this.notaryDID
    );

    return {
      ...certificate,
      notarySignature: signature,
      signatureAlgorithm: 'ECDSA-secp256k1'
    };
  }

  async recordOnBlockchain(certificate) {
    const certHash = this.hashDocument(certificate);

    return await this.notary.recordOnBlockchain({
      documentHash: certHash,
      certificateId: certificate.certificateId,
      notaryDID: this.notaryDID
    });
  }

  async generateNotarySeal(certificate) {
    return await this.notary.generateSeal({
      certificateId: certificate.certificateId,
      notaryDID: this.notaryDID,
      jurisdiction: this.jurisdiction
    });
  }

  generateCertificateId() {
    const timestamp = Date.now().toString();
    const random = crypto.randomBytes(8).toString('hex');
    return `NOTARY-${this.jurisdiction}-${timestamp}-${random}`;
  }

  generateCertificationNumber() {
    return `${this.jurisdiction}-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`.toUpperCase();
  }
}
