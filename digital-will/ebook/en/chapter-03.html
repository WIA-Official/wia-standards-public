Chapter 3: Smart Contract Execution

3.1 Smart Contract Architecture for Digital Wills

Smart contracts form the backbone of automated digital will execution in the WIA standard. These self-executing contracts encode will provisions into immutable blockchain code that automatically triggers asset transfers upon verified conditions. The architecture ensures security, transparency, and irreversibility while maintaining flexibility for complex inheritance scenarios.

The WIA smart contract framework supports multiple blockchain networks, conditional logic, time-locks, multi-signature requirements, and emergency override mechanisms. Each contract undergoes rigorous security audits and formal verification before deployment to production networks.

3.1.1 Smart Contract Trigger Events

3.3 Smart Contract Implementation Examples

3.3.1 Complete WIA Digital Will Smart Contract (Solidity)

// Example 1: Comprehensive WIA Digital Will Smart Contract
pragma solidity ^0.8.19;

import "@openzeppelin/contracts-upgradeable/proxy/utils/Initializable.sol";
import "@openzeppelin/contracts-upgradeable/access/AccessControlUpgradeable.sol";
import "@openzeppelin/contracts-upgradeable/security/PausableUpgradeable.sol";
import "@openzeppelin/contracts-upgradeable/security/ReentrancyGuardUpgradeable.sol";

contract WIADigitalWill is
    Initializable,
    AccessControlUpgradeable,
    PausableUpgradeable,
    ReentrancyGuardUpgradeable
{
    bytes32 public constant EXECUTOR_ROLE = keccak256("EXECUTOR_ROLE");
    bytes32 public constant VERIFIER_ROLE = keccak256("VERIFIER_ROLE");
    bytes32 public constant TESTATOR_ROLE = keccak256("TESTATOR_ROLE");

    struct Will {
        string willId;
        address testator;
        string testatorDID;
        WillStatus status;
        uint256 creationTime;
        uint256 triggerTime;
        uint256 executionTime;
        bytes32 contentHash;
    }

    struct TriggerCondition {
        TriggerType triggerType;
        uint256 threshold;
        uint256 verificationCount;
        mapping(address => bool) verifications;
        uint256 activationTime;
        bool activated;
    }

    struct Beneficiary {
        string beneficiaryDID;
        address beneficiaryAddress;
        uint256 allocationPercentage;
        BeneficiaryCondition[] conditions;
        bool conditionsMet;
        uint256 amountReceived;
        VestingSchedule vesting;
    }

    struct VestingSchedule {
        bool enabled;
        VestingPeriod[] periods;
        uint256 totalVested;
    }

    struct VestingPeriod {
        uint256 unlockTime;
        uint256 unlockPercentage;
        bool unlocked;
    }

    struct BeneficiaryCondition {
        ConditionType conditionType;
        bytes conditionData;
        bool verified;
    }

    struct Asset {
        string assetId;
        AssetType assetType;
        address assetAddress;
        uint256 amount;
        bytes metadata;
    }

    enum WillStatus { ACTIVE, TRIGGERED, EXECUTING, EXECUTED, CANCELLED, DISPUTED }
    enum TriggerType { DEATH_CERTIFICATE, INCAPACITATION, TIME_LOCK, INACTIVITY, MANUAL, LEGAL_ORDER }
    enum AssetType { ETH, ERC20, ERC721, ERC1155, EXTERNAL }
    enum ConditionType { AGE, IDENTITY, MARRIAGE_STATUS, EMPLOYMENT, CUSTOM }

    // State variables
    mapping(string => Will) public wills;
    mapping(string => TriggerCondition) public triggerConditions;
    mapping(string => Beneficiary[]) public willBeneficiaries;
    mapping(string => Asset[]) public willAssets;
    mapping(string => mapping(address => uint256)) public executorSignatures;

    uint256 public constant CHALLENGE_PERIOD = 30 days;
    uint256 public constant EXECUTION_DELAY = 7 days;

    // Events
    event WillCreated(string indexed willId, address indexed testator, uint256 timestamp);
    event TriggerActivated(string indexed willId, TriggerType triggerType, uint256 timestamp);
    event VerificationAdded(string indexed willId, address indexed verifier, uint256 timestamp);
    event ExecutionInitiated(string indexed willId, uint256 executionTime);
    event AssetDistributed(string indexed willId, string beneficiaryDID, uint256 amount);
    event WillChallenged(string indexed willId, address challenger, string reason);
    event EmergencyPause(string indexed willId, address pauser);

    function initialize() public initializer {
        __AccessControl_init();
        __Pausable_init();
        __ReentrancyGuard_init();
        _setupRole(DEFAULT_ADMIN_ROLE, msg.sender);
    }

    function createWill(
        string memory willId,
        string memory testatorDID,
        TriggerType triggerType,
        uint256 verificationThreshold,
        bytes32 contentHash
    ) external whenNotPaused {
        require(wills[willId].creationTime == 0, "Will already exists");

        wills[willId] = Will({
            willId: willId,
            testator: msg.sender,
            testatorDID: testatorDID,
            status: WillStatus.ACTIVE,
            creationTime: block.timestamp,
            triggerTime: 0,
            executionTime: 0,
            contentHash: contentHash
        });

        triggerConditions[willId] = TriggerCondition({
            triggerType: triggerType,
            threshold: verificationThreshold,
            verificationCount: 0,
            activationTime: 0,
            activated: false
        });

        _setupRole(TESTATOR_ROLE, msg.sender);
        emit WillCreated(willId, msg.sender, block.timestamp);
    }

    function addBeneficiary(
        string memory willId,
        string memory beneficiaryDID,
        address beneficiaryAddress,
        uint256 allocationPercentage,
        bool enableVesting,
        VestingPeriod[] memory vestingPeriods
    ) external onlyRole(TESTATOR_ROLE) {
        require(wills[willId].status == WillStatus.ACTIVE, "Will not active");
        require(wills[willId].testator == msg.sender, "Not will owner");

        Beneficiary storage beneficiary = willBeneficiaries[willId].push();
        beneficiary.beneficiaryDID = beneficiaryDID;
        beneficiary.beneficiaryAddress = beneficiaryAddress;
        beneficiary.allocationPercentage = allocationPercentage;
        beneficiary.conditionsMet = false;
        beneficiary.amountReceived = 0;
        beneficiary.vesting.enabled = enableVesting;

        if (enableVesting) {
            for (uint256 i = 0; i < vestingPeriods.length; i++) {
                beneficiary.vesting.periods.push(vestingPeriods[i]);
            }
        }
    }

    function addAsset(
        string memory willId,
        string memory assetId,
        AssetType assetType,
        address assetAddress,
        uint256 amount,
        bytes memory metadata
    ) external onlyRole(TESTATOR_ROLE) {
        require(wills[willId].status == WillStatus.ACTIVE, "Will not active");
        require(wills[willId].testator == msg.sender, "Not will owner");

        willAssets[willId].push(Asset({
            assetId: assetId,
            assetType: assetType,
            assetAddress: assetAddress,
            amount: amount,
            metadata: metadata
        }));
    }

    function verifyTrigger(string memory willId)
        external
        onlyRole(VERIFIER_ROLE)
        whenNotPaused
    {
        Will storage will = wills[willId];
        require(will.status == WillStatus.ACTIVE, "Will not active");

        TriggerCondition storage trigger = triggerConditions[willId];
        require(!trigger.verifications[msg.sender], "Already verified");

        trigger.verifications[msg.sender] = true;
        trigger.verificationCount++;

        emit VerificationAdded(willId, msg.sender, block.timestamp);

        if (trigger.verificationCount >= trigger.threshold && !trigger.activated) {
            activateTrigger(willId);
        }
    }

    function activateTrigger(string memory willId) internal {
        TriggerCondition storage trigger = triggerConditions[willId];
        Will storage will = wills[willId];

        trigger.activated = true;
        trigger.activationTime = block.timestamp;
        will.status = WillStatus.TRIGGERED;
        will.triggerTime = block.timestamp;
        will.executionTime = block.timestamp + CHALLENGE_PERIOD + EXECUTION_DELAY;

        emit TriggerActivated(willId, trigger.triggerType, block.timestamp);
    }

    function executeWill(string memory willId)
        external
        onlyRole(EXECUTOR_ROLE)
        nonReentrant
        whenNotPaused
    {
        Will storage will = wills[willId];
        require(will.status == WillStatus.TRIGGERED, "Will not triggered");
        require(block.timestamp >= will.executionTime, "Execution time not reached");

        will.status = WillStatus.EXECUTING;
        emit ExecutionInitiated(willId, block.timestamp);

        // Execute distributions
        Beneficiary[] storage beneficiaries = willBeneficiaries[willId];
        Asset[] storage assets = willAssets[willId];

        for (uint256 i = 0; i < beneficiaries.length; i++) {
            if (beneficiaries[i].conditionsMet || beneficiaries[i].conditions.length == 0) {
                distributeAssets(willId, i, assets);
            }
        }

        will.status = WillStatus.EXECUTED;
    }

    function distributeAssets(
        string memory willId,
        uint256 beneficiaryIndex,
        Asset[] storage assets
    ) internal {
        Beneficiary storage beneficiary = willBeneficiaries[willId][beneficiaryIndex];

        for (uint256 i = 0; i < assets.length; i++) {
            uint256 amount = (assets[i].amount * beneficiary.allocationPercentage) / 100;

            if (beneficiary.vesting.enabled) {
                amount = calculateVestedAmount(beneficiary, amount);
            }

            if (amount > 0) {
                transferAsset(assets[i], beneficiary.beneficiaryAddress, amount);
                beneficiary.amountReceived += amount;
                emit AssetDistributed(willId, beneficiary.beneficiaryDID, amount);
            }
        }
    }

    function calculateVestedAmount(
        Beneficiary storage beneficiary,
        uint256 totalAmount
    ) internal view returns (uint256) {
        uint256 vestedPercentage = 0;

        for (uint256 i = 0; i < beneficiary.vesting.periods.length; i++) {
            if (block.timestamp >= beneficiary.vesting.periods[i].unlockTime) {
                vestedPercentage += beneficiary.vesting.periods[i].unlockPercentage;
            }
        }

        return (totalAmount * vestedPercentage) / 100;
    }

    function transferAsset(
        Asset storage asset,
        address recipient,
        uint256 amount
    ) internal {
        if (asset.assetType == AssetType.ETH) {
            payable(recipient).transfer(amount);
        } else if (asset.assetType == AssetType.ERC20) {
            // ERC20 transfer logic
            (bool success, ) = asset.assetAddress.call(
                abi.encodeWithSignature("transfer(address,uint256)", recipient, amount)
            );
            require(success, "ERC20 transfer failed");
        }
        // Additional asset type handling...
    }

    function challengeWill(string memory willId, string memory reason)
        external
    {
        Will storage will = wills[willId];
        require(will.status == WillStatus.TRIGGERED, "Cannot challenge at this stage");
        require(block.timestamp < will.triggerTime + CHALLENGE_PERIOD, "Challenge period expired");

        will.status = WillStatus.DISPUTED;
        emit WillChallenged(willId, msg.sender, reason);
    }

    function emergencyPause(string memory willId)
        external
        onlyRole(EXECUTOR_ROLE)
    {
        Will storage will = wills[willId];
        require(will.status != WillStatus.EXECUTED, "Already executed");

        _pause();
        emit EmergencyPause(willId, msg.sender);
    }

    function verifyBeneficiaryCondition(
        string memory willId,
        uint256 beneficiaryIndex,
        uint256 conditionIndex
    ) external onlyRole(VERIFIER_ROLE) {
        Beneficiary storage beneficiary = willBeneficiaries[willId][beneficiaryIndex];
        beneficiary.conditions[conditionIndex].verified = true;

        // Check if all conditions met
        bool allMet = true;
        for (uint256 i = 0; i < beneficiary.conditions.length; i++) {
            if (!beneficiary.conditions[i].verified) {
                allMet = false;
                break;
            }
        }
        beneficiary.conditionsMet = allMet;
    }

    receive() external payable {}
}

3.3.2 Smart Contract Deployment and Management

// Example 2: Smart Contract Deployment and Management System
const { ethers, upgrades } = require('hardhat');
const { WIAContractFactory } = require('@wia/contract-factory');

class SmartContractManager {
  constructor(network, deployerPrivateKey) {
    this.provider = new ethers.providers.JsonRpcProvider(network.rpc);
    this.deployer = new ethers.Wallet(deployerPrivateKey, this.provider);
    this.factory = new WIAContractFactory(this.deployer);
  }

  async deployWillContract(configuration) {
    console.log('Deploying WIA Digital Will Smart Contract...');

    // Validate configuration
    await this.validateConfiguration(configuration);

    // Deploy implementation contract
    const WIADigitalWill = await ethers.getContractFactory('WIADigitalWill', this.deployer);

    // Deploy as upgradeable proxy
    const contract = await upgrades.deployProxy(
      WIADigitalWill,
      [],
      {
        initializer: 'initialize',
        kind: 'uups'
      }
    );

    await contract.deployed();
    console.log(`Contract deployed at: ${contract.address}`);

    // Configure will parameters
    await this.configureWill(contract, configuration);

    // Verify contract on block explorer
    await this.verifyContract(contract.address, configuration);

    // Generate audit report
    const auditReport = await this.generateAuditReport(contract, configuration);

    return {
      contractAddress: contract.address,
      transactionHash: contract.deployTransaction.hash,
      gasUsed: contract.deployTransaction.gasLimit.toString(),
      configuration: configuration,
      auditReport: auditReport
    };
  }

  async configureWill(contract, config) {
    // Create will
    const tx1 = await contract.createWill(
      config.willId,
      config.testatorDID,
      config.triggers.primaryTrigger.triggerType,
      config.triggers.verificationRequirements.minimumVerifiers,
      ethers.utils.keccak256(ethers.utils.toUtf8Bytes(JSON.stringify(config)))
    );
    await tx1.wait();
    console.log('Will created on-chain');

    // Add executors
    for (const executor of config.executors.executorList) {
      const tx2 = await contract.grantRole(
        ethers.utils.keccak256(ethers.utils.toUtf8Bytes('EXECUTOR_ROLE')),
        executor.executorAddress
      );
      await tx2.wait();
      console.log(`Executor ${executor.executorDID} added`);
    }

    // Add beneficiaries
    for (const distribution of config.distributions) {
      for (const beneficiary of distribution.beneficiaries) {
        const vestingPeriods = beneficiary.vestingSchedule?.enabled
          ? beneficiary.vestingSchedule.schedule
          : [];

        const tx3 = await contract.addBeneficiary(
          config.willId,
          beneficiary.beneficiaryDID,
          beneficiary.beneficiaryAddress,
          beneficiary.allocation.percentage,
          beneficiary.vestingSchedule?.enabled || false,
          vestingPeriods
        );
        await tx3.wait();
        console.log(`Beneficiary ${beneficiary.beneficiaryDID} added`);
      }
    }

    // Add assets
    for (const distribution of config.distributions) {
      for (const assetRef of distribution.assetReferences) {
        const tx4 = await contract.addAsset(
          config.willId,
          assetRef.assetId,
          this.getAssetTypeEnum(assetRef.assetType),
          assetRef.assetAddress,
          ethers.utils.parseUnits(assetRef.amount.toString(), 18),
          ethers.utils.toUtf8Bytes(JSON.stringify(assetRef.metadata || {}))
        );
        await tx4.wait();
        console.log(`Asset ${assetRef.assetId} added`);
      }
    }
  }

  async validateConfiguration(config) {
    // Validate total beneficiary allocations
    for (const distribution of config.distributions) {
      const totalAllocation = distribution.beneficiaries.reduce(
        (sum, b) => sum + b.allocation.percentage,
        0
      );

      if (totalAllocation !== 100) {
        throw new Error(`Distribution ${distribution.distributionId}: Total allocation must equal 100%`);
      }
    }

    // Validate executor configurations
    if (config.executors.executorList.length < config.executors.requiredSignatures) {
      throw new Error('Required signatures exceeds number of executors');
    }

    // Validate trigger configuration
    if (!config.triggers.primaryTrigger || !config.triggers.primaryTrigger.triggerType) {
      throw new Error('Primary trigger must be defined');
    }

    console.log('Configuration validation passed');
  }

  async monitorContractEvents(contractAddress, willId) {
    const contract = await ethers.getContractAt('WIADigitalWill', contractAddress);

    // Listen for trigger activation
    contract.on('TriggerActivated', (willId, triggerType, timestamp, event) => {
      console.log(`[TRIGGER] Will ${willId} triggered at ${new Date(timestamp * 1000)}`);
      this.notifyExecutors(willId, 'TRIGGER_ACTIVATED');
    });

    // Listen for execution initiation
    contract.on('ExecutionInitiated', (willId, executionTime, event) => {
      console.log(`[EXECUTION] Will ${willId} execution started`);
      this.notifyBeneficiaries(willId, 'EXECUTION_STARTED');
    });

    // Listen for asset distributions
    contract.on('AssetDistributed', (willId, beneficiaryDID, amount, event) => {
      console.log(`[DISTRIBUTION] ${amount} distributed to ${beneficiaryDID}`);
      this.recordDistribution(willId, beneficiaryDID, amount);
    });

    // Listen for challenges
    contract.on('WillChallenged', (willId, challenger, reason, event) => {
      console.log(`[CHALLENGE] Will ${willId} challenged by ${challenger}: ${reason}`);
      this.handleChallenge(willId, challenger, reason);
    });

    console.log(`Monitoring events for contract ${contractAddress}`);
  }

  async simulateExecution(contractAddress, willId) {
    // Simulate will execution without actually executing
    const contract = await ethers.getContractAt('WIADigitalWill', contractAddress);

    const will = await contract.wills(willId);
    const trigger = await contract.triggerConditions(willId);

    console.log('Execution Simulation Report:');
    console.log(`Will Status: ${will.status}`);
    console.log(`Trigger Activated: ${trigger.activated}`);
    console.log(`Verifications: ${trigger.verificationCount}/${trigger.threshold}`);

    if (will.status === 1) { // TRIGGERED
      const timeUntilExecution = will.executionTime - Math.floor(Date.now() / 1000);
      console.log(`Time until execution: ${timeUntilExecution} seconds`);

      // Simulate distributions

3.3.3 Cross-Chain Smart Contract Coordination

// Example 3: Cross-Chain Smart Contract Execution Coordinator
const { LayerZero } = require('@layerzerolabs/solidity-examples');
const { Chainlink } = require('@chainlink/contracts');

class CrossChainWillExecutor {
  constructor(networks) {
    this.networks = networks;
    this.bridgeContracts = new Map();
    this.initializeBridges();
  }

  async initializeBridges() {
    for (const [chainName, config] of Object.entries(this.networks)) {
      const provider = new ethers.providers.JsonRpcProvider(config.rpc);
      const bridge = new ethers.Contract(
        config.bridgeAddress,
        ['function sendMessage(uint16 dstChainId, bytes payload, address refundAddress) payable'],
        provider
      );
      this.bridgeContracts.set(chainName, { bridge, provider, config });
    }
  }

  async executeMultiChainWill(willConfig) {
    console.log('Initiating multi-chain will execution...');

    // Group assets by blockchain
    const assetsByChain = this.groupAssetsByChain(willConfig.distributions);

    // Create execution plan
    const executionPlan = await this.createExecutionPlan(assetsByChain);

    // Execute on each chain
    const results = [];
    for (const [chainName, execPlan] of executionPlan.entries()) {
      try {
        const result = await this.executeOnChain(chainName, execPlan);
        results.push(result);

        // Send cross-chain messages to coordinate
        await this.coordinateWithOtherChains(chainName, result, executionPlan);
      } catch (error) {
        console.error(`Execution failed on ${chainName}:`, error);
        await this.handleExecutionFailure(chainName, execPlan, error);
      }
    }

    // Verify all executions completed
    const allSuccess = results.every(r => r.success);

    return {
      success: allSuccess,
      executionResults: results,
      timestamp: Date.now()
    };
  }

  groupAssetsByChain(distributions) {
    const assetsByChain = new Map();

    for (const dist of distributions) {
      for (const assetRef of dist.assetReferences) {
        const chain = assetRef.blockchain || 'ETHEREUM';

        if (!assetsByChain.has(chain)) {
          assetsByChain.set(chain, []);
        }

        assetsByChain.get(chain).push({
          asset: assetRef,
          beneficiaries: dist.beneficiaries,
          executionSequence: dist.executionSequence
        });
      }
    }

    return assetsByChain;
  }

  async executeOnChain(chainName, executionPlan) {
    const { bridge, provider, config } = this.bridgeContracts.get(chainName);

    // Get will contract on this chain
    const willContract = new ethers.Contract(
      config.willContractAddress,
      ['function executeWill(string willId) external'],
      provider.getSigner()
    );

    // Execute will
    const tx = await willContract.executeWill(executionPlan.willId);
    const receipt = await tx.wait();

    console.log(`Executed on ${chainName}: ${receipt.transactionHash}`);

    return {
      success: true,
      chain: chainName,
      txHash: receipt.transactionHash,
      gasUsed: receipt.gasUsed.toString(),
      blockNumber: receipt.blockNumber
    };
  }

  async coordinateWithOtherChains(sourceChain, result, executionPlan) {
    // Send confirmation messages to other chains
    const { bridge, config } = this.bridgeContracts.get(sourceChain);

    for (const [targetChain, plan] of executionPlan.entries()) {
      if (targetChain !== sourceChain) {
        const message = ethers.utils.defaultAbiCoder.encode(
          ['string', 'string', 'bytes32', 'uint256'],
          [result.chain, plan.willId, result.txHash, result.blockNumber]
        );

        const dstChainId = this.getChainId(targetChain);
        const tx = await bridge.sendMessage(dstChainId, message, config.refundAddress, {
          value: ethers.utils.parseEther('0.1') // Bridge fee
        });

        await tx.wait();
        console.log(`Coordination message sent from ${sourceChain} to ${targetChain}`);
      }
    }
  }

  getChainId(chainName) {
    const chainIds = {
      'ETHEREUM': 101,
      'POLYGON': 109,
      'BINANCE_SMART_CHAIN': 102,
      'AVALANCHE': 106,
      'ARBITRUM': 110,
      'OPTIMISM': 111
    };
    return chainIds[chainName] || 0;
  }
}
