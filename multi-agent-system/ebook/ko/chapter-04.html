작업 할당 및 계획

弘益人間 (홍익인간) · 널리 인간을 이롭게 하라

효과적인 작업 할당은 작업이 에이전트 간에 공정하고 효율적으로 분배되어 개별 능력을 존중하면서 집단 생산성을 극대화하도록 보장합니다. 각 구성원이 자신의 강점에 따라 기여하는 잘 조직된 커뮤니티처럼, 다중 에이전트 시스템은 지능적인 작업 분배를 통해 최적의 결과를 달성합니다.

작업 할당 문제

작업 할당은 시스템 성능을 최적화하면서 제약 조건을 충족하는 방식으로 에이전트에 작업을 할당하는 프로세스입니다. 이것은 다중 에이전트 시스템에서 가장 근본적인 문제 중 하나입니다.

정의: 작업 할당은 에이전트 능력, 리소스 제약 및 시간적 종속성을 존중하면서 전체 시스템 유틸리티를 최적화하기 위해 어떤 에이전트가 어떤 작업을 실행해야 하는지 결정합니다.

작업 할당에서 주요 고려 사항은 다음과 같습니다:

  • 에이전트 능력: 작업을 적절한 기술을 가진 에이전트와 매칭
  • 부하 분산: 병목 현상을 방지하기 위해 작업을 균등하게 분배
  • 작업 종속성: 작업 간 우선순위 제약 존중
  • 마감 시간: 작업이 시간 제약 내에 완료되도록 보장
  • 리소스 제약: 제한된 공유 리소스 관리
  • 통신 비용: 조정 오버헤드 최소화
  • 동적 환경: 변화하는 조건과 새로운 작업에 적응

할당 전략

라운드 로빈 할당

작업이 에이전트 간에 순환적으로 분배됩니다. 간단하지만 에이전트 능력이나 현재 작업 부하를 고려하지 않습니다.

class RoundRobinAllocator {
    constructor(agents) {
        this.agents = agents;
        this.currentIndex = 0;
    }

    allocate(task) {
        const agent = this.agents[this.currentIndex];
        this.currentIndex = (this.currentIndex + 1) % this.agents.length;

        return {
            task: task,
            assignedAgent: agent.id,
            timestamp: Date.now()
        };
    }

    allocateBatch(tasks) {
        return tasks.map(task => this.allocate(task));
    }
}

능력 기반 할당

작업이 가장 잘 매칭되는 능력과 기술을 가진 에이전트에 할당됩니다.

부하 균형 할당

작업이 에이전트 작업 부하의 불균형을 최소화하도록 할당됩니다.

class LoadBalancedAllocator {
    constructor(agents) {
        this.agents = agents;
        this.workload = new Map();

        // 작업 부하 추적 초기화
        agents.forEach(agent => this.workload.set(agent.id, 0));
    }

    allocate(task) {
        // 현재 작업 부하가 가장 적은 에이전트 찾기
        let minWorkload = Infinity;
        let leastLoadedAgent = null;

        for (const agent of this.agents) {
            const currentLoad = this.workload.get(agent.id);

            if (agent.canPerform(task) && currentLoad < minWorkload) {
                minWorkload = currentLoad;
                leastLoadedAgent = agent;
            }
        }

        if (leastLoadedAgent) {
            // 작업 부하 업데이트
            const estimatedEffort = task.estimatedDuration || 1;
            this.workload.set(
                leastLoadedAgent.id,
                this.workload.get(leastLoadedAgent.id) + estimatedEffort
            );
        }

        return {
            task: task,
            assignedAgent: leastLoadedAgent?.id || null,
            currentWorkload: minWorkload,
            timestamp: Date.now()
        };
    }

    // 에이전트가 작업을 완료할 때 호출됨
    taskCompleted(agentId, task) {
        const currentLoad = this.workload.get(agentId);
        const effort = task.actualDuration || task.estimatedDuration || 1;
        this.workload.set(agentId, Math.max(0, currentLoad - effort));
    }

    getWorkloadBalance() {
        const loads = Array.from(this.workload.values());
        const avg = loads.reduce((sum, load) => sum + load, 0) / loads.length;
        const variance = loads.reduce(
            (sum, load) => sum + Math.pow(load - avg, 2),
            0
        ) / loads.length;

        return {
            average: avg,
            variance: variance,
            standardDeviation: Math.sqrt(variance)
        };
    }
}

연합 형성

일부 작업은 여러 에이전트가 함께 작업해야 합니다. 연합 형성은 복잡한 작업을 완료하기 위해 팀을 형성해야 하는 에이전트를 결정합니다.

class CoalitionFormation {
    constructor(agents, tasks) {
        this.agents = agents;
        this.tasks = tasks;
    }

    // 여러 에이전트가 필요한 작업을 위한 연합 형성
    formCoalitions() {
        const coalitions = [];

        for (const task of this.tasks) {
            if (task.requiredAgents > 1) {
                const coalition = this.findBestCoalition(task);
                if (coalition) {
                    coalitions.push({
                        task: task,
                        members: coalition.members,
                        value: coalition.value
                    });
                }
            } else {
                // 단일 에이전트 작업
                const agent = this.findBestAgent(task);
                if (agent) {
                    coalitions.push({
                        task: task,
                        members: [agent],
                        value: this.calculateValue(task, [agent])
                    });
                }
            }
        }

        return coalitions;
    }

    findBestCoalition(task) {
        const requiredSize = task.requiredAgents;
        const candidates = this.generateCombinations(this.agents, requiredSize);

        let bestCoalition = null;
        let bestValue = -Infinity;

        for (const coalition of candidates) {
            // 연합이 작업을 수행할 수 있는지 확인
            if (!this.canPerformTask(coalition, task)) continue;

            const value = this.calculateCoalitionValue(coalition, task);

            if (value > bestValue) {
                bestValue = value;
                bestCoalition = coalition;
            }
        }

        return bestCoalition ? {
            members: bestCoalition,
            value: bestValue
        } : null;
    }

    canPerformTask(coalition, task) {
        const combinedSkills = new Set();
        coalition.forEach(agent => {
            agent.getSkills().forEach(skill => combinedSkills.add(skill));
        });

        return task.requiredSkills.every(skill => combinedSkills.has(skill));
    }

    calculateCoalitionValue(coalition, task) {
        // 가치 = 작업 보상 - 연합 비용
        const reward = task.reward || 100;
        const cost = coalition.reduce(
            (sum, agent) => sum + agent.getCost(task),
            0
        );

        // 보완적 기술에 대한 보너스
        const synergyBonus = this.calculateSynergy(coalition, task);

        return reward - cost + synergyBonus;
    }

    calculateSynergy(coalition, task) {
        // 보완적 기술을 가진 에이전트가 더 잘 작업함
        const skillCoverage = new Set();
        coalition.forEach(agent => {
            agent.getSkills().forEach(skill => skillCoverage.add(skill));
        });

        const coverageRatio = skillCoverage.size / coalition.length;
        return coverageRatio * 20; // 시너지 보너스
    }
}

동적 작업 할당

동적 환경에서는 새로운 작업이 지속적으로 도착하고 에이전트의 상태가 변경됩니다. 할당은 실시간으로 적응해야 합니다.

다중 에이전트 계획

계획은 목표를 달성하는 행동 시퀀스를 결정하는 것을 포함합니다. 다중 에이전트 시스템에서 에이전트는 계획을 조정해야 합니다.

중앙 집중식 계획

중앙 계획자가 모든 에이전트를 위한 공동 계획을 만듭니다.

분산 계획

각 에이전트가 자체 계획을 만들고 충돌과 종속성을 해결하기 위해 다른 에이전트와 조정합니다.

작업 스케줄링

스케줄링은 마감 시간, 우선순위 및 리소스 가용성을 고려하여 작업을 언제 실행해야 하는지 결정합니다.

class TaskScheduler {
    constructor() {
        this.schedule = [];
        this.resources = new Map();
    }

    // 최소 마감 시간 우선 (EDF) 스케줄링
    scheduleEDF(tasks) {
        const sorted = tasks.sort((a, b) => a.deadline - b.deadline);

        for (const task of sorted) {
            const slot = this.findEarliestSlot(task);
            if (slot) {
                this.schedule.push({
                    task: task,
                    startTime: slot.start,
                    endTime: slot.end
                });
            }
        }

        return this.schedule;
    }

    // 우선순위 기반 스케줄링
    schedulePriority(tasks) {
        const sorted = tasks.sort((a, b) => b.priority - a.priority);

        for (const task of sorted) {
            const slot = this.findEarliestSlot(task);
            if (slot) {
                this.schedule.push({
                    task: task,
                    startTime: slot.start,
                    endTime: slot.end
                });
            }
        }

        return this.schedule;
    }

    findEarliestSlot(task) {
        const duration = task.duration;
        let currentTime = Date.now();

        while (currentTime < task.deadline) {
            if (this.isSlotAvailable(currentTime, duration, task.resources)) {
                return {
                    start: currentTime,
                    end: currentTime + duration
                };
            }
            currentTime += 1000; // 매초마다 확인
        }

        return null; // 마감 전에 슬롯을 찾을 수 없음
    }

    isSlotAvailable(startTime, duration, requiredResources) {
        const endTime = startTime + duration;

        // 필요한 리소스가 사용 가능한지 확인
        for (const [resource, amount] of Object.entries(requiredResources)) {
            if (!this.checkResourceAvailability(resource, amount, startTime, endTime)) {
                return false;
            }
        }

        return true;
    }
}

요약

  • 작업 할당은 시스템 성능을 최적화하기 위해 에이전트에 작업을 할당
  • 라운드 로빈은 간단한 평등 분배를 제공하지만 능력을 무시
  • 능력 기반 할당은 작업을 에이전트 전문성과 매칭
  • 부하 균형은 에이전트 간에 작업을 균등하게 분배
  • 연합 형성은 복잡한 작업을 위한 에이전트 팀을 생성
  • 동적 할당은 변화하는 조건과 새로운 작업에 적응
  • 중앙 집중식 계획은 최적화를 통해 전역 계획을 생성
  • 분산 계획은 조정을 통한 자율적 에이전트 계획을 가능하게 함
  • 스케줄링은 제약 조건을 고려하여 작업 실행 타이밍을 결정
  • 효과적인 할당은 공정하고 효율적인 작업 분배를 통해 弘益人間을 구현
