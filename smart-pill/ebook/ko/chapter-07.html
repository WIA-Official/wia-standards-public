7.1 FDA 승인 프로세스
--------------------------------------------------------------------------------

미국에서 스마트 알약은 의료기기로 FDA 규제를 받습니다.

[ID] 의료기기 분류:

    FDA_DeviceClassification {
      classI: {
        risk:        "낮음"
        controls:    "일반 관리 (General Controls)"
        examples:    "혀압자, 붕대"
        pathway:     "510(k) 면제 가능"
        smartPill:   "해당 없음 (위험 더 높음)"
      }

      classII: {
        risk:        "중간"
        controls:    "일반 + 특수 관리 (Special Controls)"
        examples: [
          "임신 테스트 키트",
          "전동 휠체어",
          "일부 진단 장치"
        ]
        pathway:     "510(k) 사전통지"
        smartPill: {
          examples: [
            "약물 순응도 센서 (Proteus)",
            "일부 단순 센서 캡슐"
          ]
          condition:   "predicate 장치 존재"
        }
      }

      classIII: {
        risk:        "높음"
        controls:    "사전 승인 (Premarket Approval)"
        examples: [
          "심장 판막",
          "심박 조율기",
          "삽입형 센서"
        ]
        pathway:     "PMA (가장 엄격)"
        smartPill: {
          examples: [
            "캡슐 내시경 (PillCam)",
            "진단 + 치료 복합 장치",
            "신규 기술"
          ]
          requirement: "안전성 및 효능 입증"
        }
      }
    }

[ID] 510(k) 경로:

    510k_Pathway {
      definition:  "기존 시판 장치와 실질적 동등성 입증"

      substantial_equivalence: {
        predicate:   "이미 시판 중인 합법 장치"
        comparison: {
          intended_use:    "동일"
          technology:      "동일 또는 다르지만 안전/효능 영향 없음"
          performance:     "동등 이상"
          biocompatibility: "동등 이상"
        }
      }

      submission: {
        contents: [
          "장치 설명 및 사용 목적",
          "Predicate 장치와 비교",
          "성능 시험 데이터",
          "생물학적 평가 (ISO 10993)",
          "소프트웨어 문서 (해당 시)",
          "전기 안전/EMC",
          "살균/멸균 검증 (해당 시)",
          "표시 및 사용 설명서"
        ]

        testing: {
          bench:       "재료, 기계적, 전기적"
          animal:      "생물학적 안전성 (필요 시)"
          clinical:    "일반적으로 불필요 (동등성 시)"
        }

        review_time: "90일 (FDA 목표) - 실제 4-12개월"
      }

      smartPill_example: {
        device:      "새로운 pH 센서 캡슐"
        predicate:   "SmartPill (2006 승인)"
        comparison: {
          use:       "동일 (GI pH, 압력, 온도)"
          sensors:   "유사 (ISFET pH, MEMS 압력)"
          wireless:  "동일 주파수 (433 MHz)"
          size:      "유사 (11 mm)"
        }
        differences: {
          battery:   "향상된 배터리 (더 긴 수명)"
          testing:   "배터리 안전성 시험"
        }
        outcome:     "510(k) 승인 가능"
      }

      denovo_pathway: {
        scenario:    "predicate 없지만 Class II 적절"
        process: [
          "510(k) 제출 → NSE (Not Substantially Equivalent)",
          "De Novo 요청",
          "FDA 검토 및 Class II 재분류",
          "승인 → 향후 predicate로 사용 가능"
        ]
        example:     "Proteus 센서 (최초 섭취형 센서)"
        timeline:    "150일 FDA 목표"
      }
    }

      post_approval: {

      smartPill_example: {

          endpoints: [

7.2 임상 시험 요구사항
--------------------------------------------------------------------------------

PMA나 높은 위험 장치는 인간 대상 임상 시험이 필수입니다.

[ID] IDE (Investigational Device Exemption):

    IDE_Process {
      purpose:     "임상 시험을 위해 미승인 장치 사용 허가"

      types: {
        nonsignificant_risk: {
          abbreviation: "NSR"
          criteria: [
            "이식형 아님",
            "생명/건강 지원 아님",
            "중대한 손상 가능성 낮음"
          ]
          approval:    "IRB만 (FDA 면제)"
          example:     "일부 단순 진단 센서 캡슐"
        }

        significant_risk: {
          abbreviation: "SR"
          criteria:     "NSR 조건 불충족"
          approval:     "IRB + FDA IDE 승인"
          example:      "약물 방출 + 진단 복합 캡슐"
        }
      }

      ide_application: {
        contents: [
          "연구 계획서 (protocol)",
          "위험 분석",
          "장치 설명 및 제조",
          "사전 동물 시험 데이터",
          "임상의 자격",
          "동의서 양식",
          "IRB 정보",
          "모니터링 계획"
        ]

        review: {
          fda:       "30일 검토"
          outcome: [
            "승인 (침묵 = 승인)",
            "임상 보류 (추가 정보 요청)",
            "disapproval"
          ]
        }
      }

      conduct: {
        gcp:         "Good Clinical Practice 준수"
        monitoring:  "독립적 모니터 (가능 시)"
        reporting: {
          progress:  "연간 보고서"
          sae:       "심각한 부작용 즉시 (10 일)"
          deviations: "프로토콜 이탈 기록"
        }
      }
    }

[ID] 임상 시험 설계:

    ClinicalTrialDesign {
      phase_analogy: {
        note:        "의료기기는 약물과 다르지만 유사 단계"

        phase1: {
          name:      "초기 안전성/타당성"
          subjects:  "10-30"
          design:    "오픈 라벨, 단일군"
          goals: [
            "안전성 (부작용)",
            "기술적 성공",
            "초기 효능 신호"
          ]
          example:   "PillCam 초기 연구 (건강 지원자)"
        }

        phase2: {
          name:      "pilot 효능"
          subjects:  "50-100"
          design:    "단일군 또는 소규모 무작위"
          goals: [
            "효능 추정 (표본 크기 계산)",
            "최적 용도 정의",
            "안전성 확장"
          ]
        }

        phase3: {
          name:      "피벗 효능"
          subjects:  "100-500+ (통계적 파워)"
          design:    "무작위, 제어, 다기관"
          goals: [
            "효능 확정 (1차 종말점)",
            "안전성 프로파일",
            "PMA 승인 기반"
          ]
        }

        phase4: {
          name:      "시판 후"
          subjects:  "수백-수천"
          goals: [
            "실제 환경 성능",
            "장기 안전성",
            "새로운 적응증"
          ]
        }
      }

      control_groups: {
        gold standard: {
          description: "기존 표준 치료"
          example:     "캡슐 내시경 vs. 소장경"
          advantage:   "직접 비교, 신뢰성"
        }

        sham: {
          description: "위약 유사 (가짜 장치)"
          challenge:   "의료기기에 어려움 (맹검)"
          example:     "비활성 캡슐"
        }

        historical: {
          description: "문헌 데이터와 비교"
          limitation:  "편향 가능성"
          use:         "드문 질환, 윤리적 이유"
        }

        self_controlled: {
          description: "환자 본인이 대조군"
          design:      "전후 비교"
          advantage:   "개인차 제거"
          limitation:  "시간 효과, 학습"
        }
      }

      endpoints: {
        primary: {
          definition:  "연구의 주요 효능 척도"
          characteristics: [
            "사전 정의 및 정당화",
            "객관적, 측정 가능",
            "임상적 의미",
            "단일 (통계적 파워)"
          ]
          examples: {
            diagnostic: {
              metric:  "진단 정확도"
              measure: "민감도, 특이도, AUC"
            }
            monitoring: {
              metric:  "검출률"
              measure: "이벤트 탐지 %"
            }
            therapeutic: {
              metric:  "임상 반응"
              measure: "증상 개선, 질병 활성도"
            }
          }
        }

        secondary: {
          purpose:     "추가 이익, 안전성"
          examples: [
            "환자 만족도",
            "삶의 질 (QoL)",
            "비용 효과",
            "아형 분석"
          ]
        }

        safety: {
          aes: {
            name:      "부작용 (Adverse Events)"
            grading:   "경증, 중등도, 중증"
            relation:  "장치와 관련성 평가"
          }

          saes: {
            name:      "심각한 부작용"
            definition: [
              "사망",
              "생명 위협",
              "입원 (또는 연장)",
              "영구 손상",
              "선천적 이상"
            ]
            reporting:  "즉시 (24-48시간)"
          }

          device_specific: [
            "폐색/천공",
            "알러지 반응",
            "배터리 누출",
            "센서 오작동"
          ]
        }
      }

      statistical_considerations: {
        sample_size: {
          factors: [
            "예상 효과 크기",
            "유의 수준 (α = 0.05)",
            "검정력 (1-β = 0.80-0.90)",
            "탈락률"
          ]
          calculation: "통계 소프트웨어 (nQuery, PASS)"
        }

        analysis_plan: {
          populations: {
            itt:       "Intention-To-Treat (모든 무작위화)"
            pp:        "Per-Protocol (프로토콜 준수자)"
            safety:    "최소 1회 장치 노출"
          }

          hypothesis_testing: {
            superiority: "새 장치 > 대조"
            noninferiority: "새 장치 < 대조 (마진 내)"
            equivalence:    "새 장치 = 대조 (마진 내)"
          }

          multiplicity: {
            problem:    "다중 검정 → 1종 오류 증가"
            adjustment: "Bonferroni, Holm, FDR"
          }
        }

        interim_analysis: {
          purpose: [
            "조기 효능 (조기 중단)",
            "무익성 (중단)",
            "안전성 (중단)"
          ]
          dmb:         "Data Monitoring Board (독립)"
          alpha_spending: "순차적 검정 방법"
        }
      }

      patient_population: {
        inclusion: {
          criteria: [
            "적응증 해당 (질병/상태)",
            "연령 범위",
            "진단 확인",
            "동의 능력"
          ]
        }

        exclusion: {
          criteria: [
            "금기증 (폐색 위험 등)",
            "임신/수유",
            "심각한 동반 질환",
            "최근 다른 연구 참여",
            "알러지"
          ]
        }

        recruitment: {
          strategies: [
            "의뢰 네트워크",
            "환자 레지스트리",
            "광고 (IRB 승인)",
            "커뮤니티 아웃리치"
          ]
          challenges:   "느린 등록, 경쟁 연구"
        }

        informed_consent: {
          elements: [
            "연구 목적 및 절차",
            "위험 및 이익",
            "대안",

7.3 국제 규제 표준
--------------------------------------------------------------------------------

스마트 알약을 전 세계 시장에 출시하려면 다양한 규제 체계를 준수해야 합니다.

[ID] 유럽 연합 (EU):

    EU_Regulations {
      mdr: {
        fullName:    "Medical Device Regulation (EU) 2017/745"
        effective:   "2021년 5월"
        replaces:    "MDD (Medical Device Directive)"

        classification: {
          rules:     "분류 규칙 (Annex VIII)"
          smartPill: "일반적으로 Class IIa 또는 IIb"

          classIIa: {
            examples:  "단순 진단 센서"
            route:     "NB 인증 + 기술 문서"
          }

          classIIb: {
            examples:  "약물 전달 + 진단"
            route:     "NB 인증 + 더 엄격한 검토"
          }
        }

        conformity_assessment: {
          notified_body: {
            role:      "독립적 평가 기관"
            designation: "EU 국가가 지정"
            tasks: [
              "기술 문서 검토",
              "품질 시스템 감사",
              "설계 검사 (해당 시)",
              "CE 마킹 승인"
            ]
          }

          technical_documentation: {
            contents: [
              "장치 설명 및 사양",
              "설계 및 제조",
              "일반 안전/성능 요구사항 (GSPR)",
              "이익-위험 분석",
              "임상 평가 (MEDDEV 2.7/1)",
              "표시 및 사용 설명",
              "사후 시판 계획"
            ]
          }

          clinical_evaluation: {
            requirement: "모든 장치"
            methods: [
              "문헌 검토 (동등 장치)",
              "임상 조사 (신규)"
            ]
            cer:         "Clinical Evaluation Report"
            updates:     "정기적 (연간 또는 PMCF 데이터)"
          }
        }

        ce_marking: {
          meaning:     "EU 요구사항 준수"
          affixing:    "장치 및 포장에 부착"
          validity:    "NB 인증서 유효 기간 (5년)"
          surveillance: "시판 후 감시 (PMS), PMCF"
        }

        unique_device_identification: {
          abbreviation: "UDI"
          components: {
            udi_di:    "Device Identifier (모델)"
            udi_pi:    "Production Identifier (로트, 일련번호)"
          }
          database:    "EUDAMED (EU 데이터베이스)"
        }
      }

      clinical_investigation: {
        regulation:  "ISO 14155"
        approval: {
          competent_authority: "국가 규제 기관"
          ethics_committee:    "윤리 위원회 (EC)"
          timeline:             "60일 (침묵 = 승인)"
        }

        conduct: {
          gcp:         "ISO 14155 준수"
          monitoring:  "스폰서 또는 CRO"
          reporting: {
            sae:       "즉시 (CA + EC)"
            annual:    "안전성 보고서"
            final:     "최종 보고서"
          }
        }
