7.3 Bash Integration

Using WIA-INTENT from shell scripts for system automation.

CLI Tool

#!/bin/bash
# wia-intent - Command-line tool for WIA-INTENT

set -euo pipefail

INTENT_FILE=""
INPUTS_FILE=""
OUTPUT_FILE=""
VERBOSE=false

# Parse command line arguments
while [[ $# -gt 0 ]]; do
  case $1 in
    -i|--intent)
      INTENT_FILE="$2"
      shift 2
      ;;
    -d|--data)
      INPUTS_FILE="$2"
      shift 2
      ;;
    -o|--output)
      OUTPUT_FILE="$2"
      shift 2
      ;;
    -v|--verbose)
      VERBOSE=true
      shift
      ;;
    *)
      echo "Unknown option: $1"
      exit 1
      ;;
  esac
done

# Validate inputs
if [[ -z "$INTENT_FILE" ]]; then
  echo "Error: Intent file required (-i)"
  exit 1
fi

# Read intent
INTENT_CODE=$(< "$INTENT_FILE")

# Read inputs if provided
INPUTS="{}"
if [[ -n "$INPUTS_FILE" ]]; then
  INPUTS=$(< "$INPUTS_FILE")
fi

# Execute intent via API
RESPONSE=$(curl -s -X POST http://localhost:8080/api/v1/execute \
  -H "Content-Type: application/json" \
  -d "{
    \"intent\": $(jq -Rs . <<< "$INTENT_CODE"),
    \"inputs\": $INPUTS
  }")

# Extract result
RESULT=$(echo "$RESPONSE" | jq -r '.result.value')
CERTAINTY=$(echo "$RESPONSE" | jq -r '.metrics.certainty')

if [[ "$VERBOSE" == true ]]; then
  echo "Certainty: $CERTAINTY"
  echo "Execution time: $(echo "$RESPONSE" | jq -r '.metrics.execution_time_ms')ms"
fi

# Output result
if [[ -n "$OUTPUT_FILE" ]]; then
  echo "$RESULT" > "$OUTPUT_FILE"
else
  echo "$RESULT"
fi

Example Usage

#!/bin/bash
# backup-system.sh - System backup using WIA-INTENT

# Define intent
cat > backup-intent.wia <<'EOF'
intent BackupSystem {
  given: source_dirs, backup_destination
  want: backup completed

  constraints {
    compression: gzip
    encryption: aes256
    verification: checksum
    time: <= 1.hour
  }

  fallback {
    if disk_full:
      cleanup_old_backups
      retry
    if network_error:
      use_local_backup
  }
}
EOF

# Define inputs
cat > backup-data.json <<EOF
{
  "source_dirs": ["/etc", "/home", "/var/www"],
  "backup_destination": "/backup/$(date +%Y%m%d)"
}
EOF

# Execute intent
wia-intent \
  --intent backup-intent.wia \
  --data backup-data.json \
  --output backup-result.json \
  --verbose

# Check result
if jq -e '.status == "success"' backup-result.json > /dev/null; then
  echo "Backup completed successfully"
  # Send notification
  notify-send "Backup Complete" "System backup finished"
else
  echo "Backup failed"
  exit 1
fi

7.5 Testing the Implementation

Unit Tests (Python)

import unittest
from intent_runner import IntentRunner, Intent

class TestIntentRunner(unittest.TestCase):
    def setUp(self):
        self.runner = IntentRunner()

    def test_simple_intent(self):
        intent_code = """
        intent DoubleNumbers {
          given: numbers
          want: each number doubled
        }
        """

        result = self.runner.execute(
            intent_code,
            inputs={'numbers': [1, 2, 3]}
        )

        self.assertEqual(result.value, [2, 4, 6])
        self.assertGreaterEqual(result.certainty, 0.9)

    def test_constraint_violation(self):
        intent_code = """
        intent SlowOperation {
          want: result
          constraints {
            time: <= 1.milliseconds
          }
        }
        """

        with self.assertRaises(ConstraintViolationError):
            self.runner.execute(intent_code, inputs={})

    def test_fallback(self):
        intent_code = """
        intent RiskyOperation {
          want: result
          certainty: >= 0.99
          fallback {
            if certainty < 0.99:
              return "fallback_result"
          }
        }
        """

        result = self.runner.execute(intent_code, inputs={})
        self.assertEqual(result.value, "fallback_result")

if __name__ == '__main__':
    unittest.main()
