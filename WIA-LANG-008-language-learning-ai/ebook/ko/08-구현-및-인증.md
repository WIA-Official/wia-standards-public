# src/services/speech_recognition.py
import asyncio
from typing import AsyncGenerator
import numpy as np
from faster_whisper import WhisperModel

class SpeechRecognitionService:
    """실시간 음성 인식 서비스"""

    def __init__(self, model_size: str = "medium"):
        self.model = WhisperModel(
            model_size,
            device="cuda",
            compute_type="float16"
        )
        self.supported_languages = [
            'ko', 'en', 'ja', 'zh', 'es', 'fr', 'de', 'ar'
        ]

    async def transcribe_stream(
        self,
        audio_stream: AsyncGenerator[bytes, None],
        language: str
    ) -> AsyncGenerator[TranscriptionResult, None]:
        """스트리밍 음성 인식"""
        audio_buffer = bytearray()
        chunk_duration = 1.0  # 1초

        async for audio_chunk in audio_stream:
            audio_buffer.extend(audio_chunk)

            # 충분한 오디오가 모이면 인식 수행
            if len(audio_buffer) >= 16000 * 2 * chunk_duration:  # 16kHz, 16-bit
                audio_np = np.frombuffer(audio_buffer, dtype=np.int16)
                audio_float = audio_np.astype(np.float32) / 32768.0

                # Whisper 인식
                segments, info = self.model.transcribe(
                    audio_float,
                    language=language,
                    vad_filter=True,
                    beam_size=5
                )

                for segment in segments:
                    yield TranscriptionResult(
                        text=segment.text,
                        start=segment.start,
                        end=segment.end,
                        confidence=segment.avg_logprob,
                        words=self._extract_words(segment)
                    )

                # 버퍼 클리어 (약간의 오버랩 유지)
                overlap = int(16000 * 2 * 0.5)  # 0.5초 오버랩
                audio_buffer = audio_buffer[-overlap:]

    def _extract_words(self, segment) -> list:
        """단어 수준 타임스탬프 추출"""
        words = []
        if hasattr(segment, 'words'):
            for word in segment.words:
                words.append({
                    'word': word.word,
                    'start': word.start,
                    'end': word.end,
                    'probability': word.probability
                })
        return words


# 발음 평가 서비스
class PronunciationAssessmentService:
    """발음 평가 서비스"""

    def __init__(self):
        self.phoneme_model = load_phoneme_model()
        self.prosody_model = load_prosody_model()

    async def assess(
        self,
        audio_data: np.ndarray,
        target_text: str,
        target_language: str
    ) -> PronunciationAssessment:
        """발음 평가 수행"""

        # 1. 음소 정렬
        alignment = await self.align_phonemes(audio_data, target_text, target_language)

        # 2. 음소별 점수 계산
        phoneme_scores = await self.score_phonemes(audio_data, alignment)

        # 3. 운율 평가 (억양, 강세, 리듬)
        prosody_scores = await self.evaluate_prosody(audio_data, target_text)

        # 4. 전체 점수 계산
        overall_score = self.calculate_overall_score(phoneme_scores, prosody_scores)

        # 5. 피드백 생성
        feedback = self.generate_feedback(phoneme_scores, prosody_scores)

        return PronunciationAssessment(
            overall_score=overall_score,
            phoneme_scores=phoneme_scores,
            prosody_scores=prosody_scores,
            fluency_score=prosody_scores['fluency'],
            feedback=feedback,
            word_scores=self.calculate_word_scores(alignment, phoneme_scores)
        )

    async def align_phonemes(
        self,
        audio: np.ndarray,
        text: str,
        language: str
    ) -> list:
        """강제 정렬을 통한 음소 매핑"""
        # G2P (Grapheme to Phoneme) 변환
        phonemes = self.text_to_phonemes(text, language)

        # 음향 모델을 사용한 강제 정렬
        alignment = self.phoneme_model.align(audio, phonemes)

        return alignment

    def generate_feedback(
        self,
        phoneme_scores: list,
        prosody_scores: dict
    ) -> list:
        """개선 피드백 생성"""
        feedback = []

        # 낮은 점수 음소 피드백
        for ps in phoneme_scores:
            if ps['score'] < 70:
                feedback.append({
                    'type': 'phoneme',
                    'phoneme': ps['phoneme'],
                    'score': ps['score'],
                    'suggestion': self.get_phoneme_tip(ps['phoneme']),
                    'example_audio': self.get_example_audio(ps['phoneme'])
                })

        # 운율 피드백
        if prosody_scores['intonation'] < 70:
            feedback.append({
                'type': 'prosody',
                'aspect': 'intonation',
                'suggestion': '문장 끝의 억양에 더 주의를 기울여 보세요.'
            })

        if prosody_scores['rhythm'] < 70:
            feedback.append({
                'type': 'prosody',
                'aspect': 'rhythm',
                'suggestion': '각 음절의 길이를 더 균등하게 유지해 보세요.'
            })

        return feedback

from openai import AsyncOpenAI
from typing import AsyncGenerator

class AITutorService:
    """AI 언어 튜터 서비스"""

    def __init__(self):
        self.client = AsyncOpenAI()
        self.conversation_history = {}

    async def get_response(
        self,
        user_id: str,
        user_message: str,
        context: TutorContext
    ) -> AsyncGenerator[str, None]:
        """AI 튜터 응답 생성 (스트리밍)"""

        # 대화 히스토리 가져오기
        history = self.get_conversation_history(user_id)

        # 시스템 프롬프트 구성
        system_prompt = self.build_system_prompt(context)

        messages = [
            {"role": "system", "content": system_prompt},
            *history,
            {"role": "user", "content": user_message}
        ]

        # 스트리밍 응답
        stream = await self.client.chat.completions.create(
            model="gpt-4-turbo",
            messages=messages,
            stream=True,
            temperature=0.7,
            max_tokens=500
        )

        full_response = ""
        async for chunk in stream:
            if chunk.choices[0].delta.content:
                content = chunk.choices[0].delta.content
                full_response += content
                yield content

        # 히스토리 업데이트
        self.update_history(user_id, user_message, full_response)

    def build_system_prompt(self, context: TutorContext) -> str:
        """컨텍스트 기반 시스템 프롬프트"""
        return f"""당신은 친절하고 인내심 있는 {context.target_language} 언어 튜터입니다.

학습자 정보:
- 현재 레벨: {context.proficiency_level}
- 모국어: {context.native_language}

4. 문화적 컨텍스트나 흥미로운 정보를 자연스럽게 포함하세요.
5. 너무 긴 응답은 피하고, 대화를 이어갈 수 있는 질문을 포함하세요.

현재 학습 주제: {context.current_topic}
세션 목표: {context.session_goal}
"""

        self,

    ) -> Exercise:

연습 유형: {exercise_type}

JSON 형식으로 출력하세요:
{{
  "instruction": "연습 지시사항",
  "content": "연습 내용",
  "expected_answer": "예상 답변",

다음 항목을 평가하세요:
1. 정확성 (0-100)

      { type: 'pronunciation_practice', duration: 300 }
    ];

    // 2. 동기화 요청
    const syncResult = await client.sync.upload({
      userId,
      events: offlineEvents,
      lastSyncTime: new Date(Date.now() - 24 * 60 * 60 * 1000)
    });

    expect(syncResult.success).toBe(true);
    expect(syncResult.conflictsResolved).toBeDefined();

    // 3. 진도 확인
    const progress = await client.progress.get(userId);
    expect(progress.completedLessons).toContain('L001');
  });
});
