```javascript
// WebSocket 연결 설정
class WIALangWebSocket {
  constructor(userId, accessToken) {
    this.userId = userId;
    this.accessToken = accessToken;
    this.ws = null;
    this.messageHandlers = new Map();
  }

  connect() {
    const wsUrl = `wss://realtime.wia.org/lang-008/v1/stream?user_id=${this.userId}`;

    this.ws = new WebSocket(wsUrl);

    // 인증
    this.ws.addEventListener('open', () => {
      this.send({
        type: 'auth',
        token: this.accessToken
      });
    });

    // 메시지 수신
    this.ws.addEventListener('message', (event) => {
      const message = JSON.parse(event.data);
      this.handleMessage(message);
    });

    // 에러 처리
    this.ws.addEventListener('error', (error) => {
      console.error('WebSocket error:', error);
      this.reconnect();
    });

    // 연결 종료
    this.ws.addEventListener('close', () => {
      console.log('WebSocket closed');
      this.reconnect();
    });
  }

  send(message) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(message));
    }
  }

  handleMessage(message) {
    const handler = this.messageHandlers.get(message.type);
    if (handler) {
      handler(message.data);
    }
  }

  on(messageType, handler) {
    this.messageHandlers.set(messageType, handler);
  }

  reconnect() {
    setTimeout(() => this.connect(), 5000);
  }
}
```

### 실시간 발음 피드백

```python
# 서버 측 WebSocket 핸들러
import asyncio
from fastapi import WebSocket

class RealTimePronunciationFeedback:
    """실시간 발음 피드백 시스템"""

    def __init__(self):
        self.asr_model = load_streaming_asr_model()
        self.pronunciation_assessor = PronunciationAssessor()

    async def handle_speech_stream(self, websocket: WebSocket, user_id: str):
        """음성 스트림 처리"""
        await websocket.accept()

        audio_buffer = bytearray()
        chunk_size = 16000 * 2  # 1초분 (16kHz, 16-bit)

        try:
            while True:
                # 오디오 청크 수신
                audio_chunk = await websocket.receive_bytes()
                audio_buffer.extend(audio_chunk)

                # 버퍼가 충분히 차면 처리
                if len(audio_buffer) >= chunk_size:
                    audio_segment = bytes(audio_buffer[:chunk_size])
                    audio_buffer = audio_buffer[chunk_size:]

                    # 비동기로 음성 인식
                    transcription_task = asyncio.create_task(
                        self.asr_model.transcribe_async(audio_segment)
                    )

                    # 중간 결과 전송
                    interim_result = await transcription_task

                    await websocket.send_json({
                        "type": "interim_transcription",
                        "data": {
                            "text": interim_result.text,
                            "confidence": interim_result.confidence
                        }
                    })

                    # 발음 평가 (백그라운드)
                    if interim_result.is_word_complete:
                        pronunciation_score = await self.assess_pronunciation_async(
                            audio_segment,
                            interim_result.text
                        )

                        if pronunciation_score.needs_feedback:
                            await websocket.send_json({
                                "type": "pronunciation_feedback",
                                "data": {
                                    "phoneme": pronunciation_score.problematic_phoneme,
                                    "score": pronunciation_score.score,
                                    "tip": pronunciation_score.improvement_tip
                                }
                            })

        except Exception as e:
            print(f"WebSocket error: {e}")
        finally:
            await websocket.close()
```

```typescript
interface SyncProtocol {

  sync_request: {

  conflict_resolution: {
    type: "resolve_conflict";
    data: {
```

**동기화 구현 예시**:

```javascript
class SyncManager {
  constructor(api, localDB) {
    this.api = api;
    this.localDB = localDB;
    this.syncInterval = 5 * 60 * 1000;  // 5분
    this.isSyncing = false;
  }

  startPeriodicSync() {
    setInterval(() => this.sync(), this.syncInterval);

    // 네트워크 연결 복구 시 즉시 동기화
    window.addEventListener('online', () => this.sync());
  }

  async sync() {
    if (this.isSyncing) return;
    this.isSyncing = true;

    try {
      // 1. 로컬에 저장된 pending 이벤트 가져오기
      const pendingEvents = await this.localDB.getPendingEvents();
      const lastSyncTime = await this.localDB.getLastSyncTime();

      // 2. 서버에 동기화 요청
      const syncResponse = await this.api.sync({
        last_sync_timestamp: lastSyncTime,
        pending_events: pendingEvents
      });

      // 3. 서버 업데이트 로컬에 반영
      await this.applyServerUpdates(syncResponse.updates);

      // 4. 충돌 처리
      if (syncResponse.conflicts.length > 0) {
        await this.resolveConflicts(syncResponse.conflicts);
      }

      // 5. Pending 이벤트 제거 및 동기화 시간 갱신
      await this.localDB.clearPendingEvents();
      await this.localDB.setLastSyncTime(syncResponse.server_timestamp);

      console.log('Sync completed successfully');
    } catch (error) {
      console.error('Sync failed:', error);
      // 다음 동기화 시도까지 로컬 데이터 유지
    } finally {
      this.isSyncing = false;
    }
  }

  async applyServerUpdates(updates) {
    // 새 콘텐츠
    for (const content of updates.new_content) {
      await this.localDB.saveContent(content);
    }

    // 진도 업데이트
    for (const progress of updates.updated_progress) {
      await this.localDB.updateProgress(progress);
    }

    // 어휘 업데이트
    for (const vocabUpdate of updates.vocabulary_updates) {
      await this.localDB.updateVocabulary(vocabUpdate);
    }
  }

  async resolveConflicts(conflicts) {
    for (const conflict of conflicts) {
      let resolution;

      // 충돌 유형별 처리
      switch (conflict.type) {
        case 'progress_score':
          // 점수 충돌: 높은 점수 선택
          resolution = conflict.server_value > conflict.client_value
            ? 'server_wins'
            : 'client_wins';
          break;

        case 'vocabulary_mastery':
          // 숙련도 충돌: 더 많은 연습 반영 (repetitions가 높은 것)
          resolution = conflict.server_data.repetitions > conflict.client_data.repetitions
            ? 'server_wins'
            : 'client_wins';
          break;

        case 'learning_event':
          // 이벤트 충돌: 병합
          resolution = 'merge';
          break;

        default:
          // 기본: 서버 우선
          resolution = 'server_wins';
      }

      await this.api.resolveConflict(conflict.id, resolution);
    }
  }
}
```

### 충돌 해결 전략

```python
class ConflictResolver:
    """데이터 충돌 해결기"""

    @staticmethod
    def resolve_progress_conflict(server_data, client_data):
        """진도 데이터 충돌 해결"""
        # 최신 타임스탬프 우선
        if server_data.timestamp > client_data.timestamp:
            return {
                "strategy": "server_wins",
                "result": server_data
            }
        else:
            return {
                "strategy": "client_wins",
                "result": client_data
            }

    @staticmethod
    def resolve_vocabulary_conflict(server_card, client_card):
        """어휘 카드 충돌 해결"""
        # 더 많은 복습 횟수를 가진 쪽 선택
        if server_card.repetitions >= client_card.repetitions:
            return {
                "strategy": "server_wins",
                "result": server_card,
                "reason": "Server has more repetitions"
            }
        else:
            return {
                "strategy": "client_wins",
                "result": client_card,
                "reason": "Client has more repetitions"
            }

    @staticmethod
    def merge_learning_events(server_events, client_events):
        """학습 이벤트 병합"""
        # 타임스탬프 기준으로 병합 및 중복 제거
        all_events = server_events + client_events

        # event_id로 중복 제거
        unique_events = {event.event_id: event for event in all_events}

        # 타임스탬프 순 정렬
        merged = sorted(
            unique_events.values(),
            key=lambda e: e.timestamp
        )

        return {
            "strategy": "merge",
            "result": merged
        }
```

```javascript
class OfflineManager {
  constructor() {
    this.db = null;
    this.initIndexedDB();
  }

  async initIndexedDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open('WIALang008', 1);

      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve();
      };

      request.onupgradeneeded = (event) => {
        const db = event.target.result;

        // Object stores 생성
        if (!db.objectStoreNames.contains('content')) {
          const contentStore = db.createObjectStore('content', { keyPath: 'content_id' });
          contentStore.createIndex('downloaded', 'downloaded', { unique: false });
        }

        if (!db.objectStoreNames.contains('events')) {
          const eventsStore = db.createObjectStore('events', {
            keyPath: 'event_id',
            autoIncrement: true
          });
          eventsStore.createIndex('synced', 'synced', { unique: false });
          eventsStore.createIndex('timestamp', 'timestamp', { unique: false });
        }

        if (!db.objectStoreNames.contains('vocabulary')) {
          db.createObjectStore('vocabulary', { keyPath: 'card_id' });
        }

        if (!db.objectStoreNames.contains('progress')) {
          db.createObjectStore('progress', { keyPath: 'user_id' });
        }
      };
    });
  }

  // 콘텐츠 다운로드 (오프라인용)
  async downloadContent(contentId) {
    const content = await this.fetchContentFromServer(contentId);

    // 미디어 파일 다운로드
    if (content.audio) {
      content.audio.local_path = await this.downloadMedia(content.audio.url);
    }
    if (content.video) {
      content.video.local_path = await this.downloadMedia(content.video.url);
    }

    // IndexedDB에 저장
    const transaction = this.db.transaction(['content'], 'readwrite');
    const store = transaction.objectStore('content');
    content.downloaded = true;
    content.download_date = new Date().toISOString();

    await store.put(content);

    return content;
  }

  // 오프라인 이벤트 기록
  async logEventOffline(event) {
    event.synced = false;
    event.timestamp = new Date().toISOString();

    const transaction = this.db.transaction(['events'], 'readwrite');
    const store = transaction.objectStore('events');

    return new Promise((resolve, reject) => {
      const request = store.add(event);
      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }

  // 동기화 대기 이벤트 조회
  async getPendingEvents() {
    const transaction = this.db.transaction(['events'], 'readonly');
    const store = transaction.objectStore('events');
    const index = store.index('synced');

    return new Promise((resolve, reject) => {
      const request = index.getAll(false);  // synced === false
      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }

  // 이벤트 동기화 완료 표시
  async markEventsSynced(eventIds) {
    const transaction = this.db.transaction(['events'], 'readwrite');
    const store = transaction.objectStore('events');

    for (const eventId of eventIds) {
      const request = store.get(eventId);
      request.onsuccess = () => {
        const event = request.result;
        if (event) {
          event.synced = true;
          store.put(event);
        }
      };
    }
  }

  // 미디어 다운로드 (Service Worker 활용)
  async downloadMedia(url) {
    const response = await fetch(url);
    const blob = await response.blob();

    // Cache API에 저장
    const cache = await caches.open('wia-lang-008-media');
    await cache.put(url, new Response(blob));

    return url;  // 캐시된 URL 반환
  }

  // 오프라인 콘텐츠 조회
  async getOfflineContent(contentId) {
    const transaction = this.db.transaction(['content'], 'readonly');
    const store = transaction.objectStore('content');

    return new Promise((resolve, reject) => {
      const request = store.get(contentId);
      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }

  // 다운로드된 모든 콘텐츠 조회
  async getAllDownloadedContent() {
    const transaction = this.db.transaction(['content'], 'readonly');
    const store = transaction.objectStore('content');
    const index = store.index('downloaded');

    return new Promise((resolve, reject) => {
      const request = index.getAll(true);  // downloaded === true
      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }
}
```

### Service Worker를 통한 오프라인 지원

```javascript
// service-worker.js
const CACHE_VERSION = 'wia-lang-008-v1';
const STATIC_CACHE = `${CACHE_VERSION}-static`;
const DYNAMIC_CACHE = `${CACHE_VERSION}-dynamic`;

const STATIC_FILES = [
  '/',
  '/index.html',
  '/css/main.css',
  '/js/app.js',
  '/js/wia-sdk.js',
  '/offline.html'
];

// 설치
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(STATIC_CACHE).then((cache) => {
      return cache.addAll(STATIC_FILES);
    })
  );
  self.skipWaiting();
});

// 활성화
self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((cacheNames) => {
      return Promise.all(
        cacheNames
          .filter((name) => name.startsWith('wia-lang-008-') && name !== STATIC_CACHE && name !== DYNAMIC_CACHE)
          .map((name) => caches.delete(name))
      );
    })
  );
  self.clients.claim();
});

// Fetch 요청 처리
self.addEventListener('fetch', (event) => {
  const { request } = event;

  // API 요청
  if (request.url.includes('/api/')) {
    event.respondWith(
      fetch(request)
        .then((response) => {
          // 성공 시 캐시 업데이트
          const responseClone = response.clone();
          caches.open(DYNAMIC_CACHE).then((cache) => {
            cache.put(request, responseClone);
          });
          return response;
        })
        .catch(() => {
          // 오프라인 시 캐시에서 반환
          return caches.match(request).then((cachedResponse) => {
            if (cachedResponse) {
              return cachedResponse;
            }
            // 캐시에도 없으면 오프라인 페이지
            return caches.match('/offline.html');
          });
        })
    );
  }
  // 정적 파일 요청
  else {
    event.respondWith(
      caches.match(request).then((cachedResponse) => {
        return cachedResponse || fetch(request).then((response) => {
          const responseClone = response.clone();
          caches.open(DYNAMIC_CACHE).then((cache) => {
            cache.put(request, responseClone);
          });
          return response;
        });
      })
    );
  }
});

// 백그라운드 동기화
self.addEventListener('sync', (event) => {

async function syncLearningEvents() {
  // IndexedDB에서 pending 이벤트 가져오기
  const db = await openIndexedDB();
  const pendingEvents = await getPendingEvents(db);

  // 서버에 전송

        body: JSON.stringify(event)
      });

      // 성공 시 동기화 표시
      await markEventSynced(db, event.event_id);
    } catch (error) {
      console.error('Failed to sync event:', error);
      // 실패 시 다음 sync까지 대기
    }
  }
}
```

### 데이터 암호화 구현

```python
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.backends import default_backend
import os

class DataEncryption:
    """데이터 암호화 유틸리티"""

    @staticmethod
    def encrypt_audio(audio_data, user_key):
        """음성 데이터 암호화 (AES-256-GCM)"""
        # 난수 생성
        nonce = os.urandom(12)  # GCM uses 12-byte nonce

        # 암호화
        cipher = Cipher(
            algorithms.AES(user_key),
            modes.GCM(nonce),
            backend=default_backend()
        )
        encryptor = cipher.encryptor()

        ciphertext = encryptor.update(audio_data) + encryptor.finalize()

        return {
            "nonce": nonce,
            "ciphertext": ciphertext,
            "tag": encryptor.tag  # Authentication tag
        }

    @staticmethod
    def decrypt_audio(encrypted_data, user_key):
        """음성 데이터 복호화"""
        cipher = Cipher(
            algorithms.AES(user_key),
            modes.GCM(encrypted_data["nonce"], encrypted_data["tag"]),
            backend=default_backend()
        )
        decryptor = cipher.decryptor()

        plaintext = decryptor.update(encrypted_data["ciphertext"]) + decryptor.finalize()

        return plaintext

    @staticmethod
    def hash_pii(personal_info):
        """개인 정보 해싱 (SHA-256)"""
        import hashlib

        return hashlib.sha256(personal_info.encode()).hexdigest()
```

### Rate Limiting 프로토콜

```python
from datetime import datetime, timedelta
from collections import defaultdict

class RateLimiter:
    """API 요청 속도 제한"""

    def __init__(self):
        self.requests = defaultdict(list)
        self.limits = {
            "default": {"requests": 100, "window": 60},  # 100 req/min
            "speech_assessment": {"requests": 20, "window": 60},  # 20 req/min
            "ai_tutor": {"requests": 50, "window": 60}  # 50 req/min
        }

    def is_allowed(self, user_id, endpoint_type="default"):
        """요청 허용 여부 확인"""
        now = datetime.now()
        limit_config = self.limits.get(endpoint_type, self.limits["default"])

        # 사용자의 최근 요청 기록
        user_requests = self.requests[user_id]

        # 시간 윈도우 밖의 요청 제거
        cutoff_time = now - timedelta(seconds=limit_config["window"])
        user_requests = [req_time for req_time in user_requests if req_time > cutoff_time]
        self.requests[user_id] = user_requests

        # 제한 초과 확인
        if len(user_requests) >= limit_config["requests"]:
            return False, {
                "allowed": False,
                "limit": limit_config["requests"],
                "window_seconds": limit_config["window"],
                "retry_after_seconds": (user_requests[0] + timedelta(seconds=limit_config["window"]) - now).total_seconds()
            }

        # 요청 기록 추가
        user_requests.append(now)

        return True, {
            "allowed": True,
            "remaining": limit_config["requests"] - len(user_requests)
        }
```
