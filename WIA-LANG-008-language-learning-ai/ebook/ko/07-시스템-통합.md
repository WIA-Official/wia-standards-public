```typescript
    timeout: number;  // ms
  };

    useCases: ['진도 동기화', '분석 데이터', '알림'];
    protocol: 'WebSocket' | 'Webhook' | 'Message Queue';
    retryPolicy: RetryPolicy;
  };

  // 배치 통합 (대량 데이터)
  batch: {
    useCases: ['사용자 동기화', '콘텐츠 가져오기', '보고서 생성'];
    schedule: string;  // cron expression
    format: 'CSV' | 'JSON' | 'XML';
  };
}

  maxRetries: number;
  initialDelay: number;
  maxDelay: number;
  backoffMultiplier: number;
```

```typescript
// LTI 1.3 통합 구현
import { LTI13 } from 'ltijs';

class LTIIntegration {
  private lti: LTI13;

  constructor() {
    this.lti = new LTI13({
      url: process.env.LTI_URL,
      clientId: process.env.LTI_CLIENT_ID,
      platform: {
        name: 'WIA-LANG-008',
        url: 'https://wia.org',
        clientId: process.env.LTI_CLIENT_ID,
        authenticationEndpoint: 'https://lms.example.com/auth',
        accesstokenEndpoint: 'https://lms.example.com/token',
        authConfig: {
          method: 'JWK_SET',
          key: 'https://lms.example.com/.well-known/jwks.json'
        }
      }
    });
  }

  // LTI 런치 핸들러
  async handleLaunch(req: Request, res: Response): Promise<void> {
    try {
      const token = res.locals.token;

      // 사용자 정보 추출
      const user = {
        id: token.user,
        name: token.userInfo?.name,
        email: token.userInfo?.email,
        roles: token.platformContext?.roles || []
      };

      // 컨텍스트 정보 (코스, 모듈 등)
      const context = {
        courseId: token.platformContext?.context?.id,
        courseName: token.platformContext?.context?.title,
        resourceId: token.platformContext?.resource?.id
      };

      // WIA 세션 생성
      const session = await this.createLearningSession(user, context);

      // 학습 인터페이스로 리다이렉트
      res.redirect(`/learn?session=${session.id}`);

    } catch (error) {
      console.error('LTI Launch Error:', error);
      res.status(500).send('Launch failed');
    }
  }

  // 점수 전송 (Assignment and Grade Services)
  async submitGrade(
    userId: string,
    resourceId: string,
    score: number,
    maxScore: number
  ): Promise<void> {
    const grade = {
      userId,
      scoreGiven: score,
      scoreMaximum: maxScore,
      activityProgress: 'Completed',
      gradingProgress: 'FullyGraded',
      timestamp: new Date().toISOString()
    };

    await this.lti.Grade.submitScore(
      res.locals.token,
      grade,
      { resourceId }
    );
  }

  // 진도 동기화 (Deep Linking)
  async syncProgress(userId: string, progress: LearningProgress): Promise<void> {
    // xAPI 형식으로 변환
    const statement = this.convertToXAPI(userId, progress);

    // LRS로 전송
    await this.sendToLRS(statement);
  }

  private convertToXAPI(userId: string, progress: LearningProgress): xAPIStatement {
    return {
      actor: {
        objectType: 'Agent',
        account: {
          homePage: 'https://wia.org',
          name: userId
        }
      },
      verb: {
        id: 'http://adlnet.gov/expapi/verbs/progressed',
        display: { 'en-US': 'progressed' }
      },
      object: {
        objectType: 'Activity',
        id: `https://wia.org/activities/${progress.activityId}`,
        definition: {
          name: { 'en-US': progress.activityName },
          type: 'http://adlnet.gov/expapi/activities/lesson'
        }
      },
      result: {
        score: {
          scaled: progress.score / 100,
          raw: progress.score,
          max: 100
        },
        completion: progress.completed,
        duration: `PT${progress.duration}S`
      },
      timestamp: new Date().toISOString()
    };
  }
}
```

```typescript
    analyzer.startAnalysis(audioStream);
  }

  // 실시간 자막 및 번역
  async enableLiveCaption(): Promise<void> {
    this.client.on('active-speaker', async (speakers) => {
      for (const speaker of speakers) {

        const transcript = await this.transcribeSpeech(speaker.audio);

        // 자막 표시
        this.displayCaption({
          speaker: speaker.userName,
          original: transcript,
          translated: translated
        });
      }
    });
  }
}
```

```typescript
// WebRTC 기반 화상 통화 + AI 튜터링
class VideoTutoringSession {
  private peerConnection: RTCPeerConnection;
  private localStream: MediaStream;
  private speechAnalyzer: SpeechAnalyzer;

  async initialize(): Promise<void> {
    // 미디어 스트림 획득
    this.localStream = await navigator.mediaDevices.getUserMedia({
      video: true,
      audio: {
        echoCancellation: true,
        noiseSuppression: true,
        sampleRate: 16000
      }
    });

    // WebRTC 연결 설정
    this.peerConnection = new RTCPeerConnection({
      iceServers: [
        { urls: 'stun:stun.wia.org:3478' },
        {
          urls: 'turn:turn.wia.org:3478',
          username: 'wia',
          credential: process.env.TURN_CREDENTIAL
        }
      ]
    });

    // 트랙 추가
    this.localStream.getTracks().forEach(track => {
      this.peerConnection.addTrack(track, this.localStream);
    });

    // 음성 분석 초기화
    this.speechAnalyzer = new SpeechAnalyzer({
      targetLanguage: 'ko',
      realtime: true
    });

    // 오디오 처리 파이프라인
    await this.setupAudioProcessing();
  }

  private async setupAudioProcessing(): Promise<void> {
    const audioContext = new AudioContext({ sampleRate: 16000 });
    const source = audioContext.createMediaStreamSource(this.localStream);

    // 오디오 분석 노드
    const analyzerNode = audioContext.createAnalyser();
    analyzerNode.fftSize = 2048;

    // 오디오 워크렛 (실시간 처리)
    await audioContext.audioWorklet.addModule('/audio-processor.js');
    const processorNode = new AudioWorkletNode(audioContext, 'speech-processor');

    // 연결
    source.connect(analyzerNode);
    analyzerNode.connect(processorNode);

    // 처리된 오디오를 WIA 분석 서버로 전송
    processorNode.port.onmessage = async (event) => {
      const audioChunk = event.data;
      const analysis = await this.speechAnalyzer.analyzeChunk(audioChunk);

      if (analysis.feedback) {
        this.displayRealTimeFeedback(analysis.feedback);
      }
    };
  }

  // 실시간 피드백 표시
  private displayRealTimeFeedback(feedback: SpeechFeedback): void {
    const feedbackPanel = document.getElementById('feedback-panel');

    feedbackPanel.innerHTML = `
      <div class="pronunciation-score">
        발음 점수: ${feedback.pronunciationScore}/100
      </div>
      <div class="fluency-indicator">
        유창성: ${feedback.fluencyLevel}
      </div>
      ${feedback.corrections.length > 0 ? `
        <div class="corrections">
          <h4>교정 사항:</h4>
          <ul>
            ${feedback.corrections.map(c => `
              <li>
                <span class="incorrect">${c.said}</span>
                → <span class="correct">${c.should}</span>
              </li>
            `).join('')}
          </ul>
        </div>
      ` : ''}
    `;
  }
}
```

```csharp
using UnityEngine;
using UnityEngine.XR;

public class WIALangVRIntegration : MonoBehaviour
{
    [SerializeField] private string apiEndpoint = "https://api.wia.org/lang-008/v1";
    [SerializeField] private string targetLanguage = "ko";

    private WIALangClient client;
    private SpeechRecognizer speechRecognizer;
    private VRAvatarController avatarController;

    async void Start()
    {
        // WIA 클라이언트 초기화
        client = new WIALangClient(apiEndpoint);
        await client.Authenticate(GetUserToken());

        // 음성 인식 초기화
        speechRecognizer = GetComponent<SpeechRecognizer>();
        speechRecognizer.OnSpeechRecognized += HandleSpeechRecognized;

        // AI 아바타 컨트롤러
        avatarController = GetComponent<VRAvatarController>();

        // VR 입력 설정
        SetupVRInput();
    }

    private void SetupVRInput()
    {
        // 컨트롤러 버튼 매핑
        InputDevices.GetDeviceAtXRNode(XRNode.RightHand)
            .TryGetFeatureValue(CommonUsages.primaryButton, out bool pressed);

        if (pressed)
        {
            StartSpeaking();
        }
    }

    // 음성 인식 결과 처리
    private async void HandleSpeechRecognized(string transcript)
    {
        // 발음 평가 요청
        var assessment = await client.AssessPronunciation(new PronunciationRequest
        {
            Text = transcript,
            TargetLanguage = targetLanguage,
            AudioData = speechRecognizer.GetAudioData()
        });

        // VR 환경에 피드백 표시
        DisplayVRFeedback(assessment);

        // AI 튜터 응답 생성
        var response = await client.GetTutorResponse(new TutorRequest
        {
            UserMessage = transcript,
            Context = GetConversationContext(),
            Difficulty = GetUserLevel()
        });

        // AI 아바타가 응답
        await avatarController.Speak(response.Text, response.AudioUrl);
    }

    private void DisplayVRFeedback(PronunciationAssessment assessment)
    {
        // 3D 공간에 피드백 패널 표시
        var feedbackPanel = Instantiate(feedbackPrefab);
        feedbackPanel.transform.position = Camera.main.transform.position
            + Camera.main.transform.forward * 2f;

        var feedbackUI = feedbackPanel.GetComponent<VRFeedbackUI>();
        feedbackUI.SetScore(assessment.OverallScore);
        feedbackUI.SetPhonemeDetails(assessment.PhonemeScores);

        // 문제 있는 발음 하이라이트
        foreach (var issue in assessment.Issues)
        {
            feedbackUI.HighlightIssue(issue.Phoneme, issue.Suggestion);
        }
    }
}
```

### 7.4.2 AR 언어 학습 (모바일)

```swift
// iOS ARKit 언어 학습
import ARKit
import Vision

class ARLanguageLearning: UIViewController, ARSessionDelegate {

    @IBOutlet var arView: ARSCNView!

    private var wiaClient: WIALangClient!
    private var objectDetector: VNCoreMLModel!

    override func viewDidLoad() {
        super.viewDidLoad()

        // AR 세션 설정
        let configuration = ARWorldTrackingConfiguration()
        configuration.planeDetection = [.horizontal, .vertical]
        arView.session.delegate = self
        arView.session.run(configuration)

        // WIA 클라이언트 초기화
        wiaClient = WIALangClient(endpoint: "https://api.wia.org/lang-008/v1")

        // 객체 인식 모델 로드
        setupObjectDetection()
    }

    // 실시간 객체 인식 및 어휘 학습
    func session(_ session: ARSession, didUpdate frame: ARFrame) {
        let pixelBuffer = frame.capturedImage

        // 객체 인식
        let request = VNCoreMLRequest(model: objectDetector) { [weak self] request, error in
            guard let results = request.results as? [VNRecognizedObjectObservation] else { return }

            for observation in results where observation.confidence > 0.8 {
                self?.handleDetectedObject(observation, in: frame)
            }
        }

        try? VNImageRequestHandler(cvPixelBuffer: pixelBuffer).perform([request])
    }

    private func handleDetectedObject(_ observation: VNRecognizedObjectObservation, in frame: ARFrame) {
        let objectLabel = observation.labels.first?.identifier ?? ""

        // WIA API로 어휘 정보 조회
        Task {
            let vocabulary = try await wiaClient.getVocabulary(
                term: objectLabel,
                targetLanguage: "ko"
            )

            // AR 공간에 어휘 정보 표시
            DispatchQueue.main.async {
                self.displayARVocabulary(vocabulary, at: observation.boundingBox)
            }
        }
    }

    private func displayARVocabulary(_ vocabulary: Vocabulary, at boundingBox: CGRect) {
        // 3D 텍스트 노드 생성
        let textGeometry = SCNText(string: vocabulary.targetTerm, extrusionDepth: 1)
        textGeometry.font = UIFont.systemFont(ofSize: 12)
        textGeometry.firstMaterial?.diffuse.contents = UIColor.white

        let textNode = SCNNode(geometry: textGeometry)

        // 바운딩 박스 위에 배치
        let position = calculateARPosition(from: boundingBox)
        textNode.position = position

        // 발음 버튼 추가
        let speakButton = createSpeakButton(vocabulary: vocabulary)
        textNode.addChildNode(speakButton)

        arView.scene.rootNode.addChildNode(textNode)

        // 탭 제스처로 발음 재생
        let tapGesture = UITapGestureRecognizer(target: self, action: #selector(handleTap))
        arView.addGestureRecognizer(tapGesture)
    }

    @objc private func handleTap(_ gesture: UITapGestureRecognizer) {
        let location = gesture.location(in: arView)
        let hitResults = arView.hitTest(location)

        if let node = hitResults.first?.node,
           let vocabulary = node.userData?["vocabulary"] as? Vocabulary {
            // 발음 재생
            playPronunciation(vocabulary.audioUrl)

            // 따라 말하기 모드 시작
            startShadowingMode(vocabulary)
        }
    }

    private func startShadowingMode(_ vocabulary: Vocabulary) {
        // 음성 녹음 시작
        let recorder = SpeechRecorder()
        recorder.startRecording()

        // 3초 후 녹음 종료 및 평가
        DispatchQueue.main.asyncAfter(deadline: .now() + 3) {
            recorder.stopRecording()

            Task {
                let assessment = try await self.wiaClient.assessPronunciation(
                    targetText: vocabulary.targetTerm,
                    audioData: recorder.audioData
                )

                self.displayAssessmentInAR(assessment)
            }
        }
    }
}
```

```typescript
// 음성 인식 및 발음 평가
export function useSpeechAssessment(targetLanguage: string) {
  const [isRecording, setIsRecording] = useState(false);
  const [assessment, setAssessment] = useState<Assessment | null>(null);

  const startRecording = async () => {
    setIsRecording(true);
    await wiaLang.speech.startRecording({
      language: targetLanguage,
      sampleRate: 16000
    });
  };

  const stopRecording = async (targetText: string) => {
    setIsRecording(false);
    const audioData = await wiaLang.speech.stopRecording();

    const result = await wiaLang.assess.pronunciation({
      targetText,
      targetLanguage,
      audioData
    });

    setAssessment(result);
    return result;
  };

  return { isRecording, assessment, startRecording, stopRecording };
}

// 오프라인 학습 지원
export function useOfflineLearning() {
  const [downloadProgress, setDownloadProgress] = useState(0);
  const [isOfflineReady, setIsOfflineReady] = useState(false);

  const downloadContent = async (courseId: string) => {
    await wiaLang.offline.downloadCourse(courseId, {
      onProgress: (progress) => setDownloadProgress(progress),
      onComplete: () => setIsOfflineReady(true)
    });
  };

  const getOfflineContent = async (contentId: string) => {
    return await wiaLang.offline.getContent(contentId);
  };

  return { downloadProgress, isOfflineReady, downloadContent, getOfflineContent };
}
```

```typescript
class EmbeddedLearningWidget {

    this.wia = wiaLang;

      <div class="wia-widget">
        <div class="wia-lesson-area" id="lesson-content"></div>
        <div class="wia-speech-area">
          <button id="record-btn" class="wia-record-btn">
            🎤 말하기
          </button>
          <div id="feedback-area" class="wia-feedback"></div>
        </div>
        <div class="wia-progress-area" id="progress"></div>
      </div>
    `;

    this.setupEventListeners();
  }

  private setupEventListeners(): void {
    const recordBtn = document.getElementById('record-btn')!;
    let isRecording = false;

    recordBtn.addEventListener('click', async () => {
      if (!isRecording) {
        isRecording = true;
        recordBtn.textContent = '⏹ 중지';
        await this.wia.speech.startRecording();
      } else {
        isRecording = false;
        recordBtn.textContent = '🎤 말하기';
        const audio = await this.wia.speech.stopRecording();
        await this.assessSpeech(audio);
      }
    });
  }

  private async assessSpeech(audioData: Blob): Promise<void> {
    const feedbackArea = document.getElementById('feedback-area')!;
    feedbackArea.innerHTML = '<p>평가 중...</p>';

    const assessment = await this.wia.assess.pronunciation({
      targetText: this.currentTargetText,
      audioData
    });

    feedbackArea.innerHTML = `
      <div class="score">점수: ${assessment.overallScore}/100</div>
      <div class="details">
        ${assessment.phonemeScores.map(p => `
          <span class="${p.score >= 80 ? 'good' : 'needs-work'}">
            ${p.phoneme}: ${p.score}
          </span>
        `).join('')}
      </div>
    `;
  }
}
```
