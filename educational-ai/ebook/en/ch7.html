7.2 LMS Integration Patterns

LTI (Learning Tools Interoperability)

LTI is the standard protocol for integrating learning applications with LMS platforms. It allows Educational AI to be embedded seamlessly within existing LMS environments. 

// LTI 1.3 Integration
import { LTI13Platform } from '@wia/lti13';

class LTIIntegration {
  private platform: LTI13Platform;

  constructor() {
    this.platform = new LTI13Platform({
      issuer: 'https://educational-ai.example.com',
      clientId: process.env.LTI_CLIENT_ID,
      authorizationServer: process.env.LTI_AUTH_SERVER,
      keySet: {
        keys: [this.getPublicKey()]
      }
    });
  }

  // Handle LTI launch request
  async handleLaunch(req: Request): Promise {
    // 1. Validate LTI message
    const validated = await this.platform.validateMessage(req);

    // 2. Extract user and context information
    const user = this.extractUserInfo(validated);
    const context = this.extractContextInfo(validated);

    // 3. Create or update user in Educational AI
    const aiUser = await this.syncUser(user);

    // 4. Create session and redirect to Educational AI
    const session = await this.createSession(aiUser, context);

    return {
      redirectUrl: `https://ai.example.com/session/${session.id}`,
      token: session.token
    };
  }

  // Send grades back to LMS
  async sendGrade(
    userId: string,
    assignmentId: string,
    score: number
  ): Promise {
    const lineItem = await this.getLineItem(assignmentId);

    await this.platform.submitScore({
      userId,
      lineItem,
      scoreGiven: score,
      scoreMaximum: 100,
      timestamp: new Date().toISOString(),
      activityProgress: 'Completed',
      gradingProgress: 'FullyGraded'
    });
  }

  // Sync roster from LMS
  async syncRoster(courseId: string): Promise {
    const members = await this.platform.getMembers(courseId);

    for (const member of members) {
      await this.syncUser({
        id: member.user_id,
        name: member.name,
        email: member.email,
        role: member.roles
      });
    }
  }
}

Deep Linking

Deep linking allows educators to select and configure Educational AI content from within their LMS, creating seamless integration experiences. 

// Deep Linking Implementation
class DeepLinkingService {
  async handleDeepLinkingRequest(req: Request): Promise {
    // 1. Validate deep linking message
    const validated = await this.validateDeepLinkMessage(req);

    // 2. Get available content based on context
    const availableContent = await this.getAvailableContent(validated.context);

    // 3. Return content selection interface
    return {
      html: this.renderContentSelector(availableContent),
      returnUrl: validated.deep_link_return_url
    };
  }

  async handleContentSelection(
    selection: ContentSelection[]
  ): Promise {
    // Create LTI resource links for selected content
    const resources = selection.map(item => ({
      type: 'ltiResourceLink',
      title: item.title,
      url: `https://ai.example.com/content/${item.id}`,
      custom: {
        contentId: item.id,
        difficulty: item.difficulty
      },
      lineItem: {
        scoreMaximum: 100,
        label: item.title,
        resourceId: item.id
      }
    }));

    return {
      type: 'LtiDeepLinkingResponse',
      data: resources
    };
  }
}

7.3 Single Sign-On (SSO) Integration

SAML Integration

Security Assertion Markup Language (SAML) enables single sign-on with educational institutions. 

// SAML SSO Implementation
import { SAML } from '@node-saml/node-saml';

class SAMLAuthProvider {
  private saml: SAML;

  constructor() {
    this.saml = new SAML({
      entryPoint: process.env.SAML_ENTRY_POINT,
      issuer: 'educational-ai-service',
      callbackUrl: 'https://ai.example.com/auth/saml/callback',
      cert: process.env.SAML_CERT,
      identifierFormat: 'urn:oasis:names:tc:SAML:2.0:nameid-format:transient'
    });
  }

  // Generate authentication request
  async generateAuthRequest(): Promise {
    return new Promise((resolve, reject) => {
      this.saml.getAuthorizeUrl({}, (err, url) => {
        if (err) reject(err);
        else resolve(url);
      });
    });
  }

  // Process SAML response
  async processResponse(body: any): Promise {
    return new Promise((resolve, reject) => {
      this.saml.validatePostResponse(body, async (err, profile) => {
        if (err) {
          reject(err);
        } else {
          // Extract user information from SAML assertion
          const user = {
            id: profile.nameID,
            email: profile.email,
            firstName: profile.firstName,
            lastName: profile.lastName,
            roles: this.parseRoles(profile.attributes)
          };

          // Create or update user in system
          const aiUser = await this.syncUser(user);
          resolve(aiUser);
        }
      });
    });
  }

  private parseRoles(attributes: any): string[] {
    const roleAttribute = attributes['urn:oid:1.3.6.1.4.1.5923.1.1.1.7'];
    return Array.isArray(roleAttribute) ? roleAttribute : [roleAttribute];
  }
}

OAuth 2.0 Integration

OAuth 2.0 provides delegated authorization for Educational AI to access resources on behalf of users. 

// OAuth 2.0 Integration
class OAuth2Provider {
  // Authorization Code Flow
  async initiateAuth(provider: string): Promise {
    const state = this.generateState();
    const params = new URLSearchParams({
      client_id: process.env.OAUTH_CLIENT_ID,
      redirect_uri: 'https://ai.example.com/auth/callback',
      response_type: 'code',
      scope: 'openid profile email',
      state
    });

    await this.saveState(state);

    return `${this.getAuthEndpoint(provider)}?${params}`;
  }

  async handleCallback(code: string, state: string): Promise {
    // Verify state
    await this.verifyState(state);

    // Exchange code for token
    const tokenResponse = await fetch(this.getTokenEndpoint(), {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({
        grant_type: 'authorization_code',
        code,
        redirect_uri: 'https://ai.example.com/auth/callback',
        client_id: process.env.OAUTH_CLIENT_ID,
        client_secret: process.env.OAUTH_CLIENT_SECRET
      })
    });

    const tokens = await tokenResponse.json();

    // Get user profile
    const profile = await this.getUserProfile(tokens.access_token);

    return profile;
  }

  // Refresh token handling
  async refreshAccessToken(refreshToken: string): Promise {
    const response = await fetch(this.getTokenEndpoint(), {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({
        grant_type: 'refresh_token',
        refresh_token: refreshToken,
        client_id: process.env.OAUTH_CLIENT_ID,
        client_secret: process.env.OAUTH_CLIENT_SECRET
      })
    });

    return response.json();
  }
}

7.4 Content Repository Integration

Content Syndication

Integrate with external content repositories to expand available learning materials. 

// Content Repository Integration
class ContentRepositoryIntegration {
  // Fetch content from external repository
  async fetchContent(repositoryId: string, contentId: string): Promise {
    const repository = await this.getRepository(repositoryId);

    const content = await fetch(`${repository.apiEndpoint}/content/${contentId}`, {
      headers: {
        'Authorization': `Bearer ${repository.apiKey}`,
        'Accept': 'application/json'
      }
    });

    const data = await content.json();

    // Transform to internal format
    return this.transformContent(data, repository.schema);
  }

  // Search across multiple repositories
  async searchContent(query: string): Promise {
    const repositories = await this.getActiveRepositories();

    const searches = repositories.map(repo =>
      this.searchRepository(repo, query)
    );

    const results = await Promise.all(searches);

    // Merge and rank results
    return this.mergeResults(results);
  }

  // Sync repository catalog
  async syncCatalog(repositoryId: string): Promise {
    const repository = await this.getRepository(repositoryId);
    const catalog = await this.fetchCatalog(repository);

    for (const item of catalog) {
      await this.upsertContent({
        externalId: item.id,
        repositoryId,
        metadata: item,
        lastSynced: new Date()
      });
    }
  }

  // SCORM package import
  async importSCORM(packageUrl: string): Promise {
    // Download SCORM package
    const packageData = await this.downloadPackage(packageUrl);

    // Extract and parse manifest
    const manifest = await this.parseManifest(packageData);

    // Import content and track structure
    const content = await this.importSCORMContent(manifest);

    return content;
  }
}

7.5 Analytics Integration

xAPI (Experience API)

xAPI captures learning experiences across platforms and tools, providing comprehensive learning analytics. 

// xAPI Integration
class XAPIService {
  async sendStatement(statement: XAPIStatement): Promise {
    const fullStatement = {
      actor: {
        objectType: 'Agent',
        name: statement.actor.name,
        mbox: `mailto:${statement.actor.email}`,
        account: {
          homePage: 'https://ai.example.com',
          name: statement.actor.id
        }
      },
      verb: {
        id: this.getVerbIRI(statement.verb),
        display: { 'en-US': statement.verb }
      },
      object: {
        id: `https://ai.example.com/activities/${statement.object.id}`,
        objectType: 'Activity',
        definition: {
          name: { 'en-US': statement.object.name },
          description: { 'en-US': statement.object.description },
          type: this.getActivityTypeIRI(statement.object.type)
        }
      },
      result: statement.result ? {
        score: statement.result.score ? {
          scaled: statement.result.score / 100,
          raw: statement.result.score,
          min: 0,
          max: 100
        } : undefined,
        success: statement.result.success,
        completion: statement.result.completion,
        duration: statement.result.duration,
        response: statement.result.response
      } : undefined,
      context: {
        contextActivities: {
          parent: statement.context?.parent ? [
            { id: statement.context.parent, objectType: 'Activity' }
          ] : undefined,
          grouping: statement.context?.course ? [
            { id: statement.context.course, objectType: 'Activity' }
          ] : undefined
        },
        platform: 'Educational AI Platform',
        language: 'en-US'
      },
      timestamp: new Date().toISOString()
    };

    // Send to LRS (Learning Record Store)
    await this.lrs.saveStatement(fullStatement);
  }

  // Common xAPI verbs for educational activities
  private getVerbIRI(verb: string): string {
    const verbs = {
      'completed': 'http://adlnet.gov/expapi/verbs/completed',
      'attempted': 'http://adlnet.gov/expapi/verbs/attempted',
      'passed': 'http://adlnet.gov/expapi/verbs/passed',
      'failed': 'http://adlnet.gov/expapi/verbs/failed',
      'answered': 'http://adlnet.gov/expapi/verbs/answered',
      'asked': 'http://adlnet.gov/expapi/verbs/asked',
      'interacted': 'http://adlnet.gov/expapi/verbs/interacted'
    };
    return verbs[verb] || verb;
  }

  // Query learning records
  async queryStatements(filter: XAPIFilter): Promise {
    const params = new URLSearchParams();

    if (filter.actor) params.append('agent', JSON.stringify(filter.actor));
    if (filter.verb) params.append('verb', this.getVerbIRI(filter.verb));
    if (filter.activity) params.append('activity', filter.activity);
    if (filter.since) params.append('since', filter.since.toISOString());
    if (filter.until) params.append('until', filter.until.toISOString());

    const response = await this.lrs.getStatements(params);
    return response.statements;
  }
}

// Zoom/Teams Integration
class VideoConferencingIntegration {

    studentId: string,

    const meeting = await this.createMeeting({

      meetingUrl: meeting.join_url,
      meetingId: meeting.id,
      aiAssistantEnabled: true
    };

    const insights = await this.aiModel.analyze(transcript, {
      identifyQuestions: true,
      extractConcepts: true,
      assessUnderstanding: true,
      suggestFollowUp: true
    });

    return insights;
  }
}

// Slack Bot Integration
class SlackBotIntegration {

}

7.7 Data Synchronization Patterns

Event-Driven Synchronization

Use events to keep Educational AI synchronized with external systems in real-time. 

// Event-driven sync with webhooks
class EventDrivenSync {
  async handleWebhook(event: WebhookEvent): Promise {
    switch (event.type) {
      case 'user.created':
        await this.syncNewUser(event.data);
        break;
      case 'enrollment.created':
        await this.syncEnrollment(event.data);
        break;
      case 'grade.updated':
        await this.syncGrade(event.data);
        break;
      case 'content.published':
        await this.syncContent(event.data);
        break;
    }
  }

  // Emit events to other systems
  async emitEvent(eventType: string, data: any): Promise {
    const event = {
      id: this.generateEventId(),
      type: eventType,
      data,
      timestamp: new Date().toISOString(),
      source: 'educational-ai'
    };

    // Send to registered webhooks
    const webhooks = await this.getWebhooks(eventType);
    await Promise.all(
      webhooks.map(webhook => this.sendWebhook(webhook, event))
    );
  }
}
