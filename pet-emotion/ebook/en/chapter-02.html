2.1 Core Data Types

EmotionType Enumeration

enum EmotionType {
  HAPPY         // Joy, pleasure
  SAD           // Dejection, sorrow
  ANXIOUS       // Worry, nervousness
  CALM          // Relaxed, peaceful
  EXCITED       // High arousal, positive
  FEARFUL       // Scared, threatened
  AFFECTIONATE  // Loving, bonding
  FRUSTRATED    // Blocked, irritated
  PLAYFUL       // Engaging, fun-seeking
  CONTENT       // Satisfied, neutral-positive
  CURIOUS       // Interested, exploring
  AGGRESSIVE    // Hostile, defensive
  BORED         // Understimulated
  TIRED         // Fatigued, sleepy
  ALERT         // Attentive, watchful
  UNKNOWN       // Insufficient data
}

2.2 Intensity Scale

Level Value Description

MODERATE 3 Obvious and consistent
STRONG 4 Dominant emotional state
INTENSE 5 Overwhelming, requires attention

2.3 Confidence Levels

CONFIDENCE CLASSIFICATION
=========================

INSUFFICIENT (0-49%):
  - Data quality too low for reliable assessment
  - Multiple conflicting signals
  - Sensor malfunction possible

LOW (50-64%):
  - Basic emotion detection possible
  - Significant uncertainty remains
  - Use with caution

MODERATE (65-84%):
  - Reliable for most purposes
  - Some ambiguity acceptable
  - Standard operational threshold

HIGH (85-100%):
  - Very reliable assessment
  - Strong signal agreement
  - Suitable for critical decisions

2.4 EmotionState Structure

interface EmotionState {
  // Core identification
  emotionType: EmotionType      // Primary emotion
  intensity: 1 | 2 | 3 | 4 | 5  // 1-5 scale

  // Confidence metrics
  confidence: number            // 0-100 percentage
  confidenceLevel: ConfidenceLevel

  // Temporal information
  duration: number              // Milliseconds in state
  stability: number             // 0-1 how stable

  // Supporting data
  contributingFactors: string[] // What triggered
  dataSourceCount: number       // Sources used
}

2.5 EmotionReading Structure

interface EmotionReading {
  // Identification
  readingId: string           // UUID
  profileId: string           // Pet profile reference
  timestamp: ISO8601DateTime  // When captured

  // Primary result
  primaryEmotion: EmotionState

  // Additional emotions detected
  secondaryEmotions: EmotionState[]

  // Overall assessment
  overallMood: MoodAssessment

  // Data sources used
  dataSources: DataSourceInfo[]

  // Quality metrics
  overallConfidence: number
  dataQuality: DataQuality
}

2.7 Data Source Types

Visual Data Sources

interface VisualDataSource {
  sourceType: "VISUAL"
  deviceType: "camera" | "depth_sensor"

  bodyPosture: {
    posture: string         // standing, sitting, lying
    tension: number         // 0-1 muscle tension
    orientation: string     // toward, away, neutral
  }

  facialExpression: {
    eyeState: string        // wide, relaxed, squinted
    mouthState: string      // open, closed, relaxed
    earPosition: string     // forward, back, relaxed
  }

  tailPosition: {
    position: string        // high, low, tucked
    movement: string        // wagging, still, twitching
    speed: number           // movement frequency
  }
}

Audio Data Sources

interface AudioDataSource {
  sourceType: "AUDIO"
  deviceType: "microphone"

  vocalization: {
    type: string            // bark, whine, purr, growl
    frequency: number       // Hz
    amplitude: number       // dB
    duration: number        // ms
    pattern: string         // continuous, intermittent
  }

  acousticFeatures: {
    pitch: number
    pitchVariation: number
    harmonicity: number
    energy: number
  }
}

Physiological Data Sources

interface PhysiologicalDataSource {
  sourceType: "PHYSIOLOGICAL"
  deviceType: "wearable" | "implant" | "external"

  heartRate: {
    bpm: number
    variability: number     // HRV in ms
    trend: string           // increasing, stable, decreasing
  }

  activity: {
    level: number           // 0-100
    stepCount: number
    movement: string        // stationary, walking, running
  }

  temperature: {
    value: number           // Celsius
    location: string        // ear, skin, core
  }
}

2.8 EmotionProfile Structure

interface EmotionProfile {
  // Identification
  profileId: string
  petId: string
  ownerId: string

  // Pet information
  species: string           // canine, feline, etc.
  breed: string
  name: string
  birthDate: Date
  weight: number

  // Baseline data
  baselineEmotions: EmotionBaseline

  // Personality assessment
  personalityTraits: PersonalityProfile

  // Known triggers
  triggers: EmotionTrigger[]

  // Metadata
  createdAt: DateTime
  updatedAt: DateTime
  lastReadingAt: DateTime
}

2.9 Baseline Calibration

interface EmotionBaseline {
  // Physiological baselines
  restingHeartRate: number      // bpm
  restingHRV: number            // ms
  normalActivityLevel: number   // 0-100
  normalBodyTemperature: number // Celsius

  // Behavioral baselines
  typicalSleepHours: number
  feedingTimes: string[]
  exercisePattern: string

  // Emotional baselines
  defaultMoodValence: number    // -1 to 1
  defaultArousalLevel: number   // 0 to 1
  emotionalStability: number    // 0 to 1

  // Calibration metadata
  calibrationDate: DateTime
  calibrationDuration: number   // hours of data
  calibrationQuality: number    // 0-100
}

2.11 Emotion Triggers

interface EmotionTrigger {
  triggerId: string
  profileId: string

  // Trigger definition
  triggerType: "environmental" | "social" | "temporal" | "physical"
  description: string
  keywords: string[]

  // Emotional response
  typicalEmotion: EmotionType
  typicalIntensity: number
  responseLatency: number       // ms to respond

  // Context
  frequency: "rare" | "occasional" | "common" | "frequent"
  lastOccurrence: DateTime

  // Management
  mitigation: string            // How to reduce impact
  positive: boolean             // Is this a positive trigger
}

Quality Level Criteria Usage
