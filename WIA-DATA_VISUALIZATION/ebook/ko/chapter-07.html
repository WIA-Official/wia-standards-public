Chapter 07: 실시간 데이터 시각화

WebSocket, SSE, 스트리밍 차트로 실시간 모니터링 구현

1. 실시간 시각화의 필요성

IoT 센서, 금융 거래, 서버 모니터링, 소셜 미디어 스트림 등에서 실시간 데이터 시각화는 필수입니다. 지연이 1초만 발생해도 수백만 원의 손실이나 시스템 장애로 이어질 수 있습니다.

실시간 시각화 사용 사례
• DevOps 모니터링: CPU, 메모리, 네트워크 트래픽 (1초 업데이트)
• 주식 거래: 실시간 가격, 거래량 (밀리초 단위)
• IoT 센서: 온도, 습도, 진동 (10ms ~ 1s)
• 소셜 미디어: 트렌딩 토픽, 감정 분석 (5-60초)
• 웹 분석: 실시간 방문자, 페이지뷰 (1-5초)
• 게임 메트릭: 동시 접속자, 서버 상태 (1초)

2. WebSocket 실시간 차트

WebSocket은 가장 널리 사용되는 실시간 통신 프로토콜입니다. HTTP 핸드셰이크 후 TCP 연결을 유지하여 지연시간을 최소화합니다.

2.1 WebSocket 서버 (Node.js)

// WebSocket 서버 (ws 라이브러리) const WebSocket = require('ws'); const wss = new WebSocket.Server({ port: 8080 }); // 연결된 클라이언트 목록 const clients = new Set(); wss.on('connection', (ws) => { console.log('클라이언트 연결됨'); clients.add(ws); // 연결 시 초기 데이터 전송 ws.send(JSON.stringify({ type: 'init', data: generateInitialData() })); // 클라이언트로부터 메시지 수신 ws.on('message', (message) => { console.log('수신:', message); const data = JSON.parse(message); if (data.type === 'subscribe') { // 구독 처리 ws.subscriptions = data.channels; } }); // 연결 해제 ws.on('close', () => { console.log('클라이언트 연결 해제'); clients.delete(ws); }); // 에러 처리 ws.on('error', (error) => { console.error('WebSocket 에러:', error); }); }); // 실시간 데이터 브로드캐스트 (1초마다) setInterval(() => { const newData = { type: 'update', timestamp: Date.now(), metrics: { cpu: Math.random() * 100, memory: Math.random() * 16, requests: Math.floor(Math.random() * 1000) } }; // 모든 연결된 클라이언트에 전송 clients.forEach(client => { if (client.readyState === WebSocket.OPEN) { client.send(JSON.stringify(newData)); } }); }, 1000); console.log('WebSocket 서버 시작: ws://localhost:8080');

3. Server-Sent Events (SSE)

SSE는 서버에서 클라이언트로 단방향 데이터 스트리밍에 최적화되어 있습니다. WebSocket보다 단순하며 HTTP/2를 사용하면 멀티플렉싱도 가능합니다.

3.1 SSE 서버 (Express.js)

// SSE 서버 const express = require('express'); const app = express(); app.use(express.static('public')); // SSE 엔드포인트 app.get('/events', (req, res) => { // SSE 헤더 설정 res.writeHead(200, { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache', 'Connection': 'keep-alive', 'Access-Control-Allow-Origin': '*' }); // 연결 유지를 위한 핑 (30초마다) const pingInterval = setInterval(() => { res.write(':ping\n\n'); }, 30000); // 실시간 데이터 전송 (1초마다) const dataInterval = setInterval(() => { const data = { timestamp: Date.now(), temperature: 20 + Math.random() * 10, humidity: 40 + Math.random() * 30, pressure: 1000 + Math.random() * 50 }; // SSE 형식으로 전송 res.write(`data: ${JSON.stringify(data)}\n\n`); }, 1000); // 클라이언트 연결 종료 시 정리 req.on('close', () => { clearInterval(pingInterval); clearInterval(dataInterval); res.end(); }); }); app.listen(3000, () => { console.log('SSE 서버 시작: http://localhost:3000'); });

3.2 SSE 클라이언트 + D3.js

// SSE EventSource const eventSource = new EventSource('http://localhost:3000/events'); // D3.js 실시간 라인 차트 const width = 800; const height = 400; const margin = { top: 20, right: 80, bottom: 30, left: 50 }; const svg = d3.select('#chart') .append('svg') .attr('width', width) .attr('height', height); const g = svg.append('g') .attr('transform', `translate(${margin.left},${margin.top})`); // 데이터 배열 const dataBuffer = []; const MAX_POINTS = 60; // 1분 (1초마다 1포인트) // 스케일 const xScale = d3.scaleTime() .range([0, width - margin.left - margin.right]); const yScale = d3.scaleLinear() .range([height - margin.top - margin.bottom, 0]) .domain([10, 40]); // 라인 생성기 const line = d3.line() .x(d => xScale(d.timestamp)) .y(d => yScale(d.temperature)) .curve(d3.curveMonotoneX); // 축 const xAxis = g.append('g') .attr('transform', `translate(0,${height - margin.top - margin.bottom})`); const yAxis = g.append('g'); // 라인 경로 const path = g.append('path') .attr('fill', 'none') .attr('stroke', '#8B5CF6') .attr('stroke-width', 2); // SSE 메시지 수신 eventSource.onmessage = (event) => { const data = JSON.parse(event.data); data.timestamp = new Date(data.timestamp); // 데이터 추가 dataBuffer.push(data); // 버퍼 크기 제한 if (dataBuffer.length > MAX_POINTS) { dataBuffer.shift(); } // 차트 업데이트 updateChart(); }; eventSource.onerror = (error) => { console.error('SSE 에러:', error); eventSource.close(); }; function updateChart() { // X 스케일 도메인 업데이트 xScale.domain(d3.extent(dataBuffer, d => d.timestamp)); // 축 업데이트 xAxis.transition() .duration(500) .call(d3.axisBottom(xScale) .ticks(5) .tickFormat(d3.timeFormat('%H:%M:%S'))); yAxis.call(d3.axisLeft(yScale)); // 라인 업데이트 path.datum(dataBuffer) .transition() .duration(500) .ease(d3.easeLinear) .attr('d', line); }

4. 스트리밍 최적화 전략

4.1 데이터 샘플링

// LTTB (Largest-Triangle-Three-Buckets) 샘플링 function downsample(data, threshold) { if (data.length <= threshold) { return data; } const sampled = []; const bucketSize = (data.length - 2) / (threshold - 2); // 첫 번째 포인트 유지 sampled.push(data[0]); for (let i = 0; i < threshold - 2; i++) { const bucketStart = Math.floor(i * bucketSize) + 1; const bucketEnd = Math.floor((i + 1) * bucketSize) + 1; const bucket = data.slice(bucketStart, bucketEnd); // 다음 버킷의 평균 포인트 const nextBucket = data.slice(bucketEnd, Math.floor((i + 2) * bucketSize) + 1); const avgNext = { x: d3.mean(nextBucket, d => d.x), y: d3.mean(nextBucket, d => d.y) }; // 이전 선택된 포인트 const prev = sampled[sampled.length - 1]; // 가장 큰 삼각형 면적을 만드는 포인트 선택 let maxArea = -1; let maxIndex = 0; bucket.forEach((point, idx) => { const area = Math.abs( (prev.x - avgNext.x) * (point.y - prev.y) - (prev.x - point.x) * (avgNext.y - prev.y) ) * 0.5; if (area > maxArea) { maxArea = area; maxIndex = bucketStart + idx; } }); sampled.push(data[maxIndex]); } // 마지막 포인트 유지 sampled.push(data[data.length - 1]); return sampled; } // 사용 const rawData = []; // 100,000 포인트 const sampledData = downsample(rawData, 1000); // 1,000 포인트로 축소

4.2 증분 업데이트

// 전체 데이터를 다시 그리지 않고 새 포인트만 추가 class IncrementalChart { constructor(containerId, maxPoints = 500) { this.maxPoints = maxPoints; this.data = []; // ECharts 초기화 this.chart = echarts.init(document.getElementById(containerId)); this.option = { xAxis: { type: 'time', boundaryGap: false }, yAxis: { type: 'value', boundaryGap: [0, '10%'] }, series: [{ type: 'line', showSymbol: false, data: [] }] }; this.chart.setOption(this.option); } addPoint(timestamp, value) { // 새 포인트 추가 this.data.push([timestamp, value]); // 최대 포인트 초과 시 제거 if (this.data.length > this.maxPoints) { this.data.shift(); } // 증분 업데이트 (전체 차트 다시 그리지 않음) this.chart.setOption({ series: [{ data: this.data }] }); } clear() { this.data = []; this.chart.setOption({ series: [{ data: [] }] }); } } // 사용 const chart = new IncrementalChart('chart-container', 1000); ws.onmessage = (event) => { const data = JSON.parse(event.data); chart.addPoint(data.timestamp, data.value); };

4.3 백프레셔 (Backpressure) 처리

// 클라이언트가 처리할 수 있는 속도로 데이터 제한 class RateLimitedStream { constructor(maxRate = 60) { // 초당 최대 60개 this.maxRate = maxRate; this.buffer = []; this.lastFlush = Date.now(); // 주기적으로 버퍼 비우기 setInterval(() => this.flush(), 1000 / maxRate); } push(data) { this.buffer.push(data); // 버퍼가 너무 크면 오래된 데이터 제거 if (this.buffer.length > this.maxRate * 2) { this.buffer = this.buffer.slice(-this.maxRate); } } flush() { if (this.buffer.length === 0) return; const data = this.buffer.shift(); this.processData(data); } processData(data) { // 차트 업데이트 chart.addPoint(data.timestamp, data.value); } } const stream = new RateLimitedStream(60); ws.onmessage = (event) => { const data = JSON.parse(event.data); stream.push(data); };

5. 성능 모니터링 대시보드

실시간 시각화의 가장 흔한 사용 사례는 시스템 성능 모니터링입니다.

5.1 완전한 모니터링 대시보드

// 실시간 모니터링 대시보드 class MonitoringDashboard { constructor() { this.ws = new WebSocket('ws://localhost:8080/metrics'); this.charts = {}; this.initCharts(); this.connectWebSocket(); } initCharts() { // CPU 차트 this.charts.cpu = new Chart(document.getElementById('cpu-chart'), { type: 'line', data: { labels: [], datasets: [{ label: 'CPU Usage (%)', data: [], borderColor: '#ef4444', backgroundColor: 'rgba(239, 68, 68, 0.1)', fill: true, tension: 0.4 }] }, options: { responsive: true, animation: false, scales: { y: { min: 0, max: 100, ticks: { callback: (value) => value + '%' } } }, plugins: { annotation: { annotations: { warningLine: { type: 'line', yMin: 80, yMax: 80, borderColor: '#f59e0b', borderWidth: 2, borderDash: [5, 5], label: { content: 'Warning: 80%', enabled: true } } } } } } }); indicator.style.color = statusMap[status].color; } formatUptime(seconds) { const days = Math.floor(seconds / 86400); const hours = Math.floor((seconds % 86400) / 3600); const minutes = Math.floor((seconds % 3600) / 60); return `${days}일 ${hours}시간 ${minutes}분`; } } // 초기화 const dashboard = new MonitoringDashboard();

핵심 요점

  • WebSocket vs SSE: 양방향 vs 단방향, 지연 1ms vs 5ms
  • 애니메이션 비활성화: 실시간 차트에서 animation: false
  • 데이터 버퍼 제한: 60-600 포인트 유지, 오래된 데이터 제거
  • LTTB 샘플링: 100,000 포인트를 1,000으로 압축
  • 증분 업데이트: 전체 재렌더링 대신 새 포인트만 추가
  • 백프레셔: 초당 60개로 속도 제한하여 과부하 방지
  • 재연결 로직: 연결 끊김 시 5초 후 자동 재연결
  • 경고 임계값: CPU 80%, 메모리 90% 초과 시 알림
  • 상태 표시: 연결/끊김/에러 상태를 시각적으로 표시
  • 성능 최적화: Canvas > SVG, ECharts > Chart.js (대용량)
