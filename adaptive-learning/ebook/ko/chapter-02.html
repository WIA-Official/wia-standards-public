2장: 시스템 아키텍처

2.1 적응형 학습 시스템의 구성요소

WIA-ADAPTIVE_LEARNING 표준에 따른 적응형 학습 시스템은 네 가지 핵심 모델과 이를 조율하는 적응 엔진으로 구성됩니다. 각 구성요소는 명확히 정의된 인터페이스를 통해 상호작용하며, 모듈식 설계로 유연한 구현과 확장이 가능합니다.

아키텍처 원칙: 느슨한 결합, 높은 응집도를 유지합니다. 각 구성요소는 독립적으로 개발, 테스트, 업그레이드할 수 있어야 합니다. 표준 인터페이스를 통해 다양한 구현체의 교체가 가능해야 합니다.

2.2 학습자 모델

학습자 모델은 개별 학습자의 상태를 표현하는 핵심 구성요소입니다:

// 학습자 모델 인터페이스
interface LearnerModel {
    // 기본 정보
    learnerId: string;
    profile: LearnerProfile;

    // 지식 상태
    knowledge: KnowledgeState;

    // 학습 특성
    characteristics: {
        learningStyle: LearningStyle;
        cognitiveProfile: CognitiveProfile;
        preferences: LearnerPreferences;
        goals: LearningGoal[];
    };

    // 정서적 상태
    affective: {
        currentMood: Mood;
        engagement: EngagementLevel;
        motivation: MotivationLevel;
        frustration: FrustrationLevel;
    };

    // 학습 이력
    history: LearningHistory;
}

interface KnowledgeState {
    concepts: Map<ConceptId, MasteryLevel>;
    skills: Map<SkillId, ProficiencyLevel>;
    misconceptions: Misconception[];
    lastUpdated: Date;
}

interface LearningHistory {
    sessions: LearningSession[];
    interactions: Interaction[];
    assessments: AssessmentResult[];
    achievements: Achievement[];
}

2.3 도메인 모델

도메인 모델은 학습 영역의 지식 구조를 표현합니다:

// 도메인 모델 인터페이스
interface DomainModel {
    // 개념 네트워크
    concepts: ConceptGraph;

    // 학습 객체
    learningObjects: LearningObjectRepository;

    // 평가 항목
    assessmentItems: AssessmentItemBank;

    // 학습 경로 템플릿
    learningPaths: LearningPathTemplate[];
}

interface ConceptGraph {
    nodes: Concept[];
    edges: ConceptRelation[];

    // 관계 유형
    getPrerequisites(conceptId: string): Concept[];
    getSuccessors(conceptId: string): Concept[];
    getRelated(conceptId: string): Concept[];
}

interface Concept {
    id: string;
    name: string;
    description: string;
    difficulty: DifficultyLevel;
    estimatedTime: number;
    learningObjectives: LearningObjective[];
    prerequisites: string[];
    tags: string[];
}

2.4 교수 모델

교수 모델은 교육 전략과 결정 규칙을 정의합니다:

구성요소 역할
교수 전략 학습 콘텐츠 제시 방법 결정
피드백 규칙 학습자 응답에 대한 피드백 생성
힌트 시스템 단계적 도움 제공 전략
평가 전략 지식 확인 시점과 방법
동기부여 전략 학습 참여 유지 기법 
// 교수 모델 인터페이스
interface PedagogicalModel {
    // 전략 선택
    selectStrategy(learner: LearnerModel, context: LearningContext): Strategy;

    // 피드백 생성
    generateFeedback(response: LearnerResponse): Feedback;

    // 힌트 제공
    provideHint(problem: Problem, attemptCount: number): Hint;

    // 다음 단계 결정
    determineNextStep(learner: LearnerModel, current: LearningObject): NextStep;
}

interface Strategy {
    type: 'mastery' | 'spiral' | 'discovery' | 'direct';
    scaffolding: ScaffoldingConfig;
    pacing: PacingConfig;
    presentation: PresentationMode;
}

2.5 적응 엔진

적응 엔진은 모든 모델의 정보를 통합하여 실시간 적응 결정을 내립니다:

// 적응 엔진 핵심 로직
class AdaptationEngine {
    private learnerModel: LearnerModel;
    private domainModel: DomainModel;
    private pedagogicalModel: PedagogicalModel;

    async adapt(event: LearningEvent): Promise<AdaptationDecision> {
        // 1. 학습자 모델 업데이트
        await this.updateLearnerModel(event);

        // 2. 현재 상태 분석
        const analysis = this.analyzeCurrentState();

        // 3. 교수 전략 결정
        const strategy = this.pedagogicalModel.selectStrategy(
            this.learnerModel,
            analysis.context
        );

        // 4. 적응 결정 생성
        const decision = this.generateDecision(analysis, strategy);

        // 5. 결정 기록 및 반환
        await this.logDecision(decision);
        return decision;
    }

    private analyzeCurrentState(): StateAnalysis {
        return {
            knowledgeGaps: this.identifyGaps(),
            strugglePoints: this.detectStruggles(),
            engagement: this.assessEngagement(),
            optimalZone: this.findZPD(),
            context: this.getCurrentContext()
        };
    }

    private generateDecision(analysis: StateAnalysis, strategy: Strategy): AdaptationDecision {
        return {
            contentSelection: this.selectContent(analysis, strategy),
            difficultyAdjustment: this.adjustDifficulty(analysis),
            pathModification: this.modifyPath(analysis),
            scaffoldingLevel: this.determineScaffolding(analysis),
            feedbackType: this.selectFeedbackType(analysis)
        };
    }
}

2.6 데이터 흐름

실시간 적응 사이클: 학습자 상호작용 → 이벤트 수집 → 학습자 모델 업데이트 → 상태 분석 → 적응 결정 → 콘텐츠/경로 조정 → 학습자에게 제시. 이 사이클은 밀리초 단위로 실행됩니다.

2.7 확장성 고려사항

  • 수평적 확장: 마이크로서비스 아키텍처로 컴포넌트별 독립 확장
  • 실시간 처리: 이벤트 스트리밍을 통한 대규모 동시 사용자 지원
  • 캐싱 전략: 학습자 모델, 콘텐츠 캐싱으로 응답 시간 최적화
  • 비동기 처리: 복잡한 분석은 백그라운드에서 처리
// 확장 가능한 아키텍처 예시
