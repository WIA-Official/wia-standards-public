7장: 평가와 분석

7.1 학습 분석의 역할

학습 분석(Learning Analytics)은 적응형 학습 시스템의 핵심 구성요소입니다. 학습자 데이터를 수집, 분석, 해석하여 학습 경험을 개선하고, 시스템 효과를 측정하며, 데이터 기반 의사결정을 지원합니다.

분석의 목적: 학습 분석은 단순한 보고를 넘어 예측과 처방을 제공해야 합니다. "무엇이 일어났는가"에서 "무엇이 일어날 것인가" 그리고 "어떻게 해야 하는가"로 발전해야 합니다.

7.2 핵심 성과 지표 (KPI)

범주 지표 설명
학습 성과 숙달 달성률 목표 숙달 수준 도달 비율
학습 성과 학습 이득 사전-사후 점수 차이
효율성 숙달까지 시간 개념 숙달에 소요된 평균 시간
효율성 시도 횟수 정답까지 평균 시도 횟수
참여도 세션 시간 평균 학습 세션 길이
참여도 복귀율 재방문 학습자 비율
적응 효과 추천 수용률 시스템 추천 따른 비율

// 교수자용 분석 인터페이스

    conceptAnalysis: {
        conceptId: string;
        classAverage: number;
        commonMisconceptions: Misconception[];
        difficultyRating: number;
        recommendedInterventions: Intervention[];
    }[];

    // 참여도 분석
    engagementMetrics: {
        activeUsers: TimeSeriesData;
        sessionDurations: Distribution;
        completionRates: Map<string, number>;
    };

    // 조기 경보 시스템
    earlyWarning: {
        learner: Learner;
        riskLevel: 'high' | 'medium' | 'low';
        indicators: RiskIndicator[];
        suggestedActions: Action[];
    }[];
}

class EarlyWarningSystem {
    assessRisk(learner: LearnerModel): RiskAssessment {
        const indicators: RiskIndicator[] = [];

        // 참여도 하락
        if (this.detectEngagementDecline(learner)) {
            indicators.push({
                type: 'engagement_decline',
                severity: 0.7,
                details: '지난 주 대비 학습 시간 50% 감소'
            });
        }

        // 성과 정체
        if (this.detectPerformancePlateau(learner)) {
            indicators.push({
                type: 'performance_plateau',
                severity: 0.5,
                details: '2주간 숙달도 향상 없음'
            });
        }

        // 좌절 신호
        if (learner.affective.frustration > 0.7) {
            indicators.push({
                type: 'frustration',
                severity: 0.8,
                details: '높은 좌절감 감지'
            });
        }

        return {
            riskScore: this.calculateOverallRisk(indicators),
            indicators,
            recommendedActions: this.suggestInterventions(indicators)
        };
    }
}

7.5 A/B 테스팅

실험 설계: 적응형 학습 시스템의 개선은 A/B 테스팅을 통해 검증되어야 합니다. 새로운 적응 알고리즘, 콘텐츠, UI 변경의 효과를 과학적으로 측정합니다. 
// A/B 테스트 프레임워크
class ABTestingFramework {
    // 실험 정의
    defineExperiment(config: ExperimentConfig): Experiment {
        return {
            id: crypto.randomUUID(),
            name: config.name,
            hypothesis: config.hypothesis,
            variants: config.variants,
            metrics: config.metrics,
            sampleSize: this.calculateSampleSize(config),
            duration: config.duration,
            status: 'draft'
        };
    }

    // 학습자를 변형에 할당
    assignVariant(experimentId: string, learnerId: string): Variant {
        // 일관된 할당을 위한 해시 기반 무작위화
        const hash = this.hashString(`${experimentId}-${learnerId}`);
        const experiment = this.getExperiment(experimentId);

        let cumulativeWeight = 0;
        for (const variant of experiment.variants) {
            cumulativeWeight += variant.weight;
            if (hash < cumulativeWeight) {
                return variant;
            }
        }
        return experiment.variants[0]; // 기본값
    }

    // 결과 분석
    analyzeResults(experimentId: string): ExperimentResults {
        const experiment = this.getExperiment(experimentId);
        const data = this.collectExperimentData(experimentId);

        return {
            sampleSizes: this.getSampleSizes(data),
            metricResults: experiment.metrics.map(metric => ({
                metric: metric.name,
                byVariant: this.calculateMetricByVariant(data, metric),
                statisticalSignificance: this.calculateSignificance(data, metric),
                confidenceInterval: this.calculateCI(data, metric)
            })),
            recommendation: this.generateRecommendation(data, experiment)
        };
    }
}

// 자동 보고서 생성
class ReportGenerator {
    async generateLearnerReport(learnerId: string): Promise<Report> {
        const data = await this.collectLearnerData(learnerId);

        return {
            title: `학습 진행 보고서 - ${data.learnerName}`,
            period: this.reportPeriod,
            sections: [
                {
                    title: '학습 성과 요약',
                    content: this.summarizePerformance(data)
                },
                {
                    title: '개념별 숙달도',
                    content: this.detailMastery(data),
                    visualization: 'mastery_chart'
                },
                {
                    title: '학습 활동 분석',
                    content: this.analyzeActivity(data)
                },
                {
                    title: '개선 권장 사항',
                    content: this.generateRecommendations(data)
                }
            ],
            generatedAt: new Date()
        };
    }
}
데이터 기반 개선: 학습 분석의 궁극적 목표는 지속적인 개선입니다. 데이터에서 얻은 통찰을 바탕으로 콘텐츠, 적응 알고리즘, 사용자 경험을 반복적으로 개선해야 합니다.
