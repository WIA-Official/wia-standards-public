5장: 데이터 보안과 프라이버시

5.1 학습 데이터의 민감성

적응형 학습 시스템은 개인화를 위해 광범위한 학습자 데이터를 수집합니다. 이 데이터는 학습자의 지적 능력, 학습 어려움, 행동 패턴 등 매우 민감한 정보를 포함합니다. 특히 아동과 청소년 학습자의 데이터는 더욱 엄격한 보호가 필요합니다.

프라이버시 원칙: 적응형 학습 시스템은 "Privacy by Design" 원칙을 따라야 합니다. 개인정보 보호는 사후 조치가 아닌 시스템 설계의 핵심 요소여야 합니다.

5.2 데이터 분류

분류 데이터 유형 보호 수준
직접 식별 이름, 이메일, 학번 최고
학습 성과 점수, 숙달도, 오답 패턴 높음
행동 데이터 클릭, 체류 시간, 탐색 경로 높음
정서 데이터 좌절감, 참여도 추론 높음
메타데이터 접속 시간, 기기 정보 중간

5.3 규정 준수

5.3.1 GDPR (일반 데이터 보호 규정)

// GDPR 준수 요구사항
interface GDPRCompliance {
    // 법적 근거
    legalBasis: 'consent' | 'contract' | 'legitimate_interest';

    // 데이터 주체 권리
    dataSubjectRights: {
        access: boolean;          // 접근권
        rectification: boolean;   // 정정권
        erasure: boolean;         // 삭제권
        portability: boolean;     // 이동권
        objection: boolean;       // 반대권
    };

    // 데이터 처리 원칙
    principles: {
        lawfulness: boolean;
        purposeLimitation: boolean;
        dataMinimization: boolean;
        accuracy: boolean;
        storageLimitation: boolean;
        integrityConfidentiality: boolean;
    };

    // 아동 데이터 (16세 미만)
    childProtection: {
        parentalConsent: boolean;
        ageVerification: boolean;
    };
}

5.3.2 FERPA (미국 가정교육권 및 프라이버시법)

  • 학생 교육 기록에 대한 부모/학생 권리
  • 제3자 공개 제한
  • 기록 수정 요청 권리
  • 디렉토리 정보 옵트아웃

5.3.3 COPPA (아동 온라인 프라이버시 보호법)

  • 13세 미만 아동 데이터 수집 시 부모 동의 필수
  • 명확한 개인정보 처리방침
  • 최소 데이터 수집 원칙
  • 부모의 데이터 접근/삭제 권리

5.4 데이터 최소화

// 데이터 최소화 구현
class DataMinimization {
    // 수집 시점 필터링
    filterCollection(data: RawData): MinimizedData {
        return {
            // 필수 데이터만 유지
            essential: this.extractEssential(data),

            // 민감 데이터 익명화
            anonymized: this.anonymize(data.sensitive),

            // 불필요 데이터 제거
            // (예: 정확한 위치 대신 국가만)
        };
    }

    // 저장 기간 관리
    applyRetentionPolicy(data: StoredData): void {
        const policy = this.getRetentionPolicy(data.type);

        if (this.isExpired(data.createdAt, policy.duration)) {
            if (policy.action === 'delete') {
                this.deleteData(data.id);
            } else if (policy.action === 'anonymize') {
                this.anonymizeData(data.id);
            } else if (policy.action === 'aggregate') {
                this.aggregateData(data.id);
            }
        }
    }

    // 목적 제한
    validatePurpose(data: LearnerData, requestedUse: string): boolean {
        const allowedPurposes = data.consentedPurposes;
        return allowedPurposes.includes(requestedUse);
    }
}

5.5 동의 관리

// 동의 관리 시스템
interface ConsentManagement {
    // 동의 수집
    collectConsent(learner: Learner, purposes: Purpose[]): ConsentRecord;

    // 동의 검증
    verifyConsent(learnerId: string, purpose: Purpose): boolean;

    // 동의 철회
    withdrawConsent(learnerId: string, purpose: Purpose): void;

    // 동의 기록 조회
    getConsentHistory(learnerId: string): ConsentRecord[];
}

class ConsentManager implements ConsentManagement {
    async collectConsent(learner: Learner, purposes: Purpose[]): Promise<ConsentRecord> {
        // 연령 확인
        if (learner.age < 16) {
            // 부모/보호자 동의 필요
            const parentConsent = await this.requestParentalConsent(learner, purposes);
            if (!parentConsent.granted) {
                throw new Error('부모 동의가 필요합니다');
            }
        }

        const record: ConsentRecord = {
            learnerId: learner.id,
            purposes: purposes,
            grantedAt: new Date(),
            method: 'explicit_opt_in',
            version: this.currentPolicyVersion,
            parentalConsent: learner.age < 16 ? parentConsent : null
        };

        await this.storeConsent(record);
        return record;
    }
}
명시적 동의: 적응형 학습에서의 데이터 수집은 명시적이고 구체적인 동의를 기반으로 해야 합니다. "묵시적 동의"나 "사전 체크된 박스"는 유효하지 않습니다.

5.6 데이터 보안

// 보안 요구사항 구현
class LearningDataSecurity {
    // 암호화
    encryption: {
        atRest: 'AES-256-GCM',
        inTransit: 'TLS 1.3',
        fields: ['personalInfo', 'assessmentResults', 'behaviorData']
    };

    // 접근 제어
    accessControl: {
        model: 'RBAC',
        roles: ['learner', 'instructor', 'admin', 'researcher'],
        auditLogging: true
    };

    // 데이터 분리
    async segregateData(learnerId: string): Promise<void> {
        // 학습자별 데이터 격리
        await this.createIsolatedNamespace(learnerId);
    }

    // 안전한 삭제
    async secureDelete(learnerId: string): Promise<void> {
        // 모든 저장소에서 완전 삭제
        await Promise.all([
            this.deleteFromDatabase(learnerId),
            this.deleteFromCache(learnerId),
            this.deleteFromBackups(learnerId),
            this.deleteFromAnalytics(learnerId)
        ]);

        // 삭제 증명 생성
        await this.generateDeletionCertificate(learnerId);
    }
}

5.8 데이터 주체 권리 구현

// 데이터 주체 권리 API
class DataSubjectRights {
    // 접근권 - 자신의 데이터 조회
    async accessData(learnerId: string): Promise<LearnerDataExport> {
        const data = await this.collectAllData(learnerId);
        return {
            format: 'JSON',
            data: data,
            exportedAt: new Date(),
            dataCategories: Object.keys(data)
        };
    }

    // 정정권 - 부정확한 데이터 수정
    async rectifyData(learnerId: string, corrections: Correction[]): Promise<void> {
        for (const correction of corrections) {
            await this.validateCorrection(correction);
            await this.applyCorrection(learnerId, correction);
            await this.logCorrection(learnerId, correction);
        }
    }

    // 삭제권 (잊힐 권리)
    async eraseData(learnerId: string, scope: 'all' | 'specific'): Promise<void> {
        if (scope === 'all') {
            await this.securityService.secureDelete(learnerId);
        } else {
            // 특정 데이터만 삭제
        }
    }

    // 이동권 - 데이터 이전
    async portData(learnerId: string, destination: DataDestination): Promise<void> {
        const data = await this.accessData(learnerId);
        const standardFormat = this.convertToStandardFormat(data);
        await this.transferData(standardFormat, destination);
    }
}
투명성: 학습자와 보호자에게 어떤 데이터가 수집되고, 어떻게 사용되며, 누가 접근할 수 있는지 명확하게 알려야 합니다. 개인정보 처리방침은 이해하기 쉬운 언어로 작성되어야 합니다.
