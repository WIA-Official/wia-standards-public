Dwell Selection Implementation

Core interaction method for AAC users:

class DwellSelectionManager {
  constructor(dwellTime = 800, cooldownTime = 500) {
    this.dwellTime = dwellTime;
    this.cooldownTime = cooldownTime;
    this.currentDwell = null;
    this.lastSelection = 0;
  }

  onGazeEnter(target, timestamp) {
    // Check cooldown period
    if (timestamp - this.lastSelection < this.cooldownTime) {
      return;
    }

    // Start new dwell
    this.currentDwell = {
      target: target,
      startTime: timestamp,
      progress: 0
    };

    this.showDwellFeedback(target, 0);
  }

  onGazeUpdate(target, timestamp) {
    if (!this.currentDwell || this.currentDwell.target !== target) {
      return;
    }

    const elapsed = timestamp - this.currentDwell.startTime;
    const progress = Math.min(elapsed / this.dwellTime, 1.0);

    this.currentDwell.progress = progress;
    this.showDwellFeedback(target, progress);

    if (progress >= 1.0) {
      this.triggerSelection(target);
    }
  }

  onGazeLeave(target, timestamp) {
    if (this.currentDwell && this.currentDwell.target === target) {
      this.cancelDwell(target);
    }
  }

  triggerSelection(target) {
    console.log('Selection triggered:', target);
    this.lastSelection = Date.now();
    this.currentDwell = null;
    this.hideDwellFeedback();

    // Trigger target action
    target.onSelect();
  }

  showDwellFeedback(target, progress) {
    // Visual feedback (e.g., progress circle)
    target.element.style.setProperty('--dwell-progress', progress);
  }
}

Adaptive Dwell Time

Automatically adjust dwell time based on user performance:

class AdaptiveDwellManager {
  constructor() {
    this.baseTime = 800;
    this.successCount = 0;
    this.errorCount = 0;
  }

  recordSuccess() {
    this.successCount++;
    if (this.successCount >= 10 && this.errorCount < 2) {
      // User is proficient, reduce dwell time
      this.baseTime = Math.max(500, this.baseTime - 50);
      this.resetCounters();
    }
  }

  recordError() {
    this.errorCount++;
    if (this.errorCount >= 3) {
      // User struggling, increase dwell time
      this.baseTime = Math.min(1500, this.baseTime + 100);
      this.resetCounters();
    }
  }

  getCurrentDwellTime() {
    return this.baseTime;
  }
}

5.2 Computer Control Integration

Gaze Mouse Emulation

class GazeMouseController {
  constructor(gazeStream) {
    this.smoothingWindow = [];
    this.smoothingSize = 5;
    this.clickMode = 'dwell'; // dwell | blink | switch
  }

  onGazePoint(point) {
    // Apply smoothing
    const smoothed = this.smoothGazePoint(point);

    // Convert to screen coordinates
    const screenPos = this.toScreenCoordinates(smoothed);

    // Move system cursor
    this.moveCursor(screenPos.x, screenPos.y);

    // Check for click trigger
    if (this.shouldTriggerClick(point)) {
      this.performClick(screenPos);
    }
  }

  smoothGazePoint(point) {
    this.smoothingWindow.push(point);
    if (this.smoothingWindow.length > this.smoothingSize) {
      this.smoothingWindow.shift();
    }

    // Average recent points
    const avgX = this.smoothingWindow.reduce((sum, p) => sum + p.x, 0) /
                 this.smoothingWindow.length;
    const avgY = this.smoothingWindow.reduce((sum, p) => sum + p.y, 0) /
                 this.smoothingWindow.length;

    return { x: avgX, y: avgY };
  }

  toScreenCoordinates(normalizedPoint) {
    return {
      x: normalizedPoint.x * screen.width,
      y: normalizedPoint.y * screen.height
    };
  }
}

Scrolling Implementation

class GazeScrollManager {
  constructor() {
    this.scrollZones = {
      top: { y: 0, height: 0.1 },      // Top 10% of screen
      bottom: { y: 0.9, height: 0.1 }  // Bottom 10% of screen
    };
    this.scrollSpeed = 2; // pixels per frame
  }

  onGazePoint(point) {
    if (point.y < this.scrollZones.top.height) {
      // Scroll up
      window.scrollBy(0, -this.scrollSpeed);
    } else if (point.y > this.scrollZones.bottom.y) {
      // Scroll down
      window.scrollBy(0, this.scrollSpeed);
    }
  }
}

5.3 Gaming Integration

Gaze-Assisted Aiming

class GazeAimingSystem {
  constructor(sensitivity = 0.7) {
    this.sensitivity = sensitivity;
    this.aimAssistRadius = 0.05; // 5% of screen
  }

  updateAim(gazePoint, targets) {
    // Find nearest target
    const nearestTarget = this.findNearestTarget(gazePoint, targets);

    if (nearestTarget && nearestTarget.distance < this.aimAssistRadius) {
      // Snap to target with aim assist
      return this.applyAimAssist(gazePoint, nearestTarget);
    }

    // Free aim
    return gazePoint;
  }

  findNearestTarget(gazePoint, targets) {
    let nearest = null;
    let minDistance = Infinity;

    for (const target of targets) {
      const dx = gazePoint.x - target.x;
      const dy = gazePoint.y - target.y;
      const distance = Math.sqrt(dx * dx + dy * dy);

      if (distance < minDistance) {
        minDistance = distance;
        nearest = { target, distance };
      }
    }

    return nearest;
  }

  applyAimAssist(gazePoint, nearestTarget) {
    // Smoothly pull aim toward target
    const pullStrength = 0.3; // 30% pull
    return {
      x: gazePoint.x + (nearestTarget.target.x - gazePoint.x) * pullStrength,
      y: gazePoint.y + (nearestTarget.target.y - gazePoint.y) * pullStrength
    };
  }
}

Attention-Based Game Mechanics

class GazeBasedGameplay {
  constructor() {
    this.gazedObjects = new Map();
    this.activationTime = 1000; // ms
  }

  onFixation(event) {
    const object = this.getGameObjectAt(event.centroid);

    if (!object) return;

    if (!this.gazedObjects.has(object.id)) {
      this.gazedObjects.set(object.id, {
        startTime: event.timestamp,
        object: object
      });
    }

    const gazeData = this.gazedObjects.get(object.id);
    const gazeDuration = event.timestamp - gazeData.startTime;

    if (gazeDuration >= this.activationTime) {
      this.activateObject(object);
      this.gazedObjects.delete(object.id);
    }
  }

  activateObject(object) {
    // Example: Looking at a door opens it
    if (object.type === 'door') {
      object.open();
    }
    // Example: Looking at an item picks it up
    else if (object.type === 'item') {
      object.pickup();
    }
  }
}

5.4 Research Applications

Reading Study Setup

class ReadingStudyRecorder {
  constructor() {
    this.gazeData = [];
    this.fixations = [];
    this.textRegions = new Map();
  }

  defineTextRegion(regionId, boundingBox, text) {
    this.textRegions.set(regionId, {
      id: regionId,
      bounds: boundingBox,
      text: text,
      fixationCount: 0,
      totalDuration: 0,
      firstFixation: null,
      lastFixation: null
    });
  }

  onFixation(event) {
    this.fixations.push(event);

    // Find which text region this fixation falls in
    for (const [regionId, region] of this.textRegions) {
      if (this.pointInBounds(event.centroid, region.bounds)) {
        region.fixationCount++;
        region.totalDuration += event.duration;

        if (!region.firstFixation) {
          region.firstFixation = event.timestamp;
        }
        region.lastFixation = event.timestamp;
      }
    }
  }

  generateReport() {
    const report = {
      totalFixations: this.fixations.length,
      averageFixationDuration: this.calculateAverageFixation(),
      regions: []
    };

    for (const [regionId, region] of this.textRegions) {
      report.regions.push({
        text: region.text,
        fixationCount: region.fixationCount,
        totalDuration: region.totalDuration,
        avgDuration: region.totalDuration / region.fixationCount,
        timeToFirstFixation: region.firstFixation - this.startTime
      });
    }

    return report;
  }
}

Heatmap Generation

class GazeHeatmapGenerator {
  constructor(width, height, resolution = 10) {
    this.width = width;
    this.height = height;
    this.resolution = resolution;
    this.grid = this.createGrid();
  }

  createGrid() {
    const cols = Math.ceil(this.width / this.resolution);
    const rows = Math.ceil(this.height / this.resolution);
    return Array(rows).fill(0).map(() => Array(cols).fill(0));
  }

  addGazePoint(point, duration = 1) {
    const x = Math.floor((point.x * this.width) / this.resolution);
    const y = Math.floor((point.y * this.height) / this.resolution);

    if (x >= 0 && x < this.grid[0].length &&
        y >= 0 && y < this.grid.length) {
      this.grid[y][x] += duration;
    }
  }

  generateHeatmap() {
    // Find max value for normalization
    const maxValue = Math.max(...this.grid.flat());

    // Create canvas
    const canvas = document.createElement('canvas');
    canvas.width = this.width;
    canvas.height = this.height;
    const ctx = canvas.getContext('2d');

    // Draw heatmap
    for (let y = 0; y < this.grid.length; y++) {
      for (let x = 0; x < this.grid[y].length; x++) {
        const intensity = this.grid[y][x] / maxValue;
        ctx.fillStyle = this.intensityToColor(intensity);
        ctx.fillRect(
          x * this.resolution,
          y * this.resolution,
          this.resolution,
          this.resolution
        );
      }
    }

    return canvas;
  }

  intensityToColor(intensity) {
    // Blue (low) → Green → Yellow → Red (high)
    const hue = (1 - intensity) * 240; // 240=blue, 0=red
    return `hsl(${hue}, 100%, 50%)`;
  }
}

5.5 Best Practices

Accessibility Guidelines

UI Design for Eye Gaze

  • Target Size: Minimum 1.5° visual angle (80px at typical viewing distance)
  • Spacing: 0.3° minimum gap between interactive elements
  • Contrast: WCAG AAA compliance (7:1 ratio minimum)
  • Feedback: Clear visual indication of dwell progress
  • Error Prevention: Confirmation for destructive actions
  • Undo/Redo: Always available for error recovery

Performance Optimization

  • Throttle Updates: Process gaze data at application frame rate (60Hz), not device rate (120Hz+)
  • Use RAF: Sync gaze processing with requestAnimationFrame
  • Spatial Indexing: Use quadtree/R-tree for fast hit testing
  • Debounce Events: Avoid rapid-fire event handlers
