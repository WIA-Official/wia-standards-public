7.2 WIA AAC Integration

7.2.1 Symbol Selection Feedback

// Haptic feedback for AAC symbol selection
interface AACHapticIntegration {
    // Feedback when hovering over symbol
    onSymbolHover: {
        pattern: 'subtle_tick',
        intensity: 0.3
    };

    // Feedback when symbol is selected
    onSymbolSelect: {
        pattern: 'confirmation.tick',
        intensity: 0.6
    };

    // Feedback when phrase is spoken
    onPhraseSpeak: {
        pattern: 'confirmation.success',
        intensity: 0.7
    };

    // Category change indication
    onCategoryChange: {
        pattern: 'navigation.direction',
        intensity: 0.5
    };
}

7.6 Unified Profile Integration

7.6.1 Cross-Standard Profile

// Unified haptic preferences across WIA standards
interface WIAHapticProfile {
    profileId: string;

    // Global haptic preferences
    globalPreferences: {
        intensity: number;
        frequencyPreference: 'low' | 'mid' | 'high';
        enabled: boolean;
    };

    // Integration-specific overrides
    integrations: {
        aac?: AACHapticPreferences;
        bci?: BCIHapticPreferences;
        eyeGaze?: GazeHapticPreferences;
        navigation?: NavigationHapticPreferences;
    };

    // Sync settings
    sync: {
        enabled: boolean;
        cloudSync: boolean;
        lastSync: string;
    };
}
