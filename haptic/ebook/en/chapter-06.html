6.1 Platform API Mapping

6.1.1 iOS Core Haptics

// WIA Haptic → iOS Core Haptics Mapping

import CoreHaptics

func playWIAHaptic(_ primitive: HapticPrimitive) throws {
    guard CHHapticEngine.capabilitiesForHardware()
        .supportsHaptics else { return }

    let engine = try CHHapticEngine()
    try engine.start()

    // Map WIA waveform to Core Haptics event
    let intensity = CHHapticEventParameter(
        parameterID: .hapticIntensity,
        value: Float(primitive.intensity)
    )

    let sharpness = CHHapticEventParameter(
        parameterID: .hapticSharpness,
        value: waveformToSharpness(primitive.waveform)
    )

    let event = CHHapticEvent(
        eventType: .hapticContinuous,
        parameters: [intensity, sharpness],
        relativeTime: 0,
        duration: Double(primitive.duration) / 1000.0
    )

    // Apply ADSR envelope
    let pattern = try createPatternWithEnvelope(
        event: event,
        envelope: primitive.envelope
    )

    let player = try engine.makePlayer(with: pattern)
    try player.start(atTime: 0)
}

6.1.2 Android Vibration API

// WIA Haptic → Android VibrationEffect Mapping

fun playWIAHaptic(primitive: HapticPrimitive) {
    val vibrator = context.getSystemService(Vibrator::class.java)

    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
        // Modern API with amplitude control
        val amplitudes = generateADSRAmplitudes(
            primitive.envelope,
            primitive.duration
        )

        val timings = LongArray(amplitudes.size) {
            primitive.duration / amplitudes.size.toLong()
        }

        val effect = VibrationEffect.createWaveform(
            timings,
            amplitudes,
            -1 // no repeat
        )

        vibrator.vibrate(effect)
    } else {
        // Legacy fallback
        vibrator.vibrate(primitive.duration.toLong())
    }
}

fun generateADSRAmplitudes(
    envelope: HapticEnvelope,
    duration: Int
): IntArray {
    // Convert ADSR to amplitude array
    val samples = duration / 10 // 10ms resolution
    return IntArray(samples) { i ->
        val t = i * 10
        calculateEnvelopeAmplitude(envelope, t, duration)
    }
}

6.1.3 Web Vibration API

// WIA Haptic → Web Vibration API Mapping

function playWIAHaptic(primitive: HapticPrimitive): void {
    if (!navigator.vibrate) {
        console.warn('Vibration API not supported');
        return;
    }

    // Web API only supports duration patterns
    // Apply graceful degradation
    const pattern = convertToWebPattern(primitive);
    navigator.vibrate(pattern);
}

function convertToWebPattern(primitive: HapticPrimitive): number[] {
    // Convert complex haptic to simple on/off pattern
    const { envelope, duration } = primitive;

    if (envelope.sustain > 0.5) {
        // Sustained pattern: continuous vibration
        return [duration];
    } else {
        // Transient pattern: pulse sequence
        const pulseCount = Math.ceil(duration / 50);
        const pattern: number[] = [];

        for (let i = 0; i < pulseCount; i++) {
            pattern.push(20);  // vibrate
            pattern.push(30);  // pause
        }

        return pattern;
    }
}

6.2 Capability Detection

6.2.1 Capability Schema

interface DeviceCapabilities {
    // Basic haptic support
    hasHaptics: boolean;

    // Advanced features
    hasAmplitudeControl: boolean;
    hasFrequencyControl: boolean;
    hasWaveformSupport: boolean;

    // Frequency range
    frequencyRange: {
        min: number;
        max: number;
    };

    // Intensity resolution
    amplitudeSteps: number;

    // Actuator type (if known)
    actuatorType?: 'erm' | 'lra' | 'piezo' | 'unknown';
}

6.2.2 Capability Detection Implementation

async function detectCapabilities(): Promise<DeviceCapabilities> {
    // Platform-specific detection
    if (isIOS()) {
        return detectIOSCapabilities();
    } else if (isAndroid()) {
        return detectAndroidCapabilities();
    } else if (isWeb()) {
        return detectWebCapabilities();
    }

    return getDefaultCapabilities();
}

function detectWebCapabilities(): DeviceCapabilities {
    return {
        hasHaptics: 'vibrate' in navigator,
        hasAmplitudeControl: false,
        hasFrequencyControl: false,
        hasWaveformSupport: false,
        frequencyRange: { min: 0, max: 0 },
        amplitudeSteps: 1,
        actuatorType: 'unknown'
    };
}

6.3 Graceful Degradation

6.3.1 Degradation Strategies

Missing Capability Degradation Strategy
No amplitude control Use on/off patterns with timing
No frequency control Vary duration and intensity instead
No ADSR support Approximate with stepped patterns
Limited resolution Quantize to available levels
No haptics at all Fall back to audio/visual cues

6.3.2 Degradation Implementation

function adaptPrimitive(
    primitive: HapticPrimitive,
    capabilities: DeviceCapabilities
): AdaptedPrimitive {

    let adapted = { ...primitive };

    // Quantize intensity
    if (capabilities.amplitudeSteps < 256) {
        adapted.intensity = quantize(
            primitive.intensity,
            capabilities.amplitudeSteps
        );
    }

    // Clamp frequency
    adapted.frequency = clamp(
        primitive.frequency,
        capabilities.frequencyRange.min,
        capabilities.frequencyRange.max
    );

    // Simplify envelope if needed
    if (!capabilities.hasWaveformSupport) {
        adapted.envelope = simplifyEnvelope(primitive.envelope);
    }

    return adapted;
}

6.5 Chapter Summary

Key Takeaways:

  1. iOS: Core Haptics with full ADSR and waveform support
  2. Android: VibrationEffect with amplitude arrays
  3. Web: Basic Vibration API with graceful degradation
  4. Capability Detection: Query device features before playback
  5. Degradation: Adapt patterns to device limitations
