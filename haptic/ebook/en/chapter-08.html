8.2 Reference Implementation

8.2.1 TypeScript SDK

// WIA Haptic SDK - Core Implementation

import { HapticPrimitive, HapticPattern, DeviceCapabilities } from './types';
import { iOSAdapter, AndroidAdapter, WebAdapter } from './adapters';

class WIAHapticEngine {
    private adapter: HapticAdapter;
    private capabilities: DeviceCapabilities;

    async initialize(): Promise<void> {
        this.capabilities = await detectCapabilities();
        this.adapter = this.selectAdapter();
        await this.adapter.initialize();
    }

    private selectAdapter(): HapticAdapter {
        if (isIOS()) return new iOSAdapter();
        if (isAndroid()) return new AndroidAdapter();
        return new WebAdapter();
    }

    async playPrimitive(primitive: HapticPrimitive): Promise<void> {
        const adapted = adaptToCapabilities(primitive, this.capabilities);
        await this.adapter.play(adapted);
    }

    async playPattern(patternId: string): Promise<void> {
        const pattern = PatternLibrary.get(patternId);
        if (!pattern) throw new Error(`Unknown pattern: ${patternId}`);

        for (const step of pattern.steps) {
            if (step.delayBefore > 0) {
                await delay(step.delayBefore);
            }

            const primitive = PrimitiveLibrary.get(step.primitiveId);
            await this.playPrimitive(primitive);
        }
    }

    async playBraille(text: string): Promise<void> {
        const encoder = new BrailleEncoder();
        const haptics = encoder.encode(text);

        for (const cell of haptics) {
            await this.playBrailleCell(cell);
            await delay(cell.encoding.cellGap);
        }
    }
}

// Export singleton
export const hapticEngine = new WIAHapticEngine();

8.2.2 Usage Example

// Initialize and use WIA Haptic
import { hapticEngine } from '@wia/haptic';

async function main() {
    await hapticEngine.initialize();

    // Play a confirmation tick
    await hapticEngine.playPattern('confirmation.tick');

    // Play Braille text
    await hapticEngine.playBraille('Hello');

    // Play custom primitive
    await hapticEngine.playPrimitive({
        id: 'custom',
        waveform: { type: 'sine' },
        envelope: { attack: 50, decay: 100, sustain: 0.7, release: 100 },
        frequency: 150,
        intensity: 0.6,
        duration: 300
    });
}

8.4 Testing Tools

8.4.1 Compliance Test Suite

// WIA Haptic Compliance Tester

import { WIAHapticTester } from '@wia/haptic-testing';

async function runComplianceTests(implementation) {
    const tester = new WIAHapticTester(implementation);

    // Primitive Tests
    const primitiveResults = await tester.testPrimitives({
        waveformAccuracy: true,
        envelopeShape: true,
        frequencyPrecision: true,
        intensityLevels: true
    });

    // Pattern Tests
    const patternResults = await tester.testPatterns({
        confirmation: true,
        notification: true,
        navigation: true,
        braille: true
    });

    // Timing Tests
    const timingResults = await tester.testTiming({
        maxLatency: 20, // ms
        jitterTolerance: 5 // ms
    });

    return {
        passed: primitiveResults.passed &&
                patternResults.passed &&
                timingResults.passed,
        details: {
            primitives: primitiveResults,
            patterns: patternResults,
            timing: timingResults
        }
    };
}

8.5 Best Practices

8.5.1 Energy Efficiency

  • Short Patterns: Prefer brief, distinct patterns over long continuous vibrations
  • Adaptive Intensity: Reduce intensity for repeated patterns
  • Coalesce Events: Batch nearby haptic events when possible
  • Idle Detection: Disable haptics when device is idle

8.5.2 User Experience

  • Intensity Calibration: Provide user intensity adjustment
  • Pattern Preview: Allow users to test patterns before selection
  • Learning Mode: Include guided tutorials for pattern vocabulary
  • Quiet Hours: Respect system do-not-disturb settings

8.5.3 Accessibility

Accessibility Checklist:

  • All patterns distinguishable without visual feedback
  • Consistent pattern meanings across app
  • User-configurable intensity levels
  • Alternative audio feedback option
  • Haptic feedback for all critical interactions

8.6 Conclusion

The WIA Haptic Standard provides a comprehensive framework for consistent, learnable haptic feedback in assistive technology. By following this standard, developers can create applications that:

  • Provide consistent tactile experiences across platforms
  • Enable users to learn patterns once and use them everywhere
  • Support Braille encoding for text communication
  • Integrate with the broader WIA accessibility ecosystem
