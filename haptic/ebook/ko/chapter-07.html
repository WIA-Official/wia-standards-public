7.2 WIA AAC 통합

7.2.1 심볼 선택 피드백

// AAC 심볼 선택을 위한 햅틱 피드백
interface AACHapticIntegration {
    // 심볼 위에 호버 시 피드백
    onSymbolHover: {
        pattern: 'subtle_tick',
        intensity: 0.3
    };

    // 심볼 선택 시 피드백
    onSymbolSelect: {
        pattern: 'confirmation.tick',
        intensity: 0.6
    };

    // 문구 발화 시 피드백
    onPhraseSpeak: {
        pattern: 'confirmation.success',
        intensity: 0.7
    };

    // 카테고리 변경 표시
    onCategoryChange: {
        pattern: 'navigation.direction',
        intensity: 0.5
    };
}

// BCI 상호작용을 위한 햅틱 피드백
interface BCIHapticIntegration {
    // 보정 피드백
    calibration: {
        targetAcquired: 'tick',
        calibrationComplete: 'success',
        calibrationFailed: 'error'
    };

    // 선택 피드백
    selection: {
        intentDetected: 'subtle_pulse',
        selectionConfirmed: 'strong_tick',
        selectionCancelled: 'fade_out'
    };

    // 피로 모니터링
    fatigue: {
        mildFatigue: 'gentle_warning',
        moderateFatigue: 'warning_pulse',
        breakRecommended: 'urgent_alert'
    };
}

7.4 WIA Eye Gaze 통합

7.4.1 드웰 선택 피드백

// 시선 드웰 선택을 위한 햅틱 피드백
interface GazeHapticIntegration {
    // 점진적 드웰 피드백
    dwellProgress: {
        type: 'progressive',
        startIntensity: 0.2,
        endIntensity: 0.8,
        curve: 'linear'
    };

    // 선택 확인
    dwellComplete: {
        pattern: 'confirmation.tick'
    };

    // 타겟에서 시선 이탈
    dwellCancelled: {
        pattern: 'fade_out',
        duration: 100
    };
}

7.5 내비게이션 시스템 통합

7.5.1 실외 내비게이션

// 내비게이션을 위한 햅틱 패턴
interface NavigationHaptics {
    // 방향 신호
    direction: {
        straight: 'forward_pulse',
        turnLeft: 'left_sweep',
        turnRight: 'right_sweep',
        uTurn: 'double_buzz'
    };

    // 거리 표시
    distance: {
        near: { intensity: 0.8, pattern: 'rapid_pulse' },
        medium: { intensity: 0.5, pattern: 'steady_pulse' },
        far: { intensity: 0.3, pattern: 'slow_pulse' }
    };

    // 장애물 감지
    obstacle: {
        detected: 'warning_texture',
        avoided: 'clear_path'
    };

    // 도착
    arrival: {
        approaching: 'crescendo',
        arrived: 'success_fanfare'
    };
}

7.6 통합 프로필 관리

7.6.1 크로스 표준 프로필

// WIA 표준 전반의 통합 햅틱 선호도
interface WIAHapticProfile {
    profileId: string;

    // 전역 햅틱 선호도
    globalPreferences: {
        intensity: number;
        frequencyPreference: 'low' | 'mid' | 'high';
        enabled: boolean;
    };

    // 통합별 재정의
    integrations: {
        aac?: AACHapticPreferences;
        bci?: BCIHapticPreferences;
        eyeGaze?: GazeHapticPreferences;
        navigation?: NavigationHapticPreferences;
    };

    // 동기화 설정
    sync: {
        enabled: boolean;
        cloudSync: boolean;
        lastSync: string;
    };
}

7.7 장 요약

핵심 내용:

  1. AAC: 심볼 선택 및 어휘 레벨 피드백

  3. Eye Gaze: 드웰 진행 및 화면 영역 표시

  5. 통합 프로필: 크로스 표준 햅틱 선호도
