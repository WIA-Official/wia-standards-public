7.1 Python Client Implementation

Basic Usage

import omni_api

# Initialize client
client = omni_api.Client(
    base_url="https://api.example.com/omni",
    api_key="your-api-key"
)

# Simple intent-based request
response = client.request(intent="get user 123")

if response.success:
    print(f"User: {response.data['name']}")
    print(f"Protocol used: {response.meta['protocol_used']}")
    print(f"Latency: {response.meta['latency_ms']}ms")
else:
    print(f"Error: {response.error['message']}")
    print(f"Comfort: {response.error['comfort']}")

    # Mother suggests recovery options
    for option in response.error['recovery_options']:
        print(f"  - {option['description']}")

Advanced Features

from omni_api import Client, WiaIntent

client = Client("https://api.example.com/omni")

# Using WIA-INTENT
intent = WiaIntent("""
    intent GetUserDashboard {
        given: user_id = 123
        want: {
            profile: complete
            orders: recent_10
            activity: today
        }
        constraints {
            latency: <= 200ms
            freshness: <= 5min
        }
        fallback {
            if timeout: return cached_version
            if partial_failure: return available_data
        }
    }
""")

response = client.request(
    intent=intent,
    hints={
        "priority": "speed",
        "protocol": "auto"  # Let Mother choose
    }
)

# Response includes rich metadata
print(f"Cache hit: {response.meta['from_cache']}")
if response.meta.get('suggestions'):
    for suggestion in response.meta['suggestions']:
        print(f"Tip: {suggestion}")

Async Support

import asyncio
from omni_api import AsyncClient

async def main():
    async with AsyncClient("https://api.example.com/omni") as client:
        # Parallel requests
        users_task = client.request(intent="list users")
        orders_task = client.request(intent="list recent orders")

        users, orders = await asyncio.gather(users_task, orders_task)

        print(f"Users: {len(users.data)}")
        print(f"Orders: {len(orders.data)}")

asyncio.run(main())

Error Handling with Recovery

from omni_api import Client, OmniError

client = Client("https://api.example.com/omni")

try:
    response = client.request(intent="get user 999")
except OmniError as e:
    print(f"Error: {e.message}")
    print(f"Comfort: {e.comfort}")

    # Auto-recover if possible
    if e.auto_recoverable:
        recovery = e.auto_recover()
        print(f"Auto-recovered: {recovery}")
    else:
        # Show recovery options to user
        for option in e.recovery_options:
            print(f"  {option['action']}: {option['description']}")
            if option['auto_applicable']:
                result = client.apply_recovery(option)
                print(f"    Result: {result}")
                break

TypeScript with Type Safety

import { OmniClient, OmniRequest, OmniResponse } from '@wia/omni-api-client';

interface User {
  id: number;
  name: string;
  email: string;
}

interface UserResponse extends OmniResponse {
  data: User;
}

const client = new OmniClient('https://api.example.com/omni');

// Type-safe request
const response = await client.request<UserResponse>({
  intent: 'get user 123'
});

if (response.success) {
  // TypeScript knows the shape of response.data
  const user: User = response.data;
  console.log(user.name.toUpperCase());
}

React Hook

import { useOmni } from '@wia/omni-api-react';

function UserProfile({ userId }) {
  const { data, loading, error, refetch } = useOmni({
    intent: `get user ${userId}`
  });

  if (loading) return <div>Loading...</div>;

  if (error) return (
    <div>
      <p>Error: {error.message}</p>
      <p>{error.comfort}</p>
      {error.recovery_options.map(option => (
        <button key={option.action} onClick={() => applyRecovery(option)}>
          {option.description}
        </button>
      ))}
    </div>
  );

  return (
    <div>
      <h1>{data.name}</h1>
      <p>{data.email}</p>
      <button onClick={refetch}>Refresh</button>
    </div>
  );
}

Node.js Server-Side

import { OmniClient } from '@wia/omni-api-client';
import express from 'express';

const client = new OmniClient({
  baseUrl: 'https://api.example.com/omni',
  apiKey: process.env.OMNI_API_KEY,
  hints: {
    protocol: 'grpc',  // Prefer gRPC for server-to-server
    priority: 'speed'
  }
});

const app = express();

app.get('/users/:id', async (req, res) => {
  try {
    const response = await client.request({
      intent: `get user ${req.params.id}`
    });

    if (response.success) {
      res.json(response.data);
    } else {
      res.status(response.error.http_status || 500)
         .json(response.error);
    }
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);

Bash Script Example

#!/bin/bash

# Bulk operations using OMNI-API
set -e

OMNI_BASE="https://api.example.com/omni"
API_KEY="your-api-key"

# Function to make OMNI request
omni_request() {
  local intent="$1"
  local payload="$2"

  curl -s -X POST "$OMNI_BASE" \
    -H "Authorization: Bearer $API_KEY" \
    -H "Content-Type: application/json" \
    -d "{\"intent\": \"$intent\", \"payload\": $payload}"
}

# Get user list
users=$(omni_request "list active users" "{}")

# Process each user
echo "$users" | jq -r '.data[].id' | while read user_id; do
  echo "Processing user $user_id..."

  # Get user details (Mother handles caching)
  user=$(omni_request "get user $user_id" "{}")

  # Extract data (Mother ensures format consistency)
  name=$(echo "$user" | jq -r '.data.name')

  echo "  User: $name"
done

echo "Done!"

TypeScript Server (Express)

import express from 'express';
import { OmniGateway } from '@wia/omni-api-server';

const app = express();
const gateway = new OmniGateway({
  evolution: {
    enabled: true,
    autoOptimize: ['caching', 'batching']
  }
});

// Register backends
gateway.registerBackend('users', {
  url: 'https://users-service.internal',
  protocol: 'rest'
});

gateway.registerBackend('orders', {
  url: 'https://orders-service.internal',
  protocol: 'grpc'
});

// Register intent handlers
gateway.onIntent('get user dashboard', async (context) => {
  // Mother coordinates calls to multiple backends
  const [user, orders] = await Promise.all([
    context.backend('users').get(context.params.user_id),
    context.backend('orders').list({ user_id: context.params.user_id })
  ]);

  return {
    success: true,
    data: { user, orders }
  };
});

// Mount gateway
app.use('/omni', gateway.middleware());

app.listen(8080);

7.5 Testing

Unit Tests (Python)

import pytest
from omni_api import Client
from omni_api.testing import MockOmni

def test_user_retrieval():
    # Mock OMNI responses
    with MockOmni() as mock:
        mock.on_intent("get user 123").returns({
            "success": True,
            "data": {"id": 123, "name": "Kim"}
        })

        client = Client(mock.url)
        response = client.request(intent="get user 123")

        assert response.success
        assert response.data["name"] == "Kim"

def test_error_handling():
    with MockOmni() as mock:
        mock.on_intent("get user 999").returns({
            "success": False,
            "error": {
                "code": "USER_NOT_FOUND",
                "message": "User not found"
            }
        })

        client = Client(mock.url)
        response = client.request(intent="get user 999")

        assert not response.success
        assert response.error["code"] == "USER_NOT_FOUND"

Integration Tests (TypeScript)

import { OmniClient } from '@wia/omni-api-client';
import { describe, it, expect } from 'vitest';

describe('OMNI API Integration', () => {
  const client = new OmniClient(process.env.OMNI_TEST_URL);

  it('should retrieve user by intent', async () => {
    const response = await client.request({
      intent: 'get user 123'
    });

    expect(response.success).toBe(true);
    expect(response.data.id).toBe(123);
    expect(response.meta.protocol_used).toBeTruthy();
  });

  it('should handle protocol translation', async () => {
    // Request via REST intent
    const restResponse = await client.request({
      intent: 'GET /users/123',
      hints: { protocol: 'rest' }
    });

    // Request via GraphQL intent
    const graphqlResponse = await client.request({
      intent: 'query { user(id: 123) { name } }',
      hints: { protocol: 'graphql' }
    });

    // Both should return same user
    expect(restResponse.data.id).toBe(graphqlResponse.data.id);
  });
});
