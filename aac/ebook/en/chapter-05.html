5.1 Overview of Phase 2

5.1.1 Purpose

Phase 2 defines the API Interface Standard - a unified programming interface that software developers use to interact with any WIA AAC compliant sensor. This layer abstracts the complexity of different sensor types behind a consistent API.

5.2 Core API Design

5.2.1 Design Principles

  1. Simplicity First

    Common operations should require minimal code. Complex operations should be possible but not required.

  2. Event-Driven Architecture

    All sensor input is handled through events, enabling reactive programming patterns.

  3. Type Safety

    Strong typing prevents common errors and enables IDE support.

  4. Async by Default

    All I/O operations are asynchronous to prevent blocking.

  5. Graceful Error Handling

    Clear error types and recovery mechanisms.

5.3.1 Constructor and Initialization

// TypeScript Example

import { WiaAAC, WiaAACConfig } from 'wia-aac';

// Basic initialization
const aac = new WiaAAC();

// With configuration
const aac = new WiaAAC({
    autoConnect: false,
    signalValidation: true,

    autoConnect?: boolean;        // Auto-connect to discovered devices
    signalValidation?: boolean;   // Validate all incoming signals

    calibrationId?: string;       // Calibration data identifier

5.3.3 Event Handling

// Signal events
aac.on('signal', (signal: AACSignal) => {
    console.log('Received signal:', signal.type);
    console.log('Data:', signal.data);
    console.log('Confidence:', signal.confidence);
});

// Specific sensor type events
aac.on('eye_tracker', (signal: EyeTrackerSignal) => {
    console.log(`Gaze: (${signal.data.gaze_x}, ${signal.data.gaze_y})`);
    if (signal.data.fixation) {
        console.log(`Dwell time: ${signal.data.dwell_time_ms}ms`);
    }
});

aac.on('switch', (signal: SwitchSignal) => {
    console.log(`Switch ${signal.data.switch_id}: ${signal.data.state}`);
});

// Selection events (abstracted)
aac.on('selection', (event: SelectionEvent) => {
    console.log('Selection made:', event.target);
    console.log('Selection method:', event.method);  // dwell, switch, blink, etc.
});

// Text output events
aac.on('text', (text: string) => {
    console.log('User typed:', text);
    // Forward to TTS, sign language, braille...
});

// Device events
aac.on('connected', (device: DeviceInfo) => {
    console.log('Connected to:', device.model);
});

aac.on('disconnected', (reason: string) => {
    console.log('Disconnected:', reason);
});

// Error events
aac.on('error', (error: AACError) => {
    console.error('Error:', error.code, error.message);
});

5.7 Rust Reference Implementation

5.7.1 Core Structure

// Rust Implementation Structure

use wia_aac::{WiaAAC, Config, Signal, SensorType};
use tokio;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Initialize
    let config = Config::builder()
        .signal_validation(true)
        .log_level("info")
        .build();

    let aac = WiaAAC::new(config)?;

    // Discover devices
    let devices = aac.discover_devices().await?;
    println!("Found {} devices", devices.len());

    // Connect
    aac.connect(SensorType::EyeTracker, None).await?;

    // Handle signals
    let mut rx = aac.subscribe();
    while let Some(signal) = rx.recv().await {
        match signal.sensor_type {
            SensorType::EyeTracker => {
                let data = signal.as_eye_tracker()?;
                println!("Gaze: ({}, {})", data.gaze_x, data.gaze_y);
            }
            SensorType::Switch => {
                let data = signal.as_switch()?;
                println!("Switch {}: {}", data.switch_id, data.state);
            }
            _ => {}
        }
    }

    Ok(())
}

5.7.3 Axum Web Server Integration

// Rust Axum API Server

use axum::{
    routing::{get, post},
    Router, Json, extract::State,
};
use wia_aac::{WiaAAC, Signal};
use std::sync::Arc;
use tokio::sync::RwLock;

type AppState = Arc<RwLock<WiaAAC>>;

#[tokio::main]
async fn main() {
    let aac = Arc::new(RwLock::new(WiaAAC::new_default().unwrap()));

    let app = Router::new()
        .route("/devices", get(list_devices))
        .route("/connect", post(connect_device))
        .route("/disconnect", post(disconnect_device))
        .route("/signal", get(get_latest_signal))
        .route("/profile", post(create_profile))
        .with_state(aac);

    axum::Server::bind(&"0.0.0.0:3000".parse().unwrap())
        .serve(app.into_make_service())
        .await
        .unwrap();
}

async fn list_devices(
    State(aac): State<AppState>
) -> Json<Vec<DeviceInfo>> {
    let aac = aac.read().await;
    Json(aac.discover_devices().await.unwrap_or_default())
}

async fn connect_device(
    State(aac): State<AppState>,
    Json(options): Json<ConnectionOptions>
) -> Result<Json<DeviceInfo>, StatusCode> {
    let mut aac = aac.write().await;
    match aac.connect(options.sensor_type, options.device_id).await {
        Ok(info) => Ok(Json(info)),
        Err(_) => Err(StatusCode::INTERNAL_SERVER_ERROR)
    }
}

5.8 Error Handling

5.8.1 Error Types

// Error type enumeration
enum AACErrorCode {
    // Connection errors
    CONNECTION_TIMEOUT = 'E001',
    DEVICE_NOT_FOUND = 'E002',
    DEVICE_BUSY = 'E003',
    CONNECTION_LOST = 'E004',

    // Signal errors
    INVALID_SIGNAL = 'E101',
    VALIDATION_FAILED = 'E102',
    SIGNAL_TIMEOUT = 'E103',

    // Calibration errors
    CALIBRATION_FAILED = 'E201',
    CALIBRATION_INCOMPLETE = 'E202',

    // Profile errors
    PROFILE_NOT_FOUND = 'E301',
    PROFILE_CORRUPT = 'E302',

    // System errors
    PERMISSION_DENIED = 'E401',
    RESOURCE_UNAVAILABLE = 'E402',
    INTERNAL_ERROR = 'E500'
}

// Error handling
try {
    await aac.connect({ type: 'eye_tracker' });
} catch (error) {
    if (error instanceof AACError) {
        switch (error.code) {
            case 'E001':
                console.log('Connection timed out. Check device is powered on.');
                break;
            case 'E002':
                console.log('Device not found. Run discovery first.');
                break;
            case 'E003':
                console.log('Device is in use by another application.');
                break;
            default:
                console.log('Error:', error.message);
        }
    }
}

5.8.2 Recovery Strategies

// Auto-reconnect configuration
const aac = new WiaAAC({
    autoReconnect: true,
    reconnectAttempts: 5,
    reconnectDelay: 1000,       // Initial delay
    reconnectBackoff: 'exponential'  // 1s, 2s, 4s, 8s, 16s
});

// Manual reconnection
aac.on('disconnected', async (reason) => {
    console.log('Disconnected:', reason);

    // Attempt reconnection
    for (let i = 0; i < 3; i++) {
        try {
            await aac.reconnect();
            console.log('Reconnected successfully');
            break;
        } catch (error) {
            console.log(`Reconnection attempt ${i + 1} failed`);
            await sleep(2000 * (i + 1));
        }
    }
});

// Fallback to different sensor
aac.on('error', async (error) => {
    if (error.code === 'E002' && currentSensor === 'eye_tracker') {
        console.log('Eye tracker unavailable, trying switch input');
        await aac.connect({ type: 'switch' });
    }
});
