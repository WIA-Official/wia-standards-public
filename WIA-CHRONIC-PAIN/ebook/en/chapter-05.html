Technical Safeguards Implementation

class HIPAASecurityManager {
  private encryptionService: EncryptionService;
  private auditLogger: AuditLogger;
  private accessController: AccessController;

  async protectPHI(data: HealthData): Promise<ProtectedData> {
    // Validate data classification
    const classification = this.classifyData(data);

    if (classification.containsPHI) {
      // Apply encryption
      const encrypted = await this.encryptionService.encrypt(data, {
        algorithm: 'AES-256-GCM',
        keyId: await this.getActiveKeyId(),
        associatedData: this.buildAAD(data)
      });

      // Log access
      await this.auditLogger.logDataAccess({
        action: 'ENCRYPT',
        dataType: classification.dataTypes,
        timestamp: new Date().toISOString(),
        userId: this.getCurrentUser(),
        purpose: 'DATA_PROTECTION'
      });

      return encrypted;
    }

    return data;
  }

  async authorizeAccess(
    userId: string,
    resource: Resource,
    action: Action
  ): Promise<AuthorizationResult> {
    // Check user permissions
    const permissions = await this.accessController.getUserPermissions(userId);

    // Verify minimum necessary access
    const isMinimumNecessary = this.verifyMinimumNecessary(
      permissions,
      resource,
      action
    );

    // Check for break-glass scenarios
    const breakGlass = await this.checkBreakGlass(userId, resource);

    // Log authorization decision
    await this.auditLogger.logAuthorization({
      userId,
      resource: resource.id,
      action,
      granted: isMinimumNecessary || breakGlass,
      timestamp: new Date().toISOString()
    });

    return {
      authorized: isMinimumNecessary || breakGlass,
      reason: isMinimumNecessary ? 'STANDARD_ACCESS' :
              breakGlass ? 'BREAK_GLASS' : 'DENIED'
    };
  }
}

class ColumnLevelEncryption {
  private kms: KeyManagementService;

  async encryptColumn(
    value: string,
    context: EncryptionContext
  ): Promise<EncryptedValue> {
    // Get data encryption key
    const dek = await this.kms.generateDataKey({
      keyId: context.masterKeyId,
      context: {
        table: context.table,
        column: context.column,
        purpose: 'PHI_PROTECTION'
      }
    });

    // Encrypt value with DEK
    const iv = crypto.randomBytes(12);
    const cipher = crypto.createCipheriv('aes-256-gcm', dek.plaintext, iv);
    cipher.setAAD(Buffer.from(context.aad));

    const encrypted = Buffer.concat([
      cipher.update(value, 'utf8'),
      cipher.final()
    ]);

    const authTag = cipher.getAuthTag();

    // Clear plaintext DEK from memory
    dek.plaintext.fill(0);

    return {
      ciphertext: encrypted.toString('base64'),
      iv: iv.toString('base64'),
      authTag: authTag.toString('base64'),
      encryptedDEK: dek.encrypted,
      keyVersion: dek.keyVersion
    };
  }
}

Consent Data Model

interface PatientConsent {
  id: string;
  patientId: string;
  version: string;
  timestamp: string;
  categories: ConsentCategory[];
  preferences: ConsentPreference[];
  signatures: ConsentSignature[];
  revocationHistory: RevocationRecord[];
}

interface ConsentCategory {
  category: string;
  description: string;
  required: boolean;
  granted: boolean;
  purposes: ConsentPurpose[];
}

enum ConsentPurpose {
  TREATMENT = 'treatment',
  CARE_COORDINATION = 'care_coordination',
  QUALITY_IMPROVEMENT = 'quality_improvement',
  RESEARCH = 'research',
  MARKETING = 'marketing',
  THIRD_PARTY_SHARING = 'third_party_sharing'
}

class ConsentManager {
  async captureConsent(
    patientId: string,
    consentData: ConsentInput
  ): Promise<PatientConsent> {
    // Validate consent completeness
    this.validateConsentCompleteness(consentData);

    // Create consent record
    const consent: PatientConsent = {
      id: generateUUID(),
      patientId,
      version: CURRENT_CONSENT_VERSION,
      timestamp: new Date().toISOString(),
      categories: this.processCategories(consentData.categories),
      preferences: consentData.preferences,
      signatures: [{
        type: 'electronic',
        timestamp: new Date().toISOString(),
        method: consentData.signatureMethod,
        ipAddress: this.hashIP(consentData.ipAddress),
        userAgent: consentData.userAgent
      }],
      revocationHistory: []
    };

    // Store with audit trail
    await this.storage.saveConsent(consent);
    await this.auditLogger.logConsentCapture(consent);

    // Notify downstream systems
    await this.publishConsentEvent({
      type: 'CONSENT_GRANTED',
      patientId,
      consentId: consent.id,
      categories: consent.categories.filter(c => c.granted).map(c => c.category)
    });

    return consent;
  }

  async checkConsent(
    patientId: string,
    purpose: ConsentPurpose
  ): Promise<ConsentCheckResult> {
    const consent = await this.getCurrentConsent(patientId);

    if (!consent) {
      return { allowed: false, reason: 'NO_CONSENT_ON_FILE' };
    }

    const category = consent.categories.find(c =>
      c.purposes.includes(purpose)
    );

    if (!category) {
      return { allowed: false, reason: 'PURPOSE_NOT_COVERED' };
    }

    return {
      allowed: category.granted,
      reason: category.granted ? 'CONSENT_GRANTED' : 'CONSENT_DENIED',
      consentId: consent.id,
      consentVersion: consent.version
    };
  }
}

Audit Logging

interface AuditLogEntry {
  id: string;
  timestamp: string;
  eventType: AuditEventType;
  actor: {
    userId: string;
    userType: 'patient' | 'provider' | 'system' | 'admin';
    ipAddress: string;
    userAgent: string;
  };
  resource: {
    type: string;
    id: string;
    patientId?: string;
  };
  action: string;
  outcome: 'success' | 'failure' | 'partial';
  details: Record<string, unknown>;
  hash: string;
}

enum AuditEventType {
  AUTHENTICATION = 'authentication',
  AUTHORIZATION = 'authorization',
  DATA_ACCESS = 'data_access',
  DATA_MODIFICATION = 'data_modification',
  DATA_EXPORT = 'data_export',
  CONSENT_CHANGE = 'consent_change',
  SECURITY_EVENT = 'security_event',
  SYSTEM_EVENT = 'system_event'
}

class AuditLogger {
  private logStore: ImmutableLogStore;
  private integrityVerifier: IntegrityVerifier;

  async log(entry: Omit<AuditLogEntry, 'id' | 'hash'>): Promise<void> {
    const id = generateUUID();
    const previousHash = await this.logStore.getLastHash();

    const hash = this.calculateHash({
      ...entry,
      id,
      previousHash
    });

    const fullEntry: AuditLogEntry = {
      ...entry,
      id,
      hash
    };

    // Write to immutable log store
    await this.logStore.append(fullEntry);

    // Real-time alerting for security events
    if (entry.eventType === AuditEventType.SECURITY_EVENT) {
      await this.alertSecurityTeam(fullEntry);
    }
  }

  async verifyLogIntegrity(startTime: string, endTime: string): Promise<IntegrityReport> {
    const logs = await this.logStore.getRange(startTime, endTime);

    let previousHash = await this.logStore.getHashBefore(startTime);
    const violations: IntegrityViolation[] = [];

    for (const log of logs) {
      const expectedHash = this.calculateHash({
        ...log,
        previousHash
      });

      if (log.hash !== expectedHash) {
        violations.push({
          logId: log.id,
          timestamp: log.timestamp,
          type: 'HASH_MISMATCH'
        });
      }

      previousHash = log.hash;
    }

    return {
      verified: violations.length === 0,
      logsChecked: logs.length,
      violations
    };
  }
}
