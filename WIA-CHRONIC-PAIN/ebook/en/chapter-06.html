SMART on FHIR Integration

class SMARTAuthClient {
  private config: SMARTConfig;

  async launchFromEHR(launchParams: LaunchParams): Promise<AuthResult> {
    // Extract launch context
    const { iss, launch } = launchParams;

    // Discover authorization endpoints
    const metadata = await this.getConformance(iss);
    const authEndpoint = metadata.rest[0].security.extension
      .find(e => e.url.includes('authorize')).valueUri;
    const tokenEndpoint = metadata.rest[0].security.extension
      .find(e => e.url.includes('token')).valueUri;

    // Build authorization request
    const authUrl = new URL(authEndpoint);
    authUrl.searchParams.set('response_type', 'code');
    authUrl.searchParams.set('client_id', this.config.clientId);
    authUrl.searchParams.set('redirect_uri', this.config.redirectUri);
    authUrl.searchParams.set('scope', this.config.scopes.join(' '));
    authUrl.searchParams.set('state', this.generateState());
    authUrl.searchParams.set('aud', iss);
    authUrl.searchParams.set('launch', launch);

    // Redirect to authorization
    return { authorizationUrl: authUrl.toString() };
  }

  async exchangeCodeForToken(
    code: string,
    tokenEndpoint: string
  ): Promise<TokenResponse> {
    const response = await fetch(tokenEndpoint, {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({
        grant_type: 'authorization_code',
        code,
        redirect_uri: this.config.redirectUri,
        client_id: this.config.clientId
      })
    });

    const tokens = await response.json();

    return {
      accessToken: tokens.access_token,
      tokenType: tokens.token_type,
      expiresIn: tokens.expires_in,
      scope: tokens.scope,
      patient: tokens.patient,
      encounter: tokens.encounter
    };
  }
}

EHR Integration Service

interface EHRIntegration {
  platform: EHRPlatform;
  capabilities: EHRCapability[];
  connection: ConnectionConfig;
  mapping: DataMapping;
}

class EHRIntegrationService {
  private adapters: Map<EHRPlatform, EHRAdapter>;

  async syncPatientData(
    patientId: string,
    ehrConnection: EHRConnection
  ): Promise<SyncResult> {
    const adapter = this.adapters.get(ehrConnection.platform);

    // Fetch patient data from EHR
    const ehrData = await adapter.fetchPatientData(
      ehrConnection,
      patientId,
      {
        resources: ['Condition', 'MedicationStatement', 'Observation'],
        dateRange: { start: '2023-01-01', end: 'now' }
      }
    );

    // Transform to internal format
    const transformedData = this.transformEHRData(ehrData, adapter.mapping);

    // Reconcile with existing data
    const reconciled = await this.reconcileData(patientId, transformedData);

    // Store synchronized data
    await this.storage.savePatientData(reconciled);

    return {
      status: 'success',
      resourcesSynced: reconciled.resources.length,
      conflicts: reconciled.conflicts,
      lastSyncTime: new Date().toISOString()
    };
  }

  async pushPainData(
    patientId: string,
    ehrConnection: EHRConnection,
    painData: PainData[]
  ): Promise<PushResult> {
    const adapter = this.adapters.get(ehrConnection.platform);

    // Transform pain data to FHIR observations
    const observations = painData.map(pd =>
      this.createPainObservation(pd, adapter.mapping)
    );

    // Create bundle for batch submission
    const bundle = {
      resourceType: 'Bundle',
      type: 'batch',
      entry: observations.map(obs => ({
        resource: obs,
        request: {
          method: 'POST',
          url: 'Observation'
        }
      }))
    };

    // Submit to EHR
    const result = await adapter.submitBundle(ehrConnection, bundle);

    return {
      status: 'success',
      submitted: observations.length,
      accepted: result.entry.filter(e => e.response.status.startsWith('2')).length
    };
  }
}

Telehealth Platform Integration

interface TelehealthIntegration {
  platform: TelehealthPlatform;
  capabilities: {
    videoVisits: boolean;
    asyncMessaging: boolean;
    remoteMonitoring: boolean;
    ePrescribing: boolean;
  };
  scheduling: SchedulingConfig;
  dataSharing: DataSharingConfig;
}

class TelehealthService {
  async scheduleVisit(
    patientId: string,
    providerId: string,
    visitType: VisitType
  ): Promise<ScheduledVisit> {
    // Check provider availability
    const availability = await this.getProviderAvailability(
      providerId,
      visitType.duration
    );

    // Get patient preferences
    const preferences = await this.getPatientPreferences(patientId);

    // Find optimal slot
    const slot = this.findOptimalSlot(availability, preferences);

    // Create visit in telehealth platform
    const visit = await this.telehealthAdapter.createVisit({
      patientId,
      providerId,
      startTime: slot.start,
      duration: visitType.duration,
      visitType: visitType.code,
      prepareInstructions: this.getPrepareInstructions(visitType)
    });

    // Pre-populate visit context with recent pain data
    await this.prepareVisitContext(visit.id, patientId);

    return visit;
  }

  async prepareVisitContext(visitId: string, patientId: string): Promise<void> {
    const recentPain = await this.getPainSummary(patientId, { days: 14 });
    const medications = await this.getCurrentMedications(patientId);

    await this.telehealthAdapter.attachVisitContext(visitId, {
      painTrend: recentPain.trend,
      averagePain: recentPain.average,
      currentMedications: medications
    });
  }
}

Patient Alert System

class ProviderAlertSystem {
  async evaluatePatientAlerts(providerId: string): Promise<PatientAlert[]> {
    const patients = await this.getProviderPatients(providerId);
    const alerts: PatientAlert[] = [];

    for (const patient of patients) {
      const painSpike = await this.checkPainSpike(patient.id);
      if (painSpike) {
        alerts.push({
          patientId: patient.id,
          alertType: 'PAIN_SPIKE',
          severity: 'high',
          message: `Pain increased to ${painSpike.current}`
        });
      }

      const adherence = await this.getAdherence(patient.id);
      if (adherence.overall < 0.7) {
        alerts.push({
          patientId: patient.id,
          alertType: 'LOW_ADHERENCE',
          severity: 'medium'
        });
      }
    }

    return alerts.sort((a, b) =>
      this.severityOrder(b.severity) - this.severityOrder(a.severity)
    );
  }
}

Pharmacy Integration

interface PharmacyIntegration {
  network: PharmacyNetwork;
  capabilities: {
    ePrescribing: boolean;
    pdmp: boolean;
    refillRequests: boolean;
    medicationHistory: boolean;
  };
}

class PharmacyService {
  async submitPrescription(
    prescription: Prescription
  ): Promise<PrescriptionResult> {
    // Validate prescription
    await this.validatePrescription(prescription);

    // Check PDMP for controlled substances
    if (prescription.isControlled) {
      const pdmpCheck = await this.checkPDMP(
        prescription.patientId,
        prescription.medication
      );

      if (pdmpCheck.alerts.length > 0) {
        return {
          status: 'requires_review',
          alerts: pdmpCheck.alerts
        };
      }
    }

    // Submit via NCPDP SCRIPT
    const result = await this.scriptAdapter.sendNewRx({
      prescriber: prescription.prescriberId,
      patient: prescription.patientId,
      medication: this.formatMedication(prescription),
      pharmacy: prescription.pharmacyId,
      sig: prescription.instructions
    });

    return {
      status: 'submitted',
      trackingNumber: result.trackingId,
      estimatedReady: result.estimatedFillTime
    };
  }
}
