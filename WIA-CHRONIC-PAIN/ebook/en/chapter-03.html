class PainMonitoringEngine {
  private dataStreams: Map<string, DataStream>;
  private patternDetector: PatternDetector;
  private alertManager: AlertManager;

  async processDataPoint(data: MonitoringDataPoint): Promise<void> {
    // Validate and normalize incoming data
    const normalized = await this.normalizeData(data);

    // Store in time-series database
    await this.storage.insert(normalized);

    // Update real-time aggregations
    await this.updateAggregations(data.patientId, normalized);

    // Check for patterns and anomalies
    const patterns = await this.patternDetector.analyze(data.patientId);

    // Generate alerts if needed
    if (patterns.alertWorthy) {
      await this.alertManager.notify(data.patientId, patterns);
    }
  }
}

Adaptive Sampling Strategy

class AdaptiveSampler {
  private baseFrequency: number = 4; // prompts per day

  calculateSamplingSchedule(patientId: string): SamplingSchedule {
    const history = this.getPatientHistory(patientId);
    const preferences = this.getPatientPreferences(patientId);

    // Increase sampling during high-variability periods
    const variabilityMultiplier = this.calculateVariability(history) > 0.5 ? 1.5 : 1.0;

    // Respect patient preferences for quiet hours
    const activeHours = preferences.activeHours || { start: 8, end: 22 };

    // Generate time-stratified random schedule
    const schedule = this.generateStratifiedSchedule({
      frequency: Math.round(this.baseFrequency * variabilityMultiplier),
      startHour: activeHours.start,
      endHour: activeHours.end,
      minimumInterval: 90, // minutes between prompts
      randomization: 0.3   // 30% randomization window
    });

    return schedule;
  }

  async triggerAssessment(patientId: string, type: AssessmentType): Promise<void> {
    // Send push notification
    await this.notifications.send(patientId, {
      title: 'Pain Check-in',
      body: 'How are you feeling right now?',
      data: { assessmentType: type },
      priority: 'high',
      ttl: 1800 // 30 minutes to respond
    });

    // Track prompt delivery
    await this.analytics.track('assessment_prompted', {
      patientId,
      type,
      scheduledTime: new Date()
    });
  }
}

class PainPatternAnalyzer {
  async analyzePainPatterns(patientId: string, window: TimeWindow): Promise<PatternAnalysis> {
    const painData = await this.fetchPainData(patientId, window);

    // Decompose time series
    const decomposition = this.decomposeTimeSeries(painData, {
      frequency: 'daily',
      model: 'additive'
    });

    // Detect trend
    const trend = this.analyzeTrend(decomposition.trend, {
      minDays: 14,
      significanceLevel: 0.05
    });

    // Find cyclic patterns
    const dailyPattern = this.detectDailyPattern(painData);
    const weeklyPattern = this.detectWeeklyPattern(painData);

    // Identify change points
    const changePoints = this.detectChangePoints(painData, {
      algorithm: 'PELT',
      minSegmentLength: 7
    });

    return {
      trend,
      dailyPattern,
      weeklyPattern,
      changePoints,
      summary: this.generateSummary(trend, dailyPattern, weeklyPattern)
    };
  }

  detectDailyPattern(data: PainDataPoint[]): DailyPattern {
    // Group by hour of day
    const hourlyAverages = this.groupByHour(data);

    // Find peak and trough hours
    const peakHour = this.findPeak(hourlyAverages);
    const troughHour = this.findTrough(hourlyAverages);

    // Calculate pattern strength
    const amplitude = hourlyAverages[peakHour] - hourlyAverages[troughHour];
    const strength = amplitude / this.mean(Object.values(hourlyAverages));

    return {
      peakHour,
      troughHour,
      amplitude,
      strength,
      hourlyProfile: hourlyAverages,
      significance: this.testPatternSignificance(hourlyAverages)
    };
  }
}

import tensorflow as tf
from tensorflow.keras import layers

class PainPredictionModel:
    def __init__(self, config):
        self.sequence_length = config.sequence_length
        self.feature_dim = config.feature_dim
        self.model = self._build_model()

    def _build_model(self):
        inputs = layers.Input(shape=(self.sequence_length, self.feature_dim))

        # LSTM layers for temporal patterns
        x = layers.LSTM(64, return_sequences=True)(inputs)
        x = layers.Dropout(0.2)(x)
        x = layers.LSTM(32)(x)
        x = layers.Dropout(0.2)(x)

        # Dense layers for prediction
        x = layers.Dense(16, activation='relu')(x)
        outputs = layers.Dense(1, activation='sigmoid')(x)

        model = tf.keras.Model(inputs, outputs)
        model.compile(
            optimizer='adam',
            loss='mse',
            metrics=['mae']
        )
        return model

    def predict_pain(self, patient_data, horizon=24):
        """Predict pain levels for next 'horizon' hours"""
        features = self.prepare_features(patient_data)
        predictions = []

        for h in range(horizon):
            pred = self.model.predict(features)
            predictions.append({
                'hour': h,
                'predicted_pain': float(pred[0][0] * 10),
                'confidence': self.calculate_confidence(features)
            })
            features = self.update_features(features, pred)

        return predictions

Trigger Identification

Trigger Categories: Physical activities, emotional states, sleep quality, weather conditions, and medication timing are potential pain triggers the system monitors.

Trigger Analysis Framework

interface TriggerAnalysis {
  trigger: {
    category: TriggerCategory;
    specific: string;
    description: string;
  };
  association: {
    correlation: number;
    temporalLag: number; // hours before pain increase
    doseResponse: boolean;
    confidence: number;
  };
  evidence: {
    occurrences: number;
    pain_increase_avg: number;
    last_occurrence: string;
    examples: TriggerExample[];
  };
}

class TriggerDetector {
  async identifyTriggers(patientId: string): Promise<TriggerAnalysis[]> {
    const painData = await this.fetchPainHistory(patientId);
    const contextData = await this.fetchContextHistory(patientId);

    const triggers: TriggerAnalysis[] = [];

    // Analyze activity triggers
    const activityTriggers = await this.analyzeActivityTriggers(
      painData,
      contextData.activities
    );
    triggers.push(...activityTriggers);

    // Analyze sleep triggers
    const sleepTriggers = await this.analyzeSleepTriggers(
      painData,
      contextData.sleep
    );
    triggers.push(...sleepTriggers);

    // Analyze weather triggers
    const weatherTriggers = await this.analyzeWeatherTriggers(
      painData,
      contextData.weather
    );
    triggers.push(...weatherTriggers);

    // Analyze medication timing
    const medicationTriggers = await this.analyzeMedicationTiming(
      painData,
      contextData.medications
    );
    triggers.push(...medicationTriggers);

    // Rank by strength and confidence
    return this.rankTriggers(triggers);
  }

  async analyzeActivityTriggers(
    painData: PainDataPoint[],
    activityData: ActivityDataPoint[]
  ): Promise<TriggerAnalysis[]> {
    const triggers: TriggerAnalysis[] = [];

    // Group activities by type
    const activityTypes = this.groupByActivityType(activityData);

    for (const [activityType, activities] of activityTypes) {
      // Find pain levels before and after activity
      const painDeltas = activities.map(activity => {
        const painBefore = this.getPainNearTime(painData, activity.startTime, -60);
        const painAfter = this.getPainNearTime(painData, activity.endTime, 120);
        return painAfter - painBefore;
      });

      // Calculate statistics
      const avgDelta = this.mean(painDeltas);
      const pValue = this.tTest(painDeltas, 0);

      if (avgDelta > 0.5 && pValue < 0.05) {
        triggers.push({
          trigger: {
            category: 'activity',
            specific: activityType,
            description: `${activityType} appears to increase pain`
          },
          association: {
            correlation: avgDelta / 10,
            temporalLag: 2,
            doseResponse: this.checkDoseResponse(activities, painDeltas),
            confidence: 1 - pValue
          },
          evidence: {
            occurrences: activities.length,
            pain_increase_avg: avgDelta,
            last_occurrence: activities[activities.length - 1].startTime,
            examples: this.getTopExamples(activities, painDeltas, 3)
          }
        });
      }
    }

    return triggers;
  }
}

Real-Time Alerting

class AlertManager {
  private alertRules: AlertRule[];
  private notificationService: NotificationService;

  async evaluateAlerts(patientId: string, currentData: MonitoringData): Promise<void> {
    const baseline = await this.getPatientBaseline(patientId);
    const preferences = await this.getAlertPreferences(patientId);

    for (const rule of this.alertRules) {
      if (this.shouldTriggerAlert(rule, currentData, baseline)) {
        await this.sendAlert({
          patientId,
          rule,
          data: currentData,
          baseline,
          preferences
        });
      }
    }
  }

  private shouldTriggerAlert(
    rule: AlertRule,
    current: MonitoringData,
    baseline: PatientBaseline
  ): boolean {
    switch (rule.type) {
      case 'PAIN_SPIKE':
        return current.painLevel > baseline.averagePain + (2 * baseline.painStdDev);

      case 'MISSED_MEDICATION':
        return current.medicationAdherence < 0.8 && rule.adherenceWindow;

      case 'PATTERN_CHANGE':
        return this.detectPatternDeviation(current, baseline) > rule.threshold;

      case 'FLARE_PREDICTION':
        return current.flareRisk > rule.riskThreshold;

      default:
        return false;
    }
  }
}
