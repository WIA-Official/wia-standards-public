Physiological Markers for Pain

Research has identified several physiological markers that correlate with chronic pain states. The WIA-CHRONIC-PAIN architecture supports collection and analysis of these markers:

Marker Pain Relevance Data Source
Heart Rate Variability Lower HRV associated with chronic pain states Wearables, ECG devices
Sleep Quality Bidirectional relationship with pain intensity Sleep trackers, actigraphy
Activity Levels Reduced activity indicates pain flares Accelerometers, step counters
Electrodermal Activity Stress and pain response indicator Specialized sensors
Skin Temperature Inflammation and autonomic changes Temperature sensors

Offline-First Data Strategy

class OfflineSyncManager {
  private db: LocalDatabase;
  private syncQueue: SyncQueue;

  async logPain(entry: PainLogEntry): Promise<void> {
    // Always save locally first
    const localId = await this.db.savePainLog(entry);

    // Queue for sync
    await this.syncQueue.add({
      type: 'PAIN_LOG',
      localId,
      data: entry,
      timestamp: Date.now(),
      priority: 'high'
    });

    // Attempt immediate sync if online
    if (await this.isOnline()) {
      this.processSyncQueue();
    }
  }

  async processSyncQueue(): Promise<void> {
    const pending = await this.syncQueue.getPending();

    for (const item of pending) {
      try {
        const response = await this.api.sync(item);
        await this.db.updateServerId(item.localId, response.serverId);
        await this.syncQueue.markComplete(item.id);
      } catch (error) {
        if (this.isRetryable(error)) {
          await this.syncQueue.reschedule(item.id);
        } else {
          await this.syncQueue.markFailed(item.id, error);
        }
      }
    }
  }
}

Data Models

Core Entity Model

-- Patient entity
CREATE TABLE patients (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    external_id VARCHAR(255) UNIQUE,
    demographics JSONB ENCRYPTED,
    conditions TEXT[] ENCRYPTED,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Pain log entries
CREATE TABLE pain_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    patient_id UUID REFERENCES patients(id),
    timestamp TIMESTAMPTZ NOT NULL,
    intensity INTEGER CHECK (intensity >= 0 AND intensity <= 10),
    locations JSONB NOT NULL,
    qualities TEXT[],
    triggers TEXT[],
    context JSONB,
    source VARCHAR(50) DEFAULT 'patient_app',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Create hypertable for time-series optimization
SELECT create_hypertable('pain_logs', 'timestamp');

-- Medication records
CREATE TABLE medications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    patient_id UUID REFERENCES patients(id),
    name VARCHAR(255) NOT NULL,
    dosage VARCHAR(100),
    frequency JSONB,
    start_date DATE,
    end_date DATE,
    prescriber VARCHAR(255),
    status VARCHAR(50) DEFAULT 'active'
);

-- Medication doses
CREATE TABLE medication_doses (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    medication_id UUID REFERENCES medications(id),
    scheduled_time TIMESTAMPTZ,
    taken_time TIMESTAMPTZ,
    status VARCHAR(50),
    notes TEXT
);

-- Wearable data
CREATE TABLE wearable_data (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    patient_id UUID REFERENCES patients(id),
    timestamp TIMESTAMPTZ NOT NULL,
    metric_type VARCHAR(50) NOT NULL,
    value NUMERIC NOT NULL,
    unit VARCHAR(20),
    source JSONB
);

Event-Driven Architecture

Event Sourcing: Pain logs and medication events are stored as immutable events, enabling complete audit trails. 
interface PainLoggedEvent {
  eventType: 'PAIN_LOGGED';
  eventId: string;
  timestamp: string;
  patientId: string;
  payload: {
    intensity: number;
    locations: PainLocation[];
    qualities: string[];
    triggers: string[];
  };
  metadata: {
    source: string;
    version: string;
    correlationId: string;
  };
}
