Chapter 5: Security & Privacy

WIA Digital Memorial Standard - Data Protection & Access Control

5.1 Security Architecture

Security and privacy are paramount in digital memorial systems. This chapter covers comprehensive security measures including encryption, access control, GDPR compliance, and privacy-preserving technologies.

Security Layers

  • Infrastructure Security: Network firewalls, DDoS protection, intrusion detection
  • Application Security: Input validation, SQL injection prevention, XSS protection
  • Data Security: Encryption at rest and in transit, key management
  • Access Control: Role-based permissions, multi-factor authentication
  • Privacy Controls: Data minimization, consent management, right to deletion
  • Compliance: GDPR, CCPA, HIPAA alignment

5.5 Implementation Examples

Example 1: JWT Authentication

Node.js - JWT Auth Middleware
const jwt = require('jsonwebtoken');
const bcrypt = require('bcryptjs');
const speakeasy = require('speakeasy');

class AuthenticationService {
  constructor(secretKey) {
    this.secretKey = secretKey;
    this.tokenExpiry = '1h';
    this.refreshTokenExpiry = '7d';
  }

  async login(email, password, mfaCode = null) {
    // Verify credentials
    const user = await this.getUserByEmail(email);
    
    if (!user || !await bcrypt.compare(password, user.passwordHash)) {
      throw new Error('Invalid credentials');
    }

    // Check if MFA is required
    if (user.mfaEnabled) {
      if (!mfaCode) {
        return { requiresMfa: true, tempToken: this.generateTempToken(user.userId) };
      }

      const verified = speakeasy.totp.verify({
        secret: user.mfaSecret,
        encoding: 'base32',
        token: mfaCode,
        window: 2
      });

      if (!verified) {
        throw new Error('Invalid MFA code');
      }
    }

    // Generate tokens
    const accessToken = this.generateAccessToken(user);
    const refreshToken = this.generateRefreshToken(user);

    // Log successful login
    await this.logSecurityEvent('login', user.userId, 'success');

    return {
      accessToken,
      refreshToken,
      expiresIn: 3600,
      user: {
        userId: user.userId,
        email: user.email,
        name: user.name
      }
    };
  }

  generateAccessToken(user) {
    return jwt.sign(
      {
        userId: user.userId,
        email: user.email,
        type: 'access'
      },
      this.secretKey,
      { expiresIn: this.tokenExpiry }
    );
  }

  generateRefreshToken(user) {
    return jwt.sign(
      {
        userId: user.userId,
        type: 'refresh'
      },
      this.secretKey,
      { expiresIn: this.refreshTokenExpiry }
    );
  }

  verifyToken(token) {
    try {
      return jwt.verify(token, this.secretKey);
    } catch (error) {
      throw new Error('Invalid or expired token');
    }
  }

  async enableMfa(userId) {
    const secret = speakeasy.generateSecret({
      name: `WIA Memorial (${userId})`
    });

    await this.saveMfaSecret(userId, secret.base32);

    return {
      secret: secret.base32,
      qrCode: secret.otpauth_url
    };
  }

  async verifyMfa(userId, code) {
    const user = await this.getUserById(userId);
    
    return speakeasy.totp.verify({
      secret: user.mfaSecret,
      encoding: 'base32',
      token: code,
      window: 2
    });
  }
}

// Middleware
function authMiddleware(req, res, next) {
  const authHeader = req.headers.authorization;
  
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'No token provided' });
  }

  const token = authHeader.substring(7);

  try {
    const decoded = authService.verifyToken(token);
    req.user = decoded;
    next();
  } catch (error) {
    return res.status(401).json({ error: 'Invalid token' });
  }
}

module.exports = { AuthenticationService, authMiddleware };

Example 2: Access Control System

Python - Role-Based Access Control
from enum import Enum
from typing import List, Set
from dataclasses import dataclass
from datetime import datetime

class Permission(Enum):
    VIEW = "view"
    EDIT = "edit"
    DELETE = "delete"
    MODERATE = "moderate"
    MANAGE_ACCESS = "manage_access"
    ADD_TRIBUTE = "add_tribute"
    ADD_MEDIA = "add_media"

class Role(Enum):
    OWNER = "owner"
    ADMINISTRATOR = "administrator"
    CONTRIBUTOR = "contributor"
    VIEWER = "viewer"

@dataclass
class AccessGrant:
    user_id: str
    memorial_id: str
    role: Role
    permissions: Set[Permission]
    granted_at: datetime
    expires_at: datetime = None

class AccessControlService:
    def __init__(self):
        self.role_permissions = {
            Role.OWNER: {
                Permission.VIEW, Permission.EDIT, Permission.DELETE,
                Permission.MODERATE, Permission.MANAGE_ACCESS,
                Permission.ADD_TRIBUTE, Permission.ADD_MEDIA
            },
            Role.ADMINISTRATOR: {
                Permission.VIEW, Permission.EDIT, Permission.MODERATE,
                Permission.ADD_TRIBUTE, Permission.ADD_MEDIA
            },
            Role.CONTRIBUTOR: {
                Permission.VIEW, Permission.ADD_TRIBUTE, Permission.ADD_MEDIA
            },
            Role.VIEWER: {
                Permission.VIEW
            }
        }

    def check_permission(self, user_id: str, memorial_id: str, 
                        required_permission: Permission) -> bool:
        """Check if user has required permission"""
        grant = self.get_access_grant(user_id, memorial_id)
        
        if not grant:
            return False

        # Check if grant is expired
        if grant.expires_at and grant.expires_at < datetime.now():
            return False

        return required_permission in grant.permissions

    def grant_access(self, grantor_id: str, user_id: str, memorial_id: str,
                    role: Role, expires_at: datetime = None) -> AccessGrant:
        """Grant access to user"""
        
        # Verify grantor has MANAGE_ACCESS permission
        if not self.check_permission(grantor_id, memorial_id, Permission.MANAGE_ACCESS):
            raise PermissionError("Insufficient permissions to grant access")

        permissions = self.role_permissions[role]
        
        grant = AccessGrant(
            user_id=user_id,
            memorial_id=memorial_id,
            role=role,
            permissions=permissions,
            granted_at=datetime.now(),
            expires_at=expires_at
        )

        self.save_access_grant(grant)
        self.log_access_change(grantor_id, user_id, memorial_id, "grant", role)

        return grant

    def revoke_access(self, revoker_id: str, user_id: str, memorial_id: str):
        """Revoke user access"""
        
        if not self.check_permission(revoker_id, memorial_id, Permission.MANAGE_ACCESS):
            raise PermissionError("Insufficient permissions to revoke access")

        self.delete_access_grant(user_id, memorial_id)
        self.log_access_change(revoker_id, user_id, memorial_id, "revoke", None)

    def get_user_memorials(self, user_id: str) -> List[str]:
        """Get all memorials user has access to"""
        grants = self.get_user_grants(user_id)
        
        # Filter out expired grants
        active_grants = [
            g for g in grants
            if not g.expires_at or g.expires_at > datetime.now()
        ]
        
        return [g.memorial_id for g in active_grants]

    def get_memorial_users(self, memorial_id: str) -> List[AccessGrant]:
        """Get all users with access to memorial"""
        return self.get_memorial_grants(memorial_id)

# Usage
ac_service = AccessControlService()

# Grant access
grant = ac_service.grant_access(
    grantor_id="USR-OWNER123",
    user_id="USR-FAMILY456",
    memorial_id="MEM-12345678",
    role=Role.ADMINISTRATOR
)

# Check permission
can_edit = ac_service.check_permission(
    "USR-FAMILY456",
    "MEM-12345678",
    Permission.EDIT
)

print(f"Can edit: {can_edit}")

Example 3: Data Encryption

Python - AES-256 Encryption
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.backends import default_backend
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2
import os
import base64

class EncryptionService:
    def __init__(self, master_key: bytes):
        self.master_key = master_key
        self.backend = default_backend()

    def encrypt_data(self, plaintext: bytes, context: str = "") -> dict:
        """Encrypt data using AES-256-GCM"""
        
        # Generate random IV
        iv = os.urandom(12)
        
        # Derive encryption key
        key = self.derive_key(context)
        
        # Create cipher
        cipher = Cipher(
            algorithms.AES(key),
            modes.GCM(iv),
            backend=self.backend
        )
        encryptor = cipher.encryptor()
        
        # Encrypt
        ciphertext = encryptor.update(plaintext) + encryptor.finalize()
        
        return {
            'ciphertext': base64.b64encode(ciphertext).decode('utf-8'),
            'iv': base64.b64encode(iv).decode('utf-8'),
            'tag': base64.b64encode(encryptor.tag).decode('utf-8'),
            'algorithm': 'AES-256-GCM'
        }

    def decrypt_data(self, encrypted_data: dict, context: str = "") -> bytes:
        """Decrypt AES-256-GCM encrypted data"""
        
        # Decode base64
        ciphertext = base64.b64decode(encrypted_data['ciphertext'])
        iv = base64.b64decode(encrypted_data['iv'])
        tag = base64.b64decode(encrypted_data['tag'])
        
        # Derive decryption key
        key = self.derive_key(context)
        
        # Create cipher
        cipher = Cipher(
            algorithms.AES(key),
            modes.GCM(iv, tag),
            backend=self.backend
        )
        decryptor = cipher.decryptor()
        
        # Decrypt
        plaintext = decryptor.update(ciphertext) + decryptor.finalize()
        
        return plaintext

    def derive_key(self, context: str) -> bytes:
        """Derive encryption key from master key"""
        
        kdf = PBKDF2(
            algorithm=hashes.SHA256(),
            length=32,
            salt=context.encode('utf-8'),
            iterations=100000,
            backend=self.backend
        )
        
        return kdf.derive(self.master_key)

    def rotate_key(self, old_context: str, new_context: str, 
                   encrypted_data: dict) -> dict:
        """Rotate encryption key"""
        
        # Decrypt with old key
        plaintext = self.decrypt_data(encrypted_data, old_context)
        
        # Re-encrypt with new key
        return self.encrypt_data(plaintext, new_context)

# Usage
master_key = os.urandom(32)
encryption_service = EncryptionService(master_key)

# Encrypt sensitive data
sensitive_data = b"Personal information about the deceased"
encrypted = encryption_service.encrypt_data(
    sensitive_data,
    context="MEM-12345678"
)

print("Encrypted:", encrypted)

# Decrypt
decrypted = encryption_service.decrypt_data(
    encrypted,
    context="MEM-12345678"
)

print("Decrypted:", decrypted.decode('utf-8'))

5.6 Certification Requirements

  • Implement AES-256 encryption for data at rest
  • Use TLS 1.3 for all data in transit
  • Require multi-factor authentication for administrators
  • Implement role-based access control (RBAC)
  • Support OAuth 2.0 and SAML 2.0 authentication
  • Maintain comprehensive audit logs for minimum 2 years
  • Implement automated vulnerability scanning
  • Provide GDPR compliance features (right to erasure, data portability)
  • Support granular privacy controls (public, protected, private)
  • Implement rate limiting on all API endpoints
  • Use secure password hashing (bcrypt, Argon2)
  • Implement session timeout and automatic logout
  • Support encrypted backups
  • Provide security headers (CSP, HSTS, X-Frame-Options)
  • Implement intrusion detection system
  • Support security event alerting
  • Provide data breach notification procedures
  • Implement secure key management (HSM or cloud KMS)
  • Support automated key rotation
  • Provide penetration testing reports (annual)
  • Implement DDoS protection
  • Support IP whitelisting/blacklisting
  • Provide consent management system
  • Implement data minimization practices
  • Support secure file upload with malware scanning
