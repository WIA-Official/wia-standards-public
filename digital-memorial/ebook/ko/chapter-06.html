Chapter 6: API Integration

WIA Digital Memorial Standard - RESTful APIs & Integration

6.1 API Architecture Overview

The WIA Memorial API provides comprehensive RESTful endpoints for memorial management, supporting OAuth 2.0 authentication, webhooks, rate limiting, and versioning for backward compatibility.

API Design Principles

  • RESTful architecture following industry standards
  • Consistent naming conventions and URL patterns
  • Comprehensive error handling with meaningful messages
  • Rate limiting to prevent abuse
  • Versioning for backward compatibility
  • Pagination for large result sets
  • Filtering, sorting, and searching capabilities
  • Webhook support for real-time notifications

Base URL Structure

https://api.wia-memorial.org/v1/{resource}

6.5 SDK & Integration Examples

Example 1: JavaScript SDK

wia-memorial-sdk.js
class WiaMemorialSDK {
  constructor(apiKey, options = {}) {
    this.apiKey = apiKey;
    this.baseUrl = options.baseUrl || 'https://api.wia-memorial.org/v1';
    this.timeout = options.timeout || 30000;
  }

  async request(endpoint, options = {}) {
    const url = `${this.baseUrl}${endpoint}`;
    const headers = {
      'Authorization': `Bearer ${this.apiKey}`,
      'Content-Type': 'application/json',
      'X-API-Version': '1.0',
      ...options.headers
    };

    try {
      const response = await fetch(url, {
        method: options.method || 'GET',
        headers,
        body: options.body ? JSON.stringify(options.body) : undefined,
        timeout: this.timeout
      });

      const data = await response.json();

      if (!response.ok) {
        throw new Error(data.error?.message || 'API request failed');
      }

      return data;
    } catch (error) {
      throw new Error(`WIA Memorial API Error: ${error.message}`);
    }
  }

  // Memorial methods
  async createMemorial(memorial) {
    return this.request('/memorials', {
      method: 'POST',
      body: memorial
    });
  }

  async getMemorial(id) {
    return this.request(`/memorials/${id}`);
  }

  async updateMemorial(id, updates) {
    return this.request(`/memorials/${id}`, {
      method: 'PUT',
      body: updates
    });
  }

  async deleteMemorial(id) {
    return this.request(`/memorials/${id}`, {
      method: 'DELETE'
    });
  }

  async listMemorials(params = {}) {
    const query = new URLSearchParams(params).toString();
    return this.request(`/memorials?${query}`);
  }

  // Tribute methods
  async submitTribute(memorialId, tribute) {
    return this.request('/tributes', {
      method: 'POST',
      body: { memorialId, ...tribute }
    });
  }

  async listTributes(memorialId, params = {}) {
    const query = new URLSearchParams(params).toString();
    return this.request(`/memorials/${memorialId}/tributes?${query}`);
  }

  // Media methods
  async uploadMedia(memorialId, file) {
    const formData = new FormData();
    formData.append('file', file);
    formData.append('memorialId', memorialId);

    return this.request('/media/upload', {
      method: 'POST',
      headers: { 'Content-Type': 'multipart/form-data' },
      body: formData
    });
  }

  // Webhook methods
  async createWebhook(webhook) {
    return this.request('/webhooks', {
      method: 'POST',
      body: webhook
    });
  }

  async listWebhooks() {
    return this.request('/webhooks');
  }
}

// Usage
const sdk = new WiaMemorialSDK('your-api-key-here');

// Create memorial
const memorial = await sdk.createMemorial({
  firstName: 'John',
  lastName: 'Doe',
  dateOfBirth: '1950-03-15',
  dateOfDeath: '2024-01-10',
  biography: 'A wonderful person...',
  templateId: 'TPL-CLASSIC01'
});

console.log('Created memorial:', memorial.data.memorialId);

Example 2: Python SDK

wia_memorial_sdk.py
import requests
from typing import Dict, List, Optional
import json

class WiaMemorialSDK:
    def __init__(self, api_key: str, base_url: str = 'https://api.wia-memorial.org/v1'):
        self.api_key = api_key
        self.base_url = base_url
        self.session = requests.Session()
        self.session.headers.update({
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json',
            'X-API-Version': '1.0'
        })

    def _request(self, method: str, endpoint: str, **kwargs) -> Dict:
        url = f'{self.base_url}{endpoint}'
        response = self.session.request(method, url, **kwargs)
        response.raise_for_status()
        return response.json()

    # Memorial methods
    def create_memorial(self, memorial: Dict) -> Dict:
        return self._request('POST', '/memorials', json=memorial)

    def get_memorial(self, memorial_id: str) -> Dict:
        return self._request('GET', f'/memorials/{memorial_id}')

    def update_memorial(self, memorial_id: str, updates: Dict) -> Dict:
        return self._request('PUT', f'/memorials/{memorial_id}', json=updates)

    def delete_memorial(self, memorial_id: str) -> Dict:
        return self._request('DELETE', f'/memorials/{memorial_id}')

    def list_memorials(self, **params) -> Dict:
        return self._request('GET', '/memorials', params=params)

    # Tribute methods
    def submit_tribute(self, memorial_id: str, tribute: Dict) -> Dict:
        data = {'memorialId': memorial_id, **tribute}
        return self._request('POST', '/tributes', json=data)

    def list_tributes(self, memorial_id: str, **params) -> Dict:
        return self._request('GET', f'/memorials/{memorial_id}/tributes', params=params)

    # Media methods
    def upload_media(self, memorial_id: str, file_path: str) -> Dict:
        with open(file_path, 'rb') as f:
            files = {'file': f}
            data = {'memorialId': memorial_id}
            return self._request('POST', '/media/upload', files=files, data=data)

    # Webhook methods
    def create_webhook(self, webhook: Dict) -> Dict:
        return self._request('POST', '/webhooks', json=webhook)

    def list_webhooks(self) -> Dict:
        return self._request('GET', '/webhooks')

# Usage

# Create memorial

    'firstName': 'John',

    'dateOfBirth': '1950-03-15',

    'biography': 'A wonderful person...',
    'templateId': 'TPL-CLASSIC01'
})

print(f"Created memorial: {memorial['data']['memorialId']}")

Example 3: Webhook Handler

webhook-handler.js
const express = require('express');
const crypto = require('crypto');

const app = express();
app.use(express.json());

// Webhook verification
function verifyWebhookSignature(payload, signature, secret) {
  const hmac = crypto.createHmac('sha256', secret);
  hmac.update(JSON.stringify(payload));
  const computedSignature = hmac.digest('hex');
  return crypto.timingSafeEqual(
    Buffer.from(signature),
    Buffer.from(computedSignature)
  );
}

// Webhook endpoint
app.post('/webhooks/wia-memorial', (req, res) => {
  const signature = req.headers['x-wia-signature'];
  const webhookSecret = process.env.WEBHOOK_SECRET;

  // Verify signature
  if (!verifyWebhookSignature(req.body, signature, webhookSecret)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const { event, data } = req.body;

  // Handle different events
  switch (event) {
    case 'memorial.created':
      console.log('New memorial created:', data.memorialId);
      // Send welcome email, setup analytics, etc.
      break;

    case 'tribute.submitted':
      console.log('New tribute submitted:', data.tributeId);
      // Notify memorial owner, moderate content, etc.
      break;

    case 'media.uploaded':
      console.log('Media uploaded:', data.mediaId);
      // Process media, generate thumbnails, etc.
      break;

    default:
      console.log('Unknown event:', event);
  }

  // Acknowledge receipt
  res.status(200).json({ received: true });
});

app.listen(3000, () => {
  console.log('Webhook server listening on port 3000');
});
