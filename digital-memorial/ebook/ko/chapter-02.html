Chapter 2: Photo/Video Archives

WIA Digital Memorial Standard - Media Management & Storage

2.1 Media Archive Architecture

Photo and video archives form the visual foundation of digital memorials, preserving precious moments and memories in perpetuity. This chapter covers the technical infrastructure, storage strategies, and best practices for implementing robust media archive systems.

A comprehensive media archive system must handle high-resolution images, video streaming, metadata management, and long-term preservation while ensuring fast access and optimal user experience.

Core Components

  • Distributed storage infrastructure with redundancy
  • Image processing pipeline for optimization
  • Video transcoding and adaptive streaming
  • Content delivery network (CDN) integration
  • Metadata extraction and indexing
  • Backup and disaster recovery systems
  • Access control and permissions management
  • Media organization and tagging capabilities

Node.js - Media Upload Handler
const multer = require('multer');
const sharp = require('sharp');
const AWS = require('aws-sdk');
const ffmpeg = require('fluent-ffmpeg');

// Configure AWS S3
const s3 = new AWS.S3({
  accessKeyId: process.env.AWS_ACCESS_KEY,
  secretAccessKey: process.env.AWS_SECRET_KEY,
  region: 'us-east-1'
});

// Configure multer for file upload
const storage = multer.memoryStorage();
const upload = multer({
  storage: storage,
  limits: { fileSize: 100 * 1024 * 1024 }, // 100MB
  fileFilter: (req, file, cb) => {
    const allowedTypes = /jpeg|jpg|png|gif|mp4|mov|avi/;
    const mimetype = allowedTypes.test(file.mimetype);
    if (mimetype) {
      cb(null, true);
    } else {
      cb(new Error('Invalid file type'));
    }
  }
});

class MediaUploadService {
  async uploadMedia(file, memorialId, userId) {
    const mediaId = `MED-${this.generateId(12)}`;
    const isVideo = file.mimetype.startsWith('video/');

    try {
      // Upload original file to S3
      const originalKey = `memorials/${memorialId}/original/${mediaId}`;
      await this.uploadToS3(file.buffer, originalKey, file.mimetype);

      // Process based on media type
      if (isVideo) {
        await this.processVideo(file, mediaId, memorialId);
      } else {
        await this.processImage(file, mediaId, memorialId);
      }

      // Create database record
      const mediaRecord = await this.createMediaRecord({
        mediaId,
        memorialId,
        userId,
        type: isVideo ? 'video' : 'photo',
        metadata: {
          filename: mediaId,
          originalFilename: file.originalname,
          mimeType: file.mimetype,
          fileSize: file.size,
          uploadDate: new Date().toISOString()
        },
        storage: {
          tier: 'hot',
          locations: [
            {
              provider: 'aws-s3',
              region: 'us-east-1',
              bucket: process.env.S3_BUCKET,
              path: originalKey,
              url: `https://${process.env.S3_BUCKET}.s3.amazonaws.com/${originalKey}`
            }
          ]
        }
      });

      return mediaRecord;

    } catch (error) {
      console.error('Error uploading media:', error);
      throw error;
    }
  }

  async processImage(file, mediaId, memorialId) {
    const variants = [
      { name: 'thumbnail', width: 200, height: 200 },
      { name: 'small', width: 640, height: 480 },
      { name: 'medium', width: 1280, height: 960 },
      { name: 'large', width: 1920, height: 1440 }
    ];

    const processedVariants = [];

    for (const variant of variants) {
      const processed = await sharp(file.buffer)
        .resize(variant.width, variant.height, {
          fit: 'inside',
          withoutEnlargement: true
        })
        .jpeg({ quality: 85 })
        .toBuffer();

      const key = `memorials/${memorialId}/${variant.name}/${mediaId}.jpg`;
      await this.uploadToS3(processed, key, 'image/jpeg');

      processedVariants.push({
        name: variant.name,
        dimensions: { width: variant.width, height: variant.height },
        fileSize: processed.length,
        url: `https://${process.env.CDN_DOMAIN}/${key}`
      });
    }

    await this.updateMediaVariants(mediaId, processedVariants);
  }

  async processVideo(file, mediaId, memorialId) {
    const outputPath = `/tmp/${mediaId}`;

    return new Promise((resolve, reject) => {
      ffmpeg(file.buffer)
        .output(`${outputPath}_720p.mp4`)
        .videoCodec('libx264')
        .size('1280x720')
        .videoBitrate('2500k')
        .audioBitrate('128k')
        .on('end', async () => {
          // Upload transcoded video to S3
          const transcodedKey = `memorials/${memorialId}/video/${mediaId}_720p.mp4`;
          await this.uploadFileToS3(`${outputPath}_720p.mp4`, transcodedKey);

          // Generate thumbnail
          await this.generateVideoThumbnail(file, mediaId, memorialId);

          resolve();
        })
        .on('error', reject)
        .run();
    });
  }

  async uploadToS3(buffer, key, contentType) {
    return s3.upload({
      Bucket: process.env.S3_BUCKET,
      Key: key,
      Body: buffer,
      ContentType: contentType,
      ACL: 'private'
    }).promise();
  }

  generateId(length) {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
    let result = '';
    for (let i = 0; i < length; i++) {
      result += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    return result;
  }
}

module.exports = new MediaUploadService();

React - Photo Gallery with Lightbox
import React, { useState, useEffect } from 'react';
import axios from 'axios';
import Lightbox from 'react-image-lightbox';
import 'react-image-lightbox/style.css';

interface MediaItem {
  mediaId: string;
  type: 'photo' | 'video';
  urls: {
    thumbnail: string;
    medium: string;
    original: string;
  };
  metadata: {
    caption?: string;
    captureDate?: string;
  };
}

const PhotoGallery: React.FC<{ memorialId: string }> = ({ memorialId }) => {
  const [media, setMedia] = useState<MediaItem[]>([]);
  const [loading, setLoading] = useState(true);
  const [lightboxOpen, setLightboxOpen] = useState(false);
  const [photoIndex, setPhotoIndex] = useState(0);
  const [filter, setFilter] = useState('all');

  useEffect(() => {
    loadMedia();
  }, [memorialId]);

  const loadMedia = async () => {
    try {
      const response = await axios.get(
        `https://api.wia-memorial.org/v1/memorials/${memorialId}/media`,
        {
          headers: {
            'Authorization': `Bearer ${localStorage.getItem('token')}`
          }
        }
      );
      setMedia(response.data.media);
    } catch (error) {
      console.error('Error loading media:', error);
    } finally {
      setLoading(false);
    }
  };

  const filteredMedia = media.filter(item =>
    filter === 'all' || item.type === filter
  );

  const images = filteredMedia
    .filter(item => item.type === 'photo')
    .map(item => item.urls.original);

  if (loading) {
    return <div className="loading">Loading gallery...</div>;
  }

  return (
    <div className="photo-gallery">
      <div className="gallery-controls">
        <button
          className={filter === 'all' ? 'active' : ''}
          onClick={() => setFilter('all')}
        >
          All Media
        </button>
        <button
          className={filter === 'photo' ? 'active' : ''}
          onClick={() => setFilter('photo')}
        >
          Photos
        </button>
        <button
          className={filter === 'video' ? 'active' : ''}
          onClick={() => setFilter('video')}
        >
          Videos
        </button>
      </div>

      <div className="gallery-grid">
        {filteredMedia.map((item, index) => (
          <div
            key={item.mediaId}
            className="gallery-item"
            onClick={() => {
              if (item.type === 'photo') {
                setPhotoIndex(index);
                setLightboxOpen(true);
              }
            }}
          >
            <img
              src={item.urls.thumbnail}
              alt={item.metadata.caption || 'Memorial photo'}
              loading="lazy"
            />
            {item.type === 'video' && (
              <div className="video-overlay">
                <span className="play-icon">▶</span>
              </div>
            )}
            {item.metadata.caption && (
              <div className="caption">{item.metadata.caption}</div>
            )}
          </div>
        ))}
      </div>

      {lightboxOpen && (
        <Lightbox
          mainSrc={images[photoIndex]}
          nextSrc={images[(photoIndex + 1) % images.length]}
          prevSrc={images[(photoIndex + images.length - 1) % images.length]}
          onCloseRequest={() => setLightboxOpen(false)}
          onMovePrevRequest={() => setPhotoIndex((photoIndex + images.length - 1) % images.length)}
          onMoveNextRequest={() => setPhotoIndex((photoIndex + 1) % images.length)}
        />
      )}
    </div>
  );
};

export default PhotoGallery;

Example 3: Video Streaming Implementation

Python - HLS Video Streaming
import os
import subprocess
from flask import Flask, Response, request, jsonify
from werkzeug.utils import secure_filename
import boto3
from botocore.exceptions import ClientError

app = Flask(__name__)
s3_client = boto3.client('s3')

class VideoStreamingService:
    def __init__(self, bucket_name):
        self.bucket_name = bucket_name
        self.allowed_extensions = {'mp4', 'avi', 'mov', 'mkv'}

    def transcode_to_hls(self, input_file, output_dir, media_id):
        """Transcode video to HLS format with multiple quality levels"""

        os.makedirs(output_dir, exist_ok=True)

        # Generate HLS playlist with multiple bitrates
        command = [
            'ffmpeg',
            '-i', input_file,
            '-profile:v', 'baseline',
            '-level', '3.0',
            '-start_number', '0',
            '-hls_time', '10',
            '-hls_list_size', '0',
            '-f', 'hls',

            # 1080p
            '-s', '1920x1080',
            '-b:v', '5000k',
            '-b:a', '192k',
            f'{output_dir}/1080p.m3u8',

            # 720p
            '-s', '1280x720',
            '-b:v', '2800k',
            '-b:a', '128k',
            f'{output_dir}/720p.m3u8',

            # 480p
            '-s', '854x480',
            '-b:v', '1400k',
            '-b:a', '96k',
            f'{output_dir}/480p.m3u8',

            # 360p
            '-s', '640x360',
            '-b:v', '800k',
            '-b:a', '64k',
            f'{output_dir}/360p.m3u8'
        ]

        try:
            subprocess.run(command, check=True, capture_output=True)
            self.create_master_playlist(output_dir)
            return True
        except subprocess.CalledProcessError as e:
            print(f"Error transcoding video: {e}")
            return False

    def create_master_playlist(self, output_dir):
        """Create HLS master playlist"""

        master_playlist = """#EXTM3U
#EXT-X-VERSION:3
#EXT-X-STREAM-INF:BANDWIDTH=5000000,RESOLUTION=1920x1080
1080p.m3u8
#EXT-X-STREAM-INF:BANDWIDTH=2800000,RESOLUTION=1280x720
720p.m3u8
#EXT-X-STREAM-INF:BANDWIDTH=1400000,RESOLUTION=854x480
480p.m3u8
#EXT-X-STREAM-INF:BANDWIDTH=800000,RESOLUTION=640x360
360p.m3u8
"""

        with open(f'{output_dir}/master.m3u8', 'w') as f:
            f.write(master_playlist)

    def upload_hls_to_s3(self, local_dir, s3_prefix):
        """Upload HLS files to S3"""

        for root, dirs, files in os.walk(local_dir):
            for file in files:
                local_path = os.path.join(root, file)
                relative_path = os.path.relpath(local_path, local_dir)
                s3_path = f'{s3_prefix}/{relative_path}'

                try:
                    s3_client.upload_file(
                        local_path,
                        self.bucket_name,
                        s3_path,
                        ExtraArgs={
                            'ContentType': self.get_content_type(file),
                            'CacheControl': 'max-age=31536000'
                        }
                    )
                except ClientError as e:
                    print(f"Error uploading {file}: {e}")
                    raise

    def get_content_type(self, filename):
        """Determine content type based on file extension"""
        if filename.endswith('.m3u8'):
            return 'application/x-mpegURL'
        elif filename.endswith('.ts'):
            return 'video/MP2T'
        else:
            return 'application/octet-stream'

    def get_streaming_url(self, media_id, memorial_id):
        """Generate CDN URL for HLS streaming"""

        cdn_domain = os.getenv('CDN_DOMAIN')
        return f'https://{cdn_domain}/memorials/{memorial_id}/stream/{media_id}/master.m3u8'

# Example usage
streaming_service = VideoStreamingService('wia-memorial-media')

@app.route('/api/v1/media/upload-video', methods=['POST'])
def upload_video():
    if 'video' not in request.files:
        return jsonify({'error': 'No video file provided'}), 400

    video_file = request.files['video']
    memorial_id = request.form.get('memorialId')
    media_id = f"MED-{generate_id(12)}"

    # Save uploaded file temporarily
    temp_path = f'/tmp/{media_id}_{secure_filename(video_file.filename)}'
    video_file.save(temp_path)

    # Transcode to HLS
    output_dir = f'/tmp/hls_{media_id}'
    success = streaming_service.transcode_to_hls(temp_path, output_dir, media_id)

    if success:
        # Upload to S3
        s3_prefix = f'memorials/{memorial_id}/stream/{media_id}'
        streaming_service.upload_hls_to_s3(output_dir, s3_prefix)

        # Generate streaming URL
        streaming_url = streaming_service.get_streaming_url(media_id, memorial_id)

        return jsonify({
            'success': True,
            'mediaId': media_id,
            'streamingUrl': streaming_url
        }), 200
    else:
        return jsonify({'error': 'Video transcoding failed'}), 500

def generate_id(length):
    import random
    import string
    return ''.join(random.choices(string.ascii_uppercase + string.digits, k=length))

2.6 Certification Requirements

To achieve WIA Digital Memorial Standard certification for photo/video archive systems:

  • Support minimum 5 image formats (JPEG, PNG, WebP, HEIC, TIFF)
  • Support minimum 4 video formats (MP4, MOV, WebM, AVI)
  • Implement automatic image optimization and resizing
  • Generate minimum 4 image variants (thumbnail, small, medium, large)
  • Support video transcoding to multiple resolutions
  • Implement adaptive bitrate streaming (HLS or DASH)
  • Provide CDN integration for media delivery
  • Implement redundant storage across multiple locations
  • Support EXIF metadata extraction and preservation
  • Provide facial recognition for photo tagging (optional)
  • Implement album/collection organization system
  • Support batch upload (minimum 50 files simultaneously)
  • Provide progress indicators for uploads and processing
  • Implement automatic backup to cold storage tier
  • Support media download in original quality
  • Provide watermarking capabilities for protection
  • Implement access control and permission management
  • Support media sharing via links with expiration
  • Provide video thumbnail generation
  • Implement lazy loading for gallery performance
  • Support 360-degree photo viewing
  • Provide slideshow functionality
  • Implement media analytics (views, downloads)
  • Support bulk tagging and metadata editing
  • Provide API rate limiting to prevent abuse
