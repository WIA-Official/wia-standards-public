Chapter 4: Memory Preservation

WIA Digital Memorial Standard - Long-term Digital Archival

4.1 Long-term Digital Preservation

Memory preservation ensures that digital memorials remain accessible and intact for generations. This chapter covers archival strategies, data migration, format obsolescence prevention, and perpetual access planning.

Preservation Challenges

  • Technology obsolescence (formats, storage media)
  • Data degradation and bit rot
  • Platform dependency and vendor lock-in
  • Storage cost management over decades
  • Legal and ownership transitions
  • Maintaining metadata integrity
  • Ensuring backward compatibility
  • Climate and disaster resilience

Preservation Principles

  • Format Migration: Regular conversion to current standards
  • Redundancy: Multiple copies across geographic locations
  • Checksum Verification: Data integrity monitoring
  • Open Standards: Use widely-adopted, non-proprietary formats
  • Documentation: Comprehensive metadata and provenance
  • Emulation: Preserve viewing experiences across platforms
  • Sustainability Planning: Financial models for perpetual care

Example 1: Automated Integrity Verification

Python - Checksum Verification System
import hashlib
import os
import json
import boto3
from datetime import datetime
from typing import Dict, List, Tuple

class IntegrityVerificationSystem:
    def __init__(self, storage_config):
        self.config = storage_config
        self.s3_client = boto3.client('s3')
        self.verification_log = []

    def compute_checksum(self, file_path: str, algorithm: str = 'sha256') -> str:
        """Compute file checksum"""
        hash_func = getattr(hashlib, algorithm)()
        
        with open(file_path, 'rb') as f:
            while chunk := f.read(8192):
                hash_func.update(chunk)
        
        return hash_func.hexdigest()

    def verify_file_integrity(self, file_path: str, expected_checksum: str, 
                             algorithm: str = 'sha256') -> Tuple[bool, str]:
        """Verify single file integrity"""
        try:
            actual_checksum = self.compute_checksum(file_path, algorithm)
            
            if actual_checksum == expected_checksum:
                return True, f"Integrity verified: {actual_checksum}"
            else:
                return False, f"Checksum mismatch! Expected: {expected_checksum}, Got: {actual_checksum}"
        
        except Exception as e:
            return False, f"Verification error: {str(e)}"

    def verify_memorial_archive(self, memorial_id: str, manifest: Dict) -> Dict:
        """Verify entire memorial archive integrity"""
        results = {
            'memorial_id': memorial_id,
            'verification_time': datetime.utcnow().isoformat(),
            'total_files': len(manifest['files']),
            'verified': 0,
            'failed': 0,
            'errors': []
        }

        for file_entry in manifest['files']:
            file_path = file_entry['path']
            expected_checksum = file_entry['checksum']
            algorithm = file_entry.get('algorithm', 'sha256')

            success, message = self.verify_file_integrity(
                file_path, expected_checksum, algorithm
            )

            if success:
                results['verified'] += 1
            else:
                results['failed'] += 1
                results['errors'].append({
                    'file': file_path,
                    'message': message
                })

            self.verification_log.append({
                'file': file_path,
                'timestamp': datetime.utcnow().isoformat(),
                'success': success,
                'message': message
            })

        results['success_rate'] = results['verified'] / results['total_files'] * 100

        return results

    def repair_corrupted_files(self, memorial_id: str, failed_files: List[str]):
        """Attempt to repair corrupted files from backup"""
        repaired = []
        
        for file_path in failed_files:
            try:
                # Retrieve from secondary backup
                backup_key = f"backups/{memorial_id}/{os.path.basename(file_path)}"
                
                self.s3_client.download_file(
                    self.config['backup_bucket'],
                    backup_key,
                    file_path
                )
                
                repaired.append(file_path)
                print(f"Repaired: {file_path}")
                
            except Exception as e:
                print(f"Failed to repair {file_path}: {e}")
        
        return repaired

    def schedule_verification(self, memorial_id: str, frequency: str = 'monthly'):
        """Schedule periodic integrity checks"""
        verification_schedule = {
            'daily': '0 2 * * *',      # 2 AM daily
            'weekly': '0 2 * * 0',      # 2 AM Sunday
            'monthly': '0 2 1 * *',     # 2 AM 1st of month
            'quarterly': '0 2 1 */3 *'  # 2 AM 1st of quarter
        }

        cron_expression = verification_schedule.get(frequency, '0 2 1 * *')
        
        # Register with scheduler (e.g., AWS EventBridge, cron)
        print(f"Scheduled verification for {memorial_id}: {cron_expression}")
        
        return {
            'memorial_id': memorial_id,
            'frequency': frequency,
            'cron_expression': cron_expression,
            'next_run': self.calculate_next_run(cron_expression)
        }

    def calculate_next_run(self, cron_expression: str) -> str:
        # Implementation would calculate next execution time
        return datetime.utcnow().isoformat()

    def generate_integrity_report(self, memorial_id: str) -> Dict:
        """Generate comprehensive integrity report"""
        return {
            'memorial_id': memorial_id,
            'report_date': datetime.utcnow().isoformat(),
            'total_verifications': len(self.verification_log),
            'successful': sum(1 for v in self.verification_log if v['success']),
            'failed': sum(1 for v in self.verification_log if not v['success']),
            'recent_checks': self.verification_log[-10:],
            'recommendation': self.get_health_recommendation()
        }

    def get_health_recommendation(self) -> str:
        """Provide health recommendation based on verification history"""
        if not self.verification_log:
            return "No verification history available"
        
        failure_rate = sum(1 for v in self.verification_log if not v['success']) / len(self.verification_log)
        
        if failure_rate == 0:
            return "Excellent: All integrity checks passing"
        elif failure_rate < 0.01:
            return "Good: Minor issues detected, monitoring recommended"
        elif failure_rate < 0.05:
            return "Fair: Action recommended to address recurring failures"
        else:
            return "Critical: Immediate action required to prevent data loss"

Example 2: Format Migration System

Node.js - Automated Format Migration
const fs = require('fs').promises;
const path = require('path');
const sharp = require('sharp');
const ffmpeg = require('fluent-ffmpeg');

class FormatMigrationService {
  constructor() {
    this.supportedMigrations = {
      images: {
        'jpg': ['webp', 'avif', 'png'],
        'png': ['webp', 'avif', 'jpg'],
        'gif': ['webp', 'mp4']
      },
      videos: {
        'avi': ['mp4', 'webm'],
        'mov': ['mp4', 'webm'],
        'wmv': ['mp4', 'webm']
      }
    };
  }

  async planMigration(memorialId, currentFormats) {
    const plan = {
      memorialId,
      migrations: [],
      estimatedTime: 0,
      estimatedCost: 0
    };

    for (const format of currentFormats) {
      if (this.isObsolete(format)) {
        const targetFormat = this.getPreferredFormat(format.type);
        plan.migrations.push({
          sourceFormat: format.format,
          targetFormat: targetFormat,
          fileCount: format.count,
          priority: this.getMigrationPriority(format.format)
        });
      }
    }

    return plan;
  }

  async migrateImageFormat(sourcePath, targetFormat, quality = 90) {
    const sourceExt = path.extname(sourcePath).slice(1);
    const targetPath = sourcePath.replace(`.${sourceExt}`, `.${targetFormat}`);

    try {
      let pipeline = sharp(sourcePath);

      switch (targetFormat) {
        case 'webp':
          await pipeline.webp({ quality }).toFile(targetPath);
          break;
        case 'avif':
          await pipeline.avif({ quality }).toFile(targetPath);
          break;
        case 'png':
          await pipeline.png({ compressionLevel: 9 }).toFile(targetPath);
          break;
        case 'jpg':
          await pipeline.jpeg({ quality }).toFile(targetPath);
          break;
        default:
          throw new Error(`Unsupported target format: ${targetFormat}`);
      }

      // Verify migration
      const sourceSize = (await fs.stat(sourcePath)).size;
      const targetSize = (await fs.stat(targetPath)).size;

      return {
        success: true,
        sourcePath,
        targetPath,
        sourceSize,
        targetSize,
        compressionRatio: ((1 - targetSize / sourceSize) * 100).toFixed(2) + '%'
      };

    } catch (error) {
      return {
        success: false,
        error: error.message
      };
    }
  }

  async migrateVideoFormat(sourcePath, targetFormat) {
    const sourceExt = path.extname(sourcePath).slice(1);
    const targetPath = sourcePath.replace(`.${sourceExt}`, `.${targetFormat}`);

    return new Promise((resolve, reject) => {
      ffmpeg(sourcePath)
        .output(targetPath)
        .videoCodec(this.getVideoCodec(targetFormat))
        .audioCodec('aac')
        .on('end', () => {
          resolve({
            success: true,
            sourcePath,
            targetPath
          });
        })
        .on('error', (error) => {
          reject({
            success: false,
            error: error.message
          });
        })
        .run();
    });
  }

  async batchMigrate(files, targetFormat) {
    const results = [];

    for (const file of files) {
      const isImage = /\.(jpg|jpeg|png|gif|webp)$/i.test(file);
      const isVideo = /\.(mp4|avi|mov|wmv)$/i.test(file);

      try {
        let result;
        if (isImage) {
          result = await this.migrateImageFormat(file, targetFormat);
        } else if (isVideo) {
          result = await this.migrateVideoFormat(file, targetFormat);
        }

        results.push(result);
      } catch (error) {
        results.push({
          success: false,
          file,
          error: error.message
        });
      }
    }

    return {
      total: files.length,
      successful: results.filter(r => r.success).length,
      failed: results.filter(r => !r.success).length,
      results
    };
  }

  isObsolete(format) {
    const obsoleteFormats = ['bmp', 'tiff', 'avi', 'wmv', 'flv'];
    return obsoleteFormats.includes(format.format.toLowerCase());
  }

  getPreferredFormat(type) {
    const preferred = {
      image: 'webp',
      video: 'mp4',
      audio: 'aac'
    };
    return preferred[type] || 'unknown';
  }

  getMigrationPriority(format) {
    const priorities = {
      'bmp': 'high',
      'tiff': 'high',
      'avi': 'high',
      'wmv': 'high',
      'jpg': 'medium',
      'png': 'low'
    };
    return priorities[format.toLowerCase()] || 'low';
  }

  getVideoCodec(format) {
    const codecs = {
      'mp4': 'libx264',
      'webm': 'libvpx-vp9',
      'mkv': 'libx264'
    };
    return codecs[format] || 'libx264';
  }
}

// Usage

  '/memorials/MEM-12345678/photos/old_photo.bmp',
  '/memorials/MEM-12345678/videos/old_video.avi'
];

  .then(results => console.log(results))
  .catch(error => console.error(error));

Solidity - Memorial Provenance Smart Contract
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MemorialProvenance {
    struct Memorial {
        string memorialId;
        address creator;
        uint256 createdAt;
        string contentHash;
        bool isActive;
    }

    struct ProvenanceEvent {
        string eventType;
        address actor;
        uint256 timestamp;
        string description;
        string dataHash;
    }

    mapping(string => Memorial) public memorials;
    mapping(string => ProvenanceEvent[]) public provenanceChain;
    mapping(string => address[]) public authorizedActors;

    event MemorialCreated(string indexed memorialId, address creator, uint256 timestamp);
    event ProvenanceRecorded(string indexed memorialId, string eventType, uint256 timestamp);
    event SuccessorDesignated(string indexed memorialId, address successor);

    modifier onlyCreatorOrAuthorized(string memory memorialId) {
        require(
            msg.sender == memorials[memorialId].creator ||
            isAuthorized(memorialId, msg.sender),
            "Not authorized"
        );
        _;
    }

    function registerMemorial(
        string memory memorialId,
        string memory contentHash
    ) public {
        require(memorials[memorialId].createdAt == 0, "Memorial already exists");

        memorials[memorialId] = Memorial({
            memorialId: memorialId,
            creator: msg.sender,
            createdAt: block.timestamp,
            contentHash: contentHash,
            isActive: true
        });

        recordProvenance(
            memorialId,
            "CREATED",
            "Memorial registered on blockchain",
            contentHash
        );

        emit MemorialCreated(memorialId, msg.sender, block.timestamp);
    }

    function recordProvenance(
        string memory memorialId,
        string memory eventType,
        string memory description,
        string memory dataHash
    ) public onlyCreatorOrAuthorized(memorialId) {
        require(memorials[memorialId].isActive, "Memorial not active");

        provenanceChain[memorialId].push(ProvenanceEvent({
            eventType: eventType,
            actor: msg.sender,
            timestamp: block.timestamp,
            description: description,
            dataHash: dataHash
        }));

        emit ProvenanceRecorded(memorialId, eventType, block.timestamp);
    }

    function designateSuccessor(
        string memory memorialId,
        address successor
    ) public onlyCreatorOrAuthorized(memorialId) {
        authorizedActors[memorialId].push(successor);

        recordProvenance(
            memorialId,
            "SUCCESSOR_DESIGNATED",
            "New successor authorized",
            ""
        );

        emit SuccessorDesignated(memorialId, successor);
    }

    function getProvenanceChain(
        string memory memorialId
    ) public view returns (ProvenanceEvent[] memory) {
        return provenanceChain[memorialId];
    }

    function verifyContentIntegrity(
        string memory memorialId,
        string memory contentHash
    ) public view returns (bool) {
        return keccak256(bytes(memorials[memorialId].contentHash)) == 
               keccak256(bytes(contentHash));
    }

    function isAuthorized(
        string memory memorialId,
        address actor
    ) internal view returns (bool) {
        address[] memory actors = authorizedActors[memorialId];
        for (uint i = 0; i < actors.length; i++) {
            if (actors[i] == actor) {
                return true;
            }
        }
        return false;
    }
}
