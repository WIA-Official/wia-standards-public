Chapter 7: Analytics & Insights

WIA Digital Memorial Standard - Visitor Analytics & Engagement Metrics

7.1 Analytics Framework

Memorial analytics provide insights into visitor engagement, tribute activity, and memorial impact while respecting privacy and maintaining appropriate sensitivity.

Key Metrics

  • Page views and unique visitors
  • Tribute submission rates
  • Media engagement (photo/video views)
  • Geographic distribution of visitors
  • Peak visiting times and patterns
  • Candle lighting frequency
  • Social media shares and referrals
  • Average session duration
  • Return visitor rates
  • Device and browser statistics

Example 1: Privacy-Respecting Analytics

analytics-tracker.js
class PrivacyAnalytics {
  constructor(memorialId) {
    this.memorialId = memorialId;
    this.sessionId = this.generateSessionId();
    this.events = [];
  }

  generateSessionId() {
    // Generate anonymous session ID (no PII)
    return 'sess_' + Math.random().toString(36).substr(2, 9);
  }

  trackPageView(page) {
    this.sendEvent({
      type: 'pageview',
      page: page,
      timestamp: Date.now(),
      sessionId: this.sessionId
    });
  }

  trackTributeView(tributeId) {
    this.sendEvent({
      type: 'tribute_view',
      tributeId: tributeId,
      timestamp: Date.now()
    });
  }

  trackMediaView(mediaId, mediaType) {
    this.sendEvent({
      type: 'media_view',
      mediaId: mediaId,
      mediaType: mediaType,
      timestamp: Date.now()
    });
  }

  trackCandleLit() {
    this.sendEvent({
      type: 'candle_lit',
      timestamp: Date.now()
    });
  }

  async sendEvent(event) {
    const payload = {
      memorialId: this.memorialId,
      ...event,
      // Add non-PII context
      viewport: {
        width: window.innerWidth,
        height: window.innerHeight
      },
      deviceType: this.getDeviceType(),
      language: navigator.language
    };

    try {
      await fetch('/api/v1/analytics/track-event', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
    } catch (error) {
      console.error('Analytics tracking error:', error);
    }
  }

  getDeviceType() {
    const ua = navigator.userAgent;
    if (/(tablet|ipad|playbook|silk)|(android(?!.*mobi))/i.test(ua)) {
      return 'tablet';
    }
    if (/Mobile|Android|iP(hone|od)|IEMobile|BlackBerry|Kindle|Silk-Accelerated|(hpw|web)OS|Opera M(obi|ini)/.test(ua)) {
      return 'mobile';
    }
    return 'desktop';
  }
}

// Usage
const analytics = new PrivacyAnalytics('MEM-12345678');
analytics.trackPageView('/memorial/view');
analytics.trackMediaView('MED-ABC123', 'photo');

Example 3: Real-time Analytics

real-time-analytics.js
const WebSocket = require('ws');

class RealTimeAnalytics {
  constructor(memorialId) {
    this.memorialId = memorialId;
    this.activeVisitors = new Map();
    this.wss = new WebSocket.Server({ port: 8080 });
    this.setupWebSocket();
  }

  setupWebSocket() {
    this.wss.on('connection', (ws, req) => {
      const visitorId = this.generateVisitorId();
      
      this.activeVisitors.set(visitorId, {
        ws: ws,
        connectedAt: Date.now(),
        lastActivity: Date.now()
      });

      // Send current active count
      ws.send(JSON.stringify({
        type: 'active_visitors',
        count: this.activeVisitors.size
      }));

      // Handle messages
      ws.on('message', (message) => {
        const data = JSON.parse(message);
        this.handleVisitorActivity(visitorId, data);
      });

      // Handle disconnect
      ws.on('close', () => {
        this.activeVisitors.delete(visitorId);
        this.broadcastActiveCount();
      });
    });

    // Cleanup inactive sessions
    setInterval(() => this.cleanupInactive(), 60000);
  }

  handleVisitorActivity(visitorId, activity) {
    const visitor = this.activeVisitors.get(visitorId);
    if (visitor) {
      visitor.lastActivity = Date.now();
      
      // Track activity
      this.trackActivity(this.memorialId, activity);
      
      // Broadcast update to admin dashboard
      this.broadcastActivity(activity);
    }
  }

  broadcastActiveCount() {
    const message = JSON.stringify({
      type: 'active_visitors',
      count: this.activeVisitors.size
    });

    this.activeVisitors.forEach(visitor => {
      if (visitor.ws.readyState === WebSocket.OPEN) {
        visitor.ws.send(message);
      }
    });
  }

  cleanupInactive() {
    const now = Date.now();
    const timeout = 5 * 60 * 1000; // 5 minutes

    this.activeVisitors.forEach((visitor, id) => {
      if (now - visitor.lastActivity > timeout) {
        visitor.ws.close();
        this.activeVisitors.delete(id);
      }
    });

    this.broadcastActiveCount();
  }

  generateVisitorId() {
    return 'vis_' + Math.random().toString(36).substr(2, 9);
  }
}

// Usage
const analytics = new RealTimeAnalytics('MEM-12345678');
console.log('Real-time analytics server started');
