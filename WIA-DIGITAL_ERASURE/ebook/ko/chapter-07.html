2. 레거시 시스템 통합

2.1 레거시 시스템 분석

시스템 유형 특징 통합 난이도 권장 접근법
Monolithic (단일체) 하나의 큰 애플리케이션 ⭐⭐⭐⭐ Strangler Fig Pattern

COBOL/Mainframe 금융/정부 시스템 ⭐⭐⭐⭐⭐ Bridge Pattern + Proxy

2.2 Strangler Fig 패턴 구현

레거시 시스템을 점진적으로 대체하는 패턴으로, 구형 나무를 감싸며 자라는 무화과나무에서 이름을 따왔습니다.

/**
 * Strangler Fig 패턴: 레거시 시스템 점진적 교체
 * @pattern Migration
 */

// 1단계: Proxy/Facade 생성
class ErasureRequestRouter {
  constructor(
    private legacySystem: LegacyUserService,
    private modernSystem: WIAErasureService
  ) {}

  async deleteUser(userId: string): Promise {
    // 라우팅 규칙: 신규 사용자는 modern, 기존 사용자는 legacy
    const user = await this.getUserInfo(userId);

    if (user.createdAt > new Date('2025-01-01')) {
      // 신규 사용자 → WIA 표준 사용
      return this.modernSystem.processErasureRequest({
        userId: userId,
        scope: this.convertLegacyScopeToWIA(user.dataScope),
        legalBasis: 'PIPA_ARTICLE_21'
      });
    } else {
      // 기존 사용자 → 레거시 시스템
      return this.legacySystem.deleteUserData(userId);
    }
  }

  // 레거시 데이터 스코프를 WIA 형식으로 변환
  private convertLegacyScopeToWIA(legacyScope: any): WIAErasureScope {
    return {
      dataCategories: this.mapLegacyCategories(legacyScope.types),
      platforms: ['WEB', 'MOBILE'],
      includeBackups: legacyScope.includeArchives || false,
      includeThirdParty: true
    };
  }
}

// 2단계: 점진적 트래픽 이동 (Feature Flag 활용)
class FeatureFlaggedErasure {
  async processRequest(userId: string) {
    const rolloutPercentage = await this.getFeatureFlag('wia-erasure-rollout');

    // 0-100% 사이로 점진적 롤아웃
    if (Math.random() * 100 < rolloutPercentage) {
      return this.wiaErasureService.process(userId);
    } else {
      return this.legacyService.process(userId);
    }
  }
}

  • reporting-service: 통계 및 보고서 생성

3.2 서비스 간 통신

통신 패턴 사용 시나리오 기술 장단점
동기식 (Synchronous) 즉시 응답 필요 gRPC, REST 간단하지만 결합도 높음
비동기식 (Asynchronous) 시간이 오래 걸리는 작업 RabbitMQ, Kafka 느슨한 결합, 복잡도 증가
이벤트 기반 (Event-Driven) 여러 서비스 반응 필요 EventBridge, SNS 확장성 우수, 디버깅 어려움
Saga Pattern 분산 트랜잭션 Orchestration / Choreography 일관성 보장, 구현 복잡

3.3 Saga 패턴 구현 예제

/**
 * Saga Pattern: 분산 트랜잭션 관리
 * @pattern Orchestration Saga
 */

class ErasureSagaOrchestrator {
  async executeErasureSaga(requestId: string) {
    const saga = new Saga(requestId);

    try {
      // Step 1: 본인 인증 검증
      await saga.step(
        'verify-identity',
        () => this.verificationService.verify(requestId),
        () => this.verificationService.rollback(requestId)  // 보상 트랜잭션
      );

      // Step 2: 주 데이터베이스 삭제
      await saga.step(
        'delete-primary-db',
        () => this.deletionService.deletePrimary(requestId),
        () => this.deletionService.restorePrimary(requestId)
      );

      // Step 3: 백업 데이터 삭제
      await saga.step(
        'delete-backups',
        () => this.deletionService.deleteBackups(requestId),
        () => this.deletionService.restoreBackups(requestId)
      );

      // Step 4: 제3자 서비스 통지
      await saga.step(
        'notify-third-party',
        () => this.thirdPartyService.notify(requestId),
        () => this.thirdPartyService.cancelNotification(requestId)
      );

      // Step 5: 완료 알림
      await saga.step(
        'send-completion-notification',
        () => this.notificationService.sendCompletion(requestId),
        () => {} // 롤백 불필요
      );

      await saga.commit();
      return { success: true, requestId };

    } catch (error) {
      // 실패 시 자동 보상 트랜잭션 실행
      await saga.rollback();
      throw new Error(`Saga failed: ${error.message}`);
    }
  }
}

5.2 카카오 통합 아키텍처

카카오는 Kafka 기반 이벤트 스트리밍과 MongoDB를 주로 활용합니다.

/**
 * 카카오 플랫폼 이벤트 기반 통합
 * @platform Kakao
 * @architecture Event-Driven
 */

// Kafka Producer: 삭제 요청 발행
import { Kafka } from 'kafkajs';

const kafka = new Kafka({
  clientId: 'kakao-wia-erasure',
  brokers: ['kafka-1.kakao.com:9092', 'kafka-2.kakao.com:9092']
});

const producer = kafka.producer();

async function publishErasureEvent(request: ErasureRequest) {
  await producer.send({
    topic: 'kakao.wia.erasure.requests',
    messages: [
      {
        key: request.userId,
        value: JSON.stringify({
          requestId: request.requestId,
          userId: request.userId,
          timestamp: new Date().toISOString(),
          scope: request.scope,
          platform: 'KAKAO'
        }),
        headers: {
          'wep-version': '1.0.0',
          'source': 'kakao-account-service',
          'correlation-id': request.requestId
        }
      }
    ]
  });
}

// Kafka Consumer: 삭제 작업 처리
const consumer = kafka.consumer({ groupId: 'wia-erasure-workers' });

async function startDeletionWorker() {
  await consumer.subscribe({ topic: 'kakao.wia.erasure.requests' });

  await consumer.run({
    eachMessage: async ({ topic, partition, message }) => {
      const request = JSON.parse(message.value.toString());

      try {
        // 1. 카카오톡 데이터 삭제
        await deleteKakaoTalkData(request.userId);

        // 2. 카카오페이 데이터 삭제
        await deleteKakaoPayData(request.userId);

        // 3. 다음 서비스 데이터 삭제
        await deleteDaumData(request.userId);

        // 4. 완료 이벤트 발행
        await publishCompletionEvent(request.requestId);

      } catch (error) {
        // Dead Letter Queue로 전송
        await publishToDLQ(request, error);
      }
    }
  });
}

// MongoDB 삭제 쿼리
async function deleteKakaoTalkData(userId: string) {
  const db = mongo.db('kakaotalk');

  // 트랜잭션으로 원자성 보장
  const session = mongo.startSession();

  try {
    await session.withTransaction(async () => {
      // 메시지 삭제
      await db.collection('messages').deleteMany({ userId }, { session });

      // 친구 관계 삭제
      await db.collection('friends').deleteMany(
        { $or: [{ userId }, { friendId: userId }] },
        { session }
      );

      // 프로필 삭제
      await db.collection('profiles').deleteOne({ userId }, { session });

      // 첨부파일 삭제 (S3 연동)
      const attachments = await db.collection('attachments')
        .find({ userId }, { session })
        .toArray();

      for (const attachment of attachments) {
        await s3.deleteObject({
          Bucket: 'kakao-attachments',
          Key: attachment.s3Key
        });
      }

      await db.collection('attachments').deleteMany({ userId }, { session });
    });

    console.log(`카카오톡 데이터 삭제 완료: ${userId}`);
  } finally {
    await session.endSession();
  }
}

# .github/workflows/deploy.yml
name: WIA-DIGITAL_ERASURE CI/CD

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

env:
  NODE_VERSION: '18'
  DOCKER_IMAGE: wia/digital-erasure

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'

      - name: Install Dependencies
        run: npm ci

      - name: Run Unit Tests
        run: npm test -- --coverage

      - name: Run E2E Tests
        run: npm run test:e2e

      - name: Upload Coverage
        uses: codecov/codecov-action@v3
        with:
          files: ./coverage/lcov.info

  security:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Run Snyk Security Scan
        uses: snyk/actions/node@master
        env:
          SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}

      - name: SonarQube Scan
        uses: sonarsource/sonarcloud-github-action@master
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          SONAR_TOKEN: ${{ secrets.SONAR_TOKEN }}

  build:
    needs: [test, security]
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v2

      - name: Login to Registry
        uses: docker/login-action@v2
        with:
          registry: ${{ secrets.REGISTRY_URL }}
          username: ${{ secrets.REGISTRY_USERNAME }}
          password: ${{ secrets.REGISTRY_PASSWORD }}

      - name: Build and Push
        uses: docker/build-push-action@v4
        with:
          context: .
          push: true
          tags: |
            ${{ env.DOCKER_IMAGE }}:${{ github.sha }}
            ${{ env.DOCKER_IMAGE }}:latest
          cache-from: type=gha
          cache-to: type=gha,mode=max

  deploy-staging:
    needs: build
    runs-on: ubuntu-latest
    environment: staging
    steps:
      - name: Deploy to Kubernetes (Staging)
        uses: azure/k8s-deploy@v4
        with:
          namespace: wia-erasure-staging
          manifests: |
            k8s/deployment.yml
            k8s/service.yml
          images: ${{ env.DOCKER_IMAGE }}:${{ github.sha }}

      - name: Run Smoke Tests
        run: |
          curl -f https://staging.wia-erasure.com/health || exit 1

  deploy-production:
    needs: deploy-staging
    if: github.ref == 'refs/heads/main'
    runs-on: ubuntu-latest
    environment: production
    steps:
      - name: Canary Deployment (10%)
        run: |
          kubectl set image deployment/wia-erasure \
            erasure-service=${{ env.DOCKER_IMAGE }}:${{ github.sha }} \
            -n production
          kubectl scale deployment/wia-erasure-canary --replicas=1

      - name: Wait and Monitor
        run: sleep 300  # 5분 대기

      - name: Check Error Rate
        run: |
          # Prometheus 메트릭 확인
          ERROR_RATE=$(curl -s prometheus.svc/api/v1/query?query=error_rate)
          if [ "$ERROR_RATE" -gt "0.01" ]; then
            echo "Error rate too high, rolling back"
            exit 1
          fi

      - name: Full Rollout (100%)
        run: |
          kubectl set image deployment/wia-erasure \
            erasure-service=${{ env.DOCKER_IMAGE }}:${{ github.sha }} \
            -n production
          kubectl scale deployment/wia-erasure --replicas=10
