// E2E 테스트 예시
describe('WIA-DIGITAL_ERASURE E2E Tests', () => {
  it('should complete full erasure within deadline', async () => {
    // 1. 삭제 요청 생성
    const request = await client.createErasureRequest({
      type: 'full',
      subject: { email: 'test@example.com' }
    });
    expect(request.status).toBe('pending');

    // 2. 상태 확인 (완료까지 대기)
    const completed = await waitForCompletion(request.id, {
      timeout: 60000,
      pollInterval: 1000
    });
    expect(completed.status).toBe('verified');

    // 3. 검증 - 데이터가 실제로 삭제되었는지
    const userData = await database.query(
      'SELECT * FROM users WHERE email = ?',
      ['test@example.com']
    );
    expect(userData).toHaveLength(0);

    // 4. 인증서 발급 확인
    const certificate = await client.getCertificate(request.id);
    expect(certificate.level).toBe('gold');
  });

  it('should handle participant failure gracefully', async () => {
    // 참여자 하나 강제 종료
    await stopParticipant('backup-system');

    const request = await client.createErasureRequest({
      type: 'full',
      subject: { email: 'test2@example.com' }
    });

    // 부분 완료 상태 확인
    const result = await waitForCompletion(request.id);
    expect(result.status).toBe('partial_complete');
    expect(result.failedSystems).toContain('backup-system');
  });
});

4.2 자동화된 리포트 생성

// 컴플라이언스 리포트 자동 생성
interface ComplianceReport {
  period: { start: Date; end: Date };
  metrics: {
    totalRequests: number;
    completedOnTime: number;
    slaCompliance: number; // 백분율
    averageCompletionTime: number; // 시간
  };
  breakdown: {
    byRegulation: Record<string, number>; // GDPR, CCPA 등
    byType: Record<string, number>; // full, partial
    byStatus: Record<string, number>;
  };
  incidents: Incident[];
  recommendations: string[];
}

async function generateMonthlyReport(
  month: Date
): Promise<ComplianceReport> {
  const start = startOfMonth(month);
  const end = endOfMonth(month);

  // 메트릭 수집
  const requests = await db.erasureRequests.findMany({
    where: {
      createdAt: { gte: start, lte: end }
    }
  });

  const completedOnTime = requests.filter(r =>
    r.status === 'completed' &&
    r.completedAt <= r.deadline
  ).length;

  // SLA 계산 (30일 이내 완료)
  const slaCompliance = (completedOnTime / requests.length) * 100;

  // 개선 권고사항 생성
  const recommendations = generateRecommendations({
    slaCompliance,
    incidents: await getMonthlyIncidents(start, end)
  });

  return {
    period: { start, end },
    metrics: {
      totalRequests: requests.length,
      completedOnTime,
      slaCompliance,
      averageCompletionTime: calculateAvgTime(requests)
    },
    breakdown: {
      byRegulation: groupBy(requests, 'legalBasis.regulation'),
      byType: groupBy(requests, 'type'),
      byStatus: groupBy(requests, 'status')
    },
    incidents: await getMonthlyIncidents(start, end),
    recommendations
  };
}

5.2 규제별 구현 체크리스트

// 규제별 요구사항 검증
const regulationRequirements = {
  GDPR: {
    maxResponseDays: 30,
    requiredVerification: ['identity'],
    mandatoryFields: ['legalBasis', 'dataCategories'],
    retentionProof: true,
    thirdPartyNotification: true
  },
  CCPA: {
    maxResponseDays: 45,
    requiredVerification: ['identity', 'authorization'],
    mandatoryFields: ['requestType'],
    retentionProof: false,
    thirdPartyNotification: false
  },
  LGPD: {
    maxResponseDays: 15,
    requiredVerification: ['identity'],
    mandatoryFields: ['legalBasis'],
    retentionProof: true,
    thirdPartyNotification: true
  }
};

function validateRequest(
  request: ErasureRequest,
  regulation: string
): ValidationResult {
  const req = regulationRequirements[regulation];
  const errors: string[] = [];

  // 필수 필드 검증
  for (const field of req.mandatoryFields) {
    if (!request[field]) {
      errors.push(`Missing required field: ${field}`);
    }
  }

  // 기한 검증
  const daysSinceRequest = daysBetween(
    request.createdAt,
    new Date()
  );
  if (daysSinceRequest > req.maxResponseDays) {
    errors.push(`Deadline exceeded: ${daysSinceRequest} days`);
  }

  return {
    isValid: errors.length === 0,
    errors,
    regulation,
    requirements: req
  };
}
