WIA-ERASURE-PROTOCOL (WEP) v1.0

1. 프로토콜 개요

WIA-ERASURE-PROTOCOL (WEP)은 서로 다른 시스템과 플랫폼 간에 개인정보 삭제 요청을 안전하고 효율적으로 전달하기 위한 통신 프로토콜입니다. API가 단일 시스템 내의 인터페이스라면, 프로토콜은 여러 시스템 간의 통신 규약입니다.

📌 학습 목표

  • WEP 프로토콜의 구조와 메시지 형식 이해
  • 시스템 간 동기화 메커니즘
  • 블록체인 기반 삭제 검증 방법
  • 제3자 플랫폼 연동 프로토콜
  • 국제 상호운용성 보장 방안

2. WEP 프로토콜 스택

계층 프로토콜 역할 기술
응용 계층 WEP-APP 비즈니스 로직, 데이터 형식 JSON-LD, Schema.org
표현 계층 WEP-DATA 데이터 인코딩, 압축 Protocol Buffers, gRPC
세션 계층 WEP-SESSION 연결 관리, 상태 유지 WebSocket, SSE
전송 계층 WEP-TRANSPORT 신뢰성 보장, 재전송 HTTP/3, QUIC
보안 계층 WEP-SECURITY 암호화, 인증, 서명 TLS 1.3, mTLS

3.1 기본 메시지 구조

 * WEP 표준 메시지 형식

 */

{
  // 메시지 헤더
  "header": {
    "version": "1.0.0",
    "messageId": "MSG-2025-0107-001234",
    "messageType": "ERASURE_REQUEST",  // REQUEST | RESPONSE | NOTIFICATION | ACK
    "timestamp": "2025-01-07T14:30:00Z",
    "sender": {
      "platformId": "NAVER_KR",
      "systemId": "naver-erasure-service",
      "certificate": "-----BEGIN CERTIFICATE-----..."
    },
    "receiver": {
      "platformId": "KAKAO_KR",
      "systemId": "kakao-data-management"
    },
    "priority": "HIGH",  // LOW | NORMAL | HIGH | URGENT
    "encryption": "AES-256-GCM",
    "signature": "SHA256-RSA"
  },

  // 메시지 본문
  "body": {
    "requestId": "ER-2025-0107-001234",
    "userId": {
      "type": "EMAIL",
      "value": "user@example.com",
      "hash": "SHA256:a3b2c1d4..."  // 프라이버시 보호
    },
    "action": "DELETE",  // DELETE | ANONYMIZE | ARCHIVE
    "scope": {
      "dataTypes": ["PROFILE", "POSTS", "MESSAGES"],
      "platforms": ["WEB", "MOBILE"],
      "includeBackups": true,
      "includeThirdParty": true
    },
    "legalBasis": "PIPA_ARTICLE_21",
    "deadline": "2025-01-17T23:59:59Z",
    "verificationProof": {
      "method": "OTP",
      "verified": true,
      "timestamp": "2025-01-07T14:25:00Z"
    }
  },

  // 메시지 푸터 (무결성 검증)
  "footer": {
    "checksum": "CRC32:a1b2c3d4",

    "chain": {
      "previousHash": "SHA256:prev_message_hash",
      "blockNumber": 12345
    }
  }
}

3.2 메시지 유형별 정의

메시지 유형 방향 설명 응답 필요
ERASURE_REQUEST Client → Server 삭제 요청 전송 ✅ 필수
ERASURE_RESPONSE Server → Client 요청 수신 확인 ✅ ACK 필요
STATUS_UPDATE Server → Client 진행 상황 통지 ❌ 선택
COMPLETION_NOTIFY Server → Client 삭제 완료 알림 ✅ ACK 필요
ERROR_NOTIFY Server → Client 오류 통지 ✅ 필수
THIRD_PARTY_RELAY Server → Third Party 제3자에게 삭제 요청 전달 ✅ 필수

4. 시스템 간 동기화 프로토콜

4.1 Pub/Sub 아키텍처

WEP는 Google Cloud Pub/Sub, AWS SNS/SQS, Apache Kafka 등과 호환되는 Pub/Sub 패턴을 사용합니다.

/**
 * WEP Pub/Sub 토픽 설계
 */

// 토픽 구조
const WEP_TOPICS = {
  // 삭제 요청 토픽
  erasureRequests: 'wep.erasure.requests.{region}.{platform}',

  // 상태 업데이트 토픽
  statusUpdates: 'wep.status.updates.{requestId}',

  // 완료 알림 토픽
  completionNotifications: 'wep.completion.notifications',

  // 제3자 전달 토픽
  thirdPartyRelay: 'wep.thirdparty.relay.{partnerId}',

  // 오류 알림 토픽
  errorNotifications: 'wep.errors.{severity}'
};

// 네이버 → 카카오 삭제 요청 전달 예제
import { PubSub } from '@google-cloud/pubsub';

const pubsub = new PubSub();

async function relayErasureRequestToKakao(request) {
  const topic = pubsub.topic('wep.thirdparty.relay.kakao');

  const message = {
    header: {
      version: '1.0.0',
      messageType: 'THIRD_PARTY_RELAY',
      sender: { platformId: 'NAVER_KR' },
      receiver: { platformId: 'KAKAO_KR' }
    },
    body: {
      requestId: request.requestId,
      userId: {
        type: 'EMAIL',
        value: request.email,
        linkedAccount: {
          platform: 'KAKAO',
          discoveredBy: 'OAUTH_LINK'
        }
      },
      action: 'DELETE',
      reason: '네이버 계정 삭제에 따른 연동 계정 삭제',
      legalBasis: 'PIPA_ARTICLE_21'
    }
  };

  await topic.publish(Buffer.from(JSON.stringify(message)));
  console.log('카카오에 삭제 요청 전달 완료');
}

5. 블록체인 기반 삭제 검증

5.1 검증 체인 구조

WEP는 선택적으로 블록체인 기술을 활용하여 삭제의 무결성을 검증합니다. 이는 개인정보보호위원회의 감사 시 증거자료로 활용될 수 있습니다.

블록 유형 내용 블록 크기 생성 주기
Genesis Block 체인 시작점 1 KB 최초 1회
Request Block 삭제 요청 기록 2-5 KB 요청 시
Verification Block 인증 완료 기록 1-2 KB 인증 시
Deletion Block 삭제 완료 기록 5-10 KB 삭제 완료 시
Audit Block 검증 및 감사 기록 3-7 KB 주기적

5.2 스마트 컨트랙트 예제

/**
 * WEP 삭제 검증 스마트 컨트랙트 (Solidity)
 * @blockchain Ethereum / Polygon
 */

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

contract WIAErasureVerification {
    struct ErasureRecord {
        string requestId;
        address requester;
        uint256 timestamp;
        bytes32 dataHash;  // 삭제된 데이터의 해시
        bool verified;
        string[] systemsCompleted;
    }

    mapping(string => ErasureRecord) public records;
    mapping(address => bool) public authorizedPlatforms;

    event ErasureRequested(string requestId, address requester, uint256 timestamp);
    event ErasureVerified(string requestId, uint256 timestamp);
    event SystemCompleted(string requestId, string systemName);

    modifier onlyAuthorized() {
        require(authorizedPlatforms[msg.sender], "Not authorized");
        _;
    }

    // 삭제 요청 기록
    function recordErasureRequest(
        string memory _requestId,
        bytes32 _dataHash
    ) public onlyAuthorized {
        require(records[_requestId].timestamp == 0, "Request already exists");

        records[_requestId] = ErasureRecord({
            requestId: _requestId,
            requester: msg.sender,
            timestamp: block.timestamp,
            dataHash: _dataHash,
            verified: false,
            systemsCompleted: new string[](0)
        });

        emit ErasureRequested(_requestId, msg.sender, block.timestamp);
    }

    // 시스템별 삭제 완료 기록
    function recordSystemCompletion(
        string memory _requestId,
        string memory _systemName
    ) public onlyAuthorized {
        require(records[_requestId].timestamp > 0, "Request not found");

        records[_requestId].systemsCompleted.push(_systemName);
        emit SystemCompleted(_requestId, _systemName);
    }

    // 최종 검증
    function verifyErasure(string memory _requestId) public onlyAuthorized {
        require(records[_requestId].timestamp > 0, "Request not found");
        require(!records[_requestId].verified, "Already verified");

        records[_requestId].verified = true;
        emit ErasureVerified(_requestId, block.timestamp);
    }

    // 검증 결과 조회
    function getErasureStatus(string memory _requestId)
        public view returns (bool verified, uint256 timestamp, uint256 systemsCount)
    {
        ErasureRecord memory record = records[_requestId];
        return (record.verified, record.timestamp, record.systemsCompleted.length);
    }
}

6. 제3자 플랫폼 연동

6.1 연동 시나리오

사용자가 네이버 계정을 삭제할 때, 네이버와 연동된 제3자 서비스(예: 배달앱, 쇼핑몰)에도 삭제 요청이 자동으로 전달되어야 합니다.

🔗 연동 플로우

  1. Step 1: 사용자가 네이버 계정 삭제 요청
  2. Step 2: 네이버가 OAuth 연동 서비스 목록 조회
  3. Step 3: 각 제3자 서비스에 WEP 메시지 전송
  4. Step 4: 제3자 서비스가 삭제 처리
  5. Step 5: 완료 응답을 네이버에 회신
  6. Step 6: 모든 연동 서비스 삭제 완료 후 네이버 계정 최종 삭제

6.2 제3자 연동 구현

/**
 * 제3자 플랫폼 연동 매니저
 */

class ThirdPartyErasureManager {
  private linkedServices: Map;

  constructor() {
    this.linkedServices = new Map();
  }

  // 사용자의 연동 서비스 목록 조회
  async getLinkedServices(userId: string): Promise {
    // OAuth 연동 기록 조회
    const oauthLinks = await db.query(`
      SELECT service_id, service_name, linked_at, access_scope
      FROM oauth_links
      WHERE user_id = ? AND status = 'ACTIVE'
    `, [userId]);

    return oauthLinks.map(link => ({
      serviceId: link.service_id,
      serviceName: link.service_name,
      linkedAt: link.linked_at,
      wepEndpoint: this.getWepEndpoint(link.service_id)
    }));
  }

  // 모든 연동 서비스에 삭제 요청 전송
  async propagateErasureRequest(requestId: string, userId: string) {
    const services = await this.getLinkedServices(userId);

    const promises = services.map(service =>
      this.sendErasureToService(service, requestId, userId)
    );

    const results = await Promise.allSettled(promises);

    // 결과 집계
    const summary = {
      total: services.length,
      success: results.filter(r => r.status === 'fulfilled').length,
      failed: results.filter(r => r.status === 'rejected').length,
      details: results.map((r, i) => ({
        service: services[i].serviceName,
        status: r.status,
        error: r.status === 'rejected' ? r.reason : null
      }))
    };

    return summary;
  }

  // 개별 서비스에 WEP 메시지 전송
  private async sendErasureToService(
    service: LinkedService,
    requestId: string,
    userId: string
  ) {
    const wepMessage = {
      header: {
        version: '1.0.0',
        messageType: 'THIRD_PARTY_RELAY',
        sender: { platformId: 'NAVER_KR' },
        receiver: { platformId: service.serviceId }
      },
      body: {
        requestId: requestId,
        userId: { type: 'NAVER_ID', value: userId },
        action: 'DELETE',
        legalBasis: 'PIPA_ARTICLE_21',
        deadline: new Date(Date.now() + 10 * 24 * 60 * 60 * 1000)  // 10일 후
      }
    };

    const response = await fetch(service.wepEndpoint, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/wep+json',
        'WEP-Version': '1.0.0',
        'Authorization': `Bearer ${await this.getServiceToken(service.serviceId)}`
      },
      body: JSON.stringify(wepMessage)
    });

    if (!response.ok) {
      throw new Error(`Service ${service.serviceName} failed: ${response.statusText}`);
    }

    return await response.json();
  }
}

7.1 지역별 프로토콜 변형

지역 프로토콜 변형 주요 차이점 법적 근거

7.2 국제 데이터 전송 프로토콜

/**
 * 국제 데이터 전송 표준 계약 조항 (SCC) 통합
 */

interface CrossBorderErasureRequest {
  requestId: string;
  sourceRegion: 'EU' | 'KR' | 'US' | 'JP' | 'CN';
  targetRegion: 'EU' | 'KR' | 'US' | 'JP' | 'CN';

  // Standard Contractual Clauses (SCC) 준수
  legalMechanism: 'SCC' | 'BCR' | 'ADEQUACY_DECISION' | 'DEROGATION';

  dataSubject: {
    id: string;
    nationality: string;
    residence: string;
  };

  // 지역별 법적 요구사항
  compliance: {
    sourceCompliance: string[];  // e.g., ['GDPR_ARTICLE_17', 'EDPB_GUIDELINES']
    targetCompliance: string[];  // e.g., ['PIPA_ARTICLE_21']
  };

  // 데이터 현지화 (Data Localization) 준수
  dataLocalization: {
    mustStayInRegion: boolean;
    allowedRegions: string[];
    transferMechanism: string;
  };
}

// 한국 → EU 삭제 요청 전송 예제
const krToEuRequest: CrossBorderErasureRequest = {
  requestId: 'ER-2025-0107-001234',
  sourceRegion: 'KR',
  targetRegion: 'EU',
  legalMechanism: 'SCC',  // Standard Contractual Clauses
  dataSubject: {
    id: 'user@example.com',
    nationality: 'KR',
    residence: 'DE'  // 독일 거주
  },
  compliance: {
    sourceCompliance: ['PIPA_ARTICLE_21'],
    targetCompliance: ['GDPR_ARTICLE_17', 'EDPB_GUIDELINES_4_2019']
  },
  dataLocalization: {
    mustStayInRegion: false,
    allowedRegions: ['EU', 'KR'],
    transferMechanism: 'EU_KR_ADEQUACY_DECISION'
  }
};

8. 프로토콜 보안

8.1 암호화 계층

🔐 보안 요구사항

  • 전송 암호화: TLS 1.3 이상 필수
  • 메시지 암호화: AES-256-GCM

  • 디지털 서명: RSA-4096 또는 Ed25519
  • 해시 알고리즘: SHA-256 이상
  • Perfect Forward Secrecy: 세션별 키 생성
  • Certificate Pinning: 중간자 공격 방지
