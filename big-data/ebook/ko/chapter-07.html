MapReduce와 WIA 통합:

// Hadoop MapReduce Job with WIA
public class WIAOrdersAnalysis {
  public static class OrderMapper
      extends Mapper {

    public void map(LongWritable key, WIARecord value, Context context)
        throws IOException, InterruptedException {
      String category = value.getString("category");
      int quantity = value.getInt("quantity");
      context.write(new Text(category), new IntWritable(quantity));
    }
  }

  public static class SumReducer
      extends Reducer {

    public void reduce(Text key, Iterable values, Context context)
        throws IOException, InterruptedException {
      int sum = 0;
      for (IntWritable val : values) {
        sum += val.get();
      }
      context.write(key, new IntWritable(sum));
    }
  }
}

from pyspark.sql import SparkSession

spark = SparkSession.builder \
    .appName("WIA Analysis") \
    .config("spark.wia.schema.registry.url", "http://schema-registry:8081") \
    .getOrCreate()

# WIA 형식으로 저장된 데이터 읽기
orders_df = spark.read \
    .format("wia") \
    .option("path", "s3://data/orders/") \
    .option("schema", "Order.v1.0") \
    .load()

# Spark SQL로 분석
orders_df.createOrReplaceTempView("orders")
result = spark.sql("""
    SELECT
        category,
        COUNT(*) as order_count,
        SUM(total_amount) as revenue,
        AVG(total_amount) as avg_order_value
    FROM orders
    WHERE order_date >= '2025-01-01'
    GROUP BY category
    ORDER BY revenue DESC
""")

# WIA 형식으로 결과 저장
result.write \
    .format("wia") \
    .option("compression", "zstd") \
    .mode("overwrite") \
    .save("s3://analytics/category-stats/")

7.2.3 Apache Flink 통합

Flink는 stateful 스트림 처리에 특화된 프레임워크로, 정확히 한 번 보장과 낮은 지연시간을 제공합니다.

// Flink에서 WIA 스트림 처리
import org.apache.flink.streaming.api.environment.StreamExecutionEnvironment;
import io.wia.flink.WIADeserializationSchema;
import io.wia.flink.WIASerializationSchema;

StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();

// WIA Kafka 소스
DataStream orders = env
    .addSource(new FlinkKafkaConsumer<>(
        "orders",
        new WIADeserializationSchema<>(Order.class, schemaRegistry),
        kafkaProps
    ))
    .uid("orders-source");

// 윈도우 집계
DataStream stats = orders
    .keyBy(order -> order.getCategory())
    .window(TumblingEventTimeWindows.of(Time.minutes(5)))
    .aggregate(new CategoryAggregator())
    .uid("category-aggregator");

// WIA Kafka 싱크
stats.addSink(new FlinkKafkaProducer<>(
    "category-stats",
    new WIASerializationSchema<>(CategoryStats.class, schemaRegistry),
    kafkaProps,
    FlinkKafkaProducer.Semantic.EXACTLY_ONCE
)).uid("category-stats-sink");

env.execute("WIA Order Analytics");

7.4.1 Tableau 연동

Tableau는 시각적 데이터 분석의 업계 표준입니다. WIA-DATA-001은 Tableau 웹 데이터 커넥터(WDC)와 JDBC 드라이버를 제공합니다.

// WIA Tableau 웹 데이터 커넥터
(function() {
  var myConnector = tableau.makeConnector();

  myConnector.getSchema = function(schemaCallback) {
    // WIA 스키마 레지스트리에서 스키마 가져오기
    fetch('http://schema-registry:8081/schemas/Order/v1.0')
      .then(response => response.json())
      .then(schema => {
        var cols = schema.fields.map(field => ({
          id: field.name,
          dataType: wiaTypeToTableauType(field.type),
          alias: field.description
        }));

        var tableSchema = {
          id: "orders",
          alias: "Orders",
          columns: cols
        };

        schemaCallback([tableSchema]);
      });
  };

  myConnector.getData = function(table, doneCallback) {
    // WIA Query API를 통해 데이터 가져오기
    fetch('http://wia-api:8080/query/orders?limit=10000')
      .then(response => response.json())
      .then(data => {
        table.appendRows(data.records);
        doneCallback();
      });
  };

  tableau.registerConnector(myConnector);
})();

from airflow import DAG
from airflow.providers.wia.operators.wia import WIAOperator
from airflow.providers.wia.sensors.wia import WIAFileSensor
from datetime import datetime, timedelta

with DAG(

    schedule_interval='0 2 * * *',  # 매일 새벽 2시

    # 1. 새 파일 대기
    wait_for_orders = WIAFileSensor(
        task_id='wait_for_orders_file',

        poke_interval=60

        task_id='validate_orders_data',
        wia_conn_id='wia_api',
        operation='validate',
        params={
            'schema': 'Order',
            'version': '1.0',
            'path': 's3://data-lake/raw/orders/{{ ds }}/'
        }
    )

    # 3. Spark 작업 실행
    process_orders = WIAOperator(
        task_id='process_orders',
        wia_conn_id='wia_emr',
        operation='submit_spark_job',
        params={
            'script': 's3://scripts/process_orders.py',
            'args': ['--date', '{{ ds }}'],
            'spark_config': {
                'spark.executor.memory': '4g',
                'spark.executor.cores': 2
            }
        }
    )

    # 4. 집계 및 통계 계산
    aggregate_stats = WIAOperator(
        task_id='aggregate_daily_stats',
        wia_conn_id='wia_api',
        operation='aggregate',
        params={
            'source': 's3://data-lake/processed/orders/{{ ds }}/',

    # 5. BigQuery로 로드
    load_to_bigquery = WIAOperator(
        task_id='load_to_bigquery',
        wia_conn_id='wia_gcp',
        operation='load',
        params={
            'source': 's3://data-lake/analytics/daily-stats/{{ ds }}/',
            'destination': 'project.dataset.daily_stats',
            'write_disposition': 'WRITE_APPEND'
        }
    )

    # 6. 알림 전송
    send_notification = WIAOperator(
        task_id='send_completion_notification',
        wia_conn_id='wia_api',
        operation='notify',
        params={
            'channel': 'slack',
            'message': 'Daily orders pipeline completed for {{ ds }}'
        }
    )

    # 작업 의존성 정의
    wait_for_orders >> validate_data >> process_orders >> aggregate_stats >> load_to_bigquery >> send_notification

7.5.2 Prefect 통합

Prefect는 최신 워크플로우 오케스트레이션 도구로 동적 워크플로우와 관찰성을 제공합니다.

from prefect import flow, task
from prefect_wia import WIAClient
from datetime import datetime

@task
def extract_orders(date: str) -> list:
    """S3에서 WIA 형식 주문 데이터 추출"""
    client = WIAClient()
    orders = client.read_from_s3(
        bucket='data-lake',
        prefix=f'raw/orders/{date}/',
        schema='Order',
        version='1.0'
    )
    return orders

@task
def transform_orders(orders: list) -> list:
    """주문 데이터 변환 및 강화"""
    transformed = []
    for order in orders:
        transformed.append({
            **order,
            'order_value_tier': categorize_value(order['total_amount']),
            'days_since_last_order': calculate_recency(order['customer_id'])
        })
    return transformed

@task
def load_to_warehouse(data: list, destination: str):
    """데이터 웨어하우스에 로드"""
    client = WIAClient()
    client.write_to_bigquery(
        data=data,
        table=destination,
        write_disposition='WRITE_APPEND'
    )

@flow(name="WIA Orders ETL")
def wia_orders_etl_flow(date: str = None):
    """WIA 주문 데이터 ETL 플로우"""
    if date is None:
        date = datetime.now().strftime('%Y-%m-%d')

    # 작업 실행
    orders = extract_orders(date)
    transformed = transform_orders(orders)
    load_to_warehouse(transformed, 'project.dataset.orders')

# 플로우 배포
if __name__ == "__main__":
    wia_orders_etl_flow.serve(

7.7 하이브리드 및 멀티클라우드 아키텍처

7.7.1 멀티클라우드 전략

시나리오 아키텍처 장점 과제
Best-of-Breed 각 클라우드의 최고 서비스 조합 최상의 기능 활용 복잡성 증가
Disaster Recovery 주 클라우드 + 백업 클라우드 고가용성 동기화 지연
Data Sovereignty 지역별 클라우드 분산 규정 준수 데이터 일관성
Cost Optimization 워크로드별 클라우드 선택 비용 절감 관리 오버헤드

멀티클라우드 WIA 파이프라인 예시:

// AWS에서 데이터 수집, GCP에서 분석, Azure에서 서빙
const multiCloudPipeline = WIA.createPipeline({
  name: 'multi-cloud-analytics'
});

// 1. AWS S3에서 원시 데이터 읽기
const rawData = multiCloudPipeline
  .read({
    provider: 'aws',
    source: 's3://aws-bucket/raw-data/',
    credentials: awsCredentials
  });

// 2. GCP Dataflow에서 처리
const processed = rawData
  .process({
    provider: 'gcp',
    engine: 'dataflow',
    script: 'gs://gcp-bucket/scripts/transform.py',
    credentials: gcpCredentials
  });

// 3. Azure Synapse에 저장
processed
  .write({
    provider: 'azure',
    destination: 'synapse://workspace/database/table',
    credentials: azureCredentials
  });

// 파이프라인 실행
await multiCloudPipeline.run();

7.8 장 요약

핵심 요점

  1. 플러그인 아키텍처: WIA-DATA-001은 Source, Sink, Transform, Router 커넥터를 통해 다양한 시스템과 통합할 수 있습니다. 커넥터는 표준화된 생명주기를 따르며 동적으로 로드, 시작, 정지할 수 있습니다.
  2. 빅데이터 프레임워크 통합: Hadoop HDFS 및 MapReduce, Spark RDD/DataFrame/SQL, Flink 스트리밍과 네이티브 통합을 제공합니다. 동일한 WIA 데이터를 모든 프레임워크에서 일관되게 처리할 수 있습니다.
  3. 클라우드 플랫폼 지원: AWS EMR/Glue/Kinesis/Redshift, GCP BigQuery/Dataproc/Dataflow/Pub/Sub, Azure Synapse/HDInsight/Event Hubs와 완전 통합됩니다. 각 클라우드의 관리형 서비스를 활용하여 운영 부담을 줄일 수 있습니다.
  4. BI 도구 연동: Tableau 웹 데이터 커넥터, Power BI 커스텀 커넥터, Looker LookML 모델을 제공하여 비즈니스 사용자가 WIA 데이터를 쉽게 시각화하고 분석할 수 있습니다.
  5. 워크플로우 오케스트레이션: Apache Airflow DAG와 Prefect 플로우를 통해 복잡한 데이터 파이프라인을 코드로 정의하고 스케줄링할 수 있습니다. WIA Operator와 Sensor로 파이프라인 단계를 표준화합니다.
  6. 데이터 거버넌스: Apache Atlas 통합으로 메타데이터 관리, 데이터 카탈로그, 데이터 계보 추적을 자동화합니다. 모든 데이터 변환 과정이 투명하게 기록되어 규정 준수와 감사를 지원합니다.
  7. 멀티클라우드 지원: Best-of-Breed, Disaster Recovery, Data Sovereignty, Cost Optimization 등 다양한 멀티클라우드 전략을 지원합니다. WIA의 표준 인터페이스로 클라우드 간 데이터 이동이 간소화됩니다.
  8. 레거시 시스템 통합: JDBC/ODBC 드라이버, REST API, 파일 기반 통합 등 다양한 방법으로 기존 시스템과 연동할 수 있어 점진적 마이그레이션이 가능합니다.
