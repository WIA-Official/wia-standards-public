Privacy in Big Data

As organizations collect and analyze increasingly large datasets, protecting individual privacy becomes paramount. The WIA-BIG_DATA standard mandates privacy-preserving techniques that enable valuable analytics while ensuring compliance with regulations like GDPR, CCPA, and HIPAA.

Privacy-preserving analytics balances two competing goals: extracting maximum value from data and protecting individuals from identification or sensitive attribute disclosure.

Differential Privacy

Differential privacy provides mathematical guarantees that the output of an analysis does not reveal whether any individual's data was included in the input. It achieves this by adding carefully calibrated noise to query results.

Differential Privacy Guarantee

A randomized algorithm M provides epsilon-differential privacy if for any two datasets D1 and D2 differing by at most one record, and for any output S:

Pr[M(D1) in S] <= exp(epsilon) * Pr[M(D2) in S]

Lower epsilon values provide stronger privacy guarantees but reduce accuracy.

# Differential Privacy Implementation with PySpark
from pyspark.sql import SparkSession
from pyspark.sql.functions import *
import numpy as np

class DifferentialPrivacyEngine:
    def __init__(self, epsilon=1.0, delta=1e-5):
        self.epsilon = epsilon
        self.delta = delta

    def laplace_mechanism(self, true_value, sensitivity):
        """Add Laplace noise for epsilon-DP"""
        scale = sensitivity / self.epsilon
        noise = np.random.laplace(0, scale)
        return true_value + noise

    def gaussian_mechanism(self, true_value, sensitivity):
        """Add Gaussian noise for (epsilon, delta)-DP"""
        sigma = sensitivity * np.sqrt(2 * np.log(1.25 / self.delta)) / self.epsilon
        noise = np.random.normal(0, sigma)
        return true_value + noise

    def private_count(self, df, condition=None):
        """Compute differentially private count"""
        if condition:
            true_count = df.filter(condition).count()
        else:
            true_count = df.count()

        # Sensitivity of count is 1
        return self.laplace_mechanism(true_count, sensitivity=1)

    def private_sum(self, df, column, clip_lower, clip_upper):
        """Compute differentially private sum with clipping"""
        # Clip values to bound sensitivity
        clipped_df = df.withColumn(
            column + "_clipped",
            when(col(column) < clip_lower, clip_lower)
            .when(col(column) > clip_upper, clip_upper)
            .otherwise(col(column))
        )

        true_sum = clipped_df.agg(sum(column + "_clipped")).collect()[0][0]
        sensitivity = clip_upper - clip_lower

        return self.laplace_mechanism(true_sum, sensitivity)

    def private_mean(self, df, column, clip_lower, clip_upper):
        """Compute differentially private mean"""
        # Split privacy budget between count and sum
        count_engine = DifferentialPrivacyEngine(self.epsilon / 2, self.delta / 2)
        sum_engine = DifferentialPrivacyEngine(self.epsilon / 2, self.delta / 2)

        private_n = count_engine.private_count(df)
        private_s = sum_engine.private_sum(df, column, clip_lower, clip_upper)

        return private_s / private_n if private_n > 0 else 0

# Usage example
dp_engine = DifferentialPrivacyEngine(epsilon=0.5)

# Private count of customers by region
private_count = dp_engine.private_count(customers_df, "region = 'US'")

# Private average order value
private_avg = dp_engine.private_mean(
    orders_df, "order_total",
    clip_lower=0, clip_upper=1000
)

K-Anonymity and L-Diversity

K-anonymity ensures that each record in a dataset is indistinguishable from at least k-1 other records with respect to quasi-identifier attributes. L-diversity extends this by ensuring diversity in sensitive attributes.

# K-Anonymity Implementation
from pyspark.sql.functions import col, count, collect_set, size

class KAnonymizer:
    def __init__(self, spark, k=5):
        self.spark = spark
        self.k = k

    def generalize_age(self, age, level):
        """Generalize age to ranges"""
        if level == 0:
            return age
        elif level == 1:
            return f"{(age // 5) * 5}-{(age // 5) * 5 + 4}"
        elif level == 2:
            return f"{(age // 10) * 10}-{(age // 10) * 10 + 9}"
        else:
            return f"{(age // 20) * 20}-{(age // 20) * 20 + 19}"

    def generalize_zipcode(self, zipcode, level):
        """Generalize zipcode by masking digits"""
        if level == 0:
            return zipcode
        else:
            return zipcode[:5-level] + "*" * level

    def check_k_anonymity(self, df, quasi_identifiers):
        """Check if dataset satisfies k-anonymity"""
        group_sizes = df.groupBy(quasi_identifiers).count()
        min_group_size = group_sizes.agg({"count": "min"}).collect()[0][0]
        return min_group_size >= self.k

    def anonymize(self, df, quasi_identifiers, generalizations):
        """Apply generalizations until k-anonymity is achieved"""
        current_df = df
        current_levels = {qi: 0 for qi in quasi_identifiers}

        while not self.check_k_anonymity(current_df, quasi_identifiers):
            # Find quasi-identifier with most distinct values
            most_distinct = None
            max_distinct = 0

            for qi in quasi_identifiers:
                distinct_count = current_df.select(qi).distinct().count()
                if distinct_count > max_distinct:
                    max_distinct = distinct_count
                    most_distinct = qi

            # Increase generalization level
            if most_distinct and current_levels[most_distinct] < 3:
                current_levels[most_distinct] += 1
                gen_func = generalizations[most_distinct]
                current_df = current_df.withColumn(
                    most_distinct,
                    gen_func(col(most_distinct), current_levels[most_distinct])
                )
            else:
                # Cannot achieve k-anonymity, suppress small groups
                current_df = self.suppress_small_groups(
                    current_df, quasi_identifiers
                )
                break

        return current_df

    def check_l_diversity(self, df, quasi_identifiers, sensitive_attr, l=3):
        """Check if dataset satisfies l-diversity"""
        diversity = df.groupBy(quasi_identifiers).agg(
            size(collect_set(sensitive_attr)).alias("diversity")
        )
        min_diversity = diversity.agg({"diversity": "min"}).collect()[0][0]
        return min_diversity >= l

# Usage
anonymizer = KAnonymizer(spark, k=5)
anonymized_df = anonymizer.anonymize(
    patient_df,
    quasi_identifiers=["age", "zipcode", "gender"],
    generalizations={
        "age": anonymizer.generalize_age,
        "zipcode": anonymizer.generalize_zipcode
    }
)

Data Masking Techniques

Masking Methods

  • Substitution: Replace with realistic fake values
  • Shuffling: Randomly reorder values within a column
  • Encryption: Format-preserving encryption
  • Tokenization: Replace with non-reversible tokens
  • Nulling: Replace with NULL values
  • Character Masking: Partially obscure values

Federated Learning

Federated learning enables training machine learning models across distributed datasets without centralizing the data. This approach preserves privacy by keeping data on local devices or servers.

# Federated Learning Framework
import numpy as np
from typing import List, Dict

class FederatedLearningServer:
    def __init__(self, model_config):
        self.model_config = model_config
        self.global_model = self.initialize_model()
        self.round_number = 0

    def initialize_model(self):
        """Initialize global model weights"""
        return {
            "weights": np.random.randn(self.model_config["input_dim"],
                                       self.model_config["output_dim"]) * 0.01,
            "bias": np.zeros(self.model_config["output_dim"])
        }

    def federated_averaging(self, client_updates: List[Dict],
                           client_weights: List[float]):
        """Aggregate client updates using FedAvg"""
        total_weight = sum(client_weights)

        new_weights = np.zeros_like(self.global_model["weights"])
        new_bias = np.zeros_like(self.global_model["bias"])

        for update, weight in zip(client_updates, client_weights):
            new_weights += (weight / total_weight) * update["weights"]
            new_bias += (weight / total_weight) * update["bias"]

        self.global_model["weights"] = new_weights
        self.global_model["bias"] = new_bias
        self.round_number += 1

        return self.global_model

    def secure_aggregation(self, encrypted_updates: List[bytes]):
        """Aggregate updates without seeing individual contributions"""
        # Uses secure multi-party computation
        pass

class FederatedLearningClient:
    def __init__(self, client_id, local_data):
        self.client_id = client_id
        self.local_data = local_data
        self.local_model = None

    def receive_global_model(self, global_model):
        """Receive and apply global model weights"""
        self.local_model = {
            "weights": global_model["weights"].copy(),
            "bias": global_model["bias"].copy()
        }

    def local_training(self, epochs=5, learning_rate=0.01):
        """Train on local data"""
        X, y = self.local_data

        for epoch in range(epochs):
            # Forward pass
            predictions = X @ self.local_model["weights"] + self.local_model["bias"]

            # Compute gradients
            error = predictions - y
            grad_weights = X.T @ error / len(X)
            grad_bias = error.mean(axis=0)

            # Update weights
            self.local_model["weights"] -= learning_rate * grad_weights
            self.local_model["bias"] -= learning_rate * grad_bias

        return self.local_model

    def compute_update_with_dp(self, epsilon=1.0):
        """Compute model update with differential privacy"""
        update = self.local_training()

        # Clip gradients
        clip_norm = 1.0
        weights_norm = np.linalg.norm(update["weights"])
        if weights_norm > clip_norm:
            update["weights"] = update["weights"] * clip_norm / weights_norm

        # Add noise
        noise_scale = clip_norm * np.sqrt(2 * np.log(1.25 / 1e-5)) / epsilon
        update["weights"] += np.random.normal(0, noise_scale,
                                               update["weights"].shape)

        return update

# Federated training loop
server = FederatedLearningServer({"input_dim": 100, "output_dim": 10})
clients = [FederatedLearningClient(i, data) for i, data in enumerate(client_data)]

for round in range(100):
    # Distribute global model
    for client in clients:
        client.receive_global_model(server.global_model)

    # Collect local updates
    updates = [client.compute_update_with_dp(epsilon=0.5) for client in clients]
    weights = [len(c.local_data[0]) for c in clients]

    # Aggregate
    server.federated_averaging(updates, weights)

Key Takeaways

  • Differential privacy provides mathematical privacy guarantees
  • K-anonymity and l-diversity protect against re-identification
  • Use appropriate masking techniques for different data types
  • Federated learning enables analytics without data centralization
  • Track and manage privacy budgets to prevent over-disclosure
