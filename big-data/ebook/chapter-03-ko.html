3장: 스트림 처리 프레임워크

학습 목표: 실시간 데이터 처리 패턴을 마스터하고, 스트림 처리의 의미론(at-least-once, exactly-once)을 이해하며, WIA-BIG_DATA 표준을 사용하여 프로덕션급 스트리밍 파이프라인을 구현합니다.

스트림 처리 소개

스트림 처리는 데이터가 도착하는 즉시 지속적으로 처리하여 실시간 분석, 모니터링 및 의사 결정을 가능하게 합니다. 바운디드 데이터셋에서 작동하는 배치 처리와 달리 스트림 처리는 최소한의 지연 시간으로 언바운디드, 지속적으로 흐르는 데이터를 처리합니다.

WIA-BIG_DATA 표준은 다양한 스트리밍 플랫폼 전반에서 신뢰성, 확장성 및 일관성을 보장하는 스트림 처리 구현에 대한 요구사항을 정의합니다.

처리 시맨틱스

전달 보장

  • At-most-once: 메시지 손실 가능, 절대 중복 안 됨 (가장 빠름)
  • At-least-once: 메시지 절대 손실 안 됨, 중복 가능 (신뢰성)
  • Exactly-once: 메시지가 정확히 한 번 처리 (가장 강력한 보장)

WIA-BIG_DATA 표준은 최소 보장으로 at-least-once 시맨틱스를 요구하며, 금융 및 중요 애플리케이션에는 exactly-once를 권장합니다.

Kafka 프로듀서 구현

// WIA-BIG_DATA 준수 Kafka 프로듀서
import org.apache.kafka.clients.producer.*;
import org.apache.kafka.common.serialization.StringSerializer;
import io.confluent.kafka.serializers.KafkaAvroSerializer;

public class WIAKafkaProducer {
    private final KafkaProducer<String, GenericRecord> producer;
    private final String topic;

    public WIAKafkaProducer(String bootstrapServers, String topic) {
        this.topic = topic;

        Properties props = new Properties();
        props.put(ProducerConfig.BOOTSTRAP_SERVERS_CONFIG, bootstrapServers);
        props.put(ProducerConfig.KEY_SERIALIZER_CLASS_CONFIG, StringSerializer.class);
        props.put(ProducerConfig.VALUE_SERIALIZER_CLASS_CONFIG, KafkaAvroSerializer.class);

        // Exactly-once 시맨틱스
        props.put(ProducerConfig.ENABLE_IDEMPOTENCE_CONFIG, true);
        props.put(ProducerConfig.ACKS_CONFIG, "all");
        props.put(ProducerConfig.RETRIES_CONFIG, Integer.MAX_VALUE);
        props.put(ProducerConfig.MAX_IN_FLIGHT_REQUESTS_PER_CONNECTION, 5);

        // 성능 튜닝
        props.put(ProducerConfig.COMPRESSION_TYPE_CONFIG, "lz4");
        props.put(ProducerConfig.BATCH_SIZE_CONFIG, 32768);
        props.put(ProducerConfig.LINGER_MS_CONFIG, 5);

        // 스키마 레지스트리
        props.put("schema.registry.url", "http://schema-registry:8081");

        this.producer = new KafkaProducer<>(props);
    }

    public void sendWithCallback(String key, GenericRecord value) {
        ProducerRecord<String, GenericRecord> record =
            new ProducerRecord<>(topic, key, value);

        producer.send(record, (metadata, exception) -> {
            if (exception != null) {
                // 실패 로깅 및 처리
                logger.error("메시지 전송 실패", exception);
                handleFailure(record, exception);
            } else {
                // 오프셋과 함께 성공 로깅
                logger.debug("파티션 {}에 오프셋 {}로 메시지 전송",
                    metadata.partition(), metadata.offset());
            }
        });
    }

    public void close() {
        producer.flush();
        producer.close();
    }
}

Apache Flink 통합

Apache Flink는 강력한 윈도잉, 상태 관리 및 exactly-once 보장을 갖춘 진정한 스트림 처리를 제공합니다. WIA-BIG_DATA 표준은 복합 이벤트 처리 및 상태 저장 연산에 Flink를 권장합니다.

// WIA-BIG_DATA Flink 스트리밍 애플리케이션
import org.apache.flink.streaming.api.scala._
import org.apache.flink.streaming.api.windowing.time.Time
import org.apache.flink.streaming.api.CheckpointingMode

object WIAStreamProcessor {
  def main(args: Array[String]): Unit = {
    val env = StreamExecutionEnvironment.getExecutionEnvironment

    // Exactly-once 체크포인팅
    env.enableCheckpointing(60000, CheckpointingMode.EXACTLY_ONCE)
    env.getCheckpointConfig.setMinPauseBetweenCheckpoints(30000)
    env.getCheckpointConfig.setCheckpointTimeout(120000)
    env.getCheckpointConfig.setMaxConcurrentCheckpoints(1)
    env.getCheckpointConfig.enableExternalizedCheckpoints(
      ExternalizedCheckpointCleanup.RETAIN_ON_CANCELLATION
    )

    // 상태 백엔드 구성
    env.setStateBackend(new RocksDBStateBackend("hdfs:///checkpoints", true))

    // 워터마크를 포함한 Kafka 소스
    val kafkaSource = KafkaSource.builder[Event]()
      .setBootstrapServers("kafka:9092")
      .setTopics("events")
      .setGroupId("wia-flink-consumer")
      .setStartingOffsets(OffsetsInitializer.committedOffsets())
      .setValueOnlyDeserializer(new EventDeserializer())
      .build()

    val eventStream = env.fromSource(
      kafkaSource,
      WatermarkStrategy
        .forBoundedOutOfOrderness[Event](Duration.ofSeconds(10))
        .withTimestampAssigner((event, _) => event.timestamp),
      "Kafka Events"
    )

    // 윈도우 집계
    val aggregatedStream = eventStream
      .keyBy(_.userId)
      .window(TumblingEventTimeWindows.of(Time.minutes(5)))
      .allowedLateness(Time.minutes(1))
      .sideOutputLateData(lateDataTag)
      .aggregate(new EventAggregator(), new WindowResultFunction())

    // 여러 출력으로 싱크
    aggregatedStream.addSink(new KafkaSink("aggregated-events"))
    aggregatedStream.addSink(new ElasticsearchSink(esConfig))

    env.execute("WIA 스트림 프로세서")
  }
}

복합 이벤트 처리 (CEP)

CEP 패턴 매칭

복합 이벤트 처리는 이벤트 스트림에서 패턴 감지를 가능하게 하며, 이는 사기 탐지, 이상 식별 및 비즈니스 규칙 시행에 필수적입니다.

// 사기 탐지를 위한 Flink CEP 패턴
import org.apache.flink.cep.scala.CEP
import org.apache.flink.cep.scala.pattern.Pattern

// 의심스러운 활동 패턴 정의
val fraudPattern = Pattern.begin[Transaction]("first")
  .where(_.amount > 1000)
  .next("second")
  .where(_.amount > 1000)
  .within(Time.minutes(10))
  .next("third")
  .where(tx => tx.amount > 5000 && tx.country != tx.homeCountry)
  .within(Time.minutes(30))

// 스트림에 패턴 적용
val patternStream = CEP.pattern(transactionStream.keyBy(_.userId), fraudPattern)

val alerts = patternStream.select(matches => {
  val first = matches("first").head
  val second = matches("second").head
  val third = matches("third").head

  FraudAlert(
    userId = first.userId,
    transactions = List(first, second, third),
    totalAmount = first.amount + second.amount + third.amount,
    riskScore = calculateRisk(first, second, third),
    timestamp = System.currentTimeMillis()
  )
})

alerts.addSink(new AlertNotificationSink())
