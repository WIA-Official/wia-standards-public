## 샘플 구현

### Python 서버 구현

```python
# app.py
from fastapi import FastAPI, HTTPException, Depends
from pydantic import BaseModel
from typing import List, Optional
import uuid

app = FastAPI(title="WIA-LANG-001 Server")

# 모델 정의
class Speaker(BaseModel):
    id: str
    age: Optional[int]
    gender: Optional[str]
    native_status: bool
    consent_given: bool

class RecordingCreate(BaseModel):
    language: str
    type: str
    speaker: Speaker
    duration: float

class Recording(RecordingCreate):
    id: str
    status: str
    created_at: str

# 데이터 저장소 (실제 구현에서는 데이터베이스 사용)
recordings_db = {}

# 엔드포인트
@app.post("/v1/languages/{language}/recordings", response_model=Recording)
async def create_recording(language: str, recording: RecordingCreate):
    """새 녹음 생성"""
    if language != recording.language:
        raise HTTPException(400, "Language mismatch")

    if not recording.speaker.consent_given:
        raise HTTPException(400, "Speaker consent required")

    rec_id = f"rec-{uuid.uuid4().hex[:8]}"
    new_recording = Recording(
        id=rec_id,
        status="created",
        created_at="2025-01-01T00:00:00Z",
        **recording.dict()
    )
    recordings_db[rec_id] = new_recording
    return new_recording

@app.get("/v1/languages/{language}/recordings", response_model=List[Recording])
async def list_recordings(
    language: str,
    limit: int = 20,
    offset: int = 0
):
    """녹음 목록 조회"""
    filtered = [r for r in recordings_db.values() if r.language == language]
    return filtered[offset:offset+limit]

@app.get("/v1/recordings/{recording_id}", response_model=Recording)
async def get_recording(recording_id: str):
    """단일 녹음 조회"""
    if recording_id not in recordings_db:
        raise HTTPException(404, "Recording not found")
    return recordings_db[recording_id]

# 헬스 체크
@app.get("/health")
async def health_check():
    return {"status": "healthy", "version": "1.0.0"}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

import { WIALangClient, Recording, Transcription } from '@wia/lang-sdk';

// 클라이언트 초기화
const client = new WIALangClient({
  apiKey: process.env.WIA_API_KEY!,
  baseUrl: 'https://api.wia-lang.org/v1'

// 언어 정보 조회
async function getLanguageInfo() {
  const language = await client.languages.get('ain');
  console.log(`언어: ${language.name.english}`);
  console.log(`상태: ${language.endangerment.status}`);
  console.log(`화자 수: ${language.speakers.native}`);
}

// 녹음 업로드
async function uploadRecording(filePath: string) {
  const recording = await client.recordings.create({
    language: 'ain',
    file: fs.createReadStream(filePath),
    metadata: {
      type: 'narrative',
      speaker: {
        id: 'spk-001',
        age: 82,
        nativeStatus: true,
        consent: { given: true }
      },
      session: {
        date: new Date().toISOString(),
        location: 'Nibutani, Hokkaido'
      },
      content: {
        genre: 'personal_narrative',
        topic: 'traditional_fishing'
      }
    }
  });

  console.log(`녹음 생성됨: ${recording.id}`);
  return recording;
}

// 전사 생성
async function createTranscription(recordingId: string) {
  const transcription = await client.transcriptions.create({
    recordingId,
    tiers: [
      { name: 'utterance', language: 'ain' },
      { name: 'translation', language: 'ko' }
    ],
    segments: [
      {
        start: 0.0,
        end: 3.5,
        data: {
          utterance: 'ウコイキ オルンオシキ イタㇰ アン',
          translation: '당신에게 말할 언어가 있습니다'
        }
      }
    ]
  });

  console.log(`전사 생성됨: ${transcription.id}`);
  return transcription;
}

// 실행
async function main() {
  await getLanguageInfo();
  const recording = await uploadRecording('./story.flac');
  const transcription = await createTranscription(recording.id);
}

main().catch(console.error);

### 데이터 검증 예제

```python
from wia_lang.validation import Validator, ValidationResult

validator = Validator(
    schema_version='1.0',
    strict_mode=True
)

# 메타데이터 검증
metadata = {
    "language": {
        "name": {"english": "Ainu"},
        "iso639-3": "ain",
        "endangerment": {
            "status": "critically_endangered"
        },
        "speakers": {
            "native": 10
        }
    }
}

result: ValidationResult = validator.validate_metadata(metadata)

if result.is_valid:
    print("✓ 메타데이터 유효")
else:
    for error in result.errors:
        print(f"✗ {error.path}: {error.message}")

# 오디오 파일 검증
audio_result = validator.validate_audio('recording.flac')
print(f"오디오 유효: {audio_result.is_valid}")
print(f"형식: {audio_result.details['format']}")
print(f"샘플레이트: {audio_result.details['sample_rate']}")

# 전체 프로젝트 검증
project_result = validator.validate_project('./my-language-project/')
print(f"프로젝트 점수: {project_result.score}/100")
for issue in project_result.issues:
    print(f"  - [{issue.severity}] {issue.message}")
```

@pytest.fixture

    with MockServer() as server:

            api_key='test-key',
            base_url=server.url
        )

class TestRecordings:
    async def test_create_recording(self, client):
        """녹음 생성 테스트"""
        recording = await client.recordings.create(
            language='ain',
            file=open('test.flac', 'rb'),
            metadata={'type': 'narrative'}
        )

        assert recording.id is not None
        assert recording.language == 'ain'
        assert recording.status == 'created'

    async def test_create_recording_without_consent_fails(self, client):
        """동의 없는 녹음 생성 실패 테스트"""
        with pytest.raises(ValidationError) as exc:
            await client.recordings.create(
                language='ain',
                metadata={
                    'speaker': {'consent': {'given': False}}
                }
            )

        assert 'consent' in str(exc.value)

class TestTranscriptions:
    async def test_create_transcription(self, client):
        """전사 생성 테스트"""
        transcription = await client.transcriptions.create(
            recording_id='rec-001',
            segments=[
                {'start': 0, 'end': 1, 'data': {'text': 'test'}}
            ]
        )

        assert transcription.id is not None
        assert len(transcription.segments) == 1

from locust import HttpUser, task, between

class WIALangUser(HttpUser):
    wait_time = between(1, 3)

    def on_start(self):
        # 인증
        response = self.client.post('/auth/token', json={
            'client_id': 'load-test',
            'client_secret': 'secret'

    @task(3)
