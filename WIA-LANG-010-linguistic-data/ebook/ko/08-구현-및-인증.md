## 4. 구현 예시

### 4.1 최소 구현 예시 (Basic Compliance)

```python
# 최소 요구사항을 충족하는 구현 예시
from dataclasses import dataclass, field
from typing import List, Optional
from datetime import datetime
import xml.etree.ElementTree as ET

@dataclass
class Metadata:
    language: str  # ISO 639-3
    title: str
    creator: str
    license: str
    created: datetime = field(default_factory=datetime.now)

@dataclass
class Token:
    id: str
    form: str
    start: int
    end: int
    lemma: Optional[str] = None
    pos: Optional[str] = None  # Universal POS

@dataclass
class Sentence:
    id: str
    text: str
    tokens: List[Token] = field(default_factory=list)

@dataclass
class Document:
    id: str
    sentences: List[Sentence] = field(default_factory=list)

@dataclass
class Corpus:
    metadata: Metadata
    documents: List[Document] = field(default_factory=list)

    def to_xml(self) -> str:
        """XML 직렬화"""
        root = ET.Element("linguisticData", {
            "xmlns": "http://wia.org/lang-010/v1.0",
            "version": "1.0.0"
        })

        # 메타데이터
        meta = ET.SubElement(root, "metadata")
        ET.SubElement(meta, "language", {"iso639-3": self.metadata.language})
        ET.SubElement(meta, "title").text = self.metadata.title
        ET.SubElement(meta, "creator").text = self.metadata.creator
        ET.SubElement(meta, "license").text = self.metadata.license
        ET.SubElement(meta, "created").text = self.metadata.created.isoformat()

        # 코퍼스
        corpus_elem = ET.SubElement(root, "corpus")
        for document in self.documents:
            doc_elem = ET.SubElement(corpus_elem, "document", {"id": document.id})
            for sentence in document.sentences:
                sent_elem = ET.SubElement(doc_elem, "sentence", {"id": sentence.id})
                ET.SubElement(sent_elem, "text").text = sentence.text

                tokens_elem = ET.SubElement(sent_elem, "tokens")
                for token in sentence.tokens:
                    token_attrs = {
                        "id": token.id,
                        "form": token.form,
                        "start": str(token.start),
                        "end": str(token.end)
                    }
                    if token.lemma:
                        token_attrs["lemma"] = token.lemma
                    if token.pos:
                        token_attrs["pos"] = token.pos
                    ET.SubElement(tokens_elem, "token", token_attrs)

        return ET.tostring(root, encoding="unicode", xml_declaration=True)

    @classmethod
    def from_xml(cls, xml_string: str) -> 'Corpus':
        """XML 역직렬화"""
        root = ET.fromstring(xml_string)

        # 메타데이터 파싱
        meta_elem = root.find(".//{http://wia.org/lang-010/v1.0}metadata")
        metadata = Metadata(
            language=meta_elem.find("language").get("iso639-3"),
            title=meta_elem.find("title").text,
            creator=meta_elem.find("creator").text,
            license=meta_elem.find("license").text,
            created=datetime.fromisoformat(meta_elem.find("created").text)
        )

        # 문서 파싱
        documents = []
        for doc_elem in root.findall(".//document"):
            sentences = []
            for sent_elem in doc_elem.findall("sentence"):
                tokens = []
                for token_elem in sent_elem.findall(".//token"):
                    token = Token(
                        id=token_elem.get("id"),
                        form=token_elem.get("form"),
                        start=int(token_elem.get("start")),
                        end=int(token_elem.get("end")),
                        lemma=token_elem.get("lemma"),
                        pos=token_elem.get("pos")
                    )
                    tokens.append(token)

                sentence = Sentence(
                    id=sent_elem.get("id"),
                    text=sent_elem.find("text").text,
                    tokens=tokens
                )
                sentences.append(sentence)

            document = Document(id=doc_elem.get("id"), sentences=sentences)
            documents.append(document)

        return cls(metadata=metadata, documents=documents)
```

### 4.2 고급 구현 예시 (Advanced Compliance)

```python
# 완전한 기능을 갖춘 고급 구현
from wia_lang_010 import (
    LinguisticData,
    Corpus,
    Document,
    Sentence,
    Token,
    Entity,
    DependencyRelation,
    SemanticRole
)
from wia_lang_010.api import WIALanguageAPI
from wia_lang_010.validation import Validator
from wia_lang_010.storage import PostgreSQLStorage
from wia_lang_010.streaming import StreamingProcessor
from wia_lang_010.integration import SpacyIntegration, TransformersIntegration

class AdvancedWIAImplementation:
    def __init__(self, config):
        self.config = config
        self.storage = PostgreSQLStorage(config['database'])
        self.validator = Validator(config['validation'])
        self.api = WIALanguageAPI(config['api'])

        # 통합 모듈
        self.spacy_integration = SpacyIntegration()
        self.transformers_integration = TransformersIntegration()

    def create_corpus_from_text(self, texts, language, auto_annotate=True):
        """텍스트에서 코퍼스 생성 (자동 주석 포함)"""
        corpus = LinguisticData(
            metadata={
                'language': language,
                'title': 'Auto-generated Corpus',
                'creator': 'Advanced WIA System',
                'license': 'CC-BY-4.0'
            }
        ).create_corpus()

        for i, text in enumerate(texts):
            doc = corpus.add_document(id=f"doc{i:05d}")

            if auto_annotate:
                # spaCy로 자동 주석
                annotated_sentences = self.spacy_integration.process(text, language)

                for sent in annotated_sentences:
                    doc.add_sentence(sent)
            else:
                # 수동 처리
                sent = doc.add_sentence(text=text)

        # 검증
        validation_result = self.validator.validate(corpus)
        if not validation_result.is_valid:
            raise ValueError(f"Validation failed: {validation_result.errors}")

        # 저장
        self.storage.save(corpus)

        return corpus

    def enhance_with_transformers(self, corpus):
        """Transformers 모델로 코퍼스 강화"""
        for document in corpus.documents:
            for sentence in document.sentences:
                # NER 추가
                entities = self.transformers_integration.extract_entities(sentence.text)
                for entity in entities:
                    sentence.add_entity(entity)

                # 감성 분석 추가
                sentiment = self.transformers_integration.analyze_sentiment(sentence.text)
                sentence.add_annotation('sentiment', sentiment)

        return corpus

    def export_multiple_formats(self, corpus, output_dir):
        """다양한 형식으로 내보내기"""
        corpus.save(f"{output_dir}/corpus.xml", format='xml')
        corpus.save(f"{output_dir}/corpus.json", format='json')
        corpus.save(f"{output_dir}/corpus.conllu", format='conllu')

        # 추가 형식
        corpus.to_dataframe().to_csv(f"{output_dir}/corpus.csv")
        corpus.to_brat(f"{output_dir}/brat/")

    def stream_process_large_corpus(self, input_path, processor_func):
        """대용량 코퍼스 스트리밍 처리"""
        streaming = StreamingProcessor(input_path)

        for batch in streaming.iter_batches(batch_size=1000):
            processed = processor_func(batch)
            self.storage.save_batch(processed)

    def serve_api(self, host='0.0.0.0', port=8000):
        """API 서버 시작"""
        self.api.add_corpus_endpoint(self.storage)
        self.api.add_search_endpoint(self.storage)
        self.api.add_annotation_endpoint(
            self.spacy_integration,
            self.transformers_integration
        )

        self.api.run(host=host, port=port)

# 사용 예시
config = {
    'database': {
        'host': 'localhost',
        'port': 5432,
        'database': 'wia_lang_010',
        'user': 'user',
        'password': 'password'
    },
    'validation': {
        'level': 'strict',
        'schema': 'wia-lang-010-v1.0.xsd'
    },
    'api': {
        'version': '1.0',
        'auth': 'jwt'
    }
}

system = AdvancedWIAImplementation(config)

# 코퍼스 생성
texts = ["문장 1", "문장 2", "문장 3"]
corpus = system.create_corpus_from_text(
    texts,
    language='kor',
    auto_annotate=True
)

# Transformers로 강화
enhanced = system.enhance_with_transformers(corpus)

# 다양한 형식으로 내보내기
system.export_multiple_formats(enhanced, "./output")

# API 서버 시작
system.serve_api()
```
