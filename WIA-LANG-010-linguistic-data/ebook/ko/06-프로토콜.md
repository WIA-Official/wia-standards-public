# 제6장: 프로토콜

## 개요

> **弘益人間 (홍익인간)** - 널리 인간을 이롭게 하라

언어 데이터의 효율적인 교환과 동기화를 위해서는 표준화된 프로토콜이 필수적입니다. 이 장에서는 WIA-LANG-010 표준이 정의하는 다양한 통신 프로토콜, 데이터 전송 메커니즘, 그리고 실시간 협업 프로토콜을 다룹니다.

```protobuf
message Metadata {

  string iso639_3 = 1;
  string name = 2;

message SentenceFilter {
```

```python
def upload_chunks(file_path, upload_id):
    def chunk_generator():
        with open(file_path, 'rb') as f:
            chunk_id = 0
            while True:
                data = f.read(1024 * 1024)  # 1MB chunks
                if not data:
                    break
                yield CorpusChunk(
                    upload_id=upload_id,
                    chunk_id=chunk_id,
                    data=data,
                    checksum=calculate_checksum(data)
                )
                chunk_id += 1
```

### 3.2 Operational Transformation (OT)

```typescript
// 작업 변환 (Operational Transformation)
interface Operation {
  type: 'insert' | 'delete' | 'update';
  path: string;
  position?: number;
  value?: any;
  oldValue?: any;
}

class OperationalTransform {
  // 두 작업을 변환
  transform(op1: Operation, op2: Operation): [Operation, Operation] {
    if (op1.path !== op2.path) {
      // 다른 경로: 충돌 없음
      return [op1, op2];
    }

    // 같은 경로에서의 충돌 해결
    if (op1.type === 'insert' && op2.type === 'insert') {
      if (op1.position <= op2.position) {
        // op2의 위치를 op1 삽입만큼 이동
        const op2Prime = {
          ...op2,
          position: op2.position + 1
        };
        return [op1, op2Prime];
      } else {
        const op1Prime = {
          ...op1,
          position: op1.position + 1
        };
        return [op1Prime, op2];
      }
    }

    if (op1.type === 'delete' && op2.type === 'delete') {
      // 같은 위치 삭제: 충돌
      if (op1.position === op2.position) {
        // op2를 무효화 (이미 삭제됨)
        return [op1, null];
      }
    }

    if (op1.type === 'update' && op2.type === 'update') {
      // 같은 필드 업데이트: Last-Write-Wins
      if (op1.timestamp > op2.timestamp) {
        return [op1, null];
      } else {
        return [null, op2];
      }
    }

    return [op1, op2];
  }

  // 작업 목록 적용
  apply(document: any, operations: Operation[]): any {
    let result = JSON.parse(JSON.stringify(document));

    for (const op of operations) {
      if (!op) continue;

      switch (op.type) {
        case 'insert':
          this.applyInsert(result, op);
          break;
        case 'delete':
          this.applyDelete(result, op);
          break;
        case 'update':
          this.applyUpdate(result, op);
          break;
      }
    }

    return result;
  }
}
```

### 3.3 Conflict Resolution (충돌 해결)

```python
from wia_lang_010.sync import ConflictResolver

resolver = ConflictResolver(strategy='three_way_merge')

# 세 가지 버전
base_version = corpus_version_40  # 공통 조상
client_a_version = corpus_version_41  # 클라이언트 A의 변경
client_b_version = corpus_version_42  # 클라이언트 B의 변경

# 자동 병합 시도
result = resolver.merge(
    base=base_version,
    left=client_a_version,
    right=client_b_version
)

if result.success:
    # 병합 성공
    merged_corpus = result.corpus
    print(f"Merged successfully. New version: {result.version}")
else:
    # 충돌 발생
    for conflict in result.conflicts:
        print(f"Conflict at: {conflict.path}")
        print(f"  Base value: {conflict.base_value}")
        print(f"  Client A: {conflict.left_value}")
        print(f"  Client B: {conflict.right_value}")

        # 수동 해결
        resolution = prompt_user(conflict)
        resolver.resolve(conflict, resolution)

    # 충돌 해결 후 재병합
    merged_corpus = resolver.apply_resolutions()

# 병합 전략
strategies = {
```

### 6.3 재시도 전략

```typescript
class RetryStrategy {
  async executeWithRetry<T>(
    operation: () => Promise<T>,
    options: RetryOptions = {}
  ): Promise<T> {
    const {
      maxRetries = 3,
      initialDelay = 1000,
      maxDelay = 10000,
      backoffMultiplier = 2,
      retryableErrors = [408, 429, 500, 502, 503, 504]
    } = options;

    let attempt = 0;
    let delay = initialDelay;

    while (attempt < maxRetries) {
      try {
        return await operation();
      } catch (error) {
        attempt++;

        if (
          !this.isRetryable(error, retryableErrors) ||
          attempt >= maxRetries
        ) {
          throw error;
        }

        // Exponential backoff with jitter
        const jitter = Math.random() * 0.1 * delay;
        await this.sleep(delay + jitter);
        delay = Math.min(delay * backoffMultiplier, maxDelay);

        console.log(
          `Retry attempt ${attempt}/${maxRetries} after ${delay}ms`
        );
      }
    }
  }

  private isRetryable(error: any, retryableCodes: number[]): boolean {
    return retryableCodes.includes(error.statusCode);
  }

  private sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```
