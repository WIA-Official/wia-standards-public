```python
restored_doc = converter.to_spacy(wia_sentence, nlp.vocab)

# spaCy 파이프라인에 WIA-LANG-010 저장 컴포넌트 추가
@spacy.component("wia_saver")
class WIASaver:
    def __init__(self, output_path):
        self.output_path = output_path
        self.corpus = LinguisticData().create_corpus()

    def __call__(self, doc):
        sentence = SpacyConverter().from_spacy(doc)
        self.corpus.add_sentence(sentence)
        return doc

    def finalize(self):
        self.corpus.save(self.output_path)

# 파이프라인 구성
nlp.add_pipe("wia_saver", config={"output_path": "output.xml"}, last=True)

# 배치 처리
texts = ["문장 1", "문장 2", "문장 3"]
for doc in nlp.pipe(texts):
    pass

# 저장
nlp.get_pipe("wia_saver").finalize()
```

### 1.2 Hugging Face Transformers 통합

```python
from transformers import AutoTokenizer, AutoModelForTokenClassification
from wia_lang_010 import Sentence, Token
from wia_lang_010.transformers import TransformersIntegration

# 모델 로드
model_name = "xlm-roberta-base"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForTokenClassification.from_pretrained(model_name)

# WIA-LANG-010 통합
integration = TransformersIntegration(tokenizer, model)

# 문장 처리
text = "서울에서 김철수가 삼성전자에 입사했다."
sentence = Sentence(text=text)

# 토큰화 (WordPiece/BPE)
tokenized = tokenizer(text, return_tensors="pt", return_offsets_mapping=True)

# 모델 예측 (NER)
outputs = model(**tokenized)
predictions = outputs.logits.argmax(-1)

# WIA-LANG-010 형식으로 변환
entities = integration.extract_entities(
    text=text,
    predictions=predictions,
    offset_mapping=tokenized["offset_mapping"][0]
)

for entity in entities:
    sentence.add_entity(
        type=entity.type,
        start=entity.start,
        end=entity.end,
        text=entity.text,
        confidence=entity.confidence
    )

# 사용자 정의 모델 파인튜닝
from wia_lang_010.transformers import WIADataset, WIADataCollator

# WIA-LANG-010 코퍼스를 Hugging Face Dataset으로 변환
corpus = LinguisticData.load("training_corpus.xml")
dataset = WIADataset.from_corpus(corpus, task="ner")

# 데이터 로더
from torch.utils.data import DataLoader

data_collator = WIADataCollator(tokenizer, task="ner")
train_loader = DataLoader(
    dataset,
    batch_size=16,
    collate_fn=data_collator
)

# 학습
from transformers import Trainer, TrainingArguments

training_args = TrainingArguments(
    output_dir="./results",
    num_train_epochs=3,
    per_device_train_batch_size=16,
    save_steps=1000,
    logging_steps=100
)

trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=dataset,
    data_collator=data_collator
)

trainer.train()

# 학습된 모델로 예측 후 WIA-LANG-010 저장
predictions = trainer.predict(dataset)
output_corpus = integration.create_corpus_from_predictions(
    corpus=corpus,
    predictions=predictions,
    label_map=model.config.id2label
)
output_corpus.save("predictions.xml")
```

```python
# Prodigy 주석 도구와 통합
from wia_lang_010.prodigy import ProdigyRecipe
import prodigy

@prodigy.recipe("wia.ner.manual")
def wia_ner_manual(dataset, source_path):
    """
    WIA-LANG-010 코퍼스에 NER 주석 추가
    """
    # WIA-LANG-010 코퍼스 로드
    corpus = LinguisticData.load(source_path)

    # Prodigy 스트림 생성
    stream = []
    for document in corpus.documents:
        for sentence in document.sentences:
            stream.append({
                "text": sentence.text,
                "meta": {
                    "doc_id": document.id,
                    "sent_id": sentence.id
                }
            })

    return {
        "dataset": dataset,
        "view_id": "ner_manual",
        "stream": stream,
        "update": update_wia_corpus
    }

def update_wia_corpus(answers):
    """주석 결과를 WIA-LANG-010으로 저장"""
    corpus = LinguisticData.load(source_path)

    for answer in answers:
        if answer["answer"] == "accept":
            sent_id = answer["meta"]["sent_id"]
            sentence = corpus.find_sentence(sent_id)

            for span in answer.get("spans", []):
                sentence.add_entity(
                    type=span["label"],
                    start=span["start"],
                    end=span["end"],
                    text=span["text"],
                    confidence=1.0
                )

    corpus.save("prodigy_annotated.xml")

# 실행: prodigy wia.ner.manual korean_ner corpus.xml -F integration.py
```

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.model_selection import train_test_split
from wia_lang_010.ml import SKLearnIntegration

# WIA-LANG-010 코퍼스 로드
corpus = LinguisticData.load("corpus.xml")
```

### 4.2 PyTorch 통합

```python
import torch
from torch.utils.data import Dataset, DataLoader
from wia_lang_010.ml import WIADataset, WIADataLoader

# WIA-LANG-010 코퍼스를 PyTorch Dataset으로 변환
class NERDataset(WIADataset):
    def __init__(self, corpus_path, tokenizer, label_map):
        super().__init__(corpus_path)
        self.tokenizer = tokenizer
        self.label_map = label_map

    def __getitem__(self, idx):
        sentence = self.corpus[idx]

        # 토큰화
        tokens = [token.form for token in sentence.tokens]
        encodings = self.tokenizer(
            tokens,
            is_split_into_words=True,
            padding='max_length',
            truncation=True,
            max_length=128,
            return_tensors='pt'
        )

        # 레이블
        labels = [self.label_map[token.entity_type or 'O'] for token in sentence.tokens]
        labels = torch.tensor(labels)

        return {
            'input_ids': encodings['input_ids'].squeeze(),
            'attention_mask': encodings['attention_mask'].squeeze(),
            'labels': labels
        }

# 데이터 로더
dataset = NERDataset(
    corpus_path="corpus.xml",
    tokenizer=tokenizer,
    label_map={'O': 0, 'B-PER': 1, 'I-PER': 2, 'B-ORG': 3, 'I-ORG': 4}
)

dataloader = DataLoader(dataset, batch_size=16, shuffle=True)

# 모델 정의
from transformers import BertForTokenClassification

model = BertForTokenClassification.from_pretrained(
    'bert-base-multilingual-cased',
    num_labels=len(dataset.label_map)
)

# 학습
optimizer = torch.optim.AdamW(model.parameters(), lr=5e-5)

model.train()
for epoch in range(3):
    for batch in dataloader:
        optimizer.zero_grad()

        outputs = model(
            input_ids=batch['input_ids'],
            attention_mask=batch['attention_mask'],
            labels=batch['labels']
        )

        loss = outputs.loss
        loss.backward()
        optimizer.step()

    print(f"Epoch {epoch + 1} completed")

# 예측 및 WIA-LANG-010으로 저장
model.eval()
predictions = []

with torch.no_grad():
    for batch in dataloader:
        outputs = model(
            input_ids=batch['input_ids'],
            attention_mask=batch['attention_mask']
        )
        preds = torch.argmax(outputs.logits, dim=2)
        predictions.extend(preds.cpu().numpy())

# 예측 결과를 코퍼스에 추가
predicted_corpus = dataset.add_predictions(predictions)
predicted_corpus.save("predicted.xml")
```

```python
from airflow import DAG
from airflow.operators.python import PythonOperator
from datetime import datetime, timedelta
from wia_lang_010 import LinguisticData

default_args = {
    'owner': 'linguistics-team',
    'depends_on_past': False,

    'email_on_retry': False,
    'retries': 3,
    'retry_delay': timedelta(minutes=5)
}

dag = DAG(
    'corpus_processing_pipeline',
    default_args=default_args,
    description='WIA-LANG-010 corpus processing pipeline',
    schedule_interval=timedelta(days=1)
)

def extract_raw_data(**context):
    """원시 데이터 추출"""
    # 데이터 소스에서 텍스트 추출
    texts = fetch_texts_from_source()
    context['ti'].xcom_push(key='raw_texts', value=texts)

def create_corpus(**context):
    """코퍼스 생성"""
    texts = context['ti'].xcom_pull(key='raw_texts')

    corpus = LinguisticData().create_corpus()
    for text in texts:
        doc = corpus.add_document()
        sentence = doc.add_sentence(text=text)

    corpus.save("/data/corpora/daily_corpus.xml")

def annotate_corpus(**context):
    """자동 주석"""
    corpus = LinguisticData.load("/data/corpora/daily_corpus.xml")

    # POS 태깅, 구문 분석 등
    annotated = run_nlp_pipeline(corpus)
    annotated.save("/data/corpora/daily_corpus_annotated.xml")

def validate_corpus(**context):
    """검증"""
    corpus = LinguisticData.load("/data/corpora/daily_corpus_annotated.xml")

    validator = Validator()
    result = validator.validate(corpus)

    if not result.is_valid:
        raise ValueError(f"Validation failed: {result.errors}")

def export_to_database(**context):
    """데이터베이스로 내보내기"""
    corpus = LinguisticData.load("/data/corpora/daily_corpus_annotated.xml")

    adapter = PostgreSQLAdapter()
    adapter.save_corpus(corpus)

# Task 정의
t1 = PythonOperator(
    task_id='extract_raw_data',
    python_callable=extract_raw_data,

)
```
