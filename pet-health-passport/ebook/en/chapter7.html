WIA Pet Health Passport Standard - Emergency Response & International Healthcare

7.1 Emergency Health Information Access

Emergency veterinary situations require immediate access to critical pet health information including current medications, chronic conditions, allergies, recent treatments, and emergency contact details. The WIA Pet Health Passport Standard defines emergency data access protocols enabling rapid retrieval of life-saving information while maintaining privacy and security safeguards. Emergency access systems must function reliably under adverse conditions including network outages, after-hours scenarios, and international locations.

Traditional paper records often prove unavailable during emergencies when pets are found injured, become ill while traveling, or require after-hours emergency care at unfamiliar facilities. Digital health passports with offline emergency capabilities, QR code access, and international emergency networks ensure critical health data remains accessible regardless of circumstances, significantly improving emergency treatment outcomes and reducing medical errors.

Example 1: Emergency Data Access (TypeScript)

interface EmergencyDataPackage {
  petId: string;
  microchipNumber: string;
  criticalInfo: {
    petName: string;
    species: string;
    breed: string;
    age: string;
    weight: { value: number; unit: string; date: string };
  };
  criticalAllergies: Array<{
    allergen: string;
    severity: string;
    reaction: string;
  }>;
  currentMedications: Array<{
    medication: string;
    dosage: string;
    frequency: string;
    criticalForLife: boolean;
  }>;
  emergencyContacts: Array<{
    name: string;
    phone: string;
    priority: number;
  }>;
}

class EmergencyAccessService {
  private apiKey: string;
  private baseUrl = 'https://api.wia-pet-health.org/v1';
  private offlineCache: Map = new Map();

  constructor(apiKey: string) {
    this.apiKey = apiKey;
    this.initializeOfflineCache();
  }

  async getEmergencyData(microchipNumber: string): Promise {
    try {
      // Try online access first
      return await this.fetchOnlineEmergencyData(microchipNumber);
    } catch (error) {
      console.warn('Online access failed, using offline cache:', error);
      return this.getOfflineEmergencyData(microchipNumber);
    }
  }

  private async fetchOnlineEmergencyData(microchip: string): Promise {
    const response = await fetch(
      `${this.baseUrl}/emergency/pets/${microchip}`,
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'X-Emergency-Access': 'true',
          'X-WIA-Version': '1.0'
        },
        // Short timeout for emergency scenarios
        signal: AbortSignal.timeout(5000)
      }
    );

    if (!response.ok) {
      throw new Error(`Emergency data fetch failed: ${response.status}`);
    }

    const data = await response.json();

    // Update offline cache
    this.updateOfflineCache(microchip, data);

    // Log emergency access
    this.logEmergencyAccess(microchip, 'microchip_scan');

    return data;
  }

  private getOfflineEmergencyData(microchip: string): EmergencyDataPackage {
    const cached = this.offlineCache.get(microchip);

    if (!cached) {
      throw new Error('No offline emergency data available for this pet');
    }

    // Check cache age
    const cacheAge = Date.now() - new Date(cached.lastUpdated).getTime();
    const maxCacheAge = 7 * 24 * 60 * 60 * 1000; // 7 days

    if (cacheAge > maxCacheAge) {
      console.warn('Offline cache is stale (> 7 days old)');
    }

    return cached;
  }

  async notifyEmergencyContacts(
    petId: string,
    emergencyType: string,
    facilityInfo: any
  ): Promise {
    await fetch(`${this.baseUrl}/emergency/notify/owner`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        petId: petId,
        emergencyType: emergencyType,
        message: `Your pet is receiving emergency care at ${facilityInfo.name}`,
        facilityName: facilityInfo.name,
        facilityPhone: facilityInfo.phone,
        facilityAddress: facilityInfo.address,
        timestamp: new Date().toISOString()
      })
    });
  }

  async logEmergencyAccess(microchip: string, accessMethod: string): Promise {
    // Log access for audit trail (fire-and-forget)
    fetch(`${this.baseUrl}/emergency/access/log`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        microchipNumber: microchip,
        accessMethod: accessMethod,
        facilityId: this.getFacilityId(),
        accessedBy: this.getUserId(),
        timestamp: new Date().toISOString(),
        reason: 'emergency_care'
      })
    }).catch(err => console.error('Failed to log access:', err));
  }

  private async initializeOfflineCache(): Promise {
    // Load cached emergency data from local storage
    try {
      const cached = localStorage.getItem('wia_emergency_cache');
      if (cached) {
        const data = JSON.parse(cached);
        this.offlineCache = new Map(Object.entries(data));
      }
    } catch (error) {
      console.error('Failed to load offline cache:', error);
    }
  }

  private updateOfflineCache(microchip: string, data: EmergencyDataPackage): void {
    this.offlineCache.set(microchip, data);

    // Persist to local storage
    try {
      const cacheObj = Object.fromEntries(this.offlineCache);
      localStorage.setItem('wia_emergency_cache', JSON.stringify(cacheObj));
    } catch (error) {
      console.error('Failed to update offline cache:', error);
    }
  }

  private getFacilityId(): string {
    return localStorage.getItem('facility_id') || 'unknown';
  }

  private getUserId(): string {
    return localStorage.getItem('user_id') || 'emergency_user';
  }
}

async function handleEmergencyPatient(microchipNumber: string) {

    // Check current medications
    console.log('Current Medications:');
    emergencyData.currentMedications.forEach(med => {

    // Notify owner
    await emergencyService.notifyEmergencyContacts(
      emergencyData.petId,
      'trauma',
      {
        name: 'Emergency Veterinary Hospital',
        phone: '555-0911',
        address: '123 Emergency Way'
      }
    );

  } catch (error) {
    console.error('Failed to retrieve emergency data:', error);
    alert('Unable to access pet health records. Proceed with standard emergency protocol.');
  }
}

Example 2: Cross-Border Emergency Network (Python)

import requests
from typing import Dict, List, Optional
from datetime import datetime
import googletrans

class InternationalEmergencyNetwork:
    """Handle emergency care across international boundaries"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://api.wia-pet-health.org/v1'
        self.translator = googletrans.Translator()

    def find_emergency_care(self, latitude: float, longitude: float,
                           species: str, country_code: str) -> List[Dict]:
        """
        Find emergency veterinary facilities near current location

        Args:
            latitude: Current latitude
            longitude: Current longitude
            species: Pet species (for specialty filtering)
            country_code: ISO country code

        Returns:
            List of emergency facilities with contact info and directions
        """
        response = requests.get(
            f'{self.base_url}/emergency/facilities/nearby',
            headers={'Authorization': f'Bearer {self.api_key}'},
            params={
                'lat': latitude,
                'lon': longitude,
                'radius': 50,  # km
                'species': species,
                'country': country_code,
                '24hour': True,
                'emergency_only': True
            }
        )

        facilities = response.json().get('facilities', [])

        # Enrich with real-time availability
        for facility in facilities:
            facility['availability'] = self.check_facility_availability(
                facility['facilityId']
            )
            facility['estimated_wait'] = self.estimate_wait_time(
                facility['facilityId']
            )

        # Sort by distance and availability
        facilities.sort(key=lambda f: (
            not f['availability']['accepting_emergencies'],
            f['distance_km']
        ))

        return facilities

    def get_international_emergency_protocol(self, country_code: str,
                                            emergency_type: str) -> Dict:
        """
        Retrieve country-specific emergency protocols and resources

        Args:
            country_code: ISO country code
            emergency_type: Type of emergency (trauma, poisoning, etc.)

        Returns:
            Protocol information including hotlines, procedures, translations
        """
        response = requests.get(
            f'{self.base_url}/emergency/international/{country_code}',
            headers={'Authorization': f'Bearer {self.api_key}'},
            params={'emergency_type': emergency_type}
        )

        protocol = response.json()

        # Add translations for key phrases
        protocol['translations'] = self.get_emergency_translations(
            country_code,
            ['My pet needs emergency care',
             'Severe bleeding',
             'Not breathing',
             'Poisoning',
             'Hit by car',
             'Where is nearest vet hospital?']
        )

        return protocol

    def initiate_cross_border_transfer(self, pet_id: str,
                                      origin_facility: str,
                                      destination_country: str) -> Dict:
        """
        Coordinate emergency medical transfer across border

        Args:
            pet_id: Pet health passport ID
            origin_facility: Current facility ID
            destination_country: Destination country code

        Returns:
            Transfer coordination details
        """
        # Get pet emergency data
        pet_data = self.get_emergency_summary(pet_id)

        # Check destination country emergency requirements
        dest_requirements = self.get_international_emergency_protocol(
            destination_country,
            'medical_transfer'
        )

        # Request expedited health certificate
        health_cert = self.request_emergency_health_certificate(
            pet_id,
            destination_country,
            reason='emergency_medical_transfer'
        )

        # Coordinate with receiving facility
        receiving_facility = self.find_emergency_care(
            dest_requirements['capital_lat'],
            dest_requirements['capital_lon'],
            pet_data['species'],
            destination_country
        )[0]  # Best available

        # Create transfer plan
        transfer_plan = {
            'transfer_id': generate_uuid(),
            'pet_id': pet_id,
            'origin_facility': origin_facility,
            'destination_facility': receiving_facility['facilityId'],
            'health_certificate': health_cert,
            'estimated_transfer_time': self.calculate_transfer_time(
                origin_facility,
                receiving_facility
            ),
            'special_requirements': dest_requirements['transfer_requirements'],
            'emergency_contacts': pet_data['emergency_contacts'],
            'status': 'coordinating'
        }

        # Notify all parties
        self.notify_transfer_parties(transfer_plan)

        return transfer_plan

    def check_facility_availability(self, facility_id: str) -> Dict:
        """Check real-time facility availability"""
        response = requests.get(
            f'{self.base_url}/emergency/facilities/{facility_id}/status',
            headers={'Authorization': f'Bearer {self.api_key}'}
        )
        return response.json()

    def estimate_wait_time(self, facility_id: str) -> str:
        """Estimate current wait time at facility"""
        # Would integrate with facility queue management
        return "15-30 minutes"

    def get_emergency_translations(self, country_code: str,
                                   phrases: List[str]) -> Dict:
        """Get emergency phrase translations for country"""
        # Map country code to primary language
        language_map = {
            'FR': 'fr', 'ES': 'es', 'DE': 'de', 'IT': 'it',
            'JP': 'ja', 'KR': 'ko', 'CN': 'zh-cn', 'RU': 'ru'
        }

        target_lang = language_map.get(country_code, 'en')

        translations = {}
        for phrase in phrases:
            try:
                translated = self.translator.translate(phrase, dest=target_lang)
                translations[phrase] = translated.text
            except:
                translations[phrase] = phrase  # Fallback to English

        return translations

    def get_emergency_summary(self, pet_id: str) -> Dict:
        """Get emergency summary for pet"""
        response = requests.get(
            f'{self.base_url}/emergency/pets/{pet_id}/summary',
            headers={'Authorization': f'Bearer {self.api_key}'}
        )
        return response.json()

    def request_emergency_health_certificate(self, pet_id: str,
                                             destination: str,
                                             reason: str) -> Dict:
        """Request expedited health certificate for emergency"""
        response = requests.post(
            f'{self.base_url}/emergency/health-certificate/expedited',
            headers={
                'Authorization': f'Bearer {self.api_key}',
                'Content-Type': 'application/json'
            },
            json={
                'pet_id': pet_id,
                'destination_country': destination,
                'reason': reason,
                'urgency': 'emergency',
                'requested_at': datetime.utcnow().isoformat()
            }
        )
        return response.json()

    def calculate_transfer_time(self, origin: str, destination: Dict) -> str:
        """Calculate estimated transfer time"""
        # Would use real routing/flight APIs
        return "6-8 hours"

    def notify_transfer_parties(self, transfer_plan: Dict):
        """Notify all parties involved in transfer"""
        requests.post(
            f'{self.base_url}/emergency/transfer/notify',
            headers={
                'Authorization': f'Bearer {self.api_key}',
                'Content-Type': 'application/json'
            },
            json=transfer_plan
        )

import java.util.*;
import java.time.LocalDateTime;

public class EmergencyTriageSystem {
    private final String apiKey;
    private final HttpClient httpClient;

    public EmergencyTriageSystem(String apiKey) {
        this.apiKey = apiKey;
        this.httpClient = HttpClient.newHttpClient();
    }

    public TriageAssessment performTriage(String petId, TriageData triageData) {
        // Calculate triage score
        int triageScore = calculateTriageScore(triageData);

        // Determine priority level
        PriorityLevel priority = determinePriorityLevel(triageScore);

        // Get recommended actions
        List recommendations = getTriageRecommendations(
            priority,
            triageData
        );

        // Create assessment
        TriageAssessment assessment = new TriageAssessment(
            UUID.randomUUID().toString(),
            petId,
            triageScore,
            priority,
            recommendations,
            LocalDateTime.now()
        );

        // Submit to WIA platform
        submitTriageAssessment(assessment);

        // Alert appropriate personnel
        if (priority == PriorityLevel.CRITICAL || priority == PriorityLevel.EMERGENCY) {
            alertEmergencyTeam(assessment);
        }

        return assessment;
    }

    private int calculateTriageScore(TriageData data) {
        int score = 0;

        // Vital signs assessment
        if (data.getHeartRate() > 180 || data.getHeartRate() < 60) {
            score += 3;
        }
        if (data.getRespiratoryRate() > 40 || data.getRespiratoryRate() < 10) {
            score += 3;
        }
        if (data.getTemperature() > 104 || data.getTemperature() < 99) {
            score += 2;
        }

        // Consciousness level
        switch (data.getConsciousnessLevel()) {
            case UNCONSCIOUS:
                score += 5;
                break;
            case SEMI_CONSCIOUS:
                score += 4;
                break;
            case DISORIENTED:
                score += 2;
                break;
            default:
                break;
        }

        // Breathing assessment
        if (data.getBreathingDifficulty() == BreathingDifficulty.SEVERE) {
            score += 5;
        } else if (data.getBreathingDifficulty() == BreathingDifficulty.MODERATE) {
            score += 3;
        }

        // Bleeding
        if (data.getBleedingSeverity() == BleedingSeverity.SEVERE_UNCONTROLLED) {
            score += 5;
        } else if (data.getBleedingSeverity() == BleedingSeverity.MODERATE) {
            score += 3;
        }

        // Pain assessment
        if (data.getPainScore() >= 8) {
            score += 3;
        } else if (data.getPainScore() >= 5) {
            score += 2;
        }

        // Trauma mechanism
        if (data.getTraumaMechanism() == TraumaMechanism.HIT_BY_CAR ||
            data.getTraumaMechanism() == TraumaMechanism.FALL_FROM_HEIGHT) {
            score += 3;
        }

        return score;
    }

    private PriorityLevel determinePriorityLevel(int score) {
        if (score >= 15) {
            return PriorityLevel.CRITICAL;  // Red - Immediate
        } else if (score >= 10) {
            return PriorityLevel.EMERGENCY;  // Orange - 15 minutes
        } else if (score >= 5) {
            return PriorityLevel.URGENT;  // Yellow - 60 minutes
        } else {
            return PriorityLevel.NON_URGENT;  // Green - 2+ hours
        }
    }

    private List getTriageRecommendations(PriorityLevel priority,
                                                   TriageData data) {
        List recommendations = new ArrayList<>();

        switch (priority) {
            case CRITICAL:
                recommendations.add("Immediate stabilization required");
                recommendations.add("Establish IV access immediately");
                recommendations.add("Prepare for possible resuscitation");
                recommendations.add("Alert veterinarian and surgery team");
                break;

            case EMERGENCY:
                recommendations.add("Priority assessment needed within 15 minutes");
                recommendations.add("Prepare oxygen if respiratory distress");
                recommendations.add("Monitor vital signs every 10 minutes");
                break;

            case URGENT:
                recommendations.add("Assess within 60 minutes");
                recommendations.add("Provide comfort measures");
                recommendations.add("Monitor for deterioration");
                break;

            case NON_URGENT:
                recommendations.add("Standard queue assessment");
                recommendations.add("Provide comfortable waiting area");
                break;
        }

        // Specific recommendations based on presenting complaint
        if (data.getBleedingSeverity() != BleedingSeverity.NONE) {
            recommendations.add("Apply direct pressure to bleeding sites");
        }

        if (data.getBreathingDifficulty() != BreathingDifficulty.NONE) {
            recommendations.add("Minimize stress, prepare oxygen");
        }

        return recommendations;
    }
