WIA Pet Health Passport Standard - Practice Management & Clinical Systems

4.1 Veterinary Practice Management Integration

Seamless integration between WIA Pet Health Passport systems and veterinary practice management software (PIMS - Practice Information Management Systems) is essential for efficient clinic workflows, accurate record keeping, and comprehensive patient care. The WIA standard defines bidirectional integration protocols enabling automatic synchronization of health records, appointments, diagnostic results, and billing information while maintaining data security and client confidentiality.

Veterinary clinics utilize diverse practice management systems including Avimark, eVetPractice, Cornerstone, VetSuite, and numerous others. The WIA standard provides vendor-neutral integration interfaces ensuring interoperability across platforms while supporting legacy systems through standardized API gateways and data transformation layers.

using System;

{
    private readonly HttpClient _httpClient;
    private readonly string _apiKey;
    private readonly string _baseUrl = "https://api.wia-pet-health.org/v1";

    public PIMSIntegrationService(string apiKey)
    {
        _apiKey = apiKey;
        _httpClient = new HttpClient();
        _httpClient.DefaultRequestHeaders.Add("Authorization", $"Bearer {apiKey}");
        _httpClient.DefaultRequestHeaders.Add("X-WIA-Version", "1.0");
    }

    public async Task SyncWithPIMS(string clinicId, PIMSVendor vendor)
    {
        // Step 1: Fetch new/updated records from PIMS
        var pimsRecords = await FetchPIMSRecords(clinicId, vendor);

        // Step 2: Map PIMS records to WIA schema
        var mappedRecords = MapPIMSToWIA(pimsRecords, vendor);

        // Step 3: Upload to WIA platform
        var syncData = new
        {
            integrationId = Guid.NewGuid(),
            clinicId = clinicId,
            pimsVendor = vendor.ToString().ToLower(),
            pimsVersion = pimsRecords.Version,
            syncTimestamp = DateTime.UtcNow,
            patientRecords = mappedRecords.Patients,
            visitRecords = mappedRecords.Visits,
            diagnosticResults = mappedRecords.DiagnosticResults
        };

        var json = JsonSerializer.Serialize(syncData);
        var content = new StringContent(json, System.Text.Encoding.UTF8, "application/json");

        var response = await _httpClient.PostAsync($"{_baseUrl}/integration/pims/sync", content);

        if (!response.IsSuccessStatusCode)
        {
            throw new Exception($"Sync failed: {await response.Content.ReadAsStringAsync()}");
        }

        var result = await JsonSerializer.DeserializeAsync(
            await response.Content.ReadAsStreamAsync()
        );

        // Step 4: Handle any sync errors
        if (result.SyncErrors?.Count > 0)
        {
            await HandleSyncErrors(result.SyncErrors);
        }

        // Step 5: Fetch updates from WIA back to PIMS
        await SyncWIAToPIMS(clinicId, result.LastSyncTime);

        return result;
    }

Example 2: Laboratory Result Integration (Python)

import hl7
import requests
from datetime import datetime
from typing import Dict, List

class LabResultIntegration:
    """Process and integrate laboratory results from external labs"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://api.wia-pet-health.org/v1'

    def process_hl7_message(self, hl7_message: str) -> Dict:
        """
        Parse HL7 v2.x message from laboratory and upload to WIA platform

        Args:
            hl7_message: HL7 formatted laboratory result message

        Returns:
            Processing result with patient match and upload status
        """
        # Parse HL7 message
        message = hl7.parse(hl7_message)

        # Extract patient identifiers
        pid_segment = message.segment('PID')
        patient_id = str(pid_segment[3])  # Patient ID
        patient_name = str(pid_segment[5])

        # Try to find microchip in extended patient IDs
        microchip = self.extract_microchip(pid_segment)

        # Match to WIA patient record
        wia_patient = self.match_patient(microchip, patient_id, patient_name)

        if not wia_patient:
            return {
                'success': False,
                'error': 'Patient not found in WIA system',
                'patient_id': patient_id
            }

        # Extract OBR (Order) and OBX (Results) segments
        results = self.parse_lab_results(message)

        # Upload to WIA platform
        upload_result = self.upload_results(wia_patient['petId'], results)

        return {
            'success': True,
            'patient_name': patient_name,
            'wia_pet_id': wia_patient['petId'],
            'tests_processed': len(results),
            'upload_id': upload_result['diagnosticId']
        }

    def extract_microchip(self, pid_segment) -> str:
        """Extract microchip number from PID segment alternate IDs"""
        # Check PID-4 (alternate patient ID) for microchip
        for alt_id in pid_segment[4]:
            if 'MICROCHIP' in str(alt_id).upper():
                return str(alt_id).split('^')[0]
        return None

    def match_patient(self, microchip: str, patient_id: str, name: str) -> Dict:
        """Match laboratory patient to WIA health record"""
        if microchip:
            # Try microchip lookup first (most reliable)
            response = requests.get(
                f'{self.base_url}/pets/microchip/{microchip}',
                headers={'Authorization': f'Bearer {self.api_key}'}
            )
            if response.status_code == 200:
                return response.json()

        # Fall back to name matching (less reliable)
        response = requests.get(
            f'{self.base_url}/pets/search?name={name}',
            headers={'Authorization': f'Bearer {self.api_key}'}
        )

        if response.status_code == 200:
            matches = response.json()['results']
            if len(matches) == 1:
                return matches[0]

        return None

    def parse_lab_results(self, message) -> List[Dict]:
        """Parse OBX segments containing test results"""
        results = []

        for obx in message.segments('OBX'):
            result = {
                'parameter': str(obx[3]),  # Observation identifier
                'value': str(obx[5]),       # Observation value
                'units': str(obx[6]) if len(obx) > 6 else '',
                'referenceRange': str(obx[7]) if len(obx) > 7 else '',
                'abnormalFlag': str(obx[8]) if len(obx) > 8 else '',
                'observationDate': str(obx[14]) if len(obx) > 14 else ''
            }

            # Determine if abnormal
            result['abnormalFlag'] = result['abnormalFlag'] in ['H', 'L', 'A', 'AA']

            results.append(result)

        return results

    def upload_results(self, pet_id: str, results: List[Dict]) -> Dict:
        """Upload parsed results to WIA platform"""
        diagnostic_data = {
            'testType': 'laboratory',
            'orderedDate': datetime.utcnow().isoformat(),
            'completedDate': datetime.utcnow().isoformat(),
            'results': results,
            'source': 'HL7_LAB_INTERFACE'
        }

        response = requests.post(
            f'{self.base_url}/integration/lab/result',
            headers={
                'Authorization': f'Bearer {self.api_key}',
                'Content-Type': 'application/json'
            },
            json={
                'petId': pet_id,
                'diagnosticResult': diagnostic_data
            }
        )

        response.raise_for_status()
        return response.json()

import org.dcm4che3.data.Attributes;
import org.dcm4che3.io.DicomInputStream;
import java.io.*;
import java.net.http.*;
import java.nio.file.*;

public class DICOMIntegrationService {
    private final String apiKey;
    private final String baseUrl = "https://api.wia-pet-health.org/v1";
    private final HttpClient httpClient;

    public DICOMIntegrationService(String apiKey) {
        this.apiKey = apiKey;
        this.httpClient = HttpClient.newHttpClient();
    }

    /**
     * Process DICOM file from imaging equipment and upload to WIA platform
     */
    public ImageUploadResult processDICOMImage(File dicomFile, String petId, String visitId)
            throws Exception {

        // Parse DICOM file to extract metadata
        DicomMetadata metadata = parseDICOMFile(dicomFile);

        // Verify patient match
        if (!verifyPatientMatch(metadata, petId)) {
            throw new Exception("Patient mismatch between DICOM and WIA record");
        }

        // Convert DICOM to viewable format (JPEG for thumbnails)
        File jpegPreview = convertDICOMToJPEG(dicomFile);

        // Upload DICOM file
        String dicomUrl = uploadFile(dicomFile, "application/dicom");

        // Upload preview image
        String previewUrl = uploadFile(jpegPreview, "image/jpeg");

        // Create imaging record in WIA system
        return createImagingRecord(petId, visitId, metadata, dicomUrl, previewUrl);
    }

    private DicomMetadata parseDICOMFile(File dicomFile) throws IOException {
        try (DicomInputStream dis = new DicomInputStream(dicomFile)) {
            Attributes attrs = dis.readDataset(-1, -1);

            return new DicomMetadata(
                attrs.getString(org.dcm4che3.data.Tag.PatientName),
                attrs.getString(org.dcm4che3.data.Tag.PatientID),
                attrs.getString(org.dcm4che3.data.Tag.StudyDate),
                attrs.getString(org.dcm4che3.data.Tag.Modality),
                attrs.getString(org.dcm4che3.data.Tag.BodyPartExamined),
                attrs.getString(org.dcm4che3.data.Tag.StudyDescription)
            );
        }
    }

    private boolean verifyPatientMatch(DicomMetadata metadata, String petId)
            throws Exception {

        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(baseUrl + "/pets/" + petId + "/health-record"))
            .header("Authorization", "Bearer " + apiKey)
            .GET()
            .build();

        HttpResponse response = httpClient.send(request,
            HttpResponse.BodyHandlers.ofString());

        if (response.statusCode() != 200) {
            return false;
        }

        JsonObject petRecord = JsonParser.parseString(response.body()).getAsJsonObject();
        String petName = petRecord.get("name").getAsString();

        // Compare names (allowing for variations)
        return comparePetNames(metadata.getPatientName(), petName);
    }

        File jpegFile = new File(dicomFile.getParent(),

        // Conversion logic here...

        return jpegFile;
    }

    private String uploadFile(File file, String contentType) throws Exception {
        // Multipart file upload to WIA storage
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(baseUrl + "/integration/imaging/upload"))
            .header("Authorization", "Bearer " + apiKey)
            .header("Content-Type", contentType)
            .POST(HttpRequest.BodyPublishers.ofFile(file.toPath()))
            .build();

        HttpResponse response = httpClient.send(request,
            HttpResponse.BodyHandlers.ofString());

        if (response.statusCode() != 200) {
            throw new Exception("Upload failed: " + response.statusCode());
        }

        JsonObject result = JsonParser.parseString(response.body()).getAsJsonObject();
        return result.get("fileUrl").getAsString();
    }

    private ImageUploadResult createImagingRecord(String petId, String visitId,
            DicomMetadata metadata, String dicomUrl, String previewUrl) throws Exception {

        JsonObject imagingRecord = new JsonObject();
        imagingRecord.addProperty("petId", petId);
        imagingRecord.addProperty("visitId", visitId);
        imagingRecord.addProperty("imagingType", metadata.getModality());
        imagingRecord.addProperty("bodyPart", metadata.getBodyPart());
        imagingRecord.addProperty("studyDate", metadata.getStudyDate());
        imagingRecord.addProperty("description", metadata.getDescription());
        imagingRecord.addProperty("dicomUrl", dicomUrl);
        imagingRecord.addProperty("previewUrl", previewUrl);

        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(baseUrl + "/integration/imaging/record"))
            .header("Authorization", "Bearer " + apiKey)
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(imagingRecord.toString()))
            .build();

        HttpResponse response = httpClient.send(request,
            HttpResponse.BodyHandlers.ofString());

        JsonObject result = JsonParser.parseString(response.body()).getAsJsonObject();

        return new ImageUploadResult(
            result.get("imagingId").getAsString(),
            dicomUrl,
            previewUrl
        );
    }

    private boolean comparePetNames(String name1, String name2) {
        // Normalize and compare names
        return name1.trim().equalsIgnoreCase(name2.trim());
    }
