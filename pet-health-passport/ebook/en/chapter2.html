Chapter 2: Vaccination Tracking Systems

WIA Pet Health Passport Standard - Immunization Management & Compliance

2.1 Introduction to Vaccination Tracking

Vaccination tracking is a critical component of pet health management, ensuring animals receive appropriate immunizations according to species-specific schedules, maintaining compliance with regulatory requirements, and protecting both individual pets and public health. The WIA Pet Health Passport Standard defines comprehensive protocols for recording, tracking, and validating vaccination records across international boundaries.

Effective vaccination tracking systems must account for multiple factors including species variations, regional disease prevalence, regulatory requirements, vaccine product differences, and individual pet health considerations. Digital vaccination records enable automated reminders, international travel verification, regulatory compliance reporting, and population-level disease surveillance.

Core Vaccination Record Components

  • Vaccine Identification: Product name, manufacturer, batch/lot number, expiration date
  • Administration Details: Date, route, dosage, site of injection, veterinarian identification
  • Next Due Date: Calculated based on vaccine protocol and species-specific schedules
  • Regulatory Compliance: Certificate numbers, official stamps, government approval documentation
  • Adverse Reactions: Any observed reactions or complications following vaccination
  • Immunity Status: Titer testing results if applicable, immunity verification
  • Travel Validity: International acceptance, country-specific requirements met

2.3 Vaccination Management API Endpoints

The WIA standard defines comprehensive API endpoints for vaccination record management, scheduling, and compliance verification.

GET /api/v1/pets/{petId}/vaccinations

Retrieve complete vaccination history for a specific pet with optional filtering by date range or vaccine type.

POST /api/v1/pets/{petId}/vaccinations

Record new vaccination event with full traceability including batch number and veterinarian credentials.

GET /api/v1/pets/{petId}/vaccinations/{vaccinationId}

Retrieve detailed information for a specific vaccination record including certificate image if available.

PUT /api/v1/pets/{petId}/vaccinations/{vaccinationId}

Update vaccination record (limited to adding titer results or adverse reactions, not modifying core data).

GET /api/v1/pets/{petId}/vaccinations/due

Get list of vaccinations currently due or upcoming within specified time window (default 30 days).

GET /api/v1/pets/{petId}/vaccinations/compliance

Check vaccination compliance status against standard schedules and regulatory requirements.

POST /api/v1/pets/{petId}/vaccinations/verify

Verify vaccination records meet requirements for specific purpose (travel, boarding, events).

GET /api/v1/pets/{petId}/vaccinations/certificate/{vaccinationId}

Generate official vaccination certificate in PDF format with digital signature and QR code.

POST /api/v1/vaccinations/batch-import

Bulk import vaccination records from legacy systems with validation and error reporting.

GET /api/v1/vaccinations/reminders

Retrieve upcoming vaccination reminders for all pets associated with owner account.

POST /api/v1/pets/{petId}/vaccinations/titer

Record titer test results and update immunity status for specific vaccine.

GET /api/v1/vaccinations/statistics

Retrieve vaccination statistics and coverage data for clinic or region (requires admin access).

Example 1: Recording Vaccination Event (TypeScript)

import axios from 'axios';

interface VaccinationRecord {
  vaccinationId?: string;
  petId: string;
  vaccineType: string;
  productName: string;
  manufacturer: string;
  batchNumber: string;
  expirationDate: string;
  administeredDate: string;
  route: string;
  site: string;
  dosage: {
    amount: number;
    unit: string;
  };
  veterinarianId: string;
  clinicId: string;
  nextDueDate: string;
  certificateNumber?: string;
}

class VaccinationService {
  private baseUrl = 'https://api.wia-pet-health.org/v1';
  private apiKey: string;

  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async recordVaccination(vaccination: VaccinationRecord): Promise {
    const endpoint = `${this.baseUrl}/pets/${vaccination.petId}/vaccinations`;

    // Calculate next due date based on vaccine type
    const nextDueDate = this.calculateNextDueDate(
      vaccination.vaccineType,
      vaccination.administeredDate
    );

    const vaccinationData = {
      ...vaccination,
      nextDueDate,
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString()
    };

    try {
      const response = await axios.post(endpoint, vaccinationData, {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
          'X-WIA-Version': '1.0'
        }
      });

      console.log(`Vaccination recorded: ${response.data.vaccinationId}`);

      // Automatically schedule reminder
      await this.scheduleReminder(
        vaccination.petId,
        nextDueDate,
        vaccination.vaccineType
      );

      return response.data;
    } catch (error) {
      console.error('Failed to record vaccination:', error);
      throw error;
    }
  }

  private calculateNextDueDate(vaccineType: string, administeredDate: string): string {
    const administered = new Date(administeredDate);
    let yearsToAdd = 1; // Default annual booster

    // Species and vaccine-specific schedules
    const scheduleMap: { [key: string]: number } = {
      'rabies': 3,           // 3-year rabies vaccine
      'distemper': 3,        // 3-year combo
      'bordetella': 1,       // Annual
      'leptospirosis': 1,    // Annual
      'feline_leukemia': 1   // Annual
    };

    yearsToAdd = scheduleMap[vaccineType] || 1;

    const nextDue = new Date(administered);
    nextDue.setFullYear(nextDue.getFullYear() + yearsToAdd);

    return nextDue.toISOString().split('T')[0];
  }

  async scheduleReminder(petId: string, dueDate: string, vaccineType: string): Promise {
    const reminderEndpoint = `${this.baseUrl}/reminders`;

    const reminder = {
      petId,
      reminderType: 'vaccination',
      dueDate,
      vaccineType,
      advanceNotice: 30 // Days before due date
    };

    await axios.post(reminderEndpoint, reminder, {
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      }
    });
  }

  async getComplianceStatus(petId: string): Promise {
    const endpoint = `${this.baseUrl}/pets/${petId}/vaccinations/compliance`;

    const response = await axios.get(endpoint, {
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'X-WIA-Version': '1.0'
      }
    });

    return response.data;
  }
}

Example 2: Travel Compliance Verification (Python)

import requests
from datetime import datetime, timedelta
from typing import List, Dict, Optional

class TravelComplianceChecker:
    """Verify pet vaccination compliance for international travel"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://api.wia-pet-health.org/v1'
        self.headers = {
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json',
            'X-WIA-Version': '1.0'
        }

    def check_travel_requirements(self, pet_id: str, destination_country: str,
                                   travel_date: str) -> Dict:
        """
        Check if pet meets vaccination requirements for travel to destination

        Args:
            pet_id: UUID of the pet
            destination_country: ISO 3166-1 alpha-2 country code
            travel_date: Intended travel date (ISO format)

        Returns:
            Compliance report with required actions
        """
        # Get country-specific requirements
        requirements = self._get_country_requirements(destination_country)

        # Get pet's current vaccination records
        vaccinations = self._get_vaccination_history(pet_id)

        # Verify each requirement
        compliance_report = {
            'compliant': True,
            'destination': destination_country,
            'travel_date': travel_date,
            'requirements_met': [],
            'requirements_missing': [],
            'actions_needed': []
        }

        for requirement in requirements:
            vaccine_type = requirement['vaccine_type']
            min_age_days = requirement.get('minimum_age_days', 0)
            min_wait_days = requirement.get('minimum_wait_after_vaccination', 21)
            max_validity_days = requirement.get('maximum_validity_days', 365)

            # Find matching vaccination
            matching_vax = self._find_valid_vaccination(
                vaccinations,
                vaccine_type,
                travel_date,
                min_wait_days,
                max_validity_days
            )

            if matching_vax:
                compliance_report['requirements_met'].append({
                    'vaccine': vaccine_type,
                    'administered': matching_vax['administeredDate'],
                    'valid_until': matching_vax.get('validUntil'),
                    'certificate': matching_vax.get('certificateNumber')
                })
            else:
                compliance_report['compliant'] = False
                compliance_report['requirements_missing'].append(vaccine_type)

                action = self._determine_required_action(
                    vaccine_type,
                    travel_date,
                    min_wait_days
                )
                compliance_report['actions_needed'].append(action)

        return compliance_report

    def _get_country_requirements(self, country_code: str) -> List[Dict]:
        """Fetch vaccination requirements for specific country"""
        endpoint = f'{self.base_url}/travel/requirements/{country_code}'
        response = requests.get(endpoint, headers=self.headers)
        response.raise_for_status()
        return response.json()['vaccination_requirements']

    def _get_vaccination_history(self, pet_id: str) -> List[Dict]:
        """Retrieve all vaccination records for pet"""
        endpoint = f'{self.base_url}/pets/{pet_id}/vaccinations'
        response = requests.get(endpoint, headers=self.headers)
        response.raise_for_status()
        return response.json()['vaccinations']

    def _find_valid_vaccination(self, vaccinations: List[Dict], vaccine_type: str,
                                travel_date: str, min_wait: int,
                                max_validity: int) -> Optional[Dict]:
        """Find vaccination that meets requirements for travel date"""
        travel_dt = datetime.fromisoformat(travel_date)

        for vax in vaccinations:
            if vax['vaccineType'] != vaccine_type:
                continue

            admin_dt = datetime.fromisoformat(vax['administeredDate'])

            # Check minimum wait period has passed
            earliest_valid = admin_dt + timedelta(days=min_wait)
            if travel_dt < earliest_valid:
                continue

            # Check vaccination hasn't expired
            latest_valid = admin_dt + timedelta(days=max_validity)
            if travel_dt > latest_valid:
                continue

            return vax

        return None

    def _determine_required_action(self, vaccine_type: str, travel_date: str,
                                   min_wait: int) -> Dict:
        """Determine what action is needed to meet requirement"""
        travel_dt = datetime.fromisoformat(travel_date)
        today = datetime.now()

        latest_vaccination_date = travel_dt - timedelta(days=min_wait)
        days_remaining = (latest_vaccination_date - today).days

        if days_remaining < 0:
            urgency = 'IMPOSSIBLE'
            message = f'Cannot meet {vaccine_type} requirement - insufficient time before travel'
        elif days_remaining < 7:
            urgency = 'URGENT'
            message = f'Must vaccinate for {vaccine_type} within {days_remaining} days'
        elif days_remaining < 30:
            urgency = 'SOON'
            message = f'Schedule {vaccine_type} vaccination within {days_remaining} days'
        else:
            urgency = 'PLANNED'
            message = f'{vaccine_type} vaccination needed before {latest_vaccination_date.date()}'

        return {
            'vaccine_type': vaccine_type,
            'urgency': urgency,
            'deadline': latest_vaccination_date.isoformat(),
            'days_remaining': days_remaining,
            'message': message
        }

    def generate_travel_certificate(self, pet_id: str, destination: str) -> bytes:
        """Generate PDF travel certificate with all required vaccinations"""
        endpoint = f'{self.base_url}/pets/{pet_id}/travel-certificate'

        response = requests.post(
            endpoint,
            headers=self.headers,
            json={'destination': destination, 'format': 'pdf'}
        )
        response.raise_for_status()

        return response.content

# Usage example
checker = TravelComplianceChecker('your-api-key')

# Check if pet can travel to EU
report = checker.check_travel_requirements(
    pet_id='pet-uuid-123',
    destination_country='FR',  # France
    travel_date='2025-06-15'
)

if report['compliant']:
    print("Pet meets all travel requirements!")
    certificate = checker.generate_travel_certificate('pet-uuid-123', 'FR')
    with open('travel_certificate.pdf', 'wb') as f:
        f.write(certificate)
else:
    print("Actions needed:")
    for action in report['actions_needed']:
        print(f"  [{action['urgency']}] {action['message']}")

package vaccination

import (
    "encoding/json"
    "fmt"
    "net/http"
    "time"

type VaccinationReminder struct {
    PetID        string    `json:"petId"`
    VaccineType  string    `json:"vaccineType"`
    DueDate      time.Time `json:"dueDate"`
    OwnerEmail   string    `json:"ownerEmail"`
    OwnerPhone   string    `json:"ownerPhone"`
    NotifyDays   []int     `json:"notifyDays"` // Days before due date to send reminders
}

type ReminderService struct {
    apiKey  string
    baseURL string
    client  *http.Client
}

func NewReminderService(apiKey string) *ReminderService {
    return &ReminderService{
        apiKey:  apiKey,
        baseURL: "https://api.wia-pet-health.org/v1",
        client:  &http.Client{Timeout: 30 * time.Second},
    }
}

// GetUpcomingReminders retrieves all vaccination reminders due within specified days
func (rs *ReminderService) GetUpcomingReminders(daysAhead int) ([]VaccinationReminder, error) {
    url := fmt.Sprintf("%s/vaccinations/reminders?days=%d", rs.baseURL, daysAhead)

    req, err := http.NewRequest("GET", url, nil)
    if err != nil {
        return nil, err
    }

    req.Header.Set("Authorization", "Bearer "+rs.apiKey)
    req.Header.Set("X-WIA-Version", "1.0")

    resp, err := rs.client.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return nil, fmt.Errorf("API returned status %d", resp.StatusCode)
    }

    var result struct {
        Reminders []VaccinationReminder `json:"reminders"`
    }

    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return nil, err
    }

    return result.Reminders, nil
}

// ProcessDailyReminders runs daily to send vaccination reminders
func (rs *ReminderService) ProcessDailyReminders() error {
    reminders, err := rs.GetUpcomingReminders(60) // Next 60 days
    if err != nil {
        return fmt.Errorf("failed to get reminders: %w", err)
    }

    today := time.Now()

    for _, reminder := range reminders {
        daysUntilDue := int(reminder.DueDate.Sub(today).Hours() / 24)

        // Check if we should send reminder today
        shouldNotify := false
        for _, notifyDay := range reminder.NotifyDays {
            if daysUntilDue == notifyDay {
                shouldNotify = true
                break
            }
        }

        if shouldNotify {
            if err := rs.sendReminder(reminder, daysUntilDue); err != nil {
                fmt.Printf("Failed to send reminder for pet %s: %v\n",
                          reminder.PetID, err)
                continue
            }
            fmt.Printf("Sent reminder for pet %s: %s due in %d days\n",
                      reminder.PetID, reminder.VaccineType, daysUntilDue)
        }
    }

    return nil
}

func (rs *ReminderService) sendReminder(reminder VaccinationReminder, daysUntil int) error {
    // Send email reminder
    if reminder.OwnerEmail != "" {
        emailBody := fmt.Sprintf(
            "Vaccination Reminder: Your pet's %s vaccination is due in %d days (on %s). "+
            "Please schedule an appointment with your veterinarian.",
            reminder.VaccineType,
            daysUntil,
            reminder.DueDate.Format("January 2, 2006"),
        )

        if err := rs.sendEmail(reminder.OwnerEmail, "Vaccination Due", emailBody); err != nil {
            return err
        }
    }

    // Send SMS if phone number provided and reminder is urgent
    if reminder.OwnerPhone != "" && daysUntil <= 7 {
        smsBody := fmt.Sprintf(
            "Pet vaccination reminder: %s due in %d days. Schedule appointment.",
            reminder.VaccineType, daysUntil,
        )

        if err := rs.sendSMS(reminder.OwnerPhone, smsBody); err != nil {
            return err
        }
    }

    return nil
}
