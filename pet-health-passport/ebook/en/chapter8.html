WIA Pet Health Passport Standard - Certification Framework & Validation

8.1 WIA Certification Framework

WIA Pet Health Passport certification ensures implementations meet rigorous standards for interoperability, security, data quality, and user experience. The certification process validates technical compliance, security posture, and operational readiness through comprehensive testing, documentation review, and performance benchmarking. Certified systems display the WIA compliance badge demonstrating adherence to international best practices and enabling participation in the global pet health data exchange network.

Certification provides multiple benefits including customer trust, regulatory compliance alignment, insurance company recognition, cross-border acceptance, and access to WIA technical support resources. Organizations pursuing certification undergo structured assessment across eight core domains covering all aspects of digital pet health passport implementation from data schemas to emergency access protocols.

Certification Levels

Level Requirements Domains Covered Validity Period Recertification

WIA Certified Standard Basic + travel, veterinary integration 5 domains 2 years Annual review + testing
WIA Certified Advanced Standard + medical records, emergency 7 domains 2 years Biannual review + testing

Certification Requirements Summary

Across all eight chapters, implementations must satisfy minimum requirements totaling 200+ certification criteria. Chapter-specific minimums: CR-1 (25 items), CR-2 (25 items), CR-3 (25 items), CR-4 (25 items), CR-5 (25 items), CR-6 (25 items), CR-7 (25 items), CR-8 (25 items).

Schedule technical testing session for API compliance, performance, and integration validation.

Retrieve detailed requirements list for specific certification level with testing criteria.

POST /api/v1/certification/validate

Run automated validation tests against implementation to check compliance before formal assessment.

GET /api/v1/certification/directory

Search directory of WIA certified organizations with filtering by level, location, and services.

POST /api/v1/certification/{certificationId}/badge/generate

Generate official certification badge for display on website and marketing materials.

GET /api/v1/certification/{certificateNumber}/verify

Verify authenticity of WIA certification badge or certificate number.

POST /api/v1/certification/recertification/schedule

Schedule required recertification review based on certification level and expiration date.

POST /api/v1/certification/complaint/file

File compliance complaint against certified organization for investigation.

GET /api/v1/certification/statistics

Retrieve certification program statistics including pass rates, common deficiencies, average timeframes.

    def run_full_certification_suite(self) -> Dict:

        results = {

            'start_time': time.time(),

        }

    def test_api_endpoint(self, method: str, endpoint: str, data: dict = None) -> bool:
        """Test specific API endpoint"""
        try:
            url = f'{self.base_url}{endpoint}'
            headers = {
                'Authorization': f'Bearer {self.api_key}',
                'Content-Type': 'application/json'
            }

            if method == 'POST':
                response = requests.post(url, headers=headers, json=data, timeout=5)
            elif method == 'GET':
                response = requests.get(url, headers=headers, timeout=5)
            elif method == 'PUT':
                response = requests.put(url, headers=headers, json=data, timeout=5)
            elif method == 'DELETE':
                response = requests.delete(url, headers=headers, timeout=5)

            return response.status_code < 500  # Accept any non-server-error response

        except Exception as e:
            print(f"Error testing {method} {endpoint}: {e}")
            return False

    def test_endpoint_exists(self, method: str, endpoint: str) -> bool:
        """Verify endpoint exists (returns any response except 404)"""
        try:
            url = f'{self.base_url}{endpoint.replace("{petId}", "test-id").replace("{microchipNumber}", "985112001234567").replace("{ownerId}", "test-owner")}'
            response = requests.request(method, url,
                headers={'Authorization': f'Bearer {self.api_key}'},
                timeout=5
            )
            return response.status_code != 404
        except:
            return False

    def generate_certification_report(self, results: Dict) -> str:
        """Generate formatted certification test report"""
        report = []
        report.append("=" * 80)
        report.append("WIA PET HEALTH PASSPORT CERTIFICATION TEST REPORT")
        report.append("=" * 80)
        report.append(f"Test Duration: {results['duration']:.2f} seconds")
        report.append(f"Overall Compliance: {results['overall_compliance']:.2f}%")
        report.append(f"Certification Eligible: {'YES' if results['eligible'] else 'NO'}")
        report.append("")
        report.append("Domain Results:")
        report.append("-" * 80)

        for domain in results['domains']:
            compliance = (domain['passed_tests'] / domain['total_tests']) * 100
            status = "✓ PASS" if compliance >= 95 else "✗ FAIL"
            report.append(f"{domain['domain_code']} - {domain['domain']}")
            report.append(f"  Tests: {domain['passed_tests']}/{domain['total_tests']} passed ({compliance:.1f}%) {status}")
            report.append("")

        return "\n".join(report)

// Performance testing for WIA certification
class PerformanceBenchmark {
  constructor(apiKey, baseUrl) {
    this.apiKey = apiKey;
    this.baseUrl = baseUrl;
    this.results = [];
  }

  async runBenchmarks() {
    console.log('Starting Performance Benchmarks...');

    const benchmarks = [
      {
        name: 'Health Record Retrieval',
        endpoint: '/api/v1/pets/test-uuid/health-record',
        method: 'GET',
        maxResponseTime: 200 // ms
      },
      {
        name: 'Microchip Lookup (Emergency)',
        endpoint: '/api/v1/emergency/pets/985112001234567',
        method: 'GET',
        maxResponseTime: 500 // ms
      },
      {
        name: 'Vaccination Record Creation',
        endpoint: '/api/v1/pets/test-uuid/vaccinations',
        method: 'POST',
        maxResponseTime: 300 // ms
      },
      {
        name: 'Travel Compliance Check',
        endpoint: '/api/v1/pets/test-uuid/travel/validate',
        method: 'POST',
        maxResponseTime: 1000 // ms
      },
      {
        name: 'Bulk Health Record Export',
        endpoint: '/api/v1/pets/test-uuid/health-record/export',
        method: 'POST',
        maxResponseTime: 2000 // ms
      }
    ];

    for (const benchmark of benchmarks) {
      const result = await this.runSingleBenchmark(benchmark);
      this.results.push(result);
    }

    return this.generateReport();
  }

  async runSingleBenchmark(benchmark) {
    const iterations = 100;
    const times = [];

    for (let i = 0; i < iterations; i++) {
      const startTime = performance.now();

      try {
        await fetch(`${this.baseUrl}${benchmark.endpoint}`, {
          method: benchmark.method,
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json'
          },
          body: benchmark.method === 'POST' ? JSON.stringify({}) : undefined
        });

        const endTime = performance.now();
        times.push(endTime - startTime);

      } catch (error) {
        console.error(`Error in ${benchmark.name}:`, error);
      }

      // Small delay between requests
      await new Promise(resolve => setTimeout(resolve, 10));
    }

    // Calculate statistics
    times.sort((a, b) => a - b);
    const avg = times.reduce((a, b) => a + b, 0) / times.length;
    const median = times[Math.floor(times.length / 2)];
    const p95 = times[Math.floor(times.length * 0.95)];
    const p99 = times[Math.floor(times.length * 0.99)];

    return {
      name: benchmark.name,
      iterations: iterations,
      average: avg.toFixed(2),
      median: median.toFixed(2),
      p95: p95.toFixed(2),
      p99: p99.toFixed(2),
      maxAllowed: benchmark.maxResponseTime,
      passed: p95 <= benchmark.maxResponseTime
    };
  }

  generateReport() {
    console.log('\n' + '='.repeat(80));
    console.log('PERFORMANCE BENCHMARK REPORT');
    console.log('='.repeat(80));

    this.results.forEach(result => {
      console.log(`\n${result.name}`);
      console.log(`  Iterations: ${result.iterations}`);
      console.log(`  Average:    ${result.average} ms`);
      console.log(`  Median:     ${result.median} ms`);
      console.log(`  95th %ile:  ${result.p95} ms`);
      console.log(`  99th %ile:  ${result.p99} ms`);
      console.log(`  Max Allowed: ${result.maxAllowed} ms`);
      console.log(`  Status:     ${result.passed ? '✓ PASS' : '✗ FAIL'}`);
    });

    const allPassed = this.results.every(r => r.passed);
    console.log('\n' + '='.repeat(80));
    console.log(`Overall Result: ${allPassed ? '✓ ALL TESTS PASSED' : '✗ SOME TESTS FAILED'}`);
    console.log('='.repeat(80));

    return {
      passed: allPassed,
      results: this.results
    };
  }
}

require 'net/http'
require 'json'
require 'openssl'

class WIASecurityAudit
  def initialize(api_key, base_url)
    @api_key = api_key
    @base_url = base_url
    @vulnerabilities = {
      critical: [],
      high: [],
      medium: [],
      low: []
    }
  end

  def run_security_audit
    puts "Running WIA Security Audit..."
    puts "=" * 80

    # Test 1: SSL/TLS Configuration
    test_ssl_tls

    # Test 2: Authentication
    test_authentication

    # Test 3: Authorization
    test_authorization

    # Test 4: SQL Injection Protection
    test_sql_injection

    # Test 5: XSS Protection
    test_xss_protection

    # Test 6: CSRF Protection
    test_csrf_protection

    # Test 7: Rate Limiting
    test_rate_limiting

    # Test 8: Data Encryption
    test_data_encryption

    # Test 9: API Key Security
    test_api_key_security

    # Test 10: Session Management
    test_session_management

    generate_audit_report
  end

  def test_ssl_tls
    uri = URI(@base_url)
    http = Net::HTTP.new(uri.host, uri.port)
    http.use_ssl = true
    http.verify_mode = OpenSSL::SSL::VERIFY_PEER

    begin
      http.start do
        if http.peer_cert.nil?
          add_vulnerability(:critical, "SSL certificate not found")
        else
          cert = http.peer_cert
          if cert.not_after < Time.now
            add_vulnerability(:critical, "SSL certificate expired")
          end

          # Check TLS version
          if http.ssl_version =~ /TLSv1\.3/
            puts "✓ TLS 1.3 supported"
          else
            add_vulnerability(:medium, "TLS 1.3 not supported")
          end
        end
      end
    rescue => e
      add_vulnerability(:critical, "SSL/TLS connection failed: #{e.message}")
    end
  end

  def test_authentication
    # Test without credentials
    response = make_request('/api/v1/pets/test/health-record', nil)

    if response.code.to_i == 401
      puts "✓ Proper authentication required"
    else
      add_vulnerability(:high, "Unauthenticated access allowed")
    end

    # Test with invalid credentials
    response = make_request('/api/v1/pets/test/health-record', 'invalid-key')

    if response.code.to_i == 401 || response.code.to_i == 403
      puts "✓ Invalid credentials rejected"
    else
      add_vulnerability(:high, "Invalid credentials accepted")
    end
  end

  def test_authorization
    # Test accessing another user's data
    response = make_request('/api/v1/pets/other-user-pet/health-record', @api_key)

    if response.code.to_i == 403
      puts "✓ Authorization checks in place"
    else
      add_vulnerability(:critical, "Authorization bypass possible")
    end
  end

  def test_sql_injection
    # Test common SQL injection patterns
    injection_patterns = [
      "' OR '1'='1",
      "'; DROP TABLE pets;--",
      "admin'--",
      "1' UNION SELECT NULL--"
    ]

    injection_patterns.each do |pattern|
      response = make_request("/api/v1/pets/#{URI.encode_www_form_component(pattern)}/health-record", @api_key)

      if response.code.to_i == 500
        add_vulnerability(:critical, "Possible SQL injection vulnerability")
        break
      end
    end

    puts "✓ SQL injection tests passed"
  end

  def test_rate_limiting
    # Make rapid requests
    requests = []
    20.times do
      requests << make_request('/api/v1/pets/test/health-record', @api_key)
    end

    too_many_requests = requests.count { |r| r.code.to_i == 429 }

    if too_many_requests > 0
      puts "✓ Rate limiting active"
    else
      add_vulnerability(:medium, "Rate limiting not detected")
    end
  end

  def make_request(path, api_key)
    uri = URI("#{@base_url}#{path}")
    request = Net::HTTP::Get.new(uri)
    request['Authorization'] = "Bearer #{api_key}" if api_key

    Net::HTTP.start(uri.hostname, uri.port, use_ssl: true) do |http|
      http.request(request)
    end
  end

  def add_vulnerability(severity, description)
    @vulnerabilities[severity] << description
  end

  def generate_audit_report
    puts "\n" + "=" * 80
    puts "SECURITY AUDIT REPORT"
    puts "=" * 80

    total_vulns = @vulnerabilities.values.flatten.length

    puts "\nVulnerabilities Found: #{total_vulns}"
    puts "  Critical: #{@vulnerabilities[:critical].length}"
    puts "  High:     #{@vulnerabilities[:high].length}"
    puts "  Medium:   #{@vulnerabilities[:medium].length}"
    puts "  Low:      #{@vulnerabilities[:low].length}"

    if total_vulns > 0
      puts "\nDetails:"
      @vulnerabilities.each do |severity, vulns|
        next if vulns.empty?

        puts "\n#{severity.upcase}:"
        vulns.each { |v| puts "  - #{v}" }
      end
    end

    passed = @vulnerabilities[:critical].empty? && @vulnerabilities[:high].empty?

    puts "\n" + "=" * 80
    puts "Audit Result: #{passed ? '✓ PASSED' : '✗ FAILED'}"
    puts "=" * 80

    { passed: passed, vulnerabilities: @vulnerabilities }
  end
end
