WIA Pet Health Passport Standard - Pet Identification & Global Registry Integration

5.1 ISO 11784/11785 Microchip Standards

Pet microchipping represents the gold standard for permanent animal identification, with over 500 million animals microchipped worldwide. The WIA Pet Health Passport Standard mandates compliance with ISO 11784 (identification code structure) and ISO 11785 (technical implementation) to ensure global interoperability and reliable pet identification across international boundaries, veterinary clinics, animal shelters, and border control checkpoints.

ISO-compliant microchips use 134.2 kHz RFID technology with 15-digit identification codes following a standardized format: 3-digit country/manufacturer code, followed by 12-digit unique identifier. This standardization enables universal reading across different scanner brands and global database lookups for lost or found pets, international travel verification, and ownership authentication.

Microchip Code Structure

Digit Position Purpose Range Example Notes
1-3 Country/Manufacturer Code 000-999 981 (USA) ISO 3166 based allocation
4-15 Unique Animal ID 000000000000-999999999999 120012345678 Manufacturer managed sequence
Full Code Complete Microchip Number 15 digits 985112001234567 Globally unique identifier

5.3 Microchip Management API Endpoints

POST /api/v1/microchips/register

Register new microchip with owner information and veterinary implantation details.

GET /api/v1/microchips/{microchipNumber}

Lookup microchip registration details including current owner and pet health record reference.

PUT /api/v1/microchips/{microchipNumber}/owner

Transfer microchip ownership with documentation of transfer reason and previous owner.

POST /api/v1/microchips/{microchipNumber}/scan

Log microchip scan event with location, scanner ID, and purpose (routine/found/border control).

GET /api/v1/microchips/{microchipNumber}/history

Retrieve complete scan history and ownership transfer records for auditing.

POST /api/v1/microchips/verify

Verify microchip number validity, checksum, and manufacturer code compliance.

POST /api/v1/microchips/batch-lookup

Bulk lookup of multiple microchip numbers for shelter or clinic processing.

GET /api/v1/microchips/owner/{ownerId}

Retrieve all microchips registered to specific owner account.

POST /api/v1/microchips/{microchipNumber}/registry/add

Add microchip registration to additional international registries (Petmaxx, AAHA, etc.).

PUT /api/v1/microchips/{microchipNumber}/deactivate

Deactivate microchip registration (pet deceased or chip failure).

GET /api/v1/microchips/found-pet

Emergency lookup interface for found pet with enhanced owner contact information.

POST /api/v1/microchips/alert/lost

Report microchipped pet as lost and activate alert system for shelters/vets in area.

Example 1: Microchip Scanner Integration (JavaScript)

// Integration with RFID scanner hardware
class MicrochipScannerService {
  constructor(apiKey, scannerPort = '/dev/ttyUSB0') {
    this.apiKey = apiKey;
    this.baseUrl = 'https://api.wia-pet-health.org/v1';
    this.scannerPort = scannerPort;
  }

  async scanMicrochip() {
    // Read from RFID scanner (example using serial port)
    const microchipNumber = await this.readFromScanner();

    if (!microchipNumber || !this.validateMicrochip(microchipNumber)) {
      throw new Error('Invalid microchip read');
    }

    // Log scan event
    await this.logScan(microchipNumber);

    // Lookup pet information
    const petInfo = await this.lookupMicrochip(microchipNumber);

    return {
      microchipNumber,
      petInfo,
      scanTimestamp: new Date().toISOString()
    };
  }

  async readFromScanner() {
    // Hardware-specific scanner integration
    // This would use actual serial port library
    return new Promise((resolve) => {
      // Simulated scanner read
      setTimeout(() => {
        resolve('985112001234567');
      }, 1000);
    });
  }

  validateMicrochip(number) {
    // Validate ISO format (15 digits)
    if (!/^[0-9]{15}$/.test(number)) {
      return false;
    }

    // Validate manufacturer code
    const prefix = number.substring(0, 3);
    const validPrefixes = ['981', '985', '900', '977', '956', '991'];

    return validPrefixes.includes(prefix);
  }

  async lookupMicrochip(microchipNumber) {
    const response = await fetch(
      `${this.baseUrl}/microchips/${microchipNumber}`,
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'X-WIA-Version': '1.0'
        }
      }
    );

    if (!response.ok) {
      if (response.status === 404) {
        return {
          registered: false,
          message: 'Microchip not registered'
        };
      }
      throw new Error(`Lookup failed: ${response.status}`);
    }

    return await response.json();
  }

  async logScan(microchipNumber) {
    await fetch(
      `${this.baseUrl}/microchips/${microchipNumber}/scan`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          scanDate: new Date().toISOString(),
          scannerId: this.getScannerId(),
          location: this.getLocation(),
          purpose: 'routine'
        })
      }
    );
  }

  getScannerId() {
    // Return unique scanner hardware ID
    return 'SCANNER-12345';
  }

  getLocation() {
    // Return clinic or facility location
    return 'Main Street Veterinary Clinic';
  }
}

Example 2: Lost Pet Alert System (Python)

import requests
from datetime import datetime, timedelta
from typing import Dict, List

class LostPetAlertSystem:
    """System for managing lost pet alerts and notifications"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://api.wia-pet-health.org/v1'
        self.headers = {
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json'
        }

    def report_lost_pet(self, microchip_number: str, lost_location: Dict,
                        owner_contact: Dict) -> str:
        """
        Report pet as lost and activate alert system

        Args:
            microchip_number: Pet's microchip number
            lost_location: {'address': str, 'city': str, 'state': str, 'zip': str, 'lat': float, 'lon': float}
            owner_contact: {'name': str, 'phone': str, 'email': str, 'alt_phone': str}

        Returns:
            Alert ID for tracking
        """
        # Verify pet ownership
        pet_info = self.lookup_microchip(microchip_number)

        alert_data = {
            'microchipNumber': microchip_number,
            'petId': pet_info['petId'],
            'lostDate': datetime.utcnow().isoformat(),
            'lostLocation': lost_location,
            'ownerContact': owner_contact,
            'alertRadius': 50,  # miles
            'description': pet_info.get('description', ''),
            'rewardOffered': False,
            'status': 'active'
        }

        response = requests.post(
            f'{self.base_url}/microchips/alert/lost',
            headers=self.headers,
            json=alert_data
        )
        response.raise_for_status()

        result = response.json()
        alert_id = result['alertId']

        # Notify nearby shelters and vets
        self.notify_facilities(lost_location, pet_info, alert_id)

        # Enable GPS tracking if pet has tracker
        if pet_info.get('hasGPSTracker'):
            self.enable_gps_tracking(microchip_number)

        return alert_id

    def notify_facilities(self, location: Dict, pet_info: Dict, alert_id: str):
        """Send notifications to nearby animal facilities"""

        # Find shelters/vets within radius
        facilities = self.find_nearby_facilities(
            location['lat'],
            location['lon'],
            radius_miles=50
        )

        notification_data = {
            'alertId': alert_id,
            'petInfo': {
                'name': pet_info['name'],
                'species': pet_info['species'],
                'breed': pet_info['breed'],
                'color': pet_info['color'],
                'microchip': pet_info['microchipNumber']
            },
            'lastSeenLocation': location,
            'ownerContact': pet_info['ownerPhone']
        }

        for facility in facilities:
            try:
                requests.post(
                    f'{self.base_url}/notifications/facility/{facility["id"]}',
                    headers=self.headers,
                    json=notification_data
                )
            except Exception as e:
                print(f"Failed to notify {facility['name']}: {e}")

    def check_alert_matches(self, microchip_number: str, found_location: Dict) -> Dict:
        """
        Check if scanned microchip matches any lost pet alerts

        Args:
            microchip_number: Scanned microchip number
            found_location: Location where pet was found

        Returns:
            Match information with owner contact details
        """
        response = requests.get(
            f'{self.base_url}/microchips/{microchip_number}/alerts/active',
            headers=self.headers
        )

        if response.status_code == 200:
            alerts = response.json().get('alerts', [])

            if len(alerts) > 0:
                alert = alerts[0]  # Most recent alert

                # Calculate distance from lost location
                distance = self.calculate_distance(
                    found_location,
                    alert['lostLocation']
                )

                # Notify owner immediately
                self.notify_owner_found(alert, found_location)

                return {
                    'match': True,
                    'alertId': alert['alertId'],
                    'petName': alert['petInfo']['name'],
                    'ownerContact': alert['ownerContact'],
                    'lostDate': alert['lostDate'],
                    'distanceFromLostLocation': distance,
                    'matchConfidence': 'HIGH'
                }

        return {'match': False}

    def notify_owner_found(self, alert: Dict, found_location: Dict):
        """Send immediate notification to owner that pet was found"""

        notification = {
            'alertId': alert['alertId'],
            'notificationType': 'PET_FOUND',
            'message': f"Great news! {alert['petInfo']['name']} has been scanned at {found_location['address']}",
            'foundLocation': found_location,
            'timestamp': datetime.utcnow().isoformat(),
            'urgency': 'HIGH'
        }

        # Send SMS
        self.send_sms(alert['ownerContact']['phone'], notification['message'])

        # Send email
        self.send_email(alert['ownerContact']['email'], notification)

        # Log notification
        requests.post(
            f"{self.base_url}/microchips/alert/{alert['alertId']}/notify",
            headers=self.headers,
            json=notification
        )

    def lookup_microchip(self, microchip_number: str) -> Dict:
        """Lookup microchip registration"""
        response = requests.get(
            f'{self.base_url}/microchips/{microchip_number}',
            headers=self.headers
        )
        response.raise_for_status()
        return response.json()

    def find_nearby_facilities(self, lat: float, lon: float, radius_miles: int) -> List[Dict]:
        """Find animal facilities near location"""
        response = requests.get(
            f'{self.base_url}/facilities/nearby',
            headers=self.headers,
            params={'lat': lat, 'lon': lon, 'radius': radius_miles}
        )
        return response.json().get('facilities', [])

    def calculate_distance(self, loc1: Dict, loc2: Dict) -> float:
        """Calculate distance between two locations in miles"""
        from math import radians, cos, sin, asin, sqrt

        lon1, lat1, lon2, lat2 = map(radians, [loc1['lon'], loc1['lat'], loc2['lon'], loc2['lat']])
        dlon = lon2 - lon1
        dlat = lat2 - lat1
        a = sin(dlat/2)**2 + cos(lat1) * cos(lat2) * sin(dlon/2)**2
        c = 2 * asin(sqrt(a))
        miles = 3956 * c
        return round(miles, 2)

    def send_sms(self, phone: str, message: str):
        # SMS service integration
        print(f"SMS to {phone}: {message}")

    def send_email(self, email: str, notification: Dict):
        # Email service integration
        print(f"Email to {email}: {notification['message']}")

import java.net.http.*;
import java.net.URI;
import com.google.gson.*;
import java.util.*;

public class MultiRegistrySync {
    private final String apiKey;
    private final HttpClient httpClient;
    private final Gson gson;

    public MultiRegistrySync(String apiKey) {
        this.apiKey = apiKey;
        this.httpClient = HttpClient.newHttpClient();
        this.gson = new Gson();
    }

    /**
     * Synchronize microchip registration across multiple international registries
     */
    public RegistrySyncResult syncToInternationalRegistries(
            String microchipNumber,
            PetRegistrationData petData) throws Exception {

        RegistrySyncResult result = new RegistrySyncResult(microchipNumber);

        // Register with AAHA Universal Pet Microchip Lookup (USA)
        if (syncToAAHA(microchipNumber, petData)) {
            result.addSuccess("AAHA");
        } else {
            result.addFailure("AAHA", "Registration failed");
        }

        // Register with Petmaxx (Europe)
        if (syncToPetmaxx(microchipNumber, petData)) {
            result.addSuccess("Petmaxx");
        } else {
            result.addFailure("Petmaxx", "Registration failed");
        }

        // Register with Europetnet (Europe)
        if (syncToEuropetnet(microchipNumber, petData)) {
            result.addSuccess("Europetnet");
        } else {
            result.addFailure("Europetnet", "Registration failed");
        }

        // Update WIA platform with registry sync status
        updateWIARegistry(microchipNumber, result);

        return result;
    }

    private boolean syncToAAHA(String microchipNumber, PetRegistrationData petData)
            throws Exception {
        // AAHA API integration
        JsonObject registrationData = new JsonObject();
        registrationData.addProperty("microchipNumber", microchipNumber);
        registrationData.addProperty("petName", petData.getPetName());
        registrationData.addProperty("species", petData.getSpecies());
        registrationData.addProperty("breed", petData.getBreed());

        // Add owner information
        JsonObject owner = new JsonObject();
        owner.addProperty("name", petData.getOwnerName());
        owner.addProperty("phone", petData.getOwnerPhone());
        owner.addProperty("email", petData.getOwnerEmail());
        registrationData.add("owner", owner);

        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create("https://api.aaha.org/v1/microchip/register"))
            .header("Authorization", "Bearer " + apiKey)
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(gson.toJson(registrationData)))
            .build();

        HttpResponse response = httpClient.send(request,
            HttpResponse.BodyHandlers.ofString());

        return response.statusCode() == 200;
    }

            throws Exception {

    }

    private boolean syncToEuropetnet(String microchipNumber, PetRegistrationData petData)
            throws Exception {

    }

    private void updateWIARegistry(String microchipNumber, RegistrySyncResult result)
            throws Exception {
        JsonObject update = new JsonObject();

        JsonObject registries = new JsonObject();
        registries.addProperty("aaha", result.isSuccessful("AAHA"));
        registries.addProperty("petmaxx", result.isSuccessful("Petmaxx"));

            .POST(HttpRequest.BodyPublishers.ofString(gson.toJson(update)))
            .build();

        httpClient.send(request, HttpResponse.BodyHandlers.ofString());
    }
}
