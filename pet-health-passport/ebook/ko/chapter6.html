WIA Pet Health Passport Standard - Comprehensive Clinical Documentation

6.1 Comprehensive Medical Record Management

Complete and accurate medical history documentation forms the foundation of quality veterinary care, enabling informed clinical decisions, continuity of care across providers, and longitudinal health monitoring. The WIA Pet Health Passport Standard defines structured medical record schemas capturing all aspects of pet health including diagnoses, treatments, surgeries, chronic conditions, medications, and preventive care interventions.

Traditional veterinary medical records vary widely in format, completeness, and accessibility. Digital standardization through WIA protocols ensures consistent documentation, enables automated clinical decision support, facilitates research and epidemiological studies, and supports seamless care transitions when pets change veterinarians or relocate geographically.

6.3 Medical Records API Endpoints

POST /api/v1/pets/{petId}/medical-records

Create new medical record entry with full SOAP documentation and associated procedures.

GET /api/v1/pets/{petId}/medical-records

Retrieve complete medical history with optional filtering by date range, record type, or diagnosis.

GET /api/v1/pets/{petId}/medical-records/{recordId}

Get detailed information for specific medical record including all attachments and linked records.

PUT /api/v1/pets/{petId}/medical-records/{recordId}

Update medical record (limited to adding addendums, cannot modify original SOAP notes).

GET /api/v1/pets/{petId}/medical-records/summary

Generate medical history summary highlighting chronic conditions, active medications, and recent visits.

GET /api/v1/pets/{petId}/medications/active

Retrieve currently active medications with dosage schedules and refill status.

POST /api/v1/pets/{petId}/medications/prescription

Create new medication prescription with electronic signature and pharmacy routing.

GET /api/v1/pets/{petId}/chronic-conditions

List all chronic conditions with management plans, medications, and monitoring schedules.

POST /api/v1/pets/{petId}/chronic-conditions

Add new chronic condition diagnosis with treatment protocol and monitoring requirements.

GET /api/v1/pets/{petId}/surgical-history

Retrieve complete surgical history including procedures, outcomes, and post-operative notes.

POST /api/v1/pets/{petId}/medical-records/attachment

Upload medical record attachment (X-ray, lab report, photo) with metadata linking.

GET /api/v1/pets/{petId}/vital-signs/trend

Analyze vital signs trends over time for weight, temperature, and other monitored parameters.

// Create comprehensive SOAP medical record
class MedicalRecordService {
  constructor(apiKey) {
    this.apiKey = apiKey;
    this.baseUrl = 'https://api.wia-pet-health.org/v1';
  }

  async createSOAPRecord(petId, visitData) {
    const record = {
      recordId: crypto.randomUUID(),
      petId: petId,
      visitDate: new Date().toISOString(),
      veterinarianId: visitData.vetId,
      clinicId: visitData.clinicId,
      recordType: visitData.visitType || 'wellness',
      chiefComplaint: visitData.chiefComplaint,

      soapNotes: {
        subjective: visitData.subjective,
        objective: {
          vitalSigns: {
            temperature: visitData.vitals.temperature,
            heartRate: visitData.vitals.heartRate,
            respiratoryRate: visitData.vitals.respiratoryRate,
            weight: visitData.vitals.weight,
            bodyConditionScore: visitData.vitals.bcs,
            painScore: visitData.vitals.painScore || 0
          },
          physicalExam: visitData.physicalExam
        },
        assessment: visitData.diagnoses.map(dx => ({
          diagnosis: dx.name,
          diagnosisCode: dx.code,
          severity: dx.severity || 'moderate',
          confidence: dx.confidence || 'presumptive'
        })),
        plan: {
          diagnosticTests: visitData.plan.tests || [],
          treatments: visitData.plan.treatments || [],
          medications: visitData.plan.medications || [],
          followUp: {
            required: visitData.plan.followUpRequired,
            timeframe: visitData.plan.followUpTimeframe,
            instructions: visitData.plan.instructions
          }
        }
      },

      procedures: visitData.procedures || [],
      medications: visitData.prescriptions || [],
      createdAt: new Date().toISOString(),
      signedBy: visitData.vetId,
      signedAt: new Date().toISOString()
    };

    const response = await fetch(
      `${this.baseUrl}/pets/${petId}/medical-records`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
          'X-WIA-Version': '1.0'
        },
        body: JSON.stringify(record)
      }
    );

    if (!response.ok) {
      throw new Error(`Failed to create record: ${response.status}`);
    }

    const result = await response.json();
    console.log(`Medical record created: ${result.recordId}`);

    // Auto-create follow-up appointment if needed
    if (record.soapNotes.plan.followUp.required) {
      await this.scheduleFollowUp(petId, record.soapNotes.plan.followUp);
    }

    return result;
  }

  async scheduleFollowUp(petId, followUpPlan) {
    // Calculate follow-up date
    const followUpDate = this.calculateFollowUpDate(followUpPlan.timeframe);

    await fetch(`${this.baseUrl}/appointments/schedule`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        petId: petId,
        appointmentType: 'followup',
        scheduledDate: followUpDate,
        notes: followUpPlan.instructions
      })
    });
  }

  calculateFollowUpDate(timeframe) {
    const now = new Date();
    const match = timeframe.match(/(\d+)\s*(day|week|month)/i);

    if (match) {
      const value = parseInt(match[1]);
      const unit = match[2].toLowerCase();

      switch(unit) {
        case 'day':
          now.setDate(now.getDate() + value);
          break;
        case 'week':
          now.setDate(now.getDate() + (value * 7));
          break;
        case 'month':
          now.setMonth(now.getMonth() + value);
          break;
      }
    }

    return now.toISOString();
  }
}

Example 2: Medication Management (Python)

from datetime import datetime, timedelta
from typing import List, Dict
import requests

class MedicationManager:
    """Manage pet medications, prescriptions, and refills"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = 'https://api.wia-pet-health.org/v1'
        self.headers = {
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json'
        }

    def prescribe_medication(self, pet_id: str, prescription: Dict) -> str:
        """
        Create new medication prescription

        Args:
            pet_id: Pet UUID
            prescription: {
                'drug_name': str,
                'dosage': str,
                'frequency': str,
                'duration': str,
                'quantity': int,
                'refills': int,
                'instructions': str
            }

        Returns:
            Prescription ID
        """
        prescription_data = {
            'medicationId': str(uuid.uuid4()),
            'drugName': prescription['drug_name'],
            'dosage': prescription['dosage'],
            'route': prescription.get('route', 'oral'),
            'frequency': prescription['frequency'],
            'duration': prescription['duration'],
            'prescribedDate': datetime.utcnow().isoformat(),
            'dispensedQuantity': prescription['quantity'],
            'refillsAuthorized': prescription['refills'],
            'instructions': prescription['instructions']
        }

        response = requests.post(
            f'{self.base_url}/pets/{pet_id}/medications/prescription',
            headers=self.headers,
            json=prescription_data
        )
        response.raise_for_status()

        result = response.json()

        # Set up refill reminders
        if prescription['refills'] > 0:
            self.schedule_refill_reminder(
                pet_id,
                result['medicationId'],
                prescription['duration']
            )

        return result['medicationId']

    def get_active_medications(self, pet_id: str) -> List[Dict]:
        """Get all currently active medications for pet"""
        response = requests.get(
            f'{self.base_url}/pets/{pet_id}/medications/active',
            headers=self.headers
        )
        response.raise_for_status()

        medications = response.json()['medications']

        # Enrich with days remaining
        for med in medications:
            med['days_remaining'] = self.calculate_days_remaining(
                med['prescribedDate'],
                med['duration']
            )
            med['refill_needed'] = med['days_remaining'] <= 7

        return medications

    def check_drug_interactions(self, pet_id: str, new_medication: str) -> Dict:
        """
        Check for potential drug interactions with active medications

        Args:
            pet_id: Pet UUID
            new_medication: Drug name to check

        Returns:
            Interaction warnings and severity
        """
        active_meds = self.get_active_medications(pet_id)
        active_drug_names = [m['drugName'] for m in active_meds]

        # Query drug interaction database
        response = requests.post(
            f'{self.base_url}/medications/interactions/check',
            headers=self.headers,
            json={
                'currentMedications': active_drug_names,
                'newMedication': new_medication
            }
        )

        interactions = response.json()

        if interactions['hasInteractions']:
            return {
                'safe': False,
                'warnings': interactions['warnings'],
                'severity': interactions['maxSeverity'],
                'recommendations': interactions['recommendations']
            }

        return {'safe': True, 'warnings': []}

    def calculate_days_remaining(self, prescribed_date: str, duration: str) -> int:
        """Calculate days of medication remaining"""
        prescribed = datetime.fromisoformat(prescribed_date.replace('Z', '+00:00'))

        # Parse duration (e.g., "14 days", "2 weeks", "1 month")
        import re
        match = re.search(r'(\d+)\s*(day|week|month)', duration, re.I)

        if match:
            value = int(match.group(1))
            unit = match.group(2).lower()

            if unit == 'day':
                duration_days = value
            elif unit == 'week':
                duration_days = value * 7
            elif unit == 'month':
                duration_days = value * 30

            end_date = prescribed + timedelta(days=duration_days)
            days_left = (end_date - datetime.utcnow()).days

            return max(0, days_left)

        return 0

    def schedule_refill_reminder(self, pet_id: str, medication_id: str,
                                 duration: str):
        """Schedule automatic refill reminder"""
        days = self.calculate_days_remaining(
            datetime.utcnow().isoformat(),
            duration
        )

        reminder_date = datetime.utcnow() + timedelta(days=days - 7)

        requests.post(
            f'{self.base_url}/reminders',
            headers=self.headers,
            json={
                'petId': pet_id,
                'reminderType': 'medication_refill',
                'medicationId': medication_id,
                'reminderDate': reminder_date.isoformat(),
                'message': 'Medication refill needed in 7 days'
            }
        )

    Severity      string    `json:"severity"`

    MonitoringPlan MonitoringPlan `json:"monitoringPlan"`
}

type MonitoringPlan struct {
    LabWorkFrequency string   `json:"labWorkFrequency"`
    ExamFrequency    string   `json:"examFrequency"`
    Parameters       []string `json:"parameters"`
    Targets          map[string]string `json:"targets"`
}

func NewChronicConditionManager(apiKey string) *ChronicConditionManager {
    return &ChronicConditionManager{
        apiKey:  apiKey,
        baseURL: "https://api.wia-pet-health.org/v1",
        client:  &http.Client{Timeout: 30 * time.Second},
    }
}

func (ccm *ChronicConditionManager) AddChronicCondition(petID string, condition ChronicCondition) error {
    condition.PetID = petID
    condition.ConditionID = generateUUID()
    condition.Status = "active"

    jsonData, err := json.Marshal(condition)
    if err != nil {
        return err
    }

    req, err := http.NewRequest("POST",
        fmt.Sprintf("%s/pets/%s/chronic-conditions", ccm.baseURL, petID),
        bytes.NewBuffer(jsonData))
    if err != nil {
        return err
    }

    req.Header.Set("Authorization", "Bearer "+ccm.apiKey)
    req.Header.Set("Content-Type", "application/json")

    resp, err := ccm.client.Do(req)
    if err != nil {
        return err
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return fmt.Errorf("failed to add condition: %d", resp.StatusCode)
    }

    // Schedule monitoring reminders
    ccm.scheduleMonitoringReminders(petID, condition)

    return nil
}

func (ccm *ChronicConditionManager) GetChronicConditions(petID string) ([]ChronicCondition, error) {
    req, err := http.NewRequest("GET",
        fmt.Sprintf("%s/pets/%s/chronic-conditions", ccm.baseURL, petID),
        nil)
    if err != nil {
        return nil, err
    }

    req.Header.Set("Authorization", "Bearer "+ccm.apiKey)

    resp, err := ccm.client.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    var result struct {
        Conditions []ChronicCondition `json:"conditions"`
    }

    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return nil, err
    }

    return result.Conditions, nil
}

        fmt.Printf("Error adding condition: %v\n", err)
        return
    }

    // Retrieve all chronic conditions
    conditions, err := manager.GetChronicConditions("pet-uuid-123")
    if err != nil {
        fmt.Printf("Error retrieving conditions: %v\n", err)
        return
    }

    for _, cond := range conditions {
        fmt.Printf("Condition: %s (Status: %s)\n", cond.Condition, cond.Status)
        fmt.Printf("  Monitoring: %s lab work, %s exams\n",
            cond.MonitoringPlan.LabWorkFrequency,
            cond.MonitoringPlan.ExamFrequency)
    }
}
