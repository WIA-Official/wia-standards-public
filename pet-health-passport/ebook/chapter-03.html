3.2 Vaccination Validation Algorithm

The vaccination validation algorithm checks if a pet's vaccinations meet the requirements for international travel:

O(n) where n = number of vaccinations

Algorithm: validate_vaccinations

  1. Get required vaccinations for destination country
  2. For each required vaccination, check if pet has a valid record
  3. Verify vaccination date is within validity period
  4. Verify issuer is an authorized veterinary authority
  5. Return validation result with any missing/expired vaccinations
vaccination_validator.rs
use chrono::{NaiveDate, Utc};

#[derive(Debug)]
pub struct ValidationResult {
    pub is_valid: bool,
    pub missing_vaccinations: Vec<String>,
    pub expired_vaccinations: Vec<String>,
    pub warnings: Vec<String>,
}

pub fn validate_vaccinations(
    passport: &PetPassport,
    destination: &str,
    travel_date: NaiveDate,
) -> ValidationResult {
    let required = get_required_vaccinations(destination);
    let mut missing = Vec::new();
    let mut expired = Vec::new();
    let mut warnings = Vec::new();

    for req_vax in &required {
        match find_vaccination(&passport.vaccinations, req_vax) {
            Some(vax) => {
                // Check if vaccination is still valid on travel date
                if vax.valid_until < travel_date {
                    expired.push(req_vax.clone());
                }

                // Check minimum waiting period (e.g., rabies needs 21 days)
                let days_since = (travel_date - vax.date_administered).num_days();
                if let Some(min_days) = get_minimum_wait_days(req_vax) {
                    if days_since < min_days as i64 {
                        warnings.push(format!(
                            "{}: {} more days required",
                            req_vax,
                            min_days as i64 - days_since
                        ));
                    }
                }
            }
            None => {
                missing.push(req_vax.clone());
            }
        }
    }

    ValidationResult {
        is_valid: missing.is_empty() && expired.is_empty(),
        missing_vaccinations: missing,
        expired_vaccinations: expired,
        warnings,
    }
}

fn get_required_vaccinations(country: &str) -> Vec<String> {
    match country {
        "EU" | "UK" => vec!["rabies".into()],
        "US" => vec!["rabies".into(), "dhpp".into()],
        "JP" => vec!["rabies".into()],
        "AU" => vec!["rabies".into(), "dhpp".into(), "leptospirosis".into()],
        "KR" => vec!["rabies".into()],
        _ => vec!["rabies".into()],
    }
}

fn get_minimum_wait_days(vaccination: &str) -> Option<u32> {
    match vaccination {
        "rabies" => Some(21),  // 21 days after primary vaccination
        _ => None,
    }
}

3.3 Quarantine Calculation Algorithm

Different countries have varying quarantine requirements based on the pet's origin country and health status:

[WARNING] Important Note

Quarantine requirements change frequently. Always verify current requirements with the destination country's official veterinary authority before travel.

quarantine_calculator.rs
#[derive(Debug)]
pub struct QuarantineResult {
    pub required: bool,
    pub duration_days: u32,
    pub location: QuarantineLocation,
    pub estimated_cost: Option<f64>,
    pub conditions: Vec<String>,
}

#[derive(Debug)]
pub enum QuarantineLocation {
    None,
    GovernmentFacility,
    ApprovedKennel,
    HomeQuarantine,
}

pub fn calculate_quarantine(
    origin_country: &str,
    destination_country: &str,
    passport: &PetPassport,
) -> QuarantineResult {
    // Check if origin is rabies-free country
    let origin_risk = get_rabies_risk_level(origin_country);
    let has_valid_rabies = has_valid_vaccination(passport, "rabies");
    let has_titer_test = has_valid_titer_test(passport);

    match destination_country {
        "AU" => calculate_australia_quarantine(origin_risk, passport),
        "NZ" => calculate_new_zealand_quarantine(origin_risk, passport),
        "JP" => calculate_japan_quarantine(origin_risk, has_titer_test),
        "UK" | "EU" => calculate_eu_quarantine(origin_risk, has_valid_rabies),
        "US" => calculate_us_quarantine(origin_risk, has_valid_rabies),
        "KR" => calculate_korea_quarantine(origin_risk, has_valid_rabies),
        _ => QuarantineResult {
            required: false,
            duration_days: 0,
            location: QuarantineLocation::None,
            estimated_cost: None,
            conditions: vec![],
        },
    }
}

fn calculate_japan_quarantine(
    origin_risk: RiskLevel,
    has_titer_test: bool,
) -> QuarantineResult {
    match (origin_risk, has_titer_test) {
        (RiskLevel::Low, true) => QuarantineResult {
            required: false,
            duration_days: 0,
            location: QuarantineLocation::None,
            estimated_cost: None,
            conditions: vec![
                "180-day waiting period after titer test".into(),
                "Must notify 40 days before arrival".into(),
            ],
        },
        (RiskLevel::Low, false) => QuarantineResult {
            required: true,
            duration_days: 180,
            location: QuarantineLocation::GovernmentFacility,
            estimated_cost: Some(3000.0),
            conditions: vec![
                "Titer test required upon arrival".into(),
            ],
        },
        _ => QuarantineResult {
            required: true,
            duration_days: 180,
            location: QuarantineLocation::GovernmentFacility,
            estimated_cost: Some(5000.0),
            conditions: vec![
                "Full quarantine required from high-risk countries".into(),
            ],
        },
    }
}

3.4 Travel Eligibility Algorithm

The comprehensive travel eligibility check combines vaccination validation, quarantine requirements, and document verification:

[CHECK] Eligibility Checklist

  • Valid passport (not expired)
  • Microchip readable and matching
  • Required vaccinations complete
  • Waiting periods satisfied
  • Health certificate issued (if required)
  • Import permit obtained (if required)
travel_eligibility.rs
#[derive(Debug)]
pub struct TravelEligibility {
    pub eligible: bool,
    pub status: EligibilityStatus,
    pub requirements_met: Vec<Requirement>,
    pub requirements_pending: Vec<Requirement>,
    pub earliest_travel_date: Option<NaiveDate>,
    pub quarantine_info: Option<QuarantineResult>,
}

#[derive(Debug)]
pub enum EligibilityStatus {
    Ready,              // Can travel immediately
    PendingDocuments,   // Missing documents
    PendingVaccination, // Vaccination not yet effective
    PendingQuarantine,  // Must complete quarantine
    Ineligible,         // Cannot travel (species banned, etc.)
}

pub fn check_travel_eligibility(
    passport: &PetPassport,
    origin: &str,
    destination: &str,
    travel_date: NaiveDate,
) -> TravelEligibility {
    let mut met = Vec::new();
    let mut pending = Vec::new();

    // 1. Passport validity check
    if passport.expiration_date.date_naive() > travel_date {
        met.push(Requirement::ValidPassport);
    } else {
        pending.push(Requirement::ValidPassport);
    }

    // 2. Microchip verification
    if verify_microchip(&passport.credential_subject.microchip_id) {
        met.push(Requirement::ValidMicrochip);
    } else {
        pending.push(Requirement::ValidMicrochip);
    }

    // 3. Vaccination validation
    let vax_result = validate_vaccinations(passport, destination, travel_date);
    if vax_result.is_valid {
        met.push(Requirement::VaccinationsComplete);
    } else {
        pending.push(Requirement::VaccinationsComplete);
    }

    // 4. Calculate quarantine requirements
    let quarantine = calculate_quarantine(origin, destination, passport);

    // 5. Check species restrictions
    if is_species_allowed(&passport.credential_subject.species, destination) {
        met.push(Requirement::SpeciesAllowed);
    } else {
        return TravelEligibility {
            eligible: false,
            status: EligibilityStatus::Ineligible,
            requirements_met: met,
            requirements_pending: vec![Requirement::SpeciesAllowed],
            earliest_travel_date: None,
            quarantine_info: Some(quarantine),
        };
    }

    // Calculate earliest possible travel date
    let earliest = calculate_earliest_date(&vax_result.warnings);

    TravelEligibility {
        eligible: pending.is_empty(),
        status: determine_status(&pending, &quarantine),
        requirements_met: met,
        requirements_pending: pending,
        earliest_travel_date: earliest,
        quarantine_info: Some(quarantine),
    }
}

3.5 Signature Verification

All credentials must be cryptographically verified to ensure authenticity:

signature_verifier.rs
use ed25519_dalek::{Signature, VerifyingKey, Verifier};

pub async fn verify_credential(
    credential: &serde_json::Value,
) -> Result<VerificationResult, VerificationError> {
    // 1. Extract proof from credential
    let proof = credential.get("proof")
        .ok_or(VerificationError::MissingProof)?;

    // 2. Get verification method (DID + key ID)
    let method = proof.get("verificationMethod")
        .and_then(|v| v.as_str())
        .ok_or(VerificationError::InvalidProof)?;

    // 3. Resolve DID to get public key
    let did_document = resolve_did(method).await?;
    let public_key = extract_public_key(&did_document, method)?;

    // 4. Reconstruct signing input
    let signing_input = create_verify_data(credential)?;

    // 5. Verify signature
    let signature_bytes = base58_decode(
        proof.get("proofValue")
            .and_then(|v| v.as_str())
            .ok_or(VerificationError::InvalidProof)?
    )?;

    let signature = Signature::from_bytes(&signature_bytes)?;
    let verifying_key = VerifyingKey::from_bytes(&public_key)?;

    verifying_key.verify(&signing_input, &signature)
        .map_err(|_| VerificationError::InvalidSignature)?;

    Ok(VerificationResult {
        verified: true,
        issuer: extract_issuer(credential)?,
        issued_at: extract_issuance_date(credential)?,
    })
}
