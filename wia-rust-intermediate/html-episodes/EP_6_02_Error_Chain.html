anyhow - 애플리케이션용 anyhow - For Applications

use anyhow::{Context, Result};
fn read_config(path: &str) -> Result<Config> {
let content = std::fs::read_to_string(path)
.context(format!("Failed to read {}", path))?;
let config: Config = toml::from_str(&content)
.context("Failed to parse config")?;
Ok(config)
}
fn main() -> Result<()> {
let config = read_config("app.toml")
.context("Failed to initialize app")?;
Ok(())
}

Context 메서드들 Context Methods

use anyhow::{Context, Result, bail!};
fn process_file(path: &str) -> Result<()> {
// context() - 에러에 메시지 추가
let data = std::fs::read(path)
.context("Cannot read file")?;
// with_context() - 지연 평가
let parsed = parse(&data)
.with_context(|| format!("Parse failed: {}", path))?;
// bail! - 즉시 에러 반환
if !parsed.is_valid() {
bail!("Invalid data in {}", path);
}
Ok(())
}

백트레이스로 추적 Track with Backtrace

// RUST_BACKTRACE=1 환경변수 설정
fn main() -> Result<()> {
if let Err(err) = run_app() {
// 에러 체인 출력
eprintln!("Error: {:?}", err);
// 원인 에러들 순회
for cause in err.chain() {
eprintln!(" Caused by: {}", cause);
}
std::process::exit(1);
}
Ok(())
}

라이브러리 vs 애플리케이션 Library vs Application

// ✅ 라이브러리: thiserror 사용
use thiserror::Error;
#[derive(Error, Debug)]
pub enum MyLibError {
#[error("validation failed")]
ValidationError,
}
// ✅ 애플리케이션: anyhow 사용
use anyhow::{Result, Context};
fn main() -> Result<()> {
my_lib::validate()
.context("App initialization failed")?;
Ok(())
}
📚 라이브러리
thiserror - 구체적 타입

🎯 애플리케이션
anyhow - 유연한 처리

💡 함께 사용 가능! 📚 Libraries
thiserror - Concrete types

🎯 Applications
anyhow - Flexible handling

💡 Can use together!

에러 체이닝 원칙 Error Chaining Principles

1️⃣ 컨텍스트 추가 - 각 레이어에서 의미 있는 정보 추가

2️⃣ 원인 보존 - source()로 하위 에러 체인 유지

3️⃣ 적절한 도구 - 라이브러리는 thiserror, 앱은 anyhow

4️⃣ 일관성 - 프로젝트 전체에서 동일한 패턴 사용

5️⃣ 로깅 - 에러 체인을 로그에 기록

6️⃣ 사용자 친화적 - 최상위 메시지는 사용자가 이해 가능하게 1️⃣ Add Context - Meaningful info at each layer

2️⃣ Preserve Cause - Keep error chain with source()

3️⃣ Right Tools - thiserror for libs, anyhow for apps

4️⃣ Consistency - Same pattern across project

5️⃣ Logging - Log error chain

6️⃣ User-Friendly - Top message understandable
