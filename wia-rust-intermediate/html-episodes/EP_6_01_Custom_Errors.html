기본 에러 Enum Basic Error Enum

use std::{fmt, error::Error};
#[derive(Debug)]
enum MyError {
NotFound(String),
InvalidInput { field: String, value: String },
Io(std::io::Error),
}
impl fmt::Display for MyError {
fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
match self {
MyError::NotFound(name) =>
write!(f, "'{}' not found", name),
MyError::InvalidInput { field, value } =>
write!(f, "Invalid {}: {}", field, value),
MyError::Io(e) => write!(f, "IO error: {}", e),
}
}
}

Error 트레이트 구현 Implement Error Trait

impl Error for MyError {
fn source(&self) -> Option<&(dyn Error + 'static)> {
match self {
MyError::Io(e) => Some(e),
_ => None,
}
}
}
// From 트레이트로 변환 편리하게
impl FromError> for MyError {
fn from(err: std::io::Error) -> Self {
MyError::Io(err)
}
}

thiserror로 간편하게 Easy with thiserror

use thiserror::Error;
#[derive(Error, Debug)]
enum DataStoreError {
#[error("data not found: {0}")]
NotFound(String),
#[error("invalid {field}: {value}")]
InvalidInput { field: String, value: String },
#[error("IO error")]
Io(#[from] std::io::Error),
#[error("parse error")]
Parse(#[from] std::num::ParseIntError),
}

에러 설계 원칙 Error Design Principles

1️⃣ 의미 있는 이름 - FileNotFoundError, InvalidInputError

2️⃣ 충분한 정보 - 에러 해결에 필요한 모든 데이터 포함

3️⃣ 적절한 세분화 - 너무 많지도, 적지도 않게

4️⃣ source() 구현 - 원인 에러 체이닝

5️⃣ From 트레이트 - 자동 변환 지원

6️⃣ 명확한 메시지 - 사용자가 이해할 수 있게 1️⃣ Meaningful Names - FileNotFoundError, InvalidInputError

2️⃣ Enough Info - Include all data needed to fix error

3️⃣ Right Granularity - Not too many, not too few

4️⃣ Implement source() - Chain underlying errors

5️⃣ From trait - Support auto conversion

6️⃣ Clear Messages - User-understandable
