macro_rules! - 선언적 매크로

macro_rules! - Declarative Macros

// 간단한 매크로 정의
macro_rules! say_hello {
() => {
println!("Hello!");
};
}
// 인자 받는 매크로
macro_rules! greet {
($name:expr) => {
println!("안녕, {}!", $name);
};
}
fn main() {
say_hello!(); // Hello!
greet!("Rust"); // 안녕, Rust!
}

매크로 패턴 매칭

Pattern Matching in Macros

macro_rules! calculate {
// 여러 패턴 매칭!
(add $a:expr, $b:expr) => {
$a + $b
};
(mul $a:expr, $b:expr) => {
$a * $b
};
}
fn main() {
let sum = calculate!(add 5, 3); // 8
let prod = calculate!(mul 4, 7); // 28
}

📝 $name:type 형식으로 캡처. type: expr, ident, ty, pat, stmt, block, item, tt 등 📝 Capture with $name:type. type: expr, ident, ty, pat, stmt, block, item, tt etc

가변 인자 매크로

Variadic Macros

// vec! 스타일 매크로
macro_rules! my_vec {
// $(...),* = 0개 이상 반복
( $( $x:expr ),* ) => {
{
let mut temp_vec = Vec::new();
$(
temp_vec.push($x);
)*
temp_vec
}
};
}
let v = my_vec![1, 2, 3, 4, 5];
println!("{:?}", v); // [1, 2, 3, 4, 5]

#[derive] Procedural Macros

// 컴파일러가 자동으로 코드 생성!
#[derive(Debug, Clone, PartialEq)]
struct User {
name: String,
age: u32,
}
// serde로 직렬화
use serde::{Serialize, Deserialize};
#[derive(Serialize, Deserialize)]
struct Config {
host: String,
port: u16,
}
// JSON으로 자동 변환 가능!
