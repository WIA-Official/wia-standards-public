Clap으로 인자 파싱

Parsing Arguments with Clap

use clap::Parser;
#[derive(Parser)]
#[command(name = "greet")]
#[command(about = "A friendly greeter")]
struct Args {
/// Name to greet
name: String,
/// Number of times
#[arg(short, long, default_value_t = 1)]
count: u8,
}
fn main() {
let args = Args::parse();
for _ in 0..args.count {
println!("Hello, {}!", args.name);
}
}

Implementing Subcommands

use clap::{Parser, Subcommand};
#[derive(Parser)]
struct Cli {
#[command(subcommand)]
command: Commands,
}
#[derive(Subcommand)]
enum Commands {
Add { name: String },
List,
Remove { id: u32 },
}
fn main() {
match Cli::parse().command {
Commands::Add { name } => println!("Adding {}", name),
Commands::List => println!("Listing all"),
Commands::Remove { id } => println!("Removing {}", id),
}
}

Reading/Writing Files

use std::fs;
use anyhow::Result;
fn read_file(path: &str) -> Result<String> {
let content = fs::read_to_string(path)?;
Ok(content)
}
fn write_file(path: &str, data: &str) -> Result<()> {
fs::write(path, data)?;
Ok(())
}

}

단어 카운터 CLI

Word Counter CLI

use clap::Parser;
use std::fs;
#[derive(Parser)]
struct Args {
/// File to count words
file: String,
#[arg(short, long)]
lines: bool,
}
fn main() -> anyhow::Result<()> {
let args = Args::parse();
let content = fs::read_to_string(&args.file)?;
if args.lines {
println!("Lines: {}", content.lines().count());
} else {
println!("Words: {}", content.split_whitespace().count());
}
Ok(())
}
