File System Traversal

use std::fs;
use std::path::Path;
fn list_files(dir: &Path) -> std::io::Result<()> {
for entry in fs::read_dir(dir)? {
let entry = entry?;
let path = entry.path();
let metadata = entry.metadata()?;
if metadata.is_dir() {
println!("[DIR] {:?}", path);
} else {
println!("[FILE] {:?} ({} bytes)",
path, metadata.len());
}
}
Ok(())
}

Recursive Directory Walk

// Cargo.toml: walkdir = "2"
use walkdir::WalkDir;
fn find_rust_files(dir: &str) {
for entry in WalkDir::new(dir)
.into_iter()
.filter_map(|e| e.ok())
.filter(|e| {
e.path().extension()
.map(|ext| ext == "rs")
.unwrap_or(false)
})
{
println!("Found: {:?}", entry.path());
}
}

Collecting System Info

// Cargo.toml: sysinfo = "0.30"
use sysinfo::System;
fn system_info() {
let mut sys = System::new_all();
sys.refresh_all();
println!("총 메모리: {} GB", sys.total_memory() / 1024 / 1024 / 1024);
println!("사용 메모리: {} GB", sys.used_memory() / 1024 / 1024 / 1024);
println!("OS: {:?}", System::name());
println!("호스트: {:?}", System::host_name());
// 프로세스 목록
for (pid, process) in sys.processes() {
println!("[{}] {}", pid, process.name());
}
}

완성 예제

시스템 모니터 도구

System Monitor Tool

use sysinfo::System;
use std::{thread, time::Duration};
fn main() {
let mut sys = System::new_all();
loop {
sys.refresh_all();
print!("\x1B[2J\x1B[1;1H"); // 화면 클리어
println!("=== 시스템 모니터 ===");
println!("메모리: {:.1}%",
sys.used_memory() as f64 / sys.total_memory() as f64 * 100.0);
println!("\n상위 CPU 사용 프로세스:");
let mut procs: Vec<_> = sys.processes().values().collect();
procs.sort_by(|a, b| b.cpu_usage().partial_cmp(&a.cpu_usage()).unwrap());
for p in procs.iter().take(5) {
println!(" {} - {:.1}%", p.name(), p.cpu_usage());
}
thread::sleep(Duration::from_secs(1));
}
}
