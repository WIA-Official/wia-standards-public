6.1 전송 계층 프로토콜

WIA-CANCER-METABOLISM 표준은 다양한 전송 계층 프로토콜을 지원하여 다양한 네트워크 환경과 사용 사례에 대응합니다.

6.1.1 HTTP/HTTPS

HTTP/HTTPS는 가장 일반적으로 사용되는 전송 프로토콜입니다. WIA API는 기본적으로 HTTPS(TLS 1.3)를 사용하여 전송 중 데이터를 암호화합니다.

프로토콜 포트 암호화 사용 사례
HTTP 80 없음 개발/테스트만 (프로덕션 비권장)
HTTPS (TLS 1.2) 443 TLS 1.2 레거시 시스템 호환성
HTTPS (TLS 1.3) 443 TLS 1.3 권장 (향상된 보안 및 성능)
HTTP/2 443 TLS 1.3 다중 요청 최적화
HTTP/3 (QUIC) 443 TLS 1.3 고지연 네트워크, 모바일

6.1.2 WebSocket

실시간 대사체 모니터링과 같은 양방향 통신이 필요한 경우 WebSocket 프로토콜을 사용합니다:

// WebSocket 연결 수립
const ws = new WebSocket('wss://api.wiastandards.com/ws/v1/realtime');

// 인증
ws.onopen = () => {
  ws.send(JSON.stringify({
    type: 'auth',
    token: 'Bearer eyJhbGciOiJSUzI1NiIs...'
  }));
};

// 프로파일 구독
ws.send(JSON.stringify({
  type: 'subscribe',
  resource: 'profiles',
  filters: {
    cancerType: 'C34.1',
    realtime: true
  }
}));

// 실시간 업데이트 수신
ws.onmessage = (event) => {
  const data = JSON.parse(event.data);

  if (data.type === 'profile.created') {
    console.log('새 프로파일:', data.payload.profileId);
  } else if (data.type === 'biomarker.alert') {
    console.log('바이오마커 경고:', data.payload.message);
  }
};

// 연결 종료
ws.onclose = (event) => {
  console.log('WebSocket 연결 종료:', event.code, event.reason);
};

6.1.3 gRPC

고성능 서버 간 통신을 위해 gRPC 프로토콜을 지원합니다:

// Protocol Buffers 정의 (cancer_metabolism.proto)
syntax = "proto3";

package wia.cancer_metabolism.v1;

service MetaboliteProfileService {
  rpc CreateProfile(CreateProfileRequest) returns (ProfileResponse);
  rpc GetProfile(GetProfileRequest) returns (ProfileResponse);
  rpc ListProfiles(ListProfilesRequest) returns (ListProfilesResponse);
  rpc StreamProfiles(StreamProfilesRequest) returns (stream ProfileResponse);
}

message CreateProfileRequest {
  string patient_id = 1;
  google.protobuf.Timestamp sample_date = 2;
  SampleType sample_type = 3;
  string cancer_type = 4;
  repeated Metabolite metabolites = 5;
}

message Metabolite {
  string metabolite_id = 1;
  string name = 2;
  double concentration = 3;
  string unit = 4;
  string pathway = 5;
}

message ProfileResponse {
  string profile_id = 1;
  string patient_id = 2;
  google.protobuf.Timestamp sample_date = 3;
  repeated Metabolite metabolites = 4;
}

enum SampleType {
  SAMPLE_TYPE_UNSPECIFIED = 0;
  SAMPLE_TYPE_SERUM = 1;
  SAMPLE_TYPE_PLASMA = 2;
  SAMPLE_TYPE_URINE = 3;
  SAMPLE_TYPE_TISSUE = 4;
}

6.4.2 종단 간 암호화 (E2EE)

특별히 민감한 데이터는 애플리케이션 레벨에서 추가 암호화를 적용합니다:

// 클라이언트 측 암호화 (Web Crypto API)
async function encryptSensitiveData(data, publicKey) {
  // 데이터를 JSON 문자열로 변환
  const jsonData = JSON.stringify(data);
  const encoder = new TextEncoder();
  const encodedData = encoder.encode(jsonData);

  // AES-GCM으로 암호화
  const key = await crypto.subtle.generateKey(
    { name: 'AES-GCM', length: 256 },
    true,
    ['encrypt', 'decrypt']
  );

  const iv = crypto.getRandomValues(new Uint8Array(12));

  const encryptedData = await crypto.subtle.encrypt(
    { name: 'AES-GCM', iv: iv },
    key,
    encodedData
  );

  // AES 키를 RSA 공개키로 암호화
  const exportedKey = await crypto.subtle.exportKey('raw', key);
  const encryptedKey = await crypto.subtle.encrypt(
    { name: 'RSA-OAEP' },
    publicKey,
    exportedKey
  );

  return {
    encryptedData: arrayBufferToBase64(encryptedData),
    encryptedKey: arrayBufferToBase64(encryptedKey),
    iv: arrayBufferToBase64(iv),
    algorithm: 'AES-GCM-256-RSA-OAEP'
  };
}

// 암호화된 데이터 전송
const profile = {
  patientId: 'PT-2025-0001-ANON',
  metabolites: [/* ... */]
};

const encrypted = await encryptSensitiveData(profile, publicKey);

await fetch('https://api.wiastandards.com/api/v1/profiles/encrypted', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': `Bearer ${accessToken}`
  },

6.4.3 디지털 서명

데이터 무결성과 출처 인증을 보장하기 위해 디지털 서명을 사용합니다:

// 데이터 서명 (Node.js)
const crypto = require('crypto');
const fs = require('fs');

function signProfile(profile, privateKeyPath) {
  // 프로파일을 정규화된 JSON으로 변환
  const canonicalJson = JSON.stringify(profile, Object.keys(profile).sort());

  // 개인키 로드
  const privateKey = fs.readFileSync(privateKeyPath, 'utf8');

  // SHA-256 해시 후 RSA 서명
  const sign = crypto.createSign('SHA256');
  sign.update(canonicalJson);
  sign.end();

  const signature = sign.sign(privateKey, 'base64');

  return {
    data: profile,
    signature: signature,
    algorithm: 'RS256',
    signedAt: new Date().toISOString()
  };
}

// 서명 검증
function verifySignature(signedProfile, publicKeyPath) {
  const { data, signature, algorithm } = signedProfile;

  const canonicalJson = JSON.stringify(data, Object.keys(data).sort());
  const publicKey = fs.readFileSync(publicKeyPath, 'utf8');

  const verify = crypto.createVerify('SHA256');
  verify.update(canonicalJson);
  verify.end();

  return verify.verify(publicKey, signature, 'base64');
}

// 사용 예제
const profile = { /* ... */ };
const signed = signProfile(profile, './private-key.pem');

// API 전송
await fetch('https://api.wiastandards.com/api/v1/profiles/signed', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify(signed)
});

6.6 연결 관리 및 재시도 전략

안정적인 통신을 위한 연결 관리와 에러 복구 메커니즘:

6.6.1 지수 백오프 재시도

async function fetchWithRetry(url, options, maxRetries = 3) {
  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      const response = await fetch(url, options);

      if (response.ok) {
        return response;
      }

      // 429 (Too Many Requests)는 재시도
      if (response.status === 429) {
        const retryAfter = response.headers.get('Retry-After');
        const delay = retryAfter ? parseInt(retryAfter) * 1000 : Math.pow(2, attempt) * 1000;

        console.log(`속도 제한, ${delay}ms 후 재시도...`);
        await sleep(delay);
        continue;
      }

      // 500-599 서버 오류는 재시도
      if (response.status >= 500) {
        const delay = Math.pow(2, attempt) * 1000;
        console.log(`서버 오류, ${delay}ms 후 재시도...`);
        await sleep(delay);
        continue;
      }

      return response;

    } catch (error) {
      if (attempt === maxRetries) {
        throw error;
      }

      const delay = Math.pow(2, attempt) * 1000;
      console.log(`네트워크 오류, ${delay}ms 후 재시도...`);
      await sleep(delay);
    }
  }

  throw new Error(`${maxRetries}회 재시도 후 실패`);
}

function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

6.6.2 연결 풀링

// HTTP/2 연결 풀 (Node.js)
const http2 = require('http2');

class HTTP2ConnectionPool {
  constructor(origin, maxConnections = 10) {
    this.origin = origin;
    this.maxConnections = maxConnections;
    this.pool = [];
  }

  getConnection() {
    // 사용 가능한 연결 찾기
    let connection = this.pool.find(conn => !conn.busy);

    if (!connection && this.pool.length < this.maxConnections) {
      // 새 연결 생성
      const client = http2.connect(this.origin);
      connection = { client, busy: false };
      this.pool.push(connection);
    }

    if (connection) {
      connection.busy = true;
      return connection.client;
    }

    // 모든 연결이 사용 중이면 대기
    return new Promise(resolve => {
      const interval = setInterval(() => {
        const conn = this.pool.find(c => !c.busy);
        if (conn) {
          clearInterval(interval);
          conn.busy = true;
          resolve(conn.client);
        }
      }, 100);
    });
  }

  releaseConnection(client) {
    const connection = this.pool.find(conn => conn.client === client);
    if (connection) {
      connection.busy = false;
    }
  }
}

  • JSON, Protocol Buffers, MessagePack 등 여러 직렬화 형식을 통해 성능과 호환성 요구사항에 대응합니다
  • 실시간 대사체 모니터링을 위해 SSE, WebSocket, gRPC 스트리밍을 지원합니다
  • TLS 1.3, 종단 간 암호화, 디지털 서명을 통해 의료 데이터의 기밀성과 무결성을 보장합니다
  • 압축, 배치 요청, 연결 풀링, 지수 백오프 재시도를 통해 네트워크 효율성과 안정성을 극대화합니다
