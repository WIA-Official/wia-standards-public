7.1.3 EHR 통합 워크플로우

// Epic FHIR 통합 예제 (Node.js)
const axios = require('axios');

class EpicFHIRIntegration {
  constructor(fhirBaseUrl, clientId, clientSecret) {
    this.fhirBaseUrl = fhirBaseUrl;
    this.clientId = clientId;
    this.clientSecret = clientSecret;
    this.accessToken = null;
  }

  // OAuth 2.0 인증
  async authenticate() {
    const response = await axios.post(`${this.fhirBaseUrl}/oauth2/token`, {
      grant_type: 'client_credentials',
      client_id: this.clientId,
      client_secret: this.clientSecret
    });

    this.accessToken = response.data.access_token;
  }

  // WIA 프로파일을 FHIR Observation으로 변환
  convertToFHIRObservation(wiaProfile, metabolite) {
    return {
      resourceType: 'Observation',
      status: 'final',
      category: [{
        coding: [{
          system: 'http://terminology.hl7.org/CodeSystem/observation-category',
          code: 'laboratory'
        }]
      }],
      code: {
        coding: [{
          system: 'https://hmdb.ca',
          code: metabolite.metaboliteId,
          display: metabolite.name
        }]
      },
      subject: {
        reference: `Patient/${wiaProfile.patientId}`
      },
      effectiveDateTime: wiaProfile.sampleDate,
      valueQuantity: {
        value: metabolite.concentration,
        unit: metabolite.unit,
        system: 'http://unitsofmeasure.org'
      }
    };
  }

  // FHIR 서버에 Observation 생성
  async createObservation(observation) {
    const response = await axios.post(
      `${this.fhirBaseUrl}/Observation`,
      observation,
      {
        headers: {
          'Authorization': `Bearer ${this.accessToken}`,
          'Content-Type': 'application/fhir+json'
        }
      }
    );

    return response.data;
  }

  // WIA 프로파일을 EHR에 통합
  async integrateProfile(wiaProfile) {
    await this.authenticate();

    const observations = [];

    for (const metabolite of wiaProfile.metabolites) {
      const fhirObs = this.convertToFHIRObservation(wiaProfile, metabolite);
      const created = await this.createObservation(fhirObs);
      observations.push(created);
    }

    return observations;
  }
}

);

// LIMS 통합 워크플로우
class LIMSIntegration {
  constructor(limsApiUrl, wiaApiUrl) {
    this.limsApiUrl = limsApiUrl;
    this.wiaApiUrl = wiaApiUrl;
  }

  // 1. LIMS에서 새 샘플 정보 가져오기
  async fetchNewSamples() {
    const response = await fetch(`${this.limsApiUrl}/samples/pending`, {
      headers: { 'Authorization': `Bearer ${this.limsToken}` }
    });

    return await response.json();
  }

  // 2. 질량분석기에서 원시 데이터 가져오기
  async fetchMassSpecData(sampleId) {
    // Waters, Thermo, Agilent 등의 기기 인터페이스
    const response = await fetch(`${this.limsApiUrl}/instruments/data/${sampleId}`);
    return await response.json();
  }

  // 3. 원시 데이터를 WIA 형식으로 변환
  convertRawDataToWIA(rawData, sampleInfo) {
    return {
      patientId: sampleInfo.patientId,
      sampleDate: sampleInfo.collectionDate,
      sampleType: sampleInfo.type,
      metabolites: rawData.peaks.map(peak => ({
        metaboliteId: peak.hmdbId,
        name: peak.compoundName,
        concentration: peak.quantification,
        unit: peak.unit,
        detectionMethod: 'LC-MS/MS',
        confidence: peak.identificationScore
      })),
      qcMetrics: {
        instrumentId: rawData.instrumentId,
        batchId: sampleInfo.batchId,
        internalStandards: rawData.internalStandards
      }
    };
  }

  // 4. WIA API에 프로파일 생성
  async createWIAProfile(wiaData) {
    const response = await fetch(`${this.wiaApiUrl}/api/v1/profiles`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.wiaToken}`
      },
      body: JSON.stringify(wiaData)
    });

    return await response.json();
  }

  // 5. LIMS에 결과 업데이트
  async updateLIMSWithResults(sampleId, wiaProfileId, results) {
    await fetch(`${this.limsApiUrl}/samples/${sampleId}/results`, {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.limsToken}`
      },
      body: JSON.stringify({
        status: 'completed',
        wiaProfileId: wiaProfileId,
        results: results,
        completedAt: new Date().toISOString()
      })
    });
  }

  // 전체 워크플로우 실행
  async processNewSamples() {
    const samples = await this.fetchNewSamples();

    for (const sample of samples) {
      try {
        // 원시 데이터 가져오기
        const rawData = await this.fetchMassSpecData(sample.id);

        // WIA 형식으로 변환
        const wiaData = this.convertRawDataToWIA(rawData, sample);

        // WIA 프로파일 생성
        const profile = await this.createWIAProfile(wiaData);

        // LIMS 업데이트
        await this.updateLIMSWithResults(sample.id, profile.profileId, wiaData.metabolites);

        console.log(`샘플 ${sample.id} 처리 완료 → WIA 프로파일 ${profile.profileId}`);

      } catch (error) {
        console.error(`샘플 ${sample.id} 처리 실패:`, error);
        // 에러 처리 및 LIMS에 실패 상태 기록
      }
    }
  }
}

// 정기적으로 실행 (예: 5분마다)
const lims = new LIMSIntegration('https://lims.hospital.com/api', 'https://api.wiastandards.com');
setInterval(() => lims.processNewSamples(), 5 * 60 * 1000);

// 바이오뱅크에서 특정 조건에 맞는 샘플 검색
async function queryBiobank(criteria) {
  const query = {
    diagnosis: criteria.cancerType,
    sampleType: criteria.sampleType,
    hasMetabolomicsData: true,
    minAge: criteria.minAge,
    maxAge: criteria.maxAge
  };

  // BBMRI Directory API 호출
  const response = await fetch('https://directory.bbmri-eric.eu/api/v1/samples/search', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(query)
  });

  const samples = await response.json();

  // WIA 프로파일 가져오기
  const profiles = await Promise.all(
    samples.map(async (sample) => {
      if (sample.wiaProfileId) {
        return await fetch(`https://api.wiastandards.com/api/v1/profiles/${sample.wiaProfileId}`);
      }
    })
  );

  return profiles.filter(p => p != null);
}

  sampleType: 'serum',
  minAge: 40,
  maxAge: 70
});

7.4.3 통합 분석 파이프라인

# Python 통합 분석 스크립트
import pandas as pd
from wia_cancer_metabolism import WIAClient
from genomics_api import GenomicsAPI
from rnaseq_api import RNASeqAPI

class MultiOmicsIntegration:
    def __init__(self):
        self.wia_client = WIAClient(api_url='https://api.wiastandards.com')
        self.genomics = GenomicsAPI()
        self.rnaseq = RNASeqAPI()

    def integrate_patient_data(self, patient_id):
        # 1. WIA 대사체 데이터 가져오기
        metabolomics = self.wia_client.profiles.search(patient_id=patient_id)

        # 2. 게놈 데이터 가져오기
        genomics = self.genomics.get_mutations(patient_id)

        # 3. 전사체 데이터 가져오기
        transcriptomics = self.rnaseq.get_expression(patient_id)

        # 4. 데이터 통합
        integrated = {
            'patient_id': patient_id,
            'metabolites': metabolomics.to_dict(),
            'mutations': genomics.to_dict(),
            'gene_expression': transcriptomics.to_dict()
        }

        # 5. 경로 분석
        pathways = self.analyze_pathways(integrated)

        return {
            **integrated,
            'pathways': pathways
        }

    def analyze_pathways(self, data):
        # KEGG 경로 농축 분석
        from pathway_enrichment import KEGGEnrichment

        enrichment = KEGGEnrichment()

        # 대사체에서 변화된 것들
        altered_metabolites = [
            m['metaboliteId'] for m in data['metabolites']
            if m.get('interpretation') == 'elevated'
        ]

        # 발현 증가된 유전자
        upregulated_genes = [
            gene for gene, expr in data['gene_expression'].items()
            if expr['logFC'] > 1.5
        ]

        # 통합 경로 분석
        result = enrichment.analyze(
            genes=upregulated_genes,
            metabolites=altered_metabolites
        )

        return result

# 사용 예제
integration = MultiOmicsIntegration()
patient_profile = integration.integrate_patient_data('PT-2025-0001-ANON')

print(f"농축된 경로: {len(patient_profile['pathways'])}")
for pathway in patient_profile['pathways'][:5]:
    print(f"  - {pathway['name']}: p={pathway['pValue']:.4f}")

7.5.3 데이터 백업 및 재해 복구

# 자동 백업 스크립트 (AWS)
#!/bin/bash

# 환경 변수
BACKUP_BUCKET="s3://wia-backups/cancer-metabolism"
DB_HOST="wia-db.cluster-xxxxx.us-east-1.rds.amazonaws.com"
DB_NAME="wia_cancer_metabolism"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)

# 1. 데이터베이스 백업
pg_dump -h $DB_HOST -U wia_admin $DB_NAME | \
  gzip > /tmp/db_backup_${TIMESTAMP}.sql.gz

# 2. S3에 업로드
aws s3 cp /tmp/db_backup_${TIMESTAMP}.sql.gz \
  ${BACKUP_BUCKET}/database/ \
  --storage-class GLACIER

# 3. 파일 스토리지 백업
aws s3 sync s3://wia-profiles/ ${BACKUP_BUCKET}/profiles/ \
  --storage-class STANDARD_IA

# 4. 오래된 백업 정리 (30일 이상)
aws s3 ls ${BACKUP_BUCKET}/database/ | \
  awk '{print $4}' | \
  while read file; do
    file_date=$(echo $file | cut -d'_' -f3 | cut -d'.' -f1)
    days_old=$(( ($(date +%s) - $(date -d $file_date +%s)) / 86400 ))
    if [ $days_old -gt 30 ]; then
      aws s3 rm ${BACKUP_BUCKET}/database/$file
    fi
  done

echo "백업 완료: db_backup_${TIMESTAMP}.sql.gz"
