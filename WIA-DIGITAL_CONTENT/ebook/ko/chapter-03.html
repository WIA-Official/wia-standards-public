3. 메타데이터 관리 구현

콘텐츠 메타데이터를 생성하고 관리하는 서비스를 구현합니다.

3.1 데이터 모델 정의

// src/models/Content.ts
import { Entity, PrimaryGeneratedColumn, Column, CreateDateColumn, UpdateDateColumn } from 'typeorm';

@Entity('contents')
export class Content {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Column({ type: 'jsonb' })
  title: {
    ko: string;
    en?: string;
    ja?: string;
  };

  @Column()
  type: 'webtoon' | 'novel' | 'video';

  @Column({ type: 'jsonb' })
  creator: {
    id: string;
    name: string;
    verified: boolean;
  };

  @Column('text', { array: true })
  genre: string[];

  @Column()
  ageRating: string;

  @Column({ type: 'jsonb' })
  rights: {
    copyrightHolder: string;
    territory: string[];
    validFrom: Date;
    validUntil: Date;
  };

  @Column({ type: 'jsonb' })
  technical: {
    format: string;
    resolution: string;
    fileSize: number;
  };

  @Column({ default: 'draft' })
  status: 'draft' | 'published' | 'archived';

  @CreateDateColumn()
  createdAt: Date;

  @UpdateDateColumn()
  updatedAt: Date;
}

3.2 메타데이터 서비스

// src/services/MetadataService.ts
import { WIADigitalContent } from '@wia/digital-content-sdk';
import { Content } from '../models/Content';

export class MetadataService {
  private wia: WIADigitalContent;

  constructor() {
    this.wia = new WIADigitalContent({
      apiKey: process.env.WIA_API_KEY,
      apiSecret: process.env.WIA_API_SECRET
    });
  }

  async createMetadata(contentData: Partial): Promise {
    // WIA 메타데이터 검증
    const validated = await this.wia.validateMetadata({
      '@context': 'https://wia.org/standards/digital-content/v1.0',
      '@type': 'Webtoon',
      title: contentData.title,
      creator: contentData.creator,
      genre: contentData.genre,
      ageRating: contentData.ageRating,
      rights: contentData.rights
    });

    if (!validated.valid) {
      throw new Error(`메타데이터 검증 실패: ${validated.errors.join(', ')}`);
    }

    // 데이터베이스에 저장
    const content = new Content();
    Object.assign(content, contentData);

    return await content.save();
  }

  async getMetadata(contentId: string): Promise {
    const content = await Content.findOne({ where: { id: contentId } });
    if (!content) {
      throw new Error('콘텐츠를 찾을 수 없습니다');
    }
    return content;
  }
}

4. DRM 암호화 구현

콘텐츠 파일을 암호화하고 사용자에게 안전하게 제공하는 시스템을 구축합니다.

4.1 DRM 서비스

// src/drm/DRMService.ts
import crypto from 'crypto';
import fs from 'fs/promises';
import { WIADRM } from '@wia/digital-content-sdk';

export class DRMService {
  private drm: WIADRM;
  private encryptionKey: Buffer;

  constructor() {
    this.drm = new WIADRM({
      apiKey: process.env.WIA_API_KEY
    });
    this.encryptionKey = Buffer.from(process.env.WIA_ENCRYPTION_KEY, 'hex');
  }

  async encryptContent(filePath: string, contentId: string, userId: string): Promise<{
    encryptedPath: string;
    watermark: string;
  }> {
    // 1. 파일 읽기
    const fileBuffer = await fs.readFile(filePath);

    // 2. 워터마크 생성
    const watermark = this.generateWatermark(contentId, userId);

    // 3. AES-256-GCM 암호화
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv('aes-256-gcm', this.encryptionKey, iv);

    const encrypted = Buffer.concat([
      cipher.update(fileBuffer),
      cipher.final()
    ]);

    const authTag = cipher.getAuthTag();

    // 4. 암호화 메타데이터 저장
    const encryptedData = {
      iv: iv.toString('hex'),
      authTag: authTag.toString('hex'),
      content: encrypted.toString('base64'),
      watermark: watermark
    };

    // 5. 암호화된 파일 저장
    const encryptedPath = `${filePath}.encrypted`;
    await fs.writeFile(encryptedPath, JSON.stringify(encryptedData));

    // 6. WIA 라이선스 서버에 키 등록
    await this.drm.registerLicense({
      contentId,
      userId,
      encryptionKey: this.encryptionKey.toString('hex'),
      iv: iv.toString('hex'),
      permissions: {
        validUntil: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000), // 30일
        maxDevices: 3,
        downloadable: true,
        offlineAccess: true
      }
    });

    return { encryptedPath, watermark };
  }

  private generateWatermark(contentId: string, userId: string): string {
    const timestamp = Date.now();
    const data = `${contentId}:${userId}:${timestamp}`;
    return crypto.createHmac('sha256', this.encryptionKey)
      .update(data)
      .digest('hex');
  }

  async issueDecryptionKey(contentId: string, userId: string, deviceId: string): Promise<{
    key: string;
    iv: string;
  }> {
    // 사용자 권한 확인
    const hasAccess = await this.checkUserAccess(contentId, userId);
    if (!hasAccess) {
      throw new Error('접근 권한이 없습니다');
    }

    // WIA 라이선스 서버에서 키 발급
    const license = await this.drm.issueKey({
      contentId,
      userId,
      deviceId
    });

    return {
      key: license.encryptionKey,
      iv: license.iv
    };
  }

  private async checkUserAccess(contentId: string, userId: string): Promise {
    // 구매 내역 확인 로직
    // 실제 구현에서는 데이터베이스 조회
    return true;
  }
}

5. API 서버 개발

RESTful API 엔드포인트를 구현하여 프론트엔드와 통신합니다.

5.1 라우터 정의

엔드포인트 메서드 설명 인증
/api/v1/contents GET 콘텐츠 목록 조회 선택
/api/v1/contents POST 새 콘텐츠 등록 필수
/api/v1/contents/:id GET 콘텐츠 상세 조회 선택
/api/v1/drm/unlock POST 복호화 키 발급 필수
/api/v1/settlements GET 수익 정산 내역 필수

5.2 컨트롤러 구현

// src/controllers/ContentController.ts
import { Request, Response } from 'express';
import { MetadataService } from '../services/MetadataService';
import { DRMService } from '../drm/DRMService';

export class ContentController {
  private metadataService: MetadataService;
  private drmService: DRMService;

  constructor() {
    this.metadataService = new MetadataService();
    this.drmService = new DRMService();
  }

  // 콘텐츠 등록
  async createContent(req: Request, res: Response) {
    try {
      const { title, creator, genre, file } = req.body;

      // 1. 메타데이터 생성
      const content = await this.metadataService.createMetadata({
        title,
        creator,
        genre,
        type: 'webtoon',
        ageRating: '15+',
        rights: {
          copyrightHolder: creator.name,
          territory: ['KR', 'JP', 'US'],
          validFrom: new Date(),
          validUntil: new Date('2030-12-31')
        }
      });

      // 2. 파일 암호화
      const { encryptedPath, watermark } = await this.drmService.encryptContent(
        file.path,
        content.id,
        creator.id
      );

      res.status(201).json({
        success: true,
        data: {
          contentId: content.id,
          encryptedPath,
          watermark
        }
      });
    } catch (error) {
      res.status(500).json({
        success: false,
        error: error.message
      });
    }
  }

  // 콘텐츠 조회
  async getContent(req: Request, res: Response) {
    try {
      const { id } = req.params;
      const content = await this.metadataService.getMetadata(id);

      res.json({
        success: true,
        data: content
      });
    } catch (error) {
      res.status(404).json({
        success: false,
        error: error.message
      });
    }
  }

  // DRM 키 발급
  async unlockContent(req: Request, res: Response) {
    try {
      const { contentId, deviceId } = req.body;
      const userId = req.user.id; // JWT에서 추출

      const decryptionKey = await this.drmService.issueDecryptionKey(
        contentId,
        userId,
        deviceId
      );

      res.json({
        success: true,
        data: decryptionKey
      });
    } catch (error) {
      res.status(403).json({
        success: false,
        error: error.message
      });
    }
  }
}
