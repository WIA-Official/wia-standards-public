6장: 디지털 트윈 통합

실시간 IoT 연결 및 건물 성능 모니터링

6.1 디지털 트윈 소개

디지털 트윈(Digital Twin)은 센서, IoT 장치 및 운영 시스템의 실시간 데이터로 지속적으로 업데이트되는 물리적 자산의 동적 실시간 디지털 표현입니다. 정적 BIM 모델과 달리 디지털 트윈은 건물의 운영 수명주기 전반에 걸쳐 발전하며, 현재 상태를 반영하고, 미래 상태를 예측하고, 지능적인 의사결정을 가능하게 합니다.

BIM에서 디지털 트윈으로의 전환은 건물 정보의 설계 문서에서 운영 인텔리전스로의 발전을 나타냅니다. BIM이 설계 및 준공 상태를 포착하는 반면, 디지털 트윈은 실시간 센서 데이터, 유지보수 기록, 재실 패턴 및 환경 조건을 통합하여 운영 상태를 유지합니다.

BIM에서 디지털 트윈으로의 진화

디지털 트윈은 BIM 기반 위에 구축되어 기하학적 데이터, 공간 관계 및 자산 정보를 상속하면서 실시간 연결, 시뮬레이션 기능 및 예측 분석을 추가합니다. 이 통합은 건물의 운영 두뇌 역할을 하는 살아있는 모델을 만듭니다.

6.3 IoT 센서 통합

IoT 센서를 디지털 트윈에 연결하려면 센서 데이터 포인트를 특정 BIM 요소에 매핑해야 합니다. WIA-BIM은 센서 등록, 데이터 매핑 및 실시간 스트리밍에 대한 표준화된 접근 방식을 정의합니다. 

// WIA-BIM 디지털 트윈: 센서 통합 스키마
interface SensorRegistration {
    sensorId: string;                 // 고유 센서 식별자
    sensorType: SensorType;           // 센서 유형
    manufacturer: string;             // 제조사
    model: string;                    // 모델
    protocol: CommunicationProtocol;  // 통신 프로토콜
    linkedElement: ElementBinding;    // BIM 요소 연결
    dataPoints: DataPoint[];          // 데이터 포인트
    location: SensorLocation;         // 위치
    calibration: CalibrationInfo;     // 교정 정보
}

type SensorType =
    | 'TEMPERATURE'      // 온도
    | 'HUMIDITY'         // 습도
    | 'CO2'              // 이산화탄소
    | 'OCCUPANCY'        // 재실
    | 'LIGHT_LEVEL'      // 조도
    | 'ENERGY_METER'     // 전력계
    | 'WATER_METER'      // 수도계량
    | 'AIR_VELOCITY'     // 풍속
    | 'PRESSURE'         // 압력
    | 'VIBRATION'        // 진동
    | 'ACOUSTIC';        // 소음

type CommunicationProtocol =
    | 'MQTT'
    | 'COAP'
    | 'MODBUS'
    | 'BACNET'
    | 'LONWORKS'
    | 'ZIGBEE'
    | 'LORAWAN';

interface ElementBinding {
    globalId: string;                 // 연결된 요소의 IFC GlobalId
    elementType: string;              // IFC 엔티티 유형
    bindingType: 'DIRECT' | 'SPATIAL' | 'SYSTEM';
    relationship: string;             // 관계 설명
}

interface DataPoint {
    pointId: string;
    name: string;
    unit: string;
    dataType: 'FLOAT' | 'INTEGER' | 'BOOLEAN' | 'STRING';
    minValue?: number;
    maxValue?: number;
    samplingInterval: number;         // 초
    retentionPeriod: number;          // 일
}

6.4 실시간 데이터 스트리밍

// WIA-BIM 디지털 트윈: 실시간 데이터 서비스
import { WIADigitalTwin, MQTTClient, TimeSeriesDB } from '@wia/digital-twin';

class RealTimeDataService {
    private mqtt: MQTTClient;
    private tsdb: TimeSeriesDB;
    private twin: WIADigitalTwin;
    private subscriptions: Map<string, Subscription> = new Map();

    constructor(config: ServiceConfig) {
        this.mqtt = new MQTTClient({
            broker: config.mqttBroker,
            clientId: `wia-twin-${config.buildingId}`,
            cleanSession: false,
            keepAlive: 60
        });

        this.tsdb = new TimeSeriesDB({
            endpoint: config.tsdbEndpoint,
            database: config.buildingId
        });

        this.twin = new WIADigitalTwin(config.twinId);
    }

    async subscribeSensor(sensorId: string): Promise<void> {
        const sensor = await this.twin.getSensor(sensorId);
        const topic = `building/${this.twin.buildingId}/sensors/${sensorId}/data`;

        await this.mqtt.subscribe(topic, async (message) => {
            const data = JSON.parse(message.payload);

            // 시계열 데이터베이스에 저장
            await this.tsdb.write({
                measurement: sensor.sensorType.toLowerCase(),
                tags: {
                    sensorId: sensorId,
                    elementId: sensor.linkedElement.globalId,
                    space: sensor.location.spaceId
                },
                fields: data.values,
                timestamp: data.timestamp
            });

            // 디지털 트윈 상태 업데이트
            await this.twin.updateElementState(
                sensor.linkedElement.globalId,
                this.mapSensorDataToState(sensor, data)
            );

            // 알림 조건 확인
            await this.checkAlertConditions(sensor, data);
        });

        this.subscriptions.set(sensorId, { topic, sensor });
    }

    async queryHistoricalData(
        sensorId: string,
        startTime: Date,
        endTime: Date,
        aggregation?: Aggregation
    ): Promise<TimeSeriesData[]> {
        const sensor = await this.twin.getSensor(sensorId);

        let query = `SELECT ${aggregation?.function || '*'}
                     FROM "${sensor.sensorType.toLowerCase()}"
                     WHERE sensorId = '${sensorId}'
                     AND time >= '${startTime.toISOString()}'
                     AND time <= '${endTime.toISOString()}'`;

        if (aggregation) {
            query += ` GROUP BY time(${aggregation.interval})`;
        }

        return await this.tsdb.query(query);
    }
}

6.5 예측 분석

디지털 트윈은 모니터링을 넘어 미래 상태를 예측하고, 이상을 감지하고, 운영을 사전에 최적화하는 예측 기능을 가능하게 합니다.

활용 사례 데이터 소스 분석 방법 결과
예측 유지보수 장비 센서, 유지보수 이력 ML 이상 감지, 생존 분석 장비 고장 예방
에너지 최적화 전력계, 날씨, 재실 회귀 모델, 최적화 알고리즘 10-30% 에너지 절감
재실 예측 출입, WiFi, 센서, 캘린더 시계열 예측 공간 활용, HVAC 일정
실내 공기질 CO2, VOC, PM2.5, 온도 임계값 모니터링, 추세 분석 재실자 건강 및 쾌적
구조 건전성 변형률계, 가속도계, 경사계 진동 분석, FEM 시뮬레이션 조기 손상 감지

디지털 트윈 성숙도 수준

  • 레벨 1 - 서술적: 기본 자산 정보가 있는 정적 모델
  • 레벨 2 - 정보적: 과거 운영 데이터가 있는 모델
  • 레벨 3 - 예측적: 예측 분석이 있는 실시간 데이터
  • 레벨 4 - 포괄적: 양방향 제어 및 최적화
  • 레벨 5 - 자율적: 자가 학습, 자가 최적화 시스템

6.6 디지털 트윈 API

// WIA-BIM SDK: 디지털 트윈 작업
import { WIADigitalTwin, TwinViewer } from '@wia/digital-twin';

class DigitalTwinService {
    private twin: WIADigitalTwin;

    async initializeTwin(bimModelId: string, buildingId: string): Promise<void> {
        // BIM 모델에서 디지털 트윈 생성
        this.twin = await WIADigitalTwin.createFromBIM({
            modelId: bimModelId,
            buildingId: buildingId,
            updateMode: 'REAL_TIME',
            retentionPolicy: {
                rawData: 30,           // 일 (원시 데이터)
                aggregated1min: 90,    // 일 (1분 집계)
                aggregated1hour: 365,  // 일 (1시간 집계)
                aggregated1day: 'FOREVER'  // 영구 (일별 집계)
            }
        });

        // 빌딩 시스템 연결
        await this.twin.connectBMS({
            protocol: 'BACNET',
            address: '192.168.1.100',
            deviceId: 1001
        });

        console.log(`디지털 트윈 초기화 완료: ${this.twin.id}`);
    }

    async getCurrentState(elementId: string): Promise<ElementState> {
        const element = await this.twin.getElement(elementId);
        const sensors = await this.twin.getSensorsByElement(elementId);

        const sensorStates = await Promise.all(
            sensors.map(s => this.twin.getLatestReading(s.sensorId))
        );

        return {
            element: element,
            geometry: await element.getGeometry(),
            properties: await element.getProperties(),
            sensorData: sensorStates,
            lastUpdated: new Date()
        };
    }

    async runSimulation(
        scenario: SimulationScenario
    ): Promise<SimulationResult> {
        return await this.twin.simulate({
            type: scenario.type,
            parameters: scenario.parameters,
            duration: scenario.duration,
            outputMetrics: scenario.metrics
        });
    }
}
