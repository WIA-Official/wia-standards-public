5장: 4D/5D/6D BIM 응용

BIM의 시간, 비용 및 지속가능성 차원

5.1 3D를 넘어서: 다차원 BIM

3D BIM이 공간 시각화 및 조정 기능을 제공하는 반면, BIM의 진정한 힘은 추가 차원이 통합될 때 나타납니다. 다차원 BIM은 3D 모델을 확장하여 시간(4D), 비용(5D), 지속가능성(6D) 및 시설 관리(7D) 데이터를 통합하여 건물 수명주기 전반에 걸쳐 의사결정을 지원하는 종합적인 디지털 표현을 만듭니다.

이러한 추가 차원은 BIM을 설계 및 조정 도구에서 시공 계획, 비용 관리, 환경 분석 및 자산 관리를 위한 강력한 플랫폼으로 변환합니다. WIA-BIM은 이러한 차원을 구현하고 통합하기 위한 표준화된 프레임워크를 제공합니다.

차원 초점 주요 단계 주요 이점
3D 기하학 및 시각화 설계 조정, 간섭 검토, 설계 검토
4D 시간 및 일정 시공 순서 계획, 물류, 진행 추적
5D 비용 및 견적 설계~시공 물량 산출, 예산 편성, 비용 관리
6D 지속가능성 설계 및 운영 에너지 분석, 탄소 발자국, LCA
7D 시설 관리 운영 자산 관리, 유지보수, 공간 계획

5.2 4D BIM: 시간 연계 일정 관리

4D BIM 개요

4D BIM은 3D 모델을 시공 일정과 연결하여 시간에 따른 시공 순서의 시각화를 가능하게 합니다. 이 강력한 기능을 통해 프로젝트 팀은 시공 과정을 시뮬레이션하고, 순서 문제를 식별하고, 물류를 최적화하고, 모든 이해관계자에게 계획을 명확하게 전달할 수 있습니다.

5.2.1 일정-모델 연결

// WIA-BIM 4D 연결 스키마
interface Schedule4DLink {
    activityId: string;              // 일정 작업 ID
    activityName: string;            // 작업 설명
    startDate: Date;
    endDate: Date;
    linkedElements: ElementLink[];   // 작업에 연결된 BIM 요소
    taskType: TaskType;
    predecessor?: string[];          // 선행 작업
    successor?: string[];            // 후행 작업
}

interface ElementLink {
    globalId: string;                // IFC GlobalId
    linkType: 'CONSTRUCT' | 'TEMPORARY' | 'DEMOLISH' | 'MAINTAIN';
    animationState: AnimationState;
}

interface AnimationState {
    beforeStart: 'HIDDEN' | 'TRANSPARENT' | 'WIREFRAME';
    duringActivity: 'HIGHLIGHTED' | 'ANIMATED' | 'NORMAL';
    afterComplete: 'VISIBLE' | 'REMOVED';
    color?: string;                  // 선택적 색상 코딩
}

type TaskType =
    | 'EXCAVATION'      // 터파기
    | 'FOUNDATION'      // 기초
    | 'STRUCTURE'       // 구조
    | 'ENVELOPE'        // 외피
    | 'MEP_ROUGH'       // MEP 러프인
    | 'MEP_FINISH'      // MEP 마감
    | 'INTERIOR'        // 내부
    | 'EXTERIOR'        // 외부
    | 'COMMISSIONING';  // 시운전

5.3 5D BIM: 비용 관리

5D BIM 개요

5D BIM은 비용 데이터를 3D 모델과 통합하여 자동 물량 산출, 실시간 비용 추정 및 프로젝트 전반에 걸친 예산 추적을 가능하게 합니다. 비용 데이터베이스를 모델 요소에 연결함으로써 설계 변경 시 비용 예측이 자동으로 업데이트되어 설계 결정의 재정적 영향에 대한 즉각적인 피드백을 제공합니다. 

// WIA-BIM 5D 비용 통합 스키마
interface CostDatabase {
    items: CostItem[];
    laborRates: LaborRate[];
    equipmentRates: EquipmentRate[];
    currency: string;                 // 통화 (KRW, USD 등)
    effectiveDate: Date;
    region: string;
}

interface CostItem {
    code: string;                    // 비용 코드 (예: 표준품셈)
    description: string;
    unit: string;                    // 측정 단위
    materialCost: number;            // 단위당 재료비
    laborCost: number;               // 단위당 노무비
    equipmentCost: number;           // 단위당 장비비
    wasteFactor: number;             // 할증률
    productivityRate: number;        // 노동시간당 단위
}

interface QuantityTakeoff {
    elementId: string;
    ifcType: string;
    quantities: ElementQuantity[];
    linkedCostCodes: CostCodeLink[];
}

interface ElementQuantity {
    quantityType: QuantityType;
    value: number;
    unit: string;
    formula?: string;                // 파생된 경우 계산 공식
}

type QuantityType =
    | 'LENGTH'        // 길이
    | 'AREA'          // 면적
    | 'VOLUME'        // 체적
    | 'COUNT'         // 개수
    | 'WEIGHT'        // 중량
    | 'GROSS_AREA'    // 총면적
    | 'NET_AREA'      // 순면적
    | 'PERIMETER';    // 둘레

// 예시: 벽 비용 계산
async function calculateWallCost(wall: IfcWall, costDb: CostDatabase): Promise<CostBreakdown> {
    const quantities = await extractQuantities(wall);
    const costItem = costDb.items.find(i => i.code === '04 21 13.13');  // 벽돌조적

    if (!costItem || !quantities.area) {
        throw new Error('비용 데이터 또는 물량 누락');
    }

    const grossArea = quantities.area.value;
    const netArea = grossArea - (quantities.openingArea?.value || 0);
    const adjustedArea = netArea * (1 + costItem.wasteFactor);

    return {
        elementId: wall.globalId,
        elementName: wall.name,
        costCode: costItem.code,
        quantity: netArea,
        unit: costItem.unit,
        materialCost: adjustedArea * costItem.materialCost,
        laborCost: adjustedArea * costItem.laborCost,
        equipmentCost: adjustedArea * costItem.equipmentCost,
        totalCost: adjustedArea * (costItem.materialCost + costItem.laborCost + costItem.equipmentCost),
        laborHours: adjustedArea / costItem.productivityRate
    };
}

5.4 6D BIM: 지속가능성 분석

6D BIM 개요

6D BIM은 에너지 분석, 수명주기 평가(LCA) 및 환경 영향 계산을 통합하여 지속가능성에 초점을 맞춥니다. 이 차원을 통해 설계자는 초기 설계 단계에서 건물 성능을 평가하고 최적화하여 탄소 발자국을 줄이고 운영 효율성을 개선하는 결정을 지원할 수 있습니다. 

// WIA-BIM 6D 지속가능성 스키마
interface SustainabilityAnalysis {
    projectId: string;
    analysisDate: Date;
    energyAnalysis: EnergyAnalysis;
    lifecycleAssessment: LCAResults;
    certificationTargets: CertificationTarget[];
}

interface EnergyAnalysis {
    annualEnergyConsumption: number;  // kWh/년
    energyUseIntensity: number;       // kWh/m2/년
    peakCoolingLoad: number;          // kW (냉방 피크 부하)
    peakHeatingLoad: number;          // kW (난방 피크 부하)
    renewableContribution: number;    // 신재생 기여율 (%)
    carbonEmissions: number;          // kg CO2e/년
    breakdownBySystem: SystemEnergy[];
}

interface SystemEnergy {
    system: 'HVAC' | 'LIGHTING' | 'EQUIPMENT' | 'DHW' | 'VERTICAL_TRANSPORT';
    consumption: number;              // kWh/년
    percentage: number;               // 전체 대비 비율
}

interface LCAResults {
    scope: 'CRADLE_TO_GATE' | 'CRADLE_TO_GRAVE' | 'CRADLE_TO_CRADLE';
    studyPeriod: number;              // 년
    embodiedCarbon: EmbodiedCarbon;   // 내재 탄소
    operationalCarbon: OperationalCarbon;  // 운영 탄소
    endOfLifeCarbon: EndOfLifeCarbon; // 수명 종료 탄소
    totalGWP: number;                 // kg CO2e (지구온난화지수)
}

interface EmbodiedCarbon {
    a1_a3_production: number;         // 원자재 및 제조
    a4_transport: number;             // 현장까지 운송
    a5_construction: number;          // 시공 과정
    total: number;                    // kg CO2e
    byMaterial: MaterialCarbon[];
}

interface MaterialCarbon {
    material: string;
    quantity: number;
    unit: string;
    carbonFactor: number;             // 단위당 kg CO2e
    totalCarbon: number;              // kg CO2e
    percentage: number;               // 전체 내재탄소 대비
}

// 예시: 콘크리트 요소의 내재 탄소 계산
async function analyzeEmbodiedCarbon(model: IfcModel): Promise<EmbodiedCarbon> {
    const concreteElements = await model.getElementsByMaterial('Concrete');
    let totalCarbon = 0;
    const breakdown: MaterialCarbon[] = [];

    for (const element of concreteElements) {
        const volume = await element.getQuantity('NetVolume');
        const grade = await element.getProperty('ConcreteGrade');
        const carbonFactor = getCarbonFactor('Concrete', grade);

        const elementCarbon = volume * carbonFactor;
        totalCarbon += elementCarbon;

        breakdown.push({
            material: `콘크리트 ${grade}`,
            quantity: volume,
            unit: 'm3',
            carbonFactor: carbonFactor,
            totalCarbon: elementCarbon,
            percentage: 0  // 합계 후 계산
        });
    }

    // 백분율 업데이트
    breakdown.forEach(b => b.percentage = (b.totalCarbon / totalCarbon) * 100);

    return {
        a1_a3_production: totalCarbon * 0.85,
        a4_transport: totalCarbon * 0.10,
        a5_construction: totalCarbon * 0.05,
        total: totalCarbon,
        byMaterial: breakdown
    };
}

녹색 건축 표준과의 통합

WIA-BIM 6D 분석은 LEED, BREEAM, WELL, Green Star 및 G-SEED를 포함한 주요 녹색 건축 인증 시스템을 지원합니다. 이 프레임워크는 BIM 데이터와 인증 크레딧 요구사항 간의 매핑을 제공하여 자동화된 문서화 및 준수 추적을 가능하게 합니다.

5.5 다차원 API

// WIA-BIM SDK: 다차원 분석 서비스
import { WIABimClient, Model4D, Model5D, Model6D } from '@wia/bim-sdk';

class MultiDimensionalBIM {
    private client: WIABimClient;

    async create4DSimulation(
        modelId: string,
        scheduleFile: string
    ): Promise<Simulation4D> {
        const model = await this.client.loadModel(modelId);
        const schedule = await this.client.importSchedule(scheduleFile);

        // 명명 및 공간 분석 기반 요소-작업 자동 연결
        const links = await Model4D.autoLink(model, schedule, {
            matchingStrategy: 'SPATIAL_AND_NAMING',
            defaultTaskMapping: DEFAULT_TASK_MAP
        });

        return Model4D.createSimulation({
            model,
            schedule,
            links,
            outputFormat: 'VIDEO',
            frameRate: 30,
            duration: 'PROPORTIONAL'
        });
    }

    async generate5DEstimate(
        modelId: string,
        costDatabase: string
    ): Promise<CostEstimate> {
        const model = await this.client.loadModel(modelId);
        const costs = await this.client.loadCostDatabase(costDatabase);

        const takeoff = await Model5D.extractQuantities(model, {
            includeOpenings: true,
            groupBy: 'SYSTEM',
            precision: 2
        });

        return Model5D.calculateCosts(takeoff, costs, {
            includeContingency: 0.10,    // 예비비 10%
            includeOverhead: 0.15,       // 간접비 15%
            includeProfit: 0.08          // 이윤 8%
        });
    }

    async perform6DAnalysis(
        modelId: string,
        location: GeoLocation
    ): Promise<SustainabilityReport> {
        const model = await this.client.loadModel(modelId);

        const energyAnalysis = await Model6D.analyzeEnergy(model, {
            location,
            weatherData: 'TMY3',
            occupancySchedule: 'ASHRAE_OFFICE',
            systemEfficiencies: DEFAULT_EFFICIENCIES
        });

        const lcaResults = await Model6D.performLCA(model, {
            scope: 'CRADLE_TO_GRAVE',
            studyPeriod: 50,
            carbonFactors: 'ICE_DATABASE'
        });

        return {
            energy: energyAnalysis,
            lca: lcaResults,
            recommendations: await Model6D.generateRecommendations(energyAnalysis, lcaResults)
        };
    }
}
