1장: IFC 표준 및 데이터 스키마

개방형 BIM 데이터 교환을 위한 Industry Foundation Classes

1.1 IFC 소개

IFC(Industry Foundation Classes)는 개방형 BIM 데이터 교환의 초석입니다. buildingSMART International이 개발하고 유지 관리하는 IFC는 서로 다른 BIM 소프트웨어 애플리케이션 간의 원활한 상호운용성을 가능하게 하는 벤더 중립적이고 표준화된 데이터 스키마를 제공합니다. IFC는 ISO 16739로 등록되어 건물 및 건설 데이터에 대한 글로벌 표준으로 확립되었습니다.

IFC는 단순한 기하학적 형상뿐만 아니라 건물 요소, 관계, 속성, 프로세스 및 건물 수명주기에 관련된 행위자를 포함하는 풍부한 의미론적 모델을 정의합니다. 이러한 포괄적인 접근 방식은 단순한 기하학적 전송이 아닌 진정한 데이터 교환을 가능하게 하여, 프로젝트의 다양한 플랫폼과 단계에 걸쳐 BIM 모델에 내장된 지능을 보존합니다.

IFC가 중요한 이유

IFC가 없다면 각 BIM 소프트웨어는 고립된 사일로로 존재하게 되어 플랫폼 간 이동 시 데이터를 수동으로 다시 입력해야 합니다. IFC는 이러한 장벽을 허물어 건축가, 엔지니어 및 시공자가 프로젝트 수명주기 전반에 걸쳐 데이터 무결성을 유지하면서 선호하는 도구를 사용하여 협업할 수 있게 합니다.

1.2 IFC 스키마 아키텍처

IFC 스키마는 네 개의 개념적 레이어로 구성되며, 각 레이어는 그 아래 레이어를 기반으로 구축됩니다. 이러한 계층화된 아키텍처는 다양한 건물 도메인과 프로젝트 유형에 걸쳐 일관성을 유지하면서 유연성을 제공합니다.

1.3 주요 IFC 엔티티

핵심 IFC 엔티티를 이해하는 것은 효과적인 BIM 구현에 필수적입니다. 이러한 엔티티는 모든 IFC 모델이 구성되는 기본 빌딩 블록을 나타냅니다.

엔티티 설명 하위 유형 예시
IfcProject 전체 프로젝트 범위를 포함하는 루트 엔티티 파일당 단일 인스턴스
IfcSite 프로젝트의 지리적 위치 좌표, 표고, 주소
IfcBuilding 사이트 내 개별 건물 건물 수준 정보
IfcBuildingStorey 건물 내 층 레벨 레벨 표고, 높이
IfcSpace 정의된 기능을 가진 둘러싸인 영역 방, 사무실, 복도
IfcWall 수직 건축 요소 IfcWallStandardCase, IfcCurtainWall
IfcSlab 수평 건축 요소 바닥, 지붕, 계단참
IfcBeam 선형 구조 부재 철골보, 콘크리트보
IfcColumn 수직 구조 부재 철골기둥, 콘크리트기둥

1.4 EXPRESS 스키마 언어

IFC는 EXPRESS 데이터 모델링 언어(ISO 10303-11)를 사용하여 공식적으로 정의됩니다. EXPRESS는 서로 다른 소프트웨어 플랫폼 간에 일관된 구현을 보장하는 엄격하고 명확한 데이터 구조, 관계 및 제약 조건의 정의를 제공합니다. 

-- EXPRESS 스키마 예제: IfcWall 정의
-- ISO 10303-11 EXPRESS 언어

ENTITY IfcWall
  SUPERTYPE OF (ONEOF
    (IfcWallStandardCase, IfcWallElementedCase))
  SUBTYPE OF (IfcBuildingElement);
    PredefinedType : OPTIONAL IfcWallTypeEnum;
  WHERE
    CorrectPredefinedType : NOT(EXISTS(PredefinedType)) OR
      (PredefinedType <> IfcWallTypeEnum.USERDEFINED) OR
      ((PredefinedType = IfcWallTypeEnum.USERDEFINED) AND
       EXISTS(SELF\IfcObject.ObjectType));
    CorrectTypeAssigned : (SIZEOF(IsTypedBy) = 0) OR
      ('IFC4X3.IFCWALLTYPE' IN TYPEOF(SELF\IfcObject.IsTypedBy[1].RelatingType));
END_ENTITY;

ENTITY IfcWallType
  SUBTYPE OF (IfcBuildingElementType);
    PredefinedType : IfcWallTypeEnum;
  WHERE
    CorrectPredefinedType : (PredefinedType <> IfcWallTypeEnum.USERDEFINED) OR
      ((PredefinedType = IfcWallTypeEnum.USERDEFINED) AND
       EXISTS(SELF\IfcElementType.ElementType));
END_ENTITY;

TYPE IfcWallTypeEnum = ENUMERATION OF
    (MOVABLE,       -- 이동식 벽
     PARAPET,       -- 파라펫
     PARTITIONING,  -- 칸막이벽
     PLUMBINGWALL,  -- 배관벽
     SHEAR,         -- 전단벽
     SOLIDWALL,     -- 솔리드벽
     STANDARD,      -- 표준벽
     POLYGONAL,     -- 다각형벽
     ELEMENTEDWALL, -- 조립벽
     RETAININGWALL, -- 옹벽
     WAVEWALL,      -- 방파벽
     USERDEFINED,   -- 사용자정의
     NOTDEFINED);   -- 미정의
END_TYPE;

1.5 IFC 파일 형식

IFC 데이터는 다양한 사용 사례와 성능 요구 사항에 적합한 여러 파일 형식으로 직렬화할 수 있습니다.

1.5.1 STEP 물리적 파일 (SPF)

전통적인 IFC 형식은 STEP Part 21 인코딩(.ifc 확장자)을 사용합니다. 이 사람이 읽을 수 있는 텍스트 형식은 널리 지원되며 디버깅 및 검증에 탁월하지만, 복잡한 모델의 경우 파일 크기가 커질 수 있습니다. 

/* IFC-SPF 예제: 간단한 벽 인스턴스 */
ISO-10303-21;
HEADER;
FILE_DESCRIPTION(('WIA-BIM 예제'),'2;1');
FILE_NAME('example.ifc','2025-01-15T10:30:00',('건축가'),('WIA'),'WIA-BIM SDK','IFC4X3_ADD2','');
FILE_SCHEMA(('IFC4X3_ADD2'));
ENDSEC;

DATA;
#1=IFCPROJECT('2m6OwYT5z9tQJy',#2,'샘플 프로젝트',$,$,$,$,(#10),#5);
#2=IFCOWNERHISTORY(#3,#4,$,.NOCHANGE.,$,$,$,1705312200);
#3=IFCPERSONANDORGANIZATION(#6,#7,$);
#4=IFCAPPLICATION(#7,'1.0','WIA-BIM SDK','WIA-BIM');
#5=IFCUNITASSIGNMENT((#8,#9));
#6=IFCPERSON($,'김','철수',$,$,$,$,$);
#7=IFCORGANIZATION($,'WIA','세계산업협회',$,$);
#8=IFCSIUNIT(*,.LENGTHUNIT.,$,.METRE.);
#9=IFCSIUNIT(*,.AREAUNIT.,$,.SQUARE_METRE.);
#10=IFCGEOMETRICREPRESENTATIONCONTEXT($,'Model',3,1.0E-05,#11,$);
#11=IFCAXIS2PLACEMENT3D(#12,$,$);
#12=IFCCARTESIANPOINT((0.,0.,0.));

/* 벽 인스턴스 */
#100=IFCWALL('3vB2YO$5X4xOQ8',#2,'벽-001','외벽',$,#101,#110,$,.STANDARD.);
#101=IFCLOCALPLACEMENT($,#102);
#102=IFCAXIS2PLACEMENT3D(#103,#104,#105);
#103=IFCCARTESIANPOINT((0.,0.,0.));
#104=IFCDIRECTION((0.,0.,1.));
#105=IFCDIRECTION((1.,0.,0.));
#110=IFCPRODUCTDEFINITIONSHAPE($,$,(#111));
#111=IFCSHAPEREPRESENTATION(#10,'Body','SweptSolid',(#112));
#112=IFCEXTRUDEDAREASOLID(#113,#11,#104,3000.);
#113=IFCRECTANGLEPROFILEDEF(.AREA.,$,$,200.,6000.);

ENDSEC;
END-ISO-10303-21;

1.6 IFC 버전 역사

IFC 스키마는 초기 출시 이후 크게 발전했으며, 각 버전은 기능을 확장하고 산업 요구를 해결했습니다.

버전 연도 주요 기능 ISO 상태
IFC 1.0 1997 초기 출시, 기본 건물 요소 -
IFC 2x3 2006 가장 널리 배포됨, 안정적인 스키마 ISO 16739:2013
IFC 4 2013 향상된 기하학, 파라메트릭 요소 ISO 16739-1:2018
IFC 4.3 2024 인프라, 철도, 도로, 교량 ISO 16739-1:2024

WIA-BIM 준수

WIA-BIM은 IFC 4.3을 기본 요구 사항으로 지정하며, IFC 2x3 모델에 대한 하위 호환성을 제공합니다. 모든 WIA 인증 BIM 소프트웨어는 신뢰할 수 있는 데이터 교환을 보장하기 위해 buildingSMART IFC 인증 프로그램을 통과해야 합니다.

1.7 API 통합 예제

// WIA-BIM SDK: 완전한 IFC 모델 작업
import { WIABimClient, IfcModel, IfcWall } from '@wia/bim-sdk';

class IFCModelManager {
    private client: WIABimClient;
    private model: IfcModel;

    constructor(config: { projectId: string; endpoint: string }) {
        this.client = new WIABimClient(config);
    }

    async loadModel(filePath: string): Promise<void> {
        this.model = await this.client.loadModel(filePath);
        console.log(`모델 로드됨: ${this.model.getProject().name}`);
        console.log(`스키마 버전: ${this.model.schemaVersion}`);
    }

    async analyzeWalls(): Promise<WallAnalysis[]> {
        const walls = await this.model.getElementsByType('IfcWall');
        const analysis: WallAnalysis[] = [];

        for (const wall of walls) {
            const geometry = await wall.getGeometry();
            const properties = await wall.getPropertySets();
            const materials = await wall.getMaterials();

            analysis.push({
                globalId: wall.globalId,
                name: wall.name,
                length: geometry.boundingBox.length,      // 길이
                height: geometry.boundingBox.height,      // 높이
                thickness: geometry.boundingBox.width,    // 두께
                area: geometry.surfaceArea,               // 면적
                volume: geometry.volume,                  // 체적
                materials: materials.map(m => m.name),    // 재료
                thermalTransmittance: properties['Pset_WallCommon']?.ThermalTransmittance  // 열관류율
            });
        }

        return analysis;
    }

    async exportToJSON(outputPath: string): Promise<void> {
        const jsonData = await this.model.toJSON();
        await this.client.writeFile(outputPath, JSON.stringify(jsonData, null, 2));
    }
}

// 사용 예시
const manager = new IFCModelManager({
    projectId: 'building-001',
    endpoint: 'https://api.wia-bim.org'
});

await manager.loadModel('architecture.ifc');
const wallAnalysis = await manager.analyzeWalls();
console.log(`${wallAnalysis.length}개의 벽 분석 완료`);
