7.1 Integration Layer Overview

The Integration layer enables WIA-BIM to connect with the broader technology ecosystem, from digital twins and IoT systems to enterprise applications and legacy platforms. This comprehensive integration capability transforms BIM from an isolated design tool into the central hub of building intelligence.

🏗️ Integration Objectives

  • Digital Twins: Create living models that mirror physical buildings
  • Enterprise Systems: Connect with ERP, CMMS, and business applications
  • IoT Networks: Integrate real-time sensor data from buildings
  • GIS Platforms: Place buildings within geographic context
  • Legacy Systems: Bridge to existing infrastructure investments

// Real-time data integration
class DigitalTwinConnector {
  async syncSensorData(twinId, sensorData) {
    const twin = await this.getTwin(twinId);

    for (const reading of sensorData) {
      // Find corresponding BIM element
      const element = await this.findElementBySensorId(
        twin.bimModelId,
        reading.sensorId
      );

      if (element) {
        // Update twin with sensor data
        await this.updateElementData(element.guid, {
          liveData: {
            [reading.property]: {
              value: reading.value,
              timestamp: reading.timestamp,
              quality: reading.quality
            }
          }
        });

        // Trigger analytics if threshold exceeded
        if (this.checkThreshold(element, reading)) {
          await this.triggerAnalytics(twin, element, reading);
        }
      }
    }
  }
}

7.3.2 ERP Integration Pattern

// WIA-BIM to ERP Connector
class ERPConnector {
  constructor(config) {
    this.endpoint = config.erpEndpoint;
    this.credentials = config.credentials;
    this.mapping = config.fieldMapping;
  }

  // Sync BIM objects to ERP assets
  async syncAssets(bimObjects) {
    const assets = bimObjects.map(obj => ({
      assetNumber: this.generateAssetNumber(obj),
      description: obj.name,
      category: this.mapCategory(obj.type),
      location: this.mapLocation(obj.containedIn),
      acquisitionDate: obj.metadata.createdAt,
      value: this.extractCost(obj),
      specifications: {
        manufacturer: obj.properties.manufacturer,
        model: obj.properties.model,
        serialNumber: obj.properties.serialNumber
      }
    }));

    return await this.erpClient.bulkCreate('assets', assets);
  }

  // Receive cost updates from ERP
  async receiveCostUpdates(erpData) {
    for (const update of erpData) {
      const bimObject = await this.findByAssetNumber(update.assetNumber);
      if (bimObject) {
        await this.bimClient.updateObject(bimObject.guid, {
          'properties.actualCost': update.totalCost,
          'properties.lastInvoiceDate': update.lastInvoiceDate
        });
      }
    }
  }
}

// Sensor data processor
class IoTProcessor {
  async processMessage(topic, payload) {
    // Parse sensor data
    const sensorData = this.parseSensorPayload(payload);

    // Find mapping
    const mapping = this.findMapping(topic);
    if (!mapping) return;

    // Transform value
    const transformedValue = mapping.transform
      ? eval(mapping.transform)(sensorData.value)
      : sensorData.value;

    // Update BIM element
    await this.bimClient.updateLiveData(mapping.bimElement, {
      [mapping.property]: {
        value: transformedValue,
        timestamp: sensorData.timestamp,
        sensorId: sensorData.id,
        quality: sensorData.quality
      }
    });

    // Check alerts
    await this.checkAlerts(mapping.bimElement, mapping.property, transformedValue);
  }
}

// Transform BIM coordinates to GIS
class GISTransformer {
  bimToGIS(bimPoint, transform) {
    // Apply rotation
    const rotated = this.rotate(bimPoint, transform.rotation);

    // Apply scale
    const scaled = {
      x: rotated.x * transform.scale,
      y: rotated.y * transform.scale,
      z: rotated.z * transform.scale
    };

    // Convert to lat/lon
    const latLon = this.projectToLatLon(scaled, transform.origin);

    return {
      latitude: latLon.lat,
      longitude: latLon.lon,
      elevation: transform.origin.elevation + scaled.z / 1000
    };
  }
}

7.6.2 Legacy Adapter Implementation

// Legacy System Adapter Pattern
class LegacyCADAdapter {
  constructor(config) {
    this.legacyPath = config.legacyPath;
    this.fileFormat = config.format; // DWG, DXF, etc.
    this.mappingRules = config.mappings;
  }

  async importFromLegacy(filePath) {
    // Read legacy format
    const legacyData = await this.readLegacyFile(filePath);

    // Transform to WIA-BIM
    const bimObjects = [];

    for (const entity of legacyData.entities) {
      const mapping = this.mappingRules[entity.type];
      if (mapping) {
        bimObjects.push({
          type: mapping.bimType,
          name: entity.name || this.generateName(entity),
          geometry: this.transformGeometry(entity.geometry),
          properties: this.mapProperties(entity, mapping.properties),
          metadata: {
            importSource: filePath,
            originalType: entity.type,
            importDate: new Date().toISOString()
          }
        });
      }
    }

    return bimObjects;
  }

  async exportToLegacy(bimObjects, targetFormat) {
    const legacyEntities = bimObjects.map(obj => ({
      type: this.reverseMapType(obj.type),
      geometry: this.reverseTransformGeometry(obj.geometry),
      attributes: this.reverseMapProperties(obj.properties)
    }));

    return await this.writeLegacyFile(legacyEntities, targetFormat);
  }
}
