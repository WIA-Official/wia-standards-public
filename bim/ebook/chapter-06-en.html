Chapter 6: Digital Twin Integration

Real-Time IoT Connectivity and Live Building Performance

6.1 Introduction to Digital Twins

A Digital Twin is a dynamic, real-time digital representation of a physical asset that is continuously updated with live data from sensors, IoT devices, and operational systems. Unlike static BIM models, digital twins evolve throughout the building's operational lifecycle, reflecting current conditions, predicting future states, and enabling intelligent decision-making.

The transition from BIM to digital twin represents the evolution of building information from design documentation to operational intelligence. While BIM captures the as-designed and as-built state, the digital twin maintains the as-operated state, incorporating real-time sensor data, maintenance records, occupancy patterns, and environmental conditions.

BIM to Digital Twin Evolution

The digital twin builds upon the BIM foundation, inheriting geometric data, spatial relationships, and asset information while adding real-time connectivity, simulation capabilities, and predictive analytics. This integration creates a living model that serves as the operational brain of the building.

6.3 IoT Sensor Integration

Connecting IoT sensors to the digital twin requires mapping sensor data points to specific BIM elements. WIA-BIM defines a standardized approach for sensor registration, data mapping, and real-time streaming. 

// WIA-BIM Digital Twin: Sensor Integration Schema
interface SensorRegistration {
    sensorId: string;                 // Unique sensor identifier
    sensorType: SensorType;           // Type of sensor
    manufacturer: string;
    model: string;
    protocol: CommunicationProtocol;
    linkedElement: ElementBinding;    // BIM element association
    dataPoints: DataPoint[];
    location: SensorLocation;
    calibration: CalibrationInfo;
}

type SensorType =
    | 'TEMPERATURE'
    | 'HUMIDITY'
    | 'CO2'
    | 'OCCUPANCY'
    | 'LIGHT_LEVEL'
    | 'ENERGY_METER'
    | 'WATER_METER'
    | 'AIR_VELOCITY'
    | 'PRESSURE'
    | 'VIBRATION'
    | 'ACOUSTIC';

type CommunicationProtocol =
    | 'MQTT'
    | 'COAP'
    | 'MODBUS'
    | 'BACNET'
    | 'LONWORKS'
    | 'ZIGBEE'
    | 'LORAWAN';

interface ElementBinding {
    globalId: string;                 // IFC GlobalId of linked element
    elementType: string;              // IFC entity type
    bindingType: 'DIRECT' | 'SPATIAL' | 'SYSTEM';
    relationship: string;             // Describes the relationship
}

interface DataPoint {
    pointId: string;
    name: string;
    unit: string;
    dataType: 'FLOAT' | 'INTEGER' | 'BOOLEAN' | 'STRING';
    minValue?: number;
    maxValue?: number;
    samplingInterval: number;         // Seconds
    retentionPeriod: number;          // Days
}

interface SensorLocation {
    spaceId?: string;                 // IfcSpace GlobalId
    coordinates: { x: number; y: number; z: number };
    mountingHeight?: number;
    orientiation?: { pitch: number; yaw: number };
}

6.4 Real-Time Data Streaming

// WIA-BIM Digital Twin: Real-Time Data Service
import { WIADigitalTwin, MQTTClient, TimeSeriesDB } from '@wia/digital-twin';

class RealTimeDataService {
    private mqtt: MQTTClient;
    private tsdb: TimeSeriesDB;
    private twin: WIADigitalTwin;
    private subscriptions: Map<string, Subscription> = new Map();

    constructor(config: ServiceConfig) {
        this.mqtt = new MQTTClient({
            broker: config.mqttBroker,
            clientId: `wia-twin-${config.buildingId}`,
            cleanSession: false,
            keepAlive: 60
        });

        this.tsdb = new TimeSeriesDB({
            endpoint: config.tsdbEndpoint,
            database: config.buildingId
        });

        this.twin = new WIADigitalTwin(config.twinId);
    }

    async subscribeSensor(sensorId: string): Promise<void> {
        const sensor = await this.twin.getSensor(sensorId);
        const topic = `building/${this.twin.buildingId}/sensors/${sensorId}/data`;

        await this.mqtt.subscribe(topic, async (message) => {
            const data = JSON.parse(message.payload);

            // Store in time series database
            await this.tsdb.write({
                measurement: sensor.sensorType.toLowerCase(),
                tags: {
                    sensorId: sensorId,
                    elementId: sensor.linkedElement.globalId,
                    space: sensor.location.spaceId
                },
                fields: data.values,
                timestamp: data.timestamp
            });

            // Update digital twin state
            await this.twin.updateElementState(
                sensor.linkedElement.globalId,
                this.mapSensorDataToState(sensor, data)
            );

            // Check for alerts
            await this.checkAlertConditions(sensor, data);
        });

        this.subscriptions.set(sensorId, { topic, sensor });
    }

    async queryHistoricalData(
        sensorId: string,
        startTime: Date,
        endTime: Date,
        aggregation?: Aggregation
    ): Promise<TimeSeriesData[]> {
        const sensor = await this.twin.getSensor(sensorId);

        let query = `SELECT ${aggregation?.function || '*'} FROM "${sensor.sensorType.toLowerCase()}"
                     WHERE sensorId = '${sensorId}'
                     AND time >= '${startTime.toISOString()}'
                     AND time <= '${endTime.toISOString()}'`;

        if (aggregation) {
            query += ` GROUP BY time(${aggregation.interval})`;
        }

        return await this.tsdb.query(query);
    }

    private async checkAlertConditions(
        sensor: SensorRegistration,
        data: SensorData
    ): Promise<void> {
        const rules = await this.twin.getAlertRules(sensor.sensorId);

        for (const rule of rules) {
            const triggered = this.evaluateRule(rule, data);
            if (triggered) {
                await this.twin.raiseAlert({
                    ruleId: rule.id,
                    sensorId: sensor.sensorId,
                    elementId: sensor.linkedElement.globalId,
                    severity: rule.severity,
                    message: this.formatAlertMessage(rule, data),
                    timestamp: new Date(),
                    data: data
                });
            }
        }
    }
}

6.5 Predictive Analytics

Digital twins enable predictive capabilities that go beyond monitoring to anticipate future conditions, detect anomalies, and optimize operations proactively.

Use Case Data Sources Analytics Method Outcome
Predictive Maintenance Equipment sensors, maintenance history ML anomaly detection, survival analysis Prevent equipment failures
Energy Optimization Energy meters, weather, occupancy Regression models, optimization algorithms 10-30% energy savings
Occupancy Prediction Access, WiFi, sensors, calendars Time series forecasting Space utilization, HVAC scheduling
Indoor Air Quality CO2, VOC, PM2.5, temperature Threshold monitoring, trend analysis Occupant health and comfort
Structural Health Strain gauges, accelerometers, tilt Vibration analysis, FEM simulation Early damage detection

Digital Twin Maturity Levels

  • Level 1 - Descriptive: Static model with basic asset information
  • Level 2 - Informative: Model with historical operational data
  • Level 3 - Predictive: Real-time data with predictive analytics
  • Level 4 - Comprehensive: Bi-directional control and optimization
  • Level 5 - Autonomous: Self-learning, self-optimizing system

6.6 Digital Twin API

// WIA-BIM SDK: Digital Twin Operations
import { WIADigitalTwin, TwinViewer } from '@wia/digital-twin';

class DigitalTwinService {
    private twin: WIADigitalTwin;

    async initializeTwin(bimModelId: string, buildingId: string): Promise<void> {
        // Create digital twin from BIM model
        this.twin = await WIADigitalTwin.createFromBIM({
            modelId: bimModelId,
            buildingId: buildingId,
            updateMode: 'REAL_TIME',
            retentionPolicy: {
                rawData: 30,           // days
                aggregated1min: 90,
                aggregated1hour: 365,
                aggregated1day: 'FOREVER'
            }
        });

        // Connect to building systems
        await this.twin.connectBMS({
            protocol: 'BACNET',
            address: '192.168.1.100',
            deviceId: 1001
        });

        console.log(`Digital twin initialized: ${this.twin.id}`);
    }

    async getCurrentState(elementId: string): Promise<ElementState> {
        const element = await this.twin.getElement(elementId);
        const sensors = await this.twin.getSensorsByElement(elementId);

        const sensorStates = await Promise.all(
            sensors.map(s => this.twin.getLatestReading(s.sensorId))
        );

        return {
            element: element,
            geometry: await element.getGeometry(),
            properties: await element.getProperties(),
            sensorData: sensorStates,
            lastUpdated: new Date()
        };
    }

    async runSimulation(
        scenario: SimulationScenario
    ): Promise<SimulationResult> {
        return await this.twin.simulate({
            type: scenario.type,
            parameters: scenario.parameters,
            duration: scenario.duration,
            outputMetrics: scenario.metrics
        });
    }
}
