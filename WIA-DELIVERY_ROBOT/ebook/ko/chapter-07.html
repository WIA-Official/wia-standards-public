7.1 테스트 전략 개요

WIA-DELIVERY_ROBOT 표준을 성공적으로 구현하기 위해서는 체계적인 테스트가 필수적입니다. 테스트는 단순히 버그를 찾는 것을 넘어 시스템이 표준 요구사항을 충족하는지 검증하고, 프로덕션 환경에서 안정적으로 작동할 것임을 보장합니다.

7.2 단위 테스트

import { RouteCalculator, RobotAssigner } from '@wia/delivery-robot-sdk'; describe('RouteCalculator', () => { let calculator; beforeEach(() => { calculator = new RouteCalculator({ algorithm: 'dijkstra' }); }); it('should find shortest path', async () => { const origin = { lat: 37.5665, lng: 126.9780 }; const destination = { lat: 37.5010, lng: 127.0390 }; const route = await calculator.calculateRoute(origin, destination); expect(route).toBeDefined(); expect(route.distance).toBeGreaterThan(0); expect(route.estimatedTime).toBeGreaterThan(0); }); it('should throw error for invalid coordinates', async () => { const origin = { lat: 999, lng: 999 }; const destination = { lat: 37.5010, lng: 127.0390 }; await expect(calculator.calculateRoute(origin, destination)) .rejects.toThrow('Invalid coordinates'); }); }); describe('RobotAssigner', () => { it('should assign nearest available robot', async () => { const robots = [ { id: 'R001', battery: 80, status: 'idle', location: { lat: 37.56, lng: 126.97 } }, { id: 'R002', battery: 45, status: 'idle', location: { lat: 37.55, lng: 126.98 } } ]; const assigner = new RobotAssigner(robots); const pickup = { lat: 37.56, lng: 126.97 }; const assigned = await assigner.assignBestRobot(pickup); expect(assigned.id).toBe('R001'); }); });

7.4 E2E 테스트

🎭 주요 E2E 테스트 시나리오

  • 정상 배송: 주문 → 로봇 할당 → 픽업 → 배송 → 완료
  • 배송 취소: 주문 → 취소 요청 → 로봇 복귀 → 환불
  • 로봇 장애: 배송 중 → 장애 발생 → 다른 로봇 재할당
  • 저배터리: 배송 중 → 배터리 부족 → 충전소 경유
  • 동시 주문: 여러 주문 → 최적 배차 → 병렬 배송
describe('E2E: Complete Delivery Flow', () => { it('should complete delivery from order to confirmation', async () => { // 1. 고객이 주문 생성 const order = await customerApp.createOrder({ items: ['피자 1판'], destination: '서울시 강남구 역삼동 123' }); expect(order.status).toBe('pending'); // 2. 시스템이 로봇 할당 await waitForCondition(() => order.status === 'assigned', 30000); expect(order.assignedRobot).toBeDefined(); // 3. 로봇이 배송지로 이동 await robotSimulator.moveToDestination(order.assignedRobot, order.destination); await waitForCondition(() => order.status === 'arrived', 120000); // 4. 고객이 수령 확인 await customerApp.confirmReceipt(order.id, { rating: 5 }); // 5. 배송 완료 확인 const completed = await customerApp.getOrder(order.id); expect(completed.status).toBe('completed'); }, 300000); });

// k6 부하 테스트 스크립트 import http from 'k6/http'; import { check, sleep } from 'k6'; export const options = { stages: [ { duration: '2m', target: 100 }, // 램프업 { duration: '5m', target: 100 }, // 유지 { duration: '2m', target: 500 }, // 스파이크 { duration: '5m', target: 500 }, // 유지 { duration: '2m', target: 0 }, // 램프다운 ], thresholds: { http_req_duration: ['p(99)<200'], // p99 < 200ms http_req_failed: ['rate<0.01'], // 에러율 < 1% }, }; export default function () { const res = http.post( 'https://api.delivery-robot.wia/v1/deliveries', JSON.stringify({ pickup: { lat: 37.5665, lng: 126.9780 }, destination: { lat: 37.5010, lng: 127.0390 } }), { headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${__ENV.TOKEN}` } } ); check(res, { 'status is 201': (r) => r.status === 201 }); sleep(1); }

7.8 장 요약

🎯 핵심 요점

  • 테스트 피라미드: 단위(70%) → 통합(20%) → E2E(10%) 비율
  • 코드 커버리지: Gold는 라인 85%, 브랜치 75% 이상
  • 통합 테스트: API, 데이터베이스, 메시지 큐 검증
  • E2E 테스트: 실제 사용자 시나리오 기반 전체 플로우
  • 성능 테스트: p99 응답시간, 처리량, 가용성 기준
  • 보안 테스트: SAST, DAST, 침투 테스트 정기 수행
  • WIA 인증: 4단계 등급, 2년 유효
