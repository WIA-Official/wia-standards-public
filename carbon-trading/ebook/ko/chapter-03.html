3장: 거래 상품

시장 깊이: WIA-CARBON_TRADING은 현물 계약, 선물, 옵션 및 구조화 상품을 포함한 포괄적인 거래 상품 모음을 지원하여 참가자가 다양한 시간 범위와 위험 프로필에 걸쳐 탄소 노출을 관리할 수 있도록 합니다.

탄소 시장은 효율적인 가격 발견, 위험 관리 및 유동성 공급을 가능하게 하는 정교한 금융 상품으로 발전했습니다. 이 장에서는 WIA-CARBON_TRADING에서 제공하는 거래 상품과 효과적인 탄소 가격 형성을 가능하게 하는 메커니즘을 탐구합니다.

주문장 역학

주문장은 다양한 가격 수준에서 매수 및 매도 주문을 집계하여 시장 깊이와 유동성에 대한 가시성을 제공합니다. 주문장 역학을 이해하는 것은 효과적인 거래 실행에 필수적입니다.

// 주문장 분석
class OrderBookAnalyzer {
    analyzeDepth(orderBook: OrderBook): MarketDepthAnalysis {
        const bidLevels = this.aggregateLevels(orderBook.bids);
        const askLevels = this.aggregateLevels(orderBook.asks);

        return {
            bidDepth: bidLevels,
            askDepth: askLevels,
            spread: askLevels[0]?.price - bidLevels[0]?.price,
            spreadPercent: (askLevels[0]?.price - bidLevels[0]?.price) / bidLevels[0]?.price * 100,
            imbalance: this.calculateImbalance(bidLevels, askLevels),
            liquidityScore: this.calculateLiquidityScore(bidLevels, askLevels)
        };
    }

    calculateImbalance(bids: PriceLevel[], asks: PriceLevel[]): number {
        const bidVolume = bids.slice(0, 5).reduce((sum, l) => sum + l.quantity, 0);
        const askVolume = asks.slice(0, 5).reduce((sum, l) => sum + l.quantity, 0);
        return (bidVolume - askVolume) / (bidVolume + askVolume);
    }

    estimateSlippage(side: 'buy' | 'sell', quantity: number, book: OrderBook): SlippageEstimate {
        const levels = side === 'buy' ? book.asks : book.bids;
        let remaining = quantity;
        let totalCost = 0;
        let executionLevels: ExecutionLevel[] = [];

        for (const level of levels) {
            const fillQty = Math.min(remaining, level.quantity);
            totalCost += fillQty * level.price;
            remaining -= fillQty;

            executionLevels.push({
                price: level.price,
                quantity: fillQty
            });

            if (remaining === 0) break;
        }

        const avgPrice = totalCost / (quantity - remaining);
        const bestPrice = levels[0].price;

        return {
            averagePrice: avgPrice,
            slippage: avgPrice - bestPrice,
            slippagePercent: (avgPrice - bestPrice) / bestPrice * 100,
            filledQuantity: quantity - remaining,
            unfilledQuantity: remaining,
            executionLevels
        };
    }
}

// 선물 가격 책정 모델
class FuturesPricer {
    calculateTheoreticalPrice(
        spotPrice: number,
        riskFreeRate: number,
        storageAndCarryCost: number,
        timeToExpiry: number,      // 연수
        convenienceYield: number
    ): number {
        // 보유 비용 모델
        // F = S * e^((r + c - y) * T)
        const netCarry = riskFreeRate + storageAndCarryCost - convenienceYield;
        return spotPrice * Math.exp(netCarry * timeToExpiry);
    }

    calculateBasis(futuresPrice: number, spotPrice: number): number {
        return futuresPrice - spotPrice;
    }

    calculateImpliedRepo(
        spotPrice: number,
        futuresPrice: number,
        timeToExpiry: number
    ): number {
        // 내재 금융 이자율
        return Math.log(futuresPrice / spotPrice) / timeToExpiry;
    }
}

탄소 옵션

탄소 옵션은 지정된 가격으로 탄소 크레딧을 매수(콜) 또는 매도(풋)할 수 있는 권리를 제공하지만 의무는 아닙니다. 옵션은 정교한 위험 관리 전략을 가능하게 하고 참가자가 가격 변동성에 대한 견해를 표현할 수 있게 합니다.

옵션 사양

// 탄소 옵션 계약
interface CarbonOption {
    optionId: string;
    underlying: string;          // 선물 계약 또는 현물
    optionType: 'call' | 'put';
    exerciseStyle: 'european' | 'american';
    strikePrice: number;
    expirationDate: Date;
    contractSize: number;
    premiumQuotation: 'per_tonne' | 'total';
    settlementMethod: 'physical' | 'cash';
}

// Black-Scholes를 사용한 옵션 가격 책정
class CarbonOptionPricer {
    private normalCDF(x: number): number {
        const a1 = 0.254829592;
        const a2 = -0.284496736;
        const a3 = 1.421413741;
        const a4 = -1.453152027;
        const a5 = 1.061405429;
        const p = 0.3275911;

        const sign = x < 0 ? -1 : 1;
        x = Math.abs(x) / Math.sqrt(2);

        const t = 1.0 / (1.0 + p * x);
        const y = 1.0 - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * Math.exp(-x * x);

        return 0.5 * (1.0 + sign * y);
    }

    priceOption(
        spotPrice: number,
        strikePrice: number,
        timeToExpiry: number,
        riskFreeRate: number,
        volatility: number,
        optionType: 'call' | 'put'
    ): OptionPricing {
        const d1 = (Math.log(spotPrice / strikePrice) +
                   (riskFreeRate + volatility * volatility / 2) * timeToExpiry) /
                   (volatility * Math.sqrt(timeToExpiry));
        const d2 = d1 - volatility * Math.sqrt(timeToExpiry);

        let price: number;
        let delta: number;

        if (optionType === 'call') {
            price = spotPrice * this.normalCDF(d1) -
                    strikePrice * Math.exp(-riskFreeRate * timeToExpiry) * this.normalCDF(d2);
            delta = this.normalCDF(d1);
        } else {
            price = strikePrice * Math.exp(-riskFreeRate * timeToExpiry) * this.normalCDF(-d2) -
                    spotPrice * this.normalCDF(-d1);
            delta = this.normalCDF(d1) - 1;
        }

        return { price, delta };
    }
}

가격 발견 메커니즘

효과적인 가격 발견은 시장 효율성에 필수적입니다. WIA-CARBON_TRADING은 정확하고 투명한 가격 형성을 보장하기 위해 여러 메커니즘을 사용합니다.

연속 거래

정규 거래 시간 동안 가격은 연속적인 주문 매칭을 통해 발견됩니다. 주문장의 최우선 매수가와 매도가가 현재 시장 스프레드를 결정하고, 체결된 거래가 거래 가격을 설정합니다.

경매 메커니즘

주기적인 경매는 주문을 집계하고 단일 청산 가격으로 실행하여 집중된 유동성 이벤트를 제공합니다. 경매는 의무 시장 허용량 배분에 특히 중요합니다.

// 경매 가격 발견
class AuctionEngine {
    private orders: AuctionOrder[] = [];

    determineAuctionPrice(): AuctionResult {
        // 매수 및 매도 주문 분리
        const buyOrders = this.orders.filter(o => o.side === 'buy')
                                     .sort((a, b) => b.price - a.price);
        const sellOrders = this.orders.filter(o => o.side === 'sell')
                                      .sort((a, b) => a.price - b.price);

        // 수요 및 공급 곡선 구축
        const demandCurve = this.buildCumulativeCurve(buyOrders, 'descending');
        const supplyCurve = this.buildCumulativeCurve(sellOrders, 'ascending');

        // 균형 가격 찾기 (최대 거래량)
        let maxVolume = 0;
        let clearingPrice = 0;

        const allPrices = [...new Set([
            ...buyOrders.map(o => o.price),
            ...sellOrders.map(o => o.price)
        ])].sort((a, b) => a - b);

        for (const price of allPrices) {
            const demandAtPrice = this.getQuantityAtPrice(demandCurve, price, 'demand');
            const supplyAtPrice = this.getQuantityAtPrice(supplyCurve, price, 'supply');
            const volumeAtPrice = Math.min(demandAtPrice, supplyAtPrice);

            if (volumeAtPrice > maxVolume) {
                maxVolume = volumeAtPrice;
                clearingPrice = price;
            }
        }

        // 청산 가격에서 주문 매칭
        const matches = this.matchAtPrice(clearingPrice, buyOrders, sellOrders);

        return {
            clearingPrice,
            totalVolume: maxVolume,
            matches,
            coverRatio: this.calculateCoverRatio(demandCurve, supplyCurve, clearingPrice)
        };
    }
}
