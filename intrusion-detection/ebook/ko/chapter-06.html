6.3.1 Snort to CEF ліҖнҷҳ

#!/usr/bin/env python3
import re
from datetime import datetime

class SnortToCEF:
    def __init__(self):
        self.cef_version = 0
        self.device_vendor = "Snort"
        self.device_product = "IDS"
        self.device_version = "2.9"

    def parse_snort_alert(self, snort_log):
        """Snort кІҪліҙ нҢҢмӢұ"""
        # Snort лЎңк·ё мҳҲмӢң:
        # [**] [1:1000001:1] SQL Injection мӢңлҸ„ нғҗм§Җ [**]
        # [Priority: 1] 12/26-10:30:45.123456
        # 192.168.1.100:54321 -> 10.0.0.50:80
        # TCP TTL:64 TOS:0x0 ID:12345 IpLen:20 DgmLen:60
        # ***A**** Seq: 0x123ABC  Ack: 0x456DEF  Win: 0x4000  TcpLen: 32

        pattern = r'\[\*\*\] \[(\d+):(\d+):(\d+)\] (.*?) \[\*\*\].*?Priority: (\d+).*?(\d{2}/\d{2}-\d{2}:\d{2}:\d{2}\.\d+).*?([\d.]+):(\d+) -> ([\d.]+):(\d+)'

        match = re.search(pattern, snort_log, re.DOTALL)
        if not match:
            return None

        gid, sid, rev, msg, priority, timestamp, src_ip, src_port, dst_ip, dst_port = match.groups()

        return {
            'gid': int(gid),
            'sid': int(sid),
            'revision': int(rev),
            'message': msg,
            'priority': int(priority),
            'timestamp': timestamp,
            'src_ip': src_ip,
            'src_port': int(src_port),
            'dst_ip': dst_ip,
            'dst_port': int(dst_port)
        }

    def to_cef(self, parsed_alert):
        """CEF нҳ•мӢқмңјлЎң ліҖнҷҳ"""
        if not parsed_alert:
            return None

        # CEF н—ӨлҚ”
        cef_header = (
            f"CEF:{self.cef_version}|"
            f"{self.device_vendor}|"
            f"{self.device_product}|"
            f"{self.device_version}|"
            f"{parsed_alert['sid']}|"  # Signature ID
            f"{parsed_alert['message']}|"  # Name
            f"{self.priority_to_severity(parsed_alert['priority'])}|"  # Severity
        )

        # CEF нҷ•мһҘ н•„л“ң
        extensions = [
            f"src={parsed_alert['src_ip']}",
            f"spt={parsed_alert['src_port']}",
            f"dst={parsed_alert['dst_ip']}",
            f"dpt={parsed_alert['dst_port']}",
            f"proto=TCP",
            f"act=blocked",
            f"rt={self.format_timestamp(parsed_alert['timestamp'])}",
            f"cs1Label=SnortRule",
            f"cs1={parsed_alert['gid']}:{parsed_alert['sid']}:{parsed_alert['revision']}"
        ]

        return cef_header + " ".join(extensions)

    def priority_to_severity(self, priority):
        """Snort мҡ°м„ мҲңмң„лҘј CEF мӢ¬к°ҒлҸ„лЎң ліҖнҷҳ"""
        # Snort: 1(лҶ’мқҢ) ~ 4(лӮ®мқҢ)
        # CEF: 0(лӮ®мқҢ) ~ 10(лҶ’мқҢ)
        severity_map = {1: 10, 2: 7, 3: 4, 4: 1}
        return severity_map.get(priority, 5)

    def format_timestamp(self, snort_timestamp):
        """нғҖмһ„мҠӨнғ¬н”„лҘј epoch л°ҖлҰ¬мҙҲлЎң ліҖнҷҳ"""
        # 12/26-10:30:45.123456 -> epoch ms
        dt = datetime.strptime(snort_timestamp.split('.')[0], "%m/%d-%H:%M:%S")
        dt = dt.replace(year=datetime.now().year)
        return int(dt.timestamp() * 1000)

# мӮ¬мҡ© мҳҲмӢң
converter = SnortToCEF()

snort_log = """
[**] [1:1000001:1] SQL Injection мӢңлҸ„ нғҗм§Җ [**]
[Priority: 1] 12/26-10:30:45.123456
192.168.1.100:54321 -> 10.0.0.50:80
TCP TTL:64 TOS:0x0 ID:12345 IpLen:20 DgmLen:60
"""

parsed = converter.parse_snort_alert(snort_log)
cef_log = converter.to_cef(parsed)
print(cef_log)

6.3.2 JSON нҳ•мӢқмңјлЎң м „мҶЎ

import json
import requests

class IDSToSIEM:
    def __init__(self, siem_endpoint, api_key):
        self.siem_endpoint = siem_endpoint
        self.api_key = api_key

    def send_alert(self, alert_data):
        """JSON нҳ•мӢқмңјлЎң SIEMм—җ кІҪліҙ м „мҶЎ"""
        # н‘ңмӨҖ JSON кө¬мЎ°
        json_alert = {
            "timestamp": alert_data['timestamp'],
            "event_type": "intrusion_detection",
            "severity": alert_data['severity'],
            "source": {
                "ip": alert_data['src_ip'],
                "port": alert_data['src_port'],
                "hostname": alert_data.get('src_hostname', 'unknown')
            },
            "destination": {
                "ip": alert_data['dst_ip'],
                "port": alert_data['dst_port'],
                "service": alert_data.get('dst_service', 'unknown')
            },
            "attack": {
                "signature_id": alert_data['signature_id'],
                "name": alert_data['attack_name'],
                "category": alert_data['category'],
                "cve": alert_data.get('cve', [])
            },
            "network": {
                "protocol": alert_data['protocol'],
                "bytes": alert_data.get('bytes', 0),
                "packets": alert_data.get('packets', 0)
            },
            "action": alert_data.get('action', 'alert'),
            "device": {
                "sensor_id": alert_data.get('sensor_id', 'ids-01'),
                "zone": alert_data.get('zone', 'internal')
            }
        }

        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }

        try:
            response = requests.post(
                self.siem_endpoint,
                json=json_alert,
                headers=headers,
                timeout=5
            )
            return response.status_code == 200
        except Exception as e:
            print(f"SIEM м „мҶЎ мӢӨнҢЁ: {e}")
            return False

# мӮ¬мҡ©
siem = IDSToSIEM("https://siem.company.com/api/events", "your-api-key")

alert = {
    'timestamp': '2024-12-26T10:30:45.123Z',
    'severity': 'high',
    'src_ip': '192.168.1.100',
    'src_port': 54321,
    'dst_ip': '10.0.0.50',
    'dst_port': 80,
    'signature_id': 1000001,
    'attack_name': 'SQL Injection Attempt',
    'category': 'web-application-attack',
    'protocol': 'TCP'
}

siem.send_alert(alert)

6.4 мқҙлІӨнҠё мғҒкҙҖ л¶„м„қ

SIEMмқҳ н•өмӢ¬ к°Җм№ҳлҠ” м—¬лҹ¬ мҶҢмҠӨмқҳ мқҙлІӨнҠёлҘј м—°кҙҖмӢңмјң лӢЁмқј мқҙлІӨнҠёлЎңлҠ” л°ңкІ¬н•ҳкё° м–ҙл Өмҡҙ ліөмһЎн•ң кіөкІ©мқ„ нғҗм§Җн•ҳлҠ” кІғмһ…лӢҲлӢӨ.

6.4.1 мғҒкҙҖ к·ңм№ҷ мң нҳ•

к·ңм№ҷ мң нҳ• м„ӨлӘ… мҳҲмӢң мӢңлӮҳлҰ¬мҳӨ
Simple лӢЁмқј мЎ°кұҙ л§Өм№ӯ IDS кІҪліҙ мӢ¬к°ҒлҸ„к°Җ 'critical'
Threshold мһ„кі„к°’ мҙҲкіј к°ҷмқҖ IPм—җм„ң 1мӢңк°„ лӮҙ IDS кІҪліҙ 10нҡҢ мқҙмғҒ
Sequence мҲңм°Ём Ғ мқҙлІӨнҠё нҸ¬нҠё мҠӨмә” вҶ’ IDS кІҪліҙ вҶ’ лЎңк·ёмқё м„ұкіө
Aggregation м—¬лҹ¬ мҶҢмҠӨ м§‘кі„ IDS + л°©нҷ”лІҪ + мӣ№ лЎңк·ё мЎ°н•©
Time-based мӢңк°„ нҢЁн„ҙ м—…л¬ҙ мҷё мӢңк°„ IDS кІҪліҙ

| join src_ip
    [search index=security sourcetype=ids attack_type!="port_scan"]
| where _time_join - _time < 600
| table _time src_ip dst_ip attack_type
| eval severity="critical"

# 2. IDS кІҪліҙмҷҖ лЎңк·ёмқё мӢӨнҢЁ мғҒкҙҖ
index=security (sourcetype=ids OR sourcetype=auth)
| eval event_type=case(
    sourcetype="ids", "ids_alert",
    sourcetype="auth" AND action="failed", "login_failed",
    1==1, "other"
  )
| transaction src_ip maxspan=5m
| where mvcount(event_type) > 1
| stats count by src_ip, dst_ip, event_type
| where count > 3

# 3. л©ҖнӢ° мҠӨн…Ңмқҙм§Җ кіөкІ© нғҗм§Җ
index=security
| eval stage=case(
    attack_type="reconnaissance", 1,
    attack_type="exploitation", 2,
    attack_type="privilege_escalation", 3,
    attack_type="lateral_movement", 4,
    attack_type="data_exfiltration", 5,
    1==1, 0
  )
| transaction src_ip maxspan=1h
| eval stages=mvdedup(stage)
| where mvcount(stages) >= 3

#!/usr/bin/env python3
from collections import defaultdict
from datetime import datetime, timedelta

class CorrelationEngine:
    def __init__(self):
        self.event_buffer = defaultdict(list)
        self.correlation_rules = []
        self.incidents = []

    def add_event(self, event):
        """мқҙлІӨнҠё м¶”к°Җ"""
        # мҶҢмҠӨ IPлі„лЎң мқҙлІӨнҠё лІ„нҚјл§Ғ
        key = event['src_ip']
        self.event_buffer[key].append(event)

        # мҳӨлһҳлҗң мқҙлІӨнҠё м ңкұ° (1мӢңк°„ мқҙмғҒ)
        cutoff = datetime.now() - timedelta(hours=1)
        self.event_buffer[key] = [
            e for e in self.event_buffer[key]
            if datetime.fromisoformat(e['timestamp']) > cutoff
        ]

        # мғҒкҙҖ к·ңм№ҷ нҸүк°Җ
        self.evaluate_rules(key)

    def add_rule(self, rule):
        """мғҒкҙҖ к·ңм№ҷ м¶”к°Җ"""
        self.correlation_rules.append(rule)

    def evaluate_rules(self, src_ip):
        """лӘЁл“  к·ңм№ҷ нҸүк°Җ"""
        events = self.event_buffer[src_ip]

        for rule in self.correlation_rules:
            if rule.matches(events):
                incident = rule.create_incident(events)
                self.incidents.append(incident)
                self.trigger_alert(incident)

    def trigger_alert(self, incident):
        """мӮ¬кі  кІҪліҙ л°ңмғқ"""
        print(f"[мӮ¬кі  нғҗм§Җ] {incident['name']}")
        print(f"  мӢ¬к°ҒлҸ„: {incident['severity']}")
        print(f"  кҙҖл Ё мқҙлІӨнҠё: {len(incident['events'])}к°ң")
        print(f"  м„ӨлӘ…: {incident['description']}")

class MultiStageAttackRule:
    """лӢӨлӢЁкі„ кіөкІ© нғҗм§Җ к·ңм№ҷ"""
    def __init__(self):
        self.name = "Multi-Stage Attack"
        self.required_stages = ['port_scan', 'exploitation', 'privilege_escalation']

    def matches(self, events):
        """к·ңм№ҷ л§Өм№ӯ"""
        if len(events) < 3:
            return False

        # мқҙлІӨнҠё мң нҳ• м¶”м¶ң
        event_types = [e.get('attack_type', '') for e in events]

        # н•„мҡ”н•ң лӘЁл“  лӢЁкі„к°Җ мһҲлҠ”м§Җ нҷ•мқё
        return all(stage in event_types for stage in self.required_stages)

    def create_incident(self, events):
        """мӮ¬кі  к°қмІҙ мғқм„ұ"""
        return {
            'name': self.name,
            'severity': 'critical',
            'timestamp': datetime.now().isoformat(),
            'events': events,
            'description': f"лӢӨлӢЁкі„ кіөкІ© нғҗм§Җ: {' -> '.join(self.required_stages)}",
            'src_ip': events[0]['src_ip'],
            'dst_ips': list(set(e['dst_ip'] for e in events))
        }

class BruteForceAfterIDSRule:
    """IDS кІҪліҙ нӣ„ л¬ҙм°Ёлі„ лҢҖмһ… кіөкІ©"""
    def __init__(self):
        self.name = "IDS Alert Followed by Brute Force"

    def matches(self, events):
        # IDS кІҪліҙк°Җ мһҲлҠ”м§Җ
        has_ids_alert = any(e.get('source') == 'ids' for e in events)

        # мқҙнӣ„ лЎңк·ёмқё мӢӨнҢЁк°Җ 5нҡҢ мқҙмғҒмқём§Җ
        login_failures = [e for e in events if e.get('event_type') == 'login_failed']

        return has_ids_alert and len(login_failures) >= 5

    def create_incident(self, events):
        return {
            'name': self.name,
            'severity': 'high',
            'timestamp': datetime.now().isoformat(),
            'events': events,
            'description': "IDS нғҗм§Җ нӣ„ л¬ҙм°Ёлі„ лҢҖмһ… кіөкІ© мӢңлҸ„",
            'src_ip': events[0]['src_ip']
        }

# мӮ¬мҡ© мҳҲмӢң

#!/usr/bin/env python3
import requests
import paramiko

class AutomatedResponse:
    def __init__(self, firewall_api, switch_api):
        self.firewall_api = firewall_api
        self.switch_api = switch_api
        self.action_log = []

    def handle_incident(self, incident):
        """мӮ¬кі м—җ л”°лҘё мһҗлҸҷ лҢҖмқ‘"""
        severity = incident['severity']
        src_ip = incident['src_ip']

        actions = []

        if severity == 'critical':
            # мҰүмӢң м°ЁлӢЁ
            actions.append(self.block_ip(src_ip))
            actions.append(self.isolate_host(src_ip))
            actions.append(self.collect_forensics(src_ip))
            actions.append(self.create_ticket(incident, priority='P1'))

        elif severity == 'high':
            # м ңн•ңм Ғ лҢҖмқ‘
            actions.append(self.rate_limit(src_ip))
            actions.append(self.create_ticket(incident, priority='P2'))
            actions.append(self.notify_analyst(incident))

        elif severity == 'medium':
            # лӘЁлӢҲн„°л§Ғ к°•нҷ”
            actions.append(self.enable_detailed_logging(src_ip))
            actions.append(self.create_ticket(incident, priority='P3'))

        else:  # low
            # кё°лЎқл§Ң
            actions.append(self.log_event(incident))

        # лӘЁл“  м•Ўм…ҳ кё°лЎқ
        self.action_log.extend(actions)

        return actions

    def block_ip(self, ip_address):
        """л°©нҷ”лІҪм—җм„ң IP м°ЁлӢЁ"""
        try:
            response = requests.post(
                f"{self.firewall_api}/block",
                json={'ip': ip_address, 'duration': 3600},  # 1мӢңк°„
                timeout=5
            )

            return {
                'action': 'block_ip',
                'ip': ip_address,
                'status': 'success' if response.ok else 'failed',
                'timestamp': datetime.now().isoformat()
            }
        except Exception as e:
            return {'action': 'block_ip', 'status': 'error', 'error': str(e)}

    def isolate_host(self, ip_address):
        """нҳёмҠӨнҠёлҘј кІ©лҰ¬ VLANмңјлЎң мқҙлҸҷ"""
        try:
            # мҠӨмң„м№ҳм—җ м ‘мҶҚн•ҳм—¬ нҸ¬нҠё VLAN ліҖкІҪ
            # (мӢӨм ңлЎңлҠ” APIлӮҳ NETCONF мӮ¬мҡ©)
            response = requests.post(
                f"{self.switch_api}/isolate",
                json={'ip': ip_address, 'vlan': 999},  # кІ©лҰ¬ VLAN
                timeout=5
            )

            return {
                'action': 'isolate_host',
                'ip': ip_address,
                'status': 'success' if response.ok else 'failed',
                'timestamp': datetime.now().isoformat()
            }
        except Exception as e:
            return {'action': 'isolate_host', 'status': 'error', 'error': str(e)}

    def collect_forensics(self, ip_address):
        """нҸ¬л ҢмӢқ лҚ°мқҙн„° мҲҳм§‘"""
        forensic_data = {
            'ip': ip_address,
            'timestamp': datetime.now().isoformat(),
            'collected': []
        }

        # 1. л©”лӘЁлҰ¬ лҚӨн”„ нҠёлҰ¬кұ°
        # 2. л„ӨнҠёмӣҢнҒ¬ мәЎмІҳ мӢңмһ‘
        # 3. н”„лЎңм„ёмҠӨ лҰ¬мҠӨнҠё мҲҳм§‘
        # 4. нҷңм„ұ м—°кІ° лӘ©лЎқ

        forensic_data['collected'] = [
            'memory_dump', 'network_capture',
            'process_list', 'network_connections'
        ]

        return {
            'action': 'collect_forensics',
            'data': forensic_data,
            'status': 'success'
        }

    def create_ticket(self, incident, priority):
        """ITSM нӢ°мј“ мғқм„ұ"""
        ticket_data = {
            'title': f"Security Incident: {incident['name']}",
            'description': incident.get('description', ''),
            'priority': priority,
            'category': 'Security',
            'affected_systems': incident.get('dst_ips', []),
            'attacker': incident.get('src_ip', 'unknown')
        }

        # ITSM API нҳём¶ң (ServiceNow, Jira л“ұ)
        # response = requests.post(itsm_api, json=ticket_data)

        return {
            'action': 'create_ticket',
            'ticket_id': 'INC-12345',  # мӢӨм ңлЎңлҠ” API мқ‘лӢөм—җм„ң
            'priority': priority,
            'status': 'success'
        }

    def notify_analyst(self, incident):
        """ліҙм•Ҳ л¶„м„қк°Җм—җкІҢ м•ҢлҰј"""
        # Slack, Email, SMS л“ұ
        notification = {
            'channel': '#security-alerts',
            'message': f"рҹҡЁ {incident['severity'].upper()} мӮ¬кі  нғҗм§Җ\n"
                      f"мң нҳ•: {incident['name']}\n"
                      f"м¶ңл°ңм§Җ: {incident['src_ip']}\n"
                      f"мӢңк°„: {incident['timestamp']}"
        }

        # Slack webhook нҳём¶ң
        # requests.post(slack_webhook, json=notification)

        return {'action': 'notify_analyst', 'status': 'success'}

    def rate_limit(self, ip_address):
        """мҶҚлҸ„ м ңн•ң м Ғмҡ©"""
        return {
            'action': 'rate_limit',
            'ip': ip_address,
            'limit': '100 req/min',
            'status': 'success'
        }

    def enable_detailed_logging(self, ip_address):
        """мғҒм„ё лЎңк№… нҷңм„ұнҷ”"""
        return {
            'action': 'enable_logging',
            'ip': ip_address,
            'status': 'success'
        }

    def log_event(self, incident):
        """мқҙлІӨнҠё кё°лЎқ"""
        return {
            'action': 'log_event',
            'incident_id': incident.get('id', 'unknown'),
            'status': 'logged'
        }

GDPR лҚ°мқҙн„° мң м¶ң м•ҢлҰј (72мӢңк°„) мң м¶ң нғҗм§Җ л°Ҹ м•ҢлҰј
ISO 27001 ліҙм•Ҳ мӮ¬кі  кҙҖлҰ¬ мӮ¬кі  кё°лЎқ л°Ҹ лҢҖмқ‘

мҡ”м•Ҫ

SIEMкіј IDS/IPSмқҳ нҶөн•©мқҖ мЎ°м§Ғмқҳ ліҙм•Ҳ к°ҖмӢңм„ұкіј лҢҖмқ‘ лҠҘл Ҙмқ„ нҒ¬кІҢ н–ҘмғҒмӢңнӮөлӢҲлӢӨ. SIEMмқҖ лЎңк·ё мҲҳм§‘, м •к·ңнҷ”, мғҒкҙҖ л¶„м„қ, мһҘкё° ліҙкҙҖ, мӢӨмӢңк°„ лӘЁлӢҲн„°л§Ғ кё°лҠҘмқ„ м ңкіөн•ҳл©°, IDS кІҪліҙлҠ” SIEMмқҳ мЈјмҡ” лҚ°мқҙн„° мҶҢмҠӨмһ…лӢҲлӢӨ. нҡЁкіјм Ғмқё нҶөн•©мқ„ мң„н•ҙм„ңлҠ” Syslog, CEF, JSON л“ұ н‘ңмӨҖ нҳ•мӢқмңјлЎң лҚ°мқҙн„°лҘј м „мҶЎн•ҳкі , SIEMм—җм„ң нҶөн•© л¶„м„қмқҙ к°ҖлҠҘн•ҳлҸ„лЎқ м •к·ңнҷ”н•ҙм•ј н•©лӢҲлӢӨ.

мқҙлІӨнҠё мғҒкҙҖ л¶„м„қмқҖ SIEMмқҳ н•өмӢ¬ к°Җм№ҳлЎң, лӢЁмқј IDS кІҪліҙл§ҢмңјлЎңлҠ” л°ңкІ¬н•ҳкё° м–ҙл Өмҡҙ лӢӨлӢЁкі„ кіөкІ©мқҙлӮҳ ліөмһЎн•ң мң„нҳ‘ нҢЁн„ҙмқ„ нғҗм§Җн•©лӢҲлӢӨ. Simple, Threshold, Sequence, Aggregation л“ұ лӢӨм–‘н•ң мғҒкҙҖ к·ңм№ҷмқ„ нҶөн•ҙ нҸ¬нҠё мҠӨмә” нӣ„ м№Ёмһ… мӢңлҸ„, IDS кІҪліҙ нӣ„ л¬ҙм°Ёлі„ лҢҖмһ… кіөкІ© л“ұмқ„ мӢқлі„н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

SOAR н”Ңлһ«нҸјкіјмқҳ нҶөн•©мңјлЎң нғҗм§Җлҗң мң„нҳ‘м—җ мһҗлҸҷмңјлЎң лҢҖмқ‘н•  мҲҳ мһҲмҠөлӢҲлӢӨ. мӢ¬к°ҒлҸ„м—җ л”°лқј IP м°ЁлӢЁ, нҳёмҠӨнҠё кІ©лҰ¬, нҸ¬л ҢмӢқ лҚ°мқҙн„° мҲҳм§‘, нӢ°мј“ мғқм„ұ л“ұмқҳ м•Ўм…ҳмқ„ мһҗлҸҷнҷ”н•ҳм—¬ лҢҖмқ‘ мӢңк°„мқ„ лҢҖнҸӯ лӢЁм¶•н•©лӢҲлӢӨ. лҢҖмӢңліҙл“ңмҷҖ мӢңк°Ғнҷ”лҠ” ліҙм•Ҳ мғҒнҷ©мқ„ м§ҒкҙҖм ҒмңјлЎң нҢҢм•…н•  мҲҳ мһҲкІҢ н•ҳл©°, PCI-DSS, HIPAA, GDPR л“ұ к·ңм • мӨҖмҲҳ мҡ”кө¬мӮ¬н•ӯмқ„ м¶©мЎұн•ҳлҠ” лҚ° н•„мҲҳм Ғмһ…лӢҲлӢӨ.

ліөмҠө л¬ём ң

  1. SIEMмқҳ 6к°Җм§Җ н•өмӢ¬ кё°лҠҘмқ„ лӮҳм—ҙн•ҳкі  к°Ғк°Ғмқ„ м„ӨлӘ…н•ҳм„ёмҡ”.
  2. IDS кІҪліҙлҘј SIEMмңјлЎң м „мҶЎн•ҳлҠ” 5к°Җм§Җ л°©лІ•мқҳ мһҘлӢЁм җмқ„ л№„көҗн•ҳм„ёмҡ”.
  3. CEF(Common Event Format)мқҳ кө¬мЎ°мҷҖ IDS лҚ°мқҙн„°лҘј CEFлЎң ліҖнҷҳн•ҳлҠ” мқҙмң лҘј м„ӨлӘ…н•ҳм„ёмҡ”.
  4. мқҙлІӨнҠё мғҒкҙҖ л¶„м„қмқҳ 5к°Җм§Җ мЈјмҡ” к·ңм№ҷ мң нҳ•мқ„ м„ӨлӘ…н•ҳкі  к°Ғк°Ғмқҳ мҳҲмӢңлҘј м ңмӢңн•ҳм„ёмҡ”.
  5. лӢӨлӢЁкі„ кіөкІ©мқ„ нғҗм§Җн•ҳкё° мң„н•ң мғҒкҙҖ к·ңм№ҷмқ„ м„Өкі„н•ҳм„ёмҡ”.
  6. SOARмқҳ к°ңл…җкіј SIEMкіјмқҳ м°Ёмқҙм җмқ„ м„ӨлӘ…н•ҳм„ёмҡ”.
  7. мӢ¬к°ҒлҸ„м—җ л”°лҘё мһҗлҸҷ лҢҖмқ‘ м „лһөмқ„ 3лӢЁкі„(Critical, High, Medium)лЎң м„Өкі„н•ҳм„ёмҡ”.
  8. ліҙм•Ҳ лҢҖмӢңліҙл“ңм—җ нҸ¬н•Ёлҗҳм–ҙм•ј н•  н•„мҲҳ мң„м Ҝ 6к°Җм§ҖлҘј м ңмӢңн•ҳкі  к°Ғк°Ғмқҳ мҡ©лҸ„лҘј м„ӨлӘ…н•ҳм„ёмҡ”.
  9. PCI-DSS, HIPAA, GDPR мӨ‘ н•ҳлӮҳлҘј м„ нғқн•ҳм—¬ IDS/SIEMмқҙ м–ҙл–»кІҢ к·ңм • мӨҖмҲҳлҘј м§Җмӣҗн•ҳлҠ”м§Җ м„ӨлӘ…н•ҳм„ёмҡ”.
  10. IDSмҷҖ SIEM нҶөн•© мӢң л°ңмғқн•  мҲҳ мһҲлҠ” м„ұлҠҘ л¬ём ңмҷҖ н•ҙкІ° л°©лІ•мқ„ л…јмқҳн•ҳм„ёмҡ”.
