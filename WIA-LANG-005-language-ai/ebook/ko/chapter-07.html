언어 학습 앱 통합

// 언어 학습 앱 통합 예시
class LanguageLearningIntegration {
  private client: WIALangClient;

  constructor(apiKey: string) {
    this.client = new WIALangClient({ apiKey });
  }

  // 발음 평가
  async evaluatePronunciation(
    audio: ArrayBuffer,
    targetText: string,
    language: string
  ): Promise<PronunciationScore> {
    // 1. 음성 인식
    const transcription = await this.client.transcribe(audio, {
      language,
      timestamps: true
    });

    // 2. 발음 점수 계산
    const score = await this.client.analyzePronunciation({
      expected: targetText,
      actual: transcription.text,
      audioFeatures: transcription.features
    });

    return {
      overallScore: score.overall,
      phonemeScores: score.phonemes,
      suggestions: score.improvements,
      nativeSpeakerComparison: score.comparison
    };
  }

  // 문법 교정
  async correctGrammar(
    text: string,
    language: string
  ): Promise<GrammarCorrection> {
    return this.client.analyze(text, {
      language,
      tasks: ['grammar', 'spelling', 'style']
    });
  }

  // 맥락 기반 어휘 학습
  async suggestVocabulary(
    context: string,
    userLevel: 'beginner' | 'intermediate' | 'advanced',
    language: string
  ): Promise<VocabularySuggestion[]> {
    return this.client.complete(context, {
      language,
      task: 'vocabulary_suggestion',
      difficulty: userLevel,
      includeExamples: true,
      includeCulturalNotes: true
    });
  }
}

실시간 채팅 번역

// WebSocket 기반 실시간 번역
class RealtimeChatTranslator {
  private ws: WebSocket;
  private translationBuffer: Map<string, string> = new Map();

  constructor(private client: WIALangClient) {}

  async connect(roomId: string, userLanguage: string) {
    this.ws = new WebSocket(`wss://chat.example.com/rooms/${roomId}`);

    this.ws.onmessage = async (event) => {
      const message = JSON.parse(event.data);

      if (message.language !== userLanguage) {
        // 다른 언어 메시지 자동 번역
        const translated = await this.client.translate({
          text: message.text,
          from: message.language,
          to: userLanguage,
          preserveEmoji: true
        });

        this.emit('translated-message', {
          original: message,
          translated: translated.text,
          confidence: translated.confidence
        });
      } else {
        this.emit('message', message);
      }
    };
  }

  async sendMessage(text: string, targetLanguages: string[]) {
    // 병렬로 여러 언어로 번역
    const translations = await Promise.all(
      targetLanguages.map(lang =>
        this.client.translate({ text, to: lang })
      )
    );

    this.ws.send(JSON.stringify({
      text,
      translations: Object.fromEntries(
        targetLanguages.map((lang, i) => [lang, translations[i].text])
      )
    }));
  }
}
