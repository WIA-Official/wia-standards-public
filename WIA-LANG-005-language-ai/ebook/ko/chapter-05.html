5.1 RESTful API 설계

API는 언어 AI 시스템에 접근하는 관문입니다. WIA-LANG-005 Phase 2는 일관되고 직관적이며 확장 가능한 API를 정의합니다.

// API 기본 구조
const BASE_URL = 'https://api.wia-lang-005.org/v1';

interface APIRequest {
  language: string;      // ISO 639-3 코드
  input: any;
  options?: RequestOptions;
}

interface APIResponse<T> {
  status: 'success' | 'error';
  data: T;
  metadata: {
    requestId: string;
    timestamp: string;
    latencyMs: number;
    modelVersion: string;
  };
  usage: {
    characters: number;
    tokens: number;
    creditsUsed: number;
  };
}

번역 API

// 번역 API 클라이언트
class TranslationClient {
  private apiKey: string;

  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async translate(request: TranslateRequest): Promise<TranslateResponse> {
    const response = await fetch(`${BASE_URL}/translate`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.apiKey}`,
        'X-WIA-Version': '1.0.0'
      },
      body: JSON.stringify({
        source_language: request.sourceLanguage,
        target_language: request.targetLanguage,
        text: request.text,
        options: {
          formality: request.formality || 'polite',
          domain: request.domain || 'general',
          alternatives: request.alternatives || 3
        }
      })
    });

    return response.json();
  }
}

// 사용 예시
const client = new TranslationClient('your-api-key');
const result = await client.translate({
  sourceLanguage: 'ain',
  targetLanguage: 'ja',
  text: 'イランカラプテ',
  formality: 'polite'
});

console.log(result.data.translation);  // 'こんにちは'
console.log(result.data.confidence);   // 0.92

음성 인식 API (ASR)

// ASR API
interface ASRRequest {
  audio: string;       // Base64 인코딩
  language: string;
  format: 'wav' | 'mp3' | 'flac';
  sampleRate?: number;
}

interface ASRResponse {
  transcription: string;
  confidence: number;
  timestamps: WordTimestamp[];
  alternatives?: Alternative[];
}

async function transcribeAudio(
  audioFile: File,
  language: string
): Promise<ASRResponse> {
  const base64Audio = await fileToBase64(audioFile);

  const response = await fetch(`${BASE_URL}/asr`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${API_KEY}`
    },
    body: JSON.stringify({
      audio: base64Audio,
      language: language,
      format: 'wav',
      options: {
        timestamps: true,
        punctuation: true,
        profanity_filter: false
      }
    })
  });

  return response.json();
}
