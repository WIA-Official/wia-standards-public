```python
from prometheus_client import Counter, Histogram, Gauge
import time

# 메트릭 정의
translation_requests_total = Counter(
    'wia_translation_requests_total',
    'Total translation requests',
    ['source_lang', 'target_lang', 'status']
)

translation_duration_seconds = Histogram(
    'wia_translation_duration_seconds',
    'Translation duration in seconds',
    ['source_lang', 'target_lang'],
    buckets=[0.1, 0.5, 1.0, 2.0, 5.0, 10.0]
)

active_requests = Gauge(
    'wia_translation_active_requests',
    'Number of active translation requests'
)

model_load_time = Gauge(
    'wia_model_load_time_seconds',
    'Time to load model',
    ['model_name']
)

# 메트릭 사용
class TranslationService:
    def translate(self, text, source_lang, target_lang):
        active_requests.inc()
        start_time = time.time()

        try:
            result = self._do_translation(text, source_lang, target_lang)
            status = 'success'
            return result
        except Exception as e:
            status = 'error'
            raise
        finally:
            duration = time.time() - start_time
            translation_duration_seconds.labels(
                source_lang=source_lang,
                target_lang=target_lang
            ).observe(duration)

            translation_requests_total.labels(
                source_lang=source_lang,
                target_lang=target_lang,
                status=status
            ).inc()

            active_requests.dec()
```

```python
from opentelemetry import trace
from opentelemetry.exporter.jaeger.thrift import JaegerExporter
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor

# 추적 설정
trace.set_tracer_provider(TracerProvider())
tracer = trace.get_tracer(__name__)

jaeger_exporter = JaegerExporter(
    agent_host_name="jaeger-agent.wia-lang-005.svc.cluster.local",
    agent_port=6831,
)

trace.get_tracer_provider().add_span_processor(
    BatchSpanProcessor(jaeger_exporter)
)

# 서비스에서 사용
class TranslationService:
    def translate(self, text, source_lang, target_lang):
        with tracer.start_as_current_span("translate") as span:
            span.set_attribute("source_lang", source_lang)
            span.set_attribute("target_lang", target_lang)
            span.set_attribute("text_length", len(text))

            # 모델 로드
            with tracer.start_as_current_span("load_model"):
                model = self.load_model(source_lang, target_lang)

            # 전처리
            with tracer.start_as_current_span("preprocess"):
                preprocessed = self.preprocess(text)

            # 번역
            with tracer.start_as_current_span("inference"):
                result = model.translate(preprocessed)

            # 후처리
            with tracer.start_as_current_span("postprocess"):
                final = self.postprocess(result)

            span.set_attribute("confidence", final.confidence)
            return final
```

```python
import logging
import json
from datetime import datetime

class StructuredLogger:
    """구조화된 JSON 로깅"""

    def __init__(self, service_name):
        self.logger = logging.getLogger(service_name)
        self.service_name = service_name

    def log(self, level, message, **kwargs):
        log_entry = {
            "timestamp": datetime.utcnow().isoformat(),
            "service": self.service_name,
            "level": level,
            "message": message,
            **kwargs
        }

        self.logger.log(
            getattr(logging, level.upper()),
            json.dumps(log_entry)
        )

    def info(self, message, **kwargs):
        self.log("info", message, **kwargs)

    def error(self, message, **kwargs):
        self.log("error", message, **kwargs)

# 사용 예
```

```python
class LanguageLearningApp:
    """WIA-LANG-005를 사용하는 언어 학습 앱"""

    def __init__(self):
        self.wia_client = WIALang005Client(
            api_key=os.getenv("WIA_API_KEY")
        )

    def practice_pronunciation(self, target_language):
        """발음 연습 기능"""
        # 1. TTS로 예문 생성
        example_text = self.get_example_sentence(target_language)
        example_audio = self.wia_client.synthesize(
            text=example_text,
            language=target_language,
            voice="default"
        )

        # 2. 학생에게 듣기
        self.play_audio(example_audio)

        # 3. 학생 발음 녹음
        student_audio = self.record_audio()

        # 4. ASR로 인식
        student_transcript = self.wia_client.transcribe(
            audio=student_audio,
            language=target_language
        )

        # 5. 정확도 평가
        accuracy = self.calculate_pronunciation_accuracy(
            expected=example_text,
            actual=student_transcript.text
        )

        return {
            "expected": example_text,
            "actual": student_transcript.text,
            "accuracy": accuracy,
            "feedback": self.generate_feedback(accuracy)
        }

    def translate_lesson(self, lesson_text, student_native_lang, target_lang):
        """수업 자료 번역"""
        # 단락별로 번역 (컨텍스트 유지)
        paragraphs = lesson_text.split('\n\n')
        translations = []

        for para in paragraphs:
            translation = self.wia_client.translate(
                text=para,
                source_lang=target_lang,
                target_lang=student_native_lang
            )
            translations.append({
                "original": para,
                "translation": translation.translation,
                "confidence": translation.confidence
            })

        return translations
```

```python
class CulturalHeritageArchive:
    """구전 문화 유산 디지털화"""

    def __init__(self):
        self.wia_client = WIALang005Client(
            api_key=os.getenv("WIA_API_KEY")
        )
        self.db = Database()

    def process_oral_recording(self, audio_file, metadata):
        """구전 녹음 처리"""
        language = metadata['language']

        # 1. 음성 인식
        transcription = self.wia_client.transcribe(
            audio=audio_file,
            language=language
        )

        # 2. 주요 언어로 번역 (접근성 향상)
        translations = {}
        for target_lang in ['en', 'ja', 'ko']:
            if target_lang != language:
                trans = self.wia_client.translate(
                    text=transcription.text,
                    source_lang=language,
                    target_lang=target_lang
                )
                translations[target_lang] = trans.translation

        # 3. 개체명 인식 (인물, 장소, 신화적 존재 등)
        entities = self.wia_client.extract_entities(
            text=transcription.text,
            language=language,
            entity_types=['PERSON', 'LOCATION', 'DEITY', 'ARTIFACT']
        )

        # 4. 데이터베이스 저장
        record_id = self.db.save_oral_tradition({
            "audio_file": audio_file,
            "language": language,
            "transcription": transcription.text,
            "translations": translations,
            "entities": entities,
            "metadata": metadata,
            "timestamp": datetime.now()
        })

        # 5. 검색 인덱스 업데이트
        self.update_search_index(record_id, transcription.text, entities)

        return record_id

    def search_stories(self, query, language='en'):
        """이야기 검색 (다국어)"""
        # 쿼리를 임베딩으로 변환
        query_embedding = self.wia_client.get_embeddings(
            texts=[query],
            language=language
        )

        # 벡터 유사도 검색
        similar_stories = self.db.vector_search(
            embedding=query_embedding[0],
            limit=10
        )

        return similar_stories
```

### 3. 실시간 통역 시스템

```python
class RealTimeInterpretation:
    """실시간 양방향 통역"""

    def __init__(self, lang_a, lang_b):
        self.wia_client = WIALang005Client(
            api_key=os.getenv("WIA_API_KEY")
        )
        self.lang_a = lang_a
        self.lang_b = lang_b

    async def interpret_stream(self, audio_stream_a, audio_stream_b):
        """양방향 오디오 스트림 통역"""

        async def process_stream(audio_stream, source_lang, target_lang, output_queue):
            """한 방향 스트림 처리"""
            async for transcript in self.wia_client.streaming_asr(
                audio_stream, source_lang
            ):
                if transcript.is_final:
                    # 번역
                    translation = await self.wia_client.translate_async(
                        text=transcript.text,
                        source_lang=source_lang,
                        target_lang=target_lang
                    )

                    # TTS
                    audio = await self.wia_client.synthesize_async(
                        text=translation.translation,
                        language=target_lang
                    )

                    # 출력 큐에 추가
                    await output_queue.put({
                        "original": transcript.text,
                        "translation": translation.translation,
                        "audio": audio
                    })

        # 양방향 처리
        queue_a_to_b = asyncio.Queue()
        queue_b_to_a = asyncio.Queue()

        await asyncio.gather(
            process_stream(audio_stream_a, self.lang_a, self.lang_b, queue_a_to_b),
            process_stream(audio_stream_b, self.lang_b, self.lang_a, queue_b_to_a)
        )

        return queue_a_to_b, queue_b_to_a
```

```yaml
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3

    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.11'

    - name: Install dependencies
      run: |
        pip install -r requirements.txt
        pip install pytest pytest-cov

    - name: Run tests
      run: |
        pytest --cov=wia_lang_005 --cov-report=xml

    - name: Upload coverage
      uses: codecov/codecov-action@v3

  build:
    needs: test
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3

    - name: Set up Docker Buildx
      uses: docker/setup-buildx-action@v2

    - name: Login to Docker Hub
      uses: docker/login-action@v2
      with:
        username: ${{ secrets.DOCKER_USERNAME }}
        password: ${{ secrets.DOCKER_TOKEN }}

    - name: Build and push
      uses: docker/build-push-action@v4
      with:
        context: .
        push: true
        tags: wia/translation:${{ github.sha }},wia/translation:latest
        cache-from: type=registry,ref=wia/translation:latest
        cache-to: type=inline

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
    - uses: actions/checkout@v3

    - name: Configure AWS credentials
      uses: aws-actions/configure-aws-credentials@v2
      with:
        aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
        aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        aws-region: us-west-2

    - name: Update kubeconfig
      run: |
        aws eks update-kubeconfig --name wia-lang-005 --region us-west-2

    - name: Deploy to Kubernetes
      run: |
        kubectl set image deployment/wia-translation \
          translation=wia/translation:${{ github.sha }} \
          -n wia-lang-005

    - name: Wait for rollout
      run: |
        kubectl rollout status deployment/wia-translation -n wia-lang-005
```
