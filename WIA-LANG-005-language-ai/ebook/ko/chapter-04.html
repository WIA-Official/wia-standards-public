4.1 메타데이터 스키마

언어 정보 (Language Metadata)

// 언어 메타데이터 인터페이스
interface LanguageMetadata {
  code: string;           // ISO 639-3 코드
  name: string;           // 영어 이름
  nativeName: string;     // 원어민 표기
  glottologId: string;    // Glottolog ID
  family: string;         // 언어 가족
  endangerment: {
    status: EndangermentLevel;
    unescoLevel: number;  // 0-5
    speakerCount: number;
    lastUpdated: string;
  };
  region: {
    countries: string[];
    areas: string[];
    coordinates: { lat: number; lng: number };
  };
  script: {
    primary: string;
    romanization: string;
    unicodeBlocks: string[];
  };
  dialects: Dialect[];
}

type EndangermentLevel =
  | 'safe'
  | 'vulnerable'
  | 'definitely_endangered'
  | 'severely_endangered'
  | 'critically_endangered'
  | 'extinct';

4.4 데이터 검증

// WIA-LANG-005 데이터 검증기
class WIALang005Validator {
  async validateDataset(path: string): Promise<ValidationResult> {
    const errors: string[] = [];

    // 1. 스키마 검증
    if (!await this.validateSchema(path)) {
      errors.push('Schema validation failed');
    }

    // 2. 메타데이터 검증
    const metadata = await this.loadMetadata(path);
    if (!this.validateMetadata(metadata)) {
      errors.push('Invalid metadata');
    }

    // 3. 언어 코드 검증 (ISO 639-3)
    if (!this.validateLanguageCode(metadata.language.code)) {
      errors.push('Invalid ISO 639-3 code');
    }

    // 4. 데이터 무결성
    if (!await this.validateDataIntegrity(path)) {
      errors.push('Data integrity failed');
    }

    // 5. 체크섬 검증
    if (!await this.validateChecksums(path)) {
      errors.push('Checksum mismatch');
    }

    return {
      valid: errors.length === 0,
      errors
    };
  }
}

4.5 프라이버시 및 문화적 민감성

ℹ️ 문화적 민감성 태그:
  • sacred: 신성한 내용 - 접근 제한, 장로 승인 필요
  • sensitive: 민감 정보 - 등록 연구자만
  • public: 공개 가능 - 오픈 액세스
// 익명화 요구사항
interface PrivacyConfig {
  anonymizationRequired: boolean;
  piiFields: string[];  // 개인정보 필드
  anonymizationMethods: {
    speakerId: 'hashed_uuid';
    location: 'generalized_to_region';
    dates: 'year_only';
  };
  retention: {
    rawAudio: 'encrypted_storage';
    transcriptions: 'hashed_speaker_ids';
    accessLogs: '90_days';
  };
}

  6. 프라이버시: 익명화와 문화적 민감성 관리를 내장합니다.
