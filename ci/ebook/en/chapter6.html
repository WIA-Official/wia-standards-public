6.1 Security Overview

WIA-CI implements comprehensive security measures to protect patient data, ensure device integrity, and prevent unauthorized access.

Security Principles

  • Defense in Depth: Multiple layers of security
  • Least Privilege: Minimum necessary access
  • Zero Trust: Always verify, never trust
  • Privacy by Design: Built-in from the ground up

Session Authentication

interface SessionAuth {
  // Token-based authentication
  tokenBased: boolean;
  tokenExpiry: number;        // seconds (3600 = 1 hour)
  refreshEnabled: boolean;

  // Challenge-response
  challengeResponse: {
    algorithm: 'ECDSA-P256';
    nonceSize: 32;            // bytes
  };
}

// Authenticate session
async function authenticateSession(deviceId: string) {
  // 1. Generate challenge
  const challenge = crypto.randomBytes(32);

  // 2. Send to device
  const response = await sendChallenge(deviceId, challenge);

  // 3. Verify signature
  const valid = verifySignature(
    challenge,
    response.signature,
    devicePublicKey
  );

  if (!valid) {
    throw new Error('Authentication failed');
  }

  // 4. Issue session token
  const token = generateSessionToken(deviceId, {
    expiresIn: 3600
  });

  return token;
}

// Encrypt sensitive data
async function encryptData(data: Uint8Array, key: CryptoKey) {
  const iv = crypto.getRandomValues(new Uint8Array(12));

  const encrypted = await crypto.subtle.encrypt(
    { name: 'AES-GCM', iv },
    key,
    data
  );

  return {
    ciphertext: new Uint8Array(encrypted),
    iv: iv,
    algorithm: 'AES-256-GCM'
  };
}

Special considerations for protecting audio data, which may contain sensitive information.

interface AudioDataProtection {
  // Storage policy
  storage: {
    neverStorePermanent: true;
    bufferOnly: true;
    maxBufferTime: 5000;      // ms
    clearOnDisconnect: true;
  };

  // Transmission
  transmission: {
    encrypted: true;
    compression: 'opus';      // Also provides obfuscation
    streamOnly: true;         // No recording
  };

  // Processing
  processing: {
    onDeviceOnly: boolean;
    cloudProcessing: 'opt-in';
    anonymizeBeforeCloud: true;
  };
}

// Audio buffer management
class SecureAudioBuffer {
  private buffer: Float32Array;
  private maxAge: number = 5000;  // ms

  push(frame: Float32Array) {
    this.buffer = frame;
    setTimeout(() => this.clear(), this.maxAge);
  }

  clear() {
    // Secure erase (overwrite with zeros)
    this.buffer.fill(0);
    this.buffer = null;
  }
}

6.5 Device Security

Firmware Security

interface FirmwareSecurity {
  // Secure boot
  secureBoot: {
    enabled: true;
    bootloaderSigned: true;
    firmwareSigned: true;
    publicKeyEmbedded: true;
  };

  // Firmware updates
  updates: {
    signatureVerification: 'mandatory';
    algorithm: 'ECDSA-P256';
    rollbackPrevention: true;
    encryptedTransmission: true;
  };
}

// Verify firmware update
async function verifyFirmwareUpdate(
  firmware: Uint8Array,
  signature: Uint8Array
) {
  // 1. Verify signature
  const valid = await crypto.subtle.verify(
    { name: 'ECDSA', hash: 'SHA-256' },
    manufacturerPublicKey,
    signature,
    firmware
  );

  if (!valid) {
    throw new Error('Invalid firmware signature');
  }

  // 2. Check version (prevent rollback)
  const version = extractVersion(firmware);
  if (version <= currentVersion) {
    throw new Error('Firmware rollback prevented');
  }

  return true;
}

async function erasePatientData(patientId: string) {
  // 1. Delete personal data
  await db.delete('patients', { id: patientId });

  // 2. Anonymize historical data
  await db.update('sessions',
    { patientId },
    { patientId: 'ANONYMIZED' }

    patientId,
    timestamp: new Date(),

  // 4. Notify connected systems
  await notifyErasure(patientId);

    permissions: [
      'manage_users',
      'view_audit_logs',

    ];

// Check permission
function checkPermission(user: User, action: string): boolean {
  const role = user.role;
  const permissions = RBACRoles[role].permissions;
  return permissions.includes(action);
}

Audit Log Structure

interface AuditLogEntry {
  id: string;
  timestamp: string;          // ISO8601
  userId: string;
  role: string;
  action: string;
  resource: string;
  result: 'success' | 'failure';
  ipAddress?: string;
  deviceId?: string;
  details?: object;
}

// Log critical actions
const auditActions = [
  'user.login',
  'user.logout',
  'config.update',
  'data.export',
  'data.delete',
  'firmware.update',
  'session.start',
  'session.end',
  'telemetry.access',
  'nrt.run'
];

async function logAction(
  userId: string,
  action: string,
  details: object
) {
  const entry: AuditLogEntry = {
    id: generateUUID(),
    timestamp: new Date().toISOString(),
    userId,
    role: getUserRole(userId),
    action,
    resource: details.resource,
    result: details.success ? 'success' : 'failure',
    ipAddress: getClientIP(),
    details
  };

  await auditLog.write(entry);

  // Alert on suspicious activity
  if (isSuspicious(entry)) {
    await alertSecurityTeam(entry);
  }
}

6.9 Incident Response

Breach Notification Process

interface IncidentResponse {
  detection: {
    monitoring: '24/7';
    alerting: 'automated';
    anomalyDetection: true;
  };

  response: {
    containment: {
      maxTime: 3600,          // seconds (1 hour)
      isolateAffected: true,
      revokeCredentials: true
    };

    investigation: {
      forensics: true,
      rootCause: true,
      impactAssessment: true
    };

    notification: {
      users: {
        maxDelay: 72,         // hours
        method: ['email', 'app_notification']
      };
      authorities: {
        maxDelay: 72,         // hours (HIPAA/GDPR)
        agencies: ['HHS', 'ICO', 'PIPC']
      };
    };
  };
}

async function handleSecurityIncident(incident: Incident) {
  // 1. Contain
  await containIncident(incident);

  // 2. Investigate
  const report = await investigateIncident(incident);

  // 3. Notify affected parties
  if (report.affectedUsers > 0) {
    await notifyUsers(report.affectedUsers);
  }

  // 4. Notify authorities (if required)
  if (report.breachNotificationRequired) {
    await notifyAuthorities(report);
  }

  // 5. Remediate
  await remediateVulnerability(report.vulnerability);

  // 6. Document
  await documentIncident(incident, report);
}

6.10 Security Best Practices

Developer Guidelines

  • Never hardcode secrets: Use secure key storage
  • Validate all inputs: Prevent injection attacks
  • Use prepared statements: Prevent SQL injection
  • Implement rate limiting: Prevent brute force
  • Regular security audits: Penetration testing
  • Keep dependencies updated: Patch vulnerabilities
  • Minimize attack surface: Disable unnecessary features

Security Checklist

  • ✓ Device pairing with MITM protection
  • ✓ Session authentication via challenge-response
  • ✓ AES-256-GCM encryption at rest
  • ✓ TLS 1.3 for data in transit
  • ✓ Audio data never stored permanently
  • ✓ Signed firmware updates only
  • ✓ HIPAA/GDPR/PIPA compliance
  • ✓ Role-based access control
  • ✓ Comprehensive audit logging
  • ✓ Incident response plan
