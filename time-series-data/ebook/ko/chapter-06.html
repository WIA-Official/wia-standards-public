6.1 이상 탐지 개요.3 실전 구현

# 6.1 이상 탐지 개요 Python 구현 예시
import pandas as pd
import numpy as np
from sklearn.ensemble import IsolationForest
from datetime import datetime, timedelta

# 시계열 데이터 생성
dates = pd.date_range(start='2025-01-01', periods=1000, freq='1H')
data = pd.DataFrame({
    'timestamp': dates,
    'value': np.random.randn(1000) * 10 + 50,
    'sensor_id': 'sensor_001'
})

# 이상 탐지 모델 학습
model = IsolationForest(contamination=0.05, random_state=42)
data['anomaly'] = model.fit_predict(data[['value']])

# 이상치 필터링
anomalies = data[data['anomaly'] == -1]
print(f"발견된 이상치: {len(anomalies)}개")

# 시각화 및 분석
import matplotlib.pyplot as plt
plt.figure(figsize=(12, 6))
plt.plot(data['timestamp'], data['value'], label='Normal', alpha=0.7)
plt.scatter(anomalies['timestamp'], anomalies['value'], 
            color='red', label='Anomaly', s=100)
plt.xlabel('Time')
plt.ylabel('Value')
plt.legend()
plt.title('6.1 이상 탐지 개요 - Anomaly Detection Results')
plt.show()

# 6.6 실시간 이상 탐지 Python 구현 예시
import pandas as pd
import numpy as np
from sklearn.ensemble import IsolationForest
from datetime import datetime, timedelta

# 시계열 데이터 생성
dates = pd.date_range(start='2025-01-01', periods=1000, freq='1H')
data = pd.DataFrame({
    'timestamp': dates,
    'value': np.random.randn(1000) * 10 + 50,
    'sensor_id': 'sensor_001'
})

# 이상 탐지 모델 학습
model = IsolationForest(contamination=0.05, random_state=42)
data['anomaly'] = model.fit_predict(data[['value']])

# 이상치 필터링
anomalies = data[data['anomaly'] == -1]
print(f"발견된 이상치: {len(anomalies)}개")

# 시각화 및 분석
import matplotlib.pyplot as plt
plt.figure(figsize=(12, 6))
plt.plot(data['timestamp'], data['value'], label='Normal', alpha=0.7)
plt.scatter(anomalies['timestamp'], anomalies['value'], 
            color='red', label='Anomaly', s=100)
plt.xlabel('Time')
plt.ylabel('Value')
plt.legend()
plt.title('6.6 실시간 이상 탐지 - Anomaly Detection Results')
plt.show()
