Chapter 6: Risk Assessment and Analysis

Identifying and Mitigating AI Decision Risks

Understanding AI Decision Risk

AI systems can fail in subtle and unexpected ways. A decision that appears correct may harbor hidden risks: embedded bias, data quality issues, model drift, edge case mishandling, or cascading failures. Effective risk assessment identifies these issues before they cause harm.

The principle of 弘益人間 compels us to proactively identify and mitigate risks, ensuring AI systems benefit rather than harm. Risk assessment transforms audit data from historical record into actionable intelligence for continuous improvement.

Multi-Dimensional Risk Scoring

Comprehensive risk scoring evaluates decisions across multiple dimensions:

interface RiskScore {
  overall: number;  // 0-100
  dimensions: {
    confidence_risk: number;      // Low model confidence
    bias_risk: number;            // Potential discrimination
    data_quality_risk: number;    // Input data issues
    drift_risk: number;           // Model performance degradation
    impact_risk: number;          // Consequences of error
    compliance_risk: number;      // Regulatory violations
    fairness_risk: number;        // Disparate impact
    explainability_risk: number;  // Lack of transparency
  };
  flags: string[];
  severity: "low" | "medium" | "high" | "critical";
}

class RiskAssessmentEngine {
  async assessRisk(log: AuditLog): Promise {
    const dimensions = {
      confidence_risk: this.assessConfidenceRisk(log),
      bias_risk: await this.assessBiasRisk(log),
      data_quality_risk: this.assessDataQualityRisk(log),
      drift_risk: await this.assessDriftRisk(log),
      impact_risk: this.assessImpactRisk(log),
      compliance_risk: await this.assessComplianceRisk(log),
      fairness_risk: await this.assessFairnessRisk(log),
      explainability_risk: this.assessExplainabilityRisk(log)
    };

    const overall = this.calculateOverallRisk(dimensions);
    const flags = this.identifyRiskFlags(log, dimensions);
    const severity = this.determineSeverity(overall, flags);

    return { overall, dimensions, flags, severity };
  }

  private calculateOverallRisk(dimensions: any): number {
    // Weighted average of dimensions
    const weights = {
      confidence_risk: 0.15,
      bias_risk: 0.20,
      data_quality_risk: 0.10,
      drift_risk: 0.10,
      impact_risk: 0.15,
      compliance_risk: 0.15,
      fairness_risk: 0.10,
      explainability_risk: 0.05
    };

    let score = 0;
    for (const [dimension, weight] of Object.entries(weights)) {
      score += dimensions[dimension] * weight;
    }

    return Math.round(score);
  }
}

Confidence Risk Assessment

Low confidence scores indicate model uncertainty and elevated error risk:

assessConfidenceRisk(log: AuditLog): number {
  const confidence = log.output.confidence;

  // Map confidence to risk (inverse relationship)
  if (confidence >= 0.95) return 5;   // Very low risk
  if (confidence >= 0.85) return 20;  // Low risk
  if (confidence >= 0.70) return 50;  // Medium risk
  if (confidence >= 0.60) return 75;  // High risk
  return 95;                          // Critical risk

  // Additional factors
  let riskAdjustment = 0;

  // Higher stakes = higher risk sensitivity
  if (log.context.business_impact === "critical") {
    riskAdjustment += 20;
  }

  // Check if near decision boundary
  const alternatives = log.output.alternatives || [];
  if (alternatives.length > 0) {
    const secondBestConfidence = alternatives[0].confidence;
    const margin = confidence - secondBestConfidence;

    if (margin < 0.10) {
      riskAdjustment += 25;  // Very close call
    }
  }

  return Math.min(100, baseRisk + riskAdjustment);
}

Bias Detection and Fairness Analysis

Systematic bias analysis identifies disparate impact across demographic groups:

class BiasDetector {
  async assessBiasRisk(log: AuditLog): Promise {
    // Check for protected attribute usage
    const protectedAttrs = this.identifyProtectedAttributes(log);
    if (protectedAttrs.length > 0) {
      return 100;  // Critical risk
    }

    // Analyze historical patterns
    const historicalAnalysis = await this.analyzeHistoricalBias(
      log.model.name,
      log.context.decision_type
    );

    return historicalAnalysis.bias_score;
  }

  async analyzeHistoricalBias(
    modelName: string,
    decisionType: string
  ): Promise {
    const decisions = await this.loadHistoricalDecisions(
      modelName,
      decisionType,
      30  // Last 30 days
    );

    // Calculate disparate impact ratio
    const groupedByDemographic = this.groupByInferredDemographic(decisions);
    const approvalRates = this.calculateApprovalRates(groupedByDemographic);

    // Compute disparate impact (80% rule)
    const minRate = Math.min(...Object.values(approvalRates));
    const maxRate = Math.max(...Object.values(approvalRates));
    const disparateImpactRatio = minRate / maxRate;

    let bias_score = 0;
    if (disparateImpactRatio < 0.80) {
      bias_score = 80;  // Fails 80% rule
    } else if (disparateImpactRatio < 0.90) {
      bias_score = 40;  // Warning zone
    } else {
      bias_score = 10;  // Acceptable
    }

    return {
      bias_score,
      disparate_impact_ratio: disparateImpactRatio,
      approval_rates: approvalRates,
      sample_size: decisions.length
    };
  }

  // Statistical parity check
  calculateStatisticalParity(
    outcomes: Map
  ): number {
    const rates = new Map();

    for (const [group, outcomeList] of outcomes) {
      const positive = outcomeList.filter(o => o).length;
      rates.set(group, positive / outcomeList.length);
    }

    const maxDifference = this.maxPairwiseDifference(rates);
    return maxDifference > 0.10 ? 70 : 20;
  }

  // Equal opportunity check
  calculateEqualOpportunity(
    outcomes: Map
  ): number {
    const tpr = new Map();  // True positive rate

    for (const [group, cases] of outcomes) {
      const positives = cases.filter(c => c.actual);
      const truePositives = positives.filter(c => c.predicted);
      tpr.set(group, truePositives.length / positives.length);
    }

    const maxDifference = this.maxPairwiseDifference(tpr);
    return maxDifference > 0.10 ? 70 : 20;
  }
}

Data Quality Risk Assessment

Poor input data quality leads to unreliable decisions:

assessDataQualityRisk(log: AuditLog): number {
  let risk = 0;

  // Check for missing values
  const missingFields = this.identifyMissingFields(log.input.raw_data);
  risk += missingFields.length * 10;

  // Check for outliers
  const outliers = this.detectOutliers(log.input.raw_data);
  risk += outliers.length * 15;

  // Check data freshness
  const dataSources = log.input.data_sources || [];
  for (const source of dataSources) {
    const age = this.getDataAge(source);
    if (age > 90) {  // Days
      risk += 20;  // Stale data
    }
  }

  // Check for data conflicts
  const conflicts = this.detectConflicts(log.input.raw_data);
  risk += conflicts.length * 25;

  return Math.min(100, risk);
}

detectOutliers(data: any): string[] {
  const outliers: string[] = [];

  for (const [field, value] of Object.entries(data)) {
    if (typeof value !== 'number') continue;

    const distribution = this.getFieldDistribution(field);
    const zScore = (value - distribution.mean) / distribution.stdDev;

    if (Math.abs(zScore) > 3) {  // 3 standard deviations
      outliers.push(field);
    }
  }

  return outliers;
}

Model Drift Detection

Models degrade over time as data distributions shift:

class DriftDetector {
  async assessDriftRisk(log: AuditLog): Promise {
    const modelId = `${log.model.name}-${log.model.version}`;

    // Compare recent performance to training performance
    const trainingMetrics = await this.getTrainingMetrics(modelId);
    const recentMetrics = await this.getRecentMetrics(modelId, 7);  // Last 7 days

    // Accuracy drift
    const accuracyDrift = Math.abs(
      trainingMetrics.accuracy - recentMetrics.accuracy
    );

    let risk = 0;

    if (accuracyDrift > 0.10) {
      risk += 60;  // Significant degradation
    } else if (accuracyDrift > 0.05) {
      risk += 30;  // Moderate degradation
    }

    // Feature distribution drift (KL divergence)
    const distributionDrift = await this.calculateKLDivergence(
      trainingMetrics.feature_distributions,
      recentMetrics.feature_distributions
    );

    if (distributionDrift > 0.5) {
      risk += 40;  // High drift
    } else if (distributionDrift > 0.2) {
      risk += 20;  // Moderate drift
    }

    // Prediction distribution drift
    const predictionDrift = this.compareDistributions(
      trainingMetrics.prediction_distribution,
      recentMetrics.prediction_distribution
    );

    if (predictionDrift > 0.3) {
      risk += 30;
    }

    return Math.min(100, risk);
  }

  async calculateKLDivergence(
    p: Distribution,
    q: Distribution
  ): Promise {
    let divergence = 0;

    for (const bin of p.bins) {
      const pVal = p.probabilities[bin] || 0.0001;
      const qVal = q.probabilities[bin] || 0.0001;
      divergence += pVal * Math.log(pVal / qVal);
    }

    return divergence;
  }
}

Impact Risk Assessment

Evaluate the potential consequences of decision errors:

assessImpactRisk(log: AuditLog): number {
  // Base risk on declared business impact
  const baseRisk = {
    "critical": 90,
    "high": 70,
    "medium": 40,
    "low": 15
  }[log.context.business_impact] || 50;

  // Adjust for decision reversibility
  const reversible = log.context.reversible || false;
  const reversibilityAdjustment = reversible ? -20 : 0;

  // Adjust for affected population size
  const populationSize = log.context.affected_population || 1;
  let populationAdjustment = 0;

  if (populationSize > 10000) {
    populationAdjustment = 25;
  } else if (populationSize > 1000) {
    populationAdjustment = 15;
  } else if (populationSize > 100) {
    populationAdjustment = 5;
  }

  // Financial impact
  const financialImpact = log.context.financial_impact || 0;
  let financialAdjustment = 0;

  if (financialImpact > 1000000) {
    financialAdjustment = 20;
  } else if (financialImpact > 100000) {
    financialAdjustment = 10;
  }

  return Math.min(100,
    baseRisk + reversibilityAdjustment +
    populationAdjustment + financialAdjustment
  );
}

Anomaly Detection

Identify unusual patterns that may indicate problems:

class AnomalyDetector {
  async detectAnomalies(log: AuditLog): Promise {
    const anomalies: Anomaly[] = [];

    // Unusual confidence for this decision type
    const typicalConfidence = await this.getTypicalConfidence(
      log.context.decision_type
    );

    if (Math.abs(log.output.confidence - typicalConfidence.mean) >
        2 * typicalConfidence.stdDev) {
      anomalies.push({
        type: "unusual_confidence",
        severity: "medium",
        description: `Confidence ${log.output.confidence} unusual for this decision type`
      });
    }

    // Unusual processing time
    const typicalDuration = await this.getTypicalDuration(log.model.name);

    if (log.timestamp.processing_duration_ms > typicalDuration.p95) {
      anomalies.push({
        type: "slow_processing",
        severity: "low",
        description: `Processing took ${log.timestamp.processing_duration_ms}ms (p95: ${typicalDuration.p95}ms)`
      });
    }

    // Unusual input patterns
    const inputAnomalies = await this.detectInputAnomalies(log.input);
    anomalies.push(...inputAnomalies);

    // Check for repeated similar decisions
    const duplicates = await this.findSimilarRecentDecisions(log);
    if (duplicates.length > 5) {
      anomalies.push({
        type: "suspicious_repetition",
        severity: "high",
        description: `${duplicates.length} similar decisions in short timeframe`
      });
    }

    return anomalies;
  }

  async detectInputAnomalies(input: any): Promise {
    // Use isolation forest or one-class SVM
    const isolationForest = await this.getAnomalyModel();
    const score = isolationForest.score(input.feature_vector);

    if (score < -0.5) {  // High anomaly score
      return [{
        type: "unusual_input_pattern",
        severity: "medium",
        description: "Input pattern significantly different from training data",
        details: { anomaly_score: score }
      }];
    }

    return [];
  }
}

Risk-Based Alerting

Alert appropriate stakeholders based on risk severity:

class RiskBasedAlertingSystem {
  async processRiskScore(log: AuditLog, risk: RiskScore): Promise {
    if (risk.severity === "critical") {
      await this.sendCriticalAlert(log, risk);
    } else if (risk.severity === "high") {
      await this.sendHighRiskAlert(log, risk);
    } else if (risk.severity === "medium") {
      await this.logMediumRisk(log, risk);
    }

    // Route high-risk decisions to human review
    if (risk.overall > 70) {
      await this.routeToHumanReview(log, risk);
    }

    // Update risk dashboard
    await this.updateDashboard(risk);
  }

  async sendCriticalAlert(log: AuditLog, risk: RiskScore): Promise {
    const alert = {
      priority: "P1",
      title: `Critical Risk Decision: ${log.decision_id}`,
      description: `Risk score: ${risk.overall}/100`,
      flags: risk.flags,
      affected_user: log.context.user_id,
      timestamp: new Date().toISOString()
    };

    await Promise.all([
      this.pagerDuty.trigger(alert),
      this.slack.sendToChannel("#ai-critical-alerts", alert),
      this.email.sendToSecurityTeam(alert),
      this.createIncident(alert)
    ]);
  }

  async routeToHumanReview(log: AuditLog, risk: RiskScore): Promise {
    await this.humanReviewQueue.add({
      decision_id: log.decision_id,
      priority: risk.severity === "critical" ? 1 : 2,
      reason: risk.flags.join(", "),
      risk_score: risk.overall,
      deadline: this.calculateReviewDeadline(risk.severity)
    });
  }
}

Chapter Summary

  • Multi-dimensional risk scoring evaluates decisions across confidence, bias, data quality, drift, impact, compliance, fairness, and explainability
  • Confidence risk assessment identifies decisions where model uncertainty creates elevated error risk
  • Bias detection analyzes historical patterns for disparate impact using statistical parity and equal opportunity metrics
  • Data quality assessment identifies missing values, outliers, stale data, and conflicts that undermine decision reliability
  • Model drift detection compares recent performance to training baselines using accuracy, feature distribution, and prediction distribution metrics
  • Impact risk assessment evaluates potential consequences considering reversibility, affected population, and financial impact
  • Anomaly detection identifies unusual patterns in confidence, processing time, inputs, and decision patterns
  • Risk-based alerting routes critical issues to immediate attention while logging medium-risk events for analysis
  • Continuous monitoring dashboards provide real-time visibility into system risk posture and trends
