Chapter 5: Compliance and Verification

Automated Policy Enforcement and Regulatory Checking

From Manual to Automated Compliance

Traditional compliance verification relies on periodic manual audits—expensive, time-consuming, and reactive. By the time violations are discovered, harm may already be done. Automated compliance checking shifts the paradigm: real-time verification, continuous monitoring, and proactive violation prevention.

The principle of 弘益人間 demands that we prevent harm before it occurs. Automated compliance systems embody this principle by detecting and blocking problematic AI decisions in real-time, protecting individuals from algorithmic harm.

Compliance Rule Engine Architecture

A compliance rule engine evaluates audit logs against regulatory requirements and organizational policies:

interface ComplianceRule {
  id: string;
  name: string;
  framework: string;  // "GDPR", "CCPA", "EU AI Act", etc.
  severity: "critical" | "high" | "medium" | "low";
  check: (log: AuditLog) => ComplianceResult;
  remediation?: (log: AuditLog) => Promise;
}

interface ComplianceResult {
  compliant: boolean;
  message: string;
  evidence?: any;
  recommendation?: string;
}

class ComplianceRuleEngine {
  private rules: Map = new Map();

  registerRule(rule: ComplianceRule): void {
    this.rules.set(rule.id, rule);
  }

  async checkCompliance(log: AuditLog): Promise {
    const results: ComplianceResult[] = [];
    const violations: ComplianceViolation[] = [];

    for (const rule of this.rules.values()) {
      const result = rule.check(log);
      results.push(result);

      if (!result.compliant) {
        violations.push({
          rule_id: rule.id,
          rule_name: rule.name,
          severity: rule.severity,
          message: result.message,
          evidence: result.evidence,
          timestamp: new Date().toISOString()
        });

        // Attempt automated remediation
        if (rule.remediation) {
          await rule.remediation(log);
        }
      }
    }

    return {
      decision_id: log.decision_id,
      overall_compliant: violations.length === 0,
      violations,
      results,
      checked_at: new Date().toISOString()
    };
  }
}

Implementing GDPR Compliance Rules

Right to Explanation Verification

const gdprRightToExplanation: ComplianceRule = {
  id: "gdpr-art22-explanation",
  name: "GDPR Article 22 - Right to Explanation",
  framework: "GDPR",
  severity: "critical",

  check: (log: AuditLog): ComplianceResult => {
    // Article 22 applies to automated decisions with legal/significant effects
    if (!log.compliance.automated) {
      return { compliant: true, message: "Not automated decision" };
    }

    if (log.context.business_impact !== "high" &&
        log.context.business_impact !== "critical") {
      return { compliant: true, message: "Not significant effect" };
    }

    // Check for meaningful explanation
    if (!log.reasoning || !log.reasoning.explanation) {
      return {
        compliant: false,
        message: "Missing explanation for significant automated decision",
        recommendation: "Add meaningful explanation to decision reasoning"
      };
    }

    // Explanation must be substantive (min 100 characters)
    if (log.reasoning.explanation.length < 100) {
      return {
        compliant: false,
        message: "Explanation insufficient - must be meaningful and detailed",
        recommendation: "Expand explanation to provide substantive information"
      };
    }

    // Must include key factors
    if (!log.reasoning.key_features || log.reasoning.key_features.length === 0) {
      return {
        compliant: false,
        message: "Explanation must identify key factors influencing decision",
        recommendation: "Add feature importance information"
      };
    }

    return { compliant: true, message: "Explanation requirements satisfied" };
  }
};

Data Minimization Check

const gdprDataMinimization: ComplianceRule = {
  id: "gdpr-art5-minimization",
  name: "GDPR Article 5 - Data Minimization",
  framework: "GDPR",
  severity: "high",

  check: (log: AuditLog): ComplianceResult => {
    // Define allowed fields per decision type
    const allowedFields = getAllowedFieldsForDecisionType(
      log.context.decision_type
    );

    const actualFields = Object.keys(log.input.raw_data);
    const excessFields = actualFields.filter(f => !allowedFields.includes(f));

    if (excessFields.length > 0) {
      return {
        compliant: false,
        message: `Excessive data collected: ${excessFields.join(', ')}`,
        evidence: { excess_fields: excessFields },
        recommendation: "Remove unnecessary fields from input data"
      };
    }

    return { compliant: true, message: "Data minimization satisfied" };
  }
};

Consent Verification

const gdprConsent: ComplianceRule = {
  id: "gdpr-art6-consent",
  name: "GDPR Article 6 - Lawful Basis (Consent)",
  framework: "GDPR",
  severity: "critical",

  check: (log: AuditLog): ComplianceResult => {
    if (log.compliance.legal_basis !== "consent") {
      // Other legal bases may apply
      return { compliant: true, message: "Different legal basis applies" };
    }

    if (!log.compliance.data_subject_consent) {
      return {
        compliant: false,
        message: "Consent required but not obtained",
        recommendation: "Obtain explicit consent before processing"
      };
    }

    // Verify consent is current (not expired)
    const consentAge = Date.now() - new Date(log.compliance.consent_date).getTime();
    const oneYear = 365 * 24 * 60 * 60 * 1000;

    if (consentAge > oneYear) {
      return {
        compliant: false,
        message: "Consent expired - must be refreshed annually",
        recommendation: "Request renewed consent"
      };
    }

    return { compliant: true, message: "Valid consent present" };
  }
};

Implementing CCPA Compliance Rules

Automated Decision Disclosure

const ccpaAutomatedDecisionDisclosure: ComplianceRule = {
  id: "ccpa-1798-185-disclosure",
  name: "CCPA Automated Decision Disclosure",
  framework: "CCPA",
  severity: "high",

  check: (log: AuditLog): ComplianceResult => {
    if (!log.compliance.automated) {
      return { compliant: true, message: "Not automated decision" };
    }

    // Verify disclosure was provided
    if (!log.compliance.disclosure_provided) {
      return {
        compliant: false,
        message: "Automated decision-making not disclosed to consumer",
        recommendation: "Provide clear disclosure of automated decision-making"
      };
    }

    // Verify explanation availability
    if (!log.compliance.explanation_available) {
      return {
        compliant: false,
        message: "Explanation not made available to consumer",
        recommendation: "Ensure consumers can access decision explanations"
      };
    }

    return { compliant: true, message: "Disclosure requirements met" };
  }
};

};

EU AI Act Compliance Rules

High-Risk System Requirements

const euAIActHighRiskRequirements: ComplianceRule = {
  id: "eu-ai-act-high-risk",
  name: "EU AI Act High-Risk System Requirements",
  framework: "EU AI Act",
  severity: "critical",

  check: (log: AuditLog): ComplianceResult => {
    if (log.context.risk_category !== "high") {
      return { compliant: true, message: "Not high-risk system" };
    }

    const violations: string[] = [];

    // Requirement: Human oversight
    if (!log.compliance.human_oversight_available) {
      violations.push("Human oversight mechanism missing");
    }

    // Requirement: Technical documentation
    if (!log.model.documentation_url) {
      violations.push("Technical documentation not linked");
    }

    // Requirement: Accuracy metrics
    if (!log.model.accuracy_metrics) {
      violations.push("Model accuracy metrics not documented");
    }

    // Requirement: Bias testing
    if (!log.model.bias_testing_completed) {
      violations.push("Bias testing not completed");
    }

    // Requirement: Transparency
    if (!log.compliance.transparency_notice_provided) {
      violations.push("Transparency notice not provided to users");
    }

    if (violations.length > 0) {
      return {
        compliant: false,
        message: `High-risk system violations: ${violations.join('; ')}`,
        evidence: { violations },
        recommendation: "Address all high-risk requirements before deployment"
      };
    }

    return { compliant: true, message: "High-risk requirements satisfied" };
  }
};

Financial Services Compliance

Fair Lending (ECOA) Compliance

const ecoaFairLending: ComplianceRule = {
  id: "ecoa-fair-lending",
  name: "Equal Credit Opportunity Act - Fair Lending",
  framework: "ECOA",
  severity: "critical",

  check: (log: AuditLog): ComplianceResult => {
    if (log.context.decision_type !== "credit_decision") {
      return { compliant: true, message: "Not credit decision" };
    }

    // Protected characteristics must not be used
    const protectedCharacteristics = [
      'race', 'color', 'religion', 'national_origin',
      'sex', 'marital_status', 'age'
    ];

    const inputFields = Object.keys(log.input.raw_data);
    const prohibitedFields = inputFields.filter(f =>
      protectedCharacteristics.includes(f.toLowerCase())
    );

    if (prohibitedFields.length > 0) {
      return {
        compliant: false,
        message: `Prohibited fields used in credit decision: ${prohibitedFields.join(', ')}`,
        evidence: { prohibited_fields: prohibitedFields },
        recommendation: "Remove protected characteristics from decision inputs"
      };
    }

    // Adverse action notice required for denials
    if (log.output.decision === "DENIED") {
      if (!log.compliance.adverse_action_notice_sent) {
        return {
          compliant: false,
          message: "Adverse action notice required for credit denial",
          recommendation: "Send adverse action notice with specific reasons"
        };
      }
    }

    return { compliant: true, message: "Fair lending requirements met" };
  }
};

Real-Time Compliance Enforcement

Real-time checking prevents non-compliant decisions from being executed:

class RealTimeComplianceEnforcer {
  async enforceCompliance(log: AuditLog): Promise {
    const report = await this.ruleEngine.checkCompliance(log);

    // Block critical violations
    const criticalViolations = report.violations.filter(
      v => v.severity === "critical"
    );

    if (criticalViolations.length > 0) {
      // Block decision
      await this.blockDecision(log.decision_id, criticalViolations);

      // Alert compliance team
      await this.alertComplianceTeam({
        decision_id: log.decision_id,
        violations: criticalViolations,
        action: "BLOCKED"
      });

      return {
        allowed: false,
        reason: "Critical compliance violations detected",
        violations: criticalViolations
      };
    }

    // Warn on high-severity violations but allow
    const highViolations = report.violations.filter(
      v => v.severity === "high"
    );

    if (highViolations.length > 0) {
      await this.alertComplianceTeam({
        decision_id: log.decision_id,
        violations: highViolations,
        action: "WARNING"
      });
    }

    return {
      allowed: true,
      warnings: highViolations
    };
  }
}

Compliance Dashboard and Reporting

Executive dashboards provide visibility into compliance status:

class ComplianceDashboard {
  async generateDashboard(timeRange: DateRange): Promise {
    const logs = await this.loadLogsInRange(timeRange);
    const reports = await Promise.all(
      logs.map(log => this.ruleEngine.checkCompliance(log))
    );

    // Overall compliance rate
    const compliantCount = reports.filter(r => r.overall_compliant).length;
    const complianceRate = compliantCount / reports.length;

    // Violations by severity
    const violationsBySeverity = this.groupBySeverity(
      reports.flatMap(r => r.violations)
    );

    // Violations by framework
    const violationsByFramework = this.groupByFramework(
      reports.flatMap(r => r.violations)
    );

    // Trend analysis
    const trend = await this.analyzeTrend(timeRange);

    return {
      overview: {
        total_decisions: logs.length,
        compliant_decisions: compliantCount,
        compliance_rate: complianceRate,
        total_violations: reports.reduce((sum, r) => sum + r.violations.length, 0)
      },
      violations_by_severity: violationsBySeverity,
      violations_by_framework: violationsByFramework,
      trend,
      top_violations: this.identifyTopViolations(reports),
      recommendations: this.generateRecommendations(reports)
    };
  }
}

Policy as Code

Express organizational policies as executable code:

// Example: Internal AI ethics policy
const ethicsPolicy: ComplianceRule = {
  id: "internal-ethics-001",
  name: "Minimum Confidence Threshold",
  framework: "Internal Ethics Policy",
  severity: "high",

  check: (log: AuditLog): ComplianceResult => {
    const minConfidence = getMinConfidenceForImpact(
      log.context.business_impact
    );

    if (log.output.confidence < minConfidence) {
      return {
        compliant: false,
        message: `Confidence ${log.output.confidence} below threshold ${minConfidence}`,
        recommendation: "Require human review for low-confidence decisions"
      };
    }

    return { compliant: true, message: "Confidence threshold met" };
  },

  // Automated remediation: route to human review
  remediation: async (log: AuditLog) => {
    await humanReviewQueue.add({
      decision_id: log.decision_id,
      reason: "Low confidence score",
      priority: "high"
    });
  }
};
